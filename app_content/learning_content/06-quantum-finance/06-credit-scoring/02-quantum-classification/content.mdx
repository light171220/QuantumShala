# Quantum Classification Algorithms

Quantum computing offers new approaches to classification problems. This lesson explores quantum algorithms that can potentially improve credit scoring models.

<Definition term="Quantum Classification">
Using quantum computers to perform classification tasks, either through quantum-enhanced classical algorithms (like quantum kernel methods) or fully quantum approaches (like variational quantum classifiers).
</Definition>

## Quantum Kernel Methods

Quantum kernels can compute similarity in exponentially large feature spaces.

```python
def quantum_kernel_concept():
    """Explain quantum kernel methods."""

    print("Quantum Kernel Methods")
    print("=" * 50)

    print("\nCore Idea:")
    print("  1. Map classical data to quantum states: x → |φ(x)⟩")
    print("  2. Kernel = overlap of quantum states: K(x,x') = |⟨φ(x)|φ(x')⟩|²")
    print("  3. This kernel can capture complex non-linear relationships")

    # Simulate quantum kernel computation
    def classical_rbf_kernel(x1, x2, gamma=1.0):
        """Classical RBF kernel."""
        return np.exp(-gamma * np.linalg.norm(x1 - x2)**2)

    def simulated_quantum_kernel(x1, x2, n_qubits=4):
        """Simulate a quantum kernel using rotation encoding."""
        # In real quantum computing, this would be:
        # |⟨0|U†(x')U(x)|0⟩|²

        # Simulate with trigonometric encoding
        phi_x1 = np.array([np.cos(x1[i % len(x1)]) for i in range(n_qubits)])
        phi_x2 = np.array([np.cos(x2[i % len(x2)]) for i in range(n_qubits)])

        # Add entanglement-like correlation
        for i in range(n_qubits - 1):
            phi_x1[i+1] *= np.cos(x1[i % len(x1)] * x1[(i+1) % len(x1)])
            phi_x2[i+1] *= np.cos(x2[i % len(x2)] * x2[(i+1) % len(x2)])

        # Overlap squared
        overlap = np.dot(phi_x1, phi_x2) / (np.linalg.norm(phi_x1) * np.linalg.norm(phi_x2))
        return overlap**2

    # Compare kernels
    np.random.seed(42)
    x1 = np.random.randn(3)
    x2 = np.random.randn(3)

    rbf = classical_rbf_kernel(x1, x2)
    quantum = simulated_quantum_kernel(x1, x2)

    print(f"\nKernel Comparison for random points:")
    print(f"  Classical RBF: {rbf:.4f}")
    print(f"  Quantum Kernel: {quantum:.4f}")

    # Show advantage scenario
    print("\nPotential Quantum Advantage:")
    print("  • Classical kernel: O(d) for d-dimensional data")
    print("  • Quantum kernel: O(2^n) effective dimension with n qubits")
    print("  • Can capture correlations inaccessible to classical kernels")

    return rbf, quantum
```

## Quantum Support Vector Machine

```python
def quantum_svm_implementation():
    """Implement a simulated Quantum SVM."""

    print("Quantum Support Vector Machine (QSVM)")
    print("=" * 55)

    # Generate credit-like data
    np.random.seed(42)
    n_samples = 200

    # Two classes with overlap (like credit data)
    class_0 = np.random.randn(n_samples // 2, 2) + np.array([1, 1])
    class_1 = np.random.randn(n_samples // 2, 2) + np.array([-1, -1])

    X = np.vstack([class_0, class_1])
    y = np.array([0] * (n_samples // 2) + [1] * (n_samples // 2))

    # Shuffle
    idx = np.random.permutation(n_samples)
    X, y = X[idx], y[idx]

    print(f"Dataset: {n_samples} samples, 2 features")
    print(f"Class distribution: {np.sum(y==0)} class 0, {np.sum(y==1)} class 1")

    # Quantum kernel function
    def quantum_kernel_matrix(X1, X2, n_qubits=4, n_layers=2):
        """Compute quantum kernel matrix."""
        n1, n2 = len(X1), len(X2)
        K = np.zeros((n1, n2))

        for i in range(n1):
            for j in range(n2):
                # Simulated quantum kernel
                x1, x2 = X1[i], X2[j]

                # Feature map simulation
                phi1 = np.zeros(2**n_qubits)
                phi2 = np.zeros(2**n_qubits)

                for q in range(2**n_qubits):
                    angle1 = sum(x1[k % len(x1)] * ((q >> k) & 1) for k in range(n_qubits))
                    angle2 = sum(x2[k % len(x2)] * ((q >> k) & 1) for k in range(n_qubits))
                    phi1[q] = np.cos(angle1) + 0.5 * np.sin(angle1)
                    phi2[q] = np.cos(angle2) + 0.5 * np.sin(angle2)

                # Normalize
                phi1 /= np.linalg.norm(phi1)
                phi2 /= np.linalg.norm(phi2)

                K[i, j] = np.dot(phi1, phi2)**2

        return K

    # Split data
    n_train = 150
    X_train, X_test = X[:n_train], X[n_train:]
    y_train, y_test = y[:n_train], y[n_train:]

    print(f"\nComputing quantum kernel matrix...")

    # Compute kernel matrices
    K_train = quantum_kernel_matrix(X_train, X_train)
    K_test = quantum_kernel_matrix(X_test, X_train)

    print(f"  Training kernel: {K_train.shape}")
    print(f"  Test kernel: {K_test.shape}")

    # Simple SVM with precomputed kernel (simplified)
    # Using pseudo-inverse for soft-margin SVM
    y_train_signed = 2 * y_train - 1  # Convert to -1, +1

    # Add regularization
    C = 1.0
    alpha = np.linalg.solve(
        K_train + np.eye(n_train) / C,
        y_train_signed
    )

    # Predictions
    train_pred = np.sign(K_train @ alpha)
    test_pred = np.sign(K_test @ alpha)

    train_acc = np.mean(train_pred == y_train_signed)
    test_acc = np.mean(test_pred == (2 * y_test - 1))

    print(f"\nQSVM Results:")
    print(f"  Training accuracy: {train_acc*100:.1f}%")
    print(f"  Test accuracy: {test_acc*100:.1f}%")

    # Compare with classical RBF kernel
    def rbf_kernel_matrix(X1, X2, gamma=0.5):
        n1, n2 = len(X1), len(X2)
        K = np.zeros((n1, n2))
        for i in range(n1):
            for j in range(n2):
                K[i, j] = np.exp(-gamma * np.linalg.norm(X1[i] - X2[j])**2)
        return K

    K_train_rbf = rbf_kernel_matrix(X_train, X_train)
    K_test_rbf = rbf_kernel_matrix(X_test, X_train)

    alpha_rbf = np.linalg.solve(K_train_rbf + np.eye(n_train) / C, y_train_signed)
    test_pred_rbf = np.sign(K_test_rbf @ alpha_rbf)
    rbf_acc = np.mean(test_pred_rbf == (2 * y_test - 1))

    print(f"\nClassical RBF SVM:")
    print(f"  Test accuracy: {rbf_acc*100:.1f}%")

    return test_acc, rbf_acc
```

## Variational Quantum Classifier

```python
def variational_quantum_classifier():
    """Implement a simulated Variational Quantum Classifier."""

    print("Variational Quantum Classifier (VQC)")
    print("=" * 55)

    class SimulatedVQC:
        """Simulated variational quantum classifier."""

        def __init__(self, n_qubits=4, n_layers=2):
            self.n_qubits = n_qubits
            self.n_layers = n_layers
            self.n_params = n_qubits * n_layers * 3  # Rx, Ry, Rz per qubit per layer
            self.params = np.random.randn(self.n_params) * 0.1

        def encode_data(self, x):
            """Encode classical data into quantum state."""
            state = np.zeros(2**self.n_qubits, dtype=complex)
            state[0] = 1.0  # Start in |00...0⟩

            # Amplitude encoding (simplified)
            for i, val in enumerate(x[:2**self.n_qubits]):
                idx = i % (2**self.n_qubits)
                angle = np.pi * val
                # Simplified rotation effect
                state[idx] += np.sin(angle) * 0.5

            state /= np.linalg.norm(state)
            return state

        def apply_variational_layer(self, state, layer_params):
            """Apply parameterized rotations (simulated)."""
            new_state = state.copy()

            for q in range(self.n_qubits):
                # Get rotation angles
                rx, ry, rz = layer_params[q*3:(q+1)*3]

                # Apply rotations (simplified simulation)
                for i in range(len(new_state)):
                    if (i >> q) & 1:  # If qubit q is |1⟩
                        phase = np.exp(1j * (rx + ry + rz))
                        new_state[i] *= phase

            # Normalize
            new_state /= np.linalg.norm(new_state)
            return new_state

        def forward(self, x):
            """Forward pass through the circuit."""
            state = self.encode_data(x)

            for layer in range(self.n_layers):
                start = layer * self.n_qubits * 3
                end = (layer + 1) * self.n_qubits * 3
                layer_params = self.params[start:end]
                state = self.apply_variational_layer(state, layer_params)

            # Measure probability of |0⟩ on first qubit
            prob_0 = sum(np.abs(state[i])**2 for i in range(len(state)) if not (i & 1))
            return prob_0

        def predict_proba(self, X):
            """Predict probabilities for all samples."""
            return np.array([self.forward(x) for x in X])

        def predict(self, X, threshold=0.5):
            """Predict classes."""
            probs = self.predict_proba(X)
            return (probs < threshold).astype(int)

        def compute_loss(self, X, y):
            """Binary cross-entropy loss."""
            probs = self.predict_proba(X)
            probs = np.clip(probs, 1e-10, 1 - 1e-10)

            # For class 0: want high prob_0
            # For class 1: want low prob_0
            loss = -np.mean(
                (1 - y) * np.log(probs) + y * np.log(1 - probs)
            )
            return loss

        def train(self, X, y, epochs=100, learning_rate=0.1):
            """Train using numerical gradient descent."""
            losses = []

            for epoch in range(epochs):
                # Numerical gradient
                grad = np.zeros_like(self.params)
                eps = 0.01

                for i in range(len(self.params)):
                    self.params[i] += eps
                    loss_plus = self.compute_loss(X, y)
                    self.params[i] -= 2 * eps
                    loss_minus = self.compute_loss(X, y)
                    self.params[i] += eps

                    grad[i] = (loss_plus - loss_minus) / (2 * eps)

                self.params -= learning_rate * grad

                if epoch % 20 == 0:
                    loss = self.compute_loss(X, y)
                    losses.append(loss)
                    acc = np.mean(self.predict(X) == y)
                    print(f"  Epoch {epoch}: Loss = {loss:.4f}, Acc = {acc:.2%}")

            return losses

    # Generate data
    np.random.seed(42)
    n_train, n_test = 50, 20

    # Credit-like features (normalized)
    X_train = np.random.randn(n_train, 4) * 0.5
    y_train = (X_train[:, 0] + X_train[:, 1] > 0).astype(int)

    X_test = np.random.randn(n_test, 4) * 0.5
    y_test = (X_test[:, 0] + X_test[:, 1] > 0).astype(int)

    print(f"Training VQC on {n_train} samples...")
    print(f"Features: 4, Qubits: 4, Layers: 2\n")

    # Train VQC
    vqc = SimulatedVQC(n_qubits=4, n_layers=2)
    vqc.train(X_train, y_train, epochs=60, learning_rate=0.5)

    # Evaluate
    train_acc = np.mean(vqc.predict(X_train) == y_train)
    test_acc = np.mean(vqc.predict(X_test) == y_test)

    print(f"\nFinal Results:")
    print(f"  Training accuracy: {train_acc:.2%}")
    print(f"  Test accuracy: {test_acc:.2%}")

    return vqc
```
