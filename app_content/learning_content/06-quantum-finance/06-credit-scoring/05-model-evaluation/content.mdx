# Model Evaluation and Validation

Proper evaluation of quantum credit scoring models requires specialized metrics and validation strategies. This lesson covers best practices for assessing model quality.

<Definition term="Model Validation">
The process of evaluating a credit scoring model's performance, stability, and compliance with regulatory requirements using held-out data, appropriate metrics, and fairness assessments.
</Definition>

## Credit Scoring Metrics

```python
import numpy as np

def credit_scoring_metrics():
    """Comprehensive metrics for credit scoring models."""

    print("Credit Scoring Evaluation Metrics")
    print("=" * 55)

    class CreditMetrics:
        """Calculate all relevant credit scoring metrics."""

        def __init__(self, y_true, y_prob):
            self.y_true = np.array(y_true)
            self.y_prob = np.array(y_prob)
            self.n_samples = len(y_true)

        def confusion_matrix(self, threshold=0.5):
            """Calculate confusion matrix."""
            y_pred = (self.y_prob >= threshold).astype(int)
            tp = np.sum((y_pred == 1) & (self.y_true == 1))
            fp = np.sum((y_pred == 1) & (self.y_true == 0))
            tn = np.sum((y_pred == 0) & (self.y_true == 0))
            fn = np.sum((y_pred == 0) & (self.y_true == 1))
            return {'tp': tp, 'fp': fp, 'tn': tn, 'fn': fn}

        def roc_auc(self, n_thresholds=100):
            """Calculate ROC-AUC."""
            thresholds = np.linspace(0, 1, n_thresholds)
            tprs, fprs = [], []

            for t in thresholds:
                cm = self.confusion_matrix(t)
                tpr = cm['tp'] / (cm['tp'] + cm['fn']) if (cm['tp'] + cm['fn']) > 0 else 0
                fpr = cm['fp'] / (cm['fp'] + cm['tn']) if (cm['fp'] + cm['tn']) > 0 else 0
                tprs.append(tpr)
                fprs.append(fpr)

            # Sort by FPR and calculate AUC
            sorted_pairs = sorted(zip(fprs, tprs))
            fprs_sorted = [p[0] for p in sorted_pairs]
            tprs_sorted = [p[1] for p in sorted_pairs]

            auc = np.trapz(tprs_sorted, fprs_sorted)
            return auc

        def gini_coefficient(self):
            """Calculate Gini coefficient."""
            return 2 * self.roc_auc() - 1

        def ks_statistic(self):
            """Calculate Kolmogorov-Smirnov statistic."""
            sorted_idx = np.argsort(self.y_prob)[::-1]
            y_sorted = self.y_true[sorted_idx]

            n_pos = np.sum(self.y_true == 1)
            n_neg = np.sum(self.y_true == 0)

            if n_pos == 0 or n_neg == 0:
                return 0

            cum_pos = np.cumsum(y_sorted) / n_pos
            cum_neg = np.cumsum(1 - y_sorted) / n_neg

            return np.max(np.abs(cum_pos - cum_neg))

        def precision_recall_curve(self, n_thresholds=100):
            """Calculate precision-recall curve."""
            thresholds = np.linspace(0.01, 0.99, n_thresholds)
            precisions, recalls = [], []

            for t in thresholds:
                cm = self.confusion_matrix(t)
                precision = cm['tp'] / (cm['tp'] + cm['fp']) if (cm['tp'] + cm['fp']) > 0 else 0
                recall = cm['tp'] / (cm['tp'] + cm['fn']) if (cm['tp'] + cm['fn']) > 0 else 0
                precisions.append(precision)
                recalls.append(recall)

            return thresholds, precisions, recalls

        def average_precision(self):
            """Calculate average precision."""
            _, precisions, recalls = self.precision_recall_curve()
            return -np.trapz(precisions, recalls)

        def brier_score(self):
            """Calculate Brier score (calibration)."""
            return np.mean((self.y_prob - self.y_true)**2)

        def full_report(self):
            """Generate comprehensive report."""
            return {
                'n_samples': self.n_samples,
                'n_positives': int(np.sum(self.y_true)),
                'base_rate': np.mean(self.y_true),
                'auc': self.roc_auc(),
                'gini': self.gini_coefficient(),
                'ks': self.ks_statistic(),
                'average_precision': self.average_precision(),
                'brier_score': self.brier_score()
            }

    # Simulated predictions
    np.random.seed(42)
    n = 1000
    y_true = np.random.binomial(1, 0.15, n)
    y_prob = np.clip(y_true * 0.4 + np.random.beta(2, 5, n), 0, 1)

    metrics = CreditMetrics(y_true, y_prob)
    report = metrics.full_report()

    print("\nModel Performance Report:")
    for key, value in report.items():
        if isinstance(value, float):
            print(f"  {key}: {value:.4f}")
        else:
            print(f"  {key}: {value}")

    return metrics, report
```

## Cross-Validation for Quantum Models

```python
def quantum_cross_validation():
    """Cross-validation strategies for quantum credit models."""

    print("Cross-Validation for Quantum Models")
    print("=" * 55)

    class QuantumCrossValidator:
        """Cross-validation with quantum-specific considerations."""

        def __init__(self, model_class, n_folds=5):
            self.model_class = model_class
            self.n_folds = n_folds

        def stratified_split(self, y, n_folds):
            """Create stratified folds."""
            pos_idx = np.where(y == 1)[0]
            neg_idx = np.where(y == 0)[0]

            np.random.shuffle(pos_idx)
            np.random.shuffle(neg_idx)

            pos_folds = np.array_split(pos_idx, n_folds)
            neg_folds = np.array_split(neg_idx, n_folds)

            folds = []
            for pf, nf in zip(pos_folds, neg_folds):
                folds.append(np.concatenate([pf, nf]))

            return folds

        def cross_validate(self, X, y, **model_params):
            """Perform k-fold cross-validation."""
            folds = self.stratified_split(y, self.n_folds)

            results = {
                'train_scores': [],
                'test_scores': [],
                'train_times': [],
                'fold_metrics': []
            }

            for fold_idx, test_idx in enumerate(folds):
                # Create train indices
                train_idx = np.concatenate([folds[i] for i in range(self.n_folds) if i != fold_idx])

                X_train, X_test = X[train_idx], X[test_idx]
                y_train, y_test = y[train_idx], y[test_idx]

                # Train model
                model = self.model_class(**model_params)

                import time
                start = time.time()
                model.fit(X_train, y_train)
                train_time = time.time() - start

                # Evaluate
                train_score = model.score(X_train, y_train)
                test_score = model.score(X_test, y_test)

                results['train_scores'].append(train_score)
                results['test_scores'].append(test_score)
                results['train_times'].append(train_time)

                print(f"  Fold {fold_idx + 1}: Train={train_score:.3f}, Test={test_score:.3f}")

            # Summary statistics
            results['mean_train'] = np.mean(results['train_scores'])
            results['mean_test'] = np.mean(results['test_scores'])
            results['std_test'] = np.std(results['test_scores'])

            return results

    # Simple model for demonstration
    class SimpleQuantumModel:
        def __init__(self, n_params=4):
            self.params = np.random.randn(n_params) * 0.1

        def fit(self, X, y, epochs=50):
            for _ in range(epochs):
                for x, label in zip(X, y):
                    pred = self.predict_proba(x.reshape(1, -1))[0]
                    error = pred - label
                    self.params -= 0.1 * error * x[:len(self.params)]
            return self

        def predict_proba(self, X):
            return 1 / (1 + np.exp(-X[:, :len(self.params)] @ self.params))

        def predict(self, X):
            return (self.predict_proba(X) >= 0.5).astype(int)

        def score(self, X, y):
            return np.mean(self.predict(X) == y)

    # Generate data
    np.random.seed(42)
    X = np.random.randn(200, 4)
    y = (X[:, 0] + X[:, 1] - 0.5 * X[:, 2] > 0).astype(int)

    print("Running 5-fold cross-validation...\n")
    cv = QuantumCrossValidator(SimpleQuantumModel, n_folds=5)
    results = cv.cross_validate(X, y)

    print(f"\nSummary:")
    print(f"  Mean train accuracy: {results['mean_train']:.3f}")
    print(f"  Mean test accuracy: {results['mean_test']:.3f} ± {results['std_test']:.3f}")

    return cv, results
```

## Model Fairness Assessment

```python
def fairness_assessment():
    """Assess fairness of credit scoring model."""

    print("Credit Model Fairness Assessment")
    print("=" * 55)

    class FairnessAnalyzer:
        """Analyze model fairness across protected groups."""

        def __init__(self, y_true, y_prob, protected_attribute):
            self.y_true = np.array(y_true)
            self.y_prob = np.array(y_prob)
            self.protected = np.array(protected_attribute)
            self.groups = np.unique(protected_attribute)

        def demographic_parity(self, threshold=0.5):
            """
            Measure demographic parity.
            Equal positive prediction rates across groups.
            """
            y_pred = (self.y_prob >= threshold).astype(int)

            rates = {}
            for group in self.groups:
                mask = self.protected == group
                rates[group] = np.mean(y_pred[mask])

            max_diff = max(rates.values()) - min(rates.values())
            return rates, max_diff

        def equal_opportunity(self, threshold=0.5):
            """
            Measure equal opportunity.
            Equal true positive rates across groups.
            """
            y_pred = (self.y_prob >= threshold).astype(int)

            tprs = {}
            for group in self.groups:
                mask = (self.protected == group) & (self.y_true == 1)
                if mask.sum() > 0:
                    tprs[group] = np.mean(y_pred[mask])
                else:
                    tprs[group] = 0

            max_diff = max(tprs.values()) - min(tprs.values())
            return tprs, max_diff

        def predictive_parity(self, threshold=0.5):
            """
            Measure predictive parity.
            Equal precision across groups.
            """
            y_pred = (self.y_prob >= threshold).astype(int)

            precisions = {}
            for group in self.groups:
                mask = (self.protected == group) & (y_pred == 1)
                if mask.sum() > 0:
                    precisions[group] = np.mean(self.y_true[mask])
                else:
                    precisions[group] = 0

            max_diff = max(precisions.values()) - min(precisions.values())
            return precisions, max_diff

        def disparate_impact_ratio(self, threshold=0.5):
            """
            Calculate disparate impact ratio.
            Ratio of positive rates between groups.
            """
            rates, _ = self.demographic_parity(threshold)

            if len(self.groups) != 2:
                return None

            rate_values = list(rates.values())
            if min(rate_values) > 0:
                return min(rate_values) / max(rate_values)
            return 0

        def full_fairness_report(self, threshold=0.5):
            """Generate comprehensive fairness report."""
            dp_rates, dp_diff = self.demographic_parity(threshold)
            eo_rates, eo_diff = self.equal_opportunity(threshold)
            pp_rates, pp_diff = self.predictive_parity(threshold)
            di_ratio = self.disparate_impact_ratio(threshold)

            return {
                'demographic_parity': {'rates': dp_rates, 'max_diff': dp_diff},
                'equal_opportunity': {'rates': eo_rates, 'max_diff': eo_diff},
                'predictive_parity': {'rates': pp_rates, 'max_diff': pp_diff},
                'disparate_impact_ratio': di_ratio,
                'fairness_assessment': 'PASS' if di_ratio and di_ratio >= 0.8 else 'REVIEW'
            }

    # Simulated data with protected attribute
    np.random.seed(42)
    n = 1000

    # Protected attribute (e.g., age group: 0=young, 1=old)
    protected = np.random.binomial(1, 0.4, n)

    # True labels (slight disparity in base rates)
    base_rates = np.where(protected == 0, 0.18, 0.12)
    y_true = np.random.binomial(1, base_rates)

    # Model predictions (capturing and potentially amplifying disparity)
    y_prob = np.clip(
        y_true * 0.5 + np.random.beta(2, 5, n) + 0.05 * protected,
        0, 1
    )

    analyzer = FairnessAnalyzer(y_true, y_prob, protected)
    report = analyzer.full_fairness_report()

    print("\nFairness Metrics:")
    print(f"\n  Demographic Parity:")
    for group, rate in report['demographic_parity']['rates'].items():
        print(f"    Group {group}: {rate:.3f}")
    print(f"    Max difference: {report['demographic_parity']['max_diff']:.3f}")

    print(f"\n  Equal Opportunity (TPR):")
    for group, rate in report['equal_opportunity']['rates'].items():
        print(f"    Group {group}: {rate:.3f}")
    print(f"    Max difference: {report['equal_opportunity']['max_diff']:.3f}")

    print(f"\n  Disparate Impact Ratio: {report['disparate_impact_ratio']:.3f}")
    print(f"  Assessment: {report['fairness_assessment']}")

    return analyzer, report

fairness_assessment()
```

<Warning>
Regulatory requirements (like ECOA in the US) prohibit discrimination in credit decisions. Quantum models must be evaluated for fairness just like classical models. Monitor for disparate impact across protected groups.
</Warning>

## Quantum vs Classical Comparison

```python
def quantum_classical_comparison():
    """Compare quantum and classical model performance."""

    print("Quantum vs Classical Model Comparison")
    print("=" * 55)

    comparison = {
        'accuracy': [],
        'auc': [],
        'training_time': [],
        'inference_time': []
    }

    # Generate synthetic credit data
    np.random.seed(42)
    n_samples = 500
    n_features = 4

    X = np.random.randn(n_samples, n_features)
    # Non-linear decision boundary
    y = ((X[:, 0] * X[:, 1] - 0.5 * X[:, 2]**2 + X[:, 3]) > 0).astype(int)

    # Split
    n_train = 400
    X_train, X_test = X[:n_train], X[n_train:]
    y_train, y_test = y[:n_train], y[n_train:]

    print(f"Data: {n_train} train, {n_samples - n_train} test\n")

    # Classical Logistic Regression
    class ClassicalLogistic:
        def __init__(self):
            self.weights = None
            self.bias = 0

        def fit(self, X, y, lr=0.1, epochs=100):
            self.weights = np.zeros(X.shape[1])
            for _ in range(epochs):
                linear = X @ self.weights + self.bias
                pred = 1 / (1 + np.exp(-linear))
                error = pred - y
                self.weights -= lr * (X.T @ error) / len(y)
                self.bias -= lr * np.mean(error)
            return self

        def predict_proba(self, X):
            return 1 / (1 + np.exp(-(X @ self.weights + self.bias)))

        def predict(self, X):
            return (self.predict_proba(X) >= 0.5).astype(int)

        def score(self, X, y):
            return np.mean(self.predict(X) == y)

    # Simulated Quantum Model
    class SimulatedQuantumModel:
        def __init__(self, n_qubits=4):
            self.n_qubits = n_qubits
            self.params = np.random.randn(n_qubits * 2) * 0.1

        def quantum_kernel(self, x1, x2):
            """Simulated quantum kernel."""
            phase1 = np.sum(x1 * self.params[:len(x1)])
            phase2 = np.sum(x2 * self.params[:len(x2)])
            return np.cos(phase1 - phase2)**2

        def fit(self, X, y, lr=0.1, epochs=50):
            for _ in range(epochs):
                for x, label in zip(X, y):
                    pred = self.predict_proba(x.reshape(1, -1))[0]
                    error = pred - label

                    # Parameter update
                    for i in range(min(len(self.params), len(x))):
                        self.params[i] -= lr * error * x[i]
            return self

        def predict_proba(self, X):
            probs = []
            for x in X:
                # Kernel-based prediction
                score = np.sum(np.sin(x[:len(self.params)] * self.params))
                prob = 1 / (1 + np.exp(-score))
                probs.append(prob)
            return np.array(probs)

        def predict(self, X):
            return (self.predict_proba(X) >= 0.5).astype(int)

        def score(self, X, y):
            return np.mean(self.predict(X) == y)

    import time

    # Train and evaluate classical
    classical = ClassicalLogistic()
    start = time.time()
    classical.fit(X_train, y_train)
    classical_train_time = time.time() - start

    start = time.time()
    classical_pred = classical.predict_proba(X_test)
    classical_inference_time = time.time() - start

    classical_acc = classical.score(X_test, y_test)

    # Train and evaluate quantum
    quantum = SimulatedQuantumModel()
    start = time.time()
    quantum.fit(X_train, y_train)
    quantum_train_time = time.time() - start

    start = time.time()
    quantum_pred = quantum.predict_proba(X_test)
    quantum_inference_time = time.time() - start

    quantum_acc = quantum.score(X_test, y_test)

    # Calculate AUC for both
    def calc_auc(y_true, y_prob):
        sorted_idx = np.argsort(y_prob)[::-1]
        y_sorted = y_true[sorted_idx]
        n_pos = np.sum(y_true)
        n_neg = len(y_true) - n_pos
        if n_pos == 0 or n_neg == 0:
            return 0.5
        tpr = np.cumsum(y_sorted) / n_pos
        fpr = np.cumsum(1 - y_sorted) / n_neg
        return np.trapz(tpr, fpr)

    classical_auc = calc_auc(y_test, classical_pred)
    quantum_auc = calc_auc(y_test, quantum_pred)

    print("Performance Comparison:")
    print("-" * 40)
    print(f"{'Metric':<20} {'Classical':<12} {'Quantum':<12}")
    print("-" * 40)
    print(f"{'Accuracy':<20} {classical_acc:.4f}       {quantum_acc:.4f}")
    print(f"{'AUC':<20} {classical_auc:.4f}       {quantum_auc:.4f}")
    print(f"{'Train time (s)':<20} {classical_train_time:.4f}       {quantum_train_time:.4f}")
    print(f"{'Inference time (s)':<20} {classical_inference_time:.6f}   {quantum_inference_time:.6f}")

    return {
        'classical': {'acc': classical_acc, 'auc': classical_auc},
        'quantum': {'acc': quantum_acc, 'auc': quantum_auc}
    }
```

## Summary

Model evaluation for quantum credit scoring:

| Metric | Purpose | Threshold |
|--------|---------|-----------|
| AUC | Discrimination | > 0.70 |
| KS | Separation | > 0.30 |
| Gini | Ranking | > 0.40 |
| Brier | Calibration | < 0.25 |
| Disparate Impact | Fairness | ≥ 0.80 |

Key takeaways:

- Use credit-specific metrics (AUC, KS, Gini) not just accuracy
- Stratified cross-validation maintains class balance
- Fairness assessment is legally required for credit models
- Compare quantum vs classical fairly with same data splits
- Document model validation for regulatory compliance
