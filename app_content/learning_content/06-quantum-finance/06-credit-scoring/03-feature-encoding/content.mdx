# Quantum Feature Encoding

Encoding classical data into quantum states is crucial for quantum machine learning. This lesson covers encoding strategies tailored for credit scoring data.

<Definition term="Quantum Feature Encoding">
The process of mapping classical data (like credit features) into quantum states that can be processed by quantum algorithms. The choice of encoding significantly impacts algorithm performance and resource requirements.
</Definition>

## Encoding Strategies Overview

## Basis Encoding

Encode discrete values as computational basis states.

```python
def basis_encoding_demo():
    """Demonstrate basis encoding for categorical credit features."""

    print("Basis Encoding for Credit Data")
    print("=" * 50)

    # Example: Encode employment status
    employment_categories = {
        'employed': 0,
        'self_employed': 1,
        'unemployed': 2,
        'retired': 3
    }

    def encode_categorical(category, n_qubits):
        """Encode category as basis state."""
        value = employment_categories.get(category, 0)
        state = np.zeros(2**n_qubits, dtype=complex)
        state[value] = 1.0
        return state

    print("\nEmployment Status Encoding (2 qubits):")
    for cat, val in employment_categories.items():
        state = encode_categorical(cat, 2)
        binary = format(val, '02b')
        print(f"  {cat}: |{binary}⟩ = {state}")

    # Multi-feature basis encoding
    def encode_multiple_categorical(features, category_sizes):
        """Encode multiple categorical features."""
        total_qubits = sum(int(np.ceil(np.log2(max(s, 2)))) for s in category_sizes)
        state = np.zeros(2**total_qubits, dtype=complex)

        # Compute combined index
        index = 0
        shift = 0
        for feat, size in zip(features, category_sizes):
            n_bits = int(np.ceil(np.log2(max(size, 2))))
            index += feat << shift
            shift += n_bits

        state[index] = 1.0
        return state, total_qubits

    # Example with multiple features
    features = [1, 2, 0]  # self_employed, education=college, homeowner=no
    sizes = [4, 4, 2]     # 4 employment, 4 education, 2 homeowner

    state, n_qubits = encode_multiple_categorical(features, sizes)
    print(f"\nMulti-feature encoding:")
    print(f"  Features: {features}")
    print(f"  Category sizes: {sizes}")
    print(f"  Qubits needed: {n_qubits}")
    print(f"  Non-zero index: {np.argmax(np.abs(state))}")

    return state
```

## Amplitude Encoding

Encode entire feature vectors in quantum amplitudes.

```python
def amplitude_encoding_demo():
    """Demonstrate amplitude encoding for credit features."""

    print("Amplitude Encoding for Credit Features")
    print("=" * 55)

    def amplitude_encode(features):
        """
        Encode feature vector as quantum amplitudes.

        |ψ⟩ = Σ x_i |i⟩ / ||x||
        """
        n = len(features)
        n_qubits = int(np.ceil(np.log2(n)))
        dim = 2**n_qubits

        # Pad to power of 2
        padded = np.zeros(dim)
        padded[:n] = features

        # Normalize
        norm = np.linalg.norm(padded)
        if norm > 0:
            state = padded / norm
        else:
            state = np.zeros(dim)
            state[0] = 1.0

        return state, n_qubits

    # Credit feature example
    credit_features = np.array([
        0.7,   # Normalized income
        0.3,   # Debt-to-income ratio
        0.8,   # Credit score (normalized)
        0.5,   # Employment years (normalized)
        0.2,   # Number of defaults (normalized)
        0.6,   # Account age (normalized)
        0.4,   # Credit utilization
        0.9    # Payment history score
    ])

    print(f"Original features ({len(credit_features)} values):")
    feature_names = ['Income', 'DTI', 'Credit Score', 'Employment',
                     'Defaults', 'Account Age', 'Utilization', 'Payment History']
    for name, val in zip(feature_names, credit_features):
        print(f"  {name}: {val:.2f}")

    state, n_qubits = amplitude_encode(credit_features)

    print(f"\nAmplitude encoded state:")
    print(f"  Qubits needed: {n_qubits}")
    print(f"  State dimension: {len(state)}")
    print(f"  Amplitudes: {state}")
    print(f"  Sum of |amplitude|²: {np.sum(np.abs(state)**2):.6f}")

    # Show probability distribution
    print(f"\nProbability of measuring each basis state:")
    probs = np.abs(state)**2
    for i, (p, name) in enumerate(zip(probs, feature_names)):
        print(f"  |{format(i, '03b')}⟩ ({name}): {p:.4f}")

    return state, n_qubits
```

## Angle Encoding

Encode features as rotation angles on individual qubits.

```python
def angle_encoding_demo():
    """Demonstrate angle encoding for credit features."""

    print("Angle Encoding for Credit Features")
    print("=" * 50)

    def angle_encode(features, encoding_type='ry'):
        """
        Encode features as rotation angles.

        Each feature x_i encoded as R_y(πx_i)|0⟩ or similar
        """
        n_features = len(features)
        n_qubits = n_features
        dim = 2**n_qubits

        state = np.zeros(dim, dtype=complex)

        if encoding_type == 'ry':
            # Product state of Ry rotations
            for i in range(dim):
                amplitude = 1.0
                for q in range(n_qubits):
                    angle = features[q] * np.pi
                    bit = (i >> q) & 1
                    if bit == 0:
                        amplitude *= np.cos(angle / 2)
                    else:
                        amplitude *= np.sin(angle / 2)
                state[i] = amplitude

        elif encoding_type == 'rx':
            # Product state of Rx rotations
            for i in range(dim):
                amplitude = complex(1.0, 0.0)
                for q in range(n_qubits):
                    angle = features[q] * np.pi
                    bit = (i >> q) & 1
                    if bit == 0:
                        amplitude *= np.cos(angle / 2)
                    else:
                        amplitude *= -1j * np.sin(angle / 2)
                state[i] = amplitude

        return state, n_qubits

    # Simplified credit features
    features = np.array([0.3, 0.7, 0.5, 0.8])
    feature_names = ['Income', 'Debt Ratio', 'Credit Score', 'History']

    print(f"Features to encode:")
    for name, val in zip(feature_names, features):
        print(f"  {name}: {val:.2f}")

    state_ry, n_qubits = angle_encode(features, 'ry')

    print(f"\nRY Angle Encoding:")
    print(f"  Qubits: {n_qubits} (one per feature)")
    print(f"  State dimension: {len(state_ry)}")

    # Show some amplitudes
    print(f"\n  Key amplitudes:")
    print(f"    |0000⟩: {state_ry[0]:.4f} (all features low)")
    print(f"    |1111⟩: {state_ry[-1]:.4f} (all features high)")

    # Verify normalization
    print(f"\n  Normalization check: {np.sum(np.abs(state_ry)**2):.6f}")

    return state_ry

angle_encoding_demo()
```

<Tip>
Angle encoding is efficient when you have few features (≤20) because it requires one qubit per feature. For high-dimensional data, amplitude encoding is more qubit-efficient but harder to prepare.
</Tip>

## IQP-Style Encoding

Encode features with entanglement for richer representations.

```python
def iqp_encoding_demo():
    """Demonstrate IQP encoding with feature interactions."""

    print("IQP-Style Encoding with Interactions")
    print("=" * 55)

    def iqp_encode(features, n_layers=1):
        """
        IQP encoding: Hadamard + diagonal gates + entanglement.

        This creates feature interactions in the quantum state.
        """
        n_qubits = len(features)
        dim = 2**n_qubits

        # Start in uniform superposition (after Hadamard)
        state = np.ones(dim, dtype=complex) / np.sqrt(dim)

        for layer in range(n_layers):
            # Apply phase based on features (diagonal gates)
            for i in range(dim):
                phase = 0

                # Single-qubit terms: Z_q(x_q)
                for q in range(n_qubits):
                    if (i >> q) & 1:
                        phase += features[q]

                # Two-qubit terms: ZZ_qr(x_q * x_r)
                for q in range(n_qubits):
                    for r in range(q + 1, n_qubits):
                        if ((i >> q) & 1) and ((i >> r) & 1):
                            phase += features[q] * features[r]

                state[i] *= np.exp(1j * phase * np.pi)

        return state, n_qubits

    # Credit features
    features = np.array([0.5, 0.3, 0.8, 0.6])
    feature_names = ['Income', 'Debt', 'Score', 'History']

    print(f"Features: {dict(zip(feature_names, features))}")

    state, n_qubits = iqp_encode(features, n_layers=1)

    print(f"\nIQP Encoded State:")
    print(f"  Dimension: {len(state)}")

    # Show that all amplitudes have same magnitude (phase encoding)
    magnitudes = np.abs(state)
    print(f"  Amplitude magnitudes: all {magnitudes[0]:.4f} (uniform)")

    # Show phases
    phases = np.angle(state)
    print(f"\n  Phases encode feature information:")
    for i in range(min(8, dim)):
        print(f"    |{format(i, '04b')}⟩: phase = {phases[i]:.3f}π")

    # Kernel between similar and different samples
    features2 = np.array([0.5, 0.3, 0.8, 0.6])  # Same
    features3 = np.array([0.1, 0.9, 0.2, 0.1])  # Different

    state2, _ = iqp_encode(features2)
    state3, _ = iqp_encode(features3)

    kernel_same = np.abs(np.vdot(state, state2))**2
    kernel_diff = np.abs(np.vdot(state, state3))**2

    print(f"\n  Kernel with same features: {kernel_same:.4f}")
    print(f"  Kernel with different features: {kernel_diff:.4f}")

    return state
```

## Handling Mixed Feature Types

Credit data often has both categorical and continuous features.

```python
def mixed_feature_encoding():
    """Handle mixed categorical and continuous features."""

    print("Mixed Feature Encoding for Credit Data")
    print("=" * 55)

    class MixedEncoder:
        """Encode mixed categorical and continuous features."""

        def __init__(self, categorical_dims, n_continuous):
            self.cat_dims = categorical_dims
            self.n_continuous = n_continuous

            # Calculate qubits needed
            self.cat_qubits = sum(int(np.ceil(np.log2(max(d, 2))))
                                  for d in categorical_dims)
            self.cont_qubits = n_continuous
            self.total_qubits = self.cat_qubits + self.cont_qubits

        def encode(self, categorical_features, continuous_features):
            """Encode mixed features."""
            dim = 2**self.total_qubits

            # First encode categorical as basis
            cat_index = 0
            shift = 0
            for feat, size in zip(categorical_features, self.cat_dims):
                n_bits = int(np.ceil(np.log2(max(size, 2))))
                cat_index += int(feat) << shift
                shift += n_bits

            # Then angle-encode continuous
            state = np.zeros(dim, dtype=complex)

            for i in range(dim):
                # Check if categorical part matches
                cat_part = i % (2**self.cat_qubits)

                if cat_part == cat_index:
                    # Compute amplitude from continuous features
                    amplitude = 1.0
                    cont_part = i >> self.cat_qubits

                    for q in range(self.n_continuous):
                        angle = continuous_features[q] * np.pi
                        bit = (cont_part >> q) & 1
                        if bit == 0:
                            amplitude *= np.cos(angle / 2)
                        else:
                            amplitude *= np.sin(angle / 2)

                    state[i] = amplitude

            # Normalize
            norm = np.linalg.norm(state)
            if norm > 0:
                state /= norm

            return state

    # Example credit application
    # Categorical: employment (4 types), education (4 types), homeowner (2 types)
    categorical = [1, 2, 1]  # Self-employed, College, Homeowner
    categorical_dims = [4, 4, 2]

    # Continuous: income, debt_ratio, credit_score, years_employed
    continuous = [0.7, 0.3, 0.85, 0.5]

    encoder = MixedEncoder(categorical_dims, len(continuous))

    print(f"Categorical features: {categorical}")
    print(f"  Category dimensions: {categorical_dims}")
    print(f"Continuous features: {continuous}")
    print(f"\nQubits breakdown:")
    print(f"  Categorical: {encoder.cat_qubits}")
    print(f"  Continuous: {encoder.cont_qubits}")
    print(f"  Total: {encoder.total_qubits}")

    state = encoder.encode(categorical, continuous)

    print(f"\nEncoded state dimension: {len(state)}")
    print(f"Non-zero amplitudes: {np.sum(np.abs(state) > 1e-10)}")

    # Show largest amplitudes
    largest_idx = np.argsort(np.abs(state))[-5:][::-1]
    print(f"\nLargest amplitude states:")
    for idx in largest_idx:
        if np.abs(state[idx]) > 1e-10:
            print(f"  |{format(idx, f'0{encoder.total_qubits}b')}⟩: {state[idx]:.4f}")

    return encoder, state

mixed_feature_encoding()
```

<Warning>
The choice of encoding significantly impacts both quantum resources and model expressivity. Amplitude encoding is qubit-efficient but may require deep circuits; angle encoding is shallow but uses many qubits.
</Warning>

## Summary

Quantum feature encoding strategies for credit scoring:

| Encoding | Qubits | Depth | Best For |
|----------|--------|-------|----------|
| Basis | O(log N) | O(1) | Categorical |
| Amplitude | O(log N) | O(N) | High-dimensional |
| Angle | O(N) | O(1) | Few features |
| IQP | O(N) | O(N) | Feature interactions |

Key takeaways:

- Basis encoding efficiently represents categorical credit features
- Amplitude encoding compresses high-dimensional data but has loading overhead
- Angle encoding uses one qubit per feature with shallow circuits
- IQP encoding captures feature interactions through entanglement
- Mixed encoding handles real credit data with both types
- Encoding choice affects expressivity, resources, and trainability
