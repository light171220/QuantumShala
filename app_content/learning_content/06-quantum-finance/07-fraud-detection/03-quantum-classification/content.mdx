```python
class FraudFeatureEncoder:
    """
    Encode transaction features for quantum fraud detection.
    """

    def __init__(self, n_qubits: int = 8):
        self.n_qubits = n_qubits

    def prepare_features(
        self,
        amount: float,
        velocity: int,
        distance: float,
        hour: int,
        is_international: bool
    ) -> np.ndarray:
        """
        Prepare raw transaction features for quantum encoding.

        Normalizes and transforms features to [0, π] range.
        """
        # Normalize each feature
        features = []

        # Amount: log transform and normalize
        log_amount = np.log1p(amount)
        norm_amount = min(log_amount / 10, 1.0) * np.pi
        features.append(norm_amount)

        # Velocity: normalize to reasonable range
        norm_velocity = min(velocity / 20, 1.0) * np.pi
        features.append(norm_velocity)

        # Distance: log transform
        log_distance = np.log1p(distance)
        norm_distance = min(log_distance / 8, 1.0) * np.pi
        features.append(norm_distance)

        # Hour: circular encoding
        hour_angle = (hour / 24) * 2 * np.pi
        features.append(np.sin(hour_angle) * np.pi / 2 + np.pi / 2)
        features.append(np.cos(hour_angle) * np.pi / 2 + np.pi / 2)

        # International flag
        features.append(np.pi if is_international else 0)

        return np.array(features[:self.n_qubits])

    def encode_batch(
        self,
        transactions: List[Dict[str, Any]]
    ) -> np.ndarray:
        """Encode batch of transactions."""
        encoded = []
        for txn in transactions:
            features = self.prepare_features(
                amount=txn['amount'],
                velocity=txn['velocity'],
                distance=txn['distance'],
                hour=txn['hour'],
                is_international=txn['is_international']
            )
            encoded.append(features)
        return np.array(encoded)


class QuantumFeatureMap:
    """
    Quantum feature map for fraud detection.

    Maps classical features to quantum states.
    """

    def __init__(
        self,
        n_qubits: int,
        encoding_type: str = 'angle',
        entanglement: str = 'linear'
    ):
        self.n_qubits = n_qubits
        self.encoding_type = encoding_type
        self.entanglement = entanglement

    def encode(self, features: np.ndarray) -> np.ndarray:
        """
        Encode features into quantum state (statevector simulation).
        """
        # Initialize state
        state = np.zeros(2 ** self.n_qubits, dtype=complex)
        state[0] = 1.0

        # Apply encoding based on type
        if self.encoding_type == 'angle':
            state = self._angle_encoding(state, features)
        elif self.encoding_type == 'amplitude':
            state = self._amplitude_encoding(state, features)
        elif self.encoding_type == 'iqp':
            state = self._iqp_encoding(state, features)

        return state

    def _angle_encoding(
        self,
        state: np.ndarray,
        features: np.ndarray
    ) -> np.ndarray:
        """
        Angle encoding: RY(θ) rotations.
        """
        # Apply Hadamards
        for q in range(self.n_qubits):
            state = self._apply_h(state, q)

        # Apply rotations
        for q in range(min(len(features), self.n_qubits)):
            state = self._apply_ry(state, q, features[q])

        # Apply entanglement
        if self.entanglement == 'linear':
            for q in range(self.n_qubits - 1):
                state = self._apply_cnot(state, q, q + 1)
        elif self.entanglement == 'full':
            for q1 in range(self.n_qubits):
                for q2 in range(q1 + 1, self.n_qubits):
                    state = self._apply_cnot(state, q1, q2)

        return state

    def _amplitude_encoding(
        self,
        state: np.ndarray,
        features: np.ndarray
    ) -> np.ndarray:
        """
        Amplitude encoding: features become amplitudes.
        """
        n_amplitudes = 2 ** self.n_qubits
        padded_features = np.zeros(n_amplitudes)
        padded_features[:len(features)] = features

        # Normalize
        norm = np.linalg.norm(padded_features)
        if norm > 0:
            padded_features = padded_features / norm

        return padded_features.astype(complex)

    def _iqp_encoding(
        self,
        state: np.ndarray,
        features: np.ndarray
    ) -> np.ndarray:
        """
        Instantaneous Quantum Polynomial (IQP) encoding.

        Creates feature interactions through ZZ gates.
        """
        # Hadamards
        for q in range(self.n_qubits):
            state = self._apply_h(state, q)

        # Diagonal rotations
        for q in range(min(len(features), self.n_qubits)):
            state = self._apply_rz(state, q, features[q])

        # ZZ interactions
        for q1 in range(self.n_qubits - 1):
            for q2 in range(q1 + 1, min(q1 + 3, self.n_qubits)):
                if q1 < len(features) and q2 < len(features):
                    angle = features[q1] * features[q2]
                    state = self._apply_rzz(state, q1, q2, angle)

        # Final Hadamards
        for q in range(self.n_qubits):
            state = self._apply_h(state, q)

        return state

    def _apply_h(self, state: np.ndarray, qubit: int) -> np.ndarray:
        """Apply Hadamard gate."""
        h = np.array([[1, 1], [1, -1]]) / np.sqrt(2)
        return self._apply_single_qubit_gate(state, h, qubit)

    def _apply_ry(self, state: np.ndarray, qubit: int, theta: float) -> np.ndarray:
        """Apply RY rotation."""
        ry = np.array([
            [np.cos(theta/2), -np.sin(theta/2)],
            [np.sin(theta/2), np.cos(theta/2)]
        ])
        return self._apply_single_qubit_gate(state, ry, qubit)

    def _apply_rz(self, state: np.ndarray, qubit: int, theta: float) -> np.ndarray:
        """Apply RZ rotation."""
        rz = np.array([
            [np.exp(-1j * theta/2), 0],
            [0, np.exp(1j * theta/2)]
        ])
        return self._apply_single_qubit_gate(state, rz, qubit)

    def _apply_cnot(self, state: np.ndarray, control: int, target: int) -> np.ndarray:
        """Apply CNOT gate."""
        n = 2 ** self.n_qubits
        new_state = np.zeros_like(state)

        for i in range(n):
            if (i >> (self.n_qubits - 1 - control)) & 1:
                # Control is |1⟩, flip target
                j = i ^ (1 << (self.n_qubits - 1 - target))
                new_state[j] = state[i]
            else:
                new_state[i] = state[i]

        return new_state

    def _apply_rzz(
        self,
        state: np.ndarray,
        q1: int,
        q2: int,
        theta: float
    ) -> np.ndarray:
        """Apply RZZ rotation."""
        n = 2 ** self.n_qubits
        new_state = np.zeros_like(state)

        for i in range(n):
            bit1 = (i >> (self.n_qubits - 1 - q1)) & 1
            bit2 = (i >> (self.n_qubits - 1 - q2)) & 1
            parity = bit1 ^ bit2

            phase = np.exp(1j * theta / 2) if parity == 0 else np.exp(-1j * theta / 2)
            new_state[i] = state[i] * phase

        return new_state

    def _apply_single_qubit_gate(
        self,
        state: np.ndarray,
        gate: np.ndarray,
        qubit: int
    ) -> np.ndarray:
        """Apply single-qubit gate."""
        n = 2 ** self.n_qubits
        new_state = np.zeros_like(state)

        for i in range(n):
            bit = (i >> (self.n_qubits - 1 - qubit)) & 1
            i_flipped = i ^ (1 << (self.n_qubits - 1 - qubit))

            if bit == 0:
                new_state[i] += gate[0, 0] * state[i]
                new_state[i] += gate[0, 1] * state[i_flipped]
            else:
                new_state[i] += gate[1, 0] * state[i_flipped]
                new_state[i] += gate[1, 1] * state[i]

        return new_state
```

## Quantum Kernel Fraud Classifier

<Definition term="Quantum Kernel">
A function that computes similarity between data points in quantum feature space: K(x, y) = |⟨φ(x)|φ(y)⟩|², where |φ(x)⟩ is the quantum feature map.
</Definition>

### Implementation

```python
class QuantumKernelFraudClassifier:
    """
    Quantum kernel classifier for fraud detection.

    Uses quantum feature map to compute kernel and classical SVM for classification.
    """

    def __init__(
        self,
        n_qubits: int = 6,
        encoding_type: str = 'iqp',
        C: float = 1.0,
        class_weight: str = 'balanced'
    ):
        self.n_qubits = n_qubits
        self.encoding_type = encoding_type
        self.C = C
        self.class_weight = class_weight

        self.feature_map = QuantumFeatureMap(n_qubits, encoding_type)
        self.support_vectors = None
        self.support_labels = None
        self.alphas = None
        self.b = 0

    def quantum_kernel(self, x1: np.ndarray, x2: np.ndarray) -> float:
        """
        Compute quantum kernel between two samples.

        K(x1, x2) = |⟨φ(x1)|φ(x2)⟩|²
        """
        state1 = self.feature_map.encode(x1)
        state2 = self.feature_map.encode(x2)

        # Inner product
        overlap = np.abs(np.vdot(state1, state2)) ** 2

        return overlap

    def compute_kernel_matrix(self, X1: np.ndarray, X2: np.ndarray = None) -> np.ndarray:
        """Compute kernel matrix."""
        if X2 is None:
            X2 = X1

        n1, n2 = len(X1), len(X2)
        K = np.zeros((n1, n2))

        for i in range(n1):
            for j in range(n2):
                K[i, j] = self.quantum_kernel(X1[i], X2[j])

        return K

    def fit(self, X: np.ndarray, y: np.ndarray):
        """
        Train quantum kernel classifier.

        Uses simplified SVM with precomputed kernel.
        """
        n_samples = len(X)

        # Compute kernel matrix
        print("Computing quantum kernel matrix...")
        K = self.compute_kernel_matrix(X)

        # Handle class imbalance
        if self.class_weight == 'balanced':
            class_counts = np.bincount(y.astype(int))
            weights = n_samples / (len(class_counts) * class_counts)
            sample_weights = weights[y.astype(int)]
        else:
            sample_weights = np.ones(n_samples)

        # Simplified kernel SVM using gradient descent
        self.alphas = np.zeros(n_samples)
        self.support_vectors = X
        self.support_labels = y

        # Train with SMO-like updates
        for iteration in range(100):
            for i in range(n_samples):
                # Compute prediction
                pred = np.sum(self.alphas * y * K[:, i]) + self.b

                # Update if misclassified
                error = y[i] - pred
                if y[i] * pred < 1:
                    self.alphas[i] += self.C * error * sample_weights[i]
                    self.alphas[i] = np.clip(self.alphas[i], 0, self.C * sample_weights[i])

            # Update bias
            support_mask = self.alphas > 1e-5
            if support_mask.any():
                predictions = K[:, support_mask] @ (self.alphas[support_mask] * y[support_mask])
                self.b = np.mean(y[support_mask] - predictions[support_mask])

        return self

    def predict(self, X: np.ndarray) -> np.ndarray:
        """Predict fraud labels."""
        K = self.compute_kernel_matrix(X, self.support_vectors)
        predictions = K @ (self.alphas * self.support_labels) + self.b
        return np.sign(predictions).astype(int)

    def predict_proba(self, X: np.ndarray) -> np.ndarray:
        """Predict fraud probabilities."""
        K = self.compute_kernel_matrix(X, self.support_vectors)
        scores = K @ (self.alphas * self.support_labels) + self.b

        # Convert to probabilities using sigmoid
        proba = 1 / (1 + np.exp(-scores))
        return proba


# Example usage
```

## Variational Quantum Classifier

<Warning>

</Warning>

### VQC for Fraud Detection

```python
class VariationalQuantumClassifier:
    """
    Variational Quantum Classifier for fraud detection.

    Learns optimal circuit parameters to classify transactions.
    """

    def __init__(
        self,
        n_qubits: int = 4,
        n_layers: int = 2,
        learning_rate: float = 0.1,
        epochs: int = 50
    ):
        self.n_qubits = n_qubits
        self.n_layers = n_layers
        self.learning_rate = learning_rate
        self.epochs = epochs

        # Initialize parameters
        n_params = n_layers * n_qubits * 3  # RY, RZ, RY per qubit per layer
        self.params = np.random.uniform(-np.pi, np.pi, n_params)
        self.threshold = 0.5

    def forward(self, features: np.ndarray) -> float:
        """
        Forward pass through VQC.

        Returns probability of fraud class.
        """
        # Initialize state
        state = np.zeros(2 ** self.n_qubits, dtype=complex)
        state[0] = 1.0

        # Encoding layer
        for q in range(min(len(features), self.n_qubits)):
            state = self._apply_ry(state, q, features[q])

        # Variational layers
        param_idx = 0
        for layer in range(self.n_layers):
            # Single-qubit rotations
            for q in range(self.n_qubits):
                state = self._apply_ry(state, q, self.params[param_idx])
                param_idx += 1
                state = self._apply_rz(state, q, self.params[param_idx])
                param_idx += 1
                state = self._apply_ry(state, q, self.params[param_idx])
                param_idx += 1

            # Entanglement
            for q in range(self.n_qubits - 1):
                state = self._apply_cnot(state, q, q + 1)
            if self.n_qubits > 1:
                state = self._apply_cnot(state, self.n_qubits - 1, 0)

        # Measure first qubit (fraud probability)
        prob_one = 0.0
        for i in range(2 ** self.n_qubits):
            if (i >> (self.n_qubits - 1)) & 1:
                prob_one += np.abs(state[i]) ** 2

        return prob_one

    def _apply_ry(self, state: np.ndarray, qubit: int, theta: float) -> np.ndarray:
        """Apply RY rotation."""
        n = 2 ** self.n_qubits
        new_state = np.zeros_like(state)

        cos_half = np.cos(theta / 2)
        sin_half = np.sin(theta / 2)

        for i in range(n):
            bit = (i >> (self.n_qubits - 1 - qubit)) & 1
            i_flipped = i ^ (1 << (self.n_qubits - 1 - qubit))

            if bit == 0:
                new_state[i] += cos_half * state[i]
                new_state[i] += sin_half * state[i_flipped]
            else:
                new_state[i] += -sin_half * state[i_flipped]
                new_state[i] += cos_half * state[i]

        return new_state

    def _apply_rz(self, state: np.ndarray, qubit: int, theta: float) -> np.ndarray:
        """Apply RZ rotation."""
        n = 2 ** self.n_qubits
        new_state = np.zeros_like(state)

        for i in range(n):
            bit = (i >> (self.n_qubits - 1 - qubit)) & 1
            phase = np.exp(-1j * theta / 2) if bit == 0 else np.exp(1j * theta / 2)
            new_state[i] = state[i] * phase

        return new_state

    def _apply_cnot(self, state: np.ndarray, control: int, target: int) -> np.ndarray:
        """Apply CNOT gate."""
        n = 2 ** self.n_qubits
        new_state = np.zeros_like(state)

        for i in range(n):
            if (i >> (self.n_qubits - 1 - control)) & 1:
                j = i ^ (1 << (self.n_qubits - 1 - target))
                new_state[j] = state[i]
            else:
                new_state[i] = state[i]

        return new_state

    def compute_gradient(
        self,
        features: np.ndarray,
        label: int,
        epsilon: float = 0.01
    ) -> np.ndarray:
        """
        Compute gradient using parameter-shift rule.
        """
        gradients = np.zeros_like(self.params)

        for i in range(len(self.params)):
            # Positive shift
            self.params[i] += np.pi / 2
            prob_plus = self.forward(features)

            # Negative shift
            self.params[i] -= np.pi
            prob_minus = self.forward(features)

            # Restore
            self.params[i] += np.pi / 2

            # Gradient of cross-entropy loss
            pred = (prob_plus + prob_minus) / 2
            pred = np.clip(pred, 1e-7, 1 - 1e-7)

            gradient_shift = (prob_plus - prob_minus) / 2
            loss_gradient = -(label / pred - (1 - label) / (1 - pred))

            gradients[i] = loss_gradient * gradient_shift

        return gradients

    def fit(self, X: np.ndarray, y: np.ndarray, verbose: bool = True):
        """Train VQC."""
        n_samples = len(X)

        for epoch in range(self.epochs):
            total_loss = 0
            correct = 0

            for i in range(n_samples):
                # Forward
                prob = self.forward(X[i])
                prob = np.clip(prob, 1e-7, 1 - 1e-7)

                # Loss
                loss = -(y[i] * np.log(prob) + (1 - y[i]) * np.log(1 - prob))
                total_loss += loss

                # Accuracy
                pred = 1 if prob > self.threshold else 0
                if pred == y[i]:
                    correct += 1

                # Gradient update
                gradients = self.compute_gradient(X[i], y[i])
                self.params -= self.learning_rate * gradients

            if verbose and (epoch + 1) % 10 == 0:
                acc = correct / n_samples
                avg_loss = total_loss / n_samples
                print(f"Epoch {epoch+1}/{self.epochs}, Loss: {avg_loss:.4f}, Acc: {acc:.3f}")

        return self

    def predict(self, X: np.ndarray) -> np.ndarray:
        """Predict fraud labels."""
        predictions = []
        for x in X:
            prob = self.forward(x)
            predictions.append(1 if prob > self.threshold else 0)
        return np.array(predictions)

    def predict_proba(self, X: np.ndarray) -> np.ndarray:
        """Predict fraud probabilities."""
        return np.array([self.forward(x) for x in X])
```

## Handling Imbalanced Data

<Formula>

</Formula>

### Cost-Sensitive Quantum Learning

```python
class BalancedQuantumClassifier:
    """
    Quantum classifier with class balancing for fraud detection.
    """

    def __init__(
        self,
        n_qubits: int = 4,
        fraud_weight: float = 10.0,
        n_layers: int = 2
    ):
        self.n_qubits = n_qubits
        self.fraud_weight = fraud_weight
        self.n_layers = n_layers
        self.vqc = VariationalQuantumClassifier(n_qubits, n_layers)

    def fit(self, X: np.ndarray, y: np.ndarray):
        """
        Train with class-weighted sampling.
        """
        n_fraud = y.sum()
        n_normal = len(y) - n_fraud

        # Oversample fraud to balance
        fraud_indices = np.where(y == 1)[0]
        normal_indices = np.where(y == 0)[0]

        # Repeat fraud samples to approximate balance
        repeat_factor = int(self.fraud_weight * n_normal / (n_fraud + 1e-8))
        repeat_factor = min(repeat_factor, 10)  # Cap to avoid too much oversampling

        if repeat_factor > 1:
            fraud_repeated = np.tile(fraud_indices, repeat_factor)
            all_indices = np.hstack([normal_indices, fraud_repeated])
        else:
            all_indices = np.arange(len(y))

        np.random.shuffle(all_indices)

        X_balanced = X[all_indices]
        y_balanced = y[all_indices]

        # Train VQC
        self.vqc.fit(X_balanced, y_balanced)

        return self

    def predict(self, X: np.ndarray) -> np.ndarray:
        """Predict with adjusted threshold for fraud detection."""
        proba = self.vqc.predict_proba(X)
        # Lower threshold to catch more fraud
        threshold = 0.3
        return (proba > threshold).astype(int)

    def predict_proba(self, X: np.ndarray) -> np.ndarray:
        return self.vqc.predict_proba(X)
```

```python
class ModelComparison:
    """
    Compare quantum and classical fraud detection performance.
    """

    @staticmethod
    def evaluate_model(
        model,
        X_test: np.ndarray,
        y_test: np.ndarray
    ) -> Dict[str, float]:
        """Evaluate model performance."""
        predictions = model.predict(X_test)

        # Convert labels if needed
        y_test_binary = (y_test > 0).astype(int)
        pred_binary = (predictions > 0).astype(int)

        tp = ((y_test_binary == 1) & (pred_binary == 1)).sum()
        tn = ((y_test_binary == 0) & (pred_binary == 0)).sum()
        fp = ((y_test_binary == 0) & (pred_binary == 1)).sum()
        fn = ((y_test_binary == 1) & (pred_binary == 0)).sum()

        precision = tp / (tp + fp) if (tp + fp) > 0 else 0
        recall = tp / (tp + fn) if (tp + fn) > 0 else 0
        f1 = 2 * precision * recall / (precision + recall) if (precision + recall) > 0 else 0
        accuracy = (tp + tn) / len(y_test)

        return {
            'accuracy': accuracy,
            'precision': precision,
            'recall': recall,
            'f1': f1,
            'tp': tp,
            'fp': fp,
            'fn': fn
        }

    @staticmethod
    def compare_models(
        quantum_model,
        classical_model,
        X_test: np.ndarray,
        y_test: np.ndarray
    ):
        """Compare quantum and classical models."""
        quantum_metrics = ModelComparison.evaluate_model(quantum_model, X_test, y_test)
        classical_metrics = ModelComparison.evaluate_model(classical_model, X_test, y_test)

        print("Model Comparison:")
        print("-" * 50)
        print(f"{'Metric':<15} {'Quantum':<15} {'Classical':<15}")
        print("-" * 50)

        for metric in ['accuracy', 'precision', 'recall', 'f1']:
            q_val = quantum_metrics[metric]
            c_val = classical_metrics[metric]
            better = '←' if q_val > c_val else '→' if c_val > q_val else '='
            print(f"{metric:<15} {q_val:<15.3f} {c_val:<15.3f} {better}")

        return quantum_metrics, classical_metrics
```
