## Case Study 1: Banking Card Fraud Detection

A major international bank implemented quantum-enhanced fraud detection for their credit card operations processing 50 million transactions daily.

### Challenge Profile

"""
Case Study: International Bank Card Fraud Detection
Processing: 50M transactions/day
Fraud Rate: 0.1% (50,000 fraudulent transactions/day)
False Positive Cost: $5 per declined legitimate transaction
Fraud Loss: Average $500 per successful fraud
SLA: < 100ms decision time
"""

from dataclasses import dataclass
from typing import Dict, List, Optional, Tuple
from enum import Enum

class FraudType(Enum):
    CARD_NOT_PRESENT = "cnp"
    COUNTERFEIT = "counterfeit"
    LOST_STOLEN = "lost_stolen"
    ACCOUNT_TAKEOVER = "ato"
    FRIENDLY_FRAUD = "friendly"

@dataclass
class BankingCaseStudy:
    """
    Case study analysis for banking fraud detection.
    """

    # Problem metrics
    daily_transactions: int = 50_000_000
    fraud_rate: float = 0.001  # 0.1%
    false_positive_cost: float = 5.0
    fraud_loss_average: float = 500.0
    latency_sla_ms: float = 100.0

    # Fraud distribution
    fraud_distribution: Dict[FraudType, float] = None

    def __post_init__(self):
        self.fraud_distribution = {
            FraudType.CARD_NOT_PRESENT: 0.45,  # 45% - Growing rapidly
            FraudType.COUNTERFEIT: 0.20,
            FraudType.LOST_STOLEN: 0.15,
            FraudType.ACCOUNT_TAKEOVER: 0.12,
            FraudType.FRIENDLY_FRAUD: 0.08
        }

    def calculate_daily_impact(self,
                               detection_rate: float,
                               false_positive_rate: float) -> Dict[str, float]:
        """
        Calculate daily financial impact of fraud detection system.
        """
        daily_frauds = self.daily_transactions * self.fraud_rate
        daily_legitimate = self.daily_transactions * (1 - self.fraud_rate)

        # Fraud detection outcomes
        caught_frauds = daily_frauds * detection_rate
        missed_frauds = daily_frauds * (1 - detection_rate)

        # False positive impact
        false_positives = daily_legitimate * false_positive_rate

        # Financial calculations
        fraud_losses = missed_frauds * self.fraud_loss_average
        fp_costs = false_positives * self.false_positive_cost
        fraud_savings = caught_frauds * self.fraud_loss_average

        return {
            "daily_frauds": daily_frauds,
            "caught_frauds": caught_frauds,
            "missed_frauds": missed_frauds,
            "false_positives": false_positives,
            "fraud_losses": fraud_losses,
            "fp_costs": fp_costs,
            "fraud_savings": fraud_savings,
            "net_daily_loss": fraud_losses + fp_costs,
            "system_value": fraud_savings - fp_costs
        }

    def compare_systems(self,
                        classical_dr: float,
                        classical_fpr: float,
                        quantum_dr: float,
                        quantum_fpr: float) -> Dict[str, Dict]:
        """
        Compare classical vs quantum-enhanced system performance.
        """
        classical = self.calculate_daily_impact(classical_dr, classical_fpr)
        quantum = self.calculate_daily_impact(quantum_dr, quantum_fpr)

        improvement = {
            key: quantum[key] - classical[key]
            for key in classical
        }

        return {
            "classical": classical,
            "quantum": quantum,
            "improvement": improvement,
            "quantum_advantage_pct": (
                (quantum["system_value"] - classical["system_value"]) /
                classical["system_value"] * 100
                if classical["system_value"] > 0 else 0
            )
        }

import numpy as np
from typing import Dict, List, Optional, Tuple
from dataclasses import dataclass

@dataclass
class BankingFraudFeatures:
    """
    Feature engineering specific to banking fraud detection.
    """

    # Transaction features
    amount: float
    amount_zscore: float  # vs customer average

    # Velocity features
    tx_count_1h: int
    tx_count_24h: int
    tx_amount_1h: float
    tx_amount_24h: float

    # Geographic features
    distance_from_home: float
    is_foreign_transaction: bool
    impossible_travel_flag: bool

    # Merchant features
    merchant_risk_score: float
    is_new_merchant: bool
    mcc_risk_category: str

    # Device/channel features
    is_card_present: bool
    device_fingerprint_match: bool
    ip_risk_score: float

    # Behavioral features
    hour_of_day_typicality: float
    day_of_week_typicality: float
    spending_category_typicality: float

    def to_quantum_features(self) -> np.ndarray:
        """
        Prepare features for quantum circuit encoding.
        Normalize to [0, π] range.
        """
        raw = np.array([
            self.amount / 10000,  # Assume max ~$10k
            (self.amount_zscore + 3) / 6,  # z-score typically in [-3, 3]
            self.tx_count_1h / 20,
            self.tx_count_24h / 100,
            self.tx_amount_1h / 5000,
            self.tx_amount_24h / 20000,
            self.distance_from_home / 5000,  # km
            float(self.is_foreign_transaction),
            float(self.impossible_travel_flag),
            self.merchant_risk_score,
            float(self.is_new_merchant),
            float(self.is_card_present),
            float(self.device_fingerprint_match),
            self.ip_risk_score,
            self.hour_of_day_typicality,
            self.day_of_week_typicality
        ])

        # Clip to [0, 1] and scale to [0, π]
        return np.clip(raw, 0, 1) * np.pi


class BankingFraudPipeline:
    """
    Production fraud detection pipeline for banking.
    Uses tiered architecture for latency management.
    """

    def __init__(self):
        self.rule_engine = RuleBasedFilter()
        self.ml_scorer = MLFraudScorer()
        self.quantum_analyzer = QuantumPatternAnalyzer()
        self.decision_engine = DecisionEngine()

    def process_transaction(self,
                           features: BankingFraudFeatures) -> Dict:
        """
        Process a transaction through the fraud detection pipeline.
        """
        result = {
            "stages_executed": [],
            "total_latency_ms": 0
        }

        # Stage 1: Rule-based pre-filtering (< 1ms)
        rule_result = self.rule_engine.evaluate(features)
        result["stages_executed"].append("rules")
        result["rule_score"] = rule_result["score"]

        # Auto-decline on hard rules (e.g., blocked merchant)
        if rule_result["hard_decline"]:
            return self._finalize(result, "decline", "rule_block", 1.0)

        # Auto-approve on safe transactions
        if rule_result["safe_pass"]:
            return self._finalize(result, "approve", "rule_pass", 0.95)

        # Stage 2: ML scoring (5-10ms)
        ml_result = self.ml_scorer.score(features)
        result["stages_executed"].append("ml")
        result["ml_score"] = ml_result["risk_score"]

        # Clear decisions based on ML
        if ml_result["risk_score"] < 0.1:
            return self._finalize(result, "approve", "ml_low_risk", 0.90)
        if ml_result["risk_score"] > 0.9:
            return self._finalize(result, "decline", "ml_high_risk", 0.92)

        # Stage 3: Quantum analysis for uncertain cases (30-50ms)
        if self._should_use_quantum(features, ml_result):
            quantum_result = self.quantum_analyzer.analyze(features)
            result["stages_executed"].append("quantum")
            result["quantum_score"] = quantum_result["fraud_probability"]

            # Combine ML and quantum predictions
            combined = self._combine_predictions(ml_result, quantum_result)
            decision = "decline" if combined["risk"] > 0.5 else "approve"
            return self._finalize(result, decision, "quantum_hybrid",
                                combined["confidence"])

        # Fallback to ML-only decision
        decision = "decline" if ml_result["risk_score"] > 0.5 else "approve"
        return self._finalize(result, decision, "ml_only",
                            0.7 + abs(ml_result["risk_score"] - 0.5) * 0.4)

    def _should_use_quantum(self,
                           features: BankingFraudFeatures,
                           ml_result: Dict) -> bool:
        """
        Determine if quantum analysis is warranted.
        """
        # High value transactions
        if features.amount > 2000:
            return True

        # ML uncertainty zone
        if 0.3 < ml_result["risk_score"] < 0.7:
            return True

        # Complex pattern indicators
        if features.impossible_travel_flag:
            return True

        return False

    def _combine_predictions(self,
                            ml_result: Dict,
                            quantum_result: Dict) -> Dict:
        """
        Combine ML and quantum predictions using weighted average.
        """
        ml_weight = 0.4
        quantum_weight = 0.6  # Quantum gets higher weight for complex cases

        combined_risk = (
            ml_result["risk_score"] * ml_weight +
            quantum_result["fraud_probability"] * quantum_weight
        )

        # Confidence based on agreement
        agreement = 1 - abs(ml_result["risk_score"] -
                           quantum_result["fraud_probability"])
        confidence = 0.7 + agreement * 0.25

        return {"risk": combined_risk, "confidence": confidence}

    def _finalize(self, result: Dict, decision: str,
                  reason: str, confidence: float) -> Dict:
        result["decision"] = decision
        result["reason"] = reason
        result["confidence"] = confidence
        return result


class RuleBasedFilter:
    """Hard rules for immediate decisions."""

    def evaluate(self, features: BankingFraudFeatures) -> Dict:
        # Blocked patterns
        if features.impossible_travel_flag and features.amount > 1000:
            return {"hard_decline": True, "safe_pass": False, "score": 1.0}

        # Safe patterns
        if (features.amount < 25 and
            features.is_card_present and
            features.device_fingerprint_match):
            return {"hard_decline": False, "safe_pass": True, "score": 0.01}

        return {"hard_decline": False, "safe_pass": False, "score": 0.5}


class MLFraudScorer:
    """Machine learning fraud scorer."""

    def score(self, features: BankingFraudFeatures) -> Dict:
        # Simplified scoring (production would use trained model)
        risk = (
            (features.amount / 10000) * 0.2 +
            min(features.tx_count_1h / 20, 1) * 0.2 +
            (1 - features.hour_of_day_typicality) * 0.15 +
            features.merchant_risk_score * 0.15 +
            features.ip_risk_score * 0.15 +
            (1 - float(features.device_fingerprint_match)) * 0.15
        )
        return {"risk_score": min(risk, 1.0)}


class QuantumPatternAnalyzer:
    """Quantum-enhanced pattern analysis."""

    def analyze(self, features: BankingFraudFeatures) -> Dict:
        # Simulated quantum analysis
        quantum_features = features.to_quantum_features()

        # Would execute quantum circuit here
        fraud_prob = np.mean(np.sin(quantum_features) ** 2) * 0.5 + 0.25

        return {
            "fraud_probability": fraud_prob,
            "pattern_type": "complex" if fraud_prob > 0.5 else "normal",
            "quantum_confidence": 0.85
        }

    Quantum algorithms excel at graph pattern matching.
    """

        # Build graph of relationships
        self.user_device_edges: Dict[str, Set[str]] = defaultdict(set)
        self.device_user_edges: Dict[str, Set[str]] = defaultdict(set)
        self.user_payment_edges: Dict[str, Set[str]] = defaultdict(set)
        self.payment_user_edges: Dict[str, Set[str]] = defaultdict(set)
        self.user_address_edges: Dict[str, Set[str]] = defaultdict(set)
        self.address_user_edges: Dict[str, Set[str]] = defaultdict(set)

    def add_transaction(self,
                        user_id: str,
                        device_id: str,
                        payment_method: str,
                        shipping_address: str):
        """Add transaction to the graph."""
        self.user_device_edges[user_id].add(device_id)
        self.device_user_edges[device_id].add(user_id)

        self.user_payment_edges[user_id].add(payment_method)
        self.payment_user_edges[payment_method].add(user_id)

        self.user_address_edges[user_id].add(shipping_address)
        self.address_user_edges[shipping_address].add(user_id)

    def find_connected_users(self,
                             user_id: str,
                             max_hops: int = 2) -> Set[str]:
        """
        Find all users connected through shared entities.
        """
        connected = {user_id}
        frontier = {user_id}

        for hop in range(max_hops):
            next_frontier = set()

            for user in frontier:
                # Through devices
                for device in self.user_device_edges.get(user, set()):
                    next_frontier.update(self.device_user_edges[device])

                # Through payment methods
                for payment in self.user_payment_edges.get(user, set()):
                    next_frontier.update(self.payment_user_edges[payment])

                # Through addresses
                for address in self.user_address_edges.get(user, set()):
                    next_frontier.update(self.address_user_edges[address])

            next_frontier -= connected
            connected.update(next_frontier)
            frontier = next_frontier

        return connected

    def calculate_ring_risk_score(self, user_ids: Set[str]) -> Dict:
        """
        Calculate fraud ring risk score for a cluster of users.
        """
        if len(user_ids) < 2:
            return {"is_ring": False, "score": 0.0}

        # Count shared entities
        shared_devices = self._count_shared_entities(
            user_ids, self.user_device_edges
        )
        shared_payments = self._count_shared_entities(
            user_ids, self.user_payment_edges
        )
        shared_addresses = self._count_shared_entities(
            user_ids, self.user_address_edges
        )

        # Ring indicators
        indicators = {
            "cluster_size": len(user_ids),
            "shared_devices": shared_devices,
            "shared_payments": shared_payments,
            "shared_addresses": shared_addresses,
            "density": self._calculate_density(user_ids)
        }

        # Risk scoring
        risk_score = (
            min(len(user_ids) / 10, 1.0) * 0.3 +
            min(shared_devices / 5, 1.0) * 0.25 +
            min(shared_payments / 3, 1.0) * 0.25 +
            min(shared_addresses / 3, 1.0) * 0.2
        )

        return {
            "is_ring": risk_score > 0.5,
            "score": risk_score,
            "indicators": indicators,
            "recommended_action": self._get_recommendation(risk_score)
        }

    def _count_shared_entities(self,
                               user_ids: Set[str],
                               edge_map: Dict[str, Set[str]]) -> int:
        """Count entities shared by multiple users."""
        entity_counts = defaultdict(int)
        for user_id in user_ids:
            for entity in edge_map.get(user_id, set()):
                entity_counts[entity] += 1

        return sum(1 for count in entity_counts.values() if count > 1)

    def _calculate_density(self, user_ids: Set[str]) -> float:
        """Calculate graph density of the cluster."""
        if len(user_ids) < 2:
            return 0.0

        total_edges = 0
        for user_id in user_ids:
            connected = self.find_connected_users(user_id, max_hops=1)
            total_edges += len(connected & user_ids) - 1

        max_edges = len(user_ids) * (len(user_ids) - 1)
        return total_edges / max_edges if max_edges > 0 else 0.0

    def _get_recommendation(self, risk_score: float) -> str:
        if risk_score > 0.8:
            return "block_cluster"
        elif risk_score > 0.5:
            return "manual_review"
        elif risk_score > 0.3:
            return "enhanced_monitoring"
        else:
            return "normal_processing"
