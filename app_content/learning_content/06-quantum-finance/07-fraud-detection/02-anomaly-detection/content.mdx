# Anomaly Detection for Fraud

Anomaly detection identifies unusual patterns that don't conform to expected behavior. This is valuable for fraud detection because fraudulent transactions are, by definition, abnormal.

## Why Anomaly Detection?

<Definition term="Anomaly Detection">
The identification of rare items, events, or observations that differ significantly from the majority of data, which may indicate fraud, errors, or novel patterns.
</Definition>

### Advantages for Fraud Detection

## Statistical Anomaly Detection

<Tip>
Statistical methods are interpretable and fast. They work well for univariate or low-dimensional data where the underlying distribution is understood.
</Tip>

### Z-Score and IQR Methods

```python
class StatisticalAnomalyDetector:
    """
    Statistical methods for anomaly detection.
    """

    def __init__(self, method: str = 'zscore', threshold: float = 3.0):
        """
        Args:
            method: 'zscore', 'iqr', or 'mad'
            threshold: Number of standard deviations (zscore/mad) or IQR multiplier
        """
        self.method = method
        self.threshold = threshold
        self.stats = {}

    def fit(self, X: np.ndarray, feature_names: List[str] = None):
        """Learn statistics from training data."""
        self.feature_names = feature_names or [f"f{i}" for i in range(X.shape[1])]

        for i, name in enumerate(self.feature_names):
            feature_data = X[:, i]

            if self.method == 'zscore':
                self.stats[name] = {
                    'mean': np.mean(feature_data),
                    'std': np.std(feature_data)
                }
            elif self.method == 'iqr':
                q1 = np.percentile(feature_data, 25)
                q3 = np.percentile(feature_data, 75)
                iqr = q3 - q1
                self.stats[name] = {
                    'q1': q1,
                    'q3': q3,
                    'iqr': iqr,
                    'lower': q1 - self.threshold * iqr,
                    'upper': q3 + self.threshold * iqr
                }
            elif self.method == 'mad':
                median = np.median(feature_data)
                mad = np.median(np.abs(feature_data - median))
                self.stats[name] = {
                    'median': median,
                    'mad': mad
                }

        return self

    def score(self, X: np.ndarray) -> np.ndarray:
        """Calculate anomaly scores."""
        n_samples = X.shape[0]
        scores = np.zeros(n_samples)

        for i, name in enumerate(self.feature_names):
            feature_data = X[:, i]

            if self.method == 'zscore':
                mean = self.stats[name]['mean']
                std = self.stats[name]['std'] + 1e-8
                z_scores = np.abs((feature_data - mean) / std)
                scores += z_scores

            elif self.method == 'iqr':
                lower = self.stats[name]['lower']
                upper = self.stats[name]['upper']
                iqr = self.stats[name]['iqr'] + 1e-8

                # Distance from bounds, normalized by IQR
                below = np.maximum(0, lower - feature_data) / iqr
                above = np.maximum(0, feature_data - upper) / iqr
                scores += below + above

            elif self.method == 'mad':
                median = self.stats[name]['median']
                mad = self.stats[name]['mad'] + 1e-8
                mad_scores = np.abs(feature_data - median) / (1.4826 * mad)
                scores += mad_scores

        # Average across features
        scores = scores / len(self.feature_names)
        return scores

    def predict(self, X: np.ndarray) -> np.ndarray:
        """Predict anomalies."""
        scores = self.score(X)
        return (scores > self.threshold).astype(int)

    def explain(self, x: np.ndarray) -> Dict[str, float]:
        """Explain which features are anomalous."""
        explanations = {}

        for i, name in enumerate(self.feature_names):
            value = x[i]

            if self.method == 'zscore':
                mean = self.stats[name]['mean']
                std = self.stats[name]['std'] + 1e-8
                z = (value - mean) / std
                explanations[name] = {
                    'value': value,
                    'z_score': z,
                    'is_anomaly': abs(z) > self.threshold
                }

            elif self.method == 'iqr':
                lower = self.stats[name]['lower']
                upper = self.stats[name]['upper']
                is_low = value < lower
                is_high = value > upper
                explanations[name] = {
                    'value': value,
                    'bounds': (lower, upper),
                    'is_anomaly': is_low or is_high
                }

        return explanations


# Example usage
np.random.seed(42)

# Generate normal transaction data
n_samples = 1000
amounts = np.random.exponential(100, n_samples)
velocities = np.random.poisson(3, n_samples)
distances = np.random.exponential(20, n_samples)

X_train = np.column_stack([amounts, velocities, distances])

# Fit detector

detector.fit(X_train, ['amount', 'velocity', 'distance'])

# Test anomalous transaction

print(f"Normal score: {detector.score(normal)[0]:.2f}")
print(f"\nExplanation: {detector.explain(anomalous[0])}")
```

## Isolation Forest

<Definition term="Isolation Forest">
An algorithm that isolates anomalies by randomly selecting features and split values. Anomalies require fewer splits to isolate, resulting in shorter path lengths.
</Definition>

### Implementation

```python
class IsolationTree:
    """Single tree in an isolation forest."""

    def __init__(self, max_depth: int, random_state: int = None):
        self.max_depth = max_depth
        self.rng = np.random.RandomState(random_state)
        self.root = None
        self.n_samples = 0

    def fit(self, X: np.ndarray):
        """Build isolation tree."""
        self.n_samples = len(X)
        self.root = self._build_tree(X, depth=0)
        return self

    def _build_tree(self, X: np.ndarray, depth: int) -> Dict:
        """Recursively build tree."""
        n_samples, n_features = X.shape

        # Stopping conditions
        if depth >= self.max_depth or n_samples <= 1:
            return {'type': 'leaf', 'size': n_samples}

        # Random feature and split
        feature = self.rng.randint(n_features)
        min_val = X[:, feature].min()
        max_val = X[:, feature].max()

        if min_val == max_val:
            return {'type': 'leaf', 'size': n_samples}

        split_value = self.rng.uniform(min_val, max_val)

        # Split data
        left_mask = X[:, feature] < split_value
        right_mask = ~left_mask

        return {
            'type': 'node',
            'feature': feature,
            'split': split_value,
            'left': self._build_tree(X[left_mask], depth + 1),
            'right': self._build_tree(X[right_mask], depth + 1)
        }

    def path_length(self, x: np.ndarray) -> float:
        """Calculate path length for a sample."""
        return self._traverse(x, self.root, 0)

    def _traverse(self, x: np.ndarray, node: Dict, depth: int) -> float:
        """Traverse tree and return path length."""
        if node['type'] == 'leaf':
            # Add expected path length for remaining samples
            return depth + self._c(node['size'])

        if x[node['feature']] < node['split']:
            return self._traverse(x, node['left'], depth + 1)
        else:
            return self._traverse(x, node['right'], depth + 1)

    def _c(self, n: int) -> float:
        """Average path length of unsuccessful search in BST."""
        if n <= 1:
            return 0
        return 2 * (np.log(n - 1) + 0.5772156649) - 2 * (n - 1) / n


class IsolationForest:
    """
    Isolation Forest anomaly detector.

    Anomalies are isolated quickly (short path lengths).
    """

    def __init__(
        self,
        n_trees: int = 100,
        max_samples: int = 256,
        contamination: float = 0.1,
        random_state: int = 42
    ):
        self.n_trees = n_trees
        self.max_samples = max_samples
        self.contamination = contamination
        self.random_state = random_state
        self.trees = []
        self.threshold = None

    def fit(self, X: np.ndarray):
        """Build isolation forest."""
        rng = np.random.RandomState(self.random_state)
        n_samples = len(X)

        # Calculate max depth
        max_depth = int(np.ceil(np.log2(min(self.max_samples, n_samples))))

        self.trees = []
        for i in range(self.n_trees):
            # Subsample
            sample_size = min(self.max_samples, n_samples)
            indices = rng.choice(n_samples, size=sample_size, replace=False)
            X_sample = X[indices]

            # Build tree
            tree = IsolationTree(max_depth, random_state=self.random_state + i)
            tree.fit(X_sample)
            self.trees.append(tree)

        # Set threshold based on contamination
        scores = self.score(X)
        self.threshold = np.percentile(scores, 100 * (1 - self.contamination))

        return self

    def score(self, X: np.ndarray) -> np.ndarray:
        """
        Calculate anomaly scores.

        Higher scores indicate more anomalous samples.
        """
        n_samples = len(X)
        path_lengths = np.zeros((n_samples, self.n_trees))

        for i, tree in enumerate(self.trees):
            for j, x in enumerate(X):
                path_lengths[j, i] = tree.path_length(x)

        # Average path length
        avg_path_length = path_lengths.mean(axis=1)

        # Normalize to [0, 1]
        c = self.trees[0]._c(self.max_samples)
        scores = 2 ** (-avg_path_length / c)

        return scores

    def predict(self, X: np.ndarray) -> np.ndarray:
        """Predict anomalies (1 for anomaly, 0 for normal)."""
        scores = self.score(X)
        return (scores >= self.threshold).astype(int)


# Example usage
np.random.seed(42)

# Normal transactions
n_normal = 1000
X_normal = np.random.randn(n_normal, 4)

# Anomalous transactions (outliers)
n_anomaly = 50
X_anomaly = np.random.randn(n_anomaly, 4) + 4  # Shifted

# Fit isolation forest

# Score all samples

print(f"Average score (normal): {scores[:n_normal].mean():.3f}")
print(f"Average score (anomaly): {scores[n_normal:].mean():.3f}")
print(f"Detected anomalies: {predictions.sum()}")
```

## Autoencoder Anomaly Detection

<Warning>
Autoencoders require more computational resources but can capture complex, nonlinear patterns in transaction behavior.
</Warning>

### Simple Autoencoder

```python
class SimpleAutoencoder:
    """
    Simple autoencoder for anomaly detection.

    Anomalies have high reconstruction error because the model
    learned to reconstruct normal patterns only.
    """

    def __init__(
        self,
        input_dim: int,
        encoding_dim: int = 8,
        hidden_dim: int = 16,
        learning_rate: float = 0.01,
        epochs: int = 100,
        random_state: int = 42
    ):
        self.input_dim = input_dim
        self.encoding_dim = encoding_dim
        self.hidden_dim = hidden_dim
        self.learning_rate = learning_rate
        self.epochs = epochs
        self.random_state = random_state

        # Initialize weights
        np.random.seed(random_state)
        self._init_weights()

        self.threshold = None

    def _init_weights(self):
        """Initialize neural network weights."""
        # Encoder
        self.W1 = np.random.randn(self.input_dim, self.hidden_dim) * 0.1
        self.b1 = np.zeros(self.hidden_dim)
        self.W2 = np.random.randn(self.hidden_dim, self.encoding_dim) * 0.1
        self.b2 = np.zeros(self.encoding_dim)

        # Decoder
        self.W3 = np.random.randn(self.encoding_dim, self.hidden_dim) * 0.1
        self.b3 = np.zeros(self.hidden_dim)
        self.W4 = np.random.randn(self.hidden_dim, self.input_dim) * 0.1
        self.b4 = np.zeros(self.input_dim)

    def _relu(self, x: np.ndarray) -> np.ndarray:
        """ReLU activation."""
        return np.maximum(0, x)

    def _relu_derivative(self, x: np.ndarray) -> np.ndarray:
        """ReLU derivative."""
        return (x > 0).astype(float)

    def _forward(self, X: np.ndarray) -> Tuple[np.ndarray, ...]:
        """Forward pass through network."""
        # Encoder
        h1 = self._relu(X @ self.W1 + self.b1)
        encoding = self._relu(h1 @ self.W2 + self.b2)

        # Decoder
        h3 = self._relu(encoding @ self.W3 + self.b3)
        reconstruction = h3 @ self.W4 + self.b4

        return h1, encoding, h3, reconstruction

    def fit(self, X: np.ndarray, contamination: float = 0.1):
        """Train autoencoder on normal data."""
        n_samples = len(X)

        for epoch in range(self.epochs):
            # Shuffle data
            indices = np.random.permutation(n_samples)
            X_shuffled = X[indices]

            total_loss = 0

            for i in range(n_samples):
                x = X_shuffled[i:i+1]

                # Forward pass
                h1, encoding, h3, reconstruction = self._forward(x)

                # Loss (MSE)
                error = reconstruction - x
                loss = np.mean(error ** 2)
                total_loss += loss

                # Backward pass (simplified gradient descent)
                d_out = 2 * error / self.input_dim

                # Decoder gradients
                d_W4 = h3.T @ d_out
                d_b4 = d_out.sum(axis=0)
                d_h3 = d_out @ self.W4.T * self._relu_derivative(h3)

                d_W3 = encoding.T @ d_h3
                d_b3 = d_h3.sum(axis=0)
                d_encoding = d_h3 @ self.W3.T * self._relu_derivative(encoding)

                # Encoder gradients
                d_W2 = h1.T @ d_encoding
                d_b2 = d_encoding.sum(axis=0)
                d_h1 = d_encoding @ self.W2.T * self._relu_derivative(h1)

                d_W1 = x.T @ d_h1
                d_b1 = d_h1.sum(axis=0)

                # Update weights
                self.W4 -= self.learning_rate * d_W4
                self.b4 -= self.learning_rate * d_b4
                self.W3 -= self.learning_rate * d_W3
                self.b3 -= self.learning_rate * d_b3
                self.W2 -= self.learning_rate * d_W2
                self.b2 -= self.learning_rate * d_b2
                self.W1 -= self.learning_rate * d_W1
                self.b1 -= self.learning_rate * d_b1

            if (epoch + 1) % 20 == 0:
                avg_loss = total_loss / n_samples
                print(f"Epoch {epoch+1}/{self.epochs}, Loss: {avg_loss:.6f}")

        # Set threshold based on training reconstruction errors
        train_scores = self.score(X)
        self.threshold = np.percentile(train_scores, 100 * (1 - contamination))

        return self

    def score(self, X: np.ndarray) -> np.ndarray:
        """Calculate reconstruction error as anomaly score."""
        _, _, _, reconstruction = self._forward(X)
        mse = np.mean((X - reconstruction) ** 2, axis=1)
        return mse

    def predict(self, X: np.ndarray) -> np.ndarray:
        """Predict anomalies."""
        scores = self.score(X)
        return (scores > self.threshold).astype(int)

    def encode(self, X: np.ndarray) -> np.ndarray:
        """Get encoded representation."""
        h1 = self._relu(X @ self.W1 + self.b1)
        encoding = self._relu(h1 @ self.W2 + self.b2)
        return encoding


# Example usage
np.random.seed(42)

# Generate normal transaction features
n_samples = 500
input_dim = 10

# Normal patterns
X_normal = np.random.randn(n_samples, input_dim)

# Train autoencoder

    input_dim=input_dim,
    encoding_dim=4,
    hidden_dim=8,
    learning_rate=0.01,
    epochs=100
)
```

## Ensemble Anomaly Detection

<Formula>
Combined Score = w₁ × Score_statistical + w₂ × Score_iforest + w₃ × Score_autoencoder
</Formula>

### Combining Multiple Detectors

```python
class EnsembleAnomalyDetector:
    """
    Combine multiple anomaly detection methods.

    Different methods capture different types of anomalies.
    """

    def __init__(
        self,
        detectors: List[Tuple[str, Any, float]] = None,
        aggregation: str = 'weighted_average'
    ):
        """
        Args:
            detectors: List of (name, detector, weight) tuples
            aggregation: 'weighted_average', 'max', or 'voting'
        """
        self.detectors = detectors or []
        self.aggregation = aggregation
        self.threshold = None

    def add_detector(self, name: str, detector: Any, weight: float = 1.0):
        """Add a detector to the ensemble."""
        self.detectors.append((name, detector, weight))

    def fit(self, X: np.ndarray, contamination: float = 0.1):
        """Fit all detectors."""
        for name, detector, _ in self.detectors:
            print(f"Fitting {name}...")
            detector.fit(X)

        # Set threshold based on combined scores
        scores = self.score(X)
        self.threshold = np.percentile(scores, 100 * (1 - contamination))

        return self

    def score(self, X: np.ndarray) -> np.ndarray:
        """Calculate combined anomaly scores."""
        n_samples = len(X)
        all_scores = np.zeros((n_samples, len(self.detectors)))
        weights = np.zeros(len(self.detectors))

        for i, (name, detector, weight) in enumerate(self.detectors):
            scores = detector.score(X)

            # Normalize scores to [0, 1]
            min_score = scores.min()
            max_score = scores.max()
            if max_score > min_score:
                scores = (scores - min_score) / (max_score - min_score)

            all_scores[:, i] = scores
            weights[i] = weight

        # Aggregate
        if self.aggregation == 'weighted_average':
            weights = weights / weights.sum()
            combined = (all_scores * weights).sum(axis=1)

        elif self.aggregation == 'max':
            combined = all_scores.max(axis=1)

        elif self.aggregation == 'voting':
            # Each detector votes based on its own threshold
            votes = np.zeros_like(all_scores)
            for i, (name, detector, _) in enumerate(self.detectors):
                if hasattr(detector, 'threshold') and detector.threshold:
                    votes[:, i] = (all_scores[:, i] > 0.5).astype(float)
            combined = votes.mean(axis=1)

        else:
            combined = all_scores.mean(axis=1)

        return combined

    def predict(self, X: np.ndarray) -> np.ndarray:
        """Predict anomalies."""
        scores = self.score(X)
        return (scores > self.threshold).astype(int)

    def explain(self, x: np.ndarray) -> Dict[str, Any]:
        """Explain anomaly detection for a single sample."""
        x_2d = x.reshape(1, -1)
        explanations = {}

        for name, detector, weight in self.detectors:
            score = detector.score(x_2d)[0]
            explanations[name] = {
                'score': score,
                'weight': weight,
                'is_anomaly': score > (detector.threshold if hasattr(detector, 'threshold') else 0.5)
            }

        combined_score = self.score(x_2d)[0]
        explanations['combined'] = {
            'score': combined_score,
            'is_anomaly': combined_score > self.threshold
        }

        return explanations


# Example usage
np.random.seed(42)

# Generate data
n_samples = 500
n_features = 5
X = np.random.randn(n_samples, n_features)

# Create ensemble
ensemble = EnsembleAnomalyDetector(aggregation='weighted_average')

# Add detectors
stat_detector = StatisticalAnomalyDetector(method='zscore', threshold=3.0)
stat_detector.fit(X, [f"f{i}" for i in range(n_features)])
ensemble.add_detector('statistical', stat_detector, weight=1.0)

iforest = IsolationForest(n_trees=50)
iforest.fit(X)
ensemble.add_detector('isolation_forest', iforest, weight=2.0)

# Set ensemble threshold
scores = ensemble.score(X)
ensemble.threshold = np.percentile(scores, 95)

# Test
X_test = np.random.randn(10, n_features)
X_test[0] = [5, 5, 5, 5, 5]  # Obvious anomaly

predictions = ensemble.predict(X_test)
print(f"Predictions: {predictions}")

explanation = ensemble.explain(X_test[0])
print(f"\nExplanation for anomaly: {explanation}")
```

## Key Takeaways

1. **No Labels Needed**: Anomaly detection works without fraud labels
2. **Statistical Methods**: Fast and interpretable for simple patterns
3. **Isolation Forest**: Effective for high-dimensional data with clear outliers
4. **Autoencoders**: Capture complex nonlinear patterns
5. **Ensemble Approach**: Combine methods for robust detection
6. **Threshold Selection**: Critical for balancing detection and false positives

Anomaly detection provides a powerful complement to supervised methods, especially for detecting novel fraud patterns and when labeled data is scarce.
