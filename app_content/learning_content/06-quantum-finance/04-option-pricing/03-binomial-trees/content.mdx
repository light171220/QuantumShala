## 1. Binomial Model Fundamentals

The binomial model discretizes stock price movements into up and down states.

<Definition term="Binomial Tree">
A discrete-time model where the stock price can move up by factor u or down by factor d at each time step. The option is priced by backward induction from expiration.
</Definition>

```python
import numpy as np
from scipy.stats import norm

class BinomialTree:
    """
    Cox-Ross-Rubinstein (CRR) binomial tree for option pricing.
    """

    def __init__(self, S, K, T, r, sigma, N, option_type='call'):
        """
        Args:
            S: Current stock price
            K: Strike price
            T: Time to expiration
            r: Risk-free rate
            sigma: Volatility
            N: Number of time steps
            option_type: 'call' or 'put'
        """
        self.S = S
        self.K = K
        self.T = T
        self.r = r
        self.sigma = sigma
        self.N = N
        self.option_type = option_type

        # Calculate tree parameters
        self.dt = T / N
        self.u = np.exp(sigma * np.sqrt(self.dt))  # Up factor
        self.d = 1 / self.u                         # Down factor
        self.p = (np.exp(r * self.dt) - self.d) / (self.u - self.d)  # Risk-neutral prob

    def build_stock_tree(self):
        """Build the stock price tree."""
        stock_tree = np.zeros((self.N + 1, self.N + 1))

        for i in range(self.N + 1):
            for j in range(i + 1):
                stock_tree[j, i] = self.S * (self.u ** (i - j)) * (self.d ** j)

        return stock_tree

    def price_european(self):
        """Price European option using binomial tree."""
        stock_tree = self.build_stock_tree()
        option_tree = np.zeros_like(stock_tree)

        # Terminal payoffs
        if self.option_type == 'call':
            option_tree[:, self.N] = np.maximum(stock_tree[:, self.N] - self.K, 0)
        else:
            option_tree[:, self.N] = np.maximum(self.K - stock_tree[:, self.N], 0)

        # Backward induction
        discount = np.exp(-self.r * self.dt)
        for i in range(self.N - 1, -1, -1):
            for j in range(i + 1):
                option_tree[j, i] = discount * (
                    self.p * option_tree[j, i + 1] +
                    (1 - self.p) * option_tree[j + 1, i + 1]
                )

        return option_tree[0, 0]

    def price_american(self):
        """Price American option with early exercise."""
        stock_tree = self.build_stock_tree()
        option_tree = np.zeros_like(stock_tree)

        # Terminal payoffs
        if self.option_type == 'call':
            option_tree[:, self.N] = np.maximum(stock_tree[:, self.N] - self.K, 0)
        else:
            option_tree[:, self.N] = np.maximum(self.K - stock_tree[:, self.N], 0)

        # Backward induction with early exercise check
        discount = np.exp(-self.r * self.dt)
        for i in range(self.N - 1, -1, -1):
            for j in range(i + 1):
                # Continuation value
                continuation = discount * (
                    self.p * option_tree[j, i + 1] +
                    (1 - self.p) * option_tree[j + 1, i + 1]
                )

                # Early exercise value
                if self.option_type == 'call':
                    exercise = stock_tree[j, i] - self.K
                else:
                    exercise = self.K - stock_tree[j, i]

                # Take maximum
                option_tree[j, i] = max(continuation, exercise)

        return option_tree[0, 0]

# Example
tree = BinomialTree(S=100, K=100, T=1.0, r=0.05, sigma=0.20, N=100)

print("Binomial Tree Parameters:")
print(f"  Up factor (u): {tree.u:.4f}")
print(f"  Down factor (d): {tree.d:.4f}")
print(f"  Risk-neutral probability (p): {tree.p:.4f}")

print("\nOption Prices (N=100 steps):")
print(f"  European Call: ${tree.price_european():.4f}")
tree.option_type = 'put'
print(f"  European Put: ${tree.price_european():.4f}")
print(f"  American Put: ${tree.price_american():.4f}")
```

## 2. CRR Model Parameters

<Formula>
Cox-Ross-Rubinstein parameters:
$$u = e^{\sigma\sqrt{\Delta t}}, \quad d = \frac{1}{u} = e^{-\sigma\sqrt{\Delta t}}$$
$$p = \frac{e^{r\Delta t} - d}{u - d}$$
where $\Delta t = T/N$ is the time step size.
</Formula>

```python
def analyze_tree_parameters(sigma, r, T, N_values):
    """
    Analyze how parameters change with number of steps.
    """
    results = []

    for N in N_values:
        dt = T / N
        u = np.exp(sigma * np.sqrt(dt))
        d = 1 / u
        p = (np.exp(r * dt) - d) / (u - d)

        results.append({
            'N': N,
            'dt': dt,
            'u': u,
            'd': d,
            'p': p,
            'u/d': u/d
        })

    return results

params = analyze_tree_parameters(sigma=0.20, r=0.05, T=1.0,
                                  N_values=[10, 50, 100, 500, 1000])

print("CRR Parameters vs Number of Steps:")
print(f"{'N':>6} {'dt':>10} {'u':>8} {'d':>8} {'p':>8}")
print("-" * 45)
for p in params:
    print(f"{p['N']:>6} {p['dt']:>10.6f} {p['u']:>8.4f} {p['d']:>8.4f} {p['p']:>8.4f}")
```

## 3. Convergence to Black-Scholes

As N increases, binomial prices converge to Black-Scholes.

```python
def convergence_analysis(S, K, T, r, sigma, N_range):
    """
    Analyze convergence of binomial prices to Black-Scholes.
    """
    # Black-Scholes price
    d1 = (np.log(S/K) + (r + sigma**2/2)*T) / (sigma*np.sqrt(T))
    d2 = d1 - sigma*np.sqrt(T)
    bs_call = S * norm.cdf(d1) - K * np.exp(-r*T) * norm.cdf(d2)
    bs_put = K * np.exp(-r*T) * norm.cdf(-d2) - S * norm.cdf(-d1)

    results = []
    for N in N_range:
        tree = BinomialTree(S, K, T, r, sigma, N, 'call')
        call_price = tree.price_european()
        tree.option_type = 'put'
        put_price = tree.price_european()

        results.append({
            'N': N,
            'call': call_price,
            'put': put_price,
            'call_error': abs(call_price - bs_call),
            'put_error': abs(put_price - bs_put)
        })

    return results, bs_call, bs_put

N_range = [10, 25, 50, 100, 200, 500, 1000]
results, bs_call, bs_put = convergence_analysis(
    S=100, K=100, T=1.0, r=0.05, sigma=0.20, N_range=N_range
)

print("Convergence to Black-Scholes:")
print(f"Black-Scholes Call: ${bs_call:.4f}")
print(f"Black-Scholes Put: ${bs_put:.4f}")
print(f"\n{'N':>6} {'Call':>10} {'Error':>10} {'Put':>10} {'Error':>10}")
print("-" * 50)
for r in results:
    print(f"{r['N']:>6} ${r['call']:>9.4f} ${r['call_error']:>9.6f} "
          f"${r['put']:>9.4f} ${r['put_error']:>9.6f}")
```

<Tip>
The binomial model converges to Black-Scholes at rate O(1/N). For practical accuracy, N=100-500 steps is usually sufficient.
</Tip>

## 4. American Option Pricing

American options can be exercised early, making them more valuable than European options.

```python
def compare_european_american(S, K, T, r, sigma, N=200):
    """
    Compare European and American option prices.
    """
    tree = BinomialTree(S, K, T, r, sigma, N)

    results = {}
    for opt_type in ['call', 'put']:
        tree.option_type = opt_type
        european = tree.price_european()
        american = tree.price_american()
        premium = american - european

        results[opt_type] = {
            'european': european,
            'american': american,
            'early_exercise_premium': premium
        }

    return results

# Compare across different moneyness levels
print("American vs European Option Prices:")
print("=" * 60)

for K in [90, 100, 110]:
    comparison = compare_european_american(S=100, K=K, T=1.0, r=0.05, sigma=0.20)
    print(f"\nStrike K = {K}:")
    for opt_type, prices in comparison.items():
        print(f"  {opt_type.capitalize()}:")
        print(f"    European: ${prices['european']:.4f}")
        print(f"    American: ${prices['american']:.4f}")
        print(f"    Early Exercise Premium: ${prices['early_exercise_premium']:.4f}")
```

<Warning>
For non-dividend paying stocks:
- American calls = European calls (never optimal to exercise early)
- American puts > European puts (early exercise can be optimal)
</Warning>

## 5. Early Exercise Boundary

Find when early exercise becomes optimal.

```python
def find_exercise_boundary(S, K, T, r, sigma, N=100):
    """
    Find the early exercise boundary for American puts.
    """
    tree = BinomialTree(S, K, T, r, sigma, N, 'put')

    stock_tree = tree.build_stock_tree()
    option_tree = np.zeros_like(stock_tree)
    exercise_boundary = np.zeros(N + 1)

    # Terminal payoffs
    option_tree[:, N] = np.maximum(K - stock_tree[:, N], 0)

    # Backward induction
    discount = np.exp(-r * tree.dt)
    for i in range(N - 1, -1, -1):
        boundary_found = False
        for j in range(i + 1):
            continuation = discount * (
                tree.p * option_tree[j, i + 1] +
                (1 - tree.p) * option_tree[j + 1, i + 1]
            )
            exercise = K - stock_tree[j, i]
            option_tree[j, i] = max(continuation, exercise)

            # Track exercise boundary
            if not boundary_found and exercise >= continuation and exercise > 0:
                exercise_boundary[i] = stock_tree[j, i]
                boundary_found = True

        if not boundary_found:
            exercise_boundary[i] = 0

    return exercise_boundary, np.linspace(0, T, N + 1)

boundary, times = find_exercise_boundary(S=100, K=100, T=1.0, r=0.05, sigma=0.20)

print("Early Exercise Boundary for American Put:")
print(f"{'Time':>8} {'Boundary':>12}")
print("-" * 22)
for t, b in zip(times[::20], boundary[::20]):  # Every 20th point
    if b > 0:
        print(f"{t:>8.3f} ${b:>11.2f}")
    else:
        print(f"{t:>8.3f}      N/A")
```

## 6. Binomial Tree with Dividends

Adjust for discrete dividends.

```python
class BinomialTreeDividend(BinomialTree):
    """
    Binomial tree with discrete dividend handling.
    """

    def __init__(self, S, K, T, r, sigma, N, option_type='call', dividends=None):
        """
        Args:
            dividends: List of (time, amount) tuples for discrete dividends
        """
        super().__init__(S, K, T, r, sigma, N, option_type)
        self.dividends = dividends or []

    def price_american_dividend(self):
        """Price American option with discrete dividends."""
        # Convert dividend times to step indices
        div_steps = {}
        for div_time, div_amount in self.dividends:
            step = int(div_time / self.dt)
            if 0 < step <= self.N:
                div_steps[step] = div_amount

        # Build tree with dividend adjustments
        stock_tree = np.zeros((self.N + 1, self.N + 1))

        # Initialize
        for i in range(self.N + 1):
            for j in range(i + 1):
                stock_tree[j, i] = self.S * (self.u ** (i - j)) * (self.d ** j)

                # Subtract PV of future dividends
                for div_step, div_amount in div_steps.items():
                    if div_step > i:
                        time_to_div = (div_step - i) * self.dt
                        stock_tree[j, i] -= div_amount * np.exp(-self.r * time_to_div)

        option_tree = np.zeros_like(stock_tree)

        # Terminal payoffs
        if self.option_type == 'call':
            option_tree[:, self.N] = np.maximum(stock_tree[:, self.N] - self.K, 0)
        else:
            option_tree[:, self.N] = np.maximum(self.K - stock_tree[:, self.N], 0)

        # Backward induction
        discount = np.exp(-self.r * self.dt)
        for i in range(self.N - 1, -1, -1):
            for j in range(i + 1):
                continuation = discount * (
                    self.p * option_tree[j, i + 1] +
                    (1 - self.p) * option_tree[j + 1, i + 1]
                )

                if self.option_type == 'call':
                    exercise = max(stock_tree[j, i] - self.K, 0)
                else:
                    exercise = max(self.K - stock_tree[j, i], 0)

                option_tree[j, i] = max(continuation, exercise)

        return option_tree[0, 0]

# Example with quarterly dividends
dividends = [(0.25, 1.0), (0.5, 1.0), (0.75, 1.0)]  # $1 every quarter

tree_no_div = BinomialTree(S=100, K=100, T=1.0, r=0.05, sigma=0.20, N=200, option_type='call')
tree_div = BinomialTreeDividend(S=100, K=100, T=1.0, r=0.05, sigma=0.20, N=200,
                                 option_type='call', dividends=dividends)

print("Impact of Dividends on Option Prices:")
print(f"  No dividends - European Call: ${tree_no_div.price_european():.4f}")
print(f"  No dividends - American Call: ${tree_no_div.price_american():.4f}")
print(f"  With dividends - American Call: ${tree_div.price_american_dividend():.4f}")
```

## 7. Greeks from Binomial Tree

Calculate Greeks numerically from the tree.

```python
def binomial_greeks(S, K, T, r, sigma, N=200, option_type='call'):
    """
    Calculate Greeks using binomial tree with finite differences.
    """
    h_S = 0.01 * S
    h_sigma = 0.001
    h_T = 1/365

    # Base price
    base = BinomialTree(S, K, T, r, sigma, N, option_type).price_european()

    # Delta
    up = BinomialTree(S + h_S, K, T, r, sigma, N, option_type).price_european()
    down = BinomialTree(S - h_S, K, T, r, sigma, N, option_type).price_european()
    delta = (up - down) / (2 * h_S)

    # Gamma
    gamma = (up - 2*base + down) / (h_S**2)

    # Vega (per 1%)
    vega_up = BinomialTree(S, K, T, r, sigma + h_sigma, N, option_type).price_european()
    vega_down = BinomialTree(S, K, T, r, sigma - h_sigma, N, option_type).price_european()
    vega = (vega_up - vega_down) / (2 * h_sigma) * 0.01

    # Theta (daily)
    if T > h_T:
        theta_down = BinomialTree(S, K, T - h_T, r, sigma, N, option_type).price_european()
        theta = (theta_down - base) / h_T
    else:
        theta = 0

    return {
        'price': base,
        'delta': delta,
        'gamma': gamma,
        'vega': vega,
        'theta': theta
    }

greeks = binomial_greeks(S=100, K=100, T=0.5, r=0.05, sigma=0.20)

print("Greeks from Binomial Tree (N=200):")
for name, value in greeks.items():
    print(f"  {name}: {value:.4f}")
```
