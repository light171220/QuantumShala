# Derivative Fundamentals

Derivatives are financial instruments whose value depends on underlying assets. They serve crucial roles in risk management, price discovery, and market efficiency. Understanding derivatives is essential before exploring quantum computing applications in this domain.

## What Are Derivatives?

A derivative is a contract between two parties whose value is derived from an underlying asset, index, or reference rate. The most common underlying assets include stocks, bonds, commodities, currencies, interest rates, and market indices.

```python
from dataclasses import dataclass
from typing import Optional
from enum import Enum
from datetime import date
import numpy as np

class DerivativeType(Enum):
    FORWARD = "forward"
    FUTURE = "future"
    OPTION = "option"
    SWAP = "swap"

class PositionType(Enum):
    LONG = "long"
    SHORT = "short"

@dataclass
class Derivative:
    """Base class for derivative instruments."""
    underlying: str
    notional: float
    expiration: date
    derivative_type: DerivativeType

    def days_to_expiry(self, current_date: date) -> int:
        """Calculate days until expiration."""
        return (self.expiration - current_date).days

    def time_to_expiry(self, current_date: date) -> float:
        """Calculate time to expiry in years."""
        return self.days_to_expiry(current_date) / 365.0

@dataclass
class ForwardContract(Derivative):
    """
    Forward contract: Agreement to buy/sell at future date.

    Key characteristics:
    - OTC (over-the-counter) traded
    - Customizable terms
    - Counterparty risk
    - No daily settlement
    """
    forward_price: float
    position: PositionType

    def __post_init__(self):
        self.derivative_type = DerivativeType.FORWARD

    def payoff(self, spot_at_expiry: float) -> float:
        """Calculate payoff at expiration."""
        if self.position == PositionType.LONG:
            return (spot_at_expiry - self.forward_price) * self.notional
        else:
            return (self.forward_price - spot_at_expiry) * self.notional

    def value(
        self,
        spot_price: float,
        risk_free_rate: float,
        current_date: date
    ) -> float:
        """
        Calculate current value of forward contract.

        For a long position:
        Value = S - K * exp(-r * T)
        """
        T = self.time_to_expiry(current_date)
        discount_factor = np.exp(-risk_free_rate * T)

        if self.position == PositionType.LONG:
            return (spot_price - self.forward_price * discount_factor) * self.notional
        else:
            return (self.forward_price * discount_factor - spot_price) * self.notional


def calculate_forward_price(
    spot_price: float,
    risk_free_rate: float,
    time_to_expiry: float,
    dividend_yield: float = 0.0,
    storage_cost: float = 0.0
) -> float:
    """
    Calculate theoretical forward price.

    F = S * exp((r - q + s) * T)

    where:
    - S: spot price
    - r: risk-free rate
    - q: dividend yield
    - s: storage cost rate
    - T: time to expiry
    """
    return spot_price * np.exp((risk_free_rate - dividend_yield + storage_cost) * time_to_expiry)
```

## Futures Contracts

Futures are standardized forward contracts traded on exchanges. The exchange clearinghouse eliminates counterparty risk through daily settlement (marking to market).

```python
@dataclass
class FuturesContract(Derivative):
    """
    Futures contract: Standardized, exchange-traded derivative.

    Key characteristics:
    - Exchange traded
    - Standardized contract sizes
    - Daily mark-to-market
    - Margin requirements
    """
    futures_price: float
    contract_size: float
    tick_size: float
    position: PositionType
    initial_margin: float
    maintenance_margin: float

    def __post_init__(self):
        self.derivative_type = DerivativeType.FUTURE

    def tick_value(self) -> float:
        """Value of one tick movement."""
        return self.tick_size * self.contract_size

    def daily_pnl(
        self,
        previous_settlement: float,
        current_settlement: float,
        num_contracts: int = 1
    ) -> float:
        """Calculate daily P&L from settlement prices."""
        price_change = current_settlement - previous_settlement

        if self.position == PositionType.LONG:
            return price_change * self.contract_size * num_contracts
        else:
            return -price_change * self.contract_size * num_contracts


class MarginAccount:
    """
    Manage futures margin account.

    Tracks margin balance and handles margin calls.
    """

    def __init__(
        self,
        initial_deposit: float,
        initial_margin: float,
        maintenance_margin: float
    ):
        self.balance = initial_deposit
        self.initial_margin = initial_margin
        self.maintenance_margin = maintenance_margin
        self.history = [{'type': 'deposit', 'amount': initial_deposit, 'balance': initial_deposit}]

    def mark_to_market(self, daily_pnl: float) -> dict:
        """
        Apply daily settlement.

        Returns status including any margin call.
        """
        self.balance += daily_pnl
        self.history.append({
            'type': 'settlement',
            'amount': daily_pnl,
            'balance': self.balance
        })

        margin_call = None
        if self.balance < self.maintenance_margin:
            margin_call = self.initial_margin - self.balance

        return {
            'balance': self.balance,
            'pnl': daily_pnl,
            'margin_call': margin_call
        }

    def deposit(self, amount: float):
        """Add funds to margin account."""
        self.balance += amount
        self.history.append({
            'type': 'deposit',
            'amount': amount,
            'balance': self.balance
        })
```

```python
class BasisAnalyzer:
    """
    Analyze basis (spot - futures price) relationships.

    Basis behavior affects hedging effectiveness.
    """

    def __init__(self):
        self.basis_history = []

    def calculate_basis(
        self,
        spot_price: float,
        futures_price: float
    ) -> float:
        """Calculate current basis."""
        basis = spot_price - futures_price
        self.basis_history.append(basis)
        return basis

    def basis_risk(self) -> float:
        """
        Calculate basis risk (volatility of basis).

        Basis risk affects hedging effectiveness.
        """
        if len(self.basis_history) < 2:
            return 0.0
        return np.std(self.basis_history)

    def is_contango(
        self,
        spot_price: float,
        futures_price: float
    ) -> bool:
        """Check if market is in contango (futures > spot)."""
        return futures_price > spot_price

    def is_backwardation(
        self,
        spot_price: float,
        futures_price: float
    ) -> bool:
        """Check if market is in backwardation (spot > futures)."""
        return spot_price > futures_price
```

## Hedging with Derivatives

One of the primary uses of derivatives is hedging - reducing risk by taking offsetting positions.

```python
class HedgeCalculator:
    """
    Calculate optimal hedge ratios and positions.
    """

    def __init__(
        self,
        asset_volatility: float,
        hedge_volatility: float,
        correlation: float
    ):
        self.asset_vol = asset_volatility
        self.hedge_vol = hedge_volatility
        self.correlation = correlation

    def minimum_variance_hedge_ratio(self) -> float:
        """
        Calculate optimal hedge ratio for minimum variance.

        h* = ρ * (σ_asset / σ_hedge)

        This minimizes the variance of the hedged position.
        """
        return self.correlation * (self.asset_vol / self.hedge_vol)

    def hedge_effectiveness(self, hedge_ratio: float) -> float:
        """
        Calculate hedge effectiveness (R-squared).

        Percentage of variance eliminated by the hedge.
        """
        return self.correlation ** 2

    def number_of_contracts(
        self,
        exposure: float,
        contract_size: float,
        spot_price: float,
        futures_price: float,
        hedge_ratio: Optional[float] = None
    ) -> int:
        """
        Calculate number of futures contracts for hedge.

        N = h * (Exposure / (Futures_price * Contract_size))
        """
        if hedge_ratio is None:
            hedge_ratio = self.minimum_variance_hedge_ratio()

        notional_per_contract = futures_price * contract_size
        num_contracts = hedge_ratio * exposure / notional_per_contract

        return round(num_contracts)


def demonstrate_hedge():
    """Demonstrate a simple hedging example."""
    # Portfolio manager holds $10M in S&P 500 stocks
    portfolio_value = 10_000_000
    spot_index = 4500

        'hedge_type': 'SHORT futures (to hedge long stock position)'
    }
```

## Introduction to Options

Options give the holder the right, but not the obligation, to buy or sell the underlying asset.

```python
class OptionType(Enum):
    CALL = "call"
    PUT = "put"

class ExerciseStyle(Enum):
    EUROPEAN = "european"  # Exercise only at expiration
    AMERICAN = "american"  # Exercise any time before expiration

@dataclass
class OptionContract(Derivative):
    """
    Option contract with key characteristics.
    """
    strike_price: float
    option_type: OptionType
    exercise_style: ExerciseStyle
    premium: float
    position: PositionType  # Long (buyer) or Short (writer)

    def __post_init__(self):
        self.derivative_type = DerivativeType.OPTION

    def intrinsic_value(self, spot_price: float) -> float:
        """Calculate intrinsic value."""
        if self.option_type == OptionType.CALL:
            return max(0, spot_price - self.strike_price)
        else:
            return max(0, self.strike_price - spot_price)

    def payoff_at_expiry(self, spot_price: float) -> float:
        """Calculate payoff at expiration."""
        intrinsic = self.intrinsic_value(spot_price)

        if self.position == PositionType.LONG:
            return intrinsic * self.notional - self.premium * self.notional
        else:
            return self.premium * self.notional - intrinsic * self.notional

    def is_in_the_money(self, spot_price: float) -> bool:
        """Check if option is in the money."""
        return self.intrinsic_value(spot_price) > 0

    def is_at_the_money(self, spot_price: float, tolerance: float = 0.01) -> bool:
        """Check if option is at the money."""
        return abs(spot_price - self.strike_price) / spot_price < tolerance

    def is_out_of_the_money(self, spot_price: float) -> bool:
        """Check if option is out of the money."""
        return not self.is_in_the_money(spot_price)

    def breakeven(self) -> float:
        """Calculate breakeven price for long position."""
        if self.option_type == OptionType.CALL:
            return self.strike_price + self.premium
        else:
            return self.strike_price - self.premium
```

## Summary

Derivatives are essential financial instruments for:

- **Risk Transfer**: Moving risk from those who don't want it to those willing to bear it
- **Price Discovery**: Revealing market expectations about future prices
- **Hedging**: Reducing exposure to adverse price movements
- **Speculation**: Taking positions on expected price movements

Key derivative types:
- **Forwards**: Customized OTC contracts for future delivery
- **Futures**: Standardized, exchange-traded with daily settlement
- **Options**: Right (not obligation) to buy/sell

Quantum computing promises to enhance derivative pricing and risk management through faster simulation and better optimization.
