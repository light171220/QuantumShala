# Exotic Options

Exotic options have features beyond standard calls and puts. These path-dependent and structurally complex instruments require advanced pricing techniques where quantum computing shows significant promise.

## Path-Dependent Options

Path-dependent options have payoffs that depend not just on the final price but on the price path taken to get there.

```python
import numpy as np
from dataclasses import dataclass
from typing import Optional, Callable, List
from enum import Enum

class AverageType(Enum):
    ARITHMETIC = "arithmetic"
    GEOMETRIC = "geometric"

class BarrierType(Enum):
    UP_AND_OUT = "up_and_out"
    UP_AND_IN = "up_and_in"
    DOWN_AND_OUT = "down_and_out"
    DOWN_AND_IN = "down_and_in"

@dataclass
class PricePath:
    """Represents a single price path."""
    prices: np.ndarray
    times: np.ndarray

    @property
    def final_price(self) -> float:
        return self.prices[-1]

    @property
    def max_price(self) -> float:
        return np.max(self.prices)

    @property
    def min_price(self) -> float:
        return np.min(self.prices)

    def arithmetic_average(self) -> float:
        return np.mean(self.prices)

    def geometric_average(self) -> float:
        return np.exp(np.mean(np.log(self.prices)))

    def crossed_barrier(self, barrier: float, direction: str) -> bool:
        """Check if path crossed a barrier."""
        if direction == "up":
            return np.any(self.prices >= barrier)
        else:
            return np.any(self.prices <= barrier)
```

## Asian Options

Asian options use the average price over a period, reducing manipulation risk and volatility exposure.

```python
class AsianOption:
    """
    Asian option with average price feature.

    Types:
    - Average price: payoff based on average vs strike
    - Average strike: payoff based on final vs average
    """

    def __init__(
        self,
        strike: float,
        is_call: bool,
        average_type: AverageType = AverageType.ARITHMETIC,
        is_average_price: bool = True  # vs average strike
    ):
        self.strike = strike
        self.is_call = is_call
        self.average_type = average_type
        self.is_average_price = is_average_price

    def payoff(self, path: PricePath) -> float:
        """Calculate payoff for a given price path."""
        if self.average_type == AverageType.ARITHMETIC:
            average = path.arithmetic_average()
        else:
            average = path.geometric_average()

        if self.is_average_price:
            # Compare average to strike
            if self.is_call:
                return max(0, average - self.strike)
            else:
                return max(0, self.strike - average)
        else:
            # Average strike: compare final to average
            if self.is_call:
                return max(0, path.final_price - average)
            else:
                return max(0, average - path.final_price)

    def monte_carlo_price(
        self,
        spot: float,
        rate: float,
        volatility: float,
        time_to_expiry: float,
        num_simulations: int = 100000,
        num_steps: int = 252
    ) -> dict:
        """
        Price Asian option using Monte Carlo simulation.

        Returns price and confidence interval.
        """
        dt = time_to_expiry / num_steps
        payoffs = []

        for _ in range(num_simulations):
            # Generate price path
            prices = [spot]
            for _ in range(num_steps):
                z = np.random.standard_normal()
                drift = (rate - 0.5 * volatility**2) * dt
                diffusion = volatility * np.sqrt(dt) * z
                prices.append(prices[-1] * np.exp(drift + diffusion))

            path = PricePath(
                prices=np.array(prices),
                times=np.linspace(0, time_to_expiry, num_steps + 1)
            )

            payoffs.append(self.payoff(path))

        # Discount payoffs
        discount_factor = np.exp(-rate * time_to_expiry)
        discounted_payoffs = np.array(payoffs) * discount_factor

        price = np.mean(discounted_payoffs)
        std_error = np.std(discounted_payoffs) / np.sqrt(num_simulations)

        return {
            'price': price,
            'std_error': std_error,
            'confidence_interval': (price - 1.96 * std_error, price + 1.96 * std_error)
        }


def geometric_asian_closed_form(
    spot: float,
    strike: float,
    rate: float,
    volatility: float,
    time_to_expiry: float,
    is_call: bool
) -> float:
    """
    Closed-form solution for geometric Asian option.

    Uses adjusted volatility and dividend yield.
    """
    from scipy.stats import norm

    # Adjusted parameters for geometric average
    adj_vol = volatility / np.sqrt(3)
    adj_rate = 0.5 * (rate - volatility**2 / 6)

    # Black-Scholes with adjusted parameters
    d1 = (np.log(spot / strike) + (adj_rate + 0.5 * adj_vol**2) * time_to_expiry) / (adj_vol * np.sqrt(time_to_expiry))
    d2 = d1 - adj_vol * np.sqrt(time_to_expiry)

    if is_call:
        price = (spot * np.exp((adj_rate - rate) * time_to_expiry) * norm.cdf(d1) -
                 strike * np.exp(-rate * time_to_expiry) * norm.cdf(d2))
    else:
        price = (strike * np.exp(-rate * time_to_expiry) * norm.cdf(-d2) -
                 spot * np.exp((adj_rate - rate) * time_to_expiry) * norm.cdf(-d1))

    return price
```

## Barrier Options

Barrier options are activated or extinguished when the underlying crosses a specified level.

```python
class BarrierOption:
    """
    Barrier option that activates/deactivates at barrier level.

    Types:
    - Knock-in: becomes active when barrier touched
    - Knock-out: becomes worthless when barrier touched
    """

    def __init__(
        self,
        strike: float,
        barrier: float,
        is_call: bool,
        barrier_type: BarrierType,
        rebate: float = 0.0  # Payment if knocked out
    ):
        self.strike = strike
        self.barrier = barrier
        self.is_call = is_call
        self.barrier_type = barrier_type
        self.rebate = rebate

    def payoff(self, path: PricePath) -> float:
        """Calculate payoff for a given price path."""
        # Check barrier condition
        if "up" in self.barrier_type.value:
            barrier_touched = path.crossed_barrier(self.barrier, "up")
        else:
            barrier_touched = path.crossed_barrier(self.barrier, "down")

        # Determine if option is active
        if "out" in self.barrier_type.value:
            # Knock-out: becomes worthless if barrier touched
            if barrier_touched:
                return self.rebate
            else:
                option_active = True
        else:
            # Knock-in: becomes active only if barrier touched
            option_active = barrier_touched

        if not option_active:
            return 0.0

        # Calculate vanilla payoff
        if self.is_call:
            return max(0, path.final_price - self.strike)
        else:
            return max(0, self.strike - path.final_price)

    def monte_carlo_price(
        self,
        spot: float,
        rate: float,
        volatility: float,
        time_to_expiry: float,
        num_simulations: int = 100000,
        num_steps: int = 252
    ) -> dict:
        """Price barrier option using Monte Carlo."""
        dt = time_to_expiry / num_steps
        payoffs = []

        for _ in range(num_simulations):
            prices = [spot]
            for _ in range(num_steps):
                z = np.random.standard_normal()
                drift = (rate - 0.5 * volatility**2) * dt
                diffusion = volatility * np.sqrt(dt) * z
                prices.append(prices[-1] * np.exp(drift + diffusion))

            path = PricePath(
                prices=np.array(prices),
                times=np.linspace(0, time_to_expiry, num_steps + 1)
            )

            payoffs.append(self.payoff(path))

        discount_factor = np.exp(-rate * time_to_expiry)
        discounted_payoffs = np.array(payoffs) * discount_factor

        return {
            'price': np.mean(discounted_payoffs),
            'std_error': np.std(discounted_payoffs) / np.sqrt(num_simulations),
            'knock_probability': np.mean([p != self.rebate for p in payoffs])
        }
```

## Lookback Options

Lookback options have payoffs based on the maximum or minimum price during the option's life.

```python
class LookbackOption:
    """
    Lookback option based on extreme prices.

    Types:
    - Fixed strike: payoff vs predetermined strike
    - Floating strike: strike is the extreme price
    """

    def __init__(
        self,
        strike: Optional[float],  # None for floating strike
        is_call: bool,
        use_max: bool  # True for max lookback, False for min
    ):
        self.strike = strike
        self.is_call = is_call
        self.use_max = use_max

    def payoff(self, path: PricePath) -> float:
        """Calculate payoff for lookback option."""
        extreme_price = path.max_price if self.use_max else path.min_price

        if self.strike is None:
            # Floating strike lookback
            if self.is_call:
                # Call: buy at minimum, sell at final
                return max(0, path.final_price - path.min_price)
            else:
                # Put: sell at maximum, buy at final
                return max(0, path.max_price - path.final_price)
        else:
            # Fixed strike lookback
            if self.is_call:
                return max(0, extreme_price - self.strike)
            else:
                return max(0, self.strike - extreme_price)

    def monte_carlo_price(
        self,
        spot: float,
        rate: float,
        volatility: float,
        time_to_expiry: float,
        num_simulations: int = 100000,
        num_steps: int = 252
    ) -> dict:
        """Price lookback option using Monte Carlo."""
        dt = time_to_expiry / num_steps
        payoffs = []

        for _ in range(num_simulations):
            prices = [spot]
            for _ in range(num_steps):
                z = np.random.standard_normal()
                drift = (rate - 0.5 * volatility**2) * dt
                diffusion = volatility * np.sqrt(dt) * z
                prices.append(prices[-1] * np.exp(drift + diffusion))

            path = PricePath(
                prices=np.array(prices),
                times=np.linspace(0, time_to_expiry, num_steps + 1)
            )

            payoffs.append(self.payoff(path))

        discount_factor = np.exp(-rate * time_to_expiry)
        discounted_payoffs = np.array(payoffs) * discount_factor

        return {
            'price': np.mean(discounted_payoffs),
            'std_error': np.std(discounted_payoffs) / np.sqrt(num_simulations)
        }
```

## Quantum Advantages for Exotic Options

Quantum computing offers significant advantages for pricing path-dependent options.

        self,

        num_samples: int
    ) -> dict:
        """
        Quantum amplitude estimation for option pricing.

        Provides O(1/ε) convergence vs O(1/ε²) classical.

        In practice, this would use Qiskit/Pennylane.
        This is a simulation of the quantum advantage.

        # Real implementation would use amplitude estimation

            sample2 = payoff_oracle(-z)

## Variance Reduction Techniques

Monte Carlo pricing benefits from variance reduction for faster convergence.

```python
class VarianceReduction:
    """
    Variance reduction techniques for Monte Carlo pricing.
    """

    @staticmethod
    def antithetic_variates(
        payoff_func: Callable,
        num_simulations: int,
        random_generator: Callable
    ) -> dict:
        """
        Antithetic variates: use z and -z together.

        Reduces variance when payoff is monotonic in z.
        """
        payoffs = []

        for _ in range(num_simulations // 2):
            z = random_generator()
            payoff_plus = payoff_func(z)
            payoff_minus = payoff_func(-z)

            # Average the pair
            payoffs.append((payoff_plus + payoff_minus) / 2)

        return {
            'mean': np.mean(payoffs),
            'std_error': np.std(payoffs) / np.sqrt(len(payoffs)),
            'variance_reduction': 'Up to 50%'
        }

    @staticmethod
    def control_variate(
        exotic_payoffs: np.ndarray,
        vanilla_payoffs: np.ndarray,
        vanilla_price: float
    ) -> dict:
        """
        Control variate using correlated vanilla option.

        Adjust estimate using known vanilla price.
        """
        # Optimal control coefficient
        covariance = np.cov(exotic_payoffs, vanilla_payoffs)[0, 1]
        variance = np.var(vanilla_payoffs)
        beta = covariance / variance if variance > 0 else 0

        # Adjusted estimate
        adjusted = exotic_payoffs - beta * (vanilla_payoffs - vanilla_price)

        return {
            'price': np.mean(adjusted),
            'std_error': np.std(adjusted) / np.sqrt(len(adjusted)),
            'beta': beta,
            'variance_reduction': f'{(1 - np.var(adjusted)/np.var(exotic_payoffs))*100:.1f}%'
        }
```

## Summary

Exotic options expand the hedging and speculation toolkit:

- **Asian options**: Use average prices, reducing manipulation risk
- **Barrier options**: Activate or deactivate at price levels
- **Lookback options**: Capture optimal prices in hindsight

Key pricing challenges:
- Path dependence requires simulation or specialized techniques
- Monte Carlo with variance reduction is the standard approach
- Quantum computing offers quadratic speedup potential

These instruments are widely used in commodities, currencies, and structured products where their specific features align with hedging needs.
