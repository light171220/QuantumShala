```python
from dataclasses import dataclass, field
from typing import Dict, List, Optional, Callable
from enum import Enum
from datetime import datetime
from abc import ABC, abstractmethod

class OrderSide(Enum):
    BUY = "buy"
    SELL = "sell"

class OrderType(Enum):
    MARKET = "market"
    LIMIT = "limit"
    STOP = "stop"
    STOP_LIMIT = "stop_limit"

class OrderStatus(Enum):
    PENDING = "pending"
    SUBMITTED = "submitted"
    FILLED = "filled"
    PARTIALLY_FILLED = "partial"
    CANCELLED = "cancelled"
    REJECTED = "rejected"

@dataclass
class Order:
    """Represents a trading order."""
    symbol: str
    side: OrderSide
    quantity: float
    order_type: OrderType
    price: Optional[float] = None  # For limit orders
    stop_price: Optional[float] = None  # For stop orders
    status: OrderStatus = OrderStatus.PENDING
    order_id: Optional[str] = None
    timestamp: datetime = field(default_factory=datetime.now)
    filled_quantity: float = 0.0
    filled_price: float = 0.0

@dataclass
class MarketData:
    """Real-time market data."""
    symbol: str
    bid: float
    ask: float
    last_price: float
    volume: int
    timestamp: datetime

    @property
    def mid_price(self) -> float:
        return (self.bid + self.ask) / 2

    @property
    def spread(self) -> float:
        return self.ask - self.bid

    @property
    def spread_bps(self) -> float:
        """Spread in basis points."""
        return (self.spread / self.mid_price) * 10000


class TradingStrategy(ABC):
    """
    Base class for all trading strategies.

    Strategies receive market data and generate trading signals.
    """

    def __init__(self, name: str, symbols: List[str]):
        self.name = name
        self.symbols = symbols
        self.positions: Dict[str, float] = {s: 0.0 for s in symbols}
        self.orders: List[Order] = []

    @abstractmethod
    def on_market_data(self, data: MarketData) -> Optional[Order]:
        """
        Process market data and optionally generate an order.

        This is the core strategy logic.
        """
        pass

    @abstractmethod
    def calculate_position_size(self, symbol: str, signal_strength: float) -> float:
        """Calculate how much to trade based on signal."""
        pass

    def update_position(self, order: Order):
        """Update position after order fill."""
        if order.status == OrderStatus.FILLED:
            multiplier = 1 if order.side == OrderSide.BUY else -1
            self.positions[order.symbol] += multiplier * order.filled_quantity


class SimpleMovingAverageCrossover(TradingStrategy):
    """
    Classic moving average crossover strategy.

    Buy when fast MA crosses above slow MA.
    Sell when fast MA crosses below slow MA.
    """

    def __init__(self, symbol: str, fast_period: int = 10, slow_period: int = 30):
        super().__init__("MA_Crossover", [symbol])
        self.symbol = symbol
        self.fast_period = fast_period
        self.slow_period = slow_period
        self.prices: List[float] = []
        self.max_position = 100  # Maximum shares

    def _calculate_ma(self, period: int) -> Optional[float]:
        if len(self.prices) < period:
            return None
        return sum(self.prices[-period:]) / period

    def on_market_data(self, data: MarketData) -> Optional[Order]:
        if data.symbol != self.symbol:
            return None

        self.prices.append(data.last_price)

        fast_ma = self._calculate_ma(self.fast_period)
        slow_ma = self._calculate_ma(self.slow_period)

        if fast_ma is None or slow_ma is None:
            return None  # Not enough data

        # Need at least 2 data points to detect crossover
        if len(self.prices) < self.slow_period + 1:
            return None

        # Calculate previous MAs
        prev_prices = self.prices[:-1]
        prev_fast_ma = sum(prev_prices[-self.fast_period:]) / self.fast_period
        prev_slow_ma = sum(prev_prices[-self.slow_period:]) / self.slow_period

        current_position = self.positions[self.symbol]

        # Detect crossover
        if fast_ma > slow_ma and prev_fast_ma <= prev_slow_ma:
            # Bullish crossover - buy signal
            if current_position <= 0:
                size = self.calculate_position_size(self.symbol, 1.0)
                return Order(
                    symbol=self.symbol,
                    side=OrderSide.BUY,
                    quantity=size,
                    order_type=OrderType.MARKET
                )

        elif fast_ma < slow_ma and prev_fast_ma >= prev_slow_ma:
            # Bearish crossover - sell signal
            if current_position >= 0:
                size = self.calculate_position_size(self.symbol, -1.0)
                return Order(
                    symbol=self.symbol,
                    side=OrderSide.SELL,
                    quantity=size,
                    order_type=OrderType.MARKET
                )

        return None

    def calculate_position_size(self, symbol: str, signal_strength: float) -> float:
        return self.max_position * abs(signal_strength)

strategy = SimpleMovingAverageCrossover("AAPL", fast_period=5, slow_period=10)
```

## Trading Performance Metrics

Understanding key metrics is essential for strategy evaluation.

```python
from dataclasses import dataclass
from typing import List, Optional
import math

@dataclass
class TradeRecord:
    """Record of a single trade."""
    symbol: str
    entry_price: float
    exit_price: float
    quantity: float
    side: str  # "long" or "short"
    entry_time: str
    exit_time: str

    @property
    def pnl(self) -> float:
        if self.side == "long":
            return (self.exit_price - self.entry_price) * self.quantity
        else:
            return (self.entry_price - self.exit_price) * self.quantity

    @property
    def return_pct(self) -> float:
        if self.side == "long":
            return (self.exit_price - self.entry_price) / self.entry_price * 100
        else:
            return (self.entry_price - self.exit_price) / self.entry_price * 100

class PerformanceAnalyzer:
    """
    Analyze trading strategy performance.
    """

    def __init__(self, trades: List[TradeRecord], initial_capital: float):
        self.trades = trades
        self.initial_capital = initial_capital

    def total_pnl(self) -> float:
        """Total profit and loss."""
        return sum(t.pnl for t in self.trades)

    def total_return(self) -> float:
        """Total return percentage."""
        return (self.total_pnl() / self.initial_capital) * 100

    def win_rate(self) -> float:
        """Percentage of winning trades."""
        if not self.trades:
            return 0.0
        winners = sum(1 for t in self.trades if t.pnl > 0)
        return (winners / len(self.trades)) * 100

    def profit_factor(self) -> float:
        """Ratio of gross profits to gross losses."""
        gross_profit = sum(t.pnl for t in self.trades if t.pnl > 0)
        gross_loss = abs(sum(t.pnl for t in self.trades if t.pnl < 0))
        if gross_loss == 0:
            return float('inf') if gross_profit > 0 else 0.0
        return gross_profit / gross_loss

    def average_win(self) -> float:
        """Average winning trade."""
        winners = [t.pnl for t in self.trades if t.pnl > 0]
        return sum(winners) / len(winners) if winners else 0.0

    def average_loss(self) -> float:
        """Average losing trade."""
        losers = [t.pnl for t in self.trades if t.pnl < 0]
        return sum(losers) / len(losers) if losers else 0.0

    def expectancy(self) -> float:
        """Expected value per trade."""
        if not self.trades:
            return 0.0
        return self.total_pnl() / len(self.trades)

    def sharpe_ratio(self, risk_free_rate: float = 0.02,
                     periods_per_year: int = 252) -> float:
        """
        Sharpe ratio (annualized).

        Assumes daily returns for periods_per_year.
        """
        if len(self.trades) < 2:
            return 0.0

        returns = [t.return_pct / 100 for t in self.trades]
        mean_return = sum(returns) / len(returns)
        variance = sum((r - mean_return) ** 2 for r in returns) / (len(returns) - 1)
        std_dev = math.sqrt(variance)

        if std_dev == 0:
            return 0.0

        # Annualize
        annual_return = mean_return * periods_per_year
        annual_std = std_dev * math.sqrt(periods_per_year)

        return (annual_return - risk_free_rate) / annual_std

    def max_drawdown(self) -> float:
        """Maximum drawdown percentage."""
        if not self.trades:
            return 0.0

        equity = self.initial_capital
        peak = equity
        max_dd = 0.0

        for trade in self.trades:
            equity += trade.pnl
            if equity > peak:
                peak = equity
            dd = (peak - equity) / peak * 100
            max_dd = max(max_dd, dd)

        return max_dd

    def summary(self) -> dict:
        """Generate performance summary."""
        return {
            "total_trades": len(self.trades),
            "total_pnl": f"${self.total_pnl():,.2f}",
            "total_return": f"{self.total_return():.2f}%",
            "win_rate": f"{self.win_rate():.1f}%",
            "profit_factor": f"{self.profit_factor():.2f}",
            "avg_win": f"${self.average_win():,.2f}",
            "avg_loss": f"${self.average_loss():,.2f}",
            "expectancy": f"${self.expectancy():,.2f}",
            "sharpe_ratio": f"{self.sharpe_ratio():.2f}",
            "max_drawdown": f"{self.max_drawdown():.2f}%"
        }


# Create sample trades
sample_trades = [
    TradeRecord("AAPL", 150.0, 155.0, 100, "long", "2024-01-01", "2024-01-05"),
    TradeRecord("AAPL", 155.0, 152.0, 100, "long", "2024-01-06", "2024-01-08"),
    TradeRecord("GOOGL", 140.0, 145.0, 50, "long", "2024-01-02", "2024-01-10"),
```
