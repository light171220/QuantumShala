## Quantum Pattern Recognition for Trading

Quantum machine learning can identify complex patterns in market data.

```python
"""
Quantum Pattern Recognition for Market Signals
"""

import numpy as np
from typing import List, Dict, Tuple, Optional
from dataclasses import dataclass
from enum import Enum

class MarketRegime(Enum):
    TRENDING = "trending"
    MEAN_REVERTING = "mean_reverting"
    VOLATILE = "volatile"
    CALM = "calm"

@dataclass
class MarketFeatures:
    """Features extracted from market data."""
    returns: np.ndarray
    volatility: float
    momentum: float
    mean_reversion_score: float
    volume_trend: float

class QuantumMarketClassifier:
    """
    Quantum-enhanced market regime classifier.
    Uses quantum feature maps and variational circuits.
    """

    def __init__(self, n_qubits: int = 4):
        self.n_qubits = n_qubits
        self.regime_history: List[MarketRegime] = []

    def extract_features(self, prices: np.ndarray,
                         volumes: np.ndarray) -> MarketFeatures:
        """Extract features for classification."""
        if len(prices) < 20:
            raise ValueError("Need at least 20 price points")

        returns = np.diff(np.log(prices))

        # Volatility (standard deviation of returns)
        volatility = np.std(returns[-20:])

        # Momentum (recent returns)
        momentum = np.sum(returns[-5:])

        # Mean reversion score (autocorrelation)
        mean_reversion = self._calculate_mean_reversion(returns)

        # Volume trend
        volume_ma_short = np.mean(volumes[-5:])
        volume_ma_long = np.mean(volumes[-20:])
        volume_trend = (volume_ma_short / volume_ma_long) - 1

        return MarketFeatures(
            returns=returns,
            volatility=volatility,
            momentum=momentum,
            mean_reversion_score=mean_reversion,
            volume_trend=volume_trend
        )

    def _calculate_mean_reversion(self, returns: np.ndarray) -> float:
        """Calculate mean reversion tendency."""
        if len(returns) < 2:
            return 0.0

        # Negative autocorrelation suggests mean reversion
        autocorr = np.corrcoef(returns[:-1], returns[1:])[0, 1]
        return -autocorr  # Higher = more mean reverting

    def encode_features(self, features: MarketFeatures) -> np.ndarray:
        """
        Encode features for quantum circuit.
        Normalize to [0, π] for rotation angles.
        """
        # Normalize each feature to [0, 1] then scale to [0, π]
        raw = np.array([
            (features.volatility * 50),  # Assume vol typically < 0.02
            (features.momentum + 0.1) / 0.2,  # Center around 0
            (features.mean_reversion_score + 1) / 2,  # [-1, 1] -> [0, 1]
            (features.volume_trend + 0.5) / 1.0  # Center around 0
        ])

        # Clip and scale to [0, π]
        normalized = np.clip(raw, 0, 1) * np.pi
        return normalized

    def classify_regime(self, features: MarketFeatures) -> Tuple[MarketRegime, float]:
        """
        Classify market regime using quantum-inspired approach.
        Returns (regime, confidence).
        """
        # Encode features
        encoded = self.encode_features(features)

        # Simulated quantum classification
        # In practice, this would run on actual quantum hardware

        # Calculate regime scores based on feature combinations
        scores = {
            MarketRegime.TRENDING: self._trending_score(features, encoded),
            MarketRegime.MEAN_REVERTING: self._mean_reversion_score(features, encoded),
            MarketRegime.VOLATILE: self._volatile_score(features, encoded),
            MarketRegime.CALM: self._calm_score(features, encoded)
        }

        # Softmax for probabilities
        max_score = max(scores.values())
        exp_scores = {k: np.exp(v - max_score) for k, v in scores.items()}
        total = sum(exp_scores.values())
        probabilities = {k: v / total for k, v in exp_scores.items()}

        # Best regime
        best_regime = max(probabilities.keys(), key=lambda k: probabilities[k])
        confidence = probabilities[best_regime]

        self.regime_history.append(best_regime)

        return best_regime, confidence

    def _trending_score(self, features: MarketFeatures, encoded: np.ndarray) -> float:
        """Score for trending regime."""
        # High momentum, low mean reversion
        return abs(features.momentum) * 20 - features.mean_reversion_score * 2

    def _mean_reversion_score(self, features: MarketFeatures, encoded: np.ndarray) -> float:
        """Score for mean-reverting regime."""
        return features.mean_reversion_score * 3 - abs(features.momentum) * 10

    def _volatile_score(self, features: MarketFeatures, encoded: np.ndarray) -> float:
        """Score for volatile regime."""
        return features.volatility * 100 + abs(features.volume_trend) * 2

    def _calm_score(self, features: MarketFeatures, encoded: np.ndarray) -> float:
        """Score for calm regime."""
        return (0.02 - features.volatility) * 100 - abs(features.momentum) * 5


class QuantumSignalGenerator:
    """
    Generate trading signals using quantum-enhanced analysis.
    """

    def __init__(self, classifier: QuantumMarketClassifier):
        self.classifier = classifier
        self.signal_history: List[Dict] = []

    def generate_signal(self, prices: np.ndarray,
                        volumes: np.ndarray) -> Dict:
        """
        Generate trading signal based on market regime.
        """
        # Extract features
        features = self.classifier.extract_features(prices, volumes)

        # Classify regime
        regime, confidence = self.classifier.classify_regime(features)

        # Generate signal based on regime
        signal = self._regime_to_signal(regime, features, confidence)

        self.signal_history.append(signal)
        return signal

    def _regime_to_signal(self, regime: MarketRegime,
                          features: MarketFeatures,
                          confidence: float) -> Dict:
        """Convert regime classification to trading signal."""

        if regime == MarketRegime.TRENDING:
            # Follow the trend
            if features.momentum > 0:
                direction = "long"
                strength = min(features.momentum * 10, 1.0)
            else:
                direction = "short"
                strength = min(abs(features.momentum) * 10, 1.0)

        elif regime == MarketRegime.MEAN_REVERTING:
            # Fade recent moves
            if features.momentum > 0:
                direction = "short"  # Fade up move
            else:
                direction = "long"  # Fade down move
            strength = min(features.mean_reversion_score, 1.0)

        elif regime == MarketRegime.VOLATILE:
            # Reduce exposure in volatile markets
            direction = "neutral"
            strength = 0.2  # Low confidence

        else:  # CALM
            # Can be more aggressive in calm markets
            if features.momentum > 0.01:
                direction = "long"
            elif features.momentum < -0.01:
                direction = "short"
            else:
                direction = "neutral"
            strength = 0.5

        # Adjust by regime confidence
        strength *= confidence

        return {
            "regime": regime.value,
            "direction": direction,
            "strength": strength,
            "confidence": confidence,
            "features": {
                "volatility": features.volatility,
                "momentum": features.momentum,
                "mean_reversion": features.mean_reversion_score
            }
        }
```

## QAOA for Portfolio Optimization

Use QAOA to optimize portfolio allocations in real-time.

```python
import numpy as np
from typing import List, Dict, Tuple
from dataclasses import dataclass

@dataclass
class Asset:
    symbol: str
    expected_return: float
    volatility: float

class QAOAPortfolioOptimizer:
    """
    Portfolio optimization using QAOA-inspired approach.
    Finds optimal discrete allocations.
    """

    def __init__(self, assets: List[Asset],
                 correlation_matrix: np.ndarray,
                 risk_aversion: float = 1.0):
        self.assets = assets
        self.correlation = correlation_matrix
        self.risk_aversion = risk_aversion
        self.n_assets = len(assets)

        # Build covariance matrix
        self.covariance = self._build_covariance()

    def _build_covariance(self) -> np.ndarray:
        """Build covariance matrix from volatilities and correlations."""
        n = self.n_assets
        cov = np.zeros((n, n))

        for i in range(n):
            for j in range(n):
                cov[i, j] = (self.assets[i].volatility *
                            self.assets[j].volatility *
                            self.correlation[i, j])

        return cov

    def objective_function(self, weights: np.ndarray) -> float:
        """
        Portfolio objective: maximize return - risk_aversion * variance
        """
        returns = np.array([a.expected_return for a in self.assets])

        portfolio_return = np.dot(weights, returns)
        portfolio_variance = np.dot(weights, np.dot(self.covariance, weights))

        # Objective to maximize
        return portfolio_return - self.risk_aversion * portfolio_variance

    def discretize_weights(self, n_levels: int = 10) -> List[np.ndarray]:
        """
        Generate discrete weight combinations.
        For quantum optimization, we work with discrete allocations.
        """
        # For small number of assets, enumerate possibilities
        from itertools import product

        levels = [i / n_levels for i in range(n_levels + 1)]
        valid_combinations = []

        for combo in product(levels, repeat=self.n_assets):
            if abs(sum(combo) - 1.0) < 0.01:  # Weights sum to ~1
                valid_combinations.append(np.array(combo))

        return valid_combinations

    def optimize_qaoa(self, n_levels: int = 10,
                      p_layers: int = 3) -> Tuple[np.ndarray, float]:
        """
        QAOA-inspired portfolio optimization.

        In a real implementation, this would:
        1. Encode the problem as QUBO
        2. Run QAOA circuits
        3. Measure and extract solution

        Here we simulate the optimization.
        """
        # Get discrete weight options
        weight_options = self.discretize_weights(n_levels)

        if not weight_options:
            # Fallback to equal weights
            return np.ones(self.n_assets) / self.n_assets, 0.0

        # Evaluate all options (QAOA would do this more efficiently)
        best_weights = None
        best_objective = float('-inf')

        for weights in weight_options:
            obj = self.objective_function(weights)
            if obj > best_objective:
                best_objective = obj
                best_weights = weights

        return best_weights, best_objective

    def optimize_with_constraints(self,
                                  min_weight: float = 0.0,
                                  max_weight: float = 0.5,
                                  sector_limits: Dict[str, float] = None) -> Tuple[np.ndarray, float]:
        """
        Optimize with additional constraints.
        """
        weight_options = self.discretize_weights(n_levels=20)

        valid_options = []
        for weights in weight_options:
            # Check min/max constraints
            if all(min_weight <= w <= max_weight for w in weights):
                valid_options.append(weights)

        if not valid_options:
            return np.ones(self.n_assets) / self.n_assets, 0.0

        best_weights = None
        best_objective = float('-inf')

        for weights in valid_options:
            obj = self.objective_function(weights)
            if obj > best_objective:
                best_objective = obj
                best_weights = weights

        return best_weights, best_objective


class DynamicQuantumRebalancer:
    """
    Dynamically rebalance portfolio using quantum optimization.
    """

    def __init__(self, optimizer: QAOAPortfolioOptimizer,
                 rebalance_threshold: float = 0.05):
        self.optimizer = optimizer
        self.rebalance_threshold = rebalance_threshold
        self.current_weights = np.ones(optimizer.n_assets) / optimizer.n_assets
        self.rebalance_history: List[Dict] = []

    def check_rebalance_needed(self, current_weights: np.ndarray) -> bool:
        """Check if rebalancing is needed based on drift."""
        optimal, _ = self.optimizer.optimize_qaoa()
        max_drift = np.max(np.abs(current_weights - optimal))
        return max_drift > self.rebalance_threshold

    def rebalance(self) -> Dict:
        """Perform rebalancing optimization."""
        old_weights = self.current_weights.copy()

        # Optimize
        new_weights, objective = self.optimizer.optimize_qaoa()
        self.current_weights = new_weights

        # Calculate trades needed
        trades = new_weights - old_weights

        result = {
            "old_weights": old_weights.tolist(),
            "new_weights": new_weights.tolist(),
            "trades": trades.tolist(),
            "objective": objective
        }

        self.rebalance_history.append(result)
        return result
```

## Quantum Momentum Strategy

Implement a momentum strategy enhanced with quantum pattern detection.

```python
import numpy as np
from typing import List, Dict, Tuple, Optional
from dataclasses import dataclass

@dataclass
class MomentumSignal:
    symbol: str
    score: float
    direction: str
    quantum_confidence: float

class QuantumMomentumStrategy:
    """
    Momentum strategy enhanced with quantum pattern detection.
    Identifies strong momentum patterns using quantum feature analysis.
    """

    def __init__(self, lookback_periods: List[int] = [5, 10, 20, 60]):
        self.lookback_periods = lookback_periods
        self.position_history: Dict[str, List[float]] = {}

    def calculate_momentum_scores(self,
                                   prices: Dict[str, np.ndarray]) -> List[MomentumSignal]:
        """
        Calculate momentum scores for all symbols.
        """
        signals = []

        for symbol, price_series in prices.items():
            if len(price_series) < max(self.lookback_periods):
                continue

            # Calculate returns for each lookback
            returns = {}
            for period in self.lookback_periods:
                ret = (price_series[-1] / price_series[-period]) - 1
                returns[period] = ret

            # Quantum-enhanced scoring
            score, confidence = self._quantum_score(returns, price_series)

            direction = "long" if score > 0 else "short"

            signals.append(MomentumSignal(
                symbol=symbol,
                score=score,
                direction=direction,
                quantum_confidence=confidence
            ))

        # Sort by absolute score
        signals.sort(key=lambda s: abs(s.score), reverse=True)

        return signals

    def _quantum_score(self,
                       returns: Dict[int, float],
                       prices: np.ndarray) -> Tuple[float, float]:
        """
        Calculate quantum-enhanced momentum score.
        Combines multiple timeframes with consistency weighting.
        """
        # Calculate weighted score
        weights = {5: 0.4, 10: 0.3, 20: 0.2, 60: 0.1}

        weighted_return = sum(
            returns.get(period, 0) * weight
            for period, weight in weights.items()
        )

        # Consistency score (do all timeframes agree?)
        signs = [np.sign(r) for r in returns.values()]
        consistency = abs(sum(signs)) / len(signs)

        # Volatility adjustment
        daily_returns = np.diff(np.log(prices[-20:]))
        volatility = np.std(daily_returns)
        vol_adjustment = 1 / (1 + volatility * 10)

        # Final score
        score = weighted_return * consistency * vol_adjustment

        # Confidence based on consistency and volatility
        confidence = consistency * (1 - min(volatility * 20, 0.5))

        return score, confidence

    def generate_portfolio(self,
                           signals: List[MomentumSignal],
                           n_long: int = 5,
                           n_short: int = 5) -> Dict[str, float]:
        """
        Generate portfolio from momentum signals.
        """
        portfolio = {}

        # Top momentum for long
        long_signals = [s for s in signals if s.direction == "long"][:n_long]

        # Bottom momentum for short (if allowed)
        short_signals = [s for s in signals if s.direction == "short"][:n_short]

        # Equal weight within long/short buckets
        long_weight = 0.5 / len(long_signals) if long_signals else 0
        short_weight = -0.5 / len(short_signals) if short_signals else 0

        for signal in long_signals:
            # Weight by confidence
            portfolio[signal.symbol] = long_weight * signal.quantum_confidence

        for signal in short_signals:
            portfolio[signal.symbol] = short_weight * signal.quantum_confidence

        return portfolio
```
