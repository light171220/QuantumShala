## 3. Operations Implementation

```python
import hashlib
import secrets
import numpy as np
from typing import Tuple

class MLKEMDemo:
    """
    Simplified ML-KEM for educational purposes.
    Production: Use FIPS-validated library!
    """
    
    def __init__(self, variant: str = 'ML-KEM-768'):
        params = MLKEMParameters.PARAMS[variant]
        self.n = params['n']
        self.k = params['k']
        self.q = params['q']
        self.variant = variant
    
    def keygen(self) -> Tuple[bytes, bytes]:
        """ML-KEM.KeyGen() → (ek, dk)"""
        # Generate random seeds
        d = secrets.token_bytes(32)
        z = secrets.token_bytes(32)
        
        # In real implementation:
        # 1. Expand d to get matrix A and secret s
        # 2. Compute t = As + e
        # 3. ek = encode(t, rho)
        # 4. dk = encode(s, ek, H(ek), z)
        
        # Simplified: return placeholder
        ek = hashlib.sha3_256(d).digest() + secrets.token_bytes(self.k * self.n // 4)
        dk = ek + z + secrets.token_bytes(self.k * self.n // 4)
        
        return ek, dk
    
    def encaps(self, ek: bytes) -> Tuple[bytes, bytes]:
        """ML-KEM.Encaps(ek) → (K, c)"""
        # Generate random m
        m = secrets.token_bytes(32)
        
        # In real implementation:
        # 1. (K_bar, r) = G(m || H(ek))
        # 2. c = encrypt using r
        # 3. K = KDF(K_bar || H(c))
        
        # Simplified
        K = hashlib.sha3_256(m + ek[:32]).digest()
        c = secrets.token_bytes(MLKEMParameters.PARAMS[self.variant]['ct_bytes'])
        
        return K, c
    
    def decaps(self, dk: bytes, c: bytes) -> bytes:
        """ML-KEM.Decaps(dk, c) → K"""
        # In real implementation:
        # 1. Decrypt c to get m'
        # 2. Re-encrypt to get c'
        # 3. If c == c': return KDF(K_bar || H(c))
        # 4. Else: return KDF(z || H(c))  ← IMPLICIT REJECTION
        
        # Simplified
        K = hashlib.sha3_256(dk[:32] + c[:32]).digest()
        return K


def demonstrate_ml_kem():
    """Demonstrate ML-KEM key exchange."""
    print("\nML-KEM KEY EXCHANGE DEMO")
    print("-" * 50)
    
    kem = MLKEMDemo('ML-KEM-768')
    
    # Server generates keys
    ek, dk = kem.keygen()
    print(f"Server KeyGen:")
    print(f"  Encapsulation key: {len(ek)} bytes")
    print(f"  Decapsulation key: {len(dk)} bytes")
    
    # Client encapsulates
    K_client, ciphertext = kem.encaps(ek)
    print(f"\nClient Encaps:")
    print(f"  Shared secret: {K_client.hex()[:32]}...")
    print(f"  Ciphertext: {len(ciphertext)} bytes")
    
    # Server decapsulates
    K_server = kem.decaps(dk, ciphertext)
    print(f"\nServer Decaps:")
    print(f"  Shared secret: {K_server.hex()[:32]}...")
    
    print("\n(Demo only - use FIPS-validated library in production)")

demonstrate_ml_kem()
```

## 4. FIPS Compliance Requirements

```python
def fips_compliance():
    """FIPS 203 compliance requirements."""
    print("\nFIPS 203 COMPLIANCE")
    print("=" * 70)
    
    print("""
    MANDATORY REQUIREMENTS:
    ━━━━━━━━━━━━━━━━━━━━━━━
    
    1. APPROVED RNG
       • SP 800-90A DRBG required
       • Proper entropy source (SP 800-90B)
       • Cannot use weak/predictable randomness
    
    2. IMPLICIT REJECTION
       • On decapsulation failure: return H(z || c)
       • Must be indistinguishable from success
       • Prevents chosen-ciphertext attacks
       
       ┌─────────────────────────────────────────────────────┐
       │ if c_prime != c:                                   │
       │     return KDF(z || H(c))  # Looks like valid K    │
       │ else:                                              │
       │     return KDF(K_bar || H(c))  # Real K            │
       └─────────────────────────────────────────────────────┘
    
    3. CONSTANT-TIME OPERATIONS
       • No secret-dependent branches
       • No secret-dependent memory access
       • Prevents timing side channels
    
    4. INPUT VALIDATION
       • Check ciphertext length
       • Validate public key format
       • Reject malformed inputs
    
    
    RECOMMENDED PRACTICES:
    ┌─────────────────────────────────────────────────────────────┐
    │ • Use ML-KEM-768 for general applications                  │
    │ • Use ML-KEM-1024 for long-term secrets (>20 years)        │
    │ • Implement hybrid with ECDH during transition             │
    │ • Use FIPS 140-3 validated modules                         │
    │ • Clear sensitive data after use                           │

    """)
```

## Summary

ML-KEM (FIPS 203) is the primary post-quantum key encapsulation standard:

1. **Three variants**: 512/768/1024 for Categories 1/3/5
2. **Based on**: Module-LWE hard problem
3. **Operations**: KeyGen, Encapsulate, Decapsulate
4. **Critical requirements**: Implicit rejection, constant-time, approved RNG
5. **Performance**: Often faster than classical ECDH
6. **Migration**: Start hybrid now, transition to PQC-only by 2035
7. **Recommendation**: Use ML-KEM-768 for most applications
