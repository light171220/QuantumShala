def phased_rollout():
    """Implementing phased rollout."""
    print("\nPHASED ROLLOUT IMPLEMENTATION")
    print("=" * 70)
    
    print("""
    TRAFFIC PERCENTAGE CONTROL:
    ━━━━━━━━━━━━━━━━━━━━━━━━━━━━
    
    Method 1: Load Balancer Rules
    ┌─────────────────────────────────────────────────────────────────┐
    │ Route 5% of traffic to hybrid-enabled servers                  │
    │ Route 95% to classical-only servers                            │
    └─────────────────────────────────────────────────────────────────┘
    
    Method 2: Feature Flags
    ┌─────────────────────────────────────────────────────────────────┐
    │ if feature_flag("hybrid_enabled", user_id):                    │
    │     use_hybrid_crypto()                                        │
    │ else:                                                          │
    │     use_classical_crypto()                                     │
    └─────────────────────────────────────────────────────────────────┘
    
    Method 3: Canary Servers
    ┌─────────────────────────────────────────────────────────────────┐
    │ Designate specific servers as canaries                         │
    │ Deploy hybrid to canaries first                                │
    │ Monitor canaries closely                                       │
    │ Promote to full fleet if successful                            │
    └─────────────────────────────────────────────────────────────────┘
    """)
    
    print('''
    ```python
    class HybridRolloutController:
        """Control hybrid cryptography rollout percentage."""
        
        def __init__(self, rollout_percentage: int = 0):
            self.rollout_percentage = rollout_percentage
        
        def should_use_hybrid(self, request_id: str) -> bool:
            """Determine if this request should use hybrid."""
            # Consistent hashing for sticky sessions
            hash_val = hash(request_id) % 100
            return hash_val < self.rollout_percentage
        
        def update_percentage(self, new_percentage: int):
            """Update rollout percentage (0-100)."""
            if 0 <= new_percentage <= 100:
                self.rollout_percentage = new_percentage
                log(f"Hybrid rollout updated to {new_percentage}%")
        
        def emergency_disable(self):
            """Immediately disable hybrid."""
            self.rollout_percentage = 0
            alert("HYBRID DISABLED - Emergency rollback triggered")
    
    
    # Usage
    controller = HybridRolloutController(rollout_percentage=5)
    
    def handle_tls_connection(client_request):
        if controller.should_use_hybrid(client_request.id):
            return negotiate_hybrid_tls(client_request)
        else:
            return negotiate_classical_tls(client_request)
    ```
    ''')

## 3. Fallback Mechanisms

```python
def fallback_mechanisms():
    """Implementing graceful fallback."""
    print("\nFALLBACK MECHANISMS")
    print("=" * 70)
    
    print("""
    NEGOTIATION WITH FALLBACK:
    ━━━━━━━━━━━━━━━━━━━━━━━━━━━
    
    Client offers algorithms in preference order:
    
    ClientHello:
        supported_groups: [
            X25519MLKEM768,    # Preferred: hybrid
            X25519,            # Fallback 1: modern classical
            prime256v1         # Fallback 2: legacy
        ]
    
    Server selects best mutually supported:
    
    ┌─────────────────────────────────────────────────────────────────┐
    │ Server Supports    │ Client Offers       │ Result              │
    ├────────────────────┼─────────────────────┼─────────────────────┤
    │ Hybrid + Classical │ Hybrid + Classical  │ Hybrid (best)       │
    │ Classical only     │ Hybrid + Classical  │ Classical (fallback)│
    │ Hybrid only        │ Classical only      │ Connection fails    │
    └─────────────────────────────────────────────────────────────────┘
    
    
    GRACEFUL DEGRADATION LEVELS:
    ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
    
    LEVEL 1: Algorithm Fallback
    • Client/server negotiate to classical
    • Automatic, transparent
    • No action needed
    
    LEVEL 2: Configuration Rollback
    • Disable hybrid in config
    • Restart not always needed
    • Fast recovery
    
    LEVEL 3: Deployment Rollback
    • Deploy previous version
    • Full rollback procedure
    • Slower but complete
    """)
    
    print('''
    ```python
    class HybridNegotiator:
        """Negotiate crypto with graceful fallback."""
        
        ALGORITHMS = [
            ('X25519MLKEM768', 'hybrid'),
            ('X25519', 'classical'),
            ('P256', 'legacy')
        ]
        
        def negotiate(self, client_supported: list, 
                     server_supported: list) -> tuple:
            """Find best mutually supported algorithm."""
            for alg, category in self.ALGORITHMS:
                if alg in client_supported and alg in server_supported:
                    return alg, category
            
            raise NegotiationError("No common algorithms")
        
        def handle_connection(self, client_hello):
            try:
                alg, category = self.negotiate(
                    client_hello.supported_groups,
                    self.server_supported
                )
                
                if category == 'hybrid':
                    return self.hybrid_handshake(client_hello, alg)
                else:
                    return self.classical_handshake(client_hello, alg)
                    
            except NegotiationError:
                metrics.increment("negotiation_failure")
                raise
            except HybridError as e:
                # Hybrid failed, try classical
                metrics.increment("hybrid_fallback")
                return self.classical_handshake(client_hello, 'X25519')
    ```
    ''')

fallback_mechanisms()
```

## Summary

Successful hybrid deployment requires careful planning:

1. **Phased Rollout**: Start small (5%), grow gradually
2. **Fallback**: Always maintain classical fallback
3. **Monitoring**: Track algorithm distribution, latency, errors
4. **Alerting**: Automated detection of issues
5. **Rollback**: Quick ability to disable hybrid

The goal is quantum protection with minimal risk to operations.
