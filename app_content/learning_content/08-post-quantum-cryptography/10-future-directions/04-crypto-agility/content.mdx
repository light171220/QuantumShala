def implementation_patterns():
    """Crypto-agile implementation patterns."""
    print("\nIMPLEMENTATION PATTERNS")
    print("=" * 70)
    
    print("""
    PATTERN 1: PROVIDER INTERFACE
    ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
    """)
    
    # Show code example
    print('''
    ```python
    from abc import ABC, abstractmethod
    
    class KEMProvider(ABC):
        """Abstract KEM provider for crypto agility."""
        
        @abstractmethod
        def keygen(self) -> tuple:
            pass
        
        @abstractmethod
        def encapsulate(self, public_key: bytes) -> tuple:
            pass
        
        @abstractmethod
        def decapsulate(self, secret_key: bytes, 
                       ciphertext: bytes) -> bytes:
            pass
    
    class MLKEMProvider(KEMProvider):
        """ML-KEM implementation."""
        def keygen(self):
            return ml_kem_keygen()
        # ... etc
    
    class X25519Provider(KEMProvider):
        """Classical fallback."""
        # ... implementation
    
    # Usage - algorithm selected at runtime
    def establish_key(provider: KEMProvider, peer_pk):
        shared_secret, ct = provider.encapsulate(peer_pk)
        return shared_secret
    ```
    ''')
    
    print("""
    
    PATTERN 2: ALGORITHM REGISTRY
    ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
    """)
    
    print('''
    ```python
    class AlgorithmRegistry:
        """Central registry for crypto algorithms."""
        
        _providers = {}
        _deprecated = set()
        
        @classmethod
        def register(cls, algorithm_id: str, provider):
            cls._providers[algorithm_id] = provider
        
        @classmethod
        def get(cls, algorithm_id: str):
            if algorithm_id in cls._deprecated:
                raise DeprecatedAlgorithmError(algorithm_id)
            return cls._providers[algorithm_id]
        
        @classmethod
        def deprecate(cls, algorithm_id: str):
            cls._deprecated.add(algorithm_id)
    
    # Registration at startup
    AlgorithmRegistry.register("ML-KEM-768", MLKEMProvider())
    AlgorithmRegistry.register("X25519", X25519Provider())
    
    # Deprecation without code change
    AlgorithmRegistry.deprecate("RSA-2048")
    ```
    ''')
    
    print("""
    
    PATTERN 3: TAGGED CIPHERTEXT
    ━━━━━━━━━━━━━━━━━━━━━━━━━━━━
    """)
    
    print('''
    ```python
    import struct
    
    def encrypt_with_tag(algorithm_id: int, key: bytes, 
                        plaintext: bytes) -> bytes:
        """Encrypt and prepend algorithm identifier."""
        provider = get_provider(algorithm_id)
        ciphertext = provider.encrypt(key, plaintext)
        
        # Tag format: 4-byte algorithm ID + ciphertext
        return struct.pack(">I", algorithm_id) + ciphertext
    
    def decrypt_tagged(key: bytes, tagged_ct: bytes) -> bytes:
        """Decrypt using tagged algorithm identifier."""
        algorithm_id = struct.unpack(">I", tagged_ct[:4])[0]
        ciphertext = tagged_ct[4:]
        
        provider = get_provider(algorithm_id)
        return provider.decrypt(key, ciphertext)
    ```
    ''')

implementation_patterns()
```

def migration_support():
    """Supporting algorithm migration in agile systems."""
    print("\nMIGRATION SUPPORT")
    print("=" * 70)
    
    print("""
    MIGRATION STRATEGIES:
    ━━━━━━━━━━━━━━━━━━━━━━
    
    1. PARALLEL SUPPORT
       
       Keep old and new algorithms active simultaneously:
       • New data encrypted with new algorithm
       • Old data decrypted with old algorithm
       • Gradual re-encryption of old data
    
    2. GRACE PERIOD DEPRECATION
       
       Timeline:
       Day 0:   New algorithm added
       Day 30:  Warning when old algorithm used
       Day 90:  Old algorithm disabled for new operations
       Day 180: Old algorithm removed entirely
    
    3. RE-ENCRYPTION PIPELINE
       
       Background job to re-encrypt stored data:
       • Read with old algorithm
       • Write with new algorithm
       • Update metadata
       • Verify and delete old version
    
    
    CERTIFICATE MIGRATION:
    ━━━━━━━━━━━━━━━━━━━━━━━

## Summary

Cryptographic agility is essential for long-term security:

1. **Abstraction**: Separate algorithms from business logic
2. **Identification**: Tag all encrypted data with algorithm ID
3. **Negotiation**: Support algorithm negotiation in protocols
4. **Configuration**: Algorithm choices in config, not code
5. **Testing**: Regularly test your ability to swap algorithms

Build crypto agility now to handle the next algorithm break gracefully.
