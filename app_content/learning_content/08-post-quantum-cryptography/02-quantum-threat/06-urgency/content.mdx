### The Delay Penalty Curve

```python
def delay_penalty_model(years_delayed: int) -> dict:
    """
    Model the penalty for delaying migration.
    """
    # Base assumptions
    base_cost = 100  # Arbitrary units
    base_risk = 10
    base_exposure = 1000  # TB
    
    # Penalties compound with delay
    cost_multiplier = 1.3 ** years_delayed  # 30% annual increase
    risk_multiplier = 1.5 ** years_delayed  # 50% annual increase
    exposure_added = 365 * years_delayed  # TB per year
    
    return {
        'years_delayed': years_delayed,
        'cost': f"{base_cost * cost_multiplier:.0f} units ({cost_multiplier:.1f}x)",
        'risk': f"{base_risk * risk_multiplier:.0f} ({risk_multiplier:.1f}x)",
        'additional_exposure': f"{exposure_added:,} TB",
        'migration_time_remaining': max(0, 10 - years_delayed)  # Assuming 10yr to CRQC
    }

print("Delay Penalty Model")
print("=" * 50)
print(f"{'Years Delayed':<15} {'Cost':<20} {'Risk':<15} {'Added Exposure'}")
print("-" * 60)

for years in [0, 1, 2, 3, 5, 7]:
    result = delay_penalty_model(years)
    print(f"{years:<15} {result['cost']:<20} {result['risk']:<15} {result['additional_exposure']}")
```

## 5. Urgency Without Panic

<Tip>
The goal is sustained action, not fear. Frame urgency around opportunity and preparedness, not disaster.
</Tip>

```python
def communication_framework():
    """
    Framework for communicating urgency appropriately.
    """
    messaging = {
        'Do Say': [
            '"We have an opportunity to lead in post-quantum security"',
            '"Early preparation gives us more options and lower costs"',
            '"Our customers expect us to protect their data long-term"',
            '"Standards are ready - we can begin confidently"',
            '"A phased approach manages risk and cost effectively"'
        ],
        'Dont Say': [
            '"Everything is going to be hacked" (causes paralysis)',
            '"We need to replace everything immediately" (unrealistic)',
            '"If we dont act now, were doomed" (fear-based)',
            '"This is the most important security issue ever" (hyperbole)',
            '"Quantum computers will destroy civilization" (alarmist)'
        ],
        'Key Messages': {
            'For Executives': 'Strategic risk management and competitive advantage',
            'For Technical Teams': 'Exciting technology challenge with clear standards',
            'For Employees': 'Continuing our commitment to data protection',
            'For Customers': 'Investing in long-term security of your data'
        }
    }
    
    print("COMMUNICATION FRAMEWORK")
    print("=" * 60)
    
    print("\n✓ DO SAY:")
    for msg in messaging['Do Say']:
        print(f"  {msg}")
    
    print("\n✗ DONT SAY:")
    for msg in messaging['Dont Say']:
        print(f"  {msg}")
    
    print("\nKEY MESSAGES BY AUDIENCE:")
    for audience, message in messaging['Key Messages'].items():
        print(f"  {audience}: {message}")

communication_framework()
```
