## 1. The Scope of Vulnerability

The quantum threat isn't theoretical—it affects virtually every secure system in use today.

<Definition term="Quantum-Vulnerable System">
Any system that relies on RSA, Diffie-Hellman, ECDH, ECDSA, or other algorithms broken by Shor's algorithm for security-critical operations like key exchange, digital signatures, or encryption.
</Definition>

## 8. Vulnerability Assessment Framework

```python
def assess_system_vulnerability(system_info: dict) -> dict:
    """
    Assess a system's quantum vulnerability.
    
    Args:
        system_info: dict with crypto_algorithms, data_sensitivity,
                    retention_period, update_difficulty
    
    Returns:
        Vulnerability assessment with priority score
    """
    score = 0
    findings = []
    
    # Check for Shor-vulnerable algorithms
    shor_vulnerable = ['RSA', 'ECDSA', 'ECDH', 'DH', 'DSA', 'ECC']
    for algo in system_info.get('crypto_algorithms', []):
        if any(v in algo.upper() for v in shor_vulnerable):
            score += 30
            findings.append(f"CRITICAL: {algo} broken by Shor's algorithm")
    
    # Check data sensitivity
    sensitivity_scores = {'low': 5, 'medium': 15, 'high': 25, 'critical': 35}
    sensitivity = system_info.get('data_sensitivity', 'medium')
    score += sensitivity_scores.get(sensitivity, 15)
    
    # Check retention period (years of secrecy needed)
    retention = system_info.get('retention_years', 5)
    if retention > 20:
        score += 20
        findings.append(f"HIGH: {retention}-year retention requires immediate action")
    elif retention > 10:
        score += 10
        findings.append(f"MEDIUM: {retention}-year retention needs planning")
    
    # Check update difficulty
    update_difficulty = system_info.get('update_difficulty', 'medium')
    difficulty_scores = {'easy': 0, 'medium': 10, 'hard': 20, 'very_hard': 30}
    score += difficulty_scores.get(update_difficulty, 10)
    if update_difficulty in ['hard', 'very_hard']:
        findings.append(f"WARNING: {update_difficulty} update process - start early")
    
    # Determine priority
    if score >= 70:
        priority = 'CRITICAL'
    elif score >= 50:
        priority = 'HIGH'
    elif score >= 30:
        priority = 'MEDIUM'
    else:
        priority = 'LOW'
    
    return {
        'vulnerability_score': score,
        'priority': priority,
        'findings': findings,
        'recommendation': get_recommendation(priority)
    }

def get_recommendation(priority: str) -> str:
    recommendations = {
        'CRITICAL': 'Begin PQC migration immediately. Implement hybrid crypto.',
        'HIGH': 'Plan migration within 1-2 years. Assess hybrid options.',
        'MEDIUM': 'Include in 3-5 year roadmap. Monitor PQC standards.',
        'LOW': 'Monitor developments. Update when standards mature.'
    }
    return recommendations.get(priority, 'Review system requirements')

# Example assessment
example_system = {
    'name': 'Healthcare Records System',
    'crypto_algorithms': ['RSA-2048', 'AES-256', 'ECDSA'],
    'data_sensitivity': 'critical',
    'retention_years': 50,
    'update_difficulty': 'hard'
}

result = assess_system_vulnerability(example_system)
print(f"Assessment for: {example_system['name']}")
print(f"Score: {result['vulnerability_score']}")
print(f"Priority: {result['priority']}")
print(f"Findings:")
for finding in result['findings']:
    print(f"  • {finding}")
print(f"Recommendation: {result['recommendation']}")
```
