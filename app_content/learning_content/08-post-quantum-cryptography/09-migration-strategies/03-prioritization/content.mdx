def prioritization_framework():

    ━━━━━━━━━━━━━━━━━━━━━━

    Maximum score: 60
    
    Priority 1 (Critical): Score > 45
    Priority 2 (High):     Score 35-45
    Priority 3 (Medium):   Score 25-34
    Priority 4 (Low):      Score 15-24
    Priority 5 (Defer):    Score < 15
    """)

prioritization_framework()


class MigrationPrioritizer:
    """Calculate migration priorities."""
    
    @staticmethod
    def calculate_priority(risk_scores: dict, effort_scores: dict, 
                          impact_scores: dict) -> dict:
        """Calculate overall priority score."""
        risk_total = sum(risk_scores.values())
        effort_total = sum(effort_scores.values())
        impact_total = sum(impact_scores.values())
        
        # Risk weighted 2x
        total = (risk_total * 2) + effort_total + impact_total
        
        if total > 45:
            priority = 1
            label = "CRITICAL"
        elif total > 35:
            priority = 2
            label = "HIGH"
        elif total > 25:
            priority = 3
            label = "MEDIUM"
        elif total > 15:
            priority = 4
            label = "LOW"
        else:
            priority = 5
            label = "DEFER"
        
        return {
            'score': total,
            'priority': priority,
            'label': label,
            'risk': risk_total,
            'effort': effort_total,
            'impact': impact_total
        }

# Example
print("\nEXAMPLE PRIORITIZATION:")
print("-" * 40)
prioritizer = MigrationPrioritizer()

# Customer API (high priority)
api_result = prioritizer.calculate_priority(
    risk_scores={'sensitivity': 4, 'duration': 4, 'exposure': 5, 'harvest': 4, 'regulatory': 3},
    effort_scores={'complexity': 3, 'dependencies': 3, 'testing': 3, 'vendor': 4, 'skills': 3},
    impact_scores={'criticality': 5, 'users': 5, 'revenue': 5, 'downtime': 5, 'compliance': 4}
)
print(f"Customer API: {api_result}")

# Internal tool (lower priority)

## Summary

Migration prioritization ensures you protect the most critical assets first:

1. **Score**: Risk, Effort, Impact on 1-5 scale
2. **Weight**: Risk gets 2x weight (security is primary driver)
3. **Categorize**: Critical, High, Medium, Low, Defer
4. **Sequence**: Respect dependencies, include quick wins
5. **Re-evaluate**: Quarterly review, immediate on security events

Prioritization is not static—re-evaluate as circumstances change.
