        """Check if actor has required capabilities."""
        return required.issubset(self.capabilities)

from dataclasses import dataclass
from datetime import datetime, timedelta
from typing import Optional

@dataclass
class DataSecurityRequirement:
    """Security lifetime requirement for data."""
    data_type: str
    required_protection_years: int
    regulatory_requirement: Optional[str] = None

    def hndl_risk_level(self, quantum_eta_years: int = 10) -> str:
        """
        Assess HNDL risk level.

        Args:
            quantum_eta_years: Estimated years until quantum threat
        """
        if self.required_protection_years > quantum_eta_years:
            if self.required_protection_years > quantum_eta_years * 2:
                return "CRITICAL"
            return "HIGH"
        elif self.required_protection_years > quantum_eta_years / 2:
            return "MEDIUM"
        return "LOW"

from dataclasses import dataclass, field
from typing import List, Dict
from enum import Enum

class AttackCategory(Enum):
    MATHEMATICAL = "Mathematical/Cryptanalytic"
    IMPLEMENTATION = "Implementation"
    SIDE_CHANNEL = "Side-Channel"
    PROTOCOL = "Protocol"
    OPERATIONAL = "Operational"

class Severity(Enum):
    CRITICAL = 4
    HIGH = 3
    MEDIUM = 2
    LOW = 1

@dataclass
class AttackVector:
    """Description of an attack vector."""
    name: str
    category: AttackCategory
    severity: Severity
    description: str
    mitigations: List[str]
    pqc_specific: bool = False  # Is this PQC-specific?

@dataclass
class ThreatModel:
    """Complete threat model for a PQC system."""
    system_name: str
    assets: List[str]
    attack_vectors: List[AttackVector] = field(default_factory=list)

    def add_vector(self, vector: AttackVector):
        self.attack_vectors.append(vector)

    def get_critical_vectors(self) -> List[AttackVector]:
        return [v for v in self.attack_vectors
                if v.severity in (Severity.CRITICAL, Severity.HIGH)]

    def summarize(self):
        """Print threat model summary."""
        print(f"Threat Model: {self.system_name}")
        print("=" * 60)
        print(f"\nProtected Assets:")
        for asset in self.assets:
            print(f"  • {asset}")

        print(f"\nAttack Vectors by Category:")
        by_category: Dict[AttackCategory, List[AttackVector]] = {}
        for v in self.attack_vectors:
            by_category.setdefault(v.category, []).append(v)

        for cat, vectors in by_category.items():
            print(f"\n  {cat.value}:")
            for v in vectors:
                pqc = " [PQC]" if v.pqc_specific else ""
                print(f"    [{v.severity.name}] {v.name}{pqc}")

## Building a PQC Threat Model

```python
from dataclasses import dataclass, field
from typing import List, Set, Dict
from enum import Enum
import json

class MitigationStatus(Enum):
    NOT_STARTED = "Not Started"
    IN_PROGRESS = "In Progress"
    IMPLEMENTED = "Implemented"
    VERIFIED = "Verified"

@dataclass
class ThreatEntry:
    """Individual threat in the model."""
    id: str
    name: str
    description: str
    category: str
    likelihood: int  # 1-5
    impact: int      # 1-5
    mitigations: List[str]
    status: MitigationStatus = MitigationStatus.NOT_STARTED

    @property
    def risk_score(self) -> int:
        return self.likelihood * self.impact

    @property
    def risk_level(self) -> str:
        score = self.risk_score
        if score >= 20:
            return "CRITICAL"
        elif score >= 12:
            return "HIGH"
        elif score >= 6:
            return "MEDIUM"
        return "LOW"

class PQCThreatModel:
    """
    Comprehensive PQC threat model builder.

    Follows STRIDE-like methodology adapted for cryptographic systems.
    """

    def __init__(self, system_name: str):
        self.system_name = system_name
        self.threats: List[ThreatEntry] = []
        self.assumptions: List[str] = []

    def add_quantum_threats(self):
        """Add standard quantum-related threats."""
        quantum_threats = [
            ThreatEntry(
                id="Q1",
                name="HNDL Attack",
                description="Adversary stores encrypted traffic for future quantum decryption",
                category="Quantum",
                likelihood=5,  # Already happening
                impact=5,      # Complete confidentiality loss
                mitigations=[
                    "Deploy PQC for key exchange immediately",
                    "Use hybrid schemes during transition",
                    "Inventory and prioritize sensitive data"
                ]
            ),
            ThreatEntry(
                id="Q2",
                name="Shor's Algorithm",
                description="Quantum factoring/discrete log breaks RSA/ECC",
                category="Quantum",
                likelihood=3,  # Future threat
                impact=5,      # Complete break
                mitigations=[
                    "Migrate to NIST PQC standards",
                    "Remove classical-only code paths",
                    "Plan certificate rotation"
                ]
            ),
            ThreatEntry(
                id="Q3",
                name="Grover's Algorithm",
                description="Quadratic speedup on symmetric key search",
                category="Quantum",
                likelihood=3,
                impact=2,      # Mitigated by larger keys
                mitigations=[
                    "Use AES-256 instead of AES-128",
                    "Use SHA-384/512 for hashing",
                    "Double symmetric key sizes"
                ]
            ),
        ]
        self.threats.extend(quantum_threats)

    def add_implementation_threats(self):
        """Add implementation-related threats."""
        impl_threats = [
            ThreatEntry(
                id="I1",
                name="Timing Side Channel",
                description="Variable-time operations leak secret information",
                category="Implementation",
                likelihood=4,
                impact=4,
                mitigations=[
                    "Use constant-time implementations",
                    "Verify with timing analysis tools",
                    "Use validated crypto libraries"
                ]
            ),
            ThreatEntry(
                id="I2",
                name="Fault Injection",
                description="Induced faults reveal secret keys",
                category="Implementation",
                likelihood=2,  # Requires physical access
                impact=5,
                mitigations=[
                    "Implement fault detection",
                    "Use redundant computations",
                    "Hardware countermeasures"
                ]
            ),
            ThreatEntry(
                id="I3",
                name="RNG Weakness",
                description="Weak randomness compromises key generation",
                category="Implementation",
                likelihood=3,
                impact=5,
                mitigations=[
                    "Use OS-provided CSPRNG",
                    "Add entropy from multiple sources",
                    "Implement RNG health checks"
                ]
            ),
        ]
        self.threats.extend(impl_threats)

    def add_protocol_threats(self):
        """Add protocol-level threats."""
        protocol_threats = [
            ThreatEntry(
                id="P1",
                name="Downgrade Attack",
                description="Force system to use weaker algorithms",
                category="Protocol",
                likelihood=4,
                impact=4,
                mitigations=[
                    "Enforce minimum algorithm strength",
                    "Signed algorithm negotiation",
                    "Reject legacy-only connections"
                ]
            ),
            ThreatEntry(
                id="P2",
                name="Hybrid Composition Flaw",
                description="Incorrect combination of classical and PQC",
                category="Protocol",
                likelihood=3,
                impact=4,
                mitigations=[
                    "Use standardized hybrid constructions",
                    "Proper key derivation (HKDF)",
                    "Security proofs for composition"
                ]
            ),
        ]
        self.threats.extend(protocol_threats)

    def build_standard_model(self) -> 'PQCThreatModel':
        """Build a standard PQC threat model."""
        self.add_quantum_threats()
        self.add_implementation_threats()
        self.add_protocol_threats()

        self.assumptions = [
            "NIST PQC parameters provide adequate security margin",
            "Quantum computers with sufficient qubits are 10-15 years away",
            "Hash functions remain secure against quantum attacks",
            "Implementation follows constant-time practices",
        ]
        return self

    def get_high_risk_threats(self) -> List[ThreatEntry]:
        """Get threats requiring immediate attention."""
        return [t for t in self.threats if t.risk_level in ("CRITICAL", "HIGH")]

    def generate_report(self) -> str:
        """Generate threat model report."""
        lines = [
            f"# Threat Model: {self.system_name}",
            "",
            "## Assumptions",
        ]
        for assumption in self.assumptions:
            lines.append(f"- {assumption}")

        lines.extend(["", "## Threat Summary", ""])

        # Group by category
        by_category: Dict[str, List[ThreatEntry]] = {}
        for t in self.threats:
            by_category.setdefault(t.category, []).append(t)

        for category, threats in by_category.items():
            lines.append(f"### {category} Threats")
            lines.append("")
            lines.append("| ID | Threat | Risk | Status |")
            lines.append("|---|---|---|---|")
            for t in sorted(threats, key=lambda x: -x.risk_score):
                lines.append(f"| {t.id} | {t.name} | {t.risk_level} | {t.status.value} |")
            lines.append("")

        # Priority recommendations
        lines.extend(["## Priority Actions", ""])
        for t in self.get_high_risk_threats():
            lines.append(f"### {t.id}: {t.name}")
            lines.append(f"Risk Score: {t.risk_score} ({t.risk_level})")
            lines.append("")
            lines.append("Mitigations:")
            for m in t.mitigations:
                lines.append(f"- {m}")
            lines.append("")

        return "\n".join(lines)

# Build and display threat model
model = PQCThreatModel("Enterprise PQC Migration")
model.build_standard_model()
print(model.generate_report())
```

## Summary

Threat modeling for PQC must account for:

1. **Quantum-specific threats**: HNDL attacks, Shor's algorithm, Grover speedup
2. **Traditional threats**: Side channels, fault injection, implementation bugs
3. **Transition threats**: Downgrade attacks, hybrid composition, algorithm agility

The key insight is that HNDL attacks are happening *now*—data encrypted today with classical cryptography may be decrypted in the future. This makes PQC migration urgent for long-lived secrets.
