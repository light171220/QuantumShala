```python
class SecureMemory:
    """
    Secure memory handling for cryptographic secrets.

    Demonstrates proper secret handling practices.
    """

    def __init__(self, size: int):
        self.size = size
        # In practice, use mlock() to prevent swapping
        self._buffer = bytearray(size)
        self._is_locked = False

    def write(self, data: bytes, offset: int = 0):
        """Write data to secure memory."""
        if offset + len(data) > self.size:
            raise ValueError("Write exceeds buffer size")
        self._buffer[offset:offset + len(data)] = data

    def read(self) -> bytes:
        """Read data from secure memory."""
        return bytes(self._buffer)

    def secure_zero(self):
        """
        Securely zero the memory.

        Must prevent compiler from optimizing this away.
        """
        # Multiple overwrites with different patterns
        for pattern in [0x00, 0xFF, 0xAA, 0x55, 0x00]:
            for i in range(self.size):
                self._buffer[i] = pattern

        # Verify zeroing completed
        if any(b != 0 for b in self._buffer):
            raise RuntimeError("Secure zeroing failed!")

    def __del__(self):
        """Destructor ensures cleanup."""
        try:
            self.secure_zero()
        except Exception:
            pass  # Best effort in destructor

    def __enter__(self):
        return self

    def __exit__(self, *args):
        self.secure_zero()
```

```python
class SecureNonceGeneration:
    """
    Secure nonce generation for ML-DSA.

    Implements hedged randomness to protect against
    RNG failures and nonce reuse.
    """

    def __init__(self, secret_seed: bytes):
        """
        Initialize with secret seed for deterministic component.

        secret_seed should be derived from the signing key.
        """
        self.secret_seed = secret_seed

    def generate_nonce(
        self,
        message: bytes,
        randomness: Optional[bytes] = None
    ) -> bytes:
        """
        Generate signing nonce with hedged randomness.

        Combines:
        1. Secret seed (deterministic, key-bound)
        2. Message hash (binds nonce to message)
        3. Fresh randomness (when available)

        Even if RNG fails, nonce differs per message.
        """
        # Get fresh randomness if not provided
        if randomness is None:
            try:
                randomness = os.urandom(32)
            except Exception:
                # RNG failed - use empty (hedging still protects)
                randomness = b""

        # Combine all inputs
        nonce_input = b"".join([
            self.secret_seed,
            hashlib.sha3_256(message).digest(),
            randomness
        ])

        # Derive nonce
        return hashlib.sha3_256(nonce_input).digest()

    def verify_uniqueness(
        self,
        nonces: list
    ) -> Tuple[bool, int]:
        """
        Verify a set of nonces are all unique.

        Returns (all_unique, duplicate_count).
        """
        seen = set()
        duplicates = 0

        for nonce in nonces:
            nonce_hex = nonce.hex()
            if nonce_hex in seen:
                duplicates += 1
            seen.add(nonce_hex)

        return duplicates == 0, duplicates
```

## Secure Coding Practices

```python
from typing import TypeVar, Generic, Callable
from contextlib import contextmanager
import os

T = TypeVar('T')

class SecureCodingPatterns:
    """
    Secure coding patterns for PQC implementations.
    """

    @staticmethod
    @contextmanager
    def secure_memory_context(size: int):
        """
        Context manager for secure memory handling.

        Ensures memory is zeroed on exit.
        """
        buffer = bytearray(size)
        try:
            yield buffer
        finally:
            # Secure zeroization
            for i in range(size):
                buffer[i] = 0
            # Memory barrier would go here in C

    @staticmethod
    def validate_or_raise(
        condition: bool,
        error_message: str
    ) -> None:
        """
        Validate condition or raise security error.

        Use for all security-critical checks.
        """
        if not condition:
            raise SecurityError(error_message)

    @staticmethod
    def constant_time_compare(a: bytes, b: bytes) -> bool:
        """
        Constant-time byte comparison.

        Prevents timing attacks on comparison.
        """
        if len(a) != len(b):
            return False

        result = 0
        for x, y in zip(a, b):
            result |= x ^ y

        return result == 0

    @staticmethod
    def secure_random_bytes(n: int) -> bytes:
        """
        Generate cryptographically secure random bytes.

        Uses OS-provided CSPRNG.
        """
        try:
            return os.urandom(n)
        except Exception as e:
            raise SecurityError(
                f"Failed to generate secure randomness: {e}"
            )


class SecurityError(Exception):
    """Security-related error."""
    pass


class SecurePQCWrapper:
    """
    Secure wrapper for PQC operations.

    Implements defense-in-depth for cryptographic operations.
    """

    def __init__(self, implementation):
        self.impl = implementation
        self.operations_count = 0
        self.max_operations = 1_000_000  # Rate limiting

    def keygen(self):
        """
        Generate key pair with security checks.
        """
        self._check_rate_limit()

        # Generate keys
        pk, sk = self.impl.keygen()

        # Validate generated keys
        if not self._validate_keypair(pk, sk):
            raise SecurityError("Generated keypair validation failed")

        self.operations_count += 1
        return pk, sk

    def encapsulate(self, pk: bytes):
        """
        Encapsulate with input validation.
        """
        self._check_rate_limit()

        # Validate public key
        if not self._validate_public_key(pk):
            raise SecurityError("Invalid public key")

        # Perform encapsulation
        ct, ss = self.impl.encapsulate(pk)

        # Validate output
        if len(ss) != 32:
            raise SecurityError("Invalid shared secret length")

        self.operations_count += 1
        return ct, ss

    def decapsulate(self, sk: bytes, ct: bytes):
        """
        Decapsulate with input validation.
        """
        self._check_rate_limit()

        # Validate inputs
        if not self._validate_secret_key(sk):
            raise SecurityError("Invalid secret key format")

        if not self._validate_ciphertext(ct):
            raise SecurityError("Invalid ciphertext format")

        # Perform decapsulation with implicit rejection
        ss = self.impl.decapsulate(sk, ct)

        self.operations_count += 1
        return ss

    def _check_rate_limit(self):
        """Check operation rate limit."""
        if self.operations_count >= self.max_operations:
            raise SecurityError("Operation rate limit exceeded")

    def _validate_public_key(self, pk: bytes) -> bool:
        """Validate public key structure."""
        # Implementation-specific validation
        return len(pk) > 0

    def _validate_secret_key(self, sk: bytes) -> bool:
        """Validate secret key structure."""
        return len(sk) > 0

    def _validate_ciphertext(self, ct: bytes) -> bool:
        """Validate ciphertext structure."""
        return len(ct) > 0

    def _validate_keypair(self, pk: bytes, sk: bytes) -> bool:
        """Validate keypair consistency."""
        # Would verify pk can be derived from sk
        return True
```

## Code Review Checklist

```
PQC Implementation Security Review Checklist:
═══════════════════════════════════════════════════════════════════

□ Memory Safety
  □ All buffers sized correctly for algorithm parameters
  □ No stack-allocated large structures (use heap)
  □ Bounds checking on all array accesses
  □ Secrets zeroed after use (verified not optimized away)
  □ No use-after-free or double-free

□ Input Validation
  □ All input lengths verified before processing
  □ Polynomial coefficients checked against bounds
  □ Ciphertext/signature structure validated
  □ Public key validity verified
  □ Reject malformed inputs early

□ Randomness
  □ Uses OS-provided CSPRNG (urandom, CryptGenRandom)
  □ Sufficient entropy for key generation
  □ Hedged randomness for signing
  □ No predictable seeds or fallbacks

□ Constant-Time Operations
  □ No secret-dependent branches
  □ No secret-dependent memory access patterns
  □ Constant-time comparison for validation checks
  □ Fixed-iteration loops for rejection sampling

□ API Safety
  □ Type-safe interfaces (can't confuse pk/sk/ct)
  □ Clear ownership semantics for key material
  □ Thread-safety documented and enforced
  □ Error handling doesn't leak information

□ Cryptographic Correctness
  □ Known Answer Tests (KAT) pass
  □ Interoperability tests with reference implementation
  □ Edge case testing (min/max parameter values)
  □ Fuzzing for crash bugs
```

## Summary

Implementation security requires attention to:

1. **Memory safety**: Buffer management, secret cleanup, allocation patterns
2. **Input validation**: Length checks, structural validation, bounds verification
3. **Nonce management**: Unique nonces, hedged randomness, no reuse
4. **Secure coding**: Constant-time operations, proper error handling, defense in depth

PQC's larger data structures and new operations introduce unique implementation challenges.
