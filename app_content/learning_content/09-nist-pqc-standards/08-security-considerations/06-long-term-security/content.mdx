# Long-Term Security Planning for PQC

Cryptographic algorithms have finite lifetimes. Planning for agility and smooth transitions is essential for maintaining security over decades.

## Crypto Agility Architecture

```python
from abc import ABC, abstractmethod
from typing import Dict, List, Optional, Type
from dataclasses import dataclass
from enum import Enum

class AlgorithmStatus(Enum):
    """Algorithm lifecycle status."""
    EXPERIMENTAL = "experimental"      # Not for production
    RECOMMENDED = "recommended"        # Current best practice
    ACCEPTABLE = "acceptable"          # Still secure, not preferred
    DEPRECATED = "deprecated"          # Phase out, security concerns
    FORBIDDEN = "forbidden"            # Must not use

@dataclass
class AlgorithmProfile:
    """Profile of a cryptographic algorithm."""
    name: str
    oid: str
    status: AlgorithmStatus
    security_bits: int
    quantum_safe: bool
    deprecation_date: Optional[str] = None
    replacement: Optional[str] = None

class CryptoAgileRegistry:
    """
    Central registry for algorithm management.

    Enables organization-wide crypto agility by providing
    a single source of truth for algorithm policies.
    """

    def __init__(self):
        self.algorithms: Dict[str, AlgorithmProfile] = {}
        self._initialize_defaults()

    def _initialize_defaults(self):
        """Initialize with standard algorithms."""
        defaults = [
            # PQC - Recommended
            AlgorithmProfile(
                "ML-KEM-768", "2.16.840.1.101.3.4.4.2",
                AlgorithmStatus.RECOMMENDED, 192, True
            ),
            AlgorithmProfile(
                "ML-KEM-1024", "2.16.840.1.101.3.4.4.3",
                AlgorithmStatus.RECOMMENDED, 256, True
            ),
            AlgorithmProfile(
                "ML-DSA-65", "2.16.840.1.101.3.4.3.18",
                AlgorithmStatus.RECOMMENDED, 192, True
            ),
            AlgorithmProfile(
                "ML-DSA-87", "2.16.840.1.101.3.4.3.19",
                AlgorithmStatus.RECOMMENDED, 256, True
            ),

            # Hybrid - Acceptable during transition
            AlgorithmProfile(
                "X25519-ML-KEM-768", "TBD",
                AlgorithmStatus.ACCEPTABLE, 192, True
            ),
            AlgorithmProfile(
                "ECDSA-P384-ML-DSA-65", "TBD",
                AlgorithmStatus.ACCEPTABLE, 192, True
            ),

            # Classical - Deprecated
            AlgorithmProfile(
                "RSA-2048", "1.2.840.113549.1.1.1",
                AlgorithmStatus.DEPRECATED, 112, False,
                deprecation_date="2030-12-31",
                replacement="ML-KEM-768"
            ),
            AlgorithmProfile(
                "ECDSA-P256", "1.2.840.10045.4.3.2",
                AlgorithmStatus.DEPRECATED, 128, False,
                deprecation_date="2030-12-31",
                replacement="ML-DSA-65"
            ),

            # Forbidden
            AlgorithmProfile(
                "RSA-1024", "1.2.840.113549.1.1.1",
                AlgorithmStatus.FORBIDDEN, 80, False
            ),
        ]

        for algo in defaults:
            self.algorithms[algo.name] = algo

    def is_allowed(self, name: str) -> bool:
        """Check if algorithm is allowed for use."""
        algo = self.algorithms.get(name)
        if not algo:
            return False
        return algo.status not in (
            AlgorithmStatus.DEPRECATED,
            AlgorithmStatus.FORBIDDEN
        )

    def get_replacement(self, name: str) -> Optional[str]:
        """Get recommended replacement for deprecated algorithm."""
        algo = self.algorithms.get(name)
        return algo.replacement if algo else None

    def get_recommended(self, quantum_safe: bool = True) -> List[str]:
        """Get list of recommended algorithms."""
        return [
            name for name, algo in self.algorithms.items()
            if algo.status == AlgorithmStatus.RECOMMENDED
            and algo.quantum_safe == quantum_safe
        ]

    def update_status(
        self,
        name: str,
        new_status: AlgorithmStatus,
        reason: str
    ):
        """Update algorithm status (e.g., after security advisory)."""
        if name in self.algorithms:
            old_status = self.algorithms[name].status
            self.algorithms[name].status = new_status
            print(f"POLICY UPDATE: {name}")
            print(f"  {old_status.value} -> {new_status.value}")
            print(f"  Reason: {reason}")


class CryptoAgileComponent(ABC):
    """
    Base class for crypto-agile components.

    Components inherit from this to gain algorithm switching capability.
    """

    def __init__(self, registry: CryptoAgileRegistry):
        self.registry = registry
        self.current_algorithm: Optional[str] = None

    @abstractmethod
    def switch_algorithm(self, algorithm: str) -> bool:
        """Switch to a different algorithm."""
        pass

    def validate_algorithm(self, algorithm: str) -> bool:
        """Validate algorithm is allowed by policy."""
        if not self.registry.is_allowed(algorithm):
            replacement = self.registry.get_replacement(algorithm)
            if replacement:
                print(f"WARNING: {algorithm} is deprecated. "
                      f"Use {replacement} instead.")
            else:
                print(f"ERROR: {algorithm} is not allowed.")
            return False
        return True
```

```

         │ ╲

     192 ─┤     ╲─────────────────────────────────────────────────
         │       ╲

         │               ╲

              Now        10yr        20yr        30yr      Time

├─ Data lifetime < 10 years: Level 3 (192-bit) sufficient

└─ Critical infrastructure: Always use highest level
```

```python
from dataclasses import dataclass
from datetime import datetime, timedelta
from typing import List, Tuple

@dataclass
class SecurityRequirement:
    """Security requirements for a data classification."""
    name: str
    protection_years: int
    minimum_security_bits: int
    must_be_quantum_safe: bool

class SecurityMarginPlanner:
    """
    Plan security margins based on data lifetime requirements.
    """

    # Assume security erodes ~2 bits per decade from cryptanalysis
    EROSION_RATE_BITS_PER_DECADE = 2

    def __init__(self):
        self.requirements: List[SecurityRequirement] = []

    def add_requirement(self, req: SecurityRequirement):
        """Add a security requirement."""
        self.requirements.append(req)

    def calculate_needed_security(
        self,
        target_security_bits: int,
        protection_years: int
    ) -> int:
        """
        Calculate security bits needed today for future protection.

        Accounts for expected erosion over time.
        """
        decades = protection_years / 10
        erosion = int(decades * self.EROSION_RATE_BITS_PER_DECADE)
        return target_security_bits + erosion

    def recommend_algorithm(
        self,
        requirement: SecurityRequirement,
        available_algorithms: dict
    ) -> Tuple[str, str]:
        """
        Recommend algorithm based on security requirement.

        Returns (algorithm_name, reasoning).
        """
        needed_bits = self.calculate_needed_security(
            requirement.minimum_security_bits,
            requirement.protection_years
        )

        # Filter by quantum safety if required
        candidates = [
            (name, info) for name, info in available_algorithms.items()
            if info['security_bits'] >= needed_bits
            and (not requirement.must_be_quantum_safe or info['quantum_safe'])
        ]

        if not candidates:
            return (None, f"No algorithm meets {needed_bits}-bit requirement")

        # Sort by security level (prefer not over-engineering)
        candidates.sort(key=lambda x: x[1]['security_bits'])
        best = candidates[0]

        reasoning = (
            f"Requirement: {requirement.minimum_security_bits} bits + "
            f"{requirement.protection_years}yr = {needed_bits} bits needed. "
            f"{best[0]} provides {best[1]['security_bits']} bits."
        )

        return (best[0], reasoning)

    def generate_policy(self) -> str:
        """Generate security policy document."""
        lines = [
            "CRYPTOGRAPHIC SECURITY POLICY",
            "=" * 50,
            "",
            "Data Classification Requirements:",
            ""
        ]

        algorithms = {
            "ML-KEM-512": {"security_bits": 128, "quantum_safe": True},
            "ML-KEM-768": {"security_bits": 192, "quantum_safe": True},
            "ML-KEM-1024": {"security_bits": 256, "quantum_safe": True},
            "ML-DSA-44": {"security_bits": 128, "quantum_safe": True},
            "ML-DSA-65": {"security_bits": 192, "quantum_safe": True},
            "ML-DSA-87": {"security_bits": 256, "quantum_safe": True},
        }

        for req in self.requirements:
            algo, reason = self.recommend_algorithm(req, algorithms)
            lines.extend([
                f"## {req.name}",
                f"   Protection Period: {req.protection_years} years",
                f"   Minimum Security: {req.minimum_security_bits} bits",
                f"   Quantum-Safe Required: {req.must_be_quantum_safe}",
                f"   Recommended Algorithm: {algo}",
                f"   Reasoning: {reason}",
                ""
            ])

        return "\n".join(lines)

# Example usage
planner = SecurityMarginPlanner()
planner.add_requirement(SecurityRequirement(
    "Top Secret", 50, 256, True
))
planner.add_requirement(SecurityRequirement(
    "Confidential", 25, 192, True
))
planner.add_requirement(SecurityRequirement(
    "Internal", 10, 128, True
))
planner.add_requirement(SecurityRequirement(
    "Session Keys", 1, 128, True
))

print(planner.generate_policy())
```

```python
from dataclasses import dataclass, field
from typing import List, Dict, Optional
from datetime import datetime, date
from enum import Enum

class MigrationPhase(Enum):
    PREPARATION = 1
    HYBRID = 2
    PQC_PRIMARY = 3
    CLASSICAL_SUNSET = 4

@dataclass
class System:
    """System requiring cryptographic migration."""
    name: str
    data_sensitivity: str  # low, medium, high, critical
    current_algorithms: List[str]
    supports_pqc: bool = False
    migration_phase: MigrationPhase = MigrationPhase.PREPARATION
    migration_date: Optional[date] = None

@dataclass
class MigrationPlan:
    """Organization-wide migration plan."""
    organization: str
    systems: List[System] = field(default_factory=list)
    target_dates: Dict[MigrationPhase, date] = field(default_factory=dict)

    def add_system(self, system: System):
        self.systems.append(system)

    def prioritize_systems(self) -> List[System]:
        """
        Prioritize systems for migration.

        Critical and high-sensitivity systems go first.
        """
        priority_order = {'critical': 0, 'high': 1, 'medium': 2, 'low': 3}
        return sorted(
            self.systems,
            key=lambda s: priority_order.get(s.data_sensitivity, 4)
        )

    def get_phase_systems(self, phase: MigrationPhase) -> List[System]:
        """Get systems in a specific migration phase."""
        return [s for s in self.systems if s.migration_phase == phase]

    def calculate_progress(self) -> Dict[str, float]:
        """Calculate migration progress by phase."""
        total = len(self.systems)
        if total == 0:
            return {}

        progress = {}
        for phase in MigrationPhase:
            count = len(self.get_phase_systems(phase))
            progress[phase.name] = (count / total) * 100

        # Calculate overall (% past preparation)
        migrated = sum(
            len(self.get_phase_systems(p))
            for p in MigrationPhase
            if p != MigrationPhase.PREPARATION
        )
        progress['OVERALL'] = (migrated / total) * 100

        return progress

    def generate_report(self) -> str:
        """Generate migration status report."""
        lines = [
            f"PQC MIGRATION STATUS: {self.organization}",
            f"Report Date: {datetime.now().strftime('%Y-%m-%d')}",
            "=" * 60,
            ""
        ]

        # Progress summary
        progress = self.calculate_progress()
        lines.append("PROGRESS BY PHASE:")
        for phase, pct in progress.items():
            bar = "█" * int(pct / 5) + "░" * (20 - int(pct / 5))
            lines.append(f"  {phase:20} [{bar}] {pct:.1f}%")

        lines.append("")
        lines.append("PRIORITY SYSTEMS (Top 10):")
        for system in self.prioritize_systems()[:10]:
            status = "✓ PQC" if system.supports_pqc else "✗ Classical"
            lines.append(
                f"  [{system.data_sensitivity:8}] {system.name:30} {status}"
            )

        lines.append("")
        lines.append("TARGET DATES:")
        for phase, target in self.target_dates.items():
            lines.append(f"  {phase.name}: {target}")

        return "\n".join(lines)

class TransitionManager:
    """
    Manage algorithm transition across organization.
    """

    def __init__(self, plan: MigrationPlan):
        self.plan = plan
        self.events: List[str] = []

    def migrate_system(
        self,
        system_name: str,
        target_phase: MigrationPhase
    ) -> bool:
        """Migrate a system to new phase."""
        for system in self.plan.systems:
            if system.name == system_name:
                old_phase = system.migration_phase
                system.migration_phase = target_phase
                system.migration_date = date.today()

                if target_phase in (MigrationPhase.HYBRID,
                                   MigrationPhase.PQC_PRIMARY):
                    system.supports_pqc = True

                self.events.append(
                    f"{datetime.now()}: {system_name} migrated "
                    f"{old_phase.name} -> {target_phase.name}"
                )
                return True
        return False

    def check_deadline_compliance(self) -> List[str]:
        """Check systems against phase deadlines."""
        warnings = []
        today = date.today()

        for phase, deadline in self.plan.target_dates.items():
            if today > deadline:
                # Find systems that should have migrated
                for system in self.plan.systems:
                    if system.migration_phase.value < phase.value:
                        warnings.append(
                            f"OVERDUE: {system.name} should be in "
                            f"{phase.name} by {deadline}"
                        )

        return warnings
```

## Version Negotiation

```python
from typing import List, Tuple, Optional
from dataclasses import dataclass
from enum import Enum

class PreferenceLevel(Enum):
    """Algorithm preference levels."""
    REQUIRED = 3      # Must use, fail if unavailable
    PREFERRED = 2     # Use if available
    ACCEPTABLE = 1    # Can use if nothing better
    REJECTED = 0      # Never use

@dataclass
class AlgorithmPreference:
    """Algorithm with preference level."""
    algorithm: str
    level: PreferenceLevel
    security_bits: int

class VersionNegotiator:
    """
    Negotiate algorithm versions between parties.

    Implements secure downgrade-resistant negotiation.
    """

    def __init__(self, local_preferences: List[AlgorithmPreference]):
        self.local = sorted(
            local_preferences,
            key=lambda x: (-x.level.value, -x.security_bits)
        )

    def negotiate(
        self,
        remote_supported: List[str]
    ) -> Tuple[Optional[str], str]:
        """
        Negotiate algorithm with remote party.

        Returns (selected_algorithm, negotiation_log).
        """
        log_lines = ["Algorithm Negotiation:"]

        # First, check for required algorithms
        for pref in self.local:
            if pref.level == PreferenceLevel.REQUIRED:
                if pref.algorithm not in remote_supported:
                    log_lines.append(
                        f"  FAIL: Required {pref.algorithm} not supported"
                    )
                    return (None, "\n".join(log_lines))

        # Find best mutually supported algorithm
        for pref in self.local:
            if pref.level == PreferenceLevel.REJECTED:
                continue

            if pref.algorithm in remote_supported:
                log_lines.append(
                    f"  Selected: {pref.algorithm} "
                    f"({pref.level.name}, {pref.security_bits}-bit)"
                )
                return (pref.algorithm, "\n".join(log_lines))

            log_lines.append(
                f"  {pref.algorithm}: not supported by remote"
            )

        log_lines.append("  FAIL: No mutually acceptable algorithm")
        return (None, "\n".join(log_lines))

    def check_downgrade_attempt(
        self,
        negotiated: str,
        expected_minimum: PreferenceLevel
    ) -> bool:
        """
        Check if negotiation resulted in downgrade attack.

        Returns True if potential downgrade detected.
        """
        for pref in self.local:
            if pref.algorithm == negotiated:
                if pref.level.value < expected_minimum.value:
                    return True  # Downgrade detected
                return False
        return True  # Unknown algorithm is suspicious

# Example secure negotiation
local_prefs = [
    AlgorithmPreference("ML-KEM-768", PreferenceLevel.PREFERRED, 192),
    AlgorithmPreference("X25519-ML-KEM-768", PreferenceLevel.ACCEPTABLE, 192),
    AlgorithmPreference("X25519", PreferenceLevel.REJECTED, 128),  # Classical only
]

negotiator = VersionNegotiator(local_prefs)

# Remote supports hybrid and classical
remote = ["X25519-ML-KEM-768", "X25519", "P-256"]
selected, log = negotiator.negotiate(remote)
print(log)
print(f"\nFinal selection: {selected}")

# Check for downgrade
if selected and negotiator.check_downgrade_attempt(
    selected, PreferenceLevel.PREFERRED
):
    print("WARNING: Potential downgrade attack detected!")
```

## Summary

Long-term security planning requires:

1. **Crypto agility**: Design systems to switch algorithms without major changes
2. **Security margins**: Account for erosion over data lifetime
3. **Migration planning**: Phased approach from hybrid to pure PQC
4. **Version negotiation**: Secure algorithm selection with downgrade protection

The quantum threat makes crypto agility more important than ever—plan today for transitions that may be necessary in the future.
