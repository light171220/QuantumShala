# Industry Standards and Guidelines for PQC

Beyond government mandates, industry standards organizations are incorporating PQC into their specifications. These standards ensure interoperability and provide implementation guidance.

```python
from dataclasses import dataclass, field
from typing import List, Optional
from enum import Enum
from datetime import date

class StandardStatus(Enum):
    DRAFT = "Draft"
    WG_ADOPTED = "WG Adopted"
    LAST_CALL = "Last Call"
    IESG_REVIEW = "IESG Review"
    RFC = "Published RFC"

@dataclass
class IETFDocument:
    """IETF document tracking."""
    name: str
    title: str
    working_group: str
    status: StandardStatus
    pqc_algorithms: List[str]
    description: str
    expected_rfc: Optional[date] = None

class IETFPQCTracker:
    """
    Track IETF PQC-related documents.
    """

    def __init__(self):
        self.documents = self._load_documents()

    def _load_documents(self) -> List[IETFDocument]:
        return [
            IETFDocument(
                "draft-ietf-tls-hybrid-design",
                "Hybrid Key Exchange in TLS 1.3",
                "TLS",
                StandardStatus.WG_ADOPTED,
                ["ML-KEM-768", "X25519"],
                "Defines hybrid key exchange for TLS 1.3 combining classical ECDH with ML-KEM",
                date(2025, 6, 1)
            ),
            IETFDocument(
                "draft-ietf-lamps-dilithium-certificates",
                "ML-DSA in Internet X.509 Certificates",
                "LAMPS",
                StandardStatus.WG_ADOPTED,
                ["ML-DSA-44", "ML-DSA-65", "ML-DSA-87"],
                "Defines encoding of ML-DSA keys and signatures in X.509 certificates",
                date(2025, 3, 1)
            ),
            IETFDocument(
                "draft-ietf-lamps-cms-kyber",
                "ML-KEM in CMS",
                "LAMPS",
                StandardStatus.WG_ADOPTED,
                ["ML-KEM-512", "ML-KEM-768", "ML-KEM-1024"],
                "Using ML-KEM in Cryptographic Message Syntax for encryption",
                date(2025, 3, 1)
            ),
            IETFDocument(
                "draft-ietf-lamps-pq-composite-sigs",
                "Composite ML-DSA Signatures",
                "LAMPS",
                StandardStatus.WG_ADOPTED,
                ["ML-DSA + ECDSA", "ML-DSA + RSA"],
                "Composite signatures combining ML-DSA with classical algorithms",
                date(2025, 6, 1)
            ),
            IETFDocument(
                "RFC 9370",
                "ML-KEM in IKEv2",
                "IPsec",
                StandardStatus.RFC,
                ["ML-KEM-768", "ML-KEM-1024"],
                "Post-quantum key exchange for IPsec VPN using ML-KEM",
                None
            ),
            IETFDocument(
                "draft-ietf-openpgp-pqc",
                "PQC in OpenPGP",
                "OpenPGP",
                StandardStatus.WG_ADOPTED,
                ["ML-KEM", "ML-DSA", "SLH-DSA"],
                "Adding PQC algorithms to OpenPGP (GnuPG, etc.)",
                date(2025, 12, 1)
            ),
        ]

    def get_by_status(self, status: StandardStatus) -> List[IETFDocument]:
        return [d for d in self.documents if d.status == status]

    def get_by_algorithm(self, algorithm: str) -> List[IETFDocument]:
        return [d for d in self.documents
                if any(algorithm in a for a in d.pqc_algorithms)]

    def generate_status_report(self) -> str:
        """Generate IETF PQC standards status report."""
        lines = [
            "IETF PQC STANDARDS STATUS",
            "=" * 60,
        ]

        # Group by status
        for status in StandardStatus:
            docs = self.get_by_status(status)
            if docs:
                lines.append(f"\n{status.value}:")
                for doc in docs:
                    lines.append(f"  {doc.name}")
                    lines.append(f"    {doc.title}")
                    lines.append(f"    WG: {doc.working_group}")
                    lines.append(f"    Algorithms: {', '.join(doc.pqc_algorithms)}")

        return "\n".join(lines)

tracker = IETFPQCTracker()
print(tracker.generate_status_report())
```

from dataclasses import dataclass
from typing import List, Dict

@dataclass
class SectorStandard:
    """Sector-specific standard with PQC implications."""
    sector: str
    standard: str
    organization: str
    current_crypto: str
    pqc_status: str
    recommended_action: str
    timeline: str

class SectorStandardsTracker:
    """
    Track sector-specific standards for PQC.
    """

    def __init__(self):
        self.standards = self._load_standards()

    def _load_standards(self) -> List[SectorStandard]:
        return [
            SectorStandard(

                "PCI SSC",
                "TLS 1.2+, approved ciphers",
                "Monitoring; PQC expected 2026-2028",
                "Begin PQC readiness assessment",
                "2026-2028"
            ),
            SectorStandard(
                "Financial",
                "ISO 20022",
                "ISO TC68",
                "XML-DSig with RSA/ECDSA",
                "No current PQC requirements",
                "Plan for ML-DSA signature support",
                "TBD"
            ),
            SectorStandard(
                "Healthcare",
                "HL7 FHIR R4",
                "HL7 International",
                "TLS for transport",
                "Follows TLS standards",
                "Ensure PQC TLS in FHIR servers",
                "With TLS updates"
            ),
            SectorStandard(
                "Energy",
                "IEC 62351",
                "IEC TC57",
                "TLS, digital signatures",
                "PQC study group formed",
                "Monitor IEC working group",
                "2027-2030"
            ),
            SectorStandard(
                "Telecom",
                "3GPP TS 33.501 (5G)",
                "3GPP SA3",
                "ECIES, AES-256",
                "PQC feasibility study complete",
                "Track 3GPP SA3 progress",
                "Release 18+"
            ),
            SectorStandard(
                "Government",
                "NIST SP 800-52 (TLS)",
                "NIST",
                "TLS 1.2/1.3 with FIPS algos",
                "Being updated for PQC",
                "Follow revised guidelines",
                "2025"
            ),
        ]

    def get_by_sector(self, sector: str) -> List[SectorStandard]:
        return [s for s in self.standards
                if sector.lower() in s.sector.lower()]

    def generate_sector_report(self, sector: str) -> str:
        """Generate sector-specific standards report."""
        standards = self.get_by_sector(sector)

        lines = [
            f"PQC STANDARDS: {sector.upper()} SECTOR",
            "=" * 50,
        ]

        if not standards:
            lines.append(f"No specific standards tracked for {sector}")
            return "\n".join(lines)

        for std in standards:
            lines.extend([
                f"\n{std.standard}",
                f"  Organization: {std.organization}",
                f"  Current Crypto: {std.current_crypto}",
                f"  PQC Status: {std.pqc_status}",
                f"  Timeline: {std.timeline}",
                f"  Action: {std.recommended_action}",
            ])

        return "\n".join(lines)

tracker = SectorStandardsTracker()
print(tracker.generate_sector_report("Financial"))
print()
print(tracker.generate_sector_report("Telecom"))

## Summary

Industry standards for PQC are rapidly developing:

1. **IETF**: Active work on TLS, certificates, CMS, and protocol integration
2. **Sector standards**: Financial, healthcare, energy, telecom all preparing
3. **Monitoring**: Track key sources for timely awareness of changes
4. **Engagement**: Participate in standards development where relevant

Standards compliance ensures interoperability and demonstrates due diligence.
