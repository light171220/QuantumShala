## Cryptographic Policy Template

```python
from dataclasses import dataclass, field
from typing import List, Dict, Optional
from datetime import date
from enum import Enum

class AlgorithmApproval(Enum):
    APPROVED = "Approved"
    DEPRECATED = "Deprecated"
    PROHIBITED = "Prohibited"
    TRANSITIONAL = "Transitional"

@dataclass
class ApprovedAlgorithm:
    """Approved algorithm entry."""
    name: str
    type: str  # KEM, Signature, Symmetric, Hash
    standard: str  # FIPS 203, etc.
    parameters: List[str]
    approval: AlgorithmApproval
    use_cases: List[str]
    sunset_date: Optional[date] = None

@dataclass
class CryptographicPolicy:
    """Organization cryptographic policy document."""
    organization: str
    version: str
    effective_date: date
    review_date: date
    owner: str
    approved_by: str

    purpose: str = ""
    scope: str = ""
    algorithms: List[ApprovedAlgorithm] = field(default_factory=list)
    key_management_requirements: List[str] = field(default_factory=list)
    compliance_requirements: List[str] = field(default_factory=list)
    exceptions_process: str = ""

    def add_algorithm(self, algo: ApprovedAlgorithm):
        self.algorithms.append(algo)

    def get_approved_for_use(self, use_case: str) -> List[ApprovedAlgorithm]:
        """Get algorithms approved for specific use case."""
        return [
            a for a in self.algorithms
            if a.approval in (AlgorithmApproval.APPROVED,
                              AlgorithmApproval.TRANSITIONAL)
            and use_case in a.use_cases
        ]

    def generate_document(self) -> str:
        """Generate policy document text."""
        lines = [
            f"CRYPTOGRAPHIC POLICY",
            f"Organization: {self.organization}",
            f"Version: {self.version}",
            f"Effective: {self.effective_date}",
            f"Next Review: {self.review_date}",
            f"Owner: {self.owner}",
            f"Approved By: {self.approved_by}",
            "=" * 60,
            "",
            "1. PURPOSE",
            self.purpose,
            "",
            "2. SCOPE",
            self.scope,
            "",
            "3. APPROVED ALGORITHMS",
        ]

        for approval in AlgorithmApproval:
            algos = [a for a in self.algorithms if a.approval == approval]
            if algos:
                lines.append(f"\n  3.{approval.name}:")
                for algo in algos:
                    lines.append(f"    {algo.name} ({algo.type})")
                    lines.append(f"      Standard: {algo.standard}")
                    lines.append(f"      Parameters: {', '.join(algo.parameters)}")
                    lines.append(f"      Use Cases: {', '.join(algo.use_cases)}")
                    if algo.sunset_date:
                        lines.append(f"      Sunset: {algo.sunset_date}")

        lines.extend([
            "",
            "4. KEY MANAGEMENT REQUIREMENTS",
        ])
        for req in self.key_management_requirements:
            lines.append(f"  • {req}")

        lines.extend([
            "",
            "5. COMPLIANCE REQUIREMENTS",
        ])
        for req in self.compliance_requirements:
            lines.append(f"  • {req}")

        lines.extend([
            "",
            "6. EXCEPTIONS PROCESS",
            self.exceptions_process,
        ])

        return "\n".join(lines)

# Create sample policy
policy = CryptographicPolicy(
    organization="Example Corporation",
    version="2.0",
    effective_date=date(2024, 1, 1),
    review_date=date(2025, 1, 1),
    owner="CISO",
    approved_by="CTO",
    purpose="Define approved cryptographic algorithms and their usage for protecting organizational data and communications.",
    scope="All systems processing, storing, or transmitting sensitive data.",
    exceptions_process="Exceptions require written approval from CISO with documented risk acceptance and remediation timeline."
)

# Add approved algorithms
policy.add_algorithm(ApprovedAlgorithm(
    "ML-KEM", "KEM", "FIPS 203",
    ["ML-KEM-768", "ML-KEM-1024"],
    AlgorithmApproval.APPROVED,
    ["Key exchange", "Data encryption key wrapping"]
))

policy.add_algorithm(ApprovedAlgorithm(
    "ML-DSA", "Signature", "FIPS 204",
    ["ML-DSA-65", "ML-DSA-87"],
    AlgorithmApproval.APPROVED,
    ["Code signing", "Document signing", "Authentication"]
))

policy.add_algorithm(ApprovedAlgorithm(
    "RSA", "KEM/Signature", "FIPS 186-5",
    ["RSA-3072", "RSA-4096"],
    AlgorithmApproval.DEPRECATED,
    ["Legacy systems only"],
    sunset_date=date(2030, 12, 31)
))

policy.add_algorithm(ApprovedAlgorithm(
    "ECDSA", "Signature", "FIPS 186-5",
    ["P-384", "P-521"],
    AlgorithmApproval.TRANSITIONAL,
    ["Hybrid with ML-DSA"],
    sunset_date=date(2030, 12, 31)
))

policy.key_management_requirements = [
    "All keys must be generated using approved DRBG",
    "Key lifetimes: Signing keys max 3 years, KEM keys max 2 years",
    "Key storage: HSM required for root and intermediate CA keys",
    "Key destruction: Secure zeroization with verification",
]

policy.compliance_requirements = [
    "FIPS 140-3 validated modules for federal systems",
    "CNSA 2.0 compliance by designated deadlines",
    "Annual cryptographic inventory update",
]

print(policy.generate_document())
```

```python
from dataclasses import dataclass
from datetime import datetime
from typing import Dict, Any, List, Optional
import json
from enum import Enum

class AuditEventType(Enum):
    KEY_GENERATION = "KEY_GENERATION"
    KEY_ROTATION = "KEY_ROTATION"
    KEY_DESTRUCTION = "KEY_DESTRUCTION"
    KEY_ACCESS = "KEY_ACCESS"
    CONFIG_CHANGE = "CONFIG_CHANGE"
    AUTH_FAILURE = "AUTH_FAILURE"
    CRYPTO_ERROR = "CRYPTO_ERROR"
    SELF_TEST = "SELF_TEST"
    POLICY_UPDATE = "POLICY_UPDATE"

@dataclass
class AuditEntry:
    """Single audit log entry."""
    timestamp: datetime
    event_type: AuditEventType
    module_id: str
    user: str
    result: str  # SUCCESS, FAILURE, WARNING
    details: Dict[str, Any]
    algorithm: Optional[str] = None
    key_id: Optional[str] = None

    def to_json(self) -> str:
        return json.dumps({
            "timestamp": self.timestamp.isoformat(),
            "event_type": self.event_type.value,
            "module_id": self.module_id,
            "algorithm": self.algorithm,
            "key_id": self.key_id,
            "user": self.user,
            "result": self.result,
            "details": self.details
        }, indent=2)

class CryptoAuditLog:
    """
    Cryptographic audit logging system.
    """

    def __init__(self, module_id: str):
        self.module_id = module_id
        self.entries: List[AuditEntry] = []

    def log(
        self,
        event_type: AuditEventType,
        user: str,
        result: str,
        details: Dict[str, Any],
        algorithm: str = None,
        key_id: str = None
    ) -> AuditEntry:
        """Log an audit event."""
        entry = AuditEntry(
            timestamp=datetime.now(),
            event_type=event_type,
            module_id=self.module_id,
            user=user,
            result=result,
            details=details,
            algorithm=algorithm,
            key_id=key_id
        )
        self.entries.append(entry)
        return entry

    def query(
        self,
        event_type: AuditEventType = None,
        start_time: datetime = None,
        end_time: datetime = None,
        key_id: str = None
    ) -> List[AuditEntry]:
        """Query audit log."""
        results = self.entries

        if event_type:
            results = [e for e in results if e.event_type == event_type]
        if start_time:
            results = [e for e in results if e.timestamp >= start_time]
        if end_time:
            results = [e for e in results if e.timestamp <= end_time]
        if key_id:
            results = [e for e in results if e.key_id == key_id]

        return results

    def generate_audit_report(
        self,
        start_date: datetime,
        end_date: datetime
    ) -> str:
        """Generate audit report for period."""
        entries = self.query(start_time=start_date, end_time=end_date)

        lines = [
            "CRYPTOGRAPHIC AUDIT REPORT",
            f"Period: {start_date.date()} to {end_date.date()}",
            f"Module: {self.module_id}",
            "=" * 60,
            "",
            f"Total Events: {len(entries)}",
            "",
            "Events by Type:",
        ]

        # Count by type
        by_type: Dict[AuditEventType, int] = {}
        for e in entries:
            by_type[e.event_type] = by_type.get(e.event_type, 0) + 1

        for event_type, count in sorted(
            by_type.items(), key=lambda x: -x[1]
        ):
            lines.append(f"  {event_type.value}: {count}")

        # Failures
        failures = [e for e in entries if e.result == "FAILURE"]
        if failures:
            lines.extend([
                "",
                f"FAILURES ({len(failures)}):",
            ])
            for f in failures[:10]:
                lines.append(
                    f"  {f.timestamp} - {f.event_type.value}: "
                    f"{f.details.get('error', 'Unknown')}"
                )

        return "\n".join(lines)

# Example usage
audit = CryptoAuditLog("pqc-module-001")

audit.log(
    AuditEventType.KEY_GENERATION,
    "admin@example.com",
    "SUCCESS",
    {"purpose": "TLS", "validity_days": 365},
    algorithm="ML-KEM-768",
    key_id="kek-001"
)

audit.log(
    AuditEventType.SELF_TEST,
    "system",
    "SUCCESS",
    {"test_suite": "FIPS 203 KAT", "duration_ms": 150}
)

audit.log(
    AuditEventType.AUTH_FAILURE,
    "unknown@example.com",
    "FAILURE",
    {"error": "Invalid credentials", "source_ip": "192.168.1.100"}
)

print(audit.generate_audit_report(
    datetime(2024, 1, 1),
    datetime.now()
))
```

## Summary

Effective audit and documentation for PQC requires:

1. **Policy documents**: Clear cryptographic policies with approved algorithms
2. **Technical documentation**: Architecture, procedures, and specifications
3. **Audit trails**: Comprehensive logging of all key management events
4. **Evidence collection**: Organized evidence for compliance audits

Preparation before audits ensures smooth compliance verification.
