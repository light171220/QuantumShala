    def get_applicable_mandates(
        self,
        entity_type: str,
        as_of: date = None
    ) -> List[RegulatoryMandate]:
        """Get mandates applicable to an entity type."""
        as_of = as_of or date.today()
        applicable = []

        for mandate in self.mandates:
            for entity in mandate.affected_entities:
                if entity_type.lower() in entity.lower():
                    if (mandate.effective_date is None or
                        mandate.effective_date <= as_of):
                        applicable.append(mandate)
                    break

        return applicable

    def generate_compliance_timeline(self) -> str:
        """Generate timeline of compliance deadlines."""
        lines = [
            "PQC COMPLIANCE TIMELINE",
            "=" * 60
        ]

        # Sort by effective date
        sorted_mandates = sorted(
            self.mandates,
            key=lambda m: m.effective_date or date.max
        )

        for mandate in sorted_mandates:
            date_str = (
                mandate.effective_date.strftime("%Y-%m-%d")
                if mandate.effective_date else "TBD"
            )
            lines.append(f"\n{date_str}: {mandate.name}")
            lines.append(f"  Status: {mandate.status.value}")
            lines.append(f"  Issuer: {mandate.issuing_body}")
            for req in mandate.key_requirements[:3]:
                lines.append(f"  • {req}")

        return "\n".join(lines)

## Compliance Planning

```python
from dataclasses import dataclass, field
from datetime import date, timedelta
from typing import List, Dict, Optional
from enum import Enum

class ComplianceStatus(Enum):
    NOT_STARTED = "Not Started"
    PLANNING = "Planning"
    IN_PROGRESS = "In Progress"
    COMPLETED = "Completed"
    VERIFIED = "Verified"

@dataclass
class ComplianceItem:
    """Individual compliance requirement."""
    requirement: str
    regulation: str
    deadline: date
    status: ComplianceStatus = ComplianceStatus.NOT_STARTED
    owner: Optional[str] = None
    notes: str = ""

@dataclass
class CompliancePlan:
    """Organizational compliance plan."""
    organization: str
    industries: List[str]
    items: List[ComplianceItem] = field(default_factory=list)

    def add_item(self, item: ComplianceItem):
        self.items.append(item)

    def get_overdue(self) -> List[ComplianceItem]:
        """Get overdue items."""
        today = date.today()
        return [
            i for i in self.items
            if i.deadline < today
            and i.status not in (ComplianceStatus.COMPLETED,
                                  ComplianceStatus.VERIFIED)
        ]

    def get_upcoming(self, days: int = 90) -> List[ComplianceItem]:
        """Get items due within specified days."""
        today = date.today()
        cutoff = today + timedelta(days=days)
        return [
            i for i in self.items
            if today <= i.deadline <= cutoff
            and i.status not in (ComplianceStatus.COMPLETED,
                                  ComplianceStatus.VERIFIED)
        ]

    def calculate_progress(self) -> Dict[str, float]:
        """Calculate compliance progress."""
        if not self.items:
            return {"overall": 0.0}

        by_regulation: Dict[str, List[ComplianceItem]] = {}
        for item in self.items:
            by_regulation.setdefault(item.regulation, []).append(item)

        progress = {}
        total_complete = 0
        total_items = len(self.items)

        for reg, items in by_regulation.items():
            completed = sum(
                1 for i in items
                if i.status in (ComplianceStatus.COMPLETED,
                                ComplianceStatus.VERIFIED)
            )
            progress[reg] = (completed / len(items)) * 100
            total_complete += completed

        progress["overall"] = (total_complete / total_items) * 100
        return progress

    def generate_report(self) -> str:
        """Generate compliance status report."""
        lines = [
            f"COMPLIANCE STATUS REPORT: {self.organization}",
            f"Date: {date.today().isoformat()}",
            f"Industries: {', '.join(self.industries)}",
            "=" * 60,
        ]

        # Progress summary
        progress = self.calculate_progress()
        lines.append("\nPROGRESS BY REGULATION:")
        for reg, pct in progress.items():
            bar = "█" * int(pct / 5) + "░" * (20 - int(pct / 5))
            lines.append(f"  {reg:30} [{bar}] {pct:.1f}%")

        # Overdue items
        overdue = self.get_overdue()
        if overdue:
            lines.append(f"\nOVERDUE ITEMS ({len(overdue)}):")
            for item in overdue:
                lines.append(
                    f"  ⚠ {item.requirement[:50]}... "
                    f"(Due: {item.deadline})"
                )

        # Upcoming items
        upcoming = self.get_upcoming()
        if upcoming:
            lines.append(f"\nUPCOMING (90 days) ({len(upcoming)}):")
            for item in upcoming[:5]:
                lines.append(
                    f"  • {item.requirement[:50]}... "
                    f"(Due: {item.deadline})"
                )

        return "\n".join(lines)

# Example compliance plan
plan = CompliancePlan(
    organization="Example Federal Contractor",
    industries=["Defense", "Federal Government"]
)

plan.add_item(ComplianceItem(
    "Complete cryptographic inventory",
    "OMB M-23-02",
    date(2023, 5, 4),
    ComplianceStatus.COMPLETED,
    "CISO Office"
))

plan.add_item(ComplianceItem(
    "Implement ML-KEM in web servers",
    "CNSA 2.0",
    date(2025, 12, 31),
    ComplianceStatus.IN_PROGRESS,
    "DevSecOps"
))

plan.add_item(ComplianceItem(
    "Validate FIPS 203/204 modules",
    "FIPS Requirements",
    date(2025, 6, 30),
    ComplianceStatus.PLANNING,
    "Security Engineering"
))

print(plan.generate_report())
```

## Summary

The regulatory landscape for PQC is rapidly evolving:

1. **US Government**: CNSA 2.0, OMB directives, and FIPS standards set clear timelines
2. **Industry**: Financial, healthcare, and critical infrastructure have emerging requirements
3. **International**: General alignment on NIST algorithms with regional variations
4. **Planning**: Organizations must assess applicability and plan transitions

Compliance deadlines are approaching—early planning is essential for meeting requirements.
