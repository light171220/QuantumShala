# PQC Certificate Formats

### OID Encoding

```python
"""
PQC Algorithm OID definitions and encoding
"""

from dataclasses import dataclass
from typing import Tuple

@dataclass
class AlgorithmOID:
    """Algorithm identifier with OID."""
    name: str
    oid: Tuple[int, ...]
    oid_string: str
    key_size: int
    sig_size: int

# NIST PQC OIDs (using NIST arc: 2.16.840.1.101.3.4.*)
PQC_ALGORITHMS = {
    # ML-KEM
    "ML-KEM-512": AlgorithmOID(
        name="ML-KEM-512",
        oid=(2, 16, 840, 1, 101, 3, 4, 4, 1),
        oid_string="2.16.840.1.101.3.4.4.1",
        key_size=800,
        sig_size=0
    ),
    "ML-KEM-768": AlgorithmOID(
        name="ML-KEM-768",
        oid=(2, 16, 840, 1, 101, 3, 4, 4, 2),
        oid_string="2.16.840.1.101.3.4.4.2",
        key_size=1184,
        sig_size=0
    ),
    "ML-KEM-1024": AlgorithmOID(
        name="ML-KEM-1024",
        oid=(2, 16, 840, 1, 101, 3, 4, 4, 3),
        oid_string="2.16.840.1.101.3.4.4.3",
        key_size=1568,
        sig_size=0
    ),

    # ML-DSA
    "ML-DSA-44": AlgorithmOID(
        name="ML-DSA-44",
        oid=(2, 16, 840, 1, 101, 3, 4, 3, 17),
        oid_string="2.16.840.1.101.3.4.3.17",
        key_size=1312,
        sig_size=2420
    ),
    "ML-DSA-65": AlgorithmOID(
        name="ML-DSA-65",
        oid=(2, 16, 840, 1, 101, 3, 4, 3, 18),
        oid_string="2.16.840.1.101.3.4.3.18",
        key_size=1952,
        sig_size=3293
    ),
    "ML-DSA-87": AlgorithmOID(
        name="ML-DSA-87",
        oid=(2, 16, 840, 1, 101, 3, 4, 3, 19),
        oid_string="2.16.840.1.101.3.4.3.19",
        key_size=2592,
        sig_size=4595
    ),
}

def encode_oid(oid: Tuple[int, ...]) -> bytes:
    """
    Encode OID to DER format.
    """
    # First two components encoded as 40*first + second
    result = bytes([40 * oid[0] + oid[1]])

    # Remaining components use base-128 encoding
    for component in oid[2:]:
        if component == 0:
            result += b'\x00'
        else:
            encoded = []
            temp = component
            while temp > 0:
                encoded.insert(0, temp & 0x7F)
                temp >>= 7
            # Set high bit on all but last byte
            for i in range(len(encoded) - 1):
                encoded[i] |= 0x80
            result += bytes(encoded)

    return result

# Display OIDs
print("PQC Algorithm OIDs")
print("=" * 60)

for name, algo in PQC_ALGORITHMS.items():
    encoded = encode_oid(algo.oid)
    print(f"\n{name}:")
    print(f"  OID: {algo.oid_string}")
    print(f"  DER: {encoded.hex()}")
    print(f"  Key: {algo.key_size} bytes, Sig: {algo.sig_size} bytes")
```

### SPKI Encoding Implementation

```python
"""
SubjectPublicKeyInfo encoding for PQC
"""

def encode_length(length: int) -> bytes:
    """Encode ASN.1 length."""
    if length < 128:
        return bytes([length])
    elif length < 256:
        return bytes([0x81, length])
    elif length < 65536:
        return bytes([0x82, length >> 8, length & 0xFF])
    else:
        return bytes([0x83, length >> 16, (length >> 8) & 0xFF, length & 0xFF])

def encode_spki(algorithm_oid: bytes, public_key: bytes) -> bytes:
    """
    Encode SubjectPublicKeyInfo for PQC algorithm.

    Args:
        algorithm_oid: DER-encoded OID
        public_key: Raw public key bytes

    Returns:
        DER-encoded SPKI
    """
    # AlgorithmIdentifier: SEQUENCE { OID }
    # (no parameters for PQC algorithms)
    oid_with_tag = bytes([0x06]) + encode_length(len(algorithm_oid)) + algorithm_oid
    algo_id = bytes([0x30]) + encode_length(len(oid_with_tag)) + oid_with_tag

    # BIT STRING for public key
    # First byte is unused bits count (0 for PQC)
    bit_string_content = bytes([0x00]) + public_key
    bit_string = bytes([0x03]) + encode_length(len(bit_string_content)) + bit_string_content

    # Outer SEQUENCE
    content = algo_id + bit_string
    spki = bytes([0x30]) + encode_length(len(content)) + content

    return spki

# Example: Encode ML-DSA-65 public key
ml_dsa_65_oid = encode_oid(PQC_ALGORITHMS["ML-DSA-65"].oid)
mock_pk = bytes(1952)  # Placeholder public key

spki = encode_spki(ml_dsa_65_oid, mock_pk)
print("SubjectPublicKeyInfo Encoding")
print("=" * 50)
print(f"Algorithm OID: {ml_dsa_65_oid.hex()}")
print(f"Public Key: {len(mock_pk)} bytes")
print(f"SPKI Total: {len(spki)} bytes")
print(f"SPKI Header: {spki[:20].hex()}...")
```

## Composite Certificates

### Hybrid Key Representation

```
┌─────────────────────────────────────────────────────────────────┐
│              Composite Key Structures                            │
├─────────────────────────────────────────────────────────────────┤
│                                                                  │
│  APPROACH 1: Composite OID                                      │
│  ┌────────────────────────────────────────────────────────────┐│
│  │  AlgorithmIdentifier: id-composite-key                      ││
│  │  SubjectPublicKey: SEQUENCE {                               ││
│  │    classicalKey SubjectPublicKeyInfo, -- ECDSA              ││
│  │    pqcKey       SubjectPublicKeyInfo  -- ML-DSA             ││
│  │  }                                                          ││
│  └────────────────────────────────────────────────────────────┘│
│  ✓ Single certificate contains both keys                       │
│  ✓ Atomic - can't separate keys                                │
│  ⚠ Requires updated parsers                                    │
│                                                                  │
│  APPROACH 2: Subject Alt Name Extension                         │
│  ┌────────────────────────────────────────────────────────────┐│
│  │  Primary key: ECDSA                                         ││
│  │  Extension: subjectAltPublicKeyInfo (ML-DSA key)           ││
│  └────────────────────────────────────────────────────────────┘│
│  ✓ Backward compatible (legacy sees ECDSA)                     │
│  ⚠ Extension may be ignored by older parsers                   │
│                                                                  │
│  APPROACH 3: Dual Certificates                                  │
│  ┌────────────────────────────────────────────────────────────┐│
│  │  Certificate 1: ECDSA key and signature                    ││
│  │  Certificate 2: ML-DSA key and signature                   ││
│  │  Both for same subject, served based on client support     ││
│  └────────────────────────────────────────────────────────────┘│
│  ✓ Simple, standard format                                     │
│  ⚠ Management overhead (two certs per entity)                 │
└─────────────────────────────────────────────────────────────────┘
```

### Composite Certificate Implementation

```python
"""
Composite certificate structures
"""

from dataclasses import dataclass
from typing import Optional, List

@dataclass
class CompositePublicKey:
    """Composite public key containing classical and PQC keys."""
    classical_algorithm: str
    classical_key: bytes
    pqc_algorithm: str
    pqc_key: bytes

@dataclass
class CompositeSignature:
    """Composite signature containing both signatures."""
    classical_signature: bytes
    pqc_signature: bytes

@dataclass
class HybridCertificate:
    """Certificate with composite keys and signatures."""
    version: int
    serial_number: bytes
    issuer: str
    subject: str
    validity_not_before: str
    validity_not_after: str
    public_key: CompositePublicKey
    extensions: dict
    signature_algorithm: str
    signature: CompositeSignature

    def get_size_breakdown(self) -> dict:
        """Calculate size breakdown of certificate."""
        return {
            "classical_key": len(self.public_key.classical_key),
            "pqc_key": len(self.public_key.pqc_key),
            "total_key": len(self.public_key.classical_key) + len(self.public_key.pqc_key),
            "classical_sig": len(self.signature.classical_signature),
            "pqc_sig": len(self.signature.pqc_signature),
            "total_sig": len(self.signature.classical_signature) + len(self.signature.pqc_signature),
        }

# Example certificate sizes
print("Hybrid Certificate Size Analysis")
print("=" * 50)

# Simulated sizes
classical_key = bytes(64)   # ECDSA P-256
pqc_key = bytes(1952)       # ML-DSA-65
classical_sig = bytes(64)   # ECDSA
pqc_sig = bytes(3293)       # ML-DSA-65

cert = HybridCertificate(
    version=3,
    serial_number=bytes(16),
    issuer="CN=Hybrid CA",
    subject="CN=example.com",
    validity_not_before="2024-01-01",
    validity_not_after="2025-01-01",
    public_key=CompositePublicKey(
        classical_algorithm="ECDSA-P256",
        classical_key=classical_key,
        pqc_algorithm="ML-DSA-65",
        pqc_key=pqc_key
    ),
    extensions={},
    signature_algorithm="id-composite-sig",
    signature=CompositeSignature(classical_sig, pqc_sig)
)

sizes = cert.get_size_breakdown()
print(f"Public Keys:")
print(f"  Classical (ECDSA): {sizes['classical_key']} bytes")
print(f"  PQC (ML-DSA-65):   {sizes['pqc_key']} bytes")
print(f"  Total:             {sizes['total_key']} bytes")
print(f"\nSignatures:")
print(f"  Classical:         {sizes['classical_sig']} bytes")
print(f"  PQC:               {sizes['pqc_sig']} bytes")
print(f"  Total:             {sizes['total_sig']} bytes")
print(f"\nCertificate Total: ~{sizes['total_key'] + sizes['total_sig'] + 200} bytes")
```

## Summary

PQC certificate formats require:

1. **New OIDs**: Standardized identifiers for each algorithm
2. **SPKI Encoding**: Standard ASN.1 structure, larger key data
3. **Composite Options**: Multiple approaches for hybrid certificates
4. **Size Planning**: Chains are significantly larger
5. **Tool Updates**: Parsers and validators need updates

Certificate infrastructure updates are essential for PQC deployment.
