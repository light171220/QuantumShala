# Infrastructure Preparation for PQC

Deploying PQC requires updating multiple infrastructure components. This lesson covers the practical aspects of preparing your infrastructure.

```python
from dataclasses import dataclass
from typing import List, Dict, Optional
import subprocess
import os

@dataclass
class ServerComponent:
    """Server component to check/update."""
    name: str
    check_command: str
    min_version: str
    current_version: Optional[str] = None
    pqc_ready: bool = False

class ServerPreparation:
    """
    Server preparation utilities for PQC deployment.
    """

    def __init__(self):
        self.components: List[ServerComponent] = []
        self._define_components()

    def _define_components(self):
        """Define components to check."""
        self.components = [
            ServerComponent(
                "OpenSSL",
                "openssl version",
                "3.0.0"
            ),
            ServerComponent(
                "Linux Kernel",
                "uname -r",
                "5.15"
            ),
            ServerComponent(
                "liboqs",
                "pkg-config --modversion liboqs",
                "0.9.0"
            ),
        ]

    def check_openssl_pqc_support(self) -> Dict:
        """Check OpenSSL for PQC algorithm support."""
        # In practice, run actual commands
        supported_kems = [
            "mlkem768",
            "mlkem1024",
            "x25519_mlkem768",
            "p384_mlkem768",
        ]

        supported_sigs = [
            "mldsa65",
            "mldsa87",
            "sphincssha2128fsimple",
        ]

        return {
            "kems": supported_kems,
            "signatures": supported_sigs,
            "hybrid_supported": True
        }

    def generate_openssl_config(self) -> str:
        """Generate OpenSSL configuration for PQC."""
        config = """
# OpenSSL configuration for PQC
# Place in openssl.cnf or include via SSL_CONF

[ssl_sect]
system_default = system_default_sect

[system_default_sect]
MinProtocol = TLSv1.3
Ciphersuites = TLS_AES_256_GCM_SHA384:TLS_CHACHA20_POLY1305_SHA256

# PQC key exchange groups (hybrid first, then pure PQC)
Groups = x25519_mlkem768:p384_mlkem768:mlkem768:mlkem1024:x25519:p-384

# PQC signature algorithms
SignatureAlgorithms = mldsa65:mldsa87:ecdsa_secp384r1_sha384

[oqs_provider_sect]
# Enable OQS provider for post-quantum algorithms
activate = 1
module = /usr/lib/ossl-modules/oqsprovider.so
"""
        return config

    def generate_nginx_config(self) -> str:
        """Generate Nginx TLS configuration for PQC."""
        config = """
# Nginx TLS configuration for PQC

server {
    listen 443 ssl http2;
    server_name example.com;

    # Certificates (hybrid recommended during transition)
    ssl_certificate /etc/nginx/certs/server-hybrid.crt;
    ssl_certificate_key /etc/nginx/certs/server-hybrid.key;

    # Alternative pure PQC certificate
    ssl_certificate /etc/nginx/certs/server-mldsa.crt;
    ssl_certificate_key /etc/nginx/certs/server-mldsa.key;

    # Protocol and ciphers
    ssl_protocols TLSv1.3;
    ssl_prefer_server_ciphers on;
    ssl_ciphers TLS_AES_256_GCM_SHA384:TLS_CHACHA20_POLY1305_SHA256;

    # Key exchange groups (PQC hybrid preferred)
    ssl_ecdh_curve x25519_mlkem768:p384_mlkem768:mlkem768:x25519:p-384;

    # Session configuration
    ssl_session_timeout 1d;
    ssl_session_cache shared:SSL:50m;
    ssl_session_tickets off;

    # OCSP stapling
    ssl_stapling on;
    ssl_stapling_verify on;

    # Additional security headers
    add_header Strict-Transport-Security "max-age=31536000" always;
}
"""
        return config

    def generate_checklist(self) -> str:
        """Generate server preparation checklist."""
        lines = [
            "SERVER PREPARATION CHECKLIST",
            "=" * 50,
            "",
            "1. OPERATING SYSTEM:",
            "   □ Verify OS version supports PQC libraries",
            "   □ Install security updates",
            "   □ Verify entropy sources (rng-tools if needed)",
            "",
            "2. CRYPTOGRAPHIC LIBRARIES:",
            "   □ Install OpenSSL 3.x with OQS provider",
            "   □ Verify FIPS validation status",
            "   □ Test basic PQC operations",
            "",
            "3. TLS CONFIGURATION:",
            "   □ Configure hybrid key exchange groups",
            "   □ Set signature algorithm preferences",
            "   □ Install PQC certificates",
            "",
            "4. RESOURCE ALLOCATION:",
            "   □ Increase memory allocation",
            "   □ Verify CPU capacity for crypto load",
            "   □ Expand network buffers",
            "",
            "5. TESTING:",
            "   □ Test TLS handshake with PQC",
            "   □ Verify fallback to classical works",
            "   □ Load test for performance",
        ]
        return "\n".join(lines)
```

## Network Component Updates

```python
from dataclasses import dataclass
from typing import List

@dataclass
class NetworkComponent:
    """Network component requiring PQC update."""
    name: str
    type: str
    pqc_considerations: List[str]
    configuration_notes: str

class NetworkPreparation:
    """
    Network infrastructure preparation for PQC.
    """

    def __init__(self):
        self.components = self._define_components()

    def _define_components(self) -> List[NetworkComponent]:
        return [
            NetworkComponent(
                "Load Balancer",
                "TLS Termination",
                [
                    "Must support PQC TLS key exchange",
                    "Larger handshake messages require buffer increase",
                    "Consider separate listener for PQC",
                    "Monitor TLS handshake latency",
                ],
                "Configure hybrid groups; increase connection timeout; expand buffers"
            ),
            NetworkComponent(
                "Web Application Firewall",
                "TLS Inspection",
                [
                    "May need firmware update for PQC",
                    "Larger certs may exceed inspection limits",
                    "Consider bypass for PQC until supported",
                ],
                "Check vendor PQC roadmap; test inspection with PQC traffic"
            ),
            NetworkComponent(
                "VPN Concentrator",
                "IPsec/IKEv2",
                [
                    "IKEv2 ML-KEM support required (RFC 9370)",
                    "Tunnel setup time may increase",
                    "Client software needs update",
                ],
                "Upgrade firmware; configure ML-KEM key exchange"
            ),
            NetworkComponent(
                "DNS/DoH/DoT",
                "Secure DNS",
                [
                    "DNSSEC may use PQC in future",
                    "DoT/DoH TLS needs PQC support",
                    "Resolver query performance impact",
                ],
                "Monitor IETF DPRIVE for PQC; update TLS stack"
            ),
            NetworkComponent(
                "Reverse Proxy",
                "TLS Termination/Re-encryption",
                [
                    "Must handle larger cert chains",
                    "Backend connections may need PQC",
                    "Session caching considerations",
                ],
                "Configure PQC for frontend; plan backend migration"
            ),
        ]

    def generate_report(self) -> str:
        """Generate network preparation report."""
        lines = [
            "NETWORK INFRASTRUCTURE PQC PREPARATION",
            "=" * 50,
        ]

        for comp in self.components:
            lines.extend([
                f"\n{comp.name} ({comp.type})",
                "Considerations:",
            ])
            for consideration in comp.pqc_considerations:
                lines.append(f"  • {consideration}")
            lines.append(f"Configuration: {comp.configuration_notes}")

        return "\n".join(lines)

network_prep = NetworkPreparation()
print(network_prep.generate_report())
```

## Summary

Infrastructure preparation for PQC involves:

1. **Server updates**: OS, TLS libraries, crypto configurations
2. **PKI transition**: Hybrid certificates, new CA hierarchy
3. **Network components**: Load balancers, WAFs, VPNs, proxies
4. **Validation**: Testing handshakes, performance, interoperability

Plan for larger resource requirements and extended timeline for infrastructure changes.
