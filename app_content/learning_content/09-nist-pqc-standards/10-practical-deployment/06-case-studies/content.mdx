```python
from dataclasses import dataclass

    """

    def __init__(self):

                    {"name": "Application", "months": 8, "effort": "35%"},
                ],

                    "systems_migrated": "650",
                    "certificates_updated": "3,200",
                    "handshake_success": "99.87%",
                    "latency_increase": "+22%",
                }
            ),
        ]

    def compare_timelines(self) -> str:
        """Compare deployment timelines across case studies."""
        lines = [
            "DEPLOYMENT TIMELINE COMPARISON",
            "=" * 50,
        ]

        for study in self.case_studies:
            lines.append(f"\n{study.name}:")
            lines.append(f"  Total Duration: {study.duration_months} months")
            lines.append("  Phases:")
            for phase in study.phases:
                bars = "█" * int(phase['months'] / 2)
                lines.append(
                    f"    {phase['name']:<15} {bars} {phase['months']}mo ({phase['effort']})"
                )

        return "\n".join(lines)

    def extract_common_challenges(self) -> List[str]:
        """Extract challenges common across case studies."""
        challenge_counts: Dict[str, int] = {}
        keywords = [
            "legacy", "vendor", "compliance", "timeline", "budget",
            "training", "third-party", "documentation"
        ]

        for study in self.case_studies:
            for challenge in study.challenges:
                for keyword in keywords:
                    if keyword in challenge.lower():
                        challenge_counts[keyword] = challenge_counts.get(keyword, 0) + 1

        # Return most common
        return [k for k, v in sorted(challenge_counts.items(), key=lambda x: -x[1])]

    def generate_summary_report(self) -> str:
        """Generate summary report from case studies."""
        lines = [
            "PQC DEPLOYMENT CASE STUDY SUMMARY",
            "=" * 50,
            "",

        for study in self.case_studies:
            lines.extend([
                f"■ {study.name}",
                f"  Industry: {study.industry}",
                f"  Size: {study.size}",
                f"  Duration: {study.duration_months} months",
                f"  Budget: {study.budget}",
                "",
                "  Key Metrics:",
            ])
            for metric, value in study.key_metrics.items():
                lines.append(f"    • {metric.replace('_', ' ').title()}: {value}")

            lines.append("\n  Top Lessons Learned:")
            for lesson in study.lessons_learned[:3]:
                lines.append(f"    • {lesson}")
            lines.append("")

        return "\n".join(lines)

# Example
analyzer = CaseStudyAnalyzer()
print(analyzer.generate_summary_report())
```

```python
from dataclasses import dataclass
from typing import List, Dict

@dataclass
class Optimization:
    """Performance optimization details."""
    name: str
    implementation: str
    impact: str
    effort: str

class CloudProviderCaseStudy:
    """
    Detailed cloud provider case study analysis.
    """

    def __init__(self):
        self.optimizations: List[Optimization] = [
            Optimization(
                "AVX-512 Vectorization",
                "Custom ML-KEM implementation using AVX-512",
                "40% reduction in CPU for crypto ops",
                "High (custom code development)"
            ),
            Optimization(
                "Connection Pooling",
                "Reuse TLS connections across requests",
                "60% reduction in handshake overhead",
                "Medium (infrastructure change)"
            ),
            Optimization(
                "Session Resumption",
                "TLS session tickets with PQC-derived keys",
                "85% of connections avoid full handshake",
                "Low (configuration change)"
            ),
            Optimization(
                "Key Pre-computation",
                "Generate ephemeral keys during idle periods",
                "30% faster key generation under load",
                "Medium (code change)"
            ),
            Optimization(
                "Certificate Caching",
                "Cache parsed certificates in memory",
                "50% reduction in parsing overhead",
                "Low (code change)"
            ),
        ]

        self.rollout_stages = [
            {"stage": "Internal dogfood", "traffic": "0.1%", "duration": "2 weeks"},
            {"stage": "Beta customers", "traffic": "1%", "duration": "4 weeks"},
            {"stage": "Early adopters", "traffic": "10%", "duration": "4 weeks"},
            {"stage": "General availability", "traffic": "50%", "duration": "6 weeks"},
            {"stage": "Default on", "traffic": "100%", "duration": "Ongoing"},
        ]

    def calculate_roi(self) -> Dict[str, any]:
        """Calculate rough ROI metrics."""
        return {
            "development_cost": "~$5M",
            "infrastructure_cost": "+8% compute",
            "security_benefit": "Quantum-resistant for 10+ years",
            "customer_retention": "Increased (competitive advantage)",
            "compliance_benefit": "Ready for federal contracts",
        }

    def generate_rollout_plan(self) -> str:
        """Generate rollout plan visualization."""
        lines = [
            "GRADUAL ROLLOUT PLAN",
            "=" * 50,
        ]

        cumulative_traffic = 0
        for stage in self.rollout_stages:
            traffic_pct = float(stage["traffic"].replace("%", ""))
            if traffic_pct > cumulative_traffic:
                cumulative_traffic = traffic_pct
            bar_length = int(cumulative_traffic / 5)
            bar = "█" * bar_length + "░" * (20 - bar_length)
            lines.append(
                f"  {stage['stage']:<20} [{bar}] {stage['traffic']:>5}"
            )
            lines.append(f"    Duration: {stage['duration']}")

        return "\n".join(lines)

    def generate_optimization_report(self) -> str:
        """Generate optimization recommendations report."""
        lines = [
            "PERFORMANCE OPTIMIZATION RECOMMENDATIONS",
            "=" * 50,
            "",
            "Based on cloud provider case study:",
            "",
        ]

        for opt in self.optimizations:
            lines.extend([
                f"■ {opt.name}",
                f"  How: {opt.implementation}",
                f"  Impact: {opt.impact}",
                f"  Effort: {opt.effort}",
                "",
            ])

        return "\n".join(lines)
```

```python
from dataclasses import dataclass
from typing import List, Dict

@dataclass
class Lesson:
    """Lesson learned from case studies."""
    category: str
    lesson: str
    recommendation: str
    priority: str  # high, medium, low

class LessonsLearnedAnalyzer:
    """
    Analyze and apply lessons learned from case studies.
    """

    def __init__(self):
        self.lessons: List[Lesson] = self._compile_lessons()

    def _compile_lessons(self) -> List[Lesson]:
        return [
            # Timeline lessons
            Lesson(
                "Timeline", "Crypto inventory takes 2-3x longer than estimated",
                "Plan 6-12 months for comprehensive inventory", "high"
            ),
            Lesson(
                "Timeline", "Vendor PQC support timelines frequently slip",
                "Add 6+ month buffer for vendor dependencies", "high"
            ),
            Lesson(
                "Timeline", "Testing requires more time than development",
                "Allocate 40% of effort to testing phases", "high"
            ),

            # Technical lessons
            Lesson(
                "Technical", "Hybrid approach is essential for safe migration",
                "Never deploy pure PQC without hybrid transition", "high"
            ),
            Lesson(
                "Technical", "Certificate size is the biggest operational issue",
                "Use ML-DSA over SLH-DSA where possible", "high"
            ),
            Lesson(
                "Technical", "Performance impact is manageable with optimization",
                "Plan for 10-25% latency increase, optimize iteratively", "medium"
            ),

            # Organizational lessons
            Lesson(
                "Organizational", "Executive sponsorship is critical for resources",
                "Get CISO/CTO commitment before starting", "high"
            ),
            Lesson(
                "Organizational", "Dedicated team is more effective than part-time",
                "Assign 5-10 FTEs for enterprise deployments", "high"
            ),
            Lesson(
                "Organizational", "Documentation is required for compliance",
                "Document all decisions and changes from day one", "medium"
            ),
        ]

    def get_by_priority(self, priority: str) -> List[Lesson]:
        """Get lessons by priority level."""
        return [l for l in self.lessons if l.priority == priority]

    def get_by_category(self, category: str) -> List[Lesson]:
        """Get lessons by category."""
        return [l for l in self.lessons if l.category == category]

    def generate_checklist(self) -> str:
        """Generate pre-deployment checklist from lessons."""
        lines = [
            "PRE-DEPLOYMENT CHECKLIST (From Case Studies)",
            "=" * 50,
            "",
        ]

        for category in ["Timeline", "Technical", "Organizational"]:
            lines.append(f"\n{category.upper()}:")
            for lesson in self.get_by_category(category):
                priority_marker = "(!)" if lesson.priority == "high" else "( )"
                lines.append(f"  □ {priority_marker} {lesson.recommendation}")

        return "\n".join(lines)

    def apply_to_organization(
        self,
        org_size: str,
        industry: str,
        timeline_months: int
    ) -> str:
        """Apply lessons to specific organization context."""
        lines = [
            "CUSTOMIZED RECOMMENDATIONS",
            "=" * 50,
            f"Organization: {org_size} in {industry}",
            f"Timeline: {timeline_months} months",
            "",
        ]

        # Adjust recommendations based on context
        if timeline_months < 24:
            lines.append("⚠ WARNING: Timeline under 24 months is aggressive")
            lines.append("  Consider: Phased approach with pilot first")
            lines.append("")

        if "government" in industry.lower():
            lines.append("GOVERNMENT-SPECIFIC:")
            lines.append("  • FIPS 140-3 validation required")
            lines.append("  • Follow OMB M-23-02 guidelines")
            lines.append("  • Coordinate with CISA")
            lines.append("")

        if "financial" in industry.lower():
            lines.append("FINANCIAL-SPECIFIC:")
            lines.append("  • PCI-DSS documentation needed")
            lines.append("  • Third-party vendor coordination critical")
            lines.append("  • HSM PQC support essential")
            lines.append("")

        lines.append("HIGH-PRIORITY LESSONS FOR YOUR CONTEXT:")
        for lesson in self.get_by_priority("high"):
            lines.append(f"  • {lesson.recommendation}")

        return "\n".join(lines)

# Example
analyzer = LessonsLearnedAnalyzer()
```

## Summary

Key takeaways from case studies:

1. **Start early**: Crypto inventory and planning take longer than expected
2. **Use hybrid**: Hybrid cryptography enables safe, gradual migration
3. **Engage vendors**: Third-party dependencies are often the bottleneck
4. **Get sponsorship**: Executive support unlocks resources and priority
5. **Document everything**: Compliance audits will require documentation
6. **Test extensively**: Allocate 40%+ of effort to testing and validation

Apply these lessons to build a realistic, achievable PQC deployment plan for your organization.
