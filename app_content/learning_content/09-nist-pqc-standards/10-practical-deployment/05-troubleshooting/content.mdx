```python
from dataclasses import dataclass
from typing import List, Dict, Optional, Tuple
from enum import Enum
import subprocess
import re

class FailureType(Enum):
    NETWORK = "network"
    TLS_VERSION = "tls_version"
    KEY_EXCHANGE = "key_exchange"
    CERTIFICATE = "certificate"
    SIGNATURE = "signature"
    TIMEOUT = "timeout"
    UNKNOWN = "unknown"

@dataclass
class DiagnosticResult:
    """Result of diagnostic check."""
    check_name: str
    passed: bool
    details: str
    recommendation: Optional[str] = None

class HandshakeDiagnostic:
    """
    Diagnose TLS handshake failures with PQC.
    """

    def __init__(self, host: str, port: int = 443):
        self.host = host
        self.port = port
        self.results: List[DiagnosticResult] = []

    def run_diagnostics(self) -> List[DiagnosticResult]:
        """Run full diagnostic suite."""
        self.results = []

        # Check connectivity
        self.results.append(self._check_connectivity())
        if not self.results[-1].passed:
            return self.results

        # Check TLS version
        self.results.append(self._check_tls_version())

        # Check key exchange
        self.results.append(self._check_key_exchange())

        # Check PQC groups specifically
        self.results.append(self._check_pqc_groups())

        # Check certificate
        self.results.append(self._check_certificate())

        return self.results

    def _check_connectivity(self) -> DiagnosticResult:
        """Check basic TCP connectivity."""
        # Simulated check
        connected = True  # Would actually attempt connection
        return DiagnosticResult(
            check_name="TCP Connectivity",
            passed=connected,
            details=f"Connection to {self.host}:{self.port} successful",
            recommendation=None if connected else "Check firewall rules and DNS"
        )

    def _check_tls_version(self) -> DiagnosticResult:
        """Check TLS version negotiation."""
        # Simulated - would run openssl s_client
        tls_version = "TLSv1.3"
        passed = tls_version == "TLSv1.3"
        return DiagnosticResult(
            check_name="TLS Version",
            passed=passed,
            details=f"Negotiated {tls_version}",
            recommendation=None if passed else "PQC requires TLS 1.3; update MinProtocol"
        )

    def _check_key_exchange(self) -> DiagnosticResult:
        """Check key exchange algorithm."""
        # Simulated
        kex = "X25519Kyber768Draft00"
        is_pqc = "kyber" in kex.lower() or "mlkem" in kex.lower()
        return DiagnosticResult(
            check_name="Key Exchange",
            passed=True,
            details=f"Using {kex} ({'PQC' if is_pqc else 'Classical'})",
            recommendation=None
        )

    def _check_pqc_groups(self) -> DiagnosticResult:
        """Check PQC group support."""
        # Simulated check for supported groups
        supported = ["x25519_mlkem768", "p384_mlkem768", "x25519"]
        pqc_groups = [g for g in supported if "mlkem" in g or "kyber" in g]

        if pqc_groups:
            return DiagnosticResult(
                check_name="PQC Groups",
                passed=True,
                details=f"PQC groups available: {', '.join(pqc_groups)}",
                recommendation=None
            )
        else:
            return DiagnosticResult(
                check_name="PQC Groups",
                passed=False,
                details="No PQC groups supported",
                recommendation="Add PQC groups to ssl_ecdh_curve or Groups config"
            )

    def _check_certificate(self) -> DiagnosticResult:
        """Check certificate chain."""
        # Simulated
        return DiagnosticResult(
            check_name="Certificate Chain",
            passed=True,
            details="Chain verified successfully with PQC signature",
            recommendation=None
        )

    def identify_failure_type(self) -> FailureType:
        """Identify the type of failure from diagnostics."""
        for result in self.results:
            if not result.passed:
                if "connectivity" in result.check_name.lower():
                    return FailureType.NETWORK
                elif "tls version" in result.check_name.lower():
                    return FailureType.TLS_VERSION
                elif "key exchange" in result.check_name.lower():
                    return FailureType.KEY_EXCHANGE
                elif "certificate" in result.check_name.lower():
                    return FailureType.CERTIFICATE
                elif "signature" in result.check_name.lower():
                    return FailureType.SIGNATURE
        return FailureType.UNKNOWN

    def generate_report(self) -> str:
        """Generate diagnostic report."""
        lines = [
            "TLS HANDSHAKE DIAGNOSTIC REPORT",
            "=" * 50,
            f"Target: {self.host}:{self.port}",
            "",
        ]

        all_passed = all(r.passed for r in self.results)
        lines.append(f"Overall Status: {'PASS' if all_passed else 'FAIL'}")
        lines.append("")

        for result in self.results:
            status = "✓" if result.passed else "✗"
            lines.append(f"{status} {result.check_name}: {result.details}")
            if result.recommendation:
                lines.append(f"  → {result.recommendation}")

        if not all_passed:
            failure_type = self.identify_failure_type()
            lines.extend([
                "",
                f"Failure Type: {failure_type.value}",
                "",
                "RECOMMENDED ACTIONS:",
            ])
            lines.extend(self._get_remediation_steps(failure_type))

        return "\n".join(lines)

    def _get_remediation_steps(self, failure_type: FailureType) -> List[str]:
        """Get remediation steps for failure type."""
        steps = {
            FailureType.NETWORK: [
                "1. Verify server is running and listening",
                "2. Check firewall allows port 443",
                "3. Verify DNS resolution",
                "4. Test from different network location",
            ],
            FailureType.TLS_VERSION: [
                "1. Update server MinProtocol to TLSv1.3",
                "2. Verify client supports TLS 1.3",
                "3. Check for TLS termination proxy issues",
            ],
            FailureType.KEY_EXCHANGE: [
                "1. Add PQC groups to server config",
                "2. Verify OQS provider is loaded",
                "3. Check group preference order",
                "4. Add classical fallback groups",
            ],
            FailureType.CERTIFICATE: [
                "1. Verify certificate is not expired",
                "2. Check full chain is being sent",
                "3. Verify trust store includes root CA",
                "4. Check certificate size limits",
            ],
        }
        return steps.get(failure_type, ["Investigate server logs for more details"])

# Example

print(diag.generate_report())
```

```python
from dataclasses import dataclass
from typing import List, Dict, Tuple
import time
import statistics

@dataclass
class PerformanceMetric:
    """Performance measurement."""
    name: str
    value_ms: float
    baseline_ms: float
    threshold_ms: float

    @property
    def delta_percent(self) -> float:
        if self.baseline_ms == 0:
            return 0
        return ((self.value_ms - self.baseline_ms) / self.baseline_ms) * 100

    @property
    def exceeds_threshold(self) -> bool:
        return self.value_ms > self.threshold_ms

class PerformanceDiagnostic:
    """
    Diagnose PQC performance issues.
    """

    def __init__(self):
        self.metrics: List[PerformanceMetric] = []
        self.baselines = {
            "handshake_total": 35.0,      # Classical handshake baseline (ms)
            "key_generation": 2.0,         # Classical keygen (ms)
            "key_exchange": 1.5,           # Classical key exchange (ms)
            "signature_verify": 0.5,       # Classical sig verify (ms)
            "certificate_parse": 1.0,      # Classical cert parse (ms)
        }
        self.thresholds = {
            "handshake_total": 100.0,      # Max acceptable (ms)
            "key_generation": 10.0,
            "key_exchange": 5.0,
            "signature_verify": 3.0,
            "certificate_parse": 5.0,
        }

    def measure_handshake(self, target: str) -> Dict[str, float]:
        """Measure handshake performance components."""
        # Simulated measurements
        measurements = {
            "handshake_total": 52.0,
            "key_generation": 3.5,
            "key_exchange": 2.8,
            "signature_verify": 1.2,
            "certificate_parse": 2.5,
        }
        return measurements

    def analyze_measurements(
        self,
        measurements: Dict[str, float]
    ) -> List[PerformanceMetric]:
        """Analyze measurements against baselines."""
        self.metrics = []

        for name, value in measurements.items():
            baseline = self.baselines.get(name, 0)
            threshold = self.thresholds.get(name, float('inf'))

            self.metrics.append(PerformanceMetric(
                name=name,
                value_ms=value,
                baseline_ms=baseline,
                threshold_ms=threshold
            ))

        return self.metrics

    def identify_bottlenecks(self) -> List[Tuple[str, str]]:
        """Identify performance bottlenecks."""
        bottlenecks = []

        for metric in self.metrics:
            if metric.exceeds_threshold:
                bottlenecks.append((
                    metric.name,
                    f"Exceeds threshold: {metric.value_ms:.1f}ms > {metric.threshold_ms:.1f}ms"
                ))
            elif metric.delta_percent > 50:
                bottlenecks.append((
                    metric.name,
                    f"Significant increase: +{metric.delta_percent:.1f}% from baseline"
                ))

        return bottlenecks

    def get_optimization_suggestions(
        self,
        bottlenecks: List[Tuple[str, str]]
    ) -> List[str]:
        """Get optimization suggestions for bottlenecks."""
        suggestions = []

        for name, _ in bottlenecks:
            if name == "handshake_total":
                suggestions.extend([
                    "Enable TLS session resumption/tickets",
                    "Consider 0-RTT for repeat connections",
                    "Review individual component times",
                ])
            elif name == "key_generation":
                suggestions.extend([
                    "Ensure AVX2/AVX-512 optimizations enabled",
                    "Consider pre-generating keys during idle time",
                    "Verify not using debug/reference implementation",
                ])
            elif name == "key_exchange":
                suggestions.extend([
                    "Check for large public key transmission",
                    "Verify network MTU settings",
                    "Consider connection pooling",
                ])
            elif name == "signature_verify":
                suggestions.extend([
                    "Cache certificate validation results",
                    "Optimize chain length (minimize intermediates)",
                    "Consider stateless verification where possible",
                ])
            elif name == "certificate_parse":
                suggestions.extend([
                    "Reduce certificate chain size",
                    "Use ML-DSA instead of SLH-DSA for smaller sigs",
                    "Enable certificate compression if available",
                ])

        return list(set(suggestions))

    def generate_report(self) -> str:
        """Generate performance diagnostic report."""
        lines = [
            "PERFORMANCE DIAGNOSTIC REPORT",
            "=" * 50,
            "",
            "MEASUREMENTS:",
        ]

        for metric in self.metrics:
            status = "⚠" if metric.exceeds_threshold else "✓"
            lines.append(
                f"  {status} {metric.name}: {metric.value_ms:.1f}ms "
                f"(baseline: {metric.baseline_ms:.1f}ms, delta: {metric.delta_percent:+.1f}%)"
            )

        bottlenecks = self.identify_bottlenecks()
        if bottlenecks:
            lines.extend(["", "BOTTLENECKS IDENTIFIED:"])
            for name, reason in bottlenecks:
                lines.append(f"  • {name}: {reason}")

            suggestions = self.get_optimization_suggestions(bottlenecks)
            if suggestions:
                lines.extend(["", "OPTIMIZATION SUGGESTIONS:"])
                for i, suggestion in enumerate(suggestions, 1):
                    lines.append(f"  {i}. {suggestion}")
        else:
            lines.extend(["", "No significant bottlenecks identified."])

        return "\n".join(lines)

# Example
perf_diag = PerformanceDiagnostic()
measurements = perf_diag.measure_handshake("example.com")
perf_diag.analyze_measurements(measurements)
print(perf_diag.generate_report())
```

```python
from dataclasses import dataclass
from typing import List, Optional

@dataclass
class CertificateInfo:
    """Certificate information."""
    subject: str
    issuer: str
    algorithm: str
    public_key_size: int
    signature_size: int
    total_size: int
    valid: bool
    error: Optional[str] = None

class CertificateDiagnostic:
    """
    Diagnose certificate chain issues.
    """

    MAX_CHAIN_SIZE = 16384  # 16KB typical limit

    def __init__(self):
        self.chain: List[CertificateInfo] = []

    def analyze_chain(self, cert_paths: List[str]) -> List[CertificateInfo]:
        """Analyze certificate chain."""
        self.chain = []

        # Simulated certificate analysis
        sample_certs = [
            CertificateInfo(
                subject="CN=example.com",
                issuer="CN=Intermediate CA",
                algorithm="ML-DSA-65",
                public_key_size=1952,
                signature_size=3309,
                total_size=6200,
                valid=True
            ),
            CertificateInfo(
                subject="CN=Intermediate CA",
                issuer="CN=Root CA",
                algorithm="ML-DSA-65",
                public_key_size=1952,
                signature_size=3309,
                total_size=6100,
                valid=True
            ),
            CertificateInfo(
                subject="CN=Root CA",
                issuer="CN=Root CA",
                algorithm="ML-DSA-65",
                public_key_size=1952,
                signature_size=3309,
                total_size=6000,
                valid=True
            ),
        ]

        self.chain = sample_certs
        return self.chain

    def get_total_chain_size(self) -> int:
        """Calculate total chain size."""
        return sum(cert.total_size for cert in self.chain)

    def check_chain_size(self) -> Tuple[bool, str]:
        """Check if chain fits within TLS limits."""
        total = self.get_total_chain_size()
        if total > self.MAX_CHAIN_SIZE:
            return False, f"Chain size {total} bytes exceeds limit of {self.MAX_CHAIN_SIZE} bytes"
        return True, f"Chain size {total} bytes within limit"

    def check_algorithm_support(
        self,
        supported_algorithms: List[str]
    ) -> List[Tuple[str, bool]]:
        """Check if all chain algorithms are supported."""
        results = []
        for cert in self.chain:
            supported = cert.algorithm in supported_algorithms
            results.append((cert.subject, supported))
        return results

    def suggest_optimizations(self) -> List[str]:
        """Suggest chain optimizations."""
        suggestions = []

        # Check chain size
        total = self.get_total_chain_size()
        if total > self.MAX_CHAIN_SIZE * 0.8:
            suggestions.append(
                f"Chain is {total/self.MAX_CHAIN_SIZE*100:.0f}% of limit - "
                "consider reducing intermediates"
            )

        # Check for large signatures
        for cert in self.chain:
            if cert.signature_size > 10000:  # SLH-DSA threshold
                suggestions.append(
                    f"{cert.subject} uses large signature ({cert.signature_size} bytes) - "
                    "consider ML-DSA for smaller size"
                )

        # Check chain depth
        if len(self.chain) > 3:
            suggestions.append(
                f"Chain depth is {len(self.chain)} - "
                "consider flattening hierarchy"
            )

        return suggestions

    def generate_report(self) -> str:
        """Generate certificate diagnostic report."""
        lines = [
            "CERTIFICATE CHAIN DIAGNOSTIC REPORT",
            "=" * 50,
            "",
            f"Chain Depth: {len(self.chain)} certificates",
            f"Total Size: {self.get_total_chain_size()} bytes",
            "",
            "CERTIFICATES:",
        ]

        for i, cert in enumerate(self.chain):
            status = "✓" if cert.valid else "✗"
            lines.extend([
                f"  [{i+1}] {cert.subject}",
                f"      Issuer: {cert.issuer}",
                f"      Algorithm: {cert.algorithm}",
                f"      Size: {cert.total_size} bytes",
                f"      Status: {status} {'Valid' if cert.valid else cert.error}",
            ])

        # Chain size check
        size_ok, size_msg = self.check_chain_size()
        lines.extend([
            "",
            f"SIZE CHECK: {'✓' if size_ok else '✗'} {size_msg}",
        ])

        # Optimization suggestions
        suggestions = self.suggest_optimizations()
        if suggestions:
            lines.extend(["", "OPTIMIZATION SUGGESTIONS:"])
            for suggestion in suggestions:
                lines.append(f"  • {suggestion}")

        return "\n".join(lines)

# Example
```

```python
from dataclasses import dataclass
from typing import Dict, List, Optional

@dataclass
class ClientCapability:
    """Client capability record."""
    name: str
    version: str
    supports_mlkem: bool
    supports_hybrid: bool
    supports_mldsa: bool
    notes: str = ""

class CompatibilityChecker:
    """
    Check client compatibility with PQC.
    """

    def __init__(self):
        self.known_clients: Dict[str, ClientCapability] = {}
        self._load_known_clients()

    def _load_known_clients(self):
        """Load known client capabilities."""
        self.known_clients = {
            "chrome": ClientCapability(
                "Chrome", "124+", True, True, False,
                "X25519Kyber768 in production"
            ),
            "firefox": ClientCapability(
                "Firefox", "128+", True, True, False,
                "X25519Kyber768 enabled by default"
            ),
            "safari": ClientCapability(
                "Safari", "unknown", False, False, False,
                "PQC support expected in future release"
            ),
            "curl": ClientCapability(
                "curl", "8.3+", True, True, False,
                "Requires OpenSSL 3.x with OQS"
            ),
            "openssl": ClientCapability(
                "OpenSSL", "3.0+", True, True, True,
                "Full support with OQS provider"
            ),
        }

    def identify_client(self, user_agent: str) -> Optional[ClientCapability]:
        """Identify client from user agent."""
        ua_lower = user_agent.lower()
        for key, client in self.known_clients.items():
            if key in ua_lower:
                return client
        return None

    def check_compatibility(
        self,
        client: ClientCapability,
        required_features: List[str]
    ) -> Dict[str, bool]:
        """Check if client meets requirements."""
        results = {}

        feature_map = {
            "mlkem": client.supports_mlkem,
            "hybrid": client.supports_hybrid,
            "mldsa": client.supports_mldsa,
        }

        for feature in required_features:
            results[feature] = feature_map.get(feature, False)

        return results

    def suggest_configuration(
        self,
        client_mix: Dict[str, float]
    ) -> List[str]:
        """Suggest server configuration based on client mix."""
        suggestions = []

        # Calculate PQC-capable percentage
        pqc_capable = 0
        for client_name, percentage in client_mix.items():
            client = self.known_clients.get(client_name.lower())
            if client and client.supports_hybrid:
                pqc_capable += percentage

        if pqc_capable < 50:
            suggestions.append(
                "Less than 50% of clients support PQC. "
                "Ensure classical fallback groups are configured."
            )

        if pqc_capable > 80:
            suggestions.append(
                "Over 80% of clients support PQC. "
                "Consider prioritizing PQC groups higher."
            )

        # Safari handling
        if "safari" in client_mix and client_mix["safari"] > 10:
            suggestions.append(
                f"Safari represents {client_mix['safari']:.0f}% of traffic. "
                "Classical fallback essential until Safari adds PQC."
            )

        return suggestions

    def generate_compatibility_report(
        self,
        client_mix: Dict[str, float]
    ) -> str:
        """Generate compatibility report for client mix."""
        lines = [
            "CLIENT COMPATIBILITY REPORT",
            "=" * 50,
            "",
            "CLIENT MIX:",
        ]

        pqc_capable_pct = 0
        for client_name, percentage in client_mix.items():
            client = self.known_clients.get(client_name.lower())
            if client:
                status = "✓ PQC" if client.supports_hybrid else "✗ Classical only"
                lines.append(f"  {client.name}: {percentage:.1f}% - {status}")
                if client.supports_hybrid:
                    pqc_capable_pct += percentage
            else:
                lines.append(f"  {client_name}: {percentage:.1f}% - Unknown")

        lines.extend([
            "",
            f"PQC-CAPABLE CLIENTS: {pqc_capable_pct:.1f}%",
            f"FALLBACK REQUIRED: {100-pqc_capable_pct:.1f}%",
            "",
        ])

        suggestions = self.suggest_configuration(client_mix)
        if suggestions:
            lines.append("RECOMMENDATIONS:")
            for suggestion in suggestions:
                lines.append(f"  • {suggestion}")

        return "\n".join(lines)
```
