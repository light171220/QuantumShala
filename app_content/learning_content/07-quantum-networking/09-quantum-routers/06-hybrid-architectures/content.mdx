# Hybrid Quantum-Classical Architectures

## Introduction

Practical quantum networks operate as hybrid systems, with classical networks providing control, management, and coordination while quantum channels carry quantum information. This lesson explores hybrid architecture design.

## Software-Defined Quantum Networking

```python
import numpy as np
from dataclasses import dataclass, field
from typing import List, Dict, Optional, Callable
from enum import Enum
from abc import ABC, abstractmethod

class NetworkPlane(Enum):
    """Network planes."""
    MANAGEMENT = "management"
    CONTROL = "control"
    DATA = "data"

@dataclass
class QuantumNetworkElement:
    """Element in quantum network."""
    element_id: str
    element_type: str
    capabilities: List[str]
    ports: int
    location: str

class ControlInterface(ABC):
    """Abstract control interface for quantum devices."""

    @abstractmethod
    def configure_switch(self, input_port: int, output_port: int) -> bool:
        pass

    @abstractmethod
    def initiate_entanglement(self, neighbor: str) -> bool:
        pass

    @abstractmethod
    def perform_bell_measurement(self, qubit1: int, qubit2: int) -> int:
        pass

    @abstractmethod
    def get_status(self) -> Dict:
        pass

class QuantumNetworkController:
    """
    Software-Defined Quantum Network Controller.
    """

    def __init__(self, controller_id: str):
        self.controller_id = controller_id
        self.elements: Dict[str, QuantumNetworkElement] = {}
        self.topology: Dict[str, List[str]] = {}
        self.element_interfaces: Dict[str, ControlInterface] = {}
        self.paths: Dict[str, List[str]] = {}

    def register_element(
        self,
        element: QuantumNetworkElement,
        interface: Optional[ControlInterface] = None
    ):
        """Register network element with controller."""
        self.elements[element.element_id] = element
        if interface:
            self.element_interfaces[element.element_id] = interface
        self.topology[element.element_id] = []

    def add_link(self, element1: str, element2: str):
        """Add link between elements."""
        if element1 in self.topology:
            self.topology[element1].append(element2)
        if element2 in self.topology:
            self.topology[element2].append(element1)

    def compute_path(self, source: str, destination: str) -> Optional[List[str]]:
        """Compute path using controller's global view."""
        # BFS for shortest path
        if source not in self.topology or destination not in self.topology:
            return None

        visited = {source}
        queue = [(source, [source])]

        while queue:
            current, path = queue.pop(0)

            if current == destination:
                return path

            for neighbor in self.topology.get(current, []):
                if neighbor not in visited:
                    visited.add(neighbor)
                    queue.append((neighbor, path + [neighbor]))

        return None

    def provision_path(
        self,
        source: str,
        destination: str
    ) -> Optional[str]:
        """Provision a path for quantum communication."""
        path = self.compute_path(source, destination)
        if not path:
            return None

        path_id = f"path-{source}-{destination}-{len(self.paths)}"

        # Configure switches along path
        for i in range(len(path) - 1):
            element = path[i]
            next_element = path[i + 1]

            interface = self.element_interfaces.get(element)
            if interface:
                # Simplified: assume port mapping known
                input_port = 0  # Would be looked up
                output_port = 1  # Would be looked up
                interface.configure_switch(input_port, output_port)

        self.paths[path_id] = path
        return path_id

    def establish_entanglement(
        self,
        path_id: str
    ) -> bool:
        """Establish entanglement along provisioned path."""
        path = self.paths.get(path_id)
        if not path:
            return False

        # Initiate entanglement on each link
        for i in range(len(path) - 1):
            element = path[i]
            neighbor = path[i + 1]

            interface = self.element_interfaces.get(element)
            if interface:
                if not interface.initiate_entanglement(neighbor):
                    return False

        return True

    def trigger_swapping(self, path_id: str) -> bool:
        """Trigger entanglement swapping at intermediate nodes."""
        path = self.paths.get(path_id)
        if not path or len(path) < 3:
            return False

        # Perform swapping at each intermediate node
        for i in range(1, len(path) - 1):
            element = path[i]
            interface = self.element_interfaces.get(element)
            if interface:
                # Bell measurement on local qubits
                interface.perform_bell_measurement(0, 1)

        return True

    def get_network_view(self) -> Dict:
        """Get global network view."""
        return {
            "elements": len(self.elements),
            "links": sum(len(neighbors) for neighbors in self.topology.values()) // 2,
            "active_paths": len(self.paths),
            "element_types": {
                e.element_type: sum(1 for x in self.elements.values()
                                   if x.element_type == e.element_type)
                for e in self.elements.values()
            }
        }
```

```python
class HybridNetworkOrchestrator:
    """
    Orchestrator for hybrid quantum-classical networks.
    """

    def __init__(self):
        self.quantum_controller = QuantumNetworkController("q-ctrl")
        self.classical_routes: Dict[str, List[str]] = {}
        self.hybrid_services: Dict[str, Dict] = {}

    def register_service(
        self,
        service_id: str,
        quantum_requirements: Dict,
        classical_requirements: Dict
    ):
        """Register a hybrid service."""
        self.hybrid_services[service_id] = {
            "quantum": quantum_requirements,
            "classical": classical_requirements,
            "status": "registered"
        }

    def provision_hybrid_connection(
        self,
        service_id: str,
        source: str,
        destination: str
    ) -> Dict:
        """Provision both quantum and classical paths."""
        service = self.hybrid_services.get(service_id)
        if not service:
            return {"success": False, "reason": "Service not found"}

        result = {
            "service_id": service_id,
            "quantum_path": None,
            "classical_path": None,
            "success": False
        }

        # Provision quantum path
        q_path_id = self.quantum_controller.provision_path(source, destination)
        if q_path_id:
            result["quantum_path"] = self.quantum_controller.paths.get(q_path_id)

        # Provision classical path (simplified)
        c_path = self._provision_classical_path(source, destination)
        if c_path:
            result["classical_path"] = c_path
            self.classical_routes[f"classical-{source}-{destination}"] = c_path

        result["success"] = bool(result["quantum_path"] and result["classical_path"])

        if result["success"]:
            service["status"] = "provisioned"

        return result

    def _provision_classical_path(
        self,
        source: str,
        destination: str
    ) -> Optional[List[str]]:
        """Provision classical path (simplified)."""
        # In practice, would interface with classical SDN controller
        return [source, f"classical-switch", destination]

    def get_service_status(self, service_id: str) -> Dict:
        """Get hybrid service status."""
        service = self.hybrid_services.get(service_id)
        if not service:
            return {"found": False}

        return {
            "found": True,
            "service_id": service_id,
            "status": service["status"],
            "quantum_requirements": service["quantum"],
            "classical_requirements": service["classical"]
        }

class HybridQKDService:
    """
    Hybrid QKD service using quantum and classical networks.
    """

    def __init__(self, orchestrator: HybridNetworkOrchestrator):
        self.orchestrator = orchestrator
        self.key_buffers: Dict[str, bytes] = {}

    def establish_secure_channel(
        self,
        alice: str,
        bob: str,
        key_rate_kbps: float = 1.0
    ) -> Dict:
        """Establish secure channel using QKD."""
        # Register service
        service_id = f"qkd-{alice}-{bob}"

        self.orchestrator.register_service(
            service_id,
            quantum_requirements={
                "type": "qkd",
                "fidelity": 0.9,
                "key_rate": key_rate_kbps
            },
            classical_requirements={
                "bandwidth_mbps": 10,
                "latency_ms": 50,
                "authenticated": True
            }
        )

        # Provision hybrid connection
        result = self.orchestrator.provision_hybrid_connection(
            service_id, alice, bob
        )

        if result["success"]:
            return {
                "status": "established",
                "service_id": service_id,
                "quantum_path": result["quantum_path"],
                "classical_path": result["classical_path"],
                "estimated_key_rate": key_rate_kbps
            }

        return {
            "status": "failed",
            "reason": "Could not provision hybrid connection"
        }
```

## Architecture Considerations

```
┌─────────────────────────────────────────────────────────────────────┐
│                    ARCHITECTURE DESIGN CONSIDERATIONS               │
├─────────────────────────────────────────────────────────────────────┤
│                                                                     │
│  TIMING AND SYNCHRONIZATION:                                        │
│  • Quantum operations require precise timing                       │
│  • Classical control must not add latency                          │
│  • Synchronization between quantum and classical planes            │
│                                                                     │
│  SCALABILITY:                                                       │
│  • Controller must handle growing number of elements               │
│  • Hierarchical or distributed control for large networks          │
│  • Balance central view vs local autonomy                          │
│                                                                     │
│  RELIABILITY:                                                       │
│  • Controller redundancy (no single point of failure)              │
│  • Graceful degradation when control plane issues                  │
│  • Local fallback for time-critical operations                     │
│                                                                     │
│  SECURITY:                                                          │
│  • Control plane must be secured                                   │
│  • Authentication between controller and elements                  │
│  • Protect against control plane attacks                           │
│                                                                     │
│  INTEROPERABILITY:                                                  │
│  • Standard interfaces between components                          │
│  • Multi-vendor element support                                    │
│  • Protocol compatibility                                          │
│                                                                     │
└─────────────────────────────────────────────────────────────────────┘
```
