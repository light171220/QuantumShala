## Link-Level Protocols

```python
import numpy as np
from dataclasses import dataclass, field
from typing import List, Dict, Optional, Tuple
from enum import Enum
from abc import ABC, abstractmethod
import random

class LinkProtocolType(Enum):
    DLCZ = "dlcz"
    MIDPOINT_BSM = "midpoint_bsm"
    DIRECT = "direct"

@dataclass
class LinkState:
    """State of a link."""
    node1: str
    node2: str
    entangled: bool = False
    fidelity: float = 0.0
    timestamp: float = 0.0

class LinkProtocol(ABC):
    """Abstract link-level protocol."""

    @abstractmethod
    def attempt_generation(self) -> Tuple[bool, float]:
        """Attempt to generate entanglement."""
        pass

    @abstractmethod
    def herald_success(self) -> Dict:
        """Process heralding signal."""
        pass

class DLCZProtocol(LinkProtocol):
    """
    DLCZ-style link protocol.

    Uses atomic ensembles to generate photon-matter entanglement,
    then BSM to create matter-matter entanglement.
    """

    def __init__(
        self,
        p_write: float = 0.01,
        p_read: float = 0.9,
        p_transmission: float = 0.5,
        p_detection: float = 0.9
    ):
        self.p_write = p_write
        self.p_read = p_read
        self.p_transmission = p_transmission
        self.p_detection = p_detection

    def attempt_generation(self) -> Tuple[bool, float]:
        """Attempt DLCZ entanglement generation."""
        # Probability of success
        p_success = self.p_write * self.p_transmission * self.p_detection

        success = random.random() < p_success

        if success:
            # Fidelity depends on read-out
            fidelity = 0.95 * self.p_read
            return True, fidelity
        return False, 0.0

    def herald_success(self) -> Dict:
        """DLCZ heralding."""
        return {
            "protocol": "DLCZ",
            "herald_type": "single_click",
            "classical_bits": 1
        }

class MidpointBSMProtocol(LinkProtocol):
    """
    Midpoint BSM link protocol.

    Both nodes send photons to midpoint for BSM.
    Heralded entanglement between nodes.
    """

    def __init__(
        self,
        p_source: float = 0.9,
        p_transmission: float = 0.5,
        p_bsm: float = 0.5
    ):
        self.p_source = p_source
        self.p_transmission = p_transmission
        self.p_bsm = p_bsm

    def attempt_generation(self) -> Tuple[bool, float]:
        """Attempt midpoint BSM entanglement."""
        # Both photons must arrive and BSM must succeed
        p_success = (self.p_source ** 2) * (self.p_transmission ** 2) * self.p_bsm

        success = random.random() < p_success

        if success:
            fidelity = 0.98  # Higher fidelity than DLCZ
            return True, fidelity
        return False, 0.0

    def herald_success(self) -> Dict:
        """Midpoint BSM heralding."""
        return {
            "protocol": "MidpointBSM",
            "herald_type": "bsm_result",
            "classical_bits": 2  # Bell state identification
        }

class LinkManager:
    """Manage link-level operations."""

    def __init__(self):
        self.links: Dict[Tuple[str, str], LinkState] = {}
        self.protocols: Dict[Tuple[str, str], LinkProtocol] = {}

    def create_link(
        self,
        node1: str,
        node2: str,
        protocol_type: LinkProtocolType = LinkProtocolType.MIDPOINT_BSM
    ):
        """Create a new link."""
        key = (min(node1, node2), max(node1, node2))
        self.links[key] = LinkState(node1, node2)

        if protocol_type == LinkProtocolType.DLCZ:
            self.protocols[key] = DLCZProtocol()
        else:
            self.protocols[key] = MidpointBSMProtocol()

    def generate_entanglement(
        self,
        node1: str,
        node2: str,
        max_attempts: int = 1000
    ) -> Tuple[bool, int, float]:
        """Generate entanglement with retries."""
        key = (min(node1, node2), max(node1, node2))

        if key not in self.protocols:
            return False, 0, 0.0

        protocol = self.protocols[key]

        for attempt in range(1, max_attempts + 1):
            success, fidelity = protocol.attempt_generation()
            if success:
                self.links[key] = LinkState(
                    node1, node2, True, fidelity, attempt
                )
                return True, attempt, fidelity

        return False, max_attempts, 0.0

# Demonstrate link protocols
print("Link-Level Protocol Demonstration")
print("=" * 60)

manager = LinkManager()
manager.create_link("Alice", "R1", LinkProtocolType.MIDPOINT_BSM)

results = []
for _ in range(100):
    success, attempts, fidelity = manager.generate_entanglement("Alice", "R1")
    if success:
        results.append(attempts)

print(f"\nMidpoint BSM Protocol:")
print(f"  Success rate: {len(results)}%")
print(f"  Avg attempts: {np.mean(results):.1f}")
print(f"  Min/Max: {min(results)}/{max(results)}")
```

## Network-Level Routing

```python
@dataclass
class RouteRequest:
    """Request for end-to-end entanglement."""
    source: str
    destination: str
    min_fidelity: float = 0.9
    timeout: float = 1.0  # seconds

@dataclass
class Route:
    """Computed route through network."""
    path: List[str]
    expected_fidelity: float
    expected_time: float
    n_swaps: int

class QuantumRouter:
    """
    Quantum network routing.

    Key differences from classical routing:
    - Entanglement is consumed (cannot copy)
    - Fidelity degrades at each hop
    - Timing constraints from memory coherence
    """

    def __init__(
        self,
        topology: Dict[str, List[str]],
        link_fidelities: Dict[Tuple[str, str], float]
    ):
        self.topology = topology
        self.link_fidelities = link_fidelities

    def compute_route(self, request: RouteRequest) -> Optional[Route]:
        """Compute best route for request."""
        paths = self._find_all_paths(
            request.source,
            request.destination
        )

        if not paths:
            return None

        # Evaluate each path
        best_route = None
        best_score = 0

        for path in paths:
            fidelity = self._path_fidelity(path)
            n_swaps = len(path) - 2

            if fidelity >= request.min_fidelity:
                # Score: higher fidelity and fewer swaps is better
                score = fidelity / (1 + 0.1 * n_swaps)
                if score > best_score:
                    best_score = score
                    best_route = Route(
                        path=path,
                        expected_fidelity=fidelity,
                        expected_time=self._estimate_time(path),
                        n_swaps=n_swaps
                    )

        return best_route

    def _find_all_paths(
        self,
        source: str,
        dest: str,
        path: List[str] = None
    ) -> List[List[str]]:
        """Find all paths (simplified DFS)."""
        if path is None:
            path = []

        path = path + [source]

        if source == dest:
            return [path]

        if source not in self.topology:
            return []

        paths = []
        for neighbor in self.topology[source]:
            if neighbor not in path:
                new_paths = self._find_all_paths(neighbor, dest, path)
                paths.extend(new_paths)

        return paths

    def _path_fidelity(self, path: List[str]) -> float:
        """Calculate expected end-to-end fidelity."""
        if len(path) < 2:
            return 1.0

        # Start with first link fidelity
        total_fidelity = 1.0

        for i in range(len(path) - 1):
            key = (min(path[i], path[i+1]), max(path[i], path[i+1]))
            link_fidelity = self.link_fidelities.get(key, 0.95)
            total_fidelity *= link_fidelity

        # Swapping degrades fidelity
        n_swaps = len(path) - 2
        swap_fidelity = 0.98 ** n_swaps  # Each swap reduces fidelity

        return total_fidelity * swap_fidelity

    def _estimate_time(self, path: List[str]) -> float:
        """Estimate time to establish entanglement."""
        # Simplified model
        n_links = len(path) - 1
        avg_attempts = 10  # Average attempts per link
        attempt_time = 1e-3  # 1 ms per attempt

        return n_links * avg_attempts * attempt_time

# Demonstrate routing
print("\n\nQuantum Routing Demonstration")
print("=" * 60)

topology = {
    "Alice": ["R1", "R2"],
    "R1": ["Alice", "R3", "R2"],
    "R2": ["Alice", "R1", "R4"],
    "R3": ["R1", "Bob"],
    "R4": ["R2", "Bob"],

    ("R1", "R2"): 0.98,
    ("R1", "R3"): 0.96,
    ("R2", "R4"): 0.97,

if route:
    print(f"\nRoute found:")
    print(f"  Path: {' -> '.join(route.path)}")
    print(f"  Expected fidelity: {route.expected_fidelity:.4f}")
    print(f"  Swaps needed: {route.n_swaps}")
    print(f"  Est. time: {route.expected_time*1000:.1f} ms")
```

```python
class QKDProtocol:
    """
    QKD protocol running on quantum internet.
    """

    def __init__(
        self,
        protocol_type: str = "E91",
        target_key_rate: float = 1000  # bits/s
    ):
        self.protocol_type = protocol_type
        self.target_key_rate = target_key_rate

    def establish_session(
        self,
        alice: str,
        bob: str,
        n_pairs: int = 1000
    ) -> Dict:
        """Establish QKD session."""
        return {
            "session_id": f"qkd_{alice}_{bob}",
            "protocol": self.protocol_type,
            "pairs_requested": n_pairs,
            "status": "establishing"
        }

    def run_protocol(self, entangled_pairs: List[Dict]) -> Dict:
        """Run QKD protocol on received pairs."""
        n_pairs = len(entangled_pairs)
        avg_fidelity = np.mean([p.get("fidelity", 0.9) for p in entangled_pairs])

        # Calculate expected key rate
        qber = 1 - avg_fidelity  # Simplified
        efficiency = 0.5  # Basis matching
        error_correction = 1.2  # Overhead

        raw_bits = n_pairs * efficiency
        secret_bits = raw_bits * (1 - self._binary_entropy(qber) * error_correction)

        return {
            "raw_bits": int(raw_bits),
            "secret_bits": max(0, int(secret_bits)),
            "qber": qber,
            "status": "complete"
        }

    def _binary_entropy(self, p: float) -> float:
        """Binary entropy function."""
        if p <= 0 or p >= 1:
            return 0
        return -p * np.log2(p) - (1-p) * np.log2(1-p)

class DistributedComputationProtocol:
    """
    Protocol for distributed quantum computation.
    """

    def __init__(self, n_parties: int = 2):
        self.n_parties = n_parties

    def setup_computation(
        self,
        parties: List[str],
        circuit_depth: int
    ) -> Dict:
        """Setup distributed computation."""
        return {
            "parties": parties,
            "entanglement_required": circuit_depth * (len(parties) - 1),
            "gates_per_party": circuit_depth,
            "communication_rounds": circuit_depth
        }

    def execute_step(
        self,
        step: int,
        entangled_pairs: Dict,
        local_operations: Dict
    ) -> Dict:
        """Execute one step of distributed computation."""
        return {
            "step": step,
            "entanglement_consumed": len(entangled_pairs),
            "classical_bits_exchanged": step * 2,
            "status": "in_progress"
        }

class SensingProtocol:
    """
    Distributed quantum sensing protocol.
    """

    def __init__(self, n_sensors: int):
        self.n_sensors = n_sensors

    def setup_sensing(
        self,
        sensor_locations: List[str],
        measurement_type: str = "phase"
    ) -> Dict:
        """Setup sensing network."""
        # GHZ states for optimal sensitivity
        return {
            "sensors": sensor_locations,
            "n_sensors": len(sensor_locations),
            "state_type": "GHZ",
            "entanglement_needed": len(sensor_locations) - 1,
            "sensitivity_gain": np.sqrt(len(sensor_locations))
        }

    def perform_measurement(
        self,
        entangled_state: Dict,
        measurement_settings: Dict
    ) -> Dict:
        """Perform coordinated measurement."""
        return {
            "measurement_complete": True,
            "classical_processing_needed": True,
            "heisenberg_limited": True
        }
```

## Protocol Challenges

```
┌─────────────────────────────────────────────────────────────────────┐
│              QUANTUM PROTOCOL DESIGN CHALLENGES                     │
├─────────────────────────────────────────────────────────────────────┤
│                                                                     │
│  1. TIMING CONSTRAINTS                                              │
│     • Memory coherence limits wait time                            │
│     • Classical communication takes time                           │
│     • Must coordinate across network                               │
│                                                                     │
│  2. PROBABILISTIC SUCCESS                                          │
│     • Operations may fail                                          │
│     • Retries consume resources and time                           │
│     • Must handle failures gracefully                              │
│                                                                     │
│  3. FIDELITY DEGRADATION                                           │
│     • Each operation reduces fidelity                              │
│     • May need purification (consumes pairs)                       │
│     • Trade-off: rate vs fidelity                                  │
│                                                                     │
│  4. RESOURCE MANAGEMENT                                            │
│     • Entanglement is consumed when used                           │
│     • Memory slots are limited                                     │
│     • Must balance multiple requests                               │
│                                                                     │
│  5. CLASSICAL OVERHEAD                                              │
│     • Many classical messages needed                               │
│     • Latency affects performance                                  │
│     • Authentication required                                       │
│                                                                     │
│  6. HETEROGENEOUS NETWORKS                                          │

│     • Interoperability challenges                                  │
```
