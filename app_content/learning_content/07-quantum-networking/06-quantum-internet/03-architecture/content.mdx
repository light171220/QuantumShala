## Node Implementation

```python
from abc import ABC, abstractmethod

class NodeType(Enum):
    END_NODE = "A"
    REPEATER = "B"
    ROUTER = "C"
    GATEWAY = "D"

@dataclass
class NodeCapabilities:
    """Capabilities of a network node."""
    memory_qubits: int
    memory_coherence: float  # seconds
    gate_fidelity: float
    photon_rate: float  # Hz
    interfaces: int
    can_route: bool
    can_compute: bool

class QuantumNode(ABC):
    """Abstract base class for quantum network nodes."""

    def __init__(
        self,
        node_id: str,
        node_type: NodeType,
        capabilities: NodeCapabilities
    ):
        self.node_id = node_id
        self.node_type = node_type
        self.capabilities = capabilities
        self.neighbors: List[str] = []
        self.entangled_pairs: Dict = {}

    @abstractmethod
    def process_photon(self, photon_state):
        """Process incoming photon."""
        pass

    @abstractmethod
    def generate_entanglement(self, target_id: str):
        """Generate entanglement with target."""
        pass

class EndNode(QuantumNode):
    """End user node with quantum capabilities."""

    def __init__(self, node_id: str):
        capabilities = NodeCapabilities(
            memory_qubits=4,
            memory_coherence=0.1,
            gate_fidelity=0.99,
            photon_rate=1e6,
            interfaces=1,
            can_route=False,
            can_compute=True
        )
        super().__init__(node_id, NodeType.END_NODE, capabilities)

    def process_photon(self, photon_state):
        """Store photon in local memory."""
        return {"action": "store", "success": True}

    def generate_entanglement(self, target_id: str):
        """Request entanglement from network."""
        return {"request": "entanglement", "target": target_id}

    def run_application(self, app_type: str, params: Dict):
        """Run quantum application."""
        supported = ["qkd", "teleportation", "computation"]
        if app_type in supported:
            return {"status": "running", "app": app_type}
        return {"status": "unsupported"}

class RepeaterNode(QuantumNode):
    """Quantum repeater node."""

    def __init__(self, node_id: str):
        capabilities = NodeCapabilities(
            memory_qubits=8,
            memory_coherence=1.0,
            gate_fidelity=0.995,
            photon_rate=1e6,
            interfaces=2,
            can_route=False,
            can_compute=False
        )
        super().__init__(node_id, NodeType.REPEATER, capabilities)

    def process_photon(self, photon_state):
        """Perform BSM or store in memory."""
        return {"action": "bsm_or_store", "success": True}

    def generate_entanglement(self, target_id: str):
        """Generate link entanglement."""
        return {"link": target_id, "status": "generating"}

    def perform_swap(self, pair1_id: str, pair2_id: str):
        """Perform entanglement swapping."""
        return {
            "operation": "swap",
            "input_pairs": [pair1_id, pair2_id],
            "output": "extended_entanglement"
        }

class RouterNode(QuantumNode):
    """Quantum router node."""

    def __init__(self, node_id: str, n_interfaces: int = 4):
        capabilities = NodeCapabilities(
            memory_qubits=16,
            memory_coherence=1.0,
            gate_fidelity=0.99,
            photon_rate=1e6,
            interfaces=n_interfaces,
            can_route=True,
            can_compute=False
        )
        super().__init__(node_id, NodeType.ROUTER, capabilities)
        self.routing_table: Dict = {}

    def process_photon(self, photon_state):
        """Route or process photon."""
        return {"action": "route_or_process"}

    def generate_entanglement(self, target_id: str):
        """Generate entanglement to any interface."""
        return {"target": target_id, "interface": self._select_interface(target_id)}

    def _select_interface(self, target_id: str) -> int:
        """Select interface for target."""
        return self.routing_table.get(target_id, {}).get("interface", 0)

    def compute_path(self, source: str, dest: str) -> List[str]:
        """Compute routing path."""
        # Simplified routing
        return [source, self.node_id, dest]
```

## Network Topology

```python
@dataclass
class NetworkLink:
    """Link between nodes."""
    node1: str
    node2: str
    distance: float  # km
    channel_type: str  # fiber, free-space, satellite
    loss_db_km: float

class NetworkTopology:
    """
    Quantum network topology.

    Defines the physical structure of the network
    and provides routing information.
    """

    def __init__(self):
        self.nodes: Dict[str, QuantumNode] = {}
        self.links: List[NetworkLink] = []

    def add_node(self, node: QuantumNode):
        """Add node to network."""
        self.nodes[node.node_id] = node

    def add_link(
        self,
        node1: str,
        node2: str,
        distance: float,
        channel_type: str = "fiber"
    ):
        """Add link between nodes."""
        loss = 0.2 if channel_type == "fiber" else 0.01
        link = NetworkLink(node1, node2, distance, channel_type, loss)
        self.links.append(link)

        # Update neighbor lists
        self.nodes[node1].neighbors.append(node2)
        self.nodes[node2].neighbors.append(node1)

    def get_path(self, source: str, dest: str) -> List[str]:
        """Find path between nodes (simple BFS)."""
        if source == dest:
            return [source]

        visited = {source}
        queue = [(source, [source])]

        while queue:
            node, path = queue.pop(0)
            for neighbor in self.nodes[node].neighbors:
                if neighbor == dest:
                    return path + [neighbor]
                if neighbor not in visited:
                    visited.add(neighbor)
                    queue.append((neighbor, path + [neighbor]))

        return []  # No path found

    def path_metrics(self, path: List[str]) -> Dict:
        """Calculate metrics for a path."""
        total_distance = 0
        n_hops = len(path) - 1
        n_repeaters = sum(1 for n in path[1:-1]
                        if self.nodes[n].node_type == NodeType.REPEATER)
        n_routers = sum(1 for n in path[1:-1]
                       if self.nodes[n].node_type == NodeType.ROUTER)

        for i in range(len(path) - 1):
            for link in self.links:
                if {link.node1, link.node2} == {path[i], path[i+1]}:
                    total_distance += link.distance
                    break

        return {
            "path": path,
            "total_distance": total_distance,
            "hops": n_hops,
            "repeaters": n_repeaters,
            "routers": n_routers
        }
```
