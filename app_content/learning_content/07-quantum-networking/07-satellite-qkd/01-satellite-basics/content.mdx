# Satellite QKD Fundamentals

## Introduction

While fiber-based quantum networks face fundamental distance limitations, satellites offer a path to truly global quantum communication. This lesson introduces the fundamentals of satellite-based QKD, explaining why space is essential for long-distance quantum links.

## Fiber vs Satellite Comparison

```python
import numpy as np
from dataclasses import dataclass
from typing import List, Dict, Tuple
from enum import Enum

class ChannelType(Enum):
    FIBER = "fiber"
    FREE_SPACE_GROUND = "free_space_ground"
    SATELLITE_LEO = "satellite_leo"
    SATELLITE_GEO = "satellite_geo"

@dataclass
class ChannelCharacteristics:
    """Characteristics of a quantum channel."""
    channel_type: ChannelType
    loss_model: str
    typical_loss_db: float
    distance_range: str
    advantages: List[str]
    disadvantages: List[str]

class ChannelComparison:
    """
    Compare different quantum channel types.
    """

    def __init__(self):
        self.channels = self._define_channels()

    def _define_channels(self) -> Dict[ChannelType, ChannelCharacteristics]:
        """Define channel characteristics."""
        return {
            ChannelType.FIBER: ChannelCharacteristics(
                channel_type=ChannelType.FIBER,
                loss_model="Exponential: 10^(-0.2L/10)",
                typical_loss_db=20,  # per 100 km
                distance_range="0-500 km (practical)",
                advantages=[
                    "All-weather operation",
                    "24/7 availability",
                    "Existing infrastructure",
                    "Low pointing requirements"
                ],
                disadvantages=[
                    "Exponential loss with distance",
                    "Needs repeaters for long distance",
                    "Infrastructure costs"
                ]
            ),
            ChannelType.SATELLITE_LEO: ChannelCharacteristics(
                channel_type=ChannelType.SATELLITE_LEO,
                loss_model="Inverse square + atmosphere",
                typical_loss_db=35,  # total for LEO pass
                distance_range="500-1200 km (LEO altitude)",
                advantages=[
                    "Global coverage possible",
                    "Lower loss than equivalent fiber",
                    "No ground infrastructure between endpoints",
                    "Intercontinental links"
                ],
                disadvantages=[
                    "Limited pass duration (~5-10 min)",
                    "Weather dependent",
                    "Day/night restrictions",
                    "Complex pointing system",
                    "High development cost"
                ]
            ),
            ChannelType.SATELLITE_GEO: ChannelCharacteristics(
                channel_type=ChannelType.SATELLITE_GEO,
                loss_model="Inverse square + atmosphere",
                typical_loss_db=60,  # much higher for GEO
                distance_range="35,786 km (GEO altitude)",
                advantages=[
                    "24/7 coverage of hemisphere",
                    "Fixed satellite position",
                    "Simpler ground tracking"
                ],
                disadvantages=[
                    "Very high loss (long distance)",
                    "Currently impractical for QKD",
                    "Large diffraction losses"
                ]
            )
        }

    def calculate_loss(
        self,
        channel_type: ChannelType,
        distance_km: float
    ) -> float:
        """Calculate channel loss in dB."""
        if channel_type == ChannelType.FIBER:
            # Fiber: exponential loss
            return 0.2 * distance_km

        elif channel_type == ChannelType.SATELLITE_LEO:
            # Satellite: inverse square + atmosphere
            # Simplified model
            geometric_loss = 20 * np.log10(distance_km * 1000)  # in meters
            atmospheric_loss = 5  # ~5 dB through atmosphere
            diffraction_loss = 10  # depends on aperture
            return geometric_loss + atmospheric_loss + diffraction_loss - 60  # normalize

        return 0

    def compare_loss_vs_distance(self) -> Dict:
        """Compare loss vs distance for different channels."""
        distances = np.array([10, 50, 100, 200, 500, 1000, 2000])

        results = {}
        for channel in [ChannelType.FIBER, ChannelType.SATELLITE_LEO]:
            losses = [self.calculate_loss(channel, d) for d in distances]
            results[channel.value] = list(zip(distances, losses))

        return results

    def print_comparison(self):
        """Print channel comparison."""
        print("Quantum Channel Comparison")
        print("=" * 70)

        for channel_type, chars in self.channels.items():
            print(f"\n{channel_type.value.upper()}")
            print("-" * 50)
            print(f"  Loss model: {chars.loss_model}")
            print(f"  Typical loss: {chars.typical_loss_db} dB")
            print(f"  Range: {chars.distance_range}")
            print(f"  Advantages:")
            for adv in chars.advantages[:2]:
                print(f"    + {adv}")
            print(f"  Disadvantages:")
            for dis in chars.disadvantages[:2]:
                print(f"    - {dis}")

# Compare channels
comparison = ChannelComparison()
comparison.print_comparison()

print("\n\nLoss Comparison (dB):")
print("-" * 50)
print(f"{'Distance (km)':<15} {'Fiber':<15} {'Satellite (LEO)'}")
for dist, fiber_loss in comparison.compare_loss_vs_distance()["fiber"]:
    sat_loss = comparison.calculate_loss(ChannelType.SATELLITE_LEO, dist)
    if dist <= 1000:  # Only show where satellite makes sense
        print(f"{dist:<15} {fiber_loss:<15.1f} {sat_loss:.1f}")
```

## Satellite QKD Link Budget

```python
@dataclass
class SatelliteLink:
    """Satellite QKD link parameters."""
    altitude_km: float
    telescope_diameter_m: float
    wavelength_nm: float
    pointing_accuracy_urad: float
    detector_efficiency: float

class SatelliteLinkBudget:
    """
    Calculate satellite QKD link budget.
    """

    def __init__(self, link: SatelliteLink):
        self.link = link
        self.wavelength_m = link.wavelength_nm * 1e-9

    def geometric_loss(self, zenith_angle_deg: float = 0) -> float:
        """Calculate geometric (inverse square) loss."""
        # Distance to satellite
        slant_range = self.link.altitude_km * 1000 / np.cos(np.radians(zenith_angle_deg))

        # Diffraction-limited beam divergence
        beam_div = 1.22 * self.wavelength_m / self.link.telescope_diameter_m

        # Spot size at ground
        spot_radius = slant_range * beam_div

        # Fraction collected by ground telescope
        collect_area = np.pi * (self.link.telescope_diameter_m / 2) ** 2
        spot_area = np.pi * spot_radius ** 2

        efficiency = collect_area / spot_area
        return -10 * np.log10(efficiency)

    def atmospheric_loss(self, zenith_angle_deg: float = 0) -> float:
        """Estimate atmospheric loss."""
        # Simplified model: ~3-8 dB depending on conditions
        base_loss = 5  # dB at zenith
        airmass = 1 / np.cos(np.radians(min(zenith_angle_deg, 85)))
        return base_loss * airmass

    def pointing_loss(self) -> float:
        """Loss due to pointing errors."""
        # Simplified model
        pointing_error_rad = self.link.pointing_accuracy_urad * 1e-6
        beam_div = 1.22 * self.wavelength_m / self.link.telescope_diameter_m
        return -10 * np.log10(np.exp(-(pointing_error_rad / beam_div) ** 2))

    def detector_loss(self) -> float:
        """Detector efficiency loss."""
        return -10 * np.log10(self.link.detector_efficiency)

    def total_link_budget(self, zenith_angle_deg: float = 0) -> Dict:
        """Calculate complete link budget."""
        geo = self.geometric_loss(zenith_angle_deg)
        atm = self.atmospheric_loss(zenith_angle_deg)
        point = self.pointing_loss()
        det = self.detector_loss()

        return {
            "geometric_loss_db": geo,
            "atmospheric_loss_db": atm,
            "pointing_loss_db": point,
            "detector_loss_db": det,
            "total_loss_db": geo + atm + point + det,
            "zenith_angle_deg": zenith_angle_deg
        }

    def print_budget(self, zenith_angle_deg: float = 0):
        """Print link budget breakdown."""
        budget = self.total_link_budget(zenith_angle_deg)

        print(f"\nSatellite QKD Link Budget")
        print(f"Altitude: {self.link.altitude_km} km")
        print(f"Zenith angle: {zenith_angle_deg}°")
        print("=" * 50)
        print(f"  Geometric loss:    {budget['geometric_loss_db']:.1f} dB")
        print(f"  Atmospheric loss:  {budget['atmospheric_loss_db']:.1f} dB")
        print(f"  Pointing loss:     {budget['pointing_loss_db']:.1f} dB")
        print(f"  Detector loss:     {budget['detector_loss_db']:.1f} dB")
        print("-" * 50)
        print(f"  TOTAL LOSS:        {budget['total_loss_db']:.1f} dB")

# Example link budget
link = SatelliteLink(
    altitude_km=500,
    telescope_diameter_m=0.3,  # 30 cm
    wavelength_nm=850,
    pointing_accuracy_urad=1.0,
    detector_efficiency=0.5
)

budget = SatelliteLinkBudget(link)

print("Link Budget Analysis")
print("=" * 60)
for angle in [0, 30, 60]:
    budget.print_budget(angle)
```
