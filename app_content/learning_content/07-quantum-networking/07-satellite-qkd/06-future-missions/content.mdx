@dataclass

    country: str
    status: MissionStatus
    launch_year: Optional[int]
    orbit: str
    primary_objectives: List[str]
    key_technologies: List[str]
    significance: str

        """Group missions by status."""
        groups = {status: [] for status in MissionStatus}
        for mission in self.missions:
            groups[mission.status].append(mission)
        return groups

    def print_missions(self):
        """Print mission summaries."""
        print("Future Satellite QKD Missions")
        print("=" * 70)

        for mission in self.missions:
            print(f"\n{mission.name} ({mission.country})")
            print("-" * 50)
            print(f"  Status: {mission.status.value}")
            if mission.launch_year:
                print(f"  Expected: ~{mission.launch_year}")
            print(f"  Orbit: {mission.orbit}")
            print(f"  Objectives:")
            for obj in mission.primary_objectives[:2]:
                print(f"    • {obj}")
            print(f"  Significance: {mission.significance}")

                    "Integration with quantum computers",
                    "Standardized global infrastructure",
                    "Commercial commodity service"

        }

    def print_roadmap(self):
        """Print development roadmap."""
        print("\nSatellite QKD Development Roadmap")
        print("=" * 70)

        for timeframe, details in self.phases.items():
            print(f"\n{timeframe}: {details['phase'].upper()}")
            print("-" * 50)
            print("  Goals:")
            for goal in details['goals']:
                print(f"    • {goal}")
            print(f"  Key missions: {', '.join(details['key_missions'])}")
