```python
import numpy as np
from dataclasses import dataclass
from typing import List, Dict, Tuple

@dataclass
class SatelliteConstellation:
    """Satellite constellation parameters."""
    name: str
    n_satellites: int
    n_planes: int
    altitude_km: float
    inclination_deg: float
    coverage_type: str

class ConstellationDesign:
    """
    Design satellite constellations for global QKD.
    """

    def __init__(self):
        self.designs = self._define_designs()

    def _define_designs(self) -> List[SatelliteConstellation]:
        """Define constellation designs."""
        return [
            SatelliteConstellation(
                name="Polar Coverage",
                n_satellites=12,
                n_planes=3,
                altitude_km=600,
                inclination_deg=90,
                coverage_type="Global including poles"
            ),
            SatelliteConstellation(
                name="Walker Delta",
                n_satellites=24,
                n_planes=6,
                altitude_km=800,
                inclination_deg=55,
                coverage_type="Mid-latitude optimized"
            ),
            SatelliteConstellation(
                name="Sun-Synchronous",
                n_satellites=8,
                n_planes=2,
                altitude_km=500,
                inclination_deg=97,
                coverage_type="Dawn-dusk, consistent lighting"
            ),
            SatelliteConstellation(
                name="Global Dense",
                n_satellites=48,
                n_planes=8,
                altitude_km=550,
                inclination_deg=70,
                coverage_type="Near-continuous global"
            )
        ]

    def calculate_coverage(self, constellation: SatelliteConstellation) -> Dict:
        """Calculate coverage metrics."""
        # Simplified coverage calculation
        sats_per_plane = constellation.n_satellites // constellation.n_planes

        # Ground track spacing
        track_spacing_deg = 360 / constellation.n_planes

        # Coverage circle radius (simplified)
        coverage_angle = np.degrees(np.arccos(
            6371 / (6371 + constellation.altitude_km)
        ))

        # Revisit time estimate
        orbital_period_min = 2 * np.pi * np.sqrt(
            (6371 + constellation.altitude_km) ** 3 / 398600
        ) / 60

        revisit_time_min = orbital_period_min / sats_per_plane

        return {
            "coverage_radius_deg": coverage_angle,
            "track_spacing_deg": track_spacing_deg,
            "orbital_period_min": orbital_period_min,
            "revisit_time_min": revisit_time_min,
            "coverage_overlap": coverage_angle * 2 > track_spacing_deg
        }

    def print_designs(self):
        """Print constellation designs."""
        print("Satellite Constellation Designs for Global QKD")
        print("=" * 70)

        for design in self.designs:
            coverage = self.calculate_coverage(design)

            print(f"\n{design.name}")
            print("-" * 50)
            print(f"  Satellites: {design.n_satellites} in {design.n_planes} planes")
            print(f"  Altitude: {design.altitude_km} km")
            print(f"  Inclination: {design.inclination_deg}°")
            print(f"  Coverage: {design.coverage_type}")
            print(f"  Orbital period: {coverage['orbital_period_min']:.0f} min")
            print(f"  Coverage radius: {coverage['coverage_radius_deg']:.0f}°")
            print(f"  Revisit time: {coverage['revisit_time_min']:.0f} min")

# Display designs
designs = ConstellationDesign()
designs.print_designs()
```

## Inter-Satellite Links

```python
class InterSatelliteLinks:
    """
    Model inter-satellite quantum links.
    """

    def __init__(self):
        self.link_types = self._define_link_types()

    def _define_link_types(self) -> Dict:
        """Define inter-satellite link types."""
        return {
            "optical_classical": {
                "name": "Classical Optical ISL",
                "data_rate": "Gbps",
                "purpose": "Timing, coordination, classical data",
                "technology": "Mature (laser comm)",
                "quantum_use": "Support only"
            },
            "quantum_isl": {
                "name": "Quantum ISL",
                "data_rate": "kbps (key)",
                "purpose": "Entanglement distribution between satellites",
                "technology": "Research phase",
                "quantum_use": "Direct quantum link"
            },
            "entanglement_relay": {
                "name": "Entanglement Relay",
                "data_rate": "Pairs/second",
                "purpose": "Extend entanglement via swapping",
                "technology": "Requires space-based BSM",
                "quantum_use": "Network backbone"
            }
        }

    def calculate_isl_loss(
        self,
        distance_km: float,
        telescope_diameter_m: float = 0.15,
        wavelength_nm: float = 1550
    ) -> float:
        """Calculate inter-satellite link loss (no atmosphere)."""
        wavelength_m = wavelength_nm * 1e-9

        # Diffraction-limited beam divergence
        beam_div = 1.22 * wavelength_m / telescope_diameter_m

        # Spot size at receiver
        spot_radius = distance_km * 1000 * beam_div

        # Collection efficiency
        collect_area = np.pi * (telescope_diameter_m / 2) ** 2
        spot_area = np.pi * spot_radius ** 2

        efficiency = collect_area / spot_area
        loss_db = -10 * np.log10(efficiency)

        return loss_db

    def print_analysis(self):
        """Print ISL analysis."""
        print("\nInter-Satellite Link Analysis")
        print("=" * 60)

        print("\nLink Types:")
        for link_id, link in self.link_types.items():
            print(f"\n  {link['name']}")
            print(f"    Data rate: {link['data_rate']}")
            print(f"    Purpose: {link['purpose']}")
            print(f"    Technology: {link['technology']}")

        print("\n\nISL Loss vs Distance (no atmosphere):")
        print("-" * 40)
        for dist in [100, 500, 1000, 2000, 5000]:
            loss = self.calculate_isl_loss(dist)
            print(f"  {dist:5} km: {loss:.1f} dB")

# Analyze ISL
isl = InterSatelliteLinks()
isl.print_analysis()
```

```python
class HybridNetwork:
    """
    Model hybrid satellite-terrestrial quantum network.
    """

    def __init__(self):
        self.segments = self._define_segments()

    def _define_segments(self) -> Dict:
        """Define network segments."""
        return {
            "metropolitan": {
                "type": "Fiber QKD",
                "distance": "0-100 km",
                "technology": "Direct fiber or trusted nodes",
                "availability": "24/7",
                "key_rate": "Mbps capable"
            },
            "regional": {
                "type": "Fiber + Satellite",
                "distance": "100-500 km",
                "technology": "Fiber backbone + satellite backup",
                "availability": "High (redundant)",
                "key_rate": "kbps-Mbps"
            },
            "continental": {
                "type": "Satellite primary",
                "distance": "500-3000 km",
                "technology": "LEO satellite constellation",
                "availability": "Weather dependent",
                "key_rate": "kbps"
            },
            "intercontinental": {
                "type": "Satellite only",
                "distance": ">3000 km",
                "technology": "Satellite relay (trusted or swapping)",
                "availability": "Limited passes",
                "key_rate": "100s bps"
            }
        }

    def optimal_path(self, distance_km: float) -> str:
        """Determine optimal path for given distance."""
        if distance_km < 100:
            return "metropolitan"
        elif distance_km < 500:
            return "regional"
        elif distance_km < 3000:
            return "continental"
        else:
            return "intercontinental"

    def print_architecture(self):
        """Print hybrid network architecture."""
        print("\nHybrid Satellite-Terrestrial Network")
        print("=" * 60)

        for segment, details in self.segments.items():
            print(f"\n{segment.upper()}")
            print("-" * 40)
            print(f"  Type: {details['type']}")
            print(f"  Distance: {details['distance']}")
            print(f"  Technology: {details['technology']}")
            print(f"  Availability: {details['availability']}")
            print(f"  Key rate: {details['key_rate']}")
```
