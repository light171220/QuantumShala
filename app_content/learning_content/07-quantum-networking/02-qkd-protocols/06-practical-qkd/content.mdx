from dataclasses import dataclass
from typing import Dict, List, Optional
from enum import Enum
import numpy as np

class DetectorType(Enum):
    APD = "Avalanche Photodiode"
    SNSPD = "Superconducting Nanowire"
    SSPD = "Superconducting Single-Photon"

class SourceType(Enum):
    WCP = "Weak Coherent Pulse"
    HSPS = "Heralded Single-Photon"
    QD = "Quantum Dot"

@dataclass
class LaserSource:
    """QKD laser source specifications."""
    source_type: SourceType
    wavelength_nm: float
    pulse_width_ps: float
    repetition_rate_mhz: float
    mean_photon_number: float  # μ for WCP
    extinction_ratio_db: float
    linewidth_khz: float

    def validate(self) -> List[str]:
        """Check specifications against requirements."""
        issues = []
        if self.extinction_ratio_db < 20:
            issues.append("Extinction ratio should be >20 dB for security")
        if self.mean_photon_number > 1 and self.source_type == SourceType.WCP:
            issues.append("Mean photon number >1 vulnerable to PNS")
        return issues

@dataclass
class SinglePhotonDetector:
    """Single-photon detector specifications."""
    detector_type: DetectorType
    efficiency: float
    dark_count_rate_hz: float
    timing_jitter_ps: float
    dead_time_ns: float
    operating_temp_k: float
    afterpulse_prob: float

    def calculate_max_count_rate(self) -> float:
        """Maximum detection rate considering dead time."""
        return 1e9 / (self.dead_time_ns + 10)  # Buffer

    def expected_snr(self, signal_rate_hz: float,
                     background_hz: float = 0) -> float:
        """Calculate signal-to-noise ratio."""
        noise = self.dark_count_rate_hz + background_hz
        return signal_rate_hz / (signal_rate_hz + noise)

@dataclass
class QKDSystemSpecs:
    """Complete QKD system specifications."""
    source: LaserSource
    detector: SinglePhotonDetector
    fiber_loss_db_km: float = 0.2
    system_loss_db: float = 3.0
    clock_rate_mhz: float = 100

    def max_distance_km(self, min_detection_prob: float = 1e-6) -> float:
        """Calculate maximum operational distance."""
        total_system_loss_db = -10 * np.log10(min_detection_prob)
        fiber_budget_db = total_system_loss_db - self.system_loss_db
        return fiber_budget_db / self.fiber_loss_db_km

    def key_rate_at_distance(self, distance_km: float,
                            qber: float = 0.03) -> float:
        """Estimate secret key rate at given distance."""
        # Channel transmission
        channel_loss_db = self.fiber_loss_db_km * distance_km
        eta_channel = 10 ** (-channel_loss_db / 10)
        eta_system = 10 ** (-self.system_loss_db / 10)

        # Detection probability
        p_detect = (self.source.mean_photon_number *
                   eta_channel * eta_system *
                   self.detector.efficiency)

        # Sifting efficiency
        sift_eff = 0.5

        # Secret fraction (simplified)
        h = lambda p: -p*np.log2(p)-(1-p)*np.log2(1-p) if 0<p<1 else 0
        secret_frac = max(0, 1 - 2*h(qber))

        # Key rate
        raw_rate = self.clock_rate_mhz * 1e6 * p_detect
        return raw_rate * sift_eff * secret_frac

# Example system

class QKDPerformanceAnalysis:
    """Analyze QKD system performance."""

    def __init__(self, system: QKDSystemSpecs):
        self.system = system

    def full_analysis(self, distances: List[float]) -> Dict:
        """Perform full performance analysis."""
        results = {
            'distances_km': distances,
            'key_rates_bps': [],
            'detection_probs': [],
            'expected_qber': [],
            'link_budget_db': []
        }

        for d in distances:
            # Link budget
            channel_loss = self.system.fiber_loss_db_km * d
            total_loss = channel_loss + self.system.system_loss_db
            results['link_budget_db'].append(total_loss)

            # Detection probability
            eta = 10 ** (-total_loss / 10) * self.system.detector.efficiency
            mu = self.system.source.mean_photon_number
            p_detect = mu * eta
            results['detection_probs'].append(p_detect)

            # QBER estimation
            dark_rate = self.system.detector.dark_count_rate_hz
            signal_rate = self.system.clock_rate_mhz * 1e6 * p_detect
            qber = dark_rate / (2 * (signal_rate + dark_rate)) + 0.01  # 1% optical errors
            results['expected_qber'].append(min(0.5, qber))

            # Key rate
            key_rate = self.system.key_rate_at_distance(d, qber)
            results['key_rates_bps'].append(max(0, key_rate))

        return results

    def find_operational_range(self, min_key_rate_bps: float = 100) -> float:
        """Find maximum distance meeting minimum key rate."""
        for d in np.arange(0, 500, 1):
            rate = self.system.key_rate_at_distance(d)
            if rate < min_key_rate_bps:
                return d - 1
        return 500

class DecoyStateAnalysis:
    """Analyze decoy-state QKD performance."""

    def __init__(self, mu: float = 0.5, nu: float = 0.1):
        """
        Args:
            mu: Signal state mean photon number
            nu: Decoy state mean photon number
        """
        self.mu = mu
        self.nu = nu

    def estimate_single_photon_yield(self,
                                     Q_mu: float,
                                     Q_nu: float) -> float:
        """
        Estimate single-photon yield Y_1 using decoy states.

        Decoy method removes PNS vulnerability.
        """
        # Simplified decoy analysis
        # Full analysis requires vacuum decoy too
        e_mu = np.exp(-self.mu)
        e_nu = np.exp(-self.nu)

        # Lower bound on Y_1
        Y_1_lower = (Q_mu * e_mu - Q_nu * e_nu * (self.mu / self.nu) ** 2) / (
            self.mu * e_mu - self.nu * e_nu * (self.mu / self.nu)
        )

        return max(0, Y_1_lower)

    def decoy_key_rate(self, distance_km: float,
                       detector_efficiency: float = 0.1,
                       dark_count: float = 1e-6) -> Dict:
        """Calculate key rate with decoy-state protocol."""
        # Channel transmission
        eta = 10 ** (-0.2 * distance_km / 10) * detector_efficiency

        # Gains
        Q_mu = 1 - np.exp(-self.mu * eta) + dark_count
        Q_nu = 1 - np.exp(-self.nu * eta) + dark_count

        # Single-photon yield estimate
        Y_1 = self.estimate_single_photon_yield(Q_mu, Q_nu)

        # Single-photon fraction
        p1 = self.mu * np.exp(-self.mu)

        # Gain from single photons
        Q_1 = Y_1 * p1

        # Secure key rate (simplified)
        h = lambda p: -p*np.log2(p)-(1-p)*np.log2(1-p) if 0<p<1 else 0
        e_1 = 0.02  # Assume 2% single-photon QBER

        key_rate = Q_1 * (1 - h(e_1)) - Q_mu * h(e_1)

        return {
            'distance_km': distance_km,
            'Q_mu': Q_mu,
            'Q_nu': Q_nu,
            'Y_1': Y_1,
            'key_rate': max(0, key_rate),
            'improvement_over_no_decoy': 'Significant at long distance'
        }

class QKDSecurityEvaluation:
    """Framework for QKD system security evaluation."""

    def __init__(self):
        self.evaluation_criteria = {
            'protocol_security': {
                'weight': 0.25,
                'checks': [
                    'Security proof published and peer-reviewed',
                    'Finite-key analysis included',
                    'Attack model clearly stated',
                    'Composable security achieved'
                ]
            },
            'implementation_security': {
                'weight': 0.30,
                'checks': [
                    'Side-channel attacks mitigated',
                    'Source characterization complete',
                    'Detector characterization complete',
                    'Timing attacks addressed'
                ]
            },
            'key_management': {
                'weight': 0.20,
                'checks': [
                    'Secure key storage',
                    'Key lifecycle management',
                    'Emergency key destruction',
                    'Audit logging'
                ]
            },
            'operational_security': {
                'weight': 0.15,
                'checks': [
                    'Physical security of endpoints',
                    'Tamper detection',
                    'Personnel security',
                    'Incident response procedures'
                ]
            },
            'classical_components': {
                'weight': 0.10,
                'checks': [
                    'Authenticated classical channel',
                    'Post-processing security',
                    'Random number quality',
                    'Software integrity'
                ]
            }
        }

    def evaluate_system(self, checklist_results: Dict) -> Dict:
        """Evaluate QKD system against criteria."""
        scores = {}
        total_score = 0

        for category, info in self.evaluation_criteria.items():
            checks = info['checks']
            passed = sum(1 for c in checks
                        if checklist_results.get(c, False))
            category_score = passed / len(checks)
            scores[category] = category_score
            total_score += category_score * info['weight']

        return {
            'category_scores': scores,
            'overall_score': total_score,
            'certification_ready': total_score >= 0.8,
            'recommendations': self._generate_recommendations(scores)
        }

    def _generate_recommendations(self, scores: Dict) -> List[str]:
        """Generate improvement recommendations."""
        recommendations = []
        for category, score in scores.items():
            if score < 0.8:
                recommendations.append(
                    f"Improve {category}: currently at {score*100:.0f}%"
                )
        return recommendations

## Summary

Practical QKD systems require careful integration of multiple components:

- **Hardware**: Sources, detectors, modulators, and QRNGs with stringent specifications
- **Performance**: Trade-offs between distance, key rate, and cost
- **Commercial**: Multiple vendors with various technology approaches
- **Standards**: ETSI, ITU-T, and ISO providing interoperability and security frameworks
- **Deployment**: Infrastructure assessment, integration planning, and security evaluation

Understanding these practical aspects is essential for successful QKD implementation.
