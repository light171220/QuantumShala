# Attack Vectors in Quantum Networks

## Introduction

While QKD offers information-theoretic security in theory, real implementations face numerous attack vectors. Understanding these vulnerabilities is essential for building secure quantum networks.

import numpy as np
from dataclasses import dataclass
from typing import List, Dict, Tuple, Optional
from enum import Enum

class AttackType(Enum):
    """Types of QKD attacks."""
    THEORETICAL = "theoretical"
    IMPLEMENTATION = "implementation"
    SIDE_CHANNEL = "side_channel"
    NETWORK = "network"

class Severity(Enum):
    """Attack severity levels."""
    LOW = 1
    MEDIUM = 2
    HIGH = 3
    CRITICAL = 4

@dataclass
class Attack:
    """Attack vector specification."""
    name: str
    attack_type: AttackType
    severity: Severity
    description: str
    mechanism: str
    countermeasures: List[str]
    known_success: bool

class AttackCatalog:
    """
    Catalog of QKD attack vectors.

    def __init__(self):

    def _define_attacks(self) -> Dict[str, Attack]:

                         "Then controls when detectors click using trigger pulses.",
                countermeasures=[
                    "Monitor detector current",

                    "Use measurement-device-independent QKD",
                    "Random detector efficiency testing"
                ],
                known_success=True
            ),
            "time_shift": Attack(
                name="Time-Shift Attack",
                attack_type=AttackType.IMPLEMENTATION,
                severity=Severity.HIGH,
                description="Exploit timing differences in detector efficiency",
                mechanism="Detectors have different efficiencies at different times. "

                countermeasures=[
                    "Equalize detector timing response",

                    "Active timing monitoring"
                ],

            ),
            "trojan_horse": Attack(

        """Get attacks by type."""
        return [a for a in self.attacks.values()
                if a.attack_type == attack_type]

    def get_critical(self) -> List[Attack]:
        """Get critical severity attacks."""
        return [a for a in self.attacks.values()
                if a.severity == Severity.CRITICAL]

    def print_summary(self):
        """Print attack catalog summary."""
        print("QKD Attack Vector Catalog")
        print("=" * 60)

        by_type = {}
        for attack in self.attacks.values():
            t = attack.attack_type.value
            if t not in by_type:
                by_type[t] = []
            by_type[t].append(attack)

        for attack_type, attacks in by_type.items():
            print(f"\n{attack_type.upper()} ATTACKS:")
            for a in attacks:
                status = "DEMONSTRATED" if a.known_success else "theoretical"
                print(f"  • {a.name} [{a.severity.name}] - {status}")

# Display catalog
catalog = AttackCatalog()
catalog.print_summary()

class DetectorBlindingSimulator:
    """
    Simulate detector blinding attack.
    """

    def __init__(self):
        self.detector_threshold = 0.5  # Photon detection threshold
        self.blinding_power = 0.0
        self.is_blinded = False

    def normal_detection(self, photon_present: bool) -> bool:
        """Normal single-photon detection."""
        if photon_present:
            # APD has ~10% efficiency typically
            return np.random.random() < 0.1
        return False

    def apply_blinding(self, cw_power: float):
        """Apply continuous wave blinding light."""
        self.blinding_power = cw_power
        # Detector transitions to linear mode above threshold
        self.is_blinded = cw_power > 1.0  # Arbitrary threshold

    def blinded_detection(
        self,
        trigger_power: float,
        trigger_present: bool
    ) -> bool:
        """Detection under blinding attack."""
        if not self.is_blinded:
            return self.normal_detection(trigger_present)

        # In linear mode, detector only responds to bright pulses
        if trigger_present and trigger_power > self.detector_threshold:
            return True  # Eve controls this!
        return False

    def simulate_attack(self, n_rounds: int = 100) -> Dict:
        """Simulate complete blinding attack."""
        results = {
            "normal_detections": 0,
            "eve_controlled": 0,
            "eve_errors": 0
        }

        # Phase 1: Normal operation
        for _ in range(n_rounds):
            if self.normal_detection(True):
                results["normal_detections"] += 1

        # Phase 2: Under attack
        self.apply_blinding(2.0)  # Blind detector

        for _ in range(n_rounds):
            # Eve decides when Bob detects
            eve_triggers = np.random.random() < 0.5
            if self.blinded_detection(1.0, eve_triggers):
                results["eve_controlled"] += 1

        return results

class SideChannelAnalyzer:
    """
    Analyze side-channel vulnerabilities.
    """

    def __init__(self):
        self.timing_data = []
        self.power_data = []

    def measure_timing(
        self,
        operation: str,
        basis: int,
        bit: int
    ) -> float:
        """Simulate timing measurement of operation."""
        # Simulated timing variations based on operation
        base_time = 100.0  # microseconds

        # Vulnerable implementation has timing differences
        if operation == "modulation":
            # Different bases might have different times
            time = base_time + (basis * 5) + (bit * 2)
        elif operation == "detection":
            time = base_time + np.random.normal(0, 1)
        else:
            time = base_time

        # Add realistic noise
        time += np.random.normal(0, 0.5)

        self.timing_data.append({
            "operation": operation,
            "basis": basis,
            "bit": bit,
            "time": time
        })

        return time

    def analyze_timing_leak(self) -> Dict:
        """Analyze timing data for information leakage."""
        if not self.timing_data:
            return {"leak_detected": False}

        # Group by basis
        basis_0_times = [d["time"] for d in self.timing_data if d["basis"] == 0]
        basis_1_times = [d["time"] for d in self.timing_data if d["basis"] == 1]

        if basis_0_times and basis_1_times:
            mean_diff = abs(np.mean(basis_0_times) - np.mean(basis_1_times))
            std = np.std(basis_0_times + basis_1_times)

            leak_detected = mean_diff > 2 * std  # Statistical significance

            return {
                "leak_detected": leak_detected,
                "mean_diff": mean_diff,
                "std": std,
                "distinguishability": mean_diff / std if std > 0 else 0
            }

        return {"leak_detected": False}

    def simulate_attack(self, n_samples: int = 1000) -> float:
        """Simulate timing attack to recover basis choices."""
        correct_guesses = 0

        for _ in range(n_samples):
            true_basis = np.random.randint(0, 2)
            true_bit = np.random.randint(0, 2)

            timing = self.measure_timing("modulation", true_basis, true_bit)

            # Eve's guess based on timing
            # Threshold learned from previous observations
            guessed_basis = 1 if timing > 103 else 0

            if guessed_basis == true_basis:
                correct_guesses += 1

        return correct_guesses / n_samples

class NetworkAttackSimulator:
    """
    Simulate network-level attacks on QKD infrastructure.
    """

    def __init__(self, network_nodes: List[str]):
        self.nodes = network_nodes
        self.compromised_nodes = set()
        self.blocked_links = set()

    def mitm_attack(
        self,
        alice: str,
        bob: str
    ) -> Dict:
        """Simulate man-in-the-middle attack."""
        # Without authentication, MITM is trivial
        return {
            "attack": "Man-in-the-Middle",
            "success_without_auth": True,
            "description": "Eve intercepts all quantum and classical channels, "
                          "runs separate QKD with Alice and Bob",
            "prevention": "Classical channel authentication required"
        }

    def trusted_node_compromise(
        self,
        compromised_node: str
    ) -> Dict:
        """Simulate trusted relay node compromise."""
        self.compromised_nodes.add(compromised_node)

        # Find all paths through this node
        affected_pairs = []
        for i, a in enumerate(self.nodes):
            for b in self.nodes[i+1:]:
                # Simplified: assume linear topology
                if compromised_node in self.nodes:
                    idx_c = self.nodes.index(compromised_node)
                    idx_a = self.nodes.index(a)
                    idx_b = self.nodes.index(b)
                    if idx_a < idx_c < idx_b or idx_b < idx_c < idx_a:
                        affected_pairs.append((a, b))

        return {
            "attack": "Trusted Node Compromise",
            "compromised": compromised_node,
            "affected_paths": len(affected_pairs),
            "description": "All keys relayed through this node are compromised",
            "prevention": "End-to-end security via quantum repeaters"
        }

    def dos_attack(
        self,
        target_link: Tuple[str, str]
    ) -> Dict:
        """Simulate denial of service attack."""
        self.blocked_links.add(target_link)

        return {
            "attack": "Denial of Service",
            "target": target_link,
            "impact": "QKD link disabled, no new keys generated",
            "description": "Physical disruption or channel jamming",
            "prevention": "Redundant paths, key reserves"
        }

    def routing_attack(
        self,
        malicious_routes: Dict[str, str]
    ) -> Dict:
        """Simulate routing manipulation attack."""
        return {
            "attack": "Routing Attack",
            "description": "Redirect traffic through adversary-controlled nodes",
            "manipulated_routes": malicious_routes,
            "prevention": "Authenticated routing protocols, path verification"
        }
