import numpy as np
from dataclasses import dataclass, field
from typing import List, Dict, Optional
from enum import Enum
from datetime import datetime

class StandardsBody(Enum):
    """Standards organizations."""
    ETSI = "ETSI"
    ISO = "ISO"
    ITU = "ITU"
    NIST = "NIST"
    BSI = "BSI"

class StandardStatus(Enum):
    """Standard document status."""
    DRAFT = "draft"
    PUBLISHED = "published"
    UNDER_REVISION = "under_revision"
    WITHDRAWN = "withdrawn"

@dataclass
class Standard:
    """Security standard specification."""
    identifier: str
    title: str
    body: StandardsBody
    status: StandardStatus
    scope: str
    requirements: List[str]
    publication_date: Optional[str] = None

            )
        }

    def get_by_body(self, body: StandardsBody) -> List[Standard]:
        """Get standards from specific body."""
        return [s for s in self.standards.values() if s.body == body]

    def check_compliance(
        self,
        implemented: Dict[str, List[str]]
    ) -> Dict:
        """Check compliance against standards."""
        results = {}

        for std_id, standard in self.standards.items():
            if std_id in implemented:
                impl = implemented[std_id]
                met = sum(1 for r in standard.requirements if r in impl)
                total = len(standard.requirements)
                results[std_id] = {
                    "standard": standard.identifier,
                    "requirements_met": met,
                    "requirements_total": total,
                    "compliance": met / total if total > 0 else 0
                }

        return results

    def print_framework(self):
        """Print standards framework."""
        print("QKD Security Standards Framework")
        print("=" * 60)

        by_body = {}
        for std in self.standards.values():
            body = std.body.value
            if body not in by_body:
                by_body[body] = []
            by_body[body].append(std)

        for body, stds in by_body.items():
            print(f"\n{body}:")
            for s in stds:
                print(f"  • {s.identifier}: {s.title}")
                print(f"    Status: {s.status.value}")

class CertificationLevel(Enum):
    """Certification assurance levels."""
    BASIC = 1
    SUBSTANTIAL = 2
    HIGH = 3

@dataclass
class CertificationRequirement:
    """Certification requirement."""
    category: str
    requirement: str
    level: CertificationLevel
    evidence_needed: List[str]

class CertificationFramework:
    """
    QKD certification framework.
    """

    def __init__(self):
        self.requirements = self._define_requirements()

    def _define_requirements(self) -> List[CertificationRequirement]:
        """Define certification requirements."""
        return [
            CertificationRequirement(
                category="Security Proofs",
                requirement="Composable security proof for implemented protocol",
                level=CertificationLevel.HIGH,
                evidence_needed=[
                    "Mathematical security proof",
                    "Finite-key analysis",
                    "Parameter estimation methodology"
                ]
            ),
            CertificationRequirement(
                category="Implementation Security",
                requirement="Protection against known implementation attacks",
                level=CertificationLevel.SUBSTANTIAL,
                evidence_needed=[
                    "Attack countermeasure documentation",
                    "Penetration testing results",
                    "Side-channel analysis report"
                ]
            ),
            CertificationRequirement(
                category="Random Number Generation",
                requirement="Cryptographic quality random numbers",
                level=CertificationLevel.HIGH,
                evidence_needed=[
                    "QRNG certification",
                    "Entropy source analysis",
                    "Statistical test results"
                ]
            ),
            CertificationRequirement(
                category="Key Management",
                requirement="Secure key lifecycle management",
                level=CertificationLevel.SUBSTANTIAL,
                evidence_needed=[
                    "Key management procedures",
                    "Secure deletion verification",
                    "Audit logging evidence"
                ]
            ),
            CertificationRequirement(
                category="Physical Security",
                requirement="Tamper-evident and tamper-resistant design",
                level=CertificationLevel.HIGH,
                evidence_needed=[
                    "Physical security design docs",
                    "Tamper testing results",
                    "Environmental testing"
                ]
            ),
            CertificationRequirement(
                category="Authentication",
                requirement="Information-theoretic or computational authentication",
                level=CertificationLevel.SUBSTANTIAL,
                evidence_needed=[
                    "Authentication protocol spec",
                    "Key bootstrapping procedure",
                    "Security analysis"
                ]
            )
        ]

    def assess_readiness(
        self,
        evidence: Dict[str, List[str]]
    ) -> Dict:
        """Assess certification readiness."""
        results = {
            "ready": [],
            "partial": [],
            "missing": [],
            "overall_score": 0
        }

        total_requirements = len(self.requirements)
        met = 0

        for req in self.requirements:
            category_evidence = evidence.get(req.category, [])
            needed = set(req.evidence_needed)
            provided = set(category_evidence)

            overlap = len(needed & provided)
            total = len(needed)

            if overlap == total:
                results["ready"].append(req.category)
                met += 1
            elif overlap > 0:
                results["partial"].append({
                    "category": req.category,
                    "provided": overlap,
                    "needed": total
                })
                met += 0.5
            else:
                results["missing"].append(req.category)

        results["overall_score"] = met / total_requirements

        return results

    def print_requirements(self):
        """Print certification requirements."""
        print("\n\nCertification Requirements")
        print("=" * 60)

        for req in self.requirements:
            print(f"\n{req.category} [{req.level.name}]:")
            print(f"  {req.requirement}")
            print(f"  Evidence needed:")
            for e in req.evidence_needed[:2]:
                print(f"    • {e}")

class SecurityEvaluation:
    """
    Security evaluation methodology for QKD.
    """

    def __init__(self):
        self.test_categories = self._define_tests()

    def _define_tests(self) -> Dict:
        """Define security test categories."""
        return {
            "optical_characterization": {
                "name": "Optical Component Characterization",
                "tests": [
                    "Source characterization (photon statistics)",
                    "Detector efficiency and timing",
                    "Wavelength and spectral analysis",
                    "Polarization extinction ratio"
                ],
                "purpose": "Verify optical components meet specifications"
            },
            "side_channel": {
                "name": "Side-Channel Analysis",
                "tests": [
                    "Timing analysis of operations",
                    "Power consumption profiling",
                    "Electromagnetic emission testing",
                    "Optical side-channel detection"
                ],
                "purpose": "Identify information leakage channels"
            },
            "attack_resistance": {
                "name": "Attack Resistance Testing",
                "tests": [
                    "Detector blinding attempt",
                    "Trojan horse injection",
                    "Time-shift exploitation",
                    "High-power laser damage"
                ],
                "purpose": "Verify countermeasure effectiveness"
            },
            "protocol_conformance": {
                "name": "Protocol Conformance Testing",
                "tests": [
                    "Parameter estimation accuracy",
                    "Privacy amplification correctness",
                    "Error correction verification",
                    "Authentication protocol testing"
                ],
                "purpose": "Verify correct protocol implementation"
            },
            "random_number": {
                "name": "Random Number Quality",
                "tests": [
                    "NIST SP 800-22 statistical tests",
                    "Entropy estimation",
                    "Bias detection",
                    "Correlation analysis"
                ],
                "purpose": "Verify cryptographic randomness quality"
            }
        }

    def evaluate(
        self,
        test_results: Dict[str, Dict]
    ) -> Dict:
        """Evaluate test results."""
        evaluation = {
            "categories": {},
            "passed": 0,
            "failed": 0,
            "overall": "PENDING"
        }

        for category, tests in self.test_categories.items():
            if category in test_results:
                results = test_results[category]
                passed = sum(1 for t in results.values() if t == "PASS")
                total = len(tests["tests"])

                evaluation["categories"][category] = {
                    "passed": passed,
                    "total": total,
                    "score": passed / total if total > 0 else 0
                }

                evaluation["passed"] += passed
                evaluation["failed"] += (total - passed)

        total_tests = evaluation["passed"] + evaluation["failed"]
        if total_tests > 0:
            pass_rate = evaluation["passed"] / total_tests
            if pass_rate >= 0.95:
                evaluation["overall"] = "PASS"
            elif pass_rate >= 0.80:
                evaluation["overall"] = "CONDITIONAL_PASS"
            else:
                evaluation["overall"] = "FAIL"

        return evaluation

    def generate_report(self, evaluation: Dict) -> str:
        """Generate evaluation report."""
        report = []
        report.append("Security Evaluation Report")
        report.append("=" * 50)
        report.append(f"\nOverall Result: {evaluation['overall']}")
        report.append(f"Tests Passed: {evaluation['passed']}")
        report.append(f"Tests Failed: {evaluation['failed']}")

        report.append("\nCategory Results:")
        for cat, results in evaluation["categories"].items():
            report.append(f"  {cat}: {results['passed']}/{results['total']} "
                         f"({results['score']:.0%})")

        return "\n".join(report)
