# Quantum Reinforcement Learning

Reinforcement learning trains agents to make decisions by interacting with environments. Quantum computing offers potential advantages through variational quantum policies, quantum exploration strategies, and quantum-enhanced value function approximation.

## Quantum Policy Networks

Variational quantum circuits can represent policy functions that map states to action probabilities.

<Definition>
**Quantum Policy**:

A quantum policy $\pi_\theta(a|s)$ uses a parameterized quantum circuit:

1. Encode state: $|0\rangle \rightarrow |\psi(s)\rangle$
2. Apply variational circuit: $|\psi(s)\rangle \rightarrow U(\theta)|\psi(s)\rangle$
3. Measure to get action probabilities: $\pi_\theta(a|s) = |\langle a|U(\theta)|\psi(s)\rangle|^2$
</Definition>

```python
import numpy as np

class QuantumPolicyNetwork:
    """Variational quantum circuit as a policy network."""

    def __init__(self, n_qubits: int, n_actions: int, n_layers: int):
        self.n_qubits = n_qubits
        self.n_actions = n_actions
        self.n_layers = n_layers
        self.dim = 2 ** n_qubits
        self.n_params = n_qubits * n_layers * 2

    def encode_state(self, state: np.ndarray) -> np.ndarray:
        """Encode environment state into quantum state."""
        qstate = np.zeros(self.dim, dtype=complex)
        qstate[0] = 1

        # Angle encoding
        for i, val in enumerate(state[:self.n_qubits]):
            theta = np.pi * np.tanh(val)
            c, s = np.cos(theta/2), np.sin(theta/2)
            new_state = np.zeros_like(qstate)
            for z in range(self.dim):
                z_flip = z ^ (1 << (self.n_qubits - 1 - i))
                new_state[z] = c * qstate[z] - s * qstate[z_flip]
                new_state[z_flip] = s * qstate[z] + c * qstate[z_flip]
            qstate = new_state

        return qstate

    def variational_circuit(self, state: np.ndarray,
                            params: np.ndarray) -> np.ndarray:
        """Apply variational circuit layers."""
        param_idx = 0

        for layer in range(self.n_layers):
            # Rotation layer
            for q in range(self.n_qubits):
                # RY rotation
                theta = params[param_idx]
                c, s = np.cos(theta/2), np.sin(theta/2)
                new_state = np.zeros_like(state)
                for z in range(self.dim):
                    z_flip = z ^ (1 << (self.n_qubits - 1 - q))
                    bit = (z >> (self.n_qubits - 1 - q)) & 1
                    new_state[z] = c * state[z] + (1-2*bit) * s * state[z_flip]
                state = new_state
                param_idx += 1

                # RZ rotation
                theta = params[param_idx]
                for z in range(self.dim):
                    bit = (z >> (self.n_qubits - 1 - q)) & 1
                    state[z] *= np.exp(1j*theta/2) if bit else np.exp(-1j*theta/2)
                param_idx += 1

            # Entangling layer
            for q in range(self.n_qubits - 1):
                new_state = np.zeros_like(state)
                for z in range(self.dim):
                    ctrl = (z >> (self.n_qubits - 1 - q)) & 1
                    z_new = z ^ (1 << (self.n_qubits-1-(q+1))) if ctrl else z
                    new_state[z_new] += state[z]
                state = new_state

        return state

    def get_action_probs(self, state: np.ndarray,
                         params: np.ndarray) -> np.ndarray:
        """Get probability distribution over actions."""
        qstate = self.encode_state(state)
        qstate = self.variational_circuit(qstate, params)

        # Measure to get action probabilities
        probs = np.zeros(self.n_actions)
        for z in range(self.dim):
            action = z % self.n_actions  # Map measurement to action
            probs[action] += np.abs(qstate[z])**2

        probs /= probs.sum() + 1e-10
        return probs

    def sample_action(self, state: np.ndarray,
                      params: np.ndarray) -> int:
        """Sample action from policy."""
        probs = self.get_action_probs(state, params)
        return np.random.choice(self.n_actions, p=probs)


# Example
print("Quantum Policy Network")
print("=" * 50)

np.random.seed(42)
policy = QuantumPolicyNetwork(n_qubits=3, n_actions=4, n_layers=2)
params = np.random.uniform(-np.pi, np.pi, policy.n_params)

state = np.array([0.5, -0.3, 0.8])
probs = policy.get_action_probs(state, params)

print(f"State: {state}")
print(f"Action probabilities: {probs.round(3)}")
print(f"Sampled action: {policy.sample_action(state, params)}")
```

## Quantum Policy Gradient (REINFORCE)

Train quantum policies using the policy gradient theorem.

```python
import numpy as np

class QuantumREINFORCE:
    """REINFORCE algorithm with quantum policy network."""

    def __init__(self, n_qubits: int, n_actions: int, n_layers: int,
                 learning_rate: float = 0.1, gamma: float = 0.99):
        self.policy = QuantumPolicyNetwork(n_qubits, n_actions, n_layers)
        self.lr = learning_rate
        self.gamma = gamma
        self.params = np.random.uniform(-np.pi, np.pi, self.policy.n_params)

    def compute_returns(self, rewards: list) -> np.ndarray:
        """Compute discounted returns."""
        returns = np.zeros(len(rewards))
        G = 0
        for t in reversed(range(len(rewards))):
            G = rewards[t] + self.gamma * G
            returns[t] = G
        return returns

    def policy_gradient(self, states: list, actions: list,
                        returns: np.ndarray) -> np.ndarray:
        """Compute policy gradient via parameter shift."""
        gradients = np.zeros_like(self.params)

        for i in range(len(self.params)):
            grad_i = 0

            for state, action, G in zip(states, actions, returns):
                # Parameter shift
                params_plus = self.params.copy()
                params_plus[i] += np.pi/2
                params_minus = self.params.copy()
                params_minus[i] -= np.pi/2

                prob_plus = self.policy.get_action_probs(state, params_plus)[action]
                prob_minus = self.policy.get_action_probs(state, params_minus)[action]

                # Gradient of log probability
                prob = self.policy.get_action_probs(state, self.params)[action]
                if prob > 1e-10:
                    grad_log_prob = (prob_plus - prob_minus) / (2 * prob)
                    grad_i += G * grad_log_prob

            gradients[i] = grad_i / len(states)

        return gradients

    def update(self, episode_data: list):
        """Update policy from episode."""
        states = [d[0] for d in episode_data]
        actions = [d[1] for d in episode_data]
        rewards = [d[2] for d in episode_data]

        returns = self.compute_returns(rewards)
        # Normalize returns
        returns = (returns - returns.mean()) / (returns.std() + 1e-10)

        gradients = self.policy_gradient(states, actions, returns)
        self.params += self.lr * gradients


class SimpleGridEnv:
    """Simple grid environment for testing."""

    def __init__(self, size: int = 4):
        self.size = size
        self.goal = (size-1, size-1)
        self.reset()

    def reset(self):
        self.pos = (0, 0)
        return self._get_state()

    def _get_state(self):
        return np.array([self.pos[0]/self.size, self.pos[1]/self.size,
                        (self.goal[0]-self.pos[0])/self.size])

    def step(self, action):
        # Actions: 0=up, 1=right, 2=down, 3=left
        moves = [(-1, 0), (0, 1), (1, 0), (0, -1)]
        dx, dy = moves[action]
        new_x = max(0, min(self.size-1, self.pos[0] + dx))
        new_y = max(0, min(self.size-1, self.pos[1] + dy))
        self.pos = (new_x, new_y)

        done = self.pos == self.goal
        reward = 1.0 if done else -0.01
        return self._get_state(), reward, done


# Training example
print("\nQuantum REINFORCE Training")
print("=" * 50)

np.random.seed(42)
env = SimpleGridEnv(size=4)
agent = QuantumREINFORCE(n_qubits=3, n_actions=4, n_layers=2,
                         learning_rate=0.3, gamma=0.99)

episode_rewards = []
for episode in range(50):
    state = env.reset()
    episode_data = []
    total_reward = 0

    for step in range(50):
        action = agent.policy.sample_action(state, agent.params)
        next_state, reward, done = env.step(action)
        episode_data.append((state, action, reward))
        total_reward += reward
        state = next_state

        if done:
            break

    agent.update(episode_data)
    episode_rewards.append(total_reward)

    if episode % 10 == 0:
        avg_reward = np.mean(episode_rewards[-10:])
        print(f"Episode {episode}: Avg Reward = {avg_reward:.2f}")

print(f"\nFinal 10-episode average: {np.mean(episode_rewards[-10:]):.2f}")
```

## Quantum Q-Learning

Approximate the Q-function using a variational quantum circuit.

```python
import numpy as np

class QuantumQNetwork:
    """Q-function approximation using quantum circuit."""

    def __init__(self, n_qubits: int, n_actions: int, n_layers: int):
        self.n_qubits = n_qubits
        self.n_actions = n_actions
        self.n_layers = n_layers
        self.dim = 2 ** n_qubits
        self.n_params = n_qubits * n_layers * 2 * n_actions

    def encode_state(self, state: np.ndarray) -> np.ndarray:
        """Encode state into quantum state."""
        qstate = np.zeros(self.dim, dtype=complex)
        qstate[0] = 1

        for i, val in enumerate(state[:self.n_qubits]):
            theta = np.pi * np.tanh(val)
            c, s = np.cos(theta/2), np.sin(theta/2)
            new_state = np.zeros_like(qstate)
            for z in range(self.dim):
                z_flip = z ^ (1 << (self.n_qubits - 1 - i))
                new_state[z] = c * qstate[z] - s * qstate[z_flip]
                new_state[z_flip] = s * qstate[z] + c * qstate[z_flip]
            qstate = new_state

        return qstate

    def circuit_for_action(self, qstate: np.ndarray,
                           params: np.ndarray) -> float:
        """Apply circuit and return Q-value (expectation)."""
        state = qstate.copy()
        param_idx = 0

        for layer in range(self.n_layers):
            for q in range(self.n_qubits):
                # RY
                theta = params[param_idx]
                c, s = np.cos(theta/2), np.sin(theta/2)
                new_state = np.zeros_like(state)
                for z in range(self.dim):
                    z_flip = z ^ (1 << (self.n_qubits - 1 - q))
                    bit = (z >> (self.n_qubits - 1 - q)) & 1
                    new_state[z] = c * state[z] + (1-2*bit) * s * state[z_flip]
                state = new_state
                param_idx += 1

                # RZ
                theta = params[param_idx]
                for z in range(self.dim):
                    bit = (z >> (self.n_qubits - 1 - q)) & 1
                    state[z] *= np.exp(1j*theta/2) if bit else np.exp(-1j*theta/2)
                param_idx += 1

        # Expectation of Z on first qubit
        exp_val = 0
        for z in range(self.dim):
            bit = (z >> (self.n_qubits - 1)) & 1
            exp_val += (1 - 2*bit) * np.abs(state[z])**2

        return exp_val

    def get_q_values(self, state: np.ndarray, params: np.ndarray) -> np.ndarray:
        """Get Q-values for all actions."""
        qstate = self.encode_state(state)
        params_per_action = len(params) // self.n_actions

        q_values = []
        for a in range(self.n_actions):
            action_params = params[a*params_per_action:(a+1)*params_per_action]
            q = self.circuit_for_action(qstate, action_params)
            q_values.append(q)

        return np.array(q_values)


class QuantumDQN:
    """DQN with quantum Q-network."""

    def __init__(self, n_qubits: int, n_actions: int, n_layers: int,
                 lr: float = 0.1, gamma: float = 0.99, epsilon: float = 0.1):
        self.q_network = QuantumQNetwork(n_qubits, n_actions, n_layers)
        self.n_actions = n_actions
        self.lr = lr
        self.gamma = gamma
        self.epsilon = epsilon
        self.params = np.random.uniform(-np.pi, np.pi, self.q_network.n_params)

    def select_action(self, state: np.ndarray) -> int:
        """Epsilon-greedy action selection."""
        if np.random.random() < self.epsilon:
            return np.random.randint(self.n_actions)
        q_values = self.q_network.get_q_values(state, self.params)
        return np.argmax(q_values)

    def update(self, state: np.ndarray, action: int,
               reward: float, next_state: np.ndarray, done: bool):
        """Update Q-network."""
        # Compute TD target
        if done:
            target = reward
        else:
            next_q = self.q_network.get_q_values(next_state, self.params)
            target = reward + self.gamma * np.max(next_q)

        # Current Q-value
        current_q = self.q_network.get_q_values(state, self.params)[action]

        # TD error
        td_error = target - current_q

        # Parameter shift gradient for the action's parameters
        params_per_action = len(self.params) // self.n_actions
        start_idx = action * params_per_action
        end_idx = start_idx + params_per_action

        for i in range(start_idx, end_idx):
            params_plus = self.params.copy()
            params_plus[i] += np.pi/2
            params_minus = self.params.copy()
            params_minus[i] -= np.pi/2

            q_plus = self.q_network.get_q_values(state, params_plus)[action]
            q_minus = self.q_network.get_q_values(state, params_minus)[action]

            gradient = (q_plus - q_minus) / 2
            self.params[i] += self.lr * td_error * gradient


# Example
print("\nQuantum Q-Learning")
print("=" * 50)

np.random.seed(42)
env = SimpleGridEnv(size=3)
agent = QuantumDQN(n_qubits=3, n_actions=4, n_layers=1,
                   lr=0.2, gamma=0.95, epsilon=0.2)

episode_rewards = []
for episode in range(30):
    state = env.reset()
    total_reward = 0

    for step in range(30):
        action = agent.select_action(state)
        next_state, reward, done = env.step(action)
        agent.update(state, action, reward, next_state, done)
        total_reward += reward
        state = next_state

        if done:
            break

    episode_rewards.append(total_reward)

    if episode % 10 == 0:
        avg = np.mean(episode_rewards[-10:]) if episode >= 10 else np.mean(episode_rewards)
        print(f"Episode {episode}: Avg Reward = {avg:.2f}")
```

## Quantum Actor-Critic

Combine quantum policy (actor) with quantum value function (critic).

```python
import numpy as np

class QuantumActorCritic:
    """Quantum actor-critic with separate policy and value networks."""

    def __init__(self, n_qubits: int, n_actions: int, n_layers: int,
                 actor_lr: float = 0.1, critic_lr: float = 0.1, gamma: float = 0.99):
        self.actor = QuantumPolicyNetwork(n_qubits, n_actions, n_layers)
        self.n_qubits = n_qubits
        self.n_layers = n_layers
        self.dim = 2 ** n_qubits
        self.gamma = gamma
        self.actor_lr = actor_lr
        self.critic_lr = critic_lr

        self.actor_params = np.random.uniform(-np.pi, np.pi, self.actor.n_params)
        self.critic_params = np.random.uniform(-np.pi, np.pi, n_qubits * n_layers * 2)

    def get_value(self, state: np.ndarray) -> float:
        """Compute state value using critic."""
        qstate = np.zeros(self.dim, dtype=complex)
        qstate[0] = 1

        # Encode state
        for i, val in enumerate(state[:self.n_qubits]):
            theta = np.pi * np.tanh(val)
            c, s = np.cos(theta/2), np.sin(theta/2)
            new_state = np.zeros_like(qstate)
            for z in range(self.dim):
                z_flip = z ^ (1 << (self.n_qubits - 1 - i))
                new_state[z] = c * qstate[z] - s * qstate[z_flip]
                new_state[z_flip] = s * qstate[z] + c * qstate[z_flip]
            qstate = new_state

        # Variational circuit
        param_idx = 0
        for _ in range(self.n_layers):
            for q in range(self.n_qubits):
                theta = self.critic_params[param_idx]
                c, s = np.cos(theta/2), np.sin(theta/2)
                new_state = np.zeros_like(qstate)
                for z in range(self.dim):
                    z_flip = z ^ (1 << (self.n_qubits - 1 - q))
                    bit = (z >> (self.n_qubits - 1 - q)) & 1
                    new_state[z] = c * qstate[z] + (1-2*bit) * s * qstate[z_flip]
                qstate = new_state
                param_idx += 1

                theta = self.critic_params[param_idx]
                for z in range(self.dim):
                    bit = (z >> (self.n_qubits - 1 - q)) & 1
                    qstate[z] *= np.exp(1j*theta/2) if bit else np.exp(-1j*theta/2)
                param_idx += 1

        # Expectation value
        exp_val = 0
        for z in range(self.dim):
            bit = (z >> (self.n_qubits - 1)) & 1
            exp_val += (1 - 2*bit) * np.abs(qstate[z])**2

        return exp_val

    def update(self, state: np.ndarray, action: int,
               reward: float, next_state: np.ndarray, done: bool):
        """Update actor and critic."""
        # TD error (advantage estimate)
        value = self.get_value(state)
        next_value = 0 if done else self.get_value(next_state)
        td_target = reward + self.gamma * next_value
        advantage = td_target - value

        # Update critic
        for i in range(len(self.critic_params)):
            params_plus = self.critic_params.copy()
            params_plus[i] += np.pi/2
            params_minus = self.critic_params.copy()
            params_minus[i] -= np.pi/2

            self.critic_params = params_plus
            v_plus = self.get_value(state)
            self.critic_params = params_minus
            v_minus = self.get_value(state)
            self.critic_params[i] = (params_plus[i] + params_minus[i]) / 2

            gradient = (v_plus - v_minus) / 2
            self.critic_params[i] += self.critic_lr * advantage * gradient

        # Update actor
        prob = self.actor.get_action_probs(state, self.actor_params)[action]
        for i in range(len(self.actor_params)):
            params_plus = self.actor_params.copy()
            params_plus[i] += np.pi/2
            params_minus = self.actor_params.copy()
            params_minus[i] -= np.pi/2

            prob_plus = self.actor.get_action_probs(state, params_plus)[action]
            prob_minus = self.actor.get_action_probs(state, params_minus)[action]

            if prob > 1e-10:
                grad_log_prob = (prob_plus - prob_minus) / (2 * prob)
                self.actor_params[i] += self.actor_lr * advantage * grad_log_prob


# Example
print("\nQuantum Actor-Critic")
print("=" * 50)

np.random.seed(42)
env = SimpleGridEnv(size=3)
agent = QuantumActorCritic(n_qubits=3, n_actions=4, n_layers=1,
                           actor_lr=0.2, critic_lr=0.3, gamma=0.95)

episode_rewards = []
for episode in range(30):
    state = env.reset()
    total_reward = 0

    for step in range(30):
        probs = agent.actor.get_action_probs(state, agent.actor_params)
        action = np.random.choice(4, p=probs)

        next_state, reward, done = env.step(action)
        agent.update(state, action, reward, next_state, done)

        total_reward += reward
        state = next_state

        if done:
            break

    episode_rewards.append(total_reward)

    if episode % 10 == 0:
        avg = np.mean(episode_rewards[-10:]) if episode >= 10 else np.mean(episode_rewards)
        print(f"Episode {episode}: Avg Reward = {avg:.2f}")
```

## Summary

Quantum reinforcement learning approaches:

1. **Quantum Policies**: Variational circuits for action selection
2. **Policy Gradient**: REINFORCE with quantum policy networks
3. **Quantum Q-Learning**: VQC for Q-function approximation
4. **Actor-Critic**: Separate quantum networks for policy and value

<Tip>
Quantum RL is particularly promising for environments with quantum structure (like quantum control) or when the state/action spaces benefit from quantum feature maps. For classical environments, the advantage comes from potential expressivity of quantum policies.
</Tip>
