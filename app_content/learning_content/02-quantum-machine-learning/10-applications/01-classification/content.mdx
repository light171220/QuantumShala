# Quantum Classification

Classification is one of the most common machine learning tasks. This lesson covers how to build and train quantum classifiers.

## Binary Classification

Binary classification assigns inputs to one of two classes. Quantum classifiers encode data, process it through variational circuits, and measure to determine class.

<Definition>
**Quantum Binary Classifier**:

A quantum classifier maps input $x$ to class label $y \in \{0, 1\}$ via:

1. Encode: $|0\rangle \rightarrow |\psi(x)\rangle$
2. Process: $|\psi(x)\rangle \rightarrow U(\theta)|\psi(x)\rangle$
3. Measure: $P(y=1) = |\langle 1|U(\theta)|\psi(x)\rangle|^2$
</Definition>

```python
import numpy as np

class QuantumBinaryClassifier:
    """Binary classifier using variational quantum circuit."""

    def __init__(self, n_qubits: int, n_layers: int):
        self.n_qubits = n_qubits
        self.n_layers = n_layers
        self.dim = 2 ** n_qubits
        self.n_params = n_qubits * n_layers * 2

    def encode_data(self, x: np.ndarray) -> np.ndarray:
        """Encode classical data into quantum state."""
        state = np.zeros(self.dim, dtype=complex)
        state[0] = 1

        # Angle encoding
        for i, val in enumerate(x[:self.n_qubits]):
            theta = np.pi * val
            c, s = np.cos(theta/2), np.sin(theta/2)
            new_state = np.zeros_like(state)
            for z in range(self.dim):
                z_flip = z ^ (1 << (self.n_qubits - 1 - i))
                new_state[z] = c * state[z] - s * state[z_flip]
                new_state[z_flip] = s * state[z] + c * state[z_flip]
            state = new_state

        return state

    def variational_layer(self, state: np.ndarray,
                          params: np.ndarray) -> np.ndarray:
        """Apply variational circuit."""
        param_idx = 0

        for _ in range(self.n_layers):
            # Rotation layer
            for q in range(self.n_qubits):
                # RY
                theta = params[param_idx]
                c, s = np.cos(theta/2), np.sin(theta/2)
                new_state = np.zeros_like(state)
                for z in range(self.dim):
                    z_flip = z ^ (1 << (self.n_qubits - 1 - q))
                    bit = (z >> (self.n_qubits - 1 - q)) & 1
                    new_state[z] = c * state[z] + (1-2*bit) * s * state[z_flip]
                state = new_state
                param_idx += 1

                # RZ
                theta = params[param_idx]
                for z in range(self.dim):
                    bit = (z >> (self.n_qubits - 1 - q)) & 1
                    phase = np.exp(1j*theta/2) if bit else np.exp(-1j*theta/2)
                    state[z] *= phase
                param_idx += 1

            # Entanglement layer
            for q in range(self.n_qubits - 1):
                new_state = np.zeros_like(state)
                for z in range(self.dim):
                    ctrl = (z >> (self.n_qubits - 1 - q)) & 1
                    z_new = z ^ (1 << (self.n_qubits-1-(q+1))) if ctrl else z
                    new_state[z_new] += state[z]
                state = new_state

        return state

    def predict_proba(self, x: np.ndarray, params: np.ndarray) -> float:
        """Predict probability of class 1."""
        state = self.encode_data(x)
        state = self.variational_layer(state, params)

        # Probability of measuring |1⟩ on first qubit
        prob_1 = 0
        for z in range(self.dim):
            if (z >> (self.n_qubits - 1)) & 1:
                prob_1 += np.abs(state[z])**2

        return prob_1

    def predict(self, x: np.ndarray, params: np.ndarray) -> int:
        """Predict class label."""
        return 1 if self.predict_proba(x, params) > 0.5 else 0

    def loss(self, X: np.ndarray, y: np.ndarray,
             params: np.ndarray) -> float:
        """Binary cross-entropy loss."""
        total_loss = 0
        for xi, yi in zip(X, y):
            prob = np.clip(self.predict_proba(xi, params), 1e-10, 1-1e-10)
            total_loss -= yi * np.log(prob) + (1-yi) * np.log(1-prob)
        return total_loss / len(y)

# Example: XOR problem
print("Quantum Binary Classifier - XOR Problem")
print("=" * 50)

X = np.array([[0, 0], [0, 1], [1, 0], [1, 1]])
y = np.array([0, 1, 1, 0])

classifier = QuantumBinaryClassifier(n_qubits=2, n_layers=2)
params = np.random.uniform(-0.1, 0.1, classifier.n_params)

# Training loop
learning_rate = 0.5
for epoch in range(100):
    # Compute gradients
    gradients = np.zeros_like(params)
    for i in range(len(params)):
        params_plus = params.copy()
        params_plus[i] += np.pi/2
        params_minus = params.copy()
        params_minus[i] -= np.pi/2
        gradients[i] = (classifier.loss(X, y, params_plus) -
                        classifier.loss(X, y, params_minus)) / 2

    params -= learning_rate * gradients

    if epoch % 20 == 0:
        loss = classifier.loss(X, y, params)
        acc = np.mean([classifier.predict(xi, params) == yi
                       for xi, yi in zip(X, y)])
        print(f"Epoch {epoch}: Loss={loss:.4f}, Accuracy={acc:.2f}")

print("\nFinal predictions:")
for xi, yi in zip(X, y):
    pred = classifier.predict(xi, params)
    prob = classifier.predict_proba(xi, params)
    print(f"  {xi} -> {pred} (prob={prob:.3f}), true={yi}")
```

## Multi-class Classification

Multi-class classification extends to more than two classes using multiple measurement outcomes.

```python
import numpy as np

class QuantumMultiClassifier:
    """Multi-class classifier using quantum circuits."""

    def __init__(self, n_qubits: int, n_classes: int, n_layers: int):
        self.n_qubits = n_qubits
        self.n_classes = n_classes
        self.n_layers = n_layers
        self.dim = 2 ** n_qubits
        self.n_params = n_qubits * n_layers * 2

        # Number of qubits for class encoding
        self.class_qubits = int(np.ceil(np.log2(n_classes)))

    def encode_and_process(self, x: np.ndarray,
                           params: np.ndarray) -> np.ndarray:
        """Encode data and apply variational circuit."""
        state = np.zeros(self.dim, dtype=complex)
        state[0] = 1

        # Encode data
        for i, val in enumerate(x[:self.n_qubits]):
            theta = np.pi * val
            c, s = np.cos(theta/2), np.sin(theta/2)
            new_state = np.zeros_like(state)
            for z in range(self.dim):
                z_flip = z ^ (1 << (self.n_qubits - 1 - i))
                new_state[z] = c * state[z] - s * state[z_flip]
                new_state[z_flip] = s * state[z] + c * state[z_flip]
            state = new_state

        # Apply variational circuit
        param_idx = 0
        for _ in range(self.n_layers):
            for q in range(self.n_qubits):
                theta = params[param_idx]
                c, s = np.cos(theta/2), np.sin(theta/2)
                new_state = np.zeros_like(state)
                for z in range(self.dim):
                    z_flip = z ^ (1 << (self.n_qubits - 1 - q))
                    bit = (z >> (self.n_qubits - 1 - q)) & 1
                    new_state[z] = c * state[z] + (1-2*bit) * s * state[z_flip]
                state = new_state
                param_idx += 1

                theta = params[param_idx]
                for z in range(self.dim):
                    bit = (z >> (self.n_qubits - 1 - q)) & 1
                    state[z] *= np.exp(1j*theta/2) if bit else np.exp(-1j*theta/2)
                param_idx += 1

            for q in range(self.n_qubits - 1):
                new_state = np.zeros_like(state)
                for z in range(self.dim):
                    ctrl = (z >> (self.n_qubits - 1 - q)) & 1
                    z_new = z ^ (1 << (self.n_qubits-1-(q+1))) if ctrl else z
                    new_state[z_new] += state[z]
                state = new_state

        return state

    def predict_proba(self, x: np.ndarray, params: np.ndarray) -> np.ndarray:
        """Predict class probabilities."""
        state = self.encode_and_process(x, params)

        probs = np.zeros(self.n_classes)
        for z in range(self.dim):
            # Use first class_qubits to determine class
            class_idx = z >> (self.n_qubits - self.class_qubits)
            if class_idx < self.n_classes:
                probs[class_idx] += np.abs(state[z])**2

        # Normalize
        probs /= probs.sum() + 1e-10
        return probs

    def predict(self, x: np.ndarray, params: np.ndarray) -> int:
        """Predict class label."""
        return np.argmax(self.predict_proba(x, params))

# Example: 3-class problem
print("\nQuantum Multi-class Classifier")
print("=" * 50)

# Simple 3-class dataset
np.random.seed(42)
n_samples = 30

X = np.random.randn(n_samples, 4)
y = np.zeros(n_samples, dtype=int)
y[n_samples//3:2*n_samples//3] = 1
y[2*n_samples//3:] = 2

classifier = QuantumMultiClassifier(n_qubits=4, n_classes=3, n_layers=2)
params = np.random.uniform(-0.1, 0.1, classifier.n_params)

print(f"Dataset: {n_samples} samples, 3 classes")
print(f"Circuit: {classifier.n_qubits} qubits, {classifier.n_params} parameters")

# Quick training demo
for epoch in range(50):
    # Simplified gradient computation
    gradients = np.zeros_like(params)
    for i in range(min(5, len(params))):  # Update only first 5 params
        for xi, yi in zip(X, y):
            prob = classifier.predict_proba(xi, params)
            params_plus = params.copy()
            params_plus[i] += np.pi/2
            prob_plus = classifier.predict_proba(xi, params_plus)
            gradients[i] -= (prob_plus[yi] - prob[yi])

    params[:5] += 0.1 * gradients[:5]

predictions = [classifier.predict(xi, params) for xi in X]
accuracy = np.mean([p == yi for p, yi in zip(predictions, y)])
print(f"Final accuracy: {accuracy:.2f}")
```

## Quantum Kernel Classification

Use quantum kernel methods for classification without explicit variational optimization.

```python
import numpy as np

class QuantumKernelClassifier:
    """Classification using quantum kernels."""

    def __init__(self, n_qubits: int):
        self.n_qubits = n_qubits
        self.dim = 2 ** n_qubits

    def feature_map(self, x: np.ndarray) -> np.ndarray:
        """Map classical data to quantum state."""
        state = np.zeros(self.dim, dtype=complex)
        state[0] = 1

        # First layer: angle encoding
        for i, val in enumerate(x[:self.n_qubits]):
            theta = val
            c, s = np.cos(theta/2), np.sin(theta/2)
            new_state = np.zeros_like(state)
            for z in range(self.dim):
                z_flip = z ^ (1 << (self.n_qubits - 1 - i))
                new_state[z] = c * state[z] - s * state[z_flip]
                new_state[z_flip] = s * state[z] + c * state[z_flip]
            state = new_state

        # Entanglement
        for q in range(self.n_qubits - 1):
            new_state = np.zeros_like(state)
            for z in range(self.dim):
                ctrl = (z >> (self.n_qubits - 1 - q)) & 1
                z_new = z ^ (1 << (self.n_qubits-1-(q+1))) if ctrl else z
                new_state[z_new] += state[z]
            state = new_state

        # Second layer with products
        for i in range(self.n_qubits):
            for j in range(i+1, min(i+2, len(x))):
                if j < len(x):
                    theta = x[i] * x[j]
                    for z in range(self.dim):
                        bit = (z >> (self.n_qubits - 1 - i)) & 1
                        state[z] *= np.exp(1j*theta/2) if bit else np.exp(-1j*theta/2)

        return state

    def kernel(self, x1: np.ndarray, x2: np.ndarray) -> float:
        """Compute quantum kernel between two points."""
        state1 = self.feature_map(x1)
        state2 = self.feature_map(x2)
        return np.abs(np.vdot(state1, state2))**2

    def compute_kernel_matrix(self, X: np.ndarray) -> np.ndarray:
        """Compute full kernel matrix."""
        n = len(X)
        K = np.zeros((n, n))
        for i in range(n):
            for j in range(i, n):
                K[i, j] = self.kernel(X[i], X[j])
                K[j, i] = K[i, j]
        return K

    def fit_predict(self, X_train: np.ndarray, y_train: np.ndarray,
                    X_test: np.ndarray, reg: float = 0.1) -> np.ndarray:
        """Train and predict using kernel method."""
        # Compute kernel matrices
        K_train = self.compute_kernel_matrix(X_train)
        K_test = np.array([[self.kernel(xt, xtr)
                           for xtr in X_train] for xt in X_test])

        # Kernel ridge regression
        alpha = np.linalg.solve(K_train + reg * np.eye(len(K_train)), y_train)
        predictions = K_test @ alpha

        return (predictions > 0.5).astype(int)

# Example
print("\nQuantum Kernel Classification")
print("=" * 50)

# Create dataset
np.random.seed(42)
X_train = np.random.randn(20, 4)
y_train = (X_train[:, 0] + X_train[:, 1] > 0).astype(int)
X_test = np.random.randn(10, 4)
y_test = (X_test[:, 0] + X_test[:, 1] > 0).astype(int)

classifier = QuantumKernelClassifier(n_qubits=4)
predictions = classifier.fit_predict(X_train, y_train, X_test)

accuracy = np.mean(predictions == y_test)
print(f"Test accuracy: {accuracy:.2f}")
```

## Summary

Quantum classification techniques:

1. **Binary Classification**: Measure single qubit for 0/1 prediction
2. **Multi-class**: Use multiple qubits to encode class labels
3. **Quantum Kernels**: Implicit feature maps via kernel methods
4. **Hybrid Approaches**: Combine quantum and classical processing

<Tip>
Choose the right approach based on your data size and available quantum resources. Kernel methods work well for small datasets, while variational methods scale better.
</Tip>
