# Natural Gradients

Natural gradients account for the geometry of parameter space, leading to more efficient optimization by moving in the steepest direction on the manifold of quantum states.

## Learning Objectives

After completing this lesson, you will be able to:

- Understand parameter space geometry
- Compute the Fubini-Study metric
- Implement natural gradient descent
- Apply quantum natural gradients
- Compare with vanilla gradients

## 1. The Problem with Vanilla Gradients

Standard gradients ignore the geometry of the parameter space.

<Definition term="Natural Gradient">
The natural gradient is the steepest descent direction in the manifold of probability distributions or quantum states, computed as g̃ = F⁻¹g where F is the Fisher information matrix and g is the standard gradient.
</Definition>

```python
import numpy as np

def vanilla_gradient_problem():
    """
    Illustrate why vanilla gradients can be inefficient.
    """
    print("The Problem with Vanilla Gradients:")
    print("=" * 50)

    print("\n1. Parameter Space Geometry:")
    print("-" * 40)
    print("  Different parameters affect the quantum state differently")
    print("  θ₁ might change |ψ⟩ drastically")
    print("  θ₂ might barely affect |ψ⟩")
    print("  Vanilla gradient treats them equally!")

    print("\n\n2. Illustrative Example:")
    print("-" * 40)

    # Highly anisotropic cost function
    def cost(params):
        # θ₁ has small effect, θ₂ has large effect
        return params[0]**2 + 100 * params[1]**2

    def vanilla_gradient(params):
        return np.array([2 * params[0], 200 * params[1]])

    # Natural gradient with proper metric
    def natural_gradient(params):
        # Metric accounts for sensitivity
        F = np.diag([1, 100])  # Fisher-like metric
        g = vanilla_gradient(params)
        return np.linalg.solve(F, g)

    params = np.array([1.0, 1.0])

    print(f"  At params = {params}:")
    print(f"  Vanilla gradient:  {vanilla_gradient(params)}")
    print(f"  Natural gradient:  {natural_gradient(params)}")

    print("\n\n3. Optimization Comparison:")
    print("-" * 40)

    lr = 0.01

    # Vanilla gradient descent
    p_vanilla = np.array([1.0, 1.0])
    vanilla_history = [cost(p_vanilla)]
    for _ in range(50):
        p_vanilla = p_vanilla - lr * vanilla_gradient(p_vanilla)
        vanilla_history.append(cost(p_vanilla))

    # Natural gradient descent
    p_natural = np.array([1.0, 1.0])
    natural_history = [cost(p_natural)]
    for _ in range(50):
        p_natural = p_natural - lr * natural_gradient(p_natural)
        natural_history.append(cost(p_natural))

    print(f"\n  After 50 steps (lr={lr}):")
    print(f"  Vanilla: cost = {vanilla_history[-1]:.6f}")
    print(f"  Natural: cost = {natural_history[-1]:.6f}")

vanilla_gradient_problem()
```

<Tip>
Natural gradients are particularly valuable when parameters have very different sensitivities, which is common in quantum circuits due to varying gate effects.
</Tip>

## 2. The Fubini-Study Metric

The quantum natural gradient uses the Fubini-Study metric tensor.

```python
import numpy as np

def fubini_study_metric():
    """
    Introduce the Fubini-Study metric for quantum states.
    """
    print("Fubini-Study Metric:")
    print("=" * 50)

    print("\n1. Definition:")
    print("-" * 40)
    print("  The Fubini-Study metric measures 'distance' between")
    print("  quantum states in terms of their distinguishability.")
    print()
    print("  g_ij = Re[⟨∂ᵢψ|∂ⱼψ⟩ - ⟨∂ᵢψ|ψ⟩⟨ψ|∂ⱼψ⟩]")
    print()
    print("  where |∂ᵢψ⟩ = ∂|ψ⟩/∂θᵢ")

    print("\n\n2. Quantum Fisher Information:")
    print("-" * 40)
    print("  For variational circuits:")
    print("  F_ij = 4 × g_ij")
    print()
    print("  This relates to how sensitive measurements are")
    print("  to parameter changes.")

    # Numerical demonstration
    print("\n\n3. Example Calculation:")
    print("-" * 40)

    def param_state(theta):
        """State |ψ(θ)⟩ = cos(θ/2)|0⟩ + sin(θ/2)|1⟩"""
        return np.array([np.cos(theta/2), np.sin(theta/2)])

    def state_derivative(theta):
        """∂|ψ⟩/∂θ"""
        return np.array([-np.sin(theta/2)/2, np.cos(theta/2)/2])

    theta = 0.5

    psi = param_state(theta)
    dpsi = state_derivative(theta)

    # Fubini-Study metric component
    term1 = np.real(np.vdot(dpsi, dpsi))
    term2 = np.abs(np.vdot(dpsi, psi))**2
    g = term1 - term2

    print(f"  θ = {theta}")
    print(f"  |ψ⟩ = {psi.round(4)}")
    print(f"  |∂ψ⟩ = {dpsi.round(4)}")
    print(f"  ")
    print(f"  ⟨∂ψ|∂ψ⟩ = {term1:.4f}")
    print(f"  |⟨∂ψ|ψ⟩|² = {term2:.4f}")
    print(f"  g(θ) = {g:.4f}")
    print(f"  ")
    print(f"  Note: For this ansatz, g = 1/4 always")

fubini_study_metric()
```

<Formula>
Quantum Fisher Information Matrix:
$$F_{ij} = 4 \text{Re}\left[\langle\partial_i\psi|\partial_j\psi\rangle - \langle\partial_i\psi|\psi\rangle\langle\psi|\partial_j\psi\rangle\right]$$
</Formula>

## 3. Computing the Metric on Hardware

Methods to estimate the Fubini-Study metric on quantum devices.

```python
import numpy as np

def hardware_metric_estimation():
    """
    Estimate the Fubini-Study metric on quantum hardware.
    """
    print("Hardware Metric Estimation:")
    print("=" * 50)

    print("\n1. Parameter-Shift for Metric:")
    print("-" * 40)
    print("  F_ij can be computed using parameter-shift:")
    print()
    print("  For i ≠ j:")
    print("  F_ij = -(1/2)[⟨Z⟩_{++} - ⟨Z⟩_{+-} - ⟨Z⟩_{-+} + ⟨Z⟩_{--}]")
    print()
    print("  where ± means shift θᵢ and θⱼ by ±π/2")

    def estimate_metric_element(circuit_fn, params, i, j, shift=np.pi/2):
        """
        Estimate F_ij using parameter-shift.

        Args:
            circuit_fn: Function returning expectation value
            params: Parameters
            i, j: Indices
            shift: Shift amount
        """
        if i == j:
            # Diagonal element
            p0 = params.copy()
            p_plus = params.copy(); p_plus[i] += shift
            p_minus = params.copy(); p_minus[i] -= shift

            f0 = circuit_fn(p0)
            f_plus = circuit_fn(p_plus)
            f_minus = circuit_fn(p_minus)

            return 0.5 * (1 - (f_plus + f_minus) / 2 + (f_plus - f_minus)**2 / 4)
        else:
            # Off-diagonal element
            shifts = [(1, 1), (1, -1), (-1, 1), (-1, -1)]
            f_vals = []

            for si, sj in shifts:
                p = params.copy()
                p[i] += si * shift
                p[j] += sj * shift
                f_vals.append(circuit_fn(p))

            return -0.125 * (f_vals[0] - f_vals[1] - f_vals[2] + f_vals[3])

    print("\n\n2. Simulated Example:")
    print("-" * 40)

    def simulated_circuit(params):
        """Simulated circuit: expectation of product rotations."""
        return np.cos(params[0]) * np.cos(params[1])

    params = np.array([0.5, 0.8])

    # Compute full metric
    n = len(params)
    F = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            F[i, j] = estimate_metric_element(simulated_circuit, params, i, j)

    print(f"  Parameters: {params}")
    print(f"\n  Estimated Fisher Information Matrix:")
    print(f"    {F.round(4)}")

    print("\n\n3. Practical Considerations:")
    print("-" * 40)

    considerations = [
        "O(n²) circuit evaluations for full metric",
        "Shot noise affects metric accuracy",
        "Block-diagonal approximations reduce cost",
        "Regularization needed for numerical stability"
    ]

    for c in considerations:
        print(f"  • {c}")

hardware_metric_estimation()
```

## 4. Natural Gradient Descent

Implementing the quantum natural gradient optimizer.

```python
import numpy as np

class QuantumNaturalGradient:
    """
    Quantum Natural Gradient optimizer.
    """

    def __init__(self, circuit_fn, n_params, regularization=0.01):
        """
        Args:
            circuit_fn: Circuit expectation value function
            n_params: Number of parameters
            regularization: Regularization for matrix inversion
        """
        self.circuit = circuit_fn
        self.n_params = n_params
        self.reg = regularization

    def compute_gradient(self, params, shift=np.pi/2):
        """Compute vanilla gradient using parameter-shift."""
        grad = np.zeros(self.n_params)
        for i in range(self.n_params):
            p_plus = params.copy(); p_plus[i] += shift
            p_minus = params.copy(); p_minus[i] -= shift
            grad[i] = 0.5 * (self.circuit(p_plus) - self.circuit(p_minus))
        return grad

    def compute_metric(self, params, shift=np.pi/2):
        """Compute Fubini-Study metric tensor."""
        F = np.zeros((self.n_params, self.n_params))

        for i in range(self.n_params):
            for j in range(i, self.n_params):
                if i == j:
                    # Diagonal
                    p_plus = params.copy(); p_plus[i] += shift
                    p_minus = params.copy(); p_minus[i] -= shift

                    f_plus = self.circuit(p_plus)
                    f_minus = self.circuit(p_minus)

                    F[i, i] = 0.25 * (1 - 0.5 * (f_plus + f_minus))
                else:
                    # Off-diagonal
                    p = [params.copy() for _ in range(4)]
                    p[0][i] += shift; p[0][j] += shift
                    p[1][i] += shift; p[1][j] -= shift
                    p[2][i] -= shift; p[2][j] += shift
                    p[3][i] -= shift; p[3][j] -= shift

                    f = [self.circuit(pp) for pp in p]
                    F[i, j] = -0.125 * (f[0] - f[1] - f[2] + f[3])
                    F[j, i] = F[i, j]

        return F

    def natural_gradient(self, params):
        """Compute natural gradient."""
        g = self.compute_gradient(params)
        F = self.compute_metric(params)

        # Regularize for numerical stability
        F_reg = F + self.reg * np.eye(self.n_params)

        # Natural gradient: F^(-1) @ g
        try:
            g_natural = np.linalg.solve(F_reg, g)
        except np.linalg.LinAlgError:
            g_natural = g  # Fall back to vanilla gradient

        return g_natural

    def step(self, params, learning_rate):
        """Perform one optimization step."""
        g_nat = self.natural_gradient(params)
        return params - learning_rate * g_nat

# Demonstration
print("Quantum Natural Gradient Optimizer:")
print("=" * 50)

def quantum_cost(params):
    """Simulated quantum cost function."""
    return 0.5 * (1 - np.cos(params[0]) * np.cos(params[1]) * np.cos(params[2]))

params = np.array([1.0, 0.5, 0.8])
qng = QuantumNaturalGradient(quantum_cost, n_params=3)

print(f"\nInitial params: {params}")
print(f"Initial cost: {quantum_cost(params):.4f}")

# Vanilla gradient
vanilla_grad = qng.compute_gradient(params)
print(f"\nVanilla gradient: {vanilla_grad.round(4)}")

# Natural gradient
natural_grad = qng.natural_gradient(params)
print(f"Natural gradient: {natural_grad.round(4)}")

# Metric tensor
F = qng.compute_metric(params)
print(f"\nMetric tensor:\n{F.round(4)}")
```

<Warning>
Computing the full metric tensor requires O(n²) circuit evaluations. For large n, consider block-diagonal or other approximations.
</Warning>

## 5. Approximations and Efficiency

Techniques to reduce the cost of natural gradients.

```python
import numpy as np

def efficient_natural_gradients():
    """
    Efficient approximations to the full natural gradient.
    """
    print("Efficient Natural Gradient Approximations:")
    print("=" * 50)

    print("\n1. Block-Diagonal Approximation:")
    print("-" * 40)
    print("  Assume metric is block-diagonal per layer")
    print("  Cost: O(L × k²) instead of O(n²)")
    print("  where L = layers, k = params per layer")

    def block_diagonal_metric(params, block_sizes):
        """Compute block-diagonal approximation."""
        n = len(params)
        F = np.zeros((n, n))

        idx = 0
        for size in block_sizes:
            # Fill in this block
            for i in range(size):
                for j in range(size):
                    F[idx + i, idx + j] = 0.25 if i == j else 0.05
            idx += size

        return F

    print("\n\n2. Diagonal Approximation:")
    print("-" * 40)
    print("  Only compute diagonal elements")
    print("  Cost: O(n) instead of O(n²)")
    print("  Trade-off: Ignores parameter correlations")

    def diagonal_metric(params):
        """Diagonal approximation - just variances."""
        return np.diag([0.25] * len(params))

    print("\n\n3. Comparison:")
    print("-" * 40)

    n_params = 12
    params = np.random.randn(n_params)

    print(f"  {n_params} parameters, 4 layers of 3:")
    print()
    print(f"  Full metric:     {n_params**2} = {n_params**2} evaluations")
    print(f"  Block-diagonal:  4 × 3² = {4 * 9} evaluations")
    print(f"  Diagonal only:   {n_params} evaluations")

    print("\n\n4. Stochastic Natural Gradient:")
    print("-" * 40)
    print("  Sample subset of metric elements each step")
    print("  Build up accurate metric estimate over time")
    print("  Similar to stochastic gradient methods")

    print("\n\n5. Layer-wise Updates:")
    print("-" * 40)
    print("  Update one layer at a time")
    print("  Compute metric only for active layer")
    print("  Coordinate descent in natural gradient space")

efficient_natural_gradients()
```

## 6. Comparison with Vanilla Gradients

Empirical comparison on optimization tasks.

```python
import numpy as np

def gradient_comparison():
    """
    Compare vanilla and natural gradient optimization.
    """
    print("Vanilla vs Natural Gradient Comparison:")
    print("=" * 50)

    def quantum_cost(params):
        """Simulated anisotropic cost landscape."""
        # Different sensitivities for different parameters
        weights = np.array([0.1, 1.0, 10.0])
        return np.sum(weights * np.sin(params)**2)

    def vanilla_gradient(params):
        grad = np.zeros(len(params))
        shift = np.pi / 2
        for i in range(len(params)):
            p_plus = params.copy(); p_plus[i] += shift
            p_minus = params.copy(); p_minus[i] -= shift
            grad[i] = 0.5 * (quantum_cost(p_plus) - quantum_cost(p_minus))
        return grad

    def natural_gradient(params, reg=0.01):
        # Simplified: use known sensitivity as metric
        weights = np.array([0.1, 1.0, 10.0])
        F = np.diag(weights)
        g = vanilla_gradient(params)
        return np.linalg.solve(F + reg * np.eye(3), g)

    # Run optimization
    np.random.seed(42)
    n_steps = 100
    lr = 0.1

    # Vanilla
    params_v = np.array([1.0, 1.0, 1.0])
    history_v = [quantum_cost(params_v)]
    for _ in range(n_steps):
        params_v = params_v - lr * vanilla_gradient(params_v)
        history_v.append(quantum_cost(params_v))

    # Natural
    params_n = np.array([1.0, 1.0, 1.0])
    history_n = [quantum_cost(params_n)]
    for _ in range(n_steps):
        params_n = params_n - lr * natural_gradient(params_n)
        history_n.append(quantum_cost(params_n))

    print("\n  Optimization Results:")
    print("-" * 40)
    print(f"\n  {'Step':<8} {'Vanilla':<15} {'Natural':<15}")
    print("  " + "-" * 38)

    for i in [0, 10, 25, 50, 100]:
        print(f"  {i:<8} {history_v[i]:<15.4f} {history_n[i]:<15.4f}")

    print(f"\n  Final parameters:")
    print(f"    Vanilla: {params_v.round(3)}")
    print(f"    Natural: {params_n.round(3)}")

    print("\n\n  Key Observations:")
    print("-" * 40)
    print("  • Natural gradient converges faster")
    print("  • Equal update step in 'state space'")
    print("  • Particularly helpful for ill-conditioned problems")

gradient_comparison()
```

## Summary

Natural gradients improve optimization by respecting geometry:

1. **Motivation**: Vanilla gradients ignore parameter sensitivity
2. **Metric**: Fubini-Study measures quantum state distance
3. **Computation**: Parameter-shift can estimate metric on hardware
4. **Formula**: Natural gradient = F⁻¹ × vanilla gradient
5. **Efficiency**: Use block-diagonal or diagonal approximations
6. **Benefits**: Faster convergence, especially for ill-conditioned problems

Natural gradients are valuable but computationally expensive—use approximations.
