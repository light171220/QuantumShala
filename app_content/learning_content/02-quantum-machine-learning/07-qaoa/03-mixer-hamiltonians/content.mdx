# Mixer Hamiltonians

The mixer Hamiltonian determines how QAOA explores the solution space. Different problems require different mixers to maintain feasibility and ensure good performance.

## Role of the Mixer

The mixer enables transitions between computational basis states.

<Definition>
**Mixer Requirements**:
1. Should not commute with cost Hamiltonian (enables exploration)
2. Ground state should include all feasible solutions
3. For constrained problems, should preserve feasibility
</Definition>

```python
import numpy as np

def analyze_mixer(mixer_H, cost_H, name="Mixer"):
    """Analyze mixer properties."""
    n = int(np.log2(len(mixer_H)))

    # Check commutation
    commutator = mixer_H @ cost_H - cost_H @ mixer_H
    commutes = np.allclose(commutator, 0)

    # Eigenstructure
    eigenvalues = np.linalg.eigvalsh(mixer_H)

    print(f"\n{name} Analysis:")
    print(f"  Commutes with cost: {commutes}")
    print(f"  Eigenvalues: {np.unique(eigenvalues.round(4))}")

    # Check if uniform superposition is ground state
    uniform = np.ones(2**n) / np.sqrt(2**n)
    energy = np.real(uniform.conj() @ mixer_H @ uniform)
    ground_energy = eigenvalues.min()
    is_ground = np.isclose(energy, ground_energy)
    print(f"  |+⟩^n is ground state: {is_ground}")

# Example: standard X mixer vs diagonal cost
n = 3
X_mixer = np.zeros((8, 8))
for q in range(n):
    for z in range(8):
        z_flip = z ^ (1 << (n-1-q))
        X_mixer[z, z_flip] += 1

cost_H = np.diag([0, 1, 1, 2, 1, 2, 2, 3])  # Count 1s

analyze_mixer(X_mixer, cost_H, "Standard X-mixer")
```

## Standard X-Mixer

The simplest and most common mixer.

<Formula>
**X-Mixer**: Sum of Pauli X on each qubit:

$$B = \sum_{j=1}^{n} X_j$$

The unitary $e^{-i\beta B}$ applies RX rotations to each qubit.
</Formula>

```python
import numpy as np
from scipy.linalg import expm

class XMixer:
    """Standard X mixer implementation."""

    def __init__(self, n_qubits):
        self.n = n_qubits
        self.dim = 2 ** n_qubits
        self.hamiltonian = self._build_hamiltonian()

    def _build_hamiltonian(self):
        """Build B = sum X_j."""
        B = np.zeros((self.dim, self.dim))
        for q in range(self.n):
            for z in range(self.dim):
                z_flip = z ^ (1 << (self.n - 1 - q))
                B[z, z_flip] += 1
        return B

    def unitary(self, beta):
        """Compute e^{-i beta B}."""
        return expm(-1j * beta * self.hamiltonian)

    def apply(self, state, beta):
        """Apply mixer unitary to state."""
        return self.unitary(beta) @ state

    def ground_state(self):
        """Return ground state of mixer (|+⟩^n)."""
        return np.ones(self.dim) / np.sqrt(self.dim)

# Example
mixer = XMixer(3)
print("Standard X-Mixer (3 qubits)")
print("=" * 40)

# Show Hamiltonian structure
print("\nHamiltonian B (each 1 represents X connection):")
print((mixer.hamiltonian > 0).astype(int))

# Effect on basis states
print("\nEffect of mixer unitary (β = π/4):")
beta = np.pi/4
U = mixer.unitary(beta)

for z in [0, 7]:  # |000⟩ and |111⟩
    state = np.zeros(8)
    state[z] = 1
    new_state = U @ state

    print(f"\n  |{format(z, '03b')}⟩ →")
    for i, amp in enumerate(new_state):
        if np.abs(amp) > 0.01:
            print(f"    {amp:.3f} |{format(i, '03b')}⟩")
```

## XY-Mixer for Particle Conservation

The XY-mixer preserves the number of 1s in the bitstring.

<Definition>
**XY-Mixer**: For each pair of qubits:

$$B_{XY} = \sum_{i<j} (X_i X_j + Y_i Y_j) / 2$$

This swaps |01⟩ ↔ |10⟩ while leaving |00⟩ and |11⟩ unchanged.
</Definition>

<Warning>
The XY-mixer only connects states with the same Hamming weight. Starting from a specific particle number sector, the optimization stays within that sector.
</Warning>

```python
import numpy as np
from scipy.linalg import expm

class XYMixer:
    """XY mixer that preserves particle number."""

    def __init__(self, n_qubits, connectivity='all'):
        """
        Initialize XY mixer.

        Args:
            n_qubits: Number of qubits
            connectivity: 'all' for all pairs, 'linear' for nearest neighbor
        """
        self.n = n_qubits
        self.dim = 2 ** n_qubits
        self.connectivity = connectivity
        self.hamiltonian = self._build_hamiltonian()

    def _build_hamiltonian(self):
        """Build B_XY."""
        B = np.zeros((self.dim, self.dim), dtype=complex)

        pairs = self._get_pairs()

        for i, j in pairs:
            for z in range(self.dim):
                bi = (z >> (self.n - 1 - i)) & 1
                bj = (z >> (self.n - 1 - j)) & 1

                # XY swaps |01⟩ ↔ |10⟩
                if bi != bj:
                    z_swap = z ^ (1 << (self.n - 1 - i)) ^ (1 << (self.n - 1 - j))
                    B[z, z_swap] += 0.5  # (XX + YY)/2 coefficient

        return B

    def _get_pairs(self):
        """Get qubit pairs based on connectivity."""
        if self.connectivity == 'all':
            return [(i, j) for i in range(self.n) for j in range(i+1, self.n)]
        elif self.connectivity == 'linear':
            return [(i, i+1) for i in range(self.n - 1)]

    def unitary(self, beta):
        """Compute e^{-i beta B}."""
        return expm(-1j * beta * self.hamiltonian)

    def verify_conservation(self, state, beta):
        """Verify particle number is conserved."""
        new_state = self.unitary(beta) @ state

        def particle_sector(state_vec):
            sectors = {}
            for z, amp in enumerate(state_vec):
                if np.abs(amp) > 1e-10:
                    n_particles = bin(z).count('1')
                    sectors[n_particles] = sectors.get(n_particles, 0) + np.abs(amp)**2
            return sectors

        before = particle_sector(state)
        after = particle_sector(new_state)

        return before, after

# Example
print("XY-Mixer (Particle Conserving)")
print("=" * 40)

xy = XYMixer(4, connectivity='all')

# Start with 2-particle state
state = np.zeros(16)
state[0b0011] = 1.0  # |0011⟩ - 2 particles

print(f"\nInitial state: |0011⟩ (2 particles)")

before, after = xy.verify_conservation(state, np.pi/4)
print(f"Particle sectors before: {before}")
print(f"Particle sectors after: {after}")
print(f"Conservation verified: {before == after}")

# Show final state
new_state = xy.unitary(np.pi/4) @ state
print("\nFinal state components:")
for z, amp in enumerate(new_state):
    if np.abs(amp) > 0.01:
        bits = format(z, '04b')
        n_ones = bits.count('1')
        print(f"  |{bits}⟩ (n={n_ones}): {np.abs(amp):.4f}")
```

## Ring Mixer for Cyclic Problems

For problems with cyclic structure like TSP.

```python
import numpy as np
from scipy.linalg import expm

class RingMixer:
    """Ring mixer for cyclic/periodic problems."""

    def __init__(self, n_qubits):
        self.n = n_qubits
        self.dim = 2 ** n_qubits
        self.hamiltonian = self._build_hamiltonian()

    def _build_hamiltonian(self):
        """Build ring mixer B = sum X_i X_{i+1} + Y_i Y_{i+1}."""
        B = np.zeros((self.dim, self.dim), dtype=complex)

        for i in range(self.n):
            j = (i + 1) % self.n  # Wrap around

            for z in range(self.dim):
                bi = (z >> (self.n - 1 - i)) & 1
                bj = (z >> (self.n - 1 - j)) & 1

                if bi != bj:
                    z_swap = z ^ (1 << (self.n - 1 - i)) ^ (1 << (self.n - 1 - j))
                    B[z, z_swap] += 0.5

        return B

# Example
ring = RingMixer(4)
print("\nRing Mixer (4 qubits)")
print("=" * 40)
print("Connects: 0↔1, 1↔2, 2↔3, 3↔0")

# Show connectivity
B = ring.hamiltonian
print("\nConnectivity matrix:")
print((np.abs(B) > 0).astype(int))
```

## Grover Mixer

The Grover mixer uses the uniform superposition projector.

<Formula>
**Grover Mixer**:

$$B_G = |s\rangle\langle s| - I$$

where $|s\rangle$ is the uniform superposition over feasible states.
</Formula>

```python
import numpy as np
from scipy.linalg import expm

class GroverMixer:
    """Grover-style mixer for constraint satisfaction."""

    def __init__(self, feasible_states):
        """
        Initialize with feasible state indices.

        Args:
            feasible_states: List of feasible computational basis state indices
        """
        self.feasible = feasible_states
        self.n_feasible = len(feasible_states)
        self.dim = max(feasible_states) + 1

        # Build uniform superposition over feasible states
        self.s = np.zeros(self.dim, dtype=complex)
        for z in feasible_states:
            self.s[z] = 1.0 / np.sqrt(self.n_feasible)

        self.hamiltonian = self._build_hamiltonian()

    def _build_hamiltonian(self):
        """Build B = |s⟩⟨s| - I (restricted to feasible subspace)."""
        # Projector onto s
        projector = np.outer(self.s, self.s.conj())
        return projector - np.eye(self.dim)

    def unitary(self, beta):
        """Compute Grover mixer unitary."""
        return expm(-1j * beta * self.hamiltonian)

# Example: 2-of-4 feasible states
feasible = [0b0011, 0b0101, 0b0110, 0b1001, 0b1010, 0b1100]  # Exactly 2 ones
grover = GroverMixer(feasible)

print("Grover Mixer Example")
print("=" * 40)
print(f"Feasible states (2 of 4): {[format(z, '04b') for z in feasible]}")
print(f"Dimension of |s⟩: {grover.n_feasible}")
```

## Choosing the Right Mixer

Different problems benefit from different mixers.

```python
def mixer_selection_guide():
    """Guide for choosing QAOA mixer."""

    recommendations = {
        'Unconstrained (MaxCut, etc.)': {
            'mixer': 'Standard X-mixer',
            'reason': 'Full exploration of Hilbert space',
            'initial_state': '|+⟩^n'
        },
        'Fixed particle number': {
            'mixer': 'XY-mixer',
            'reason': 'Preserves Hamming weight',
            'initial_state': 'Dicke state with k ones'
        },
        'Permutation problems (TSP)': {
            'mixer': 'Swap network or XY on rows/columns',
            'reason': 'Preserves one-hot constraints',
            'initial_state': 'Uniform over valid permutations'
        },
        'General constraints': {
            'mixer': 'Grover mixer',
            'reason': 'Only connects feasible states',
            'initial_state': 'Uniform over feasible states'
        }
    }

    print("Mixer Selection Guide")
    print("=" * 55)

    for problem, info in recommendations.items():
        print(f"\n{problem}:")
        print(f"  Recommended: {info['mixer']}")
        print(f"  Reason: {info['reason']}")
        print(f"  Initial state: {info['initial_state']}")

mixer_selection_guide()
```

## Summary

Mixer Hamiltonians in QAOA:

1. **X-mixer** for unconstrained problems
2. **XY-mixer** preserves particle number
3. **Ring mixer** for cyclic structures
4. **Grover mixer** for general constraint satisfaction
5. **Custom mixers** for specific problem symmetries
