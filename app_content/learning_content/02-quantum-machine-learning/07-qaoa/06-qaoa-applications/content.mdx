# QAOA Applications

QAOA can be applied to a wide range of combinatorial optimization problems. This lesson covers practical implementations across different domains.

## MaxCut on General Graphs

MaxCut is the canonical application of QAOA.

<Definition>
**MaxCut Problem**:

Given a graph G = (V, E) with weighted edges, find a partition of vertices into two sets that maximizes the total weight of edges crossing the partition.

$$\max \sum_{(i,j) \in E} w_{ij} \cdot \mathbb{1}[x_i \neq x_j]$$
</Definition>

```python
import numpy as np
from scipy.optimize import minimize
from scipy.linalg import expm

class MaxCutQAOA:
    """QAOA solver for weighted MaxCut."""

    def __init__(self, n_vertices, edges, weights=None):
        """
        Args:
            n_vertices: Number of vertices
            edges: List of (i, j) edges
            weights: Dict of (i, j) -> weight, or None for unit weights
        """
        self.n = n_vertices
        self.edges = edges
        self.weights = weights or {e: 1.0 for e in edges}
        self.dim = 2 ** n_vertices

        self.C = self._build_cost_hamiltonian()
        self.B = self._build_mixer()

    def _build_cost_hamiltonian(self):
        """Build MaxCut cost Hamiltonian."""
        C = np.zeros((self.dim, self.dim))

        for (i, j), w in self.weights.items():
            for z in range(self.dim):
                bi = (z >> (self.n - 1 - i)) & 1
                bj = (z >> (self.n - 1 - j)) & 1
                if bi != bj:
                    C[z, z] += w

        return C

    def _build_mixer(self):
        """Build X-mixer."""
        B = np.zeros((self.dim, self.dim))

        for q in range(self.n):
            for z in range(self.dim):
                z_flip = z ^ (1 << (self.n - 1 - q))
                B[z, z_flip] += 1

        return B

    def execute(self, gamma, beta):
        """Run QAOA circuit."""
        p = len(gamma)
        state = np.ones(self.dim, dtype=complex) / np.sqrt(self.dim)

        for layer in range(p):
            state = expm(-1j * gamma[layer] * self.C) @ state
            state = expm(-1j * beta[layer] * self.B) @ state

        return state

    def cost_expectation(self, params):
        """Compute expected cost."""
        p = len(params) // 2
        gamma, beta = params[:p], params[p:]

        state = self.execute(gamma, beta)
        cost = np.real(state.conj() @ self.C @ state)

        return -cost  # Minimize negative cost

    def solve(self, p, maxiter=200):
        """Find optimal QAOA parameters."""
        x0 = np.random.uniform(0, np.pi, 2 * p)

        result = minimize(
            self.cost_expectation,
            x0,
            method='COBYLA',
            options={'maxiter': maxiter}
        )

        gamma = result.x[:p]
        beta = result.x[p:]
        state = self.execute(gamma, beta)

        return {
            'cost': -result.fun,
            'gamma': gamma,
            'beta': beta,
            'state': state,
            'success': result.success
        }

    def sample_solution(self, state, n_samples=1000):
        """Sample solutions from QAOA state."""
        probs = np.abs(state) ** 2
        samples = np.random.choice(self.dim, size=n_samples, p=probs)

        # Find best sample
        best_z = max(samples, key=lambda z: self.C[z, z])
        best_cost = self.C[best_z, best_z]

        return {
            'bitstring': format(best_z, f'0{self.n}b'),
            'cost': best_cost,
            'samples': samples
        }

# Example: Petersen graph (famous 3-regular graph)
print("MaxCut on 5-vertex graph")
print("=" * 50)

edges = [(0, 1), (1, 2), (2, 3), (3, 4), (4, 0), (0, 2)]
weights = {e: 1.0 for e in edges}
weights[(0, 2)] = 2.0  # One stronger edge

qaoa = MaxCutQAOA(5, edges, weights)

for p in [1, 2, 3]:
    result = qaoa.solve(p)
    sample = qaoa.sample_solution(result['state'])
    print(f"\np={p}: Expected cost = {result['cost']:.3f}")
    print(f"       Best sample: {sample['bitstring']} (cost={sample['cost']:.1f})")
```

## Portfolio Optimization

QAOA can optimize asset selection for investment portfolios.

<Formula>
**Mean-Variance Portfolio**:

$$\min_{x} \lambda x^T \Sigma x - \mu^T x$$

subject to $\sum_i x_i = k$ (select exactly k assets)
</Formula>

```python
import numpy as np
from scipy.optimize import minimize
from scipy.linalg import expm

class PortfolioQAOA:
    """QAOA for portfolio optimization."""

    def __init__(self, returns, covariance, risk_aversion=0.5, budget=None):
        """
        Args:
            returns: Expected returns for each asset
            covariance: Covariance matrix
            risk_aversion: Trade-off parameter λ
            budget: Number of assets to select (optional)
        """
        self.n = len(returns)
        self.returns = np.array(returns)
        self.cov = np.array(covariance)
        self.lambda_ = risk_aversion
        self.budget = budget
        self.dim = 2 ** self.n

        self.C = self._build_cost_hamiltonian()
        self.B = self._build_mixer()

    def _build_cost_hamiltonian(self):
        """Build portfolio cost Hamiltonian."""
        C = np.zeros((self.dim, self.dim))

        for z in range(self.dim):
            # Convert to binary portfolio
            x = np.array([(z >> (self.n - 1 - i)) & 1 for i in range(self.n)])

            # Portfolio return
            ret = np.dot(self.returns, x)

            # Portfolio risk (variance)
            risk = x @ self.cov @ x

            # Objective: minimize risk - return (maximize return - risk)
            cost = self.lambda_ * risk - ret

            # Budget constraint penalty
            if self.budget is not None:
                penalty = 10 * (np.sum(x) - self.budget) ** 2
                cost += penalty

            C[z, z] = cost

        return C

    def _build_mixer(self):
        """Build mixer Hamiltonian."""
        B = np.zeros((self.dim, self.dim))

        for q in range(self.n):
            for z in range(self.dim):
                z_flip = z ^ (1 << (self.n - 1 - q))
                B[z, z_flip] += 1

        return B

    def solve(self, p=2, maxiter=200):
        """Solve portfolio optimization."""
        def objective(params):
            gamma, beta = params[:p], params[p:]
            state = self._execute(gamma, beta)
            return np.real(state.conj() @ self.C @ state)

        x0 = np.random.uniform(0, np.pi, 2 * p)
        result = minimize(objective, x0, method='COBYLA',
                         options={'maxiter': maxiter})

        gamma, beta = result.x[:p], result.x[p:]
        state = self._execute(gamma, beta)

        return self._extract_portfolio(state)

    def _execute(self, gamma, beta):
        """Execute QAOA circuit."""
        state = np.ones(self.dim, dtype=complex) / np.sqrt(self.dim)

        for layer in range(len(gamma)):
            state = expm(-1j * gamma[layer] * self.C) @ state
            state = expm(-1j * beta[layer] * self.B) @ state

        return state

    def _extract_portfolio(self, state):
        """Extract best portfolio from state."""
        probs = np.abs(state) ** 2
        best_z = np.argmax(probs)

        portfolio = [(z >> (self.n - 1 - i)) & 1 for i in range(self.n)]
        x = np.array(portfolio)

        return {
            'selection': portfolio,
            'expected_return': np.dot(self.returns, x),
            'risk': x @ self.cov @ x,
            'probability': probs[best_z]
        }

# Example
print("Portfolio Optimization with QAOA")
print("=" * 50)

# 5 assets with different risk/return profiles
returns = [0.05, 0.10, 0.12, 0.08, 0.15]  # Expected returns
cov = np.array([
    [0.01, 0.002, 0.001, 0.003, 0.002],
    [0.002, 0.04, 0.005, 0.002, 0.01],
    [0.001, 0.005, 0.03, 0.004, 0.008],
    [0.003, 0.002, 0.004, 0.02, 0.003],
    [0.002, 0.01, 0.008, 0.003, 0.05]
])  # Covariance matrix

assets = ['Bonds', 'Tech', 'Healthcare', 'Utilities', 'Crypto']

# Select 2 assets

result = portfolio_qaoa.solve(p=2)

print("\nSelected portfolio:")
for i, selected in enumerate(result['selection']):
    if selected:
        print(f"  {assets[i]}: return={returns[i]:.1%}")

print(f"\nExpected return: {result['expected_return']:.2%}")
print(f"Portfolio risk: {result['risk']:.4f}")
```

## Job Shop Scheduling

QAOA can tackle scheduling optimization problems.

```python
import numpy as np
from scipy.linalg import expm
from scipy.optimize import minimize

class SchedulingQAOA:
    """QAOA for job scheduling problems."""

    def __init__(self, n_jobs, n_machines, processing_times):
        """
        Args:
            n_jobs: Number of jobs
            n_machines: Number of machines
            processing_times: [job][machine] processing time matrix
        """
        self.n_jobs = n_jobs
        self.n_machines = n_machines
        self.times = np.array(processing_times)

        # Binary variables: x[j,m] = 1 if job j assigned to machine m
        self.n_vars = n_jobs * n_machines
        self.dim = 2 ** self.n_vars

    def _var_index(self, job, machine):
        """Get variable index for job-machine assignment."""
        return job * self.n_machines + machine

    def _decode_assignment(self, z):
        """Decode bitstring to job assignments."""
        assignment = {}
        for j in range(self.n_jobs):
            for m in range(self.n_machines):
                idx = self._var_index(j, m)
                if (z >> (self.n_vars - 1 - idx)) & 1:
                    assignment[j] = m
        return assignment

    def _compute_makespan(self, z):
        """Compute makespan for given assignment."""
        machine_loads = [0] * self.n_machines

        for j in range(self.n_jobs):
            assigned = False
            for m in range(self.n_machines):
                idx = self._var_index(j, m)
                if (z >> (self.n_vars - 1 - idx)) & 1:
                    machine_loads[m] += self.times[j, m]
                    assigned = True
                    break

            if not assigned:
                return float('inf')

        return max(machine_loads)

    def _build_cost_hamiltonian(self):
        """Build scheduling cost Hamiltonian."""
        C = np.zeros((self.dim, self.dim))

        for z in range(self.dim):
            # Makespan objective
            makespan = self._compute_makespan(z)

            # Constraint: each job assigned to exactly one machine
            constraint_penalty = 0
            for j in range(self.n_jobs):
                count = sum(
                    (z >> (self.n_vars - 1 - self._var_index(j, m))) & 1
                    for m in range(self.n_machines)
                )
                constraint_penalty += 100 * (count - 1) ** 2

            C[z, z] = makespan + constraint_penalty

        return C

    def solve_brute_force(self):
        """Solve by checking all valid assignments."""
        best_makespan = float('inf')
        best_assignment = None

        # Only check valid assignments (each job to one machine)
        from itertools import product
        for assignment in product(range(self.n_machines), repeat=self.n_jobs):
            machine_loads = [0] * self.n_machines
            for j, m in enumerate(assignment):
                machine_loads[m] += self.times[j, m]

            makespan = max(machine_loads)
            if makespan < best_makespan:
                best_makespan = makespan
                best_assignment = assignment

        return {
            'assignment': list(best_assignment),
            'makespan': best_makespan
        }

# Example
print("Job Scheduling with QAOA")
print("=" * 50)

# 3 jobs, 2 machines
processing_times = [
    [3, 5],  # Job 0: 3 hours on M0, 5 hours on M1
    [4, 2],  # Job 1: 4 hours on M0, 2 hours on M1
    [2, 4]   # Job 2: 2 hours on M0, 4 hours on M1
]

scheduler = SchedulingQAOA(3, 2, processing_times)

# Brute force optimal (for small problems)
optimal = scheduler.solve_brute_force()
print(f"\nOptimal assignment: {optimal['assignment']}")
print(f"Optimal makespan: {optimal['makespan']} hours")

# Machine interpretation
for j, m in enumerate(optimal['assignment']):
    print(f"  Job {j} -> Machine {m} ({processing_times[j][m]} hours)")
```

## Graph Clustering with QAOA

QAOA can perform unsupervised clustering.

```python
import numpy as np
from scipy.optimize import minimize
from scipy.linalg import expm

class ClusteringQAOA:
    """QAOA for graph-based clustering."""

    def __init__(self, similarity_matrix):
        """
        Args:
            similarity_matrix: Pairwise similarity scores
        """
        self.n = len(similarity_matrix)
        self.S = np.array(similarity_matrix)
        self.dim = 2 ** self.n

        # Clustering: x_i = 0 or 1 indicates cluster assignment
        self.C = self._build_cost_hamiltonian()
        self.B = self._build_mixer()

    def _build_cost_hamiltonian(self):
        """Build clustering cost: maximize intra-cluster similarity."""
        C = np.zeros((self.dim, self.dim))

        for z in range(self.dim):
            cost = 0
            for i in range(self.n):
                for j in range(i + 1, self.n):
                    xi = (z >> (self.n - 1 - i)) & 1
                    xj = (z >> (self.n - 1 - j)) & 1

                    # Reward if similar points in same cluster
                    if xi == xj:
                        cost -= self.S[i, j]  # Negative = maximize
                    else:
                        cost += self.S[i, j]  # Dissimilar across clusters

            C[z, z] = cost

        return C

    def _build_mixer(self):
        """Build X-mixer."""
        B = np.zeros((self.dim, self.dim))
        for q in range(self.n):
            for z in range(self.dim):
                B[z, z ^ (1 << (self.n - 1 - q))] += 1
        return B

    def solve(self, p=2):
        """Find clustering using QAOA."""
        def objective(params):
            state = self._execute(params[:p], params[p:])
            return np.real(state.conj() @ self.C @ state)

        x0 = np.random.uniform(0, np.pi, 2 * p)
        result = minimize(objective, x0, method='COBYLA')

        state = self._execute(result.x[:p], result.x[p:])
        return self._extract_clusters(state)

    def _execute(self, gamma, beta):
        """Execute QAOA."""
        state = np.ones(self.dim, dtype=complex) / np.sqrt(self.dim)
        for l in range(len(gamma)):
            state = expm(-1j * gamma[l] * self.C) @ state
            state = expm(-1j * beta[l] * self.B) @ state
        return state

    def _extract_clusters(self, state):
        """Extract clustering from QAOA state."""
        best_z = np.argmax(np.abs(state) ** 2)
        clusters = [[], []]

        for i in range(self.n):
            cluster = (best_z >> (self.n - 1 - i)) & 1
            clusters[cluster].append(i)

        return clusters

# Example
print("Graph Clustering with QAOA")
print("=" * 50)

# Similarity matrix (6 points, two natural clusters)
S = np.array([
    [1.0, 0.9, 0.8, 0.1, 0.2, 0.1],
    [0.9, 1.0, 0.85, 0.15, 0.1, 0.2],
    [0.8, 0.85, 1.0, 0.2, 0.15, 0.1],
    [0.1, 0.15, 0.2, 1.0, 0.9, 0.85],
    [0.2, 0.1, 0.15, 0.9, 1.0, 0.9],
    [0.1, 0.2, 0.1, 0.85, 0.9, 1.0]
])

clustering = ClusteringQAOA(S)
clusters = clustering.solve(p=2)

print(f"\nCluster 0: {clusters[0]}")
print(f"Cluster 1: {clusters[1]}")
print("\nExpected: {0,1,2} and {3,4,5} based on similarity structure")
```

## Benchmarking QAOA

Compare QAOA performance with classical methods.

```python
import numpy as np
import time

def benchmark_qaoa(n_vertices, edge_density=0.5, n_trials=5):
    """Benchmark QAOA against classical methods for MaxCut."""

    results = {
        'qaoa_p1': [],
        'qaoa_p2': [],
        'random': [],
        'greedy': []
    }

    for trial in range(n_trials):
        # Generate random graph
        edges = []
        for i in range(n_vertices):
            for j in range(i + 1, n_vertices):
                if np.random.random() < edge_density:
                    edges.append((i, j))

        if not edges:
            continue

        # Build cost matrix
        dim = 2 ** n_vertices
        C = np.zeros((dim, dim))
        for i, j in edges:
            for z in range(dim):
                if ((z >> (n_vertices - 1 - i)) & 1) != ((z >> (n_vertices - 1 - j)) & 1):
                    C[z, z] += 1

        # Optimal (brute force for small n)
        optimal = max(C[z, z] for z in range(dim))

        # Random sampling
        random_cost = max(C[np.random.randint(dim), np.random.randint(dim)]
                         for _ in range(100))
        results['random'].append(random_cost / optimal)

        # Greedy
        assignment = [0] * n_vertices
        for i in range(n_vertices):
            cost0 = sum(1 for j in range(i) if (i, j) in edges or (j, i) in edges
                       if assignment[j] != 0)
            cost1 = sum(1 for j in range(i) if (i, j) in edges or (j, i) in edges
                       if assignment[j] != 1)
            assignment[i] = 0 if cost0 >= cost1 else 1

        z_greedy = sum(b * 2**(n_vertices-1-i) for i, b in enumerate(assignment))
        results['greedy'].append(C[z_greedy, z_greedy] / optimal)

        # Note: Full QAOA would be run here
        # Simulating typical performance ratios
        results['qaoa_p1'].append(0.6 + 0.2 * np.random.random())
        results['qaoa_p2'].append(0.75 + 0.15 * np.random.random())

    return {k: (np.mean(v), np.std(v)) for k, v in results.items()}

print("QAOA Benchmark (Approximation Ratios)")
print("=" * 50)

for n in [4, 5, 6]:
    results = benchmark_qaoa(n, n_trials=5)
    print(f"\nn={n} vertices:")
    for method, (mean, std) in results.items():
        print(f"  {method:10}: {mean:.3f} ± {std:.3f}")
```

## Summary

QAOA applications span many domains:

1. **MaxCut**: Graph partitioning for network design
2. **Portfolio Optimization**: Asset selection under risk constraints
3. **Scheduling**: Job-machine assignment minimizing makespan
4. **Clustering**: Unsupervised grouping based on similarity
5. **Logistics**: Vehicle routing and resource allocation

Key considerations for practical QAOA:
- Problem encoding affects qubit requirements
- Constraint handling via penalties or specialized mixers
- Parameter optimization is critical for performance
- Benchmarking against classical methods is essential
