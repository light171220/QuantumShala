# QNN Fundamentals

Quantum Neural Networks (QNNs) are parameterized quantum circuits designed to learn patterns from data. They combine the principles of quantum computing with machine learning.

## What is a Quantum Neural Network?

A QNN consists of a sequence of quantum gates with trainable parameters.

<Definition>
**Quantum Neural Network**:

A QNN is a parameterized unitary operation:

$$U(\theta) = \prod_{l=1}^{L} U_l(\theta_l) W_l$$

where $U_l(\theta_l)$ are parameterized gates and $W_l$ are fixed entangling gates.
</Definition>

```python
import numpy as np
from scipy.linalg import expm

class QuantumNeuralNetwork:
    """Basic quantum neural network implementation."""

    def __init__(self, n_qubits, n_layers):
        self.n_qubits = n_qubits
        self.n_layers = n_layers
        self.dim = 2 ** n_qubits

        # Parameters: 3 rotations per qubit per layer
        self.n_params = 3 * n_qubits * n_layers

    def _rx(self, theta, qubit):
        """Single-qubit RX gate."""
        U = np.eye(self.dim, dtype=complex)
        c, s = np.cos(theta/2), np.sin(theta/2)

        for z in range(self.dim):
            bit = (z >> (self.n_qubits - 1 - qubit)) & 1
            z_flip = z ^ (1 << (self.n_qubits - 1 - qubit))

            if z < z_flip:
                U[z, z] = c
                U[z, z_flip] = -1j * s
                U[z_flip, z] = -1j * s
                U[z_flip, z_flip] = c

        return U

    def _ry(self, theta, qubit):
        """Single-qubit RY gate."""
        U = np.eye(self.dim, dtype=complex)
        c, s = np.cos(theta/2), np.sin(theta/2)

        for z in range(self.dim):
            z_flip = z ^ (1 << (self.n_qubits - 1 - qubit))
            if z < z_flip:
                U[z, z] = c
                U[z, z_flip] = -s
                U[z_flip, z] = s
                U[z_flip, z_flip] = c

        return U

    def _rz(self, theta, qubit):
        """Single-qubit RZ gate."""
        U = np.eye(self.dim, dtype=complex)

        for z in range(self.dim):
            bit = (z >> (self.n_qubits - 1 - qubit)) & 1
            phase = np.exp(-1j * theta/2) if bit == 0 else np.exp(1j * theta/2)
            U[z, z] = phase

        return U

    def _cnot(self, control, target):
        """CNOT gate."""
        U = np.zeros((self.dim, self.dim), dtype=complex)

        for z in range(self.dim):
            ctrl_bit = (z >> (self.n_qubits - 1 - control)) & 1
            if ctrl_bit:
                z_new = z ^ (1 << (self.n_qubits - 1 - target))
            else:
                z_new = z
            U[z_new, z] = 1

        return U

    def forward(self, params, input_state=None):
        """Execute the QNN."""
        if input_state is None:
            state = np.zeros(self.dim, dtype=complex)
            state[0] = 1  # |0...0⟩
        else:
            state = input_state.copy()

        param_idx = 0
        for layer in range(self.n_layers):
            # Rotation layer
            for q in range(self.n_qubits):
                state = self._rx(params[param_idx], q) @ state
                param_idx += 1
                state = self._ry(params[param_idx], q) @ state
                param_idx += 1
                state = self._rz(params[param_idx], q) @ state
                param_idx += 1

            # Entangling layer
            for q in range(self.n_qubits - 1):
                state = self._cnot(q, q + 1) @ state

        return state

# Example
qnn = QuantumNeuralNetwork(n_qubits=3, n_layers=2)
print(f"QNN with {qnn.n_qubits} qubits, {qnn.n_layers} layers")
print(f"Number of parameters: {qnn.n_params}")

params = np.random.uniform(0, 2*np.pi, qnn.n_params)
state = qnn.forward(params)

print(f"\nOutput state probabilities:")
probs = np.abs(state)**2
for z in range(8):
    if probs[z] > 0.01:
        print(f"  |{format(z, '03b')}⟩: {probs[z]:.4f}")
```

## Classical vs Quantum Neural Networks

Understanding the key differences helps design effective QNNs.

<Tip>
QNNs operate in exponentially large Hilbert space but have limited access through measurement. This is both their power and limitation.
</Tip>

```python
import numpy as np

def compare_nn_architectures():
    """Compare classical and quantum neural network properties."""

    comparison = {
        'State Space': {
            'Classical': 'Real-valued vectors in R^n',
            'Quantum': 'Complex vectors in C^(2^n), unit norm'
        },
        'Operations': {
            'Classical': 'Matrix multiplication + nonlinearity',
            'Quantum': 'Unitary transformations (reversible)'
        },
        'Parameters': {
            'Classical': 'O(n^2) for dense layer',
            'Quantum': 'O(n) for single layer'
        },
        'Nonlinearity': {
            'Classical': 'ReLU, sigmoid, tanh (explicit)',
            'Quantum': 'Measurement (probabilistic)'
        },
        'Expressibility': {
            'Classical': 'Universal with enough width/depth',
            'Quantum': 'Universal with enough depth'
        },
        'Training': {
            'Classical': 'Backpropagation',
            'Quantum': 'Parameter shift rule, finite differences'
        }
    }

    print("Classical vs Quantum Neural Networks")
    print("=" * 60)

    for aspect, values in comparison.items():
        print(f"\n{aspect}:")
        print(f"  Classical: {values['Classical']}")
        print(f"  Quantum:   {values['Quantum']}")

compare_nn_architectures()
```

## Expressibility

Expressibility measures how well a QNN can explore the space of quantum states.

<Formula>
**Expressibility Measure**:

$$\mathcal{E} = D_{KL}(P_{QNN}(\theta) \| P_{Haar})$$

where $P_{QNN}$ is the distribution of states from the QNN and $P_{Haar}$ is the uniform (Haar) distribution.
</Formula>

```python
import numpy as np

def compute_expressibility(qnn, n_samples=1000):
    """Estimate QNN expressibility via state overlap statistics."""

    fidelities = []

    for _ in range(n_samples):
        # Two random parameter sets
        params1 = np.random.uniform(0, 2*np.pi, qnn.n_params)
        params2 = np.random.uniform(0, 2*np.pi, qnn.n_params)

        state1 = qnn.forward(params1)
        state2 = qnn.forward(params2)

        # Fidelity between states
        fidelity = np.abs(np.vdot(state1, state2))**2
        fidelities.append(fidelity)

    fidelities = np.array(fidelities)

    # Compare to Haar distribution
    # For Haar random states, mean fidelity = 1/(d+1) where d = 2^n
    d = qnn.dim
    haar_mean = 1 / (d + 1)

    expressibility = {
        'mean_fidelity': np.mean(fidelities),
        'haar_mean': haar_mean,
        'std_fidelity': np.std(fidelities),
        'expressibility_score': 1 - abs(np.mean(fidelities) - haar_mean) / haar_mean
    }

    return expressibility

# Compare different QNN depths
print("Expressibility vs Circuit Depth")
print("=" * 50)

for n_layers in [1, 2, 4, 8]:
    qnn = QuantumNeuralNetwork(n_qubits=4, n_layers=n_layers)
    expr = compute_expressibility(qnn, n_samples=500)

    print(f"\nLayers={n_layers}:")
    print(f"  Mean fidelity: {expr['mean_fidelity']:.4f}")
    print(f"  Haar target:   {expr['haar_mean']:.4f}")
    print(f"  Score:         {expr['expressibility_score']:.3f}")
```

## Entangling Capability

Entanglement is crucial for QNN expressivity beyond classical simulation.

```python
import numpy as np

def compute_entanglement_entropy(state, n_qubits, partition):
    """Compute entanglement entropy for bipartition."""
    n_A = len(partition)
    n_B = n_qubits - n_A
    dim_A = 2 ** n_A
    dim_B = 2 ** n_B

    # Reshape to bipartite form
    # This is simplified - proper implementation needs qubit reordering
    rho = np.outer(state, state.conj())

    # Partial trace over subsystem B
    rho_A = np.zeros((dim_A, dim_A), dtype=complex)

    complement = [q for q in range(n_qubits) if q not in partition]

    for i in range(dim_A):
        for j in range(dim_A):
            for k in range(dim_B):
                # Construct indices
                idx1 = 0
                idx2 = 0
                for q in range(n_qubits):
                    if q in partition:
                        pos = partition.index(q)
                        idx1 |= ((i >> (n_A - 1 - pos)) & 1) << (n_qubits - 1 - q)
                        idx2 |= ((j >> (n_A - 1 - pos)) & 1) << (n_qubits - 1 - q)
                    else:
                        pos = complement.index(q)
                        idx1 |= ((k >> (n_B - 1 - pos)) & 1) << (n_qubits - 1 - q)
                        idx2 |= ((k >> (n_B - 1 - pos)) & 1) << (n_qubits - 1 - q)

                rho_A[i, j] += rho[idx1, idx2]

    # Von Neumann entropy
    eigenvalues = np.linalg.eigvalsh(rho_A)
    eigenvalues = eigenvalues[eigenvalues > 1e-12]
    entropy = -np.sum(eigenvalues * np.log2(eigenvalues))

    return entropy

def analyze_entanglement_capability(qnn, n_samples=100):
    """Analyze entanglement capability of QNN."""

    entropies = []
    partition = list(range(qnn.n_qubits // 2))

    for _ in range(n_samples):
        params = np.random.uniform(0, 2*np.pi, qnn.n_params)
        state = qnn.forward(params)
        entropy = compute_entanglement_entropy(state, qnn.n_qubits, partition)
        entropies.append(entropy)

    max_entropy = min(len(partition), qnn.n_qubits - len(partition))

    return {
        'mean_entropy': np.mean(entropies),
        'max_entropy': max_entropy,
        'entangling_power': np.mean(entropies) / max_entropy
    }

print("\nEntanglement Capability Analysis")
print("=" * 50)

qnn = QuantumNeuralNetwork(n_qubits=4, n_layers=3)
ent = analyze_entanglement_capability(qnn, n_samples=100)

print(f"Mean entanglement entropy: {ent['mean_entropy']:.3f}")
print(f"Maximum possible entropy:  {ent['max_entropy']:.3f}")
print(f"Entangling power:          {ent['entangling_power']:.3f}")
```

## QNN Architectures

Common QNN architectures for different applications.

```python
import numpy as np

class QNNArchitectures:
    """Collection of QNN architecture patterns."""

    @staticmethod
    def hardware_efficient(n_qubits, n_layers):
        """Hardware-efficient ansatz."""
        structure = []
        for l in range(n_layers):
            # Single-qubit rotations
            for q in range(n_qubits):
                structure.append(('RY', q, f'θ_{l}_{q}_1'))
                structure.append(('RZ', q, f'θ_{l}_{q}_2'))

            # Linear entanglement
            for q in range(n_qubits - 1):
                structure.append(('CNOT', q, q + 1))

        return structure

    @staticmethod
    def strongly_entangling(n_qubits, n_layers):
        """Strongly entangling ansatz."""
        structure = []
        for l in range(n_layers):
            # Full rotation layer
            for q in range(n_qubits):
                structure.append(('RX', q, f'θ_{l}_{q}_1'))
                structure.append(('RY', q, f'θ_{l}_{q}_2'))
                structure.append(('RZ', q, f'θ_{l}_{q}_3'))

            # All-to-all entanglement
            for q1 in range(n_qubits):
                for q2 in range(q1 + 1, n_qubits):
                    structure.append(('CZ', q1, q2))

        return structure

    @staticmethod
    def alternating(n_qubits, n_layers):
        """Alternating layered ansatz."""
        structure = []
        for l in range(n_layers):
            # Rotation layer
            for q in range(n_qubits):
                structure.append(('RY', q, f'θ_{l}_{q}'))

            # Alternating entanglement
            if l % 2 == 0:
                pairs = [(i, i+1) for i in range(0, n_qubits-1, 2)]
            else:
                pairs = [(i, i+1) for i in range(1, n_qubits-1, 2)]

            for q1, q2 in pairs:
                structure.append(('CNOT', q1, q2))

        return structure

# Display architectures
print("QNN Architecture Patterns")
print("=" * 50)

for name, builder in [
    ('Hardware-Efficient', QNNArchitectures.hardware_efficient),
    ('Strongly-Entangling', QNNArchitectures.strongly_entangling),
    ('Alternating', QNNArchitectures.alternating)
]:
    structure = builder(n_qubits=4, n_layers=2)
    n_params = sum(1 for op in structure if op[0] in ['RX', 'RY', 'RZ'])
    n_entangling = sum(1 for op in structure if op[0] in ['CNOT', 'CZ'])

    print(f"\n{name}:")
    print(f"  Parameters: {n_params}")
    print(f"  Entangling gates: {n_entangling}")
    print(f"  First 5 operations: {structure[:5]}")
```

## Summary

Key concepts in Quantum Neural Networks:

1. **Architecture**: Parameterized unitaries with entangling layers
2. **Expressibility**: How well the circuit explores state space
3. **Entanglement**: Enables quantum correlations beyond classical
4. **Parameter efficiency**: QNNs can have fewer parameters than classical NNs
5. **Training**: Requires quantum-specific gradient methods
