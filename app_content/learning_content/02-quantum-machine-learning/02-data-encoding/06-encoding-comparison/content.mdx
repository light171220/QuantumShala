Quantitative comparison of resource requirements.

```python
import numpy as np

def resource_comparison(n_values, n_features):
    """
    Compare resource requirements for encoding n_values data points
    with n_features each.
    """
    print(f"Resource Comparison: {n_values} values, {n_features} features")
    print("=" * 60)

    comparisons = {
        "Basis Encoding": {
            "qubits": int(np.ceil(np.log2(n_values))),
            "gates_per_point": int(np.ceil(np.log2(n_values))),  # Worst case
            "can_superpose": True,
            "parallel_encoding": False
        },
        "Amplitude Encoding": {
            "qubits": int(np.ceil(np.log2(n_values))),
            "gates_per_point": n_values,  # General case
            "can_superpose": True,
            "parallel_encoding": False
        },
        "Angle Encoding (parallel)": {
            "qubits": n_features,
            "gates_per_point": n_features,
            "can_superpose": False,
            "parallel_encoding": True
        },
        "Angle Encoding (sequential)": {
            "qubits": 1,
            "gates_per_point": n_features,
            "can_superpose": False,
            "parallel_encoding": False
        },
        "Kernel (ZZ-map)": {
            "qubits": n_features,
            "gates_per_point": n_features * 2 + n_features * (n_features - 1) // 2,
            "can_superpose": False,
            "parallel_encoding": True
        }
    }

    print(f"\n{'Method':<25} {'Qubits':<10} {'Gates':<10} {'Superpose':<12}")
    print("-" * 60)

    for method, stats in comparisons.items():
        print(f"{method:<25} {stats['qubits']:<10} {stats['gates_per_point']:<10} "
              f"{'Yes' if stats['can_superpose'] else 'No':<12}")

    return comparisons

# Example: 1024 data values, 10 features each
resource_comparison(1024, 10)
```

```python
def data_reuploading_circuit(x, params, n_layers=2):
    """
    Simple data re-uploading circuit.

    Alternates between data encoding and variational layers.
    """
    state = np.array([1, 0], dtype=complex)

    for layer in range(n_layers):
        # Data encoding layer
        theta_data = np.pi * x
        Ry_data = np.array([
            [np.cos(theta_data/2), -np.sin(theta_data/2)],
            [np.sin(theta_data/2), np.cos(theta_data/2)]
        ])
        state = Ry_data @ state

        # Variational layer
        theta_var = params[layer]
        Rz_var = np.array([
            [np.exp(-1j * theta_var / 2), 0],
            [0, np.exp(1j * theta_var / 2)]
        ])
        state = Rz_var @ state

    # Expectation value of Z
    prob_0 = np.abs(state[0]) ** 2
    return 2 * prob_0 - 1  # Maps to [-1, 1]

# Demonstrate

    output = data_reuploading_circuit(x, params)
    print(f"  x={x:.2f} → {output:+.4f}")
```

<Tip>
There's no universally best encoding method. The optimal choice depends on your data, hardware, algorithm, and performance goals. Start with the simplest encoding that could work, then iterate based on results.
</Tip>

## Summary

In this lesson, you learned to compare and choose data encodings:

1. **Basis Encoding**: Best for discrete data, simple circuits
2. **Amplitude Encoding**: Best for dense vectors, exponential compression
3. **Angle Encoding**: Best for NISQ, variational algorithms
4. **Kernel Encoding**: Best for classification, feature maps
5. **Hybrid Strategies**: Combine methods for complex tasks
6. **Decision Guide**: Match encoding to data, hardware, and goals

<Warning>
Encoding choice can make or break your QML application. An encoding that's theoretically optimal may be impractical on real hardware. Always validate your choice with experiments on representative data.
</Warning>
