## 1. Computational Basis States

Basis encoding stores classical information directly in the computational basis.

<Definition term="Computational Basis">
The set of orthonormal states |0...0⟩, |0...1⟩, ..., |1...1⟩ that form the standard measurement basis for n qubits. There are 2ⁿ basis states for n qubits.
</Definition>

```python
import numpy as np

def show_computational_basis(n_qubits):
    """Display all computational basis states for n qubits."""
    n_states = 2 ** n_qubits

    print(f"Computational Basis for {n_qubits} qubits:")
    print("=" * 40)
    print(f"{'Index':<8} {'Binary':<12} {'Ket Notation':<15}")
    print("-" * 40)

    for i in range(n_states):
        binary = format(i, f'0{n_qubits}b')
        ket = f"|{binary}⟩"
        print(f"{i:<8} {binary:<12} {ket:<15}")

show_computational_basis(3)
```

<Formula>
$$|i\rangle = |b_{n-1}b_{n-2}...b_1b_0\rangle \text{ where } i = \sum_{k=0}^{n-1} b_k \cdot 2^k$$
</Formula>

## 2. Encoding Binary Strings

The simplest form of basis encoding: directly map bit strings to qubit states.

```python
import numpy as np

def binary_to_state(binary_string):
    """
    Convert a binary string to a quantum state vector.

    Args:
        binary_string: String of 0s and 1s (e.g., "101")

    Returns:
        State vector as numpy array
    """
    n_qubits = len(binary_string)
    dim = 2 ** n_qubits

    # Convert binary to integer index
    index = int(binary_string, 2)

    # Create state vector with 1 at the index position
    state = np.zeros(dim)
    state[index] = 1.0

    return state

def state_to_binary(state, n_qubits):
    """Convert a basis state back to binary string."""
    index = np.argmax(np.abs(state))
    return format(index, f'0{n_qubits}b')

# Examples
print("Binary String to Quantum State:")
print("=" * 50)

test_strings = ["00", "01", "10", "11", "101", "110"]
for bs in test_strings:
    state = binary_to_state(bs)
    recovered = state_to_binary(state, len(bs))
    print(f"|{bs}⟩ → index {int(bs, 2)}, recovered: {recovered}")
```

## 3. Encoding Integers

Integers are encoded by their binary representation.

```python
import numpy as np

def encode_integer(value, n_qubits=None):
    """
    Encode an integer into a quantum basis state.

    Args:
        value: Non-negative integer to encode
        n_qubits: Number of qubits (auto-calculated if None)

    Returns:
        Tuple of (state_vector, n_qubits_used)
    """
    if value < 0:
        raise ValueError("Cannot encode negative integers")

    # Calculate minimum qubits needed
    if n_qubits is None:
        n_qubits = max(1, int(np.ceil(np.log2(value + 1))))

    if value >= 2 ** n_qubits:
        raise ValueError(f"Value {value} requires more than {n_qubits} qubits")

    dim = 2 ** n_qubits
    state = np.zeros(dim)
    state[value] = 1.0

    return state, n_qubits

# Encoding examples
print("Integer Encoding:")
print("=" * 50)
print(f"{'Value':<10} {'Qubits':<10} {'Binary':<15} {'State Index':<12}")
print("-" * 50)

for value in [0, 1, 5, 7, 15, 42]:
    try:
        state, n_q = encode_integer(value)
        binary = format(value, f'0{n_q}b')
        idx = np.argmax(state)
        print(f"{value:<10} {n_q:<10} {binary:<15} {idx:<12}")
    except ValueError as e:
        print(f"{value:<10} Error: {e}")
```

<Tip>
Basis encoding is efficient in qubits: log₂(N) qubits can represent N distinct values. This exponential compression is one of quantum computing's key features.
</Tip>

## 4. Encoding Circuits

Building circuits that prepare basis-encoded states.

```python
import numpy as np

def basis_encoding_circuit(binary_string):
    """
    Generate the gates needed to encode a binary string.

    Starting from |0...0⟩, apply X gates where bits are 1.

    Args:
        binary_string: Target binary string

    Returns:
        List of (gate, qubit) operations
    """
    operations = []
    n_qubits = len(binary_string)

    for i, bit in enumerate(binary_string):
        if bit == '1':
            # Apply X gate to flip |0⟩ to |1⟩
            operations.append(('X', i))

    return operations

def simulate_encoding(binary_string):
    """Simulate the encoding circuit."""
    n = len(binary_string)

    # Start with |0...0⟩
    state = np.zeros(2 ** n)
    state[0] = 1.0

    # X gate matrix
    X = np.array([[0, 1], [1, 0]])
    I = np.eye(2)

    # Apply X gates where needed
    for i, bit in enumerate(binary_string):
        if bit == '1':
            # Build full operator
            op = np.array([[1.0]])
            for j in range(n):
                if j == i:
                    op = np.kron(op, X)
                else:
                    op = np.kron(op, I)
            state = op @ state

    return state

# Example
print("Basis Encoding Circuit:")
print("=" * 50)

target = "101"
ops = basis_encoding_circuit(target)
print(f"Target state: |{target}⟩")
print(f"Operations from |000⟩:")
for gate, qubit in ops:
    print(f"  {gate} on qubit {qubit}")

state = simulate_encoding(target)
print(f"\nResulting state vector:")
print(f"  Non-zero at index {np.argmax(state)} = {int(target, 2)}")
```

## 5. Superposition of Values

Basis encoding enables creating superpositions over multiple values.

```python
import numpy as np

def create_superposition(values, n_qubits):
    """
    Create equal superposition over multiple integer values.

    Args:
        values: List of integers to superpose
        n_qubits: Number of qubits

    Returns:
        Normalized quantum state
    """
    dim = 2 ** n_qubits
    state = np.zeros(dim, dtype=complex)

    for v in values:
        if 0 <= v < dim:
            state[v] = 1.0

    # Normalize
    state = state / np.linalg.norm(state)
    return state

def uniform_superposition(n_qubits):
    """Create uniform superposition over all basis states (H⊗n|0⟩)."""
    dim = 2 ** n_qubits
    return np.ones(dim) / np.sqrt(dim)

# Examples
print("Superposition States:")
print("=" * 50)

# Superposition of specific values
values = [0, 2, 4, 6]  # Even numbers 0-7
state = create_superposition(values, 3)
print(f"\nSuperposition of {values}:")
print(f"Amplitudes: {np.round(state, 4)}")
print(f"Probabilities: {np.round(np.abs(state)**2, 4)}")

# Uniform superposition
uniform = uniform_superposition(3)
print(f"\nUniform superposition (H⊗3|000⟩):")
print(f"All amplitudes: {uniform[0]:.4f}")
print(f"All probabilities: {np.abs(uniform[0])**2:.4f}")
```

<Warning>
Creating arbitrary superpositions requires careful circuit design. While uniform superposition is easy (just Hadamard gates), specific value superpositions may need more complex preparation.
</Warning>

```python
def basis_encoding_limitations():
    """Discuss limitations of basis encoding."""

    print("Basis Encoding Limitations:")
    print("=" * 50)

    limitations = [
        {
            "issue": "Discrete data only",
            "detail": "Cannot directly encode continuous values",
            "workaround": "Discretize continuous data first"

            "detail": "Single basis state has no learnable structure",
            "workaround": "Use superpositions or combine with other encodings"
```

In this lesson, you learned about basis encoding:

2. **Binary Encoding**: Direct mapping of bit strings to qubit states

</Tip>
