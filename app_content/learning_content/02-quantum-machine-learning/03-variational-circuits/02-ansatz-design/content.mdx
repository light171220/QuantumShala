# Ansatz Design Principles

The ansatz (parameterized quantum circuit) is the heart of any variational algorithm. Its design critically impacts both the algorithm's expressiveness and trainability.

## Learning Objectives

After completing this lesson, you will be able to:

- Understand ansatz design tradeoffs
- Learn common ansatz structures
- Evaluate expressibility vs trainability
- Implement different ansatz types
- Choose appropriate ansatz for problems

## 1. Ansatz Design Fundamentals

An ansatz must balance expressibility (ability to represent target states) with trainability (ease of optimization).

<Definition term="Ansatz">
A parameterized quantum circuit U(θ) that maps input states to output states. The parameters θ are optimized to minimize a cost function. The choice of ansatz structure determines which states can be reached.
</Definition>

## 2. Common Ansatz Structures

Several standard ansatz structures have emerged for different applications.

```python
import numpy as np

class AnsatzLibrary:
    """
    Library of common ansatz structures.
    """

    def __init__(self, n_qubits):
        self.n_qubits = n_qubits
        self.dim = 2**n_qubits

    def hardware_efficient(self, params, n_layers=2):
        """
        Hardware-efficient ansatz: alternating rotation and entanglement layers.
        """
        state = np.zeros(self.dim, dtype=complex)
        state[0] = 1.0

        param_idx = 0
        for layer in range(n_layers):
            # Rotation layer
            for q in range(self.n_qubits):
                state = self._ry(state, q, params[param_idx])
                param_idx += 1
                state = self._rz(state, q, params[param_idx])
                param_idx += 1

            # Entanglement layer (linear connectivity)
            for q in range(self.n_qubits - 1):
                state = self._cnot(state, q, q + 1)

        return state

    def strongly_entangling(self, params, n_layers=2):
        """
        Strongly entangling ansatz with all-to-all connectivity pattern.
        """
        state = np.zeros(self.dim, dtype=complex)
        state[0] = 1.0

        param_idx = 0
        for layer in range(n_layers):
            # Full rotation layer
            for q in range(self.n_qubits):
                state = self._rx(state, q, params[param_idx])
                param_idx += 1
                state = self._ry(state, q, params[param_idx])
                param_idx += 1
                state = self._rz(state, q, params[param_idx])
                param_idx += 1

            # Entanglement with offset
            offset = layer % self.n_qubits
            for q in range(self.n_qubits):
                target = (q + 1 + offset) % self.n_qubits
                if q != target:
                    state = self._cnot(state, q, target)

        return state

    def _ry(self, state, qubit, theta):
        c, s = np.cos(theta/2), np.sin(theta/2)
        new_state = np.zeros_like(state)
        for i in range(self.dim):
            bit = (i >> (self.n_qubits - 1 - qubit)) & 1
            partner = i ^ (1 << (self.n_qubits - 1 - qubit))
            if bit == 0:
                new_state[i] += c * state[i] - s * state[partner]
                new_state[partner] += s * state[i] + c * state[partner]
        return new_state

    def _rz(self, state, qubit, theta):
        new_state = state.copy()
        for i in range(self.dim):
            bit = (i >> (self.n_qubits - 1 - qubit)) & 1
            phase = np.exp(-1j * theta / 2) if bit == 0 else np.exp(1j * theta / 2)
            new_state[i] *= phase
        return new_state

    def _rx(self, state, qubit, theta):
        c, s = np.cos(theta/2), np.sin(theta/2)
        new_state = np.zeros_like(state)
        for i in range(self.dim):
            bit = (i >> (self.n_qubits - 1 - qubit)) & 1
            partner = i ^ (1 << (self.n_qubits - 1 - qubit))
            if bit == 0:
                new_state[i] += c * state[i] - 1j * s * state[partner]
                new_state[partner] += -1j * s * state[i] + c * state[partner]
        return new_state

    def _cnot(self, state, control, target):
        new_state = state.copy()
        for i in range(self.dim):
            ctrl_bit = (i >> (self.n_qubits - 1 - control)) & 1
            if ctrl_bit == 1:
                partner = i ^ (1 << (self.n_qubits - 1 - target))
                new_state[i], new_state[partner] = state[partner], state[i]
        return new_state

# Demonstration
print("Common Ansatz Structures:")
print("=" * 50)

n_qubits = 3
lib = AnsatzLibrary(n_qubits)

# Hardware-efficient ansatz
n_layers = 2
n_params_hw = n_qubits * 2 * n_layers
params_hw = np.random.uniform(0, 2*np.pi, n_params_hw)

print(f"\n1. Hardware-Efficient Ansatz:")
print(f"   Qubits: {n_qubits}, Layers: {n_layers}")
print(f"   Parameters: {n_params_hw}")

print(f"   State norm: {np.linalg.norm(state_hw):.6f}")

# Strongly entangling ansatz

print(f"   Qubits: {n_qubits}, Layers: {n_layers}")
```

## 3. Expressibility Analysis

Measuring how well an ansatz can explore the state space.

```python
import numpy as np

def measure_expressibility(ansatz_fn, n_params, n_samples=500):
    """
    Measure ansatz expressibility using state fidelity distribution.
    """
    fidelities = []

    for _ in range(n_samples):
        # Generate two random parameter sets
        params1 = np.random.uniform(0, 2*np.pi, n_params)
        params2 = np.random.uniform(0, 2*np.pi, n_params)

        state1 = ansatz_fn(params1)
        state2 = ansatz_fn(params2)

        # Compute fidelity
        fidelity = np.abs(np.vdot(state1, state2))**2
        fidelities.append(fidelity)

    return np.array(fidelities)

def expressibility_comparison():
    """
    Compare expressibility of different ansätze.
    """
    print("Expressibility Analysis:")
    print("=" * 50)

    n_qubits = 2
    lib = AnsatzLibrary(n_qubits)

    # Define ansätze with different depths
    ansatze = {
        'HW-1 layer': (lambda p: lib.hardware_efficient(p, 1), 4),
        'HW-2 layers': (lambda p: lib.hardware_efficient(p, 2), 8),
        'HW-4 layers': (lambda p: lib.hardware_efficient(p, 4), 16),
        'SE-2 layers': (lambda p: lib.strongly_entangling(p, 2), 12),
    }

    # Haar random baseline (for 2 qubits)
    def haar_fidelity_mean():
        """Mean fidelity for Haar-random states in d-dim space."""
        d = 2**n_qubits
        return 1.0 / d

    print(f"\nHaar random mean fidelity: {haar_fidelity_mean():.4f}")

    print("\nAnsatz expressibility (mean fidelity → lower is more expressive):")
    print("-" * 50)

    for name, (ansatz_fn, n_params) in ansatze.items():
        fidelities = measure_expressibility(ansatz_fn, n_params, n_samples=200)
        mean_f = np.mean(fidelities)
        std_f = np.std(fidelities)
        print(f"  {name:15s}: mean = {mean_f:.4f} ± {std_f:.4f}")

expressibility_comparison()
```

<Warning>
High expressibility can lead to barren plateaus where gradients vanish exponentially. Balance expressibility with trainability.
</Warning>

## 4. Entanglement Patterns

Different entanglement structures suit different problems.

```python
import numpy as np

def entanglement_patterns():
    """
    Explore different entanglement patterns.
    """
    print("Entanglement Patterns:")
    print("=" * 50)

    patterns = [
        {
            'name': 'Linear (Nearest-neighbor)',
            'description': 'Each qubit connects to adjacent qubits',
            'pros': 'Hardware-friendly, shallow depth',
            'cons': 'Limited long-range correlations',
            'gates': 'CZ/CNOT between (0,1), (1,2), ...'
        },
        {
            'name': 'Circular',
            'description': 'Linear plus connection between first and last',
            'pros': 'Periodic boundary conditions',
            'cons': 'May need SWAP gates on some hardware',
            'gates': 'Linear + (n-1, 0)'
        },
        {
            'name': 'All-to-all',
            'description': 'Every qubit connects to every other',
            'pros': 'Maximum entanglement capability',
            'cons': 'Requires many SWAP gates on limited connectivity',
            'gates': 'All pairs (i,j) for i < j'
        },
        {
            'name': 'Ladder',
            'description': 'Two chains with cross-connections',
            'pros': 'Good for 2D lattice problems',
            'cons': 'Requires even number of qubits',
            'gates': '2 parallel chains with rungs'
        }
    ]

    for p in patterns:
        print(f"\n{p['name']}:")
        print(f"  {p['description']}")
        print(f"  Pros: {p['pros']}")
        print(f"  Cons: {p['cons']}")

    # Demonstrate entanglement entropy for different patterns
    print("\n\nEntanglement Entropy Comparison:")
    print("-" * 40)

    def compute_entanglement_entropy(state, n_qubits):
        """Compute bipartite entanglement entropy."""
        mid = n_qubits // 2
        dim_a = 2**mid
        dim_b = 2**(n_qubits - mid)

        # Reshape and compute Schmidt values
        reshaped = state.reshape(dim_a, dim_b)
        schmidt = np.linalg.svd(reshaped, compute_uv=False)
        schmidt = schmidt[schmidt > 1e-10]

        entropy = -np.sum(schmidt**2 * np.log2(schmidt**2 + 1e-15))
        return entropy

    n_qubits = 4
    lib = AnsatzLibrary(n_qubits)

    # Compare entanglement for random parameters
    np.random.seed(42)
    params = np.random.uniform(0, 2*np.pi, 24)

    for n_layers in [1, 2, 4]:
        n_params = n_qubits * 2 * n_layers
        state = lib.hardware_efficient(params[:n_params], n_layers)
        entropy = compute_entanglement_entropy(state, n_qubits)
        max_entropy = n_qubits // 2
        print(f"  {n_layers} layers: S = {entropy:.4f} (max = {max_entropy})")
```

## 5. Problem-Specific Ansätze

Designing ansätze that respect problem structure.

```python
import numpy as np

def problem_specific_ansatze():
    """
    Discuss problem-specific ansatz design.
    """
    print("Problem-Specific Ansatz Design:")
    print("=" * 50)

    ansatz_types = [
        {
            'name': 'UCCSD (Chemistry)',
            'application': 'Molecular ground states',
            'structure': 'Unitary Coupled Cluster with Singles and Doubles',
            'key_feature': 'Preserves particle number and spin symmetry'
        },
        {
            'name': 'QAOA (Optimization)',
            'application': 'Combinatorial optimization',
            'structure': 'Alternating mixer and problem Hamiltonians',
            'key_feature': 'Problem structure encoded in circuit'
        },
        {
            'name': 'HVA (Condensed Matter)',
            'application': 'Many-body physics',
            'structure': 'Hamiltonian Variational Ansatz',
            'key_feature': 'Matches Hamiltonian terms'
        },
        {
            'name': 'IQP (Machine Learning)',
            'application': 'Classification/sampling',
            'structure': 'Instantaneous Quantum Polynomial',
            'key_feature': 'Classically hard to simulate'
        }
    ]

    for ansatz in ansatz_types:
        print(f"\n{ansatz['name']}:")
        print(f"  Application: {ansatz['application']}")
        print(f"  Structure: {ansatz['structure']}")
        print(f"  Key feature: {ansatz['key_feature']}")

    # Simple QAOA-style ansatz demonstration
    print("\n\nQAOA-style Ansatz Example:")
    print("-" * 40)

    def qaoa_layer(state, gamma, beta, n_qubits):
        """Simple QAOA layer for Max-Cut on a ring."""
        # Problem unitary (ZZ interactions)
        for q in range(n_qubits):
            next_q = (q + 1) % n_qubits
            # Simplified ZZ rotation
            for i in range(len(state)):
                b1 = (i >> (n_qubits - 1 - q)) & 1
                b2 = (i >> (n_qubits - 1 - next_q)) & 1
                parity = 1 if b1 == b2 else -1
                state[i] *= np.exp(-1j * gamma * parity)

        # Mixer unitary (X rotations)
        lib = AnsatzLibrary(n_qubits)
        for q in range(n_qubits):
            state = lib._rx(state, q, 2 * beta)

        return state

    n_qubits = 4
    n_layers = 2

    # Initialize in uniform superposition
    state = np.ones(2**n_qubits, dtype=complex) / np.sqrt(2**n_qubits)

    gammas = [0.3, 0.5]
    betas = [0.4, 0.2]

    for p in range(n_layers):
        state = qaoa_layer(state, gammas[p], betas[p], n_qubits)

    probs = np.abs(state)**2
    print(f"  Top 3 measurement outcomes:")
    top_indices = np.argsort(probs)[-3:][::-1]
    for idx in top_indices:
        bitstring = format(idx, f'0{n_qubits}b')
        print(f"    |{bitstring}⟩: {probs[idx]:.4f}")

problem_specific_ansatze()
```

<Tip>
When possible, incorporate known problem symmetries into your ansatz. This reduces the parameter space and often improves trainability.
</Tip>

## Summary

Effective ansatz design requires balancing multiple factors:

1. **Expressibility**: Ability to reach target states
2. **Trainability**: Gradient behavior during optimization
3. **Entanglement**: Pattern matching problem structure
4. **Depth**: Hardware constraints and noise
5. **Problem Structure**: Leveraging symmetries

The optimal ansatz depends on the specific problem and available hardware.
