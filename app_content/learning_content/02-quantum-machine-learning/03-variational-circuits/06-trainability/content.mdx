# Trainability of Variational Circuits

While expressibility determines what states an ansatz can reach, trainability determines whether we can find the optimal parameters through gradient-based optimization.

## Learning Objectives

After completing this lesson, you will be able to:

- Define and measure trainability
- Identify trainability issues
- Understand gradient variance scaling
- Apply trainability-preserving strategies
- Balance expressibility and trainability

## 1. What is Trainability?

Trainability measures how effectively we can optimize circuit parameters using gradients.

<Definition term="Trainability">
The degree to which gradient-based optimization can effectively navigate the loss landscape to find good solutions. A trainable circuit has non-vanishing gradients that provide useful optimization directions.
</Definition>

## 2. Gradient Variance Analysis

The variance of gradients is a key indicator of trainability.

```python
import numpy as np

def gradient_variance_analysis():
    """
    Analyze gradient variance for different circuits.
    """
    print("Gradient Variance Analysis:")
    print("=" * 50)

    def compute_gradient_variance(gradient_fn, n_params, n_samples=100):
        """Sample gradient variance."""
        gradients = []
        for _ in range(n_samples):
            params = np.random.uniform(0, 2*np.pi, n_params)
            grad = gradient_fn(params)
            gradients.append(grad)

        gradients = np.array(gradients)
        return np.var(gradients, axis=0)

    # Simulated gradient functions
    def shallow_gradient(params):
        # Shallow circuit: reasonable gradients
        return np.sin(params) * 0.5

    def deep_gradient(params):
        # Deep circuit: exponentially small gradients
        n = len(params)
        return np.sin(params) * (0.5 ** (n/4))

    print("\n1. Gradient Variance Scaling:")
    print("-" * 40)

    print(f"\n  {'N params':<12} {'Shallow Var':<15} {'Deep Var':<15}")
    print("  " + "-" * 40)

    for n_params in [4, 8, 12, 16, 20]:
        var_shallow = compute_gradient_variance(shallow_gradient, n_params)
        var_deep = compute_gradient_variance(deep_gradient, n_params)

        mean_var_shallow = np.mean(var_shallow)
        mean_var_deep = np.mean(var_deep)

        print(f"  {n_params:<12} {mean_var_shallow:<15.6f} {mean_var_deep:<15.6f}")

    print("\n\n2. Implications:")
    print("-" * 40)
    print("  • Shallow: Variance stays reasonable")
    print("  • Deep: Variance shrinks exponentially")
    print("  • Vanishing variance = untrainable")

    # Threshold analysis
    print("\n\n3. Practical Thresholds:")
    print("-" * 40)

    thresholds = [
        (1e-2, "Good trainability"),
        (1e-4, "Marginal trainability"),
        (1e-6, "Poor trainability"),
        (1e-8, "Essentially untrainable")
    ]

    for threshold, label in thresholds:
        print(f"  Var > {threshold:.0e}: {label}")

gradient_variance_analysis()
```

## 3. Barren Plateau Detection

Identifying when your circuit suffers from barren plateaus.

```python
import numpy as np

class BarrenPlateauDetector:
    """
    Detect barren plateau behavior in variational circuits.
    """

    def __init__(self, cost_fn, n_params):
        self.cost_fn = cost_fn
        self.n_params = n_params

    def estimate_gradient_variance(self, n_samples=100, epsilon=0.01):
        """Estimate gradient variance via finite differences."""
        gradients = []

        for _ in range(n_samples):
            params = np.random.uniform(0, 2*np.pi, self.n_params)
            grad = self._numerical_gradient(params, epsilon)
            gradients.append(grad)

        gradients = np.array(gradients)
        return np.var(gradients, axis=0), np.mean(gradients, axis=0)

    def _numerical_gradient(self, params, epsilon):
        """Compute gradient via central differences."""
        grad = np.zeros(self.n_params)
        for i in range(self.n_params):
            params_plus = params.copy()
            params_plus[i] += epsilon

            params_minus = params.copy()
            params_minus[i] -= epsilon

            grad[i] = (self.cost_fn(params_plus) - self.cost_fn(params_minus)) / (2 * epsilon)

        return grad

    def diagnose(self, n_samples=100):
        """Run barren plateau diagnosis."""
        variance, mean = self.estimate_gradient_variance(n_samples)

        mean_var = np.mean(variance)
        mean_abs_grad = np.mean(np.abs(mean))

        diagnosis = {
            'gradient_variance': mean_var,
            'mean_gradient_magnitude': mean_abs_grad,
            'status': 'healthy' if mean_var > 1e-4 else 'warning' if mean_var > 1e-6 else 'barren'
        }

        return diagnosis

# Demonstration
print("Barren Plateau Detection:")
print("=" * 50)

# Trainable cost function
def trainable_cost(params):
    return np.sum(np.sin(params)**2)

# Barren cost function (simulated)
def barren_cost(params):
    n = len(params)
    scale = 0.5 ** (n / 2)  # Exponential suppression
    return scale * np.sum(np.sin(params)**2)

np.random.seed(42)

for name, cost_fn, n_params in [("Trainable (8 params)", trainable_cost, 8),
                                  ("Trainable (16 params)", trainable_cost, 16),
                                  ("Barren (8 params)", barren_cost, 8),
                                  ("Barren (16 params)", barren_cost, 16)]:
    detector = BarrenPlateauDetector(cost_fn, n_params)
    result = detector.diagnose(50)

    print(f"\n{name}:")
    print(f"  Gradient variance: {result['gradient_variance']:.2e}")
    print(f"  Status: {result['status']}")
```

<Tip>
Always check gradient variance before committing to a full optimization. A few hundred random samples can reveal trainability issues early.
</Tip>

## 4. Trainability-Preserving Strategies

Techniques to maintain trainability while achieving expressibility.

```python
import numpy as np

def trainability_strategies():
    """
    Strategies for maintaining trainability.
    """
    print("Trainability-Preserving Strategies:")
    print("=" * 50)

    strategies = [
        {
            'name': 'Shallow Circuits',
            'description': 'Limit circuit depth',
            'mechanism': 'Fewer layers = less gradient dilution',
            'tradeoff': 'May reduce expressibility'
        },
        {
            'name': 'Local Cost Functions',
            'description': 'Use cost functions acting on few qubits',
            'mechanism': 'Local observables have polynomial gradient variance',
            'tradeoff': 'May not match global problem structure'
        },
        {
            'name': 'Layerwise Training',
            'description': 'Train one layer at a time',
            'mechanism': 'Each layer sees non-vanishing gradients',
            'tradeoff': 'May find suboptimal solutions'
        },
        {
            'name': 'Initialization Strategies',
            'description': 'Start near identity or known good solution',
            'mechanism': 'Avoid random initialization pitfalls',
            'tradeoff': 'Requires problem-specific knowledge'
        },
        {
            'name': 'Parameter Correlation',
            'description': 'Reduce effective parameter count',
            'mechanism': 'Same parameter used in multiple places',
            'tradeoff': 'Reduces flexibility'
        }
    ]

    for s in strategies:
        print(f"\n{s['name']}:")
        print(f"  Description: {s['description']}")
        print(f"  Mechanism: {s['mechanism']}")
        print(f"  Tradeoff: {s['tradeoff']}")

    # Demonstrate initialization
    print("\n\nInitialization Strategy Example:")
    print("-" * 40)

    def identity_init(n_params):
        """Initialize near identity."""
        return np.random.normal(0, 0.1, n_params)

    def random_init(n_params):
        """Standard random initialization."""
        return np.random.uniform(0, 2*np.pi, n_params)

    n_params = 10
    id_params = identity_init(n_params)
    rand_params = random_init(n_params)

    print(f"  Identity init: |θ| = {np.linalg.norm(id_params):.4f}")
    print(f"  Random init: |θ| = {np.linalg.norm(rand_params):.4f}")

trainability_strategies()
```

## 5. Balancing Expressibility and Trainability

Finding the sweet spot between being able to reach solutions and being able to find them.


## Summary

Trainability is essential for successful variational algorithms:

1. **Definition**: Ability to optimize using gradients
2. **Key Metric**: Gradient variance over random initializations
3. **Barren Plateaus**: Exponentially vanishing gradients
4. **Strategies**: Shallow circuits, local costs, smart initialization
5. **Balance**: Trade off with expressibility based on problem

Always verify trainability before committing to expensive optimization runs.
