# Problem-Inspired Ansätze

Problem-inspired ansätze leverage domain knowledge to create circuits that naturally respect problem structure, often leading to better trainability and solution quality.

## Learning Objectives

After completing this lesson, you will be able to:

- Understand problem-inspired ansatz design
- Implement chemistry-inspired circuits
- Build optimization-focused ansätze
- Leverage symmetries in circuit design
- Compare problem-specific vs generic approaches

## 1. Why Problem-Inspired Design?

Generic ansätze explore the full Hilbert space, but problem solutions often lie in structured subspaces.

## 2. Chemistry-Inspired Ansätze (UCCSD)

The Unitary Coupled Cluster ansatz is designed for molecular electronic structure.

```python
import numpy as np

def uccsd_overview():
    """
    Overview of UCCSD ansatz for quantum chemistry.
    """
    print("UCCSD Ansatz for Quantum Chemistry:")
    print("=" * 50)

    print("\n1. Coupled Cluster Theory:")
    print("-" * 40)
    print("  Classical CC: |ψ⟩ = e^T |HF⟩")
    print("  Quantum UCC: |ψ⟩ = e^(T-T†) |HF⟩")
    print("\n  T = T₁ + T₂ + ...")
    print("  T₁ = Σᵢₐ tᵢₐ a†ₐaᵢ  (singles)")
    print("  T₂ = Σᵢⱼₐᵦ tᵢⱼₐᵦ a†ₐa†ᵦaᵢaⱼ  (doubles)")

    print("\n\n2. Key Properties:")
    print("-" * 40)

    properties = [
        "Preserves particle number (N electrons)",
        "Preserves spin symmetry (S², Sz)",
        "Starts from Hartree-Fock reference",
        "Systematically improvable (add triples, etc.)",
        "Exponentially many terms in general"
    ]

    for p in properties:
        print(f"  • {p}")

    print("\n\n3. Parameter Scaling:")
    print("-" * 40)

    def uccsd_params(n_electrons, n_orbitals):
        n_occ = n_electrons // 2
        n_virt = n_orbitals - n_occ
        singles = n_occ * n_virt
        doubles = singles * (singles + 1) // 2
        return singles, doubles

    print(f"  {'Orbitals':<12} {'Electrons':<12} {'Singles':<10} {'Doubles':<10}")
    print("  " + "-" * 44)

    for n_orb, n_elec in [(4, 2), (6, 4), (8, 4), (10, 6), (12, 8)]:
        s, d = uccsd_params(n_elec, n_orb)
        print(f"  {n_orb:<12} {n_elec:<12} {s:<10} {d:<10}")

uccsd_overview()
```

## 3. Optimization-Inspired Ansätze (QAOA)

QAOA naturally encodes combinatorial optimization problems.

```python
import numpy as np

class QAOAAnsatz:
    """
    QAOA ansatz for optimization problems.
    """

    def __init__(self, n_qubits, cost_terms):
        """
        Args:
            n_qubits: Number of qubits
            cost_terms: List of (qubits, coefficient) for ZZ terms
        """
        self.n_qubits = n_qubits
        self.dim = 2**n_qubits
        self.cost_terms = cost_terms

    def initial_state(self):
        """Prepare uniform superposition."""
        state = np.ones(self.dim, dtype=complex) / np.sqrt(self.dim)
        return state

    def cost_layer(self, state, gamma):
        """Apply cost Hamiltonian evolution e^{-iγH_C}."""
        for qubits, coeff in self.cost_terms:
            state = self._apply_zz(state, qubits[0], qubits[1], gamma * coeff)
        return state

    def mixer_layer(self, state, beta):
        """Apply mixer Hamiltonian evolution e^{-iβH_M}."""
        for q in range(self.n_qubits):
            state = self._rx(state, q, 2 * beta)
        return state

    def forward(self, gammas, betas):
        """Execute QAOA circuit."""
        state = self.initial_state()
        for gamma, beta in zip(gammas, betas):
            state = self.cost_layer(state, gamma)
            state = self.mixer_layer(state, beta)
        return state

    def _apply_zz(self, state, q1, q2, angle):
        new_state = state.copy()
        for i in range(self.dim):
            b1 = (i >> (self.n_qubits - 1 - q1)) & 1
            b2 = (i >> (self.n_qubits - 1 - q2)) & 1
            parity = 1 if b1 == b2 else -1
            new_state[i] *= np.exp(-1j * angle * parity)
        return new_state

    def _rx(self, state, qubit, theta):
        c, s = np.cos(theta/2), np.sin(theta/2)
        new_state = np.zeros_like(state)
        for i in range(self.dim):
            bit = (i >> (self.n_qubits - 1 - qubit)) & 1
            partner = i ^ (1 << (self.n_qubits - 1 - qubit))
            if bit == 0:
                new_state[i] += c * state[i] - 1j * s * state[partner]
                new_state[partner] += -1j * s * state[i] + c * state[partner]
        return new_state

# Demonstration: MaxCut on a triangle
print("QAOA for MaxCut:")
print("=" * 50)

n_qubits = 3
# Triangle graph: edges (0,1), (1,2), (0,2)
cost_terms = [([0, 1], 0.5), ([1, 2], 0.5), ([0, 2], 0.5)]

qaoa = QAOAAnsatz(n_qubits, cost_terms)

print("\nGraph: Triangle (3 vertices, 3 edges)")
print("Optimal cut: 2 edges (any 2-1 partition)")

# Try different parameter values
print("\nQAOA optimization (p=1):")

best_cost = float('inf')
best_params = None

for gamma in np.linspace(0, np.pi, 10):
    for beta in np.linspace(0, np.pi/2, 10):
        state = qaoa.forward([gamma], [beta])
        probs = np.abs(state)**2

        # Expected cost (edges cut)
        exp_cost = 0
        for i in range(2**n_qubits):
            bits = [(i >> (n_qubits - 1 - q)) & 1 for q in range(n_qubits)]
            cut_value = sum(1 for (q1, q2), _ in cost_terms if bits[q1] != bits[q2])
            exp_cost += probs[i] * cut_value

        if exp_cost > -best_cost:  # Maximizing
            best_cost = -exp_cost
            best_params = (gamma, beta)

print(f"  Best γ: {best_params[0]:.4f}")
print(f"  Best β: {best_params[1]:.4f}")
print(f"  Expected cut: {-best_cost:.4f}")
```

<Definition term="QAOA">
Quantum Approximate Optimization Algorithm: A variational algorithm that alternates between cost Hamiltonian evolution (encoding the optimization objective) and mixer Hamiltonian evolution (enabling exploration).
</Definition>

## 4. Hamiltonian Variational Ansatz (HVA)

HVA uses problem Hamiltonian terms as building blocks.

```python
import numpy as np

def hva_overview():
    """
    Overview of Hamiltonian Variational Ansatz.
    """
    print("Hamiltonian Variational Ansatz (HVA):")
    print("=" * 50)

    print("\n1. Core Idea:")
    print("-" * 40)
    print("  Use terms of the problem Hamiltonian as ansatz generators")
    print("  |ψ(θ)⟩ = ∏ₖ e^{-iθₖHₖ} |ψ₀⟩")
    print("  where H = Σₖ Hₖ")

    print("\n\n2. Example: Heisenberg Model")
    print("-" * 40)
    print("  H = J Σᵢ (XᵢXᵢ₊₁ + YᵢYᵢ₊₁ + ZᵢZᵢ₊₁)")
    print("\n  HVA ansatz:")
    print("    Layer 1: e^{-iθ₁ΣXX} e^{-iθ₂ΣYY} e^{-iθ₃ΣZZ}")
    print("    Layer 2: e^{-iθ₄ΣXX} e^{-iθ₅ΣYY} e^{-iθ₆ΣZZ}")
    print("    ...")

    print("\n\n3. Advantages:")
    print("-" * 40)

    advantages = [
        "Naturally respects problem symmetries",
        "Adiabatic-inspired: connects to known limits",
        "Often more trainable than generic ansätze",
        "Physical interpretation of parameters"
    ]

    for a in advantages:
        print(f"  • {a}")

    print("\n\n4. Parameter Comparison:")
    print("-" * 40)

    # Compare parameter counts
    n_qubits = 6
    n_layers = 3

    hw_params = 2 * n_qubits * n_layers  # Hardware-efficient
    hva_params = 3 * n_layers  # HVA for Heisenberg (3 terms per layer)

    print(f"  {n_qubits} qubits, {n_layers} layers:")
    print(f"    Hardware-efficient: {hw_params} parameters")
    print(f"    HVA (Heisenberg): {hva_params} parameters")

hva_overview()
```

## 5. Symmetry-Preserving Design

Incorporating symmetries constrains the search space and improves results.

```python
import numpy as np

def symmetry_preserving():
    """
    Demonstrate symmetry-preserving ansatz design.
    """
    print("Symmetry-Preserving Ansatz Design:")
    print("=" * 50)

    print("\n1. Common Symmetries in Physics:")
    print("-" * 40)

    symmetries = [
        {
            'symmetry': 'Particle Number',
            'system': 'Electronic structure',
            'generator': 'N = Σᵢ nᵢ',
            'implementation': 'Number-preserving gates'
        },
        {
            'symmetry': 'Total Spin',
            'system': 'Magnetic systems',
            'generator': 'S² and Sᵤ',
            'implementation': 'Spin-adapted operators'
        },
        {
            'symmetry': 'Parity',
            'system': 'Many quantum systems',
            'generator': 'P = ∏ᵢ Zᵢ',
            'implementation': 'Even parity gates'
        },
        {
            'symmetry': 'Time Reversal',
            'system': 'Closed systems',
            'generator': 'Complex conjugation',
            'implementation': 'Real amplitudes'
        }
    ]

    for s in symmetries:
        print(f"\n  {s['symmetry']}:")
        print(f"    System: {s['system']}")
        print(f"    Implementation: {s['implementation']}")

    print("\n\n2. Particle-Number Preserving Gates:")
    print("-" * 40)

    print("  Standard gates that preserve particle number:")
    print("    • RZ: Phase rotation (preserves occupation)")
    print("    • Givens rotation: Mixes |01⟩ ↔ |10⟩")
    print("    • fSWAP: Fermionic swap gate")

    # Demonstrate Givens rotation
    print("\n  Givens rotation G(θ) on |01⟩, |10⟩ subspace:")
    print("    G(θ)|00⟩ = |00⟩")
    print("    G(θ)|01⟩ = cos(θ)|01⟩ + sin(θ)|10⟩")
    print("    G(θ)|10⟩ = -sin(θ)|01⟩ + cos(θ)|10⟩")
    print("    G(θ)|11⟩ = |11⟩")

    print("\n\n3. Search Space Reduction:")
    print("-" * 40)

    def count_n_particle_states(n_qubits, n_particles):
        from math import comb
        return comb(n_qubits, n_particles)

    n_qubits = 10
    print(f"  {n_qubits} qubits:")
    print(f"    Full Hilbert space: {2**n_qubits}")
    for n_particles in [2, 4, 5]:
        n_states = count_n_particle_states(n_qubits, n_particles)
        reduction = 2**n_qubits / n_states
        print(f"    {n_particles}-particle sector: {n_states} ({reduction:.1f}x reduction)")

symmetry_preserving()
```

<Warning>
Breaking symmetries during optimization can lead to unphysical solutions. Always verify that your ansatz preserves required symmetries.
</Warning>

## Summary

Problem-inspired ansätze offer significant advantages:

1. **Reduced Search Space**: Focus on physically relevant states
2. **UCCSD**: Chemically motivated for molecular problems
3. **QAOA**: Natural encoding for optimization
4. **HVA**: Uses Hamiltonian structure directly
5. **Symmetries**: Constrain to valid solution subspaces

Choose ansatz structure based on your problem's domain knowledge.
