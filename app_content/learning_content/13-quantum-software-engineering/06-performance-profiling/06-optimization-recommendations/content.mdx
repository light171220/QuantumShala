## 2. Recommendation Engine Framework

```python
from dataclasses import dataclass, field
from typing import Dict, List, Optional, Callable
from enum import Enum
import numpy as np

class Priority(Enum):
    CRITICAL = 4
    HIGH = 3
    MEDIUM = 2
    LOW = 1

class Effort(Enum):
    MINIMAL = 1  # Minutes to implement
    LOW = 2      # Hours to implement
    MEDIUM = 3   # Days to implement
    HIGH = 4     # Weeks to implement

@dataclass
class Recommendation:
    """A single optimization recommendation."""
    id: str
    title: str
    description: str
    category: str
    priority: Priority
    effort: Effort
    estimated_improvement: float  # Percentage
    implementation_steps: List[str]
    code_example: Optional[str] = None
    prerequisites: List[str] = field(default_factory=list)

    @property
    def roi_score(self) -> float:
        """Return on investment score."""
        return (self.estimated_improvement * self.priority.value) / self.effort.value


class RecommendationEngine:
    """Generate optimization recommendations from profiling data."""

    def __init__(self):
        self.rules: List[Callable] = []
        self.recommendations: List[Recommendation] = []

    def add_rule(self, rule: Callable):
        """Add a recommendation rule."""
        self.rules.append(rule)

    def analyze(self, profile_data: Dict) -> List[Recommendation]:
        """Analyze profiling data and generate recommendations."""
        self.recommendations = []

        for rule in self.rules:
            recs = rule(profile_data)
            if recs:
                self.recommendations.extend(recs)

        # Sort by ROI score
        self.recommendations.sort(key=lambda x: x.roi_score, reverse=True)
        return self.recommendations

    def get_quick_wins(self) -> List[Recommendation]:
        """Get high-impact, low-effort recommendations."""
        return [r for r in self.recommendations
                if r.effort in [Effort.MINIMAL, Effort.LOW]
                and r.priority in [Priority.HIGH, Priority.CRITICAL]]

    def generate_report(self) -> str:
        """Generate formatted recommendation report."""
        report = []
        report.append("=" * 70)
        report.append("OPTIMIZATION RECOMMENDATIONS")
        report.append("=" * 70)

        for i, rec in enumerate(self.recommendations, 1):
            report.append(f"\n{i}. [{rec.priority.name}] {rec.title}")
            report.append(f"   Category: {rec.category}")
            report.append(f"   Effort: {rec.effort.name}")
            report.append(f"   Expected Improvement: {rec.estimated_improvement}%")
            report.append(f"   ROI Score: {rec.roi_score:.2f}")
            report.append(f"   Description: {rec.description}")
            report.append("   Steps:")
            for step in rec.implementation_steps:
                report.append(f"   • {step}")

        return "\n".join(report)


# Define recommendation rules
def transpilation_rule(profile: Dict) -> List[Recommendation]:
    """Check for transpilation optimization opportunities."""
    recommendations = []

    transpile_pct = profile.get('transpilation_percentage', 0)
    opt_level = profile.get('optimization_level', 2)

    if transpile_pct > 50 and opt_level >= 2:
        recommendations.append(Recommendation(
            id="trans-001",
            title="Reduce Transpilation Overhead",
            description="Transpilation is consuming over 50% of execution time.",
            category="transpilation",
            priority=Priority.HIGH,
            effort=Effort.LOW,
            estimated_improvement=30,
            implementation_steps=[
                "Try optimization_level=1 for development",
                "Cache transpiled circuits for repeated executions",
                "Pre-transpile circuits in batch"
            ],
            code_example="""
# Cache transpiled circuits
from functools import lru_cache

@lru_cache(maxsize=100)
def get_transpiled(circuit_key, backend_name):
    return transpile(circuits[circuit_key], backend)
"""
        ))

    return recommendations


def memory_rule(profile: Dict) -> List[Recommendation]:
    """Check for memory optimization opportunities."""
    recommendations = []

    peak_memory = profile.get('peak_memory_mb', 0)
    num_qubits = profile.get('num_qubits', 0)

    # Check if memory usage is higher than expected
    expected_memory = (2 ** num_qubits * 16) / (1024 * 1024) if num_qubits else 0

    if peak_memory > expected_memory * 2:
        recommendations.append(Recommendation(
            id="mem-001",
            title="Reduce Memory Overhead",
            description=f"Memory usage ({peak_memory:.0f}MB) is 2x higher than expected.",
            category="memory",
            priority=Priority.MEDIUM,
            effort=Effort.MEDIUM,
            estimated_improvement=40,
            implementation_steps=[
                "Clean up intermediate results",
                "Use gc.collect() between circuit executions",
                "Consider streaming processing for large batches"
            ]
        ))

    return recommendations


# Create engine with rules
```

## 3. Pattern-Based Recommendations

```python
class PatternMatcher:
    """Match profiling patterns to recommendations."""

    def __init__(self):
        self.patterns = {}

    def register_pattern(self, name: str, condition: Callable,
                         recommendation: Recommendation):
        """Register a pattern and its recommendation."""
        self.patterns[name] = {
            'condition': condition,
            'recommendation': recommendation
        }

    def match(self, profile: Dict) -> List[Recommendation]:
        """Match patterns and return applicable recommendations."""
        matches = []
        for name, pattern in self.patterns.items():
            if pattern['condition'](profile):
                matches.append(pattern['recommendation'])
        return matches


# Define common patterns
def create_pattern_matcher() -> PatternMatcher:
    """Create matcher with common patterns."""
    matcher = PatternMatcher()

    # Pattern: High queue wait time
```

## 4. Impact Estimation

```python
class ImpactEstimator:
    """Estimate impact of optimizations."""

    def __init__(self):
        self.historical_data: List[Dict] = []

    def record_optimization(self, before: Dict, after: Dict,
                           optimization_type: str):
        """Record before/after metrics for an optimization."""
        self.historical_data.append({
            'type': optimization_type,
            'before': before,
            'after': after,
            'improvement': self._calculate_improvement(before, after)
        })

    def _calculate_improvement(self, before: Dict, after: Dict) -> Dict:
        """Calculate improvement metrics."""
        improvements = {}

        for key in before:
            if key in after and isinstance(before[key], (int, float)):
                before_val = before[key]
                after_val = after[key]
                if before_val != 0:
                    pct_change = ((before_val - after_val) / before_val) * 100
                    improvements[key] = pct_change

        return improvements

    def estimate_impact(self, optimization_type: str,
                        current_profile: Dict) -> Dict:
        """Estimate impact based on historical data."""
        relevant = [d for d in self.historical_data
                   if d['type'] == optimization_type]

        if not relevant:
            return {'estimated': False, 'reason': 'No historical data'}

        # Average improvements across historical data
        improvements = {}
        for key in relevant[0]['improvement']:
            values = [d['improvement'].get(key, 0) for d in relevant]
            improvements[key] = np.mean(values)

        # Apply to current profile
        projected = {}
        for key, pct in improvements.items():
            if key in current_profile:
                projected[key] = current_profile[key] * (1 - pct / 100)

        return {
            'estimated': True,
            'average_improvements': improvements,
            'projected_metrics': projected
        }


class ROICalculator:
    """Calculate return on investment for optimizations."""

    def __init__(self, hourly_cost: float = 50.0):
        self.hourly_cost = hourly_cost

    def calculate_roi(self, recommendation: Recommendation,
                      current_runtime_s: float,
                      executions_per_day: int) -> Dict:
        """Calculate ROI for a recommendation."""

        # Estimate implementation effort in hours
        effort_hours = {
            Effort.MINIMAL: 0.5,
            Effort.LOW: 4,
            Effort.MEDIUM: 20,
            Effort.HIGH: 80
        }

        impl_cost = effort_hours[recommendation.effort] * self.hourly_cost

        # Calculate runtime savings
        improvement = recommendation.estimated_improvement / 100
        time_saved_per_exec = current_runtime_s * improvement
        daily_savings_s = time_saved_per_exec * executions_per_day

        # Convert to monetary value (assuming runtime costs)
        daily_savings_cost = (daily_savings_s / 3600) * self.hourly_cost

        # Calculate payback period
        if daily_savings_cost > 0:
            payback_days = impl_cost / daily_savings_cost
        else:
            payback_days = float('inf')

        return {
            'implementation_cost': impl_cost,
            'daily_time_saved_s': daily_savings_s,
            'daily_cost_saved': daily_savings_cost,
            'payback_days': payback_days,
            'annual_savings': daily_savings_cost * 365,
            'roi_percentage': ((daily_savings_cost * 365 - impl_cost) / impl_cost) * 100
        }
```

## 5. Automated Optimization Pipeline

```python
class OptimizationPipeline:
    """Automated pipeline for generating and applying optimizations."""

    def __init__(self):
        self.engine = RecommendationEngine()
        self.pattern_matcher = create_pattern_matcher()
        self.impact_estimator = ImpactEstimator()
        self.applied_optimizations: List[Dict] = []

    def run_analysis(self, profile: Dict) -> Dict:
        """Run complete optimization analysis."""

        # Collect recommendations from all sources
        all_recommendations = []

        # Rule-based recommendations
        all_recommendations.extend(self.engine.analyze(profile))

        # Pattern-based recommendations
        all_recommendations.extend(self.pattern_matcher.match(profile))

        # Deduplicate by ID
        seen_ids = set()
        unique_recommendations = []
        for rec in all_recommendations:
            if rec.id not in seen_ids:
                seen_ids.add(rec.id)
                unique_recommendations.append(rec)

        # Sort by ROI
        unique_recommendations.sort(key=lambda x: x.roi_score, reverse=True)

        # Generate action plan
        action_plan = self._create_action_plan(unique_recommendations)

        return {
            'recommendations': unique_recommendations,
            'action_plan': action_plan,
            'quick_wins': [r for r in unique_recommendations
                          if r.effort in [Effort.MINIMAL, Effort.LOW]
                          and r.priority in [Priority.HIGH, Priority.CRITICAL]],
            'summary': self._create_summary(unique_recommendations)
        }

    def _create_action_plan(self, recommendations: List[Recommendation]) -> List[Dict]:
        """Create prioritized action plan."""
        plan = []

        for i, rec in enumerate(recommendations[:5], 1):  # Top 5
            plan.append({
                'order': i,
                'recommendation': rec.title,
                'priority': rec.priority.name,
                'effort': rec.effort.name,
                'expected_improvement': f"{rec.estimated_improvement}%",
                'steps': rec.implementation_steps
            })

        return plan

    def _create_summary(self, recommendations: List[Recommendation]) -> Dict:
        """Create analysis summary."""
        if not recommendations:
            return {'status': 'optimal', 'message': 'No optimizations needed'}

        total_potential = sum(r.estimated_improvement for r in recommendations)
        critical_count = sum(1 for r in recommendations
                            if r.priority == Priority.CRITICAL)

        return {
            'total_recommendations': len(recommendations),
            'critical_issues': critical_count,
            'potential_improvement': f"{min(total_potential, 80)}%",
            'top_priority': recommendations[0].title if recommendations else None
        }

    def apply_and_verify(self, recommendation: Recommendation,
                         apply_fn: Callable,
                         profile_fn: Callable) -> Dict:
        """Apply an optimization and verify results."""

        # Get before profile
        before = profile_fn()

        # Apply optimization
        apply_fn(recommendation)

        # Get after profile
        after = profile_fn()

        # Calculate actual improvement
        improvement = {}
        for key in before:
            if key in after and isinstance(before[key], (int, float)):
                if before[key] != 0:
                    improvement[key] = ((before[key] - after[key]) / before[key]) * 100

        # Record for future estimates
        self.impact_estimator.record_optimization(before, after, recommendation.category)

        result = {
            'recommendation': recommendation.id,
            'expected_improvement': recommendation.estimated_improvement,
            'actual_improvement': improvement,
            'success': any(v > 0 for v in improvement.values()),
            'before': before,
            'after': after
        }

        self.applied_optimizations.append(result)
        return result


def generate_optimization_report(profile: Dict) -> str:
    """Generate complete optimization report."""
    pipeline = OptimizationPipeline()

    # Add custom rules
    pipeline.engine.add_rule(transpilation_rule)
    pipeline.engine.add_rule(memory_rule)

    # Run analysis
    results = pipeline.run_analysis(profile)

    # Format report
    report = []
    report.append("=" * 70)
    report.append("QUANTUM WORKFLOW OPTIMIZATION REPORT")
    report.append("=" * 70)

    # Summary
    summary = results['summary']
    report.append(f"\nSummary:")
    report.append(f"  Total Recommendations: {summary['total_recommendations']}")
    report.append(f"  Critical Issues: {summary['critical_issues']}")
    report.append(f"  Potential Improvement: {summary['potential_improvement']}")

    # Quick wins
    report.append("\nQuick Wins (High Impact, Low Effort):")
    for qw in results['quick_wins']:
        report.append(f"  • {qw.title} ({qw.estimated_improvement}% improvement)")

    # Action plan
    report.append("\nAction Plan:")
    for item in results['action_plan']:
        report.append(f"\n  {item['order']}. {item['recommendation']}")
        report.append(f"     Priority: {item['priority']}, Effort: {item['effort']}")
        report.append(f"     Expected: {item['expected_improvement']}")

    return "\n".join(report)
```

## 6. Continuous Optimization

```python
class ContinuousOptimizer:
    """Continuously monitor and suggest optimizations."""

    def __init__(self, check_interval_minutes: int = 60):
        self.check_interval = check_interval_minutes
        self.baseline: Optional[Dict] = None
        self.history: List[Dict] = []
        self.pipeline = OptimizationPipeline()

    def set_baseline(self, profile: Dict):
        """Set baseline performance metrics."""
        self.baseline = profile
        self.history.append({
            'timestamp': time.time(),
            'profile': profile,
            'type': 'baseline'
        })

    def check_regression(self, current: Dict) -> List[str]:
        """Check for performance regressions."""
        if not self.baseline:
            return []

        regressions = []
        threshold = 1.2  # 20% slower is a regression

        for key in self.baseline:
            if key in current and isinstance(self.baseline[key], (int, float)):
                if self.baseline[key] > 0:
                    ratio = current[key] / self.baseline[key]
                    if ratio > threshold:
                        regressions.append(
                            f"{key}: {ratio:.1f}x slower than baseline"
                        )

        return regressions

    def analyze_trend(self, metric: str) -> Dict:
        """Analyze trend for a specific metric."""
        values = [h['profile'].get(metric, 0) for h in self.history
                 if metric in h.get('profile', {})]

        if len(values) < 3:
            return {'trend': 'insufficient_data'}

        # Simple linear regression
        x = np.arange(len(values))
        coeffs = np.polyfit(x, values, 1)
        slope = coeffs[0]

        if slope > 0.05 * np.mean(values):
            trend = 'increasing'
        elif slope < -0.05 * np.mean(values):
            trend = 'decreasing'
        else:
            trend = 'stable'

        return {
            'trend': trend,
            'slope': slope,
            'current': values[-1],
            'baseline': values[0],
            'change_pct': ((values[-1] - values[0]) / values[0] * 100)
            if values[0] != 0 else 0
        }

    def get_proactive_recommendations(self, current: Dict) -> List[Recommendation]:
        """Get proactive recommendations based on trends."""
        recommendations = []

        # Check all tracked metrics for concerning trends
        for metric in current:
            trend = self.analyze_trend(metric)
            if trend.get('trend') == 'increasing' and trend.get('change_pct', 0) > 20:
                recommendations.append(Recommendation(
                    id=f"trend-{metric}",
                    title=f"Address {metric} Growth",
                    description=f"{metric} has increased {trend['change_pct']:.0f}% from baseline",
                    category="trend",
                    priority=Priority.MEDIUM,
                    effort=Effort.LOW,
                    estimated_improvement=20,
                    implementation_steps=[
                        f"Investigate recent changes affecting {metric}",
                        "Review commits since baseline was set",
                        "Profile specific workflows in detail"
                    ]
                ))

        return recommendations
```
