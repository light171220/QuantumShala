## 1. Understanding Bottlenecks in Quantum Workflows

Performance bottlenecks can occur at any stage of quantum computing workflows. Identifying these bottlenecks is crucial for optimization.

## 2. Bottleneck Detection Framework

```python
import time
from dataclasses import dataclass, field
from typing import Dict, List, Optional, Callable, Any
from enum import Enum
import numpy as np

class BottleneckSeverity(Enum):
    LOW = "low"
    MEDIUM = "medium"
    HIGH = "high"
    CRITICAL = "critical"

@dataclass
class BottleneckInfo:
    """Information about a detected bottleneck."""
    phase: str
    duration_ms: float
    percentage: float
    severity: BottleneckSeverity
    description: str
    recommendations: List[str] = field(default_factory=list)

@dataclass
class PipelineMetrics:
    """Metrics for each pipeline stage."""
    phase: str
    duration_ms: float
    memory_mb: float
    iterations: int = 1
    metadata: Dict = field(default_factory=dict)

class BottleneckDetector:
    """Detect performance bottlenecks in quantum workflows."""

    def __init__(self, threshold_percentile: float = 0.5):
        self.threshold_percentile = threshold_percentile
        self.metrics: List[PipelineMetrics] = []
        self.bottlenecks: List[BottleneckInfo] = []

    def record_metric(self, phase: str, duration_ms: float,
                      memory_mb: float = 0, **metadata):
        """Record a metric for a pipeline stage."""
        self.metrics.append(PipelineMetrics(
            phase=phase,
            duration_ms=duration_ms,
            memory_mb=memory_mb,
            metadata=metadata
        ))

    def analyze(self) -> List[BottleneckInfo]:
        """Analyze metrics to detect bottlenecks."""
        if not self.metrics:
            return []

        total_time = sum(m.duration_ms for m in self.metrics)
        self.bottlenecks = []

        for metric in self.metrics:
            percentage = (metric.duration_ms / total_time) * 100

            if percentage > self.threshold_percentile * 100:
                severity = self._determine_severity(percentage)
                recommendations = self._get_recommendations(metric.phase, percentage)

                self.bottlenecks.append(BottleneckInfo(
                    phase=metric.phase,
                    duration_ms=metric.duration_ms,
                    percentage=percentage,
                    severity=severity,
                    description=f"{metric.phase} takes {percentage:.1f}% of total time",
                    recommendations=recommendations
                ))

        return sorted(self.bottlenecks,
                     key=lambda x: x.duration_ms, reverse=True)

    def _determine_severity(self, percentage: float) -> BottleneckSeverity:
        """Determine severity based on percentage."""
        if percentage > 80:
            return BottleneckSeverity.CRITICAL
        elif percentage > 60:
            return BottleneckSeverity.HIGH
        elif percentage > 40:
            return BottleneckSeverity.MEDIUM
        else:
            return BottleneckSeverity.LOW

    def _get_recommendations(self, phase: str,
                             percentage: float) -> List[str]:
        """Get recommendations based on bottleneck phase."""
        recommendations = {
            'transpilation': [
                "Try lower optimization levels",
                "Pre-transpile circuits when possible",
                "Use cached transpilation results"
            ],
            'execution': [
                "Consider using simulators for development",
                "Batch circuits to reduce overhead",
                "Choose backend with shorter queue"
            ],
            'queue_wait': [
                "Use IBM Quantum fair-share scheduling",
                "Consider dedicated access",
                "Schedule jobs during off-peak hours"
            ],
            'result_processing': [
                "Process results incrementally",
                "Use efficient data structures",
                "Parallelize post-processing"
            ],
            'circuit_creation': [
                "Use circuit templates",
                "Pre-build parameterized circuits",
                "Optimize circuit construction"
            ]
        }

        return recommendations.get(phase, ["Profile this phase in detail"])

    def print_report(self):
        """Print bottleneck analysis report."""
        bottlenecks = self.analyze()

        print("\n" + "=" * 70)
        print("BOTTLENECK DETECTION REPORT")
        print("=" * 70)

        total_time = sum(m.duration_ms for m in self.metrics)
        print(f"\nTotal Pipeline Time: {total_time:.2f} ms")
        print(f"Threshold: {self.threshold_percentile * 100:.0f}%")

        print("\nPhase Breakdown:")
        print("-" * 50)
        for metric in sorted(self.metrics,
                            key=lambda x: x.duration_ms, reverse=True):
            pct = (metric.duration_ms / total_time) * 100
            bar = "█" * int(pct / 5) + "░" * (20 - int(pct / 5))
            flag = " ⚠️" if pct > self.threshold_percentile * 100 else ""
            print(f"  {metric.phase:20s} {metric.duration_ms:8.2f} ms "
                  f"{bar} {pct:5.1f}%{flag}")

        if bottlenecks:
            print("\n" + "=" * 70)
            print("DETECTED BOTTLENECKS")
            print("=" * 70)

            for bn in bottlenecks:
                severity_icon = {
                    BottleneckSeverity.CRITICAL: "🔴",
                    BottleneckSeverity.HIGH: "🟠",
                    BottleneckSeverity.MEDIUM: "🟡",
                    BottleneckSeverity.LOW: "🟢"
                }
                print(f"\n{severity_icon[bn.severity]} {bn.phase} "
                      f"({bn.severity.value.upper()})")
                print(f"   {bn.description}")
                print("   Recommendations:")
                for rec in bn.recommendations:
                    print(f"   • {rec}")


# Example usage
detector = BottleneckDetector(threshold_percentile=0.3)

# Simulate pipeline metrics
detector.record_metric("circuit_creation", 50.0)
detector.record_metric("transpilation", 500.0)
detector.record_metric("execution", 150.0)
detector.record_metric("result_processing", 30.0)

detector.print_report()
```

## 3. Transpilation Bottleneck Analysis

```python
class TranspilationAnalyzer:
    """Analyze transpilation bottlenecks in detail."""

    def __init__(self):
        self.pass_timings: Dict[str, float] = {}
        self.circuit_stats: Dict[str, Any] = {}

    def profile_transpilation(self, circuit, backend,
                               optimization_level: int = 2) -> Dict:
        """Profile transpilation with pass-level detail."""
        from qiskit import transpile
        from qiskit.transpiler import PassManager
        from qiskit.transpiler.preset_passmanagers import generate_preset_pass_manager

        # Get the preset pass manager
        pm = generate_preset_pass_manager(
            optimization_level=optimization_level,
            backend=backend
        )

        results = {
            'original_depth': circuit.depth(),
            'original_gates': sum(circuit.count_ops().values()),
            'optimization_level': optimization_level,
            'passes': []
        }

        # Time full transpilation
        start = time.perf_counter()
        transpiled = transpile(circuit, backend=backend,
                               optimization_level=optimization_level)
        total_time = (time.perf_counter() - start) * 1000

        results['total_time_ms'] = total_time
        results['transpiled_depth'] = transpiled.depth()
        results['transpiled_gates'] = sum(transpiled.count_ops().values())
        results['depth_reduction'] = results['original_depth'] - results['transpiled_depth']
        results['gate_reduction'] = results['original_gates'] - results['transpiled_gates']

        return results

    def identify_slow_passes(self, results: Dict,
                              threshold_pct: float = 20) -> List[str]:
        """Identify transpiler passes taking excessive time."""
        slow_passes = []
        total = results.get('total_time_ms', 0)

        for pass_info in results.get('passes', []):
            pct = (pass_info['time_ms'] / total) * 100 if total > 0 else 0
            if pct > threshold_pct:
                slow_passes.append(pass_info['name'])

        return slow_passes

    def analyze_scaling(self, circuit_sizes: List[int],
                        backend) -> Dict:
        """Analyze how transpilation scales with circuit size."""
        from qiskit.circuit.random import random_circuit

        scaling_data = []

        for size in circuit_sizes:
            circuit = random_circuit(num_qubits=5, depth=size)

            times = []
            for _ in range(3):  # Average over 3 runs
                start = time.perf_counter()
                transpiled = transpile(circuit, backend=backend,
                                       optimization_level=2)
                times.append((time.perf_counter() - start) * 1000)

            scaling_data.append({
                'circuit_depth': size,
                'mean_time_ms': np.mean(times),
                'std_time_ms': np.std(times),
                'transpiled_depth': transpiled.depth()
            })

        # Fit scaling model
        depths = [d['circuit_depth'] for d in scaling_data]
        times = [d['mean_time_ms'] for d in scaling_data]

        # Linear fit
        coeffs = np.polyfit(depths, times, 1)

        return {
            'data': scaling_data,
            'scaling_coefficient': coeffs[0],
            'base_overhead_ms': coeffs[1],
            'scaling_model': f"time_ms ≈ {coeffs[0]:.3f} × depth + {coeffs[1]:.1f}"
        }


class OptimizationLevelAnalyzer:
    """Analyze tradeoffs between optimization levels."""

    def compare_levels(self, circuit, backend,
                       levels: List[int] = [0, 1, 2, 3]) -> Dict:
        """Compare transpilation across optimization levels."""
        from qiskit import transpile

        results = {}

        for level in levels:
            times = []
            depths = []
            gates = []

            for _ in range(3):  # Average multiple runs
                start = time.perf_counter()
                transpiled = transpile(circuit, backend=backend,
                                       optimization_level=level)
                times.append((time.perf_counter() - start) * 1000)
                depths.append(transpiled.depth())
                gates.append(sum(transpiled.count_ops().values()))

            results[level] = {
                'mean_time_ms': np.mean(times),
                'std_time_ms': np.std(times),
                'mean_depth': np.mean(depths),
                'mean_gates': np.mean(gates)
            }

        return results

    def find_optimal_level(self, comparison: Dict,
                           time_weight: float = 0.3,
                           quality_weight: float = 0.7) -> int:
        """Find optimal optimization level balancing time and quality."""
        scores = {}

        # Normalize metrics
        times = [v['mean_time_ms'] for v in comparison.values()]
        depths = [v['mean_depth'] for v in comparison.values()]

        max_time = max(times)
        max_depth = max(depths)

        for level, metrics in comparison.items():
            time_score = 1 - (metrics['mean_time_ms'] / max_time)
            quality_score = 1 - (metrics['mean_depth'] / max_depth)

            scores[level] = (time_weight * time_score +
                           quality_weight * quality_score)

        return max(scores, key=scores.get)
```

## 4. Execution Bottleneck Analysis

```python
class ExecutionBottleneckAnalyzer:
    """Analyze quantum execution bottlenecks."""

    def __init__(self):
        self.execution_history: List[Dict] = []

    def profile_execution(self, circuit, backend,
                          shots: int = 1000) -> Dict:
        """Profile circuit execution in detail."""
        from qiskit import transpile

        # Pre-transpile
        transpiled = transpile(circuit, backend=backend)

        # Measure submission time
        submit_start = time.perf_counter()
        job = backend.run(transpiled, shots=shots)
        submit_time = (time.perf_counter() - submit_start) * 1000

        # Measure wait and execution
        wait_start = time.perf_counter()
        result = job.result()
        total_wait = (time.perf_counter() - wait_start) * 1000

        # Get execution metadata if available
        execution_info = {
            'submit_time_ms': submit_time,
            'total_wait_ms': total_wait,
            'shots': shots,
            'circuit_depth': transpiled.depth()
        }

        self.execution_history.append(execution_info)
        return execution_info

    def detect_queue_bottleneck(self, threshold_ms: float = 1000) -> Dict:
        """Detect if queue wait times are a bottleneck."""
        if not self.execution_history:
            return {'is_bottleneck': False, 'reason': 'No data'}

        wait_times = [e['total_wait_ms'] for e in self.execution_history]

        return {
            'is_bottleneck': np.mean(wait_times) > threshold_ms,
            'mean_wait_ms': np.mean(wait_times),
            'max_wait_ms': np.max(wait_times),
            'variability': np.std(wait_times) / np.mean(wait_times),
            'recommendation': self._queue_recommendation(np.mean(wait_times))
        }

    def _queue_recommendation(self, mean_wait: float) -> str:
        """Generate recommendation based on wait time."""
        if mean_wait > 60000:  # > 1 minute
            return "Consider dedicated access or schedule during off-peak"
        elif mean_wait > 10000:  # > 10 seconds
            return "Try alternative backends with shorter queues"
        elif mean_wait > 1000:  # > 1 second
            return "Queue times are moderate; batch jobs to amortize wait"
        else:
            return "Queue times are acceptable"


class ShotsAnalyzer:
    """Analyze shot count impact on performance."""

    def profile_shot_scaling(self, circuit, backend,
                             shot_counts: List[int]) -> Dict:
        """Profile how execution scales with shots."""
        from qiskit import transpile

        transpiled = transpile(circuit, backend=backend)
        scaling_data = []

        for shots in shot_counts:
            times = []
            for _ in range(3):
                start = time.perf_counter()
                job = backend.run(transpiled, shots=shots)
                result = job.result()
                times.append((time.perf_counter() - start) * 1000)

            scaling_data.append({
                'shots': shots,
                'mean_time_ms': np.mean(times),
                'std_time_ms': np.std(times),
                'time_per_shot_us': np.mean(times) * 1000 / shots
            })

        return {
            'data': scaling_data,
            'fixed_overhead_ms': self._estimate_overhead(scaling_data),
            'marginal_cost_per_shot_us': self._estimate_marginal_cost(scaling_data)
        }

    def _estimate_overhead(self, data: List[Dict]) -> float:
        """Estimate fixed overhead independent of shots."""
        if len(data) < 2:
            return 0

        shots = [d['shots'] for d in data]
        times = [d['mean_time_ms'] for d in data]

        # Linear fit: time = overhead + slope * shots
        coeffs = np.polyfit(shots, times, 1)
        return coeffs[1]

    def _estimate_marginal_cost(self, data: List[Dict]) -> float:
        """Estimate marginal cost per shot."""
        if len(data) < 2:
            return 0

        shots = [d['shots'] for d in data]
        times = [d['mean_time_ms'] for d in data]

        coeffs = np.polyfit(shots, times, 1)
        return coeffs[0] * 1000  # Convert to microseconds
```

## 5. Automated Bottleneck Detection

```python
class AutomatedBottleneckDetector:
    """Automatically detect and classify bottlenecks."""

    def __init__(self):
        self.detector = BottleneckDetector()
        self.transpilation_analyzer = TranspilationAnalyzer()
        self.execution_analyzer = ExecutionBottleneckAnalyzer()

    def full_pipeline_analysis(self, circuit, backend,
                                shots: int = 1000) -> Dict:
        """Run full pipeline analysis."""
        from qiskit import transpile

        results = {
            'phases': {},
            'bottlenecks': [],
            'recommendations': []
        }

        # Phase 1: Circuit Analysis
        start = time.perf_counter()
        circuit_info = {
            'depth': circuit.depth(),
            'width': circuit.num_qubits,
            'gates': sum(circuit.count_ops().values())
        }
        circuit_time = (time.perf_counter() - start) * 1000
        results['phases']['circuit_analysis'] = circuit_time

        # Phase 2: Transpilation
        start = time.perf_counter()
        transpiled = transpile(circuit, backend=backend, optimization_level=2)
        transpile_time = (time.perf_counter() - start) * 1000
        results['phases']['transpilation'] = transpile_time

        # Phase 3: Execution
        start = time.perf_counter()
        job = backend.run(transpiled, shots=shots)
        result = job.result()
        exec_time = (time.perf_counter() - start) * 1000
        results['phases']['execution'] = exec_time

        # Phase 4: Result Processing
        start = time.perf_counter()
        counts = result.get_counts()
        process_time = (time.perf_counter() - start) * 1000
        results['phases']['result_processing'] = process_time

        # Analyze bottlenecks
        total_time = sum(results['phases'].values())
        for phase, time_ms in results['phases'].items():
            pct = (time_ms / total_time) * 100
            if pct > 40:  # Significant bottleneck
                results['bottlenecks'].append({
                    'phase': phase,
                    'percentage': pct,
                    'time_ms': time_ms
                })

        # Generate recommendations
        results['recommendations'] = self._generate_recommendations(results)

        return results

    def _generate_recommendations(self, results: Dict) -> List[str]:
        """Generate actionable recommendations."""
        recommendations = []

        for bn in results['bottlenecks']:
            phase = bn['phase']

            if phase == 'transpilation':
                recommendations.append(
                    "Transpilation is the bottleneck. "
                    "Consider caching or using lower optimization levels."
                )
            elif phase == 'execution':
                recommendations.append(
                    "Execution is the bottleneck. "
                    "Check queue times or use batch execution."
                )
            elif phase == 'result_processing':
                recommendations.append(
                    "Result processing is the bottleneck. "
                    "Consider streaming or parallel processing."
                )

        return recommendations

    def compare_configurations(self, circuit, backend,
                                configs: List[Dict]) -> Dict:
        """Compare different configurations."""
        from qiskit import transpile

        config_results = []

        for config in configs:
            opt_level = config.get('optimization_level', 2)
            shots = config.get('shots', 1000)

            start = time.perf_counter()
            transpiled = transpile(circuit, backend=backend,
                                   optimization_level=opt_level)
            job = backend.run(transpiled, shots=shots)
            result = job.result()
            total_time = (time.perf_counter() - start) * 1000

            config_results.append({
                'config': config,
                'total_time_ms': total_time,
                'transpiled_depth': transpiled.depth()
            })

        best_config = min(config_results, key=lambda x: x['total_time_ms'])

        return {
            'results': config_results,
            'best_config': best_config['config'],
            'best_time_ms': best_config['total_time_ms']
        }
```

## 6. Real-Time Monitoring

```python
class RealTimeMonitor:
    """Real-time bottleneck monitoring."""

    def __init__(self, window_size: int = 10):
        self.window_size = window_size
        self.metrics_history: List[Dict] = []
        self.alerts: List[Dict] = []

    def record_execution(self, metrics: Dict):
        """Record execution metrics."""
        metrics['timestamp'] = time.time()
        self.metrics_history.append(metrics)

        # Keep only recent history
        if len(self.metrics_history) > self.window_size * 2:
            self.metrics_history = self.metrics_history[-self.window_size:]

        # Check for anomalies
        self._check_anomalies(metrics)

    def _check_anomalies(self, metrics: Dict):
        """Check for performance anomalies."""
        if len(self.metrics_history) < self.window_size:
            return

        recent = self.metrics_history[-self.window_size:]

        for key in ['transpile_ms', 'execute_ms', 'total_ms']:
            if key not in metrics:
                continue

            values = [m.get(key, 0) for m in recent[:-1]]
            mean = np.mean(values)
            std = np.std(values)

            if std > 0 and abs(metrics[key] - mean) > 2 * std:
                self.alerts.append({
                    'timestamp': time.time(),
                    'metric': key,
                    'value': metrics[key],
                    'expected': mean,
                    'deviation': (metrics[key] - mean) / std
                })

    def get_current_status(self) -> Dict:
        """Get current performance status."""
        if not self.metrics_history:
            return {'status': 'no_data'}

        recent = self.metrics_history[-self.window_size:]

        return {
            'status': 'ok' if not self.alerts else 'alert',
            'samples': len(recent),
            'avg_total_ms': np.mean([m.get('total_ms', 0) for m in recent]),
            'recent_alerts': len([a for a in self.alerts
                                  if time.time() - a['timestamp'] < 300])
        }
```

## Summary

| Bottleneck Type | Common Causes | Detection Method | Quick Fixes |
|-----------------|---------------|------------------|-------------|
| Transpilation | High opt level, large circuits | Time profiling | Lower opt level, caching |
| Queue Wait | Backend load, scheduling | Monitor queue times | Off-peak scheduling |
| Execution | Deep circuits, many shots | Execution profiling | Circuit optimization |
| Results | Large datasets, complex analysis | Memory/CPU profiling | Streaming processing |

## Best Practices

1. **Profile before optimizing** - Measure first, then optimize
2. **Check all phases** - Bottlenecks can shift after optimization
3. **Monitor continuously** - Performance can degrade over time
4. **Compare configurations** - Test different settings systematically
5. **Document findings** - Track what works for future reference
