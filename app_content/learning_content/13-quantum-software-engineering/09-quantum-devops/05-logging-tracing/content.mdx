## 1. The Observability Challenge in Quantum Systems

Quantum applications span multiple execution domains: classical preprocessing, quantum circuit execution (on simulators or hardware), and classical post-processing. Understanding system behavior requires tracing requests across all these domains.

<Definition term="Distributed Tracing">
A method of tracking requests as they flow through a distributed system, recording timing information and metadata at each step. For quantum applications, this includes tracking from parameter preparation through quantum execution to result analysis.
</Definition>

### The Quantum Execution Flow

```
Request → Classical Prep → Transpilation → Quantum Exec → Post-Process → Response
   │           │               │               │              │           │
   └───────────┴───────────────┴───────────────┴──────────────┴───────────┘
                              Trace Context
```

<Formula>
$$\text{Total Latency} = T_{prep} + T_{transpile} + T_{queue} + T_{exec} + T_{postprocess}$$
</Formula>

Understanding where time is spent requires tracing across all phases.

## 2. Structured Logging for Quantum Applications

Structured logging produces machine-readable log entries that can be queried and analyzed.

```python
# src/quantum_app/logging/structured_logger.py
"""Structured logging for quantum applications."""
import logging
import json
from datetime import datetime, timezone
from typing import Any, Dict, Optional
from dataclasses import dataclass, asdict
from contextvars import ContextVar
import traceback

# Context variables for request tracking
current_trace_id: ContextVar[Optional[str]] = ContextVar('trace_id', default=None)
current_span_id: ContextVar[Optional[str]] = ContextVar('span_id', default=None)
current_experiment_id: ContextVar[Optional[str]] = ContextVar('experiment_id', default=None)


@dataclass
class QuantumLogRecord:
    """Structured log record for quantum operations."""
    timestamp: str
    level: str
    message: str
    logger_name: str

    # Trace context
    trace_id: Optional[str] = None
    span_id: Optional[str] = None
    parent_span_id: Optional[str] = None

    # Quantum context
    experiment_id: Optional[str] = None
    circuit_name: Optional[str] = None
    backend: Optional[str] = None
    job_id: Optional[str] = None

    # Execution details
    num_qubits: Optional[int] = None
    circuit_depth: Optional[int] = None
    shots: Optional[int] = None

    # Error info
    error_type: Optional[str] = None
    error_message: Optional[str] = None
    stack_trace: Optional[str] = None

    # Custom fields
    extra: Optional[Dict[str, Any]] = None

    def to_json(self) -> str:
        """Serialize to JSON string."""
        data = {k: v for k, v in asdict(self).items() if v is not None}
        return json.dumps(data)


class StructuredFormatter(logging.Formatter):
    """Formats log records as structured JSON."""

    def format(self, record: logging.LogRecord) -> str:
        log_record = QuantumLogRecord(
            timestamp=datetime.now(timezone.utc).isoformat(),
            level=record.levelname,
            message=record.getMessage(),
            logger_name=record.name,
            trace_id=current_trace_id.get(),
            span_id=current_span_id.get(),
            experiment_id=current_experiment_id.get()
        )

        # Add extra fields from record
        if hasattr(record, 'quantum_context'):
            ctx = record.quantum_context
            log_record.circuit_name = ctx.get('circuit_name')
            log_record.backend = ctx.get('backend')
            log_record.job_id = ctx.get('job_id')
            log_record.num_qubits = ctx.get('num_qubits')
            log_record.circuit_depth = ctx.get('circuit_depth')
            log_record.shots = ctx.get('shots')

        # Add exception info
        if record.exc_info:
            log_record.error_type = record.exc_info[0].__name__
            log_record.error_message = str(record.exc_info[1])
            log_record.stack_trace = ''.join(traceback.format_exception(*record.exc_info))

        # Add any extra fields
        if hasattr(record, 'extra_fields'):
            log_record.extra = record.extra_fields

        return log_record.to_json()


class QuantumLogger:
    """Logger with quantum-specific context support."""

    def __init__(self, name: str):
        self.logger = logging.getLogger(name)
        self._context: Dict[str, Any] = {}

    def set_context(self, **kwargs):
        """Set persistent context for all subsequent logs."""
        self._context.update(kwargs)

    def clear_context(self):
        """Clear the logging context."""
        self._context.clear()

    def _log(self, level: int, message: str, **kwargs):
        """Internal logging method with context injection."""
        extra = {
            'quantum_context': {**self._context, **kwargs.get('quantum_context', {})},
            'extra_fields': kwargs.get('extra', {})
        }
        self.logger.log(level, message, extra=extra)

    def debug(self, message: str, **kwargs):
        self._log(logging.DEBUG, message, **kwargs)

    def info(self, message: str, **kwargs):
        self._log(logging.INFO, message, **kwargs)

    def warning(self, message: str, **kwargs):
        self._log(logging.WARNING, message, **kwargs)

    def error(self, message: str, **kwargs):
        self._log(logging.ERROR, message, **kwargs)

    def circuit_submitted(
        self,
        circuit_name: str,
        backend: str,
        job_id: str,
        num_qubits: int,
        depth: int,
        shots: int
    ):
        """Log circuit submission."""
        self.info(
            f"Circuit '{circuit_name}' submitted to {backend}",
            quantum_context={
                'circuit_name': circuit_name,
                'backend': backend,
                'job_id': job_id,
                'num_qubits': num_qubits,
                'circuit_depth': depth,
                'shots': shots
            }
        )

    def circuit_completed(
        self,
        circuit_name: str,
        job_id: str,
        duration_seconds: float,
        success: bool,
        fidelity: Optional[float] = None
    ):
        """Log circuit completion."""
        self.info(
            f"Circuit '{circuit_name}' completed",
            quantum_context={
                'circuit_name': circuit_name,
                'job_id': job_id
            },
            extra={
                'duration_seconds': duration_seconds,
                'success': success,
                'fidelity': fidelity
            }
        )


def setup_logging(
    level: int = logging.INFO,
    json_output: bool = True,
    log_file: Optional[str] = None
):
    """Configure logging for the application."""
    root_logger = logging.getLogger()
    root_logger.setLevel(level)

    # Remove existing handlers
    root_logger.handlers.clear()

    # Create handler
    if log_file:
        handler = logging.FileHandler(log_file)
    else:
        handler = logging.StreamHandler()

    # Set formatter
    if json_output:
        handler.setFormatter(StructuredFormatter())
    else:
        handler.setFormatter(logging.Formatter(
            '%(asctime)s - %(name)s - %(levelname)s - %(message)s'
        ))

    root_logger.addHandler(handler)

    return root_logger
```

## 3. Distributed Tracing with OpenTelemetry

OpenTelemetry provides a vendor-neutral way to instrument applications for distributed tracing.

```python
# src/quantum_app/tracing/otel_tracing.py
"""OpenTelemetry tracing for quantum applications."""
from opentelemetry import trace
from opentelemetry.sdk.trace import TracerProvider
from opentelemetry.sdk.trace.export import BatchSpanProcessor
from opentelemetry.exporter.otlp.proto.grpc.trace_exporter import OTLPSpanExporter
from opentelemetry.sdk.resources import Resource
from opentelemetry.semconv.resource import ResourceAttributes
from opentelemetry.trace import Status, StatusCode
from opentelemetry.trace.propagation.tracecontext import TraceContextTextMapPropagator

from contextlib import contextmanager
from typing import Optional, Dict, Any, Generator
from functools import wraps
import time


def setup_tracing(
    service_name: str,
    otlp_endpoint: str = "http://localhost:4317",
    environment: str = "development"
) -> trace.Tracer:
    """Initialize OpenTelemetry tracing."""
    resource = Resource.create({
        ResourceAttributes.SERVICE_NAME: service_name,
        ResourceAttributes.DEPLOYMENT_ENVIRONMENT: environment,
        "quantum.service": "true"
    })

    provider = TracerProvider(resource=resource)
    exporter = OTLPSpanExporter(endpoint=otlp_endpoint, insecure=True)
    provider.add_span_processor(BatchSpanProcessor(exporter))

    trace.set_tracer_provider(provider)

    return trace.get_tracer(service_name)


class QuantumTracer:
    """Tracing utilities for quantum operations."""

    def __init__(self, tracer: trace.Tracer):
        self.tracer = tracer
        self.propagator = TraceContextTextMapPropagator()

    @contextmanager
    def trace_circuit_execution(
        self,
        circuit_name: str,
        backend: str,
        num_qubits: int,
        depth: int,
        shots: int
    ) -> Generator[trace.Span, None, None]:
        """Create a span for circuit execution."""
        with self.tracer.start_as_current_span(
            "quantum.circuit.execute",
            attributes={
                "quantum.circuit.name": circuit_name,
                "quantum.backend": backend,
                "quantum.num_qubits": num_qubits,
                "quantum.circuit.depth": depth,
                "quantum.shots": shots
            }
        ) as span:
            try:
                yield span
            except Exception as e:
                span.set_status(Status(StatusCode.ERROR, str(e)))
                span.record_exception(e)
                raise

    @contextmanager
    def trace_transpilation(
        self,
        circuit_name: str,
        optimization_level: int
    ) -> Generator[trace.Span, None, None]:
        """Create a span for circuit transpilation."""
        with self.tracer.start_as_current_span(
            "quantum.circuit.transpile",
            attributes={
                "quantum.circuit.name": circuit_name,
                "quantum.optimization_level": optimization_level
            }
        ) as span:
            yield span

    @contextmanager
    def trace_optimization_iteration(
        self,
        algorithm: str,
        iteration: int,
        experiment_id: str
    ) -> Generator[trace.Span, None, None]:
        """Create a span for an optimization iteration."""
        with self.tracer.start_as_current_span(
            "quantum.optimization.iteration",
            attributes={
                "quantum.algorithm": algorithm,
                "quantum.iteration": iteration,
                "quantum.experiment_id": experiment_id
            }
        ) as span:
            yield span

    def add_result_attributes(
        self,
        span: trace.Span,
        success: bool,
        fidelity: Optional[float] = None,
        duration_ms: Optional[float] = None,
        job_id: Optional[str] = None
    ):
        """Add result attributes to a span."""
        span.set_attribute("quantum.success", success)
        if fidelity is not None:
            span.set_attribute("quantum.fidelity", fidelity)
        if duration_ms is not None:
            span.set_attribute("quantum.duration_ms", duration_ms)
        if job_id is not None:
            span.set_attribute("quantum.job_id", job_id)

    def inject_context(self, carrier: Dict[str, str]):
        """Inject trace context into a carrier for propagation."""
        self.propagator.inject(carrier)

    def extract_context(self, carrier: Dict[str, str]):
        """Extract trace context from a carrier."""
        return self.propagator.extract(carrier)


def traced(span_name: Optional[str] = None):
    """Decorator to automatically trace a function."""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            tracer = trace.get_tracer(__name__)
            name = span_name or f"{func.__module__}.{func.__name__}"

            with tracer.start_as_current_span(name) as span:
                start_time = time.time()
                try:
                    result = func(*args, **kwargs)
                    span.set_status(Status(StatusCode.OK))
                    return result
                except Exception as e:
                    span.set_status(Status(StatusCode.ERROR, str(e)))
                    span.record_exception(e)
                    raise
                finally:
                    duration = (time.time() - start_time) * 1000
                    span.set_attribute("duration_ms", duration)

        return wrapper
    return decorator
```

## 4. Correlation IDs and Request Tracking

Track requests across the entire quantum pipeline with correlation IDs.

```python
# src/quantum_app/tracing/correlation.py
"""Correlation ID management for request tracking."""
import uuid
from contextvars import ContextVar
from dataclasses import dataclass, field
from datetime import datetime, timezone
from typing import Optional, Dict, Any, List
import json

# Context variables
request_id: ContextVar[Optional[str]] = ContextVar('request_id', default=None)
experiment_id: ContextVar[Optional[str]] = ContextVar('experiment_id', default=None)
job_ids: ContextVar[List[str]] = ContextVar('job_ids', default=[])


@dataclass
class RequestContext:
    """Context for tracking a quantum request through the pipeline."""
    request_id: str
    experiment_id: Optional[str] = None
    user_id: Optional[str] = None
    created_at: str = field(default_factory=lambda: datetime.now(timezone.utc).isoformat())

    # Pipeline stages
    preprocessing_complete: bool = False
    transpilation_complete: bool = False
    execution_complete: bool = False
    postprocessing_complete: bool = False

    # Job tracking
    job_ids: List[str] = field(default_factory=list)

    # Timing
    timings: Dict[str, float] = field(default_factory=dict)

    # Results
    success: Optional[bool] = None
    error: Optional[str] = None

    def mark_stage(self, stage: str, duration_ms: float):
        """Mark a pipeline stage as complete."""
        setattr(self, f"{stage}_complete", True)
        self.timings[stage] = duration_ms

    def add_job(self, job_id: str):
        """Add a quantum job ID."""
        self.job_ids.append(job_id)

    def complete(self, success: bool, error: Optional[str] = None):
        """Mark the request as complete."""
        self.success = success
        self.error = error

    def to_dict(self) -> Dict[str, Any]:
        """Convert to dictionary for logging."""
        return {
            'request_id': self.request_id,
            'experiment_id': self.experiment_id,
            'user_id': self.user_id,
            'created_at': self.created_at,
            'stages': {
                'preprocessing': self.preprocessing_complete,
                'transpilation': self.transpilation_complete,
                'execution': self.execution_complete,
                'postprocessing': self.postprocessing_complete
            },
            'job_ids': self.job_ids,
            'timings': self.timings,
            'success': self.success,
            'error': self.error,
            'total_duration_ms': sum(self.timings.values())
        }


class CorrelationManager:
    """Manages correlation IDs and request context."""

    def __init__(self):
        self._contexts: Dict[str, RequestContext] = {}

    def create_context(
        self,
        experiment_id: Optional[str] = None,
        user_id: Optional[str] = None
    ) -> RequestContext:
        """Create a new request context."""
        req_id = str(uuid.uuid4())
        ctx = RequestContext(
            request_id=req_id,
            experiment_id=experiment_id,
            user_id=user_id
        )

        self._contexts[req_id] = ctx

        # Set context variables
        request_id.set(req_id)
        if experiment_id:
            globals()['experiment_id'].set(experiment_id)

        return ctx

    def get_context(self, req_id: Optional[str] = None) -> Optional[RequestContext]:
        """Get a request context by ID or current."""
        if req_id is None:
            req_id = request_id.get()
        return self._contexts.get(req_id)

    def get_current_request_id(self) -> Optional[str]:
        """Get the current request ID."""
        return request_id.get()

    def cleanup_context(self, req_id: str):
        """Clean up a completed request context."""
        if req_id in self._contexts:
            del self._contexts[req_id]


# Global correlation manager
correlation_manager = CorrelationManager()


def get_request_id() -> Optional[str]:
    """Get the current request ID."""
    return request_id.get()


def with_correlation(func):
    """Decorator to ensure correlation context exists."""
    @wraps(func)
    def wrapper(*args, **kwargs):
        if request_id.get() is None:
            # Create a new context if none exists
            correlation_manager.create_context()
        return func(*args, **kwargs)
    return wrapper
```

```python
# src/quantum_app/algorithms/traced_vqe.py
"""VQE implementation with full tracing."""
from typing import Optional, Callable
import time

from opentelemetry import trace

from ..tracing.otel_tracing import QuantumTracer, traced
from ..tracing.correlation import correlation_manager, get_request_id
from ..logging.structured_logger import QuantumLogger

logger = QuantumLogger(__name__)


class TracedVQE:
    """VQE with comprehensive tracing and logging."""

    def __init__(
        self,
        ansatz,
        hamiltonian,
        optimizer,
        backend,
        tracer: QuantumTracer
    ):
        self.ansatz = ansatz
        self.hamiltonian = hamiltonian
        self.optimizer = optimizer
        self.backend = backend
        self.tracer = tracer

    def run(
        self,
        initial_params,
        max_iterations: int = 100,
        experiment_id: Optional[str] = None
    ):
        """Run VQE with full tracing."""
        # Create correlation context
        ctx = correlation_manager.create_context(experiment_id=experiment_id)
        logger.set_context(experiment_id=experiment_id, request_id=ctx.request_id)

        logger.info(f"Starting VQE optimization", extra={
            'max_iterations': max_iterations,
            'num_parameters': len(initial_params)
        })

        otel_tracer = trace.get_tracer(__name__)

        with otel_tracer.start_as_current_span(
            "vqe.optimization",
            attributes={
                "request_id": ctx.request_id,
                "experiment_id": experiment_id,
                "max_iterations": max_iterations
            }
        ) as root_span:

            try:
                # Stage 1: Preprocessing
                start = time.time()
                with otel_tracer.start_as_current_span("vqe.preprocessing"):
                    circuit = self._prepare_circuit()
                ctx.mark_stage('preprocessing', (time.time() - start) * 1000)

                # Stage 2: Optimization loop
                params = initial_params
                best_energy = float('inf')
                best_params = None

                for iteration in range(max_iterations):
                    with self.tracer.trace_optimization_iteration(
                        algorithm="VQE",
                        iteration=iteration,
                        experiment_id=experiment_id
                    ) as iter_span:

                        # Transpile
                        start = time.time()
                        with self.tracer.trace_transpilation(
                            circuit_name="vqe_ansatz",
                            optimization_level=3
                        ):
                            transpiled = self._transpile(circuit, params)
                        transpile_time = (time.time() - start) * 1000

                        # Execute
                        start = time.time()
                        with self.tracer.trace_circuit_execution(
                            circuit_name="vqe_ansatz",
                            backend=self.backend.name,
                            num_qubits=circuit.num_qubits,
                            depth=transpiled.depth(),
                            shots=4000
                        ) as exec_span:
                            energy, job_id = self._execute(transpiled)
                            ctx.add_job(job_id)
                            self.tracer.add_result_attributes(
                                exec_span,
                                success=True,
                                duration_ms=(time.time() - start) * 1000,
                                job_id=job_id
                            )
                        exec_time = (time.time() - start) * 1000

                        # Log iteration
                        logger.info(
                            f"VQE iteration {iteration}: energy={energy:.6f}",
                            quantum_context={
                                'circuit_name': 'vqe_ansatz',
                                'backend': self.backend.name,
                                'job_id': job_id
                            },
                            extra={
                                'iteration': iteration,
                                'energy': energy,
                                'transpile_time_ms': transpile_time,
                                'exec_time_ms': exec_time
                            }
                        )

                        # Update best
                        if energy < best_energy:
                            best_energy = energy
                            best_params = params.copy()

                        # Optimizer step
                        params = self.optimizer.step(params, energy)

                        iter_span.set_attribute("energy", energy)
                        iter_span.set_attribute("best_energy", best_energy)

                # Stage 3: Post-processing
                start = time.time()
                with otel_tracer.start_as_current_span("vqe.postprocessing"):
                    result = self._postprocess(best_params, best_energy)
                ctx.mark_stage('postprocessing', (time.time() - start) * 1000)

                ctx.mark_stage('execution', sum(ctx.timings.values()))
                ctx.complete(success=True)

                root_span.set_attribute("final_energy", best_energy)
                root_span.set_attribute("total_iterations", max_iterations)

                logger.info(
                    f"VQE completed: energy={best_energy:.6f}",
                    extra=ctx.to_dict()
                )

                return result

            except Exception as e:
                ctx.complete(success=False, error=str(e))
                logger.error(f"VQE failed: {e}", extra=ctx.to_dict())
                raise

    def _prepare_circuit(self):
        """Prepare the ansatz circuit."""
        return self.ansatz

    def _transpile(self, circuit, params):
        """Transpile circuit with current parameters."""
        bound = circuit.bind_parameters(params)
        return self.backend.transpile(bound, optimization_level=3)

    def _execute(self, circuit):
        """Execute circuit and return energy."""
        job = self.backend.run(circuit, shots=4000)
        result = job.result()
        energy = self._compute_energy(result)
        return energy, job.job_id()
```

Create comprehensive audit trails for quantum experiments.

```python
# src/quantum_app/audit/audit_trail.py
"""Audit trail for quantum experiment reproducibility."""
from dataclasses import dataclass, field, asdict
from datetime import datetime, timezone
from typing import Dict, List, Any, Optional
import hashlib
import json
from pathlib import Path


@dataclass
class AuditEntry:
    """Single audit trail entry."""
    timestamp: str
    event_type: str
    description: str
    actor: str
    data: Dict[str, Any] = field(default_factory=dict)
    hash: str = ""

    def __post_init__(self):
        if not self.hash:
            self.hash = self._compute_hash()

    def _compute_hash(self) -> str:
        """Compute hash of entry for integrity verification."""
        content = json.dumps({
            'timestamp': self.timestamp,
            'event_type': self.event_type,
            'description': self.description,
            'actor': self.actor,
            'data': self.data
        }, sort_keys=True)
        return hashlib.sha256(content.encode()).hexdigest()[:16]


@dataclass
class ExperimentAuditTrail:
    """Complete audit trail for a quantum experiment."""
    experiment_id: str
    created_at: str
    created_by: str

    # Experiment configuration
    circuit_hash: str
    backend: str
    shots: int
    optimization_level: int
    error_mitigation: Dict[str, Any]

    # Version information
    git_commit: Optional[str] = None
    qiskit_version: str = ""
    dependencies: Dict[str, str] = field(default_factory=dict)

    # Entries
    entries: List[AuditEntry] = field(default_factory=list)

    # Chain hash for tamper detection
    chain_hash: str = ""

    def add_entry(
        self,
        event_type: str,
        description: str,
        actor: str,
        data: Optional[Dict] = None
    ):
        """Add an entry to the audit trail."""
        entry = AuditEntry(
            timestamp=datetime.now(timezone.utc).isoformat(),
            event_type=event_type,
            description=description,
            actor=actor,
            data=data or {}
        )
        self.entries.append(entry)
        self._update_chain_hash(entry)

    def _update_chain_hash(self, entry: AuditEntry):
        """Update chain hash with new entry."""
        content = f"{self.chain_hash}{entry.hash}"
        self.chain_hash = hashlib.sha256(content.encode()).hexdigest()[:16]

    def verify_integrity(self) -> bool:
        """Verify the integrity of the audit trail."""
        computed_hash = ""
        for entry in self.entries:
            # Verify entry hash
            expected_hash = entry._compute_hash()
            if entry.hash != expected_hash:
                return False

            # Update chain
            computed_hash = hashlib.sha256(
                f"{computed_hash}{entry.hash}".encode()
            ).hexdigest()[:16]

        return computed_hash == self.chain_hash

    def save(self, directory: str = "audit"):
        """Save audit trail to file."""
        path = Path(directory)
        path.mkdir(parents=True, exist_ok=True)

        filename = f"audit_{self.experiment_id}.json"
        filepath = path / filename

        with open(filepath, 'w') as f:
            json.dump(asdict(self), f, indent=2)

        return str(filepath)

    @classmethod
    def load(cls, filepath: str) -> "ExperimentAuditTrail":
        """Load audit trail from file."""
        with open(filepath, 'r') as f:
            data = json.load(f)

        # Reconstruct entries
        entries = [AuditEntry(**e) for e in data.pop('entries', [])]
        data['entries'] = entries

        return cls(**data)


class AuditLogger:
    """Logger that creates audit trail entries."""

    def __init__(self, trail: ExperimentAuditTrail, actor: str):
        self.trail = trail
        self.actor = actor

    def circuit_submitted(self, circuit_name: str, job_id: str, **kwargs):
        """Log circuit submission."""
        self.trail.add_entry(
            event_type="CIRCUIT_SUBMITTED",
            description=f"Circuit '{circuit_name}' submitted as job {job_id}",
            actor=self.actor,
            data={'circuit_name': circuit_name, 'job_id': job_id, **kwargs}
        )

    def circuit_completed(self, job_id: str, success: bool, **kwargs):
        """Log circuit completion."""
        self.trail.add_entry(
            event_type="CIRCUIT_COMPLETED",
            description=f"Job {job_id} completed: {'success' if success else 'failure'}",
            actor=self.actor,
            data={'job_id': job_id, 'success': success, **kwargs}
        )

    def parameters_updated(self, iteration: int, params: List[float], cost: float):
        """Log parameter update."""
        self.trail.add_entry(
            event_type="PARAMETERS_UPDATED",
            description=f"Iteration {iteration}: cost={cost:.6f}",
            actor=self.actor,
            data={'iteration': iteration, 'cost': cost, 'num_params': len(params)}
        )

    def experiment_completed(self, final_result: Dict[str, Any]):
        """Log experiment completion."""
        self.trail.add_entry(
            event_type="EXPERIMENT_COMPLETED",
            description="Experiment completed successfully",
            actor=self.actor,
            data=final_result
        )
```
