## 2. KAK Decomposition

The Cartan/KAK decomposition: U = (A1 ⊗ A2) @ exp(-i(xx*XX + yy*YY + zz*ZZ)) @ (B1 ⊗ B2).

```python
from scipy.linalg import expm

class KAKDecomposition:
    """
    KAK decomposition for two-qubit unitaries.

    U = (A1 ⊗ A2) @ K @ (B1 ⊗ B2)

    where K = exp(-i(c1*XX + c2*YY + c3*ZZ))
    """

    def __init__(self, U: np.ndarray):
        self.U = U
        self.decompose()

    def decompose(self):
        """Perform KAK decomposition."""
        # Magic basis transformation
        M = np.array([
            [1, 0, 0, 1j],
            [0, 1j, 1, 0],
            [0, 1j, -1, 0],
            [1, 0, 0, -1j]
        ], dtype=complex) / np.sqrt(2)

        # Transform to magic basis
        U_magic = M.conj().T @ self.U @ M

        # Compute U^T @ U
        UtU = U_magic.T @ U_magic

        # Eigendecompose to get interaction coefficients
        eigenvalues, _ = np.linalg.eig(UtU)

        # Extract c1, c2, c3 from eigenvalues
        phases = np.angle(eigenvalues) / 2
        phases = np.sort(phases)

        # Canonical ordering
        self.c1 = (phases[3] - phases[0]) / 2
        self.c2 = (phases[2] - phases[1]) / 2
        self.c3 = (phases[3] + phases[0] - phases[2] - phases[1]) / 4

        # Compute local unitaries (simplified)
        self.A1, self.A2, self.B1, self.B2 = self._compute_locals()

    def _compute_locals(self) -> Tuple[np.ndarray, ...]:
        """Compute local single-qubit unitaries."""

    def to_cnot_circuit(self) -> List[Tuple]:

        gates = []

        # Pre-rotation B1, B2
```

## 5. CZ vs CNOT Conversion

```python
def cx_to_cz() -> List[Tuple]:
    """Convert CNOT to CZ using Hadamards."""
    # CX(0,1) = H(1) @ CZ(0,1) @ H(1)
    return [
        ('H', 1),
        ('CZ', 0, 1),
        ('H', 1)
    ]

def cz_to_cx() -> List[Tuple]:
    """Convert CZ to CNOT using Hadamards."""
    # CZ(0,1) = H(1) @ CX(0,1) @ H(1)
    return [
        ('H', 1),
        ('CX', 0, 1),
        ('H', 1)
    ]

class TwoQubitDecomposer:
    """Decompose two-qubit gates for specific backends."""

    def __init__(self, native_two_qubit: str = 'CX'):
        self.native = native_two_qubit

    def decompose(self, gate: str, qubits: Tuple[int, int]) -> List[Tuple]:
        """Decompose gate to native operations."""
        if gate == self.native:
            return [(gate, qubits[0], qubits[1])]

        if gate == 'SWAP':
            if self.native == 'CX':
                return [
                    ('CX', qubits[0], qubits[1]),
                    ('CX', qubits[1], qubits[0]),
                    ('CX', qubits[0], qubits[1]),
                ]
            elif self.native == 'CZ':
                # SWAP with CZ: more complex
                pass

        if gate == 'CZ' and self.native == 'CX':
            return [
                ('H', qubits[1]),
                ('CX', qubits[0], qubits[1]),
                ('H', qubits[1]),
            ]

        if gate == 'CX' and self.native == 'CZ':
            return [
                ('H', qubits[1]),
                ('CZ', qubits[0], qubits[1]),
```

## Summary

Two-qubit decomposition is essential for transpilation:
1. **KAK decomposition**: Optimal CNOT count (0-3)
2. **SWAP decomposition**: 3 CNOTs
3. **Controlled-U**: ABC decomposition
4. **Native gate conversion**: CX <-> CZ via Hadamards
