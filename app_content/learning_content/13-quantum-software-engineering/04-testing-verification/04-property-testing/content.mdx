## 1. Property-Based Testing Paradigm

Property-based testing verifies that code satisfies mathematical properties for arbitrary inputs:

```
Traditional Testing        Property-Based Testing
------------------         ----------------------
test(H, |0>) -> |+>        forall state: H(H(state)) == state
test(H, |1>) -> |->        forall gate: is_unitary(gate)
Fixed inputs               Generated inputs
```

## 2. Quantum Gate Properties

```python
import numpy as np
from typing import Callable, List, Tuple, Any
from dataclasses import dataclass

@dataclass
class QuantumProperty:
    """Defines a testable property."""
    name: str
    predicate: Callable[..., bool]
    generators: List[Callable]
    description: str

class QuantumPropertyTester:
    """Property-based testing framework for quantum circuits."""

    def __init__(self, num_samples: int = 100, tolerance: float = 1e-10):
        self.num_samples = num_samples
        self.tolerance = tolerance
        self.failures: List[dict] = []

    def check_property(self, prop: QuantumProperty) -> bool:
        """Check property holds for generated inputs."""
        self.failures = []

        for _ in range(self.num_samples):
            # Generate inputs from each generator
            inputs = [gen() for gen in prop.generators]

            try:
                if not prop.predicate(*inputs):
                    self.failures.append({
                        'property': prop.name,
                        'inputs': inputs,
                        'error': 'Property violated'
                    })
            except Exception as e:
                self.failures.append({
                    'property': prop.name,
                    'inputs': inputs,
                    'error': str(e)
                })

        return len(self.failures) == 0

    def shrink_failure(self, prop: QuantumProperty,
                       failing_inputs: List) -> List:
        """Attempt to find minimal failing example."""
        # Simple shrinking: try smaller values
        minimal = failing_inputs

        for i, inp in enumerate(failing_inputs):
            if isinstance(inp, np.ndarray):
                # Try simpler matrices
                for simpler in self._shrink_matrix(inp):
                    test_inputs = failing_inputs.copy()
                    test_inputs[i] = simpler
                    if not prop.predicate(*test_inputs):
                        minimal = test_inputs
                        break

        return minimal

    def _shrink_matrix(self, matrix: np.ndarray) -> List[np.ndarray]:
        """Generate simpler versions of a matrix."""
        shrunk = []
        # Try identity
        shrunk.append(np.eye(matrix.shape[0]))
        # Try with smaller values
        shrunk.append(matrix / 2)
        # Try with rounded values
        shrunk.append(np.round(matrix, 2))
        return shrunk
```

## 3. Essential Quantum Properties

```python
class QuantumProperties:
    """Library of quantum computing properties."""

    @staticmethod
    def unitarity() -> QuantumProperty:
        """Gates must be unitary: U^dag U = I."""
        def check_unitary(matrix: np.ndarray) -> bool:
            product = matrix.conj().T @ matrix
            identity = np.eye(matrix.shape[0])
            return np.allclose(product, identity, atol=1e-10)

        return QuantumProperty(
            name="unitarity",
            predicate=check_unitary,
            generators=[GateGenerators.random_unitary],
            description="U^dagger U = I"
        )

    @staticmethod
    def hermiticity() -> QuantumProperty:
        """Observables must be Hermitian: O^dag = O."""
        def check_hermitian(matrix: np.ndarray) -> bool:
            return np.allclose(matrix, matrix.conj().T, atol=1e-10)

        return QuantumProperty(
            name="hermiticity",
            predicate=check_hermitian,
            generators=[GateGenerators.random_observable],
            description="O^dagger = O"
        )

    @staticmethod
    def involution() -> QuantumProperty:
        """Self-inverse gates: G @ G = I (e.g., Pauli, Hadamard)."""
        def check_involution(gate: np.ndarray) -> bool:
            product = gate @ gate
            return np.allclose(product, np.eye(gate.shape[0]), atol=1e-10)

        return QuantumProperty(
            name="involution",
            predicate=check_involution,
            generators=[GateGenerators.random_involution],
            description="G @ G = I"
        )

    @staticmethod
    def trace_preservation() -> QuantumProperty:
        """Quantum channels preserve trace: Tr(E(rho)) = Tr(rho)."""
        def check_trace_preserving(channel: Callable,
                                   rho: np.ndarray) -> bool:
            original_trace = np.trace(rho)
            evolved = channel(rho)
            new_trace = np.trace(evolved)
            return np.isclose(original_trace, new_trace, atol=1e-10)

        return QuantumProperty(
            name="trace_preservation",
            predicate=check_trace_preserving,
            generators=[GateGenerators.random_channel,
                       GateGenerators.random_density_matrix],
            description="Tr(E(rho)) = Tr(rho)"
        )

    @staticmethod
    def positivity_preservation() -> QuantumProperty:
        """Channels preserve positive semidefiniteness."""
        def check_positive(channel: Callable, rho: np.ndarray) -> bool:
            evolved = channel(rho)
            eigenvalues = np.linalg.eigvalsh(evolved)
            return np.all(eigenvalues >= -1e-10)

        return QuantumProperty(
            name="positivity",
            predicate=check_positive,
            generators=[GateGenerators.random_channel,
                       GateGenerators.random_density_matrix],
            description="E(rho) >= 0 for rho >= 0"
        )

    @staticmethod
    def normalization() -> QuantumProperty:
        """State vectors must be normalized."""
        def check_normalized(state: np.ndarray) -> bool:
            norm = np.linalg.norm(state)
            return np.isclose(norm, 1.0, atol=1e-10)

        return QuantumProperty(
            name="normalization",
            predicate=check_normalized,
            generators=[GateGenerators.random_statevector],
            description="||psi|| = 1"
        )
```

## 4. Input Generators

```python
class GateGenerators:
    """Random generators for quantum objects."""

    @staticmethod
    def random_unitary(dim: int = 2) -> np.ndarray:
        """Generate random unitary matrix using QR decomposition."""
        # Random complex matrix
        real = np.random.randn(dim, dim)
        imag = np.random.randn(dim, dim)
        random_matrix = real + 1j * imag

        # QR decomposition gives orthogonal Q
        q, r = np.linalg.qr(random_matrix)

        # Ensure determinant is 1 (special unitary)
        diagonal = np.diag(r)
        phase = diagonal / np.abs(diagonal)
        return q @ np.diag(phase)

    @staticmethod
    def random_statevector(num_qubits: int = 2) -> np.ndarray:
        """Generate random normalized statevector."""
        dim = 2 ** num_qubits
        real = np.random.randn(dim)
        imag = np.random.randn(dim)
        state = real + 1j * imag
        return state / np.linalg.norm(state)

    @staticmethod
    def random_density_matrix(dim: int = 2) -> np.ndarray:
        """Generate random valid density matrix."""
        # Random state for pure density matrix
        state = GateGenerators.random_statevector(
            int(np.log2(dim))
        )
        return np.outer(state, state.conj())

    @staticmethod
    def random_observable(dim: int = 2) -> np.ndarray:
        """Generate random Hermitian observable."""
        real = np.random.randn(dim, dim)
        imag = np.random.randn(dim, dim)
        matrix = real + 1j * imag
        # Make Hermitian: (A + A^dag) / 2
        return (matrix + matrix.conj().T) / 2

    @staticmethod
    def random_involution(dim: int = 2) -> np.ndarray:
        """Generate random self-inverse unitary."""
        # Construct from reflection formula
        v = GateGenerators.random_statevector(int(np.log2(dim)))
        # Householder reflection: I - 2|v><v|
        return np.eye(dim) - 2 * np.outer(v, v.conj())

    @staticmethod
    def random_channel() -> Callable:
        """Generate random quantum channel (depolarizing)."""
        p = np.random.uniform(0, 1)

        def depolarizing(rho: np.ndarray) -> np.ndarray:
            dim = rho.shape[0]
            return (1 - p) * rho + p * np.eye(dim) / dim

        return depolarizing

    @staticmethod
    def random_rotation_angle() -> float:
        """Generate random rotation angle."""
        return np.random.uniform(0, 2 * np.pi)

    @staticmethod
    def random_pauli() -> np.ndarray:
        """Return random Pauli matrix."""
        paulis = [
            np.eye(2),
            np.array([[0, 1], [1, 0]]),
            np.array([[0, -1j], [1j, 0]]),
            np.array([[1, 0], [0, -1]])
        ]
        return paulis[np.random.randint(4)]
```

## 5. Algebraic Properties of Gate Composition

```python
class AlgebraicProperties:
    """Properties related to gate algebra."""

    @staticmethod
    def composition_associativity() -> QuantumProperty:
        """Matrix multiplication is associative: (AB)C = A(BC)."""
        def check_associative(A: np.ndarray, B: np.ndarray,
                              C: np.ndarray) -> bool:
            left = (A @ B) @ C
            right = A @ (B @ C)
            return np.allclose(left, right, atol=1e-10)

        return QuantumProperty(
            name="associativity",
            predicate=check_associative,
            generators=[GateGenerators.random_unitary] * 3,
            description="(AB)C = A(BC)"
        )

    @staticmethod
    def tensor_product_properties() -> QuantumProperty:
        """Tensor product distributes: (A x B)(C x D) = AC x BD."""
        def check_tensor_distribution(A: np.ndarray, B: np.ndarray,
                                      C: np.ndarray, D: np.ndarray) -> bool:
            left = np.kron(A, B) @ np.kron(C, D)
            right = np.kron(A @ C, B @ D)
            return np.allclose(left, right, atol=1e-10)

        return QuantumProperty(
            name="tensor_distribution",
            predicate=check_tensor_distribution,
            generators=[GateGenerators.random_unitary] * 4,
            description="(A x B)(C x D) = AC x BD"
        )

    @staticmethod
    def commutator_properties() -> QuantumProperty:
        """[A, B] = AB - BA is anti-Hermitian for Hermitian A, B."""
        def check_commutator_anti_hermitian(A: np.ndarray,
                                            B: np.ndarray) -> bool:
            commutator = A @ B - B @ A
            # Should be anti-Hermitian: [A,B]^dag = -[A,B]
            return np.allclose(commutator.conj().T, -commutator, atol=1e-10)

        return QuantumProperty(
            name="commutator_anti_hermitian",
            predicate=check_commutator_anti_hermitian,
            generators=[GateGenerators.random_observable] * 2,
            description="[H1, H2]^dag = -[H1, H2]"
        )

    @staticmethod
    def rotation_composition() -> QuantumProperty:
        """Rotation gates compose: Rz(a)Rz(b) = Rz(a+b)."""
        def rz(theta: float) -> np.ndarray:
            return np.array([
                [np.exp(-1j * theta / 2), 0],
                [0, np.exp(1j * theta / 2)]
            ])

        def check_rotation_composition(a: float, b: float) -> bool:
            composed = rz(a) @ rz(b)
            direct = rz(a + b)
            # Check equality up to global phase
            if np.abs(composed[0, 0]) < 1e-10:
                return True
            phase = direct[0, 0] / composed[0, 0]
            return np.allclose(composed * phase, direct, atol=1e-10)

        return QuantumProperty(
            name="rotation_composition",
            predicate=check_rotation_composition,
            generators=[GateGenerators.random_rotation_angle] * 2,
            description="Rz(a)Rz(b) = Rz(a+b)"
        )
```

## 6. Invariant Testing

```python
class InvariantTester:
    """Test that operations preserve specific invariants."""

    def __init__(self, tolerance: float = 1e-10):
        self.tolerance = tolerance

    def test_invariant(self, operation: Callable,
                       invariant: Callable,
                       generator: Callable,
                       num_samples: int = 100) -> dict:
        """
        Test that invariant(x) == invariant(operation(x)).

        Args:
            operation: Function to test
            invariant: Property that should be preserved
            generator: Generates random inputs
            num_samples: Number of test cases
        """
        results = {
            'passed': 0,
            'failed': 0,
            'failures': []
        }

        for _ in range(num_samples):
            x = generator()
            before = invariant(x)
            after = invariant(operation(x))

            if np.isclose(before, after, atol=self.tolerance):
                results['passed'] += 1
            else:
                results['failed'] += 1
                results['failures'].append({
                    'input': x,
                    'before': before,
                    'after': after
                })

        return results

    def test_entropy_invariant(self, channel: Callable) -> dict:
        """Test that von Neumann entropy doesn't decrease for unital channels."""
        def entropy(rho: np.ndarray) -> float:
            eigenvalues = np.linalg.eigvalsh(rho)
            eigenvalues = eigenvalues[eigenvalues > 1e-12]
            return -np.sum(eigenvalues * np.log2(eigenvalues))

        return self.test_invariant(
            operation=channel,
            invariant=entropy,
            generator=lambda: GateGenerators.random_density_matrix(2)
        )

    def test_purity_under_unitary(self) -> dict:
        """Test that purity Tr(rho^2) is invariant under unitary evolution."""
        def purity(rho: np.ndarray) -> float:
            return np.real(np.trace(rho @ rho))

        def random_unitary_evolution(rho: np.ndarray) -> np.ndarray:
            U = GateGenerators.random_unitary(rho.shape[0])
            return U @ rho @ U.conj().T

        return self.test_invariant(
            operation=random_unitary_evolution,
            invariant=purity,
            generator=lambda: GateGenerators.random_density_matrix(2)
        )

    def test_eigenvalue_invariant(self) -> dict:
        """Test that unitary evolution preserves eigenvalues."""
        def eigenvalues_sorted(rho: np.ndarray) -> np.ndarray:
            return np.sort(np.linalg.eigvalsh(rho))

        def unitary_evolution(rho: np.ndarray) -> np.ndarray:
            U = GateGenerators.random_unitary(rho.shape[0])
            return U @ rho @ U.conj().T

        results = {'passed': 0, 'failed': 0, 'failures': []}

        for _ in range(100):
            rho = GateGenerators.random_density_matrix(2)
            before = eigenvalues_sorted(rho)
            after = eigenvalues_sorted(unitary_evolution(rho))

            if np.allclose(before, after, atol=1e-10):
                results['passed'] += 1
            else:
                results['failed'] += 1

        return results
```

## 7. Hypothesis-Style Test Framework

```python
class HypothesisTester:
    """Hypothesis-style property testing with shrinking."""

    def __init__(self, max_examples: int = 100):
        self.max_examples = max_examples

    def given(self, *strategies):
        """Decorator for property-based tests."""
        def decorator(test_func: Callable) -> Callable:
            def wrapper():
                for _ in range(self.max_examples):
                    # Generate inputs
                    inputs = [strategy() for strategy in strategies]

                    try:
                        test_func(*inputs)
                    except AssertionError as e:
                        # Shrink inputs
                        shrunk = self._shrink_inputs(inputs, strategies,
                                                     test_func)
                        raise AssertionError(
                            f"Property failed for: {shrunk}\n"
                            f"Original: {inputs}\n"
                            f"Error: {e}"
                        )

                return True
            return wrapper
        return decorator

    def _shrink_inputs(self, inputs: List, strategies: List,
                       test_func: Callable) -> List:
        """Attempt to find minimal failing example."""
        minimal = inputs

        for i, inp in enumerate(inputs):
            for smaller in self._shrink_value(inp):
                test_inputs = list(minimal)
                test_inputs[i] = smaller

                try:
                    test_func(*test_inputs)
                except AssertionError:
                    minimal = test_inputs
                    break

        return minimal

    def _shrink_value(self, value: Any) -> List:
        """Generate smaller versions of a value."""
        if isinstance(value, np.ndarray):
            return [
                np.eye(value.shape[0]),
                value / 2,
                np.round(value, 1)
            ]
        elif isinstance(value, float):
            return [0.0, value / 2, round(value, 1)]
        elif isinstance(value, int):
            return [0, value // 2, value - 1]
        return []


# Usage example
tester = HypothesisTester()

@tester.given(GateGenerators.random_unitary)
def test_unitary_determinant_is_one(U: np.ndarray):
    """Determinant of unitary has magnitude 1."""
    det = np.linalg.det(U)
    assert np.isclose(np.abs(det), 1.0, atol=1e-10), \
        f"|det(U)| = {np.abs(det)}"

@tester.given(GateGenerators.random_statevector,
              GateGenerators.random_unitary)
def test_unitary_preserves_norm(state: np.ndarray, U: np.ndarray):
    """Unitary transformation preserves state norm."""
    original_norm = np.linalg.norm(state)
    new_state = U @ state
    new_norm = np.linalg.norm(new_state)
    assert np.isclose(original_norm, new_norm, atol=1e-10)
```

## 8. Parameterized Circuit Properties

```python
class ParameterizedCircuitProperties:
    """Properties for variational quantum circuits."""

    @staticmethod
    def gradient_consistency() -> QuantumProperty:
        """Numerical and analytical gradients should match."""
        def check_gradient(params: np.ndarray) -> bool:
            def circuit_output(p: np.ndarray) -> float:
                # Simple parameterized circuit: Rz(p[0])Ry(p[1])|0>
                theta, phi = p[0], p[1]
                state = np.array([1, 0])
                # Rz
                rz = np.array([[np.exp(-1j*theta/2), 0],
                              [0, np.exp(1j*theta/2)]])
                # Ry
                ry = np.array([[np.cos(phi/2), -np.sin(phi/2)],
                              [np.sin(phi/2), np.cos(phi/2)]])
                state = ry @ rz @ state
                return np.abs(state[0])**2  # P(0)

            # Parameter shift gradient
            shift = np.pi / 2
            analytical_grad = np.zeros(len(params))
            for i in range(len(params)):
                params_plus = params.copy()
                params_plus[i] += shift
                params_minus = params.copy()
                params_minus[i] -= shift
                analytical_grad[i] = (circuit_output(params_plus) -
                                     circuit_output(params_minus)) / 2

            # Numerical gradient
            epsilon = 1e-7
            numerical_grad = np.zeros(len(params))
            for i in range(len(params)):
                params_plus = params.copy()
                params_plus[i] += epsilon
                params_minus = params.copy()
                params_minus[i] -= epsilon
                numerical_grad[i] = (circuit_output(params_plus) -
                                    circuit_output(params_minus)) / (2 * epsilon)

            return np.allclose(analytical_grad, numerical_grad, atol=1e-5)

        return QuantumProperty(
            name="gradient_consistency",
            predicate=check_gradient,
            generators=[lambda: np.random.uniform(0, 2*np.pi, 2)],
            description="Analytical gradient matches numerical"
        )

    @staticmethod
    def periodicity() -> QuantumProperty:
        """Rotation gates are 2pi periodic."""
        def check_periodicity(theta: float) -> bool:
            def ry(angle: float) -> np.ndarray:
                return np.array([
                    [np.cos(angle/2), -np.sin(angle/2)],
                    [np.sin(angle/2), np.cos(angle/2)]
                ])

            gate1 = ry(theta)
            gate2 = ry(theta + 2 * np.pi)
            return np.allclose(gate1, gate2, atol=1e-10)

        return QuantumProperty(
            name="rotation_periodicity",
            predicate=check_periodicity,
            generators=[GateGenerators.random_rotation_angle],
            description="Ry(theta) = Ry(theta + 2pi)"
        )
```

## Summary

Property-based testing for quantum programs:

| Property Type | Examples | Verification Method |
|--------------|----------|---------------------|
| **Structural** | Unitarity, Hermiticity | Matrix equations |
| **Algebraic** | Associativity, Distributivity | Composition tests |
| **Invariants** | Trace, Norm, Purity | Before/after comparison |
| **Parameterized** | Periodicity, Gradients | Parameter sweep |

Best practices:
1. Define properties from mathematical specifications
2. Use random generators for comprehensive coverage
3. Implement input shrinking for debugging
4. Test composition and algebraic relations
5. Verify invariants are preserved under operations
