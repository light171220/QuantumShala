## 1. Circuit Composition

```python
import numpy as np
from typing import List, Optional

class Circuit:
    def __init__(self, n_qubits: int, n_clbits: int = 0):
        self.n_qubits = n_qubits
        self.n_clbits = n_clbits
        self.operations = []

    def compose(self, other: 'Circuit', qubits: Optional[List[int]] = None):
        """Compose circuits sequentially."""
        if qubits is None:
            qubits = list(range(other.n_qubits))

        for op in other.operations:
            mapped_qubits = [qubits[q] for q in op['qubits']]
            self.operations.append({
                'gate': op['gate'],
                'qubits': mapped_qubits,
                'params': op.get('params', [])
            })
        return self

    def tensor(self, other: 'Circuit'):
        """Tensor product of circuits."""
        new_circ = Circuit(self.n_qubits + other.n_qubits)
        for op in self.operations:
            new_circ.operations.append(op.copy())
        for op in other.operations:
            shifted = op.copy()
            shifted['qubits'] = [q + self.n_qubits for q in op['qubits']]
            new_circ.operations.append(shifted)
        return new_circ

    def inverse(self):
        """Return inverse circuit."""
        inv = Circuit(self.n_qubits)
        for op in reversed(self.operations):
            inv_op = op.copy()
            inv_op['params'] = [-p for p in op.get('params', [])]
            inv.operations.append(inv_op)
        return inv
```

## 2. Classical Control

```python
class ConditionalGate:
    """Gate conditioned on classical register."""

    def __init__(self, gate, clbit: int, value: int):
        self.gate = gate
        self.clbit = clbit
        self.value = value

def add_conditional(circuit, gate, qubits, clbit, value):
    """Add classically controlled gate."""
    circuit.operations.append({
        'gate': gate,
        'qubits': qubits,
        'condition': {'clbit': clbit, 'value': value}
    })
```

## Summary

Circuit composition enables modular quantum programming:
1. Sequential composition for program flow
2. Tensor product for parallel operations
3. Inverse for uncomputation
4. Classical conditioning for adaptive circuits
