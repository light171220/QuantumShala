## 1. Hybrid Algorithm Structure

```python
from abc import ABC, abstractmethod
from typing import Callable, List
import numpy as np

class HybridAlgorithm(ABC):
    """Base class for hybrid algorithms."""

    def __init__(self, quantum_backend, classical_optimizer):
        self.backend = quantum_backend
        self.optimizer = classical_optimizer
        self.iteration_data = []

    @abstractmethod
    def quantum_circuit(self, params: np.ndarray):
        """Build parameterized quantum circuit."""
        pass

    @abstractmethod
    def cost_function(self, measurement_results) -> float:
        """Compute cost from quantum measurements."""
        pass

    def run(self, initial_params: np.ndarray, max_iterations: int = 100):
        """Execute hybrid optimization loop."""
        params = initial_params.copy()

        for iteration in range(max_iterations):
            # Quantum step
            circuit = self.quantum_circuit(params)
            results = self.backend.execute(circuit)

            # Classical step
            cost = self.cost_function(results)
            self.iteration_data.append({'params': params.copy(), 'cost': cost})

            # Update parameters
            params = self.optimizer.step(params, cost)

        return params, self.iteration_data

class SimpleVQE(HybridAlgorithm):
    """Simple VQE implementation."""

    def __init__(self, hamiltonian, ansatz, backend, optimizer):
        super().__init__(backend, optimizer)
        self.hamiltonian = hamiltonian
        self.ansatz = ansatz

    def quantum_circuit(self, params):
        return self.ansatz.build(params)

    def cost_function(self, results):
        return self.hamiltonian.expectation(results)
```

## 2. Parameter Management

```python
class ParameterManager:
    """Manage variational parameters."""

    def __init__(self, bounds: List[tuple] = None):
        self.params = None
        self.bounds = bounds
        self.history = []

    def initialize(self, shape, method='random'):
        if method == 'random':
            self.params = np.random.uniform(-np.pi, np.pi, shape)
        elif method == 'zeros':
            self.params = np.zeros(shape)
        return self.params

    def update(self, new_params):
        self.history.append(self.params.copy())
        if self.bounds:
            new_params = np.clip(new_params,
                                [b[0] for b in self.bounds],
                                [b[1] for b in self.bounds])
        self.params = new_params
        return self.params
```

## Summary

Hybrid programming combines strengths of both paradigms:
1. Quantum for sampling/interference
2. Classical for optimization/control
3. Iterative feedback loop
