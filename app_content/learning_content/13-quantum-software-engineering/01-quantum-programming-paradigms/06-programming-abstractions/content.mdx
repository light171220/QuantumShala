## 1. Circuit Decorators

```python
from functools import wraps

def quantum_function(num_qubits):
    """Decorator to define quantum functions."""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            circuit = Circuit(num_qubits)
            result = func(circuit, *args, **kwargs)
            return circuit
        return wrapper
    return decorator

@quantum_function(num_qubits=2)
def bell_state(qc):
    qc.h(0)
    qc.cx(0, 1)
    return qc
```

## 2. Context Managers

```python
class QuantumContext:
    """Context manager for quantum circuit building."""

    def __init__(self, num_qubits):
        self.circuit = Circuit(num_qubits)

    def __enter__(self):
        return self.circuit

    def __exit__(self, *args):
        pass

# Usage
with QuantumContext(3) as qc:
    qc.h(0)
    qc.cx(0, 1)
    qc.cx(1, 2)
```

## 3. Fluent Interface

```python
class FluentCircuit:
    """Fluent API for circuit building."""

    def __init__(self, n):
        self.ops = []
        self.n = n

    def h(self, q):
        self.ops.append(('H', q))
        return self  # Enable chaining

    def cx(self, c, t):
        self.ops.append(('CX', c, t))
        return self

    def measure(self):
        self.ops.append(('MEASURE_ALL',))
        return self

# Usage
circuit = FluentCircuit(2).h(0).cx(0, 1).measure()
```

## Summary

Good abstractions make quantum programming accessible:
1. Decorators for reusable patterns
2. Context managers for resource management
3. Fluent interfaces for readability
