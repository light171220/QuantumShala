## 1. Programming Abstraction Levels

Quantum programs can be written at different abstraction levels:

```
Application Layer     (VQE, QAOA, Shor's)
        |
Algorithm Layer       (Circuits, Operations)
        |
Gate Layer            (Native gates, Calibration)
        |
Pulse Layer           (Microwave pulses, Timing)
        |
Hardware Layer        (Qubits, Control electronics)
```

## 2. Circuit Model Programming

```python
from dataclasses import dataclass
from typing import List, Tuple

@dataclass
class Gate:
    """Quantum gate representation."""
    name: str
    qubits: Tuple[int, ...]
    params: Tuple[float, ...] = ()

class QuantumCircuit:
    """Simple circuit representation."""

    def __init__(self, num_qubits: int):
        self.num_qubits = num_qubits
        self.gates: List[Gate] = []

    def h(self, qubit: int):
        """Add Hadamard gate."""
        self.gates.append(Gate('H', (qubit,)))
        return self

    def cx(self, control: int, target: int):
        """Add CNOT gate."""
        self.gates.append(Gate('CX', (control, target)))
        return self

    def rx(self, qubit: int, theta: float):
        """Add RX rotation."""
        self.gates.append(Gate('RX', (qubit,), (theta,)))
        return self

    def measure_all(self):
        """Add measurements on all qubits."""
        for q in range(self.num_qubits):
            self.gates.append(Gate('MEASURE', (q,)))
        return self

    def to_qasm(self) -> str:
        """Convert to OpenQASM string."""
        lines = [f"OPENQASM 2.0;", f"qreg q[{self.num_qubits}];"]
        for gate in self.gates:
            if gate.name == 'H':
                lines.append(f"h q[{gate.qubits[0]}];")
            elif gate.name == 'CX':
                lines.append(f"cx q[{gate.qubits[0]}], q[{gate.qubits[1]}];")
            elif gate.name == 'RX':
                lines.append(f"rx({gate.params[0]}) q[{gate.qubits[0]}];")
        return '\n'.join(lines)

# Example
qc = QuantumCircuit(2)
qc.h(0).cx(0, 1).measure_all()
print(qc.to_qasm())
```

## 3. Pulse-Level Programming

```python
import numpy as np

@dataclass
class Pulse:
    """Pulse waveform."""
    channel: str
    samples: np.ndarray
    duration: float

def gaussian_pulse(duration: float, amplitude: float, sigma: float) -> np.ndarray:
    """Generate Gaussian pulse."""
    dt = 0.1  # Time resolution in ns
    t = np.arange(0, duration, dt)
    center = duration / 2
    return amplitude * np.exp(-((t - center)**2) / (2 * sigma**2))

def drag_pulse(duration: float, amplitude: float, sigma: float, beta: float) -> np.ndarray:
    """Generate DRAG pulse for reduced leakage."""
    dt = 0.1
    t = np.arange(0, duration, dt)
    center = duration / 2
    gauss = amplitude * np.exp(-((t - center)**2) / (2 * sigma**2))
    deriv = -gauss * (t - center) / sigma**2
    return gauss + 1j * beta * deriv

class PulseSchedule:
    """Pulse-level circuit representation."""

    def __init__(self):
        self.pulses: List[Tuple[float, Pulse]] = []  # (start_time, pulse)

    def add_pulse(self, start_time: float, pulse: Pulse):
        self.pulses.append((start_time, pulse))

    def total_duration(self) -> float:
        return max(t + p.duration for t, p in self.pulses)
```

## 4. High-Level Abstractions

```python
class VQEProgram:
    """High-level VQE abstraction."""

    def __init__(self, hamiltonian, ansatz, optimizer):
        self.hamiltonian = hamiltonian
        self.ansatz = ansatz
        self.optimizer = optimizer

    def run(self, backend, initial_params=None):
        """Execute VQE optimization."""
        def cost_fn(params):
            circuit = self.ansatz.build(params)
            result = backend.execute(circuit)
            return self.hamiltonian.expectation(result)

        return self.optimizer.minimize(cost_fn, initial_params)
```

## Summary

Programming paradigms offer different trade-offs:
1. **Circuit model**: Portable, high-level
2. **Pulse level**: Fine control, hardware-specific
3. **Application level**: Problem-focused, automatic compilation
