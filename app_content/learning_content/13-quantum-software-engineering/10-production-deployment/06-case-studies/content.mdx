def analyze_case_study(case: CaseStudy) -> Dict:
    """Analyze a case study and extract key insights."""

    analysis = {
        'summary': {
            'industry': case.industry,
            'deployment_date': case.deployment_date.isoformat(),
            'title': case.title
        },
        'key_success_factors': [],
        'risk_factors': [],
        'applicability_score': {}
    }

    # Identify success factors
    if 'classical_fallback' in case.technical_details:
        analysis['key_success_factors'].append(
            "Implemented classical fallback for reliability"
        )

    if 'error_mitigation' in case.technical_details:
        analysis['key_success_factors'].append(
            "Applied error mitigation techniques"
        )

    if len(case.technical_details.get('backends', [])) > 1:
        analysis['key_success_factors'].append(
            "Multi-backend strategy for redundancy"
        )

    # Identify risks from challenges
    for challenge in case.outcomes.get('challenges', []):
        if 'queue' in challenge.lower():
            analysis['risk_factors'].append("Queue time variability")
        if 'calibration' in challenge.lower():
            analysis['risk_factors'].append("Hardware calibration dependency")
        if 'fidelity' in challenge.lower():
            analysis['risk_factors'].append("Fidelity estimation accuracy")

    return analysis

POST-INCIDENT REVIEW
{'='*60}

Incident ID: {incident.incident_id}
Severity: {incident.severity}
Title: {incident.title}
Date: {incident.date.isoformat()}
Duration: {incident.duration_minutes} minutes

IMPACT
------
{incident.impact}

ROOT CAUSE
----------
{incident.root_cause}

TIMELINE
--------
"""
    for entry in incident.timeline:
        review += f"  {entry['time']}: {entry['event']}\n"

    review += f"""
RESOLUTION
----------
{incident.resolution}

LESSONS LEARNED
---------------
"""
    for i, lesson in enumerate(incident.lessons_learned, 1):
        review += f"  {i}. {lesson}\n"

    review += """
ACTION ITEMS
------------
"""
    for item in incident.action_items:
        review += f"  [ ] {item['action']}\n"
        review += f"      Owner: {item['owner']}, Due: {item['due']}\n"

    return review

@dataclass
class DeploymentPattern:
    """A deployment pattern or anti-pattern."""
    name: str
    category: str  # "pattern" or "anti-pattern"
    description: str
    context: str
    solution: str
    consequences: List[str]
    examples: List[str]

# Collection of patterns learned from case studies
deployment_patterns = [
    DeploymentPattern(
        name="Hybrid Decomposition",
        category="pattern",
        description="Decompose problems to use quantum only for suitable subproblems",
        context="When facing a complex problem that doesn't fit entirely on available quantum hardware",
        solution="Identify the NP-hard or quantum-amenable subproblem and solve only that part with quantum, using classical methods for the rest",
        consequences=[
            "+ Practical with current hardware limitations",
            "+ Easier to validate quantum contribution",
            "+ Graceful fallback if quantum fails",
            "- Requires careful problem decomposition",
            "- May not capture full quantum advantage"
        ],
        examples=[
            "QAOA for clustering in vehicle routing",
            "Quantum kernel in hybrid ML pipeline",
            "VQE for active space in molecular simulation"
        ]
    ),

    DeploymentPattern(
        name="Shadow Mode Validation",
        category="pattern",
        description="Run quantum system in parallel with classical for comparison before going live",
        context="When deploying quantum system to production for the first time",
        solution="Execute both quantum and classical solutions for every request, compare results, but only serve classical results to users",
        consequences=[
            "+ Builds confidence in quantum results",
            "+ Collects comparison data for optimization",
            "+ Zero risk to production users",
            "- Doubles computational cost during shadow period",
            "- Requires infrastructure for parallel execution"
        ],
        examples=[
            "Logistics company ran 3 months shadow mode",
            "Financial services validated for 6 months before live trading"
        ]
    ),

    DeploymentPattern(
        name="Ensemble Aggregation",
        category="pattern",
        description="Run multiple quantum executions and aggregate results",
        context="When quantum results have high variance due to hardware noise",
        solution="Execute the same circuit multiple times (possibly on different backends) and use statistical aggregation of results",
        consequences=[
            "+ Reduces impact of outlier results",
            "+ Enables confidence interval reporting",
            "+ Can leverage multiple backend diversity",
            "- Increases cost and latency",
            "- Requires aggregation strategy design"
        ],
        examples=[
            "Financial optimization uses median of 5 runs",
            "Chemistry calculations report mean with error bars"
        ]
    ),

    DeploymentPattern(
        name="Provider API Trust",
        category="anti-pattern",
        description="Trusting provider status APIs without independent verification",
        context="Assuming provider-reported calibration and health status is always accurate",
        solution="Implement independent health monitoring using test circuits and fidelity tracking",
        consequences=[
            "- Missed degradation when provider API lagged reality",
            "- Customer impact before detection",
            "+ Independent verification catches issues earlier",
            "+ Enables faster automated response"
        ],
        examples=[
            "Calibration drift incident where provider reported 'healthy'",
            "Queue time estimates that didn't match actual wait"
        ]
    ),

    DeploymentPattern(
        name="Quantum-Only Architecture",
        category="anti-pattern",
        description="Designing system with quantum as sole computation path",
        context="Over-optimism about quantum reliability and availability",
        solution="Always implement classical fallback for business continuity",
        consequences=[
            "- Complete outage when quantum unavailable",
            "- No comparison baseline for result validation",
            "- Single point of failure",
            "+ Classical fallback ensures continuity",
            "+ Enables A/B comparison"
        ],
        examples=[
            "System outage during provider maintenance window",
            "Unable to serve requests during queue congestion"
        ]
    ),

    DeploymentPattern(
        name="Ignoring Time-of-Day Effects",
        category="anti-pattern",
        description="Not accounting for hardware performance variation throughout the day",
        context="Quantum hardware quality varies based on calibration schedule and usage patterns",
        solution="Track performance by time of day and schedule critical workloads during optimal periods",
        consequences=[
            "- Lower quality results during degraded periods",
            "- Inconsistent user experience",
            "+ Scheduling optimization improves average quality",
            "+ Alerting can warn users of degraded periods"
        ],
        examples=[
            "Afternoon results consistently lower quality than morning",
            "Weekend performance better due to lower queue utilization"
        ]
    )
]

def generate_pattern_guide() -> str:
    """Generate a pattern reference guide."""

    guide = """
QUANTUM PRODUCTION DEPLOYMENT PATTERNS
{'='*60}

RECOMMENDED PATTERNS
--------------------
"""
    for pattern in deployment_patterns:
        if pattern.category == "pattern":
            guide += f"""
{pattern.name}
{'─' * len(pattern.name)}
{pattern.description}

When to use: {pattern.context}

Solution: {pattern.solution}

Trade-offs:
"""
            for c in pattern.consequences:
                guide += f"  {c}\n"

            guide += "\nExamples:\n"
            for e in pattern.examples:
                guide += f"  • {e}\n"

    guide += """
ANTI-PATTERNS TO AVOID
----------------------
"""
    for pattern in deployment_patterns:
        if pattern.category == "anti-pattern":
            guide += f"""
{pattern.name}
{'─' * len(pattern.name)}
Problem: {pattern.description}

Context: {pattern.context}

Better approach: {pattern.solution}
"""

    return guide

## 7. Decision Framework

```python
class DeploymentDecisionFramework:
    """
    Framework for making quantum production deployment decisions.

    Based on lessons learned from case studies.
    """

    def __init__(self):
        self.criteria = self._setup_criteria()

    def _setup_criteria(self) -> Dict:
        """Set up evaluation criteria."""
        return {
            'problem_fit': {
                'weight': 0.25,
                'questions': [
                    "Is the problem NP-hard or exponentially complex?",
                    "Does the problem have natural quantum representation?",
                    "Can the problem fit on available qubit counts?",
                    "Is approximate solution acceptable?"
                ]
            },
            'business_value': {
                'weight': 0.20,
                'questions': [
                    "What is the potential ROI from quantum improvement?",
                    "Is there competitive advantage from early adoption?",
                    "Can we quantify the value of solution quality improvement?",
                    "Is the business willing to accept experimental technology?"
                ]
            },
            'technical_readiness': {
                'weight': 0.25,
                'questions': [
                    "Do we have quantum engineering expertise?",
                    "Is classical fallback implemented?",
                    "Are error mitigation strategies validated?",
                    "Is monitoring and observability in place?"
                ]
            },
            'operational_maturity': {
                'weight': 0.20,
                'questions': [
                    "Are SLOs defined and achievable?",
                    "Is disaster recovery tested?",
                    "Are runbooks documented?",
                    "Is on-call rotation staffed?"
                ]
            },
            'risk_tolerance': {
                'weight': 0.10,
                'questions': [
                    "Can we tolerate occasional degraded results?",
                    "Is the application user-facing or internal?",
                    "What is the cost of quantum result being wrong?",
                    "Do we have time for extended validation?"
                ]
            }
        }

    def evaluate_readiness(self, scores: Dict[str, float]) -> Dict:
        """
        Evaluate deployment readiness based on scores.

        Args:
            scores: Dict with scores 0-1 for each criterion category

        Returns:
            Evaluation result with recommendation
        """
        weighted_score = 0
        category_results = {}

        for category, config in self.criteria.items():
            score = scores.get(category, 0)
            weighted = score * config['weight']
            weighted_score += weighted

            category_results[category] = {
                'score': score,
                'weight': config['weight'],
                'weighted': weighted,
                'status': 'pass' if score >= 0.7 else 'needs_work'
            }

        # Generate recommendation
        if weighted_score >= 0.8:
            recommendation = "READY FOR PRODUCTION"
            guidance = "Proceed with production deployment following standard procedures"
        elif weighted_score >= 0.6:
            recommendation = "READY FOR LIMITED PRODUCTION"
            guidance = "Deploy to limited scope (specific users/use cases) with enhanced monitoring"
        elif weighted_score >= 0.4:
            recommendation = "NEEDS MORE PREPARATION"
            guidance = "Address gaps before production; consider extended pilot"
        else:
            recommendation = "NOT READY"
            guidance = "Significant work needed; focus on lowest-scoring areas"

        # Identify blocking issues
        blocking = [
            cat for cat, result in category_results.items()
            if result['score'] < 0.5
        ]

        return {
            'overall_score': weighted_score,
            'recommendation': recommendation,
            'guidance': guidance,
            'category_results': category_results,
            'blocking_issues': blocking,
            'next_steps': self._generate_next_steps(category_results)
        }

    def _generate_next_steps(self, results: Dict) -> List[str]:
        """Generate prioritized next steps."""
        steps = []

        # Sort by lowest scores first
        sorted_cats = sorted(
            results.items(),
            key=lambda x: x[1]['score']
        )

        for category, result in sorted_cats:
            if result['status'] == 'needs_work':
                if category == 'problem_fit':
                    steps.append("Validate problem-quantum fit with proof of concept")
                elif category == 'business_value':
                    steps.append("Develop business case with quantified value")
                elif category == 'technical_readiness':
                    steps.append("Complete technical readiness checklist")
                elif category == 'operational_maturity':
                    steps.append("Establish operational procedures and test DR")
                elif category == 'risk_tolerance':
                    steps.append("Define risk acceptance criteria with stakeholders")

        return steps[:3]  # Top 3 priorities
```

## Summary

Key takeaways from production deployment case studies:

1. **Hybrid Approaches Win**: All successful deployments used quantum for specific subproblems, not entire applications
2. **Shadow Mode is Essential**: Parallel classical execution builds confidence and provides validation
3. **Expect Variance**: Ensemble methods and error bars are necessary for reliable results
4. **Independent Monitoring**: Never trust provider APIs alone; implement your own health checks
5. **Domain Expertise Critical**: Successful deployments involved close collaboration with domain experts

The path to production quantum applications requires patience, rigorous validation, and a willingness to learn from both successes and failures.
