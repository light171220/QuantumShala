```

Key Metrics:
┌─────────────────────────────────────────────────────────────────┐
│ RTO (Recovery Time Objective)                                   │

│ Quantum challenge: Backend availability is external             │
├─────────────────────────────────────────────────────────────────┤
│ RPO (Recovery Point Objective)                                  │

1. QPU provider outage (external dependency)
2. Classical infrastructure failure (internal)
3. Data corruption or loss
```

```python
    tier: RecoveryTier

class DisasterRecoveryPlanner:
    """
    Plans and manages disaster recovery for quantum applications.
    """

    TIER_DEFAULTS = {
        RecoveryTier.TIER_1: {'rto': 60, 'rpo': 5},
        RecoveryTier.TIER_2: {'rto': 240, 'rpo': 60},
        RecoveryTier.TIER_3: {'rto': 1440, 'rpo': 240},

    def __init__(self, application_name: str):
        self.application_name = application_name
        self.services: Dict[str, RecoveryObjectives] = {}
        self._setup_default_services()

    def _setup_default_services(self):
        """Configure default quantum service recovery objectives."""

        # Core quantum execution - highest priority
        self.services['quantum_executor'] = RecoveryObjectives(

            tier=RecoveryTier.TIER_1,
            rto_minutes=60,
            rpo_minutes=5,
            dependencies=['backend_router', 'result_store'],

        # Result storage - critical for data integrity
        self.services['result_store'] = RecoveryObjectives(

            tier=RecoveryTier.TIER_1,
            rto_minutes=30,
            rpo_minutes=0,  # No data loss acceptable
            dependencies=['database', 'cache'],
            recovery_procedure='result_store_recovery'
        )

        # Backend router - needed for job distribution
        self.services['backend_router'] = RecoveryObjectives(
            service_name='backend_router',
            tier=RecoveryTier.TIER_2,
            rto_minutes=120,
            rpo_minutes=60,
            dependencies=['backend_registry'],
            recovery_procedure='router_recovery'
        )

        # Job queue - can rebuild from pending jobs
        self.services['job_queue'] = RecoveryObjectives(
            service_name='job_queue',
            tier=RecoveryTier.TIER_2,
            rto_minutes=180,
            rpo_minutes=30,
            dependencies=['database'],
            recovery_procedure='queue_recovery'
        )

        # Circuit compiler - stateless, quick recovery
        self.services['circuit_compiler'] = RecoveryObjectives(
            service_name='circuit_compiler',
            tier=RecoveryTier.TIER_3,
            rto_minutes=240,
            rpo_minutes=0,  # Stateless
            dependencies=[],
            recovery_procedure='compiler_recovery'
        )

    def calculate_overall_rto(self) -> Dict:
        """Calculate overall RTO considering dependencies."""
        # Build dependency graph and find critical path
        visited = set()
        max_rto = 0

        def get_service_rto(service_name: str, depth: int = 0) -> int:
            if service_name in visited or depth > 10:
                return 0
            visited.add(service_name)

            if service_name not in self.services:
                return 0

            service = self.services[service_name]
            dep_rto = max(
                (get_service_rto(dep, depth + 1) for dep in service.dependencies),
                default=0
            )

            return service.rto_minutes + dep_rto

        service_rtos = {}
        for name in self.services:
            visited.clear()
            service_rtos[name] = get_service_rto(name)

        max_rto = max(service_rtos.values()) if service_rtos else 0

        return {
            'overall_rto_minutes': max_rto,
            'overall_rto_hours': max_rto / 60,
            'service_rtos': service_rtos,
            'critical_path': max(service_rtos.items(), key=lambda x: x[1])[0] if service_rtos else None
        }

    def generate_dr_plan(self) -> Dict:
        """Generate comprehensive disaster recovery plan."""
        rto_analysis = self.calculate_overall_rto()

        plan = {
            'application': self.application_name,
            'generated_at': datetime.now().isoformat(),
            'summary': {
                'total_services': len(self.services),
                'tier_1_services': sum(1 for s in self.services.values() if s.tier == RecoveryTier.TIER_1),
                'overall_rto': rto_analysis['overall_rto_hours'],
                'critical_path': rto_analysis['critical_path']
            },
            'services': {},
            'recovery_order': self._calculate_recovery_order(),
            'contacts': {
                'primary_oncall': 'oncall@example.com',
                'escalation': 'escalation@example.com'
            }
        }

        for name, service in self.services.items():
            plan['services'][name] = {
                'tier': service.tier.value,
                'rto_minutes': service.rto_minutes,
                'rpo_minutes': service.rpo_minutes,
                'dependencies': service.dependencies,
                'recovery_procedure': service.recovery_procedure,
                'last_tested': service.tested_at.isoformat() if service.tested_at else None
            }

        return plan

    def _calculate_recovery_order(self) -> List[str]:
        """Calculate optimal recovery order based on dependencies."""
        # Topological sort of services based on dependencies
        order = []
        visited = set()
        temp_visited = set()

        def visit(name: str):
            if name in temp_visited:
                return  # Cycle detected, skip
            if name in visited:
                return

            temp_visited.add(name)

            if name in self.services:
                for dep in self.services[name].dependencies:
                    visit(dep)

            temp_visited.remove(name)
            visited.add(name)
            order.append(name)

        for name in self.services:
            visit(name)

        return order
```

## 3. Backup and Restore Procedures

```python
from abc import ABC, abstractmethod
import gzip
from io import BytesIO

class BackupType(Enum):
    """Types of backups."""
    FULL = "full"
    INCREMENTAL = "incremental"
    DIFFERENTIAL = "differential"
    SNAPSHOT = "snapshot"

@dataclass
class BackupMetadata:
    """Metadata for a backup."""
    backup_id: str
    backup_type: BackupType
    created_at: datetime
    size_bytes: int
    checksum: str
    source_service: str
    parent_backup_id: Optional[str] = None
    retention_days: int = 30

class BackupStrategy(ABC):
    """Abstract backup strategy."""

    @abstractmethod
    def backup(self, data: Dict) -> BackupMetadata:
        """Create a backup."""
        pass

    @abstractmethod
    def restore(self, backup_id: str) -> Dict:
        """Restore from backup."""
        pass

class QuantumWorkloadBackup:
    """
    Backup system for quantum workloads.

    Backs up:
    - Circuit definitions
    - Job configurations
    - Results history
    - Calibration data
    - User configurations
    """

    def __init__(self, storage_backend: str = "local"):
        self.storage_backend = storage_backend
        self.backups: Dict[str, BackupMetadata] = {}
        self.backup_data: Dict[str, bytes] = {}  # In-memory for demo

    def backup_circuits(
        self,
        circuits: List[Dict],
        backup_type: BackupType = BackupType.FULL
    ) -> BackupMetadata:
        """Backup circuit definitions."""
        backup_id = f"circuits_{datetime.now().strftime('%Y%m%d_%H%M%S')}"

        # Serialize and compress
        data_json = json.dumps(circuits, sort_keys=True)
        compressed = gzip.compress(data_json.encode())

        # Calculate checksum
        checksum = hashlib.sha256(compressed).hexdigest()

        metadata = BackupMetadata(
            backup_id=backup_id,
            backup_type=backup_type,
            created_at=datetime.now(),
            size_bytes=len(compressed),
            checksum=checksum,
            source_service="circuit_store"
        )

        self.backups[backup_id] = metadata
        self.backup_data[backup_id] = compressed

        return metadata

    def backup_job_results(
        self,
        results: List[Dict],
        last_backup_id: Optional[str] = None
    ) -> BackupMetadata:
        """Backup job results with incremental support."""
        if last_backup_id and last_backup_id in self.backups:
            backup_type = BackupType.INCREMENTAL
        else:
            backup_type = BackupType.FULL

        backup_id = f"results_{datetime.now().strftime('%Y%m%d_%H%M%S')}"

        # Include result metadata but optionally exclude large measurement data
        backup_results = []
        for result in results:
            backup_result = {
                'job_id': result.get('job_id'),
                'circuit_id': result.get('circuit_id'),
                'backend': result.get('backend'),
                'status': result.get('status'),
                'submitted_at': result.get('submitted_at'),
                'completed_at': result.get('completed_at'),
                'shots': result.get('shots'),
                'counts_summary': self._summarize_counts(result.get('counts', {}))
            }
            backup_results.append(backup_result)

        data_json = json.dumps(backup_results, sort_keys=True, default=str)
        compressed = gzip.compress(data_json.encode())
        checksum = hashlib.sha256(compressed).hexdigest()

        metadata = BackupMetadata(
            backup_id=backup_id,
            backup_type=backup_type,
            created_at=datetime.now(),
            size_bytes=len(compressed),
            checksum=checksum,
            source_service="result_store",
            parent_backup_id=last_backup_id
        )

        self.backups[backup_id] = metadata
        self.backup_data[backup_id] = compressed

        return metadata

    def _summarize_counts(self, counts: Dict) -> Dict:
        """Summarize measurement counts for backup."""
        if not counts:
            return {}

        total = sum(counts.values())
        return {
            'total_shots': total,
            'unique_outcomes': len(counts),
            'top_outcomes': dict(sorted(counts.items(), key=lambda x: -x[1])[:10])
        }

    def restore_backup(self, backup_id: str) -> Dict:
        """Restore data from backup."""
        if backup_id not in self.backups:
            raise ValueError(f"Backup {backup_id} not found")

        metadata = self.backups[backup_id]
        compressed = self.backup_data[backup_id]

        # Verify checksum
        actual_checksum = hashlib.sha256(compressed).hexdigest()
        if actual_checksum != metadata.checksum:
            raise ValueError("Backup checksum verification failed - data corrupted")

        # Decompress and deserialize
        decompressed = gzip.decompress(compressed)
        data = json.loads(decompressed.decode())

        return {
            'backup_id': backup_id,
            'restored_at': datetime.now().isoformat(),
            'backup_type': metadata.backup_type.value,
            'original_date': metadata.created_at.isoformat(),
            'data': data
        }

    def list_backups(self, service: str = None) -> List[Dict]:
        """List available backups."""
        backups = []
        for backup_id, metadata in self.backups.items():
            if service and metadata.source_service != service:
                continue

            backups.append({
                'backup_id': backup_id,
                'type': metadata.backup_type.value,
                'created_at': metadata.created_at.isoformat(),
                'size_bytes': metadata.size_bytes,
                'service': metadata.source_service
            })

        return sorted(backups, key=lambda x: x['created_at'], reverse=True)

    def cleanup_old_backups(self, retention_days: int = 30) -> int:
        """Remove backups older than retention period."""
        cutoff = datetime.now() - timedelta(days=retention_days)
        removed = 0

        for backup_id in list(self.backups.keys()):
            if self.backups[backup_id].created_at < cutoff:
                del self.backups[backup_id]
                del self.backup_data[backup_id]
                removed += 1

        return removed
```

```python
class FailoverState(Enum):
    """Failover system states."""
    NORMAL = "normal"          # Primary active
    FAILING_OVER = "failing_over"  # Transition in progress
    FAILED_OVER = "failed_over"    # Secondary active
    RECOVERING = "recovering"      # Returning to primary

@dataclass
class FailoverTarget:
    """A failover target configuration."""
    name: str
    endpoint: str
    region: str
    priority: int  # Lower = higher priority
    health_check_url: str
    is_active: bool = False
    last_health_check: Optional[datetime] = None
    health_status: str = "unknown"

class QuantumFailoverManager:
    """
    Manages failover between quantum backend providers and regions.

    Handles:
    - Automatic failover on primary failure
    - Manual failover for maintenance
    - Failback to primary when recovered
    - Health monitoring of all targets
    """

    def __init__(self):
        self.state = FailoverState.NORMAL
        self.targets: Dict[str, FailoverTarget] = {}
        self.active_target: Optional[str] = None
        self.failover_history: List[Dict] = []

    def register_target(self, target: FailoverTarget):
        """Register a failover target."""
        self.targets[target.name] = target

        # Set first target as active if none set
        if self.active_target is None:
            target.is_active = True
            self.active_target = target.name

    def check_health(self, target_name: str) -> bool:
        """Check health of a failover target."""
        if target_name not in self.targets:
            return False

        target = self.targets[target_name]

        # Simulate health check (would be actual HTTP check in production)
        is_healthy = True  # Placeholder

        target.last_health_check = datetime.now()
        target.health_status = "healthy" if is_healthy else "unhealthy"

        return is_healthy

    def check_all_targets(self) -> Dict[str, bool]:
        """Check health of all targets."""
        results = {}
        for name in self.targets:
            results[name] = self.check_health(name)
        return results

    def initiate_failover(self, reason: str, manual: bool = False) -> Dict:
        """
        Initiate failover to next available target.
        """
        if self.state == FailoverState.FAILING_OVER:
            return {'success': False, 'error': 'Failover already in progress'}

        current_target = self.active_target
        self.state = FailoverState.FAILING_OVER

        # Find next healthy target
        sorted_targets = sorted(
            self.targets.values(),
            key=lambda t: t.priority
        )

        new_target = None
        for target in sorted_targets:
            if target.name != current_target:
                if self.check_health(target.name):
                    new_target = target
                    break

        if not new_target:
            self.state = FailoverState.NORMAL
            return {
                'success': False,
                'error': 'No healthy failover targets available'
            }

        # Execute failover
        if self.active_target and self.active_target in self.targets:
            self.targets[self.active_target].is_active = False

        new_target.is_active = True
        self.active_target = new_target.name
        self.state = FailoverState.FAILED_OVER

        # Record failover event
        event = {
            'timestamp': datetime.now().isoformat(),
            'from_target': current_target,
            'to_target': new_target.name,
            'reason': reason,
            'manual': manual
        }
        self.failover_history.append(event)

        return {
            'success': True,
            'previous_target': current_target,
            'new_target': new_target.name,
            'event': event
        }

    def initiate_failback(self, target_name: str) -> Dict:
        """
        Failback to a specific target (typically primary).
        """
        if target_name not in self.targets:
            return {'success': False, 'error': 'Target not found'}

        if not self.check_health(target_name):
            return {'success': False, 'error': 'Target is not healthy'}

        self.state = FailoverState.RECOVERING

        # Execute failback
        previous = self.active_target
        if previous and previous in self.targets:
            self.targets[previous].is_active = False

        self.targets[target_name].is_active = True
        self.active_target = target_name
        self.state = FailoverState.NORMAL

        event = {
            'timestamp': datetime.now().isoformat(),
            'from_target': previous,
            'to_target': target_name,
            'reason': 'failback',
            'manual': True
        }
        self.failover_history.append(event)

        return {
            'success': True,
            'previous_target': previous,
            'new_target': target_name
        }

    def get_status(self) -> Dict:
        """Get current failover status."""
        return {
            'state': self.state.value,
            'active_target': self.active_target,
            'targets': {
                name: {
                    'region': t.region,
                    'priority': t.priority,
                    'is_active': t.is_active,
                    'health_status': t.health_status,
                    'last_check': t.last_health_check.isoformat() if t.last_health_check else None
                }
                for name, t in self.targets.items()
            },
            'recent_failovers': self.failover_history[-5:]
        }
```

```python
        restore_time = datetime.now()

        # Validate restored data
        restored_data = restored['data'][0] if restored['data'] else None

        validations = []

        if restored_data == test_data:
            validations.append({'check': 'data_integrity', 'passed': True})
        else:
            validations.append({'check': 'data_integrity', 'passed': False})
            findings.append("CRITICAL: Restored data does not match original")

        # Calculate times
        backup_duration = (backup_time - test_start).total_seconds()
        restore_duration = (restore_time - restore_start).total_seconds()

        result = {
            'test_type': 'backup_restore',
            'executed_at': test_start.isoformat(),
            'backup_duration_seconds': backup_duration,
            'restore_duration_seconds': restore_duration,
            'backup_size_bytes': metadata.size_bytes,
            'success': all(v['passed'] for v in validations),
            'validations': validations,
            'findings': findings
        }

        self.test_history.append(result)
        return result

    def generate_test_report(self) -> Dict:
        """Generate comprehensive DR test report."""
        if not self.test_history:
            return {'message': 'No tests executed'}

        total_tests = len(self.test_history)
        passed_tests = sum(1 for t in self.test_history if t.get('success', False))

        return {
            'report_generated': datetime.now().isoformat(),
            'summary': {
                'total_tests': total_tests,
                'passed': passed_tests,
                'failed': total_tests - passed_tests,
                'pass_rate': passed_tests / total_tests if total_tests > 0 else 0
            },
            'tests': self.test_history,
            'recommendations': self._generate_recommendations()
        }

    def _generate_recommendations(self) -> List[str]:
        """Generate recommendations based on test results."""
        recommendations = []

        # Check for failed tests
        failed_tests = [t for t in self.test_history if not t.get('success', False)]
        if failed_tests:
            recommendations.append(
                f"Address {len(failed_tests)} failed DR tests before next production deployment"
            )

        # Check test coverage
        if not any(t['test_type'] == 'failover' for t in self.test_history):
            recommendations.append("Schedule failover tests to validate backend redundancy")

        if not any(t['test_type'] == 'backup_restore' for t in self.test_history):
            recommendations.append("Execute backup/restore tests to validate data recovery")

        return recommendations
```

## Summary

Disaster recovery for quantum systems requires:

1. **Recovery Objectives**: Define RTO/RPO for each service considering dependencies
2. **Backup Procedures**: Regular backups of circuits, results, and configurations
3. **Failover Mechanisms**: Automatic and manual failover across providers/regions
4. **Recovery Testing**: Regular testing to validate DR procedures
5. **Business Continuity**: Comprehensive plans for various disaster scenarios

The unique challenge of quantum DR is balancing the need for specialized QPU access with the requirement for service continuity when that access is disrupted.
