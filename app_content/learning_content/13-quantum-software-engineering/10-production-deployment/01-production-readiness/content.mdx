## 1. What Makes Quantum Production Different?

Quantum applications face unique production challenges that classical software does not encounter:

```
QUANTUM PRODUCTION CHALLENGES:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Classical Software         Quantum Software
───────────────────────    ───────────────────────
Deterministic output       Probabilistic output
Stable hardware            Drifting calibration
Local execution            Remote QPU access
Unlimited retries          Shot-limited execution
Predictable costs          Variable queue times

Key Metrics:
- Fidelity stability: F(t) = F₀ · e^(-t/T₁)
- Result variance: σ² = p(1-p)/n_shots
- Queue latency: L_queue ~ O(demand/capacity)
```

from dataclasses import dataclass, field
from typing import List, Dict, Optional, Callable
from enum import Enum
from datetime import datetime
import hashlib

class ReadinessLevel(Enum):
    """Production readiness maturity levels."""
    NOT_READY = 0
    DEVELOPMENT = 1
    STAGING = 2
    PRE_PRODUCTION = 3
    PRODUCTION = 4

@dataclass
class ReadinessCriterion:
    """Single readiness criterion."""
    name: str
    category: str
    description: str
    check_fn: Callable[[], bool]
    required_level: ReadinessLevel
    weight: float = 1.0

    def evaluate(self) -> bool:
        """Evaluate this criterion."""
        try:
            return self.check_fn()
        except Exception as e:
            print(f"Criterion {self.name} failed: {e}")
            return False

@dataclass
class QuantumProductionReadiness:
    """
    Production readiness assessment for quantum applications.

    Based on the Google SRE production readiness review framework,
    adapted for quantum computing workloads.
    """

    application_name: str
    version: str
    criteria: List[ReadinessCriterion] = field(default_factory=list)
    assessment_date: datetime = field(default_factory=datetime.now)

    def add_criterion(self, criterion: ReadinessCriterion):
        """Add a readiness criterion."""
        self.criteria.append(criterion)

    def evaluate_all(self) -> Dict[str, any]:
        """Evaluate all readiness criteria."""
        results = {
            'application': self.application_name,
            'version': self.version,
            'timestamp': self.assessment_date.isoformat(),
            'categories': {},
            'overall_score': 0.0,
            'ready_for_production': False
        }

        total_weight = 0
        weighted_score = 0

        for criterion in self.criteria:
            category = criterion.category
            if category not in results['categories']:
                results['categories'][category] = {
                    'criteria': [],
                    'passed': 0,
                    'failed': 0
                }

            passed = criterion.evaluate()
            results['categories'][category]['criteria'].append({
                'name': criterion.name,
                'passed': passed,
                'required_level': criterion.required_level.name
            })

            if passed:
                results['categories'][category]['passed'] += 1
                weighted_score += criterion.weight
            else:
                results['categories'][category]['failed'] += 1

            total_weight += criterion.weight

        results['overall_score'] = weighted_score / total_weight if total_weight > 0 else 0
        results['ready_for_production'] = results['overall_score'] >= 0.9

        return results

    def generate_report(self) -> str:
        """Generate human-readable readiness report."""
        results = self.evaluate_all()

        report = f"""
QUANTUM APPLICATION PRODUCTION READINESS REPORT
{'='*60}
Application: {results['application']}
Version: {results['version']}
Assessment Date: {results['timestamp']}
Overall Score: {results['overall_score']*100:.1f}%
Production Ready: {'YES' if results['ready_for_production'] else 'NO'}

CATEGORY BREAKDOWN:
"""
        for category, data in results['categories'].items():
            report += f"\n{category}:\n"
            report += f"  Passed: {data['passed']}, Failed: {data['failed']}\n"
            for crit in data['criteria']:
                status = "PASS" if crit['passed'] else "FAIL"
                report += f"    [{status}] {crit['name']}\n"

        return report

from typing import Callable
import time

@dataclass
class MetricDefinition:
    """Definition of a quantum metric to monitor."""
    name: str
    unit: str
    description: str
    collection_fn: Callable[[], float]
    warning_threshold: Optional[float] = None
    critical_threshold: Optional[float] = None

@dataclass
class AlertRule:
    """Alerting rule configuration."""
    name: str
    metric: str
    condition: str  # "above", "below", "equals"
    threshold: float
    duration_seconds: int
    severity: str  # "warning", "critical", "page"
    notification_channels: List[str]

class QuantumProductionMonitor:
    """
    Production monitoring system for quantum applications.

    Tracks quantum-specific metrics:
    - Circuit fidelity
    - Result distribution stability
    - Queue times
    - Error rates
    """

    def __init__(self, application_name: str):
        self.application_name = application_name
        self.metrics: Dict[str, MetricDefinition] = {}
        self.alerts: Dict[str, AlertRule] = {}
        self.metric_history: Dict[str, List[tuple]] = {}

        self._setup_default_metrics()
        self._setup_default_alerts()

    def _setup_default_metrics(self):
        """Configure default quantum metrics."""

        self.metrics['circuit_fidelity'] = MetricDefinition(
            name="circuit_fidelity",
            unit="ratio",
            description="Estimated circuit fidelity based on gate errors",

            critical_threshold=0.80
        )

        self.metrics['result_entropy'] = MetricDefinition(
            name="result_entropy",
            unit="bits",
            description="Entropy of measurement results (stability indicator)",
            collection_fn=lambda: 0.8,
            warning_threshold=1.5,  # High entropy indicates instability
            critical_threshold=2.0
        )

        self.metrics['queue_wait_time'] = MetricDefinition(
            name="queue_wait_time",
            unit="seconds",
            description="Time spent waiting in QPU queue",
            collection_fn=lambda: 30.0,
            warning_threshold=300,
            critical_threshold=600
        )

        self.metrics['execution_success_rate'] = MetricDefinition(
            name="execution_success_rate",
            unit="ratio",
            description="Ratio of successful circuit executions",
            collection_fn=lambda: 0.99,
            warning_threshold=0.95,
            critical_threshold=0.90
        )

        self.metrics['error_mitigation_overhead'] = MetricDefinition(
            name="error_mitigation_overhead",
            unit="ratio",
            description="Overhead factor from error mitigation",
            collection_fn=lambda: 1.5,
            warning_threshold=3.0,
            critical_threshold=5.0
        )

    def _setup_default_alerts(self):
        """Configure default alerting rules."""

        self.alerts['low_fidelity'] = AlertRule(
            name="Low Circuit Fidelity",
            metric="circuit_fidelity",
            condition="below",
            threshold=0.85,
            duration_seconds=300,
            severity="critical",
            notification_channels=["pagerduty", "slack"]
        )

        self.alerts['long_queue'] = AlertRule(
            name="Extended Queue Wait",
            metric="queue_wait_time",
            condition="above",
            threshold=600,
            duration_seconds=60,
            severity="warning",
            notification_channels=["slack"]
        )

        self.alerts['execution_failures'] = AlertRule(
            name="High Execution Failure Rate",
            metric="execution_success_rate",
            condition="below",
            threshold=0.90,
            duration_seconds=180,
            severity="page",
            notification_channels=["pagerduty", "slack", "email"]
        )

    def collect_metrics(self) -> Dict[str, float]:
        """Collect all configured metrics."""
        timestamp = time.time()
        values = {}

        for name, metric in self.metrics.items():
            try:
                value = metric.collection_fn()
                values[name] = value

                # Store in history
                if name not in self.metric_history:
                    self.metric_history[name] = []
                self.metric_history[name].append((timestamp, value))

                # Keep only last hour of data
                cutoff = timestamp - 3600
                self.metric_history[name] = [
                    (t, v) for t, v in self.metric_history[name] if t > cutoff
                ]
            except Exception as e:
                print(f"Error collecting metric {name}: {e}")
                values[name] = None

        return values

    def check_alerts(self, current_values: Dict[str, float]) -> List[Dict]:
        """Check all alert rules against current values."""
        triggered = []

        for alert_name, rule in self.alerts.items():
            if rule.metric not in current_values:
                continue

            value = current_values[rule.metric]
            if value is None:
                continue

            triggered_flag = False
            if rule.condition == "above" and value > rule.threshold:
                triggered_flag = True
            elif rule.condition == "below" and value < rule.threshold:
                triggered_flag = True
            elif rule.condition == "equals" and value == rule.threshold:
                triggered_flag = True

            if triggered_flag:
                triggered.append({
                    'alert_name': alert_name,
                    'metric': rule.metric,
                    'current_value': value,
                    'threshold': rule.threshold,
                    'severity': rule.severity,
                    'channels': rule.notification_channels
                })

        return triggered

    def generate_dashboard_config(self) -> Dict:
        """Generate dashboard configuration for visualization."""
        return {
            'dashboard_name': f"{self.application_name} Quantum Monitoring",
            'refresh_interval': 30,
            'panels': [

                    'title': 'Circuit Fidelity',
                    'type': 'gauge',
                    'metric': 'circuit_fidelity',
                    'thresholds': [0.8, 0.9, 1.0],

## 5. Operational Runbooks

```python
@dataclass
class RunbookStep:
    """Single step in a runbook procedure."""
    step_number: int
    action: str
    expected_outcome: str
    commands: List[str] = field(default_factory=list)
    verification: Optional[str] = None
    rollback_action: Optional[str] = None

@dataclass
class Runbook:
    """Operational runbook for quantum systems."""

    id: str
    title: str
    description: str
    trigger_conditions: List[str]
    severity: str
    estimated_time_minutes: int
    required_permissions: List[str]
    steps: List[RunbookStep] = field(default_factory=list)

    def to_markdown(self) -> str:
        """Generate markdown documentation."""
        md = f"""# {self.title}

**ID:** {self.id}
**Severity:** {self.severity}
**Estimated Time:** {self.estimated_time_minutes} minutes

## Description
{self.description}

## Trigger Conditions
"""
        for condition in self.trigger_conditions:
            md += f"- {condition}\n"

        md += "\n## Required Permissions\n"
        for perm in self.required_permissions:
            md += f"- {perm}\n"

        md += "\n## Procedure\n\n"
        for step in self.steps:
            md += f"### Step {step.step_number}: {step.action}\n\n"
            md += f"**Expected Outcome:** {step.expected_outcome}\n\n"

            if step.commands:
                md += "**Commands:**\n```bash\n"
                for cmd in step.commands:
                    md += f"{cmd}\n"
                md += "```\n\n"

            if step.verification:
                md += f"**Verification:** {step.verification}\n\n"

            if step.rollback_action:
                md += f"**Rollback:** {step.rollback_action}\n\n"

        return md

@dataclass
class GoLiveGate:
    """Go-live decision gate."""
    name: str
    category: str
    criteria: str
    threshold: str

class GoLiveDecisionMatrix:
    """
    Decision matrix for quantum application go-live.
    """

    def __init__(self, application_name: str):

        self.gates: List[GoLiveGate] = []
        self.decision_date = datetime.now()
        self._initialize_gates()

            # Security Gates
            GoLiveGate(

                status="pass"
            ),
            GoLiveGate(
                name="Secrets Management",

                threshold="No hardcoded secrets",
                current_value="Verified",
                status="pass"
            ),

            # Business Gates
            GoLiveGate(
                name="Stakeholder Approval",
                category="Business",
                criteria="Business stakeholder sign-off",
                threshold="Approved",
                current_value="Pending",
                status="fail"
            ),
            GoLiveGate(
                name="Documentation",
                category="Business",
                criteria="User documentation complete",
                threshold="Published",
                current_value="Published",
                status="pass"
            ),
        ]

    def evaluate_readiness(self) -> Dict:
        """Evaluate go-live readiness."""
        total = len(self.gates)
        passed = sum(1 for g in self.gates if g.status == "pass")
        failed = sum(1 for g in self.gates if g.status == "fail")
        waived = sum(1 for g in self.gates if g.status == "waived")

        by_category = {}
        for gate in self.gates:
            if gate.category not in by_category:
                by_category[gate.category] = {'pass': 0, 'fail': 0, 'waived': 0}
            by_category[gate.category][gate.status] += 1

        can_proceed = failed == 0 or all(
            g.waiver_reason for g in self.gates if g.status == "fail"
        )

        return {
            'total_gates': total,
            'passed': passed,
            'failed': failed,
            'waived': waived,
            'by_category': by_category,
            'can_proceed': can_proceed,
            'recommendation': "APPROVED" if can_proceed else "BLOCKED"
        }

    def generate_decision_document(self) -> str:
        """Generate go-live decision document."""
        result = self.evaluate_readiness()

        doc = f"""
GO-LIVE DECISION DOCUMENT
{'='*60}
Application: {self.application_name}
Decision Date: {self.decision_date.isoformat()}

SUMMARY:
  Total Gates: {result['total_gates']}
  Passed: {result['passed']}
  Failed: {result['failed']}
  Waived: {result['waived']}

  RECOMMENDATION: {result['recommendation']}

GATE DETAILS:
"""
        for gate in self.gates:
            status_icon = {"pass": "[PASS]", "fail": "[FAIL]", "waived": "[WAIVED]"}[gate.status]
            doc += f"\n{status_icon} {gate.name} ({gate.category})\n"
            doc += f"    Criteria: {gate.criteria}\n"
            doc += f"    Threshold: {gate.threshold}\n"
            doc += f"    Current: {gate.current_value}\n"
            if gate.waiver_reason:
                doc += f"    Waiver: {gate.waiver_reason}\n"

        doc += """
APPROVALS:
  Technical Lead:    _________________ Date: _______
  Operations Lead:   _________________ Date: _______
  Security Lead:     _________________ Date: _______
  Product Owner:     _________________ Date: _______
  Executive Sponsor: _________________ Date: _______
"""
        return doc

## Summary

Production readiness for quantum applications requires:

1. **Readiness Framework**: Structured criteria evaluation across multiple categories
2. **Deployment Checklist**: Comprehensive pre-deployment verification
3. **Monitoring Setup**: Quantum-specific metrics and alerting
4. **Operational Runbooks**: Documented procedures for common issues
5. **Go-Live Gates**: Clear decision criteria for production launch

The unique challenges of quantum computing (probabilistic outputs, hardware drift, remote execution) require additional rigor beyond traditional software deployment practices.
