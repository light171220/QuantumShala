## 1. The EPR Argument and Local Realism

In 1935, Einstein, Podolsky, and Rosen (EPR) argued that quantum mechanics is incomplete. Their reasoning relied on two assumptions:

<Definition term="Local Realism">
The conjunction of two principles:
1. **Realism**: Physical properties have definite values independent of measurement
2. **Locality**: No influence can travel faster than light; distant measurements cannot affect local outcomes
</Definition>

EPR showed that if quantum mechanics is complete, it must violate local realism. In 1964, John Bell proved this could be experimentally tested.

## 2. Bell's Theorem and the CHSH Inequality

<Definition term="CHSH Inequality">
For any local hidden variable theory with measurement settings a, a' for Alice and b, b' for Bob:
$$|S| = |E(a,b) - E(a,b') + E(a',b) + E(a',b')| \leq 2$$
where E(x,y) is the correlation between measurement outcomes.
</Definition>

<Formula>
$$S_{\text{classical}} \leq 2$$
$$S_{\text{quantum}} \leq 2\sqrt{2} \approx 2.828 \text{ (Tsirelson bound)}$$
</Formula>

```python
import numpy as np

def classical_correlation(hidden_variable, a, b, strategy):
    """
    Classical correlation function based on hidden variable.

    In any local hidden variable theory, Alice's outcome depends
    only on her setting (a) and the hidden variable (lambda).
    Same for Bob.
    """
    # Example deterministic strategy
    if strategy == 'always_same':
        alice_outcome = 1 if hidden_variable < np.cos(a)**2 else -1
        bob_outcome = 1 if hidden_variable < np.cos(b)**2 else -1
    else:
        alice_outcome = 1 if np.random.random() < 0.5 else -1
        bob_outcome = alice_outcome  # Perfect correlation

    return alice_outcome * bob_outcome

def test_chsh_classical(num_trials=100000):
    """
    Test CHSH inequality with classical correlations.
    Shows that no classical strategy can violate S <= 2.
    """
    # Optimal classical settings
    a, a_prime = 0, np.pi/2
    b, b_prime = np.pi/4, 3*np.pi/4

    # Try different classical strategies
    strategies = ['always_same', 'random']

    print("Classical CHSH Test:")
    print("=" * 50)

    for strategy in strategies:
        E_ab = np.mean([classical_correlation(np.random.random(), a, b, strategy)
                        for _ in range(num_trials)])
        E_ab_p = np.mean([classical_correlation(np.random.random(), a, b_prime, strategy)
                          for _ in range(num_trials)])
        E_a_pb = np.mean([classical_correlation(np.random.random(), a_prime, b, strategy)
                          for _ in range(num_trials)])
        E_a_pb_p = np.mean([classical_correlation(np.random.random(), a_prime, b_prime, strategy)
                            for _ in range(num_trials)])

        S = E_ab - E_ab_p + E_a_pb + E_a_pb_p

        print(f"Strategy: {strategy}")
        print(f"  S = {S:.4f} (bound: 2)")
        print(f"  Violated: {abs(S) > 2}")

test_chsh_classical()
```

## 3. Quantum Violation of Bell Inequalities

Quantum mechanics can achieve correlations that violate the CHSH inequality.

```python
import numpy as np

def quantum_correlation(theta_a, theta_b, state='singlet'):
    """
    Calculate quantum correlation E(a,b) for measurements
    at angles theta_a and theta_b.

    For singlet state |Psi->: E(a,b) = -cos(theta_a - theta_b)
    For Bell state |Phi+>: E(a,b) = cos(theta_a - theta_b)
    """
    if state == 'singlet':
        return -np.cos(theta_a - theta_b)
    else:  # phi_plus
        return np.cos(theta_a - theta_b)

def calculate_chsh_quantum(a, a_prime, b, b_prime, state='singlet'):
    """
    Calculate CHSH value for quantum state with given measurement angles.
    """
    E_ab = quantum_correlation(a, b, state)
    E_ab_prime = quantum_correlation(a, b_prime, state)
    E_a_prime_b = quantum_correlation(a_prime, b, state)
    E_a_prime_b_prime = quantum_correlation(a_prime, b_prime, state)

    S = E_ab - E_ab_prime + E_a_prime_b + E_a_prime_b_prime
    return S, (E_ab, E_ab_prime, E_a_prime_b, E_a_prime_b_prime)

# Find optimal angles for maximal violation
print("Quantum CHSH Analysis:")
print("=" * 50)

# Optimal angles for singlet state
# a = 0, a' = pi/2, b = pi/4, b' = -pi/4
a_opt = 0
a_prime_opt = np.pi/2
b_opt = np.pi/4
b_prime_opt = -np.pi/4

S, correlations = calculate_chsh_quantum(a_opt, a_prime_opt, b_opt, b_prime_opt)

print(f"Optimal measurement angles:")
print(f"  Alice: a = 0, a' = pi/2")
print(f"  Bob: b = pi/4, b' = -pi/4")
print(f"\nCorrelations:")
print(f"  E(a,b) = {correlations[0]:.4f}")
print(f"  E(a,b') = {correlations[1]:.4f}")
print(f"  E(a',b) = {correlations[2]:.4f}")
print(f"  E(a',b') = {correlations[3]:.4f}")
print(f"\nCHSH value: S = {S:.4f}")
print(f"Tsirelson bound: 2*sqrt(2) = {2*np.sqrt(2):.4f}")
print(f"Classical bound: 2")
print(f"\nViolation: {abs(S) - 2:.4f} above classical bound")
```

</Tip>

## 4. Tsirelson's Bound

Why does quantum mechanics achieve exactly $2\sqrt{2}$ and not more?

<Definition term="Tsirelson's Bound">
For any quantum state and measurements:
$$|S| \leq 2\sqrt{2}$$
This is the maximum CHSH value achievable by quantum mechanics, reached by maximally entangled states with optimal measurements.
</Definition>

```python
def verify_tsirelson_bound(num_random_states=1000):
    """
    Verify that random quantum states never exceed Tsirelson's bound.
    """
    max_S = 0
    max_angles = None

    for _ in range(num_random_states):
        # Random measurement angles
        a = np.random.uniform(0, np.pi)
        a_prime = np.random.uniform(0, np.pi)
        b = np.random.uniform(0, np.pi)
        b_prime = np.random.uniform(0, np.pi)

        S, _ = calculate_chsh_quantum(a, a_prime, b, b_prime)

        if abs(S) > max_S:
            max_S = abs(S)
            max_angles = (a, a_prime, b, b_prime)

    print(f"Maximum S found: {max_S:.4f}")
    print(f"Tsirelson bound: {2*np.sqrt(2):.4f}")
    print(f"Below bound: {max_S <= 2*np.sqrt(2) + 1e-10}")

    return max_S

verify_tsirelson_bound()
```

<Warning>
Theories beyond quantum mechanics (like some post-quantum theories) could in principle achieve S = 4, the algebraic maximum. That quantum mechanics stops at Tsirelson's bound is a fundamental constraint that may reveal deep principles about nature.
</Warning>

## 5. Simulating a Bell Test

```python
import numpy as np

def simulate_bell_test(num_pairs=10000, noise=0.0):
    """
    Simulate a complete Bell test experiment.

    Args:
        num_pairs: number of entangled pairs
        noise: probability of random outcome (depolarizing noise)

    Returns:
        S: experimental CHSH value
        statistics: detailed breakdown
    """
    # Measurement settings
    settings = {
        'a': 0, 'a_prime': np.pi/2,
        'b': np.pi/4, 'b_prime': -np.pi/4
    }

    # Store outcomes for each setting combination
    outcomes = {
        'ab': [], 'ab_prime': [],
        'a_prime_b': [], 'a_prime_b_prime': []
    }

    for _ in range(num_pairs):
        # Randomly choose settings for this pair
        alice_setting = np.random.choice(['a', 'a_prime'])
        bob_setting = np.random.choice(['b', 'b_prime'])

        theta_a = settings[alice_setting]
        theta_b = settings[bob_setting]

        # Quantum measurement simulation
        # For singlet state, correlation is -cos(theta_a - theta_b)
        correlation = -np.cos(theta_a - theta_b)

        # Convert to individual outcomes
        # P(same) = (1 + correlation) / 2
        p_same = (1 + correlation) / 2

        if np.random.random() < noise:
            # Random outcome due to noise
            alice_outcome = np.random.choice([1, -1])
            bob_outcome = np.random.choice([1, -1])
        else:
            alice_outcome = np.random.choice([1, -1])
            if np.random.random() < p_same:
                bob_outcome = alice_outcome
            else:
                bob_outcome = -alice_outcome

        # Record outcome
        key = alice_setting + '_' + bob_setting
        key = key.replace('a_b', 'ab').replace('a_a', 'a').replace('b_b', 'b')
        if alice_setting == 'a':
            if bob_setting == 'b':
                outcomes['ab'].append(alice_outcome * bob_outcome)
            else:
                outcomes['ab_prime'].append(alice_outcome * bob_outcome)
        else:
            if bob_setting == 'b':
                outcomes['a_prime_b'].append(alice_outcome * bob_outcome)
            else:
                outcomes['a_prime_b_prime'].append(alice_outcome * bob_outcome)

    # Calculate correlations
    E = {}
    for key in outcomes:
        if len(outcomes[key]) > 0:
            E[key] = np.mean(outcomes[key])
        else:
            E[key] = 0

    # CHSH value
    S = E['ab'] - E['ab_prime'] + E['a_prime_b'] + E['a_prime_b_prime']

    return S, E, outcomes

# Run simulation
print("Bell Test Simulation:")
print("=" * 50)

for noise in [0.0, 0.05, 0.10, 0.20]:
    S, E, _ = simulate_bell_test(num_pairs=100000, noise=noise)
    violation = abs(S) > 2
    print(f"\nNoise level: {noise*100:.0f}%")
    print(f"  S = {S:.4f}")
    print(f"  Violation: {violation}")
```

## 6. Loopholes in Bell Tests

<Definition term="Locality Loophole">
If the measurement events are not spacelike separated, one measurement could influence the other through a subluminal signal, allowing a local hidden variable explanation.
</Definition>

<Definition term="Detection Loophole">
If detection efficiency is low, the detected subset might not represent the full ensemble, potentially allowing local hidden variable models that exploit the undetected pairs.
</Definition>

```python
def simulate_with_detection_loophole(num_pairs=10000, efficiency=0.9):
    """
    Simulate Bell test with imperfect detection.

    Low efficiency can make violations disappear if we
    only count detected pairs.
    """
    detected_outcomes = {
        'ab': [], 'ab_prime': [],
        'a_prime_b': [], 'a_prime_b_prime': []
    }

    # Settings
    settings = {'a': 0, 'a_prime': np.pi/2, 'b': np.pi/4, 'b_prime': -np.pi/4}

    for _ in range(num_pairs):
        # Random settings
        alice_setting = np.random.choice(['a', 'a_prime'])
        bob_setting = np.random.choice(['b', 'b_prime'])

        # Check if both detectors fire
        alice_detected = np.random.random() < efficiency
        bob_detected = np.random.random() < efficiency

        if not (alice_detected and bob_detected):
            continue  # Only count coincidences

        theta_a = settings[alice_setting]
        theta_b = settings[bob_setting]
        correlation = -np.cos(theta_a - theta_b)

        # Generate outcomes
        p_same = (1 + correlation) / 2
        alice_outcome = np.random.choice([1, -1])
        bob_outcome = alice_outcome if np.random.random() < p_same else -alice_outcome

        # Record
        if alice_setting == 'a':
            key = 'ab' if bob_setting == 'b' else 'ab_prime'
        else:
            key = 'a_prime_b' if bob_setting == 'b' else 'a_prime_b_prime'
        detected_outcomes[key].append(alice_outcome * bob_outcome)

    # Calculate S
    E = {k: np.mean(v) if len(v) > 0 else 0 for k, v in detected_outcomes.items()}
    S = E['ab'] - E['ab_prime'] + E['a_prime_b'] + E['a_prime_b_prime']

    total_coincidences = sum(len(v) for v in detected_outcomes.values())

    return S, total_coincidences

print("Detection Loophole Analysis:")
print("=" * 50)

for eff in [1.0, 0.9, 0.8, 0.7]:
    S, coincidences = simulate_with_detection_loophole(100000, eff)
    print(f"\nEfficiency: {eff*100:.0f}%")
    print(f"  Coincidences: {coincidences}")
    print(f"  S = {S:.4f}")
    print(f"  Violation: {abs(S) > 2}")
```

<Tip>
Modern "loophole-free" Bell tests (2015 onwards) close both loopholes simultaneously, providing the strongest evidence yet against local hidden variable theories.
</Tip>

## 7. Device-Independent Applications

Bell violations enable protocols that don't require trust in devices.

```python
def device_independent_qkd_estimate(S_observed, num_rounds):
    """
    Estimate secure key rate from observed CHSH violation.

    In device-independent QKD, the security comes from
    the Bell violation itself, not from trusting the devices.
    """
    # Key rate lower bound from CHSH value
    # Based on DIQKD security proofs

    if abs(S_observed) <= 2:
        # No violation, no security
        return 0

    # Simplified estimate of key rate
    # Actual formulas are more complex
    tsirelson = 2 * np.sqrt(2)
    normalized_violation = (abs(S_observed) - 2) / (tsirelson - 2)

    # Approximate key rate (bits per round)
    key_rate = normalized_violation * 0.5  # Simplified

    total_bits = key_rate * num_rounds

    return key_rate, total_bits

# Example
S_exp = 2.7  # Typical experimental value
key_rate, total = device_independent_qkd_estimate(S_exp, 1000000)
print(f"Device-Independent QKD Estimate:")
print(f"  Observed S: {S_exp}")
print(f"  Key rate: {key_rate:.4f} bits/round")
print(f"  Total secure bits (1M rounds): {total:.0f}")
```

## 8. Beyond CHSH: Other Bell Inequalities

```python
def mermin_inequality(n_parties):
    """
    The Mermin inequality generalizes Bell tests to n parties.

    Classical bound: 1
    Quantum max: 2^((n-1)/2)

    The violation grows exponentially with the number of parties!
    """
    classical_bound = 1
    quantum_max = 2**((n_parties - 1) / 2)

    return classical_bound, quantum_max

print("Multipartite Bell Inequalities (Mermin):")
print("=" * 50)
print(f"{'Parties':<10} {'Classical':<12} {'Quantum Max':<12} {'Ratio':<10}")
print("-" * 44)

for n in range(2, 8):
    c_bound, q_max = mermin_inequality(n)
    ratio = q_max / c_bound
    print(f"{n:<10} {c_bound:<12.1f} {q_max:<12.2f} {ratio:<10.2f}")
```
