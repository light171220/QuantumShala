## 1. Vision of the Quantum Internet

</Definition>

```python
import numpy as np

class QuantumNetworkNode:
    """
    Model of a quantum network node.
    """

    def __init__(self, node_id, node_type='end_node'):
        self.node_id = node_id
        self.node_type = node_type  # end_node, repeater, or router
        self.connections = {}
        self.memory_qubits = []
        self.entangled_pairs = {}

    def connect(self, other_node, channel_length_km):
        """Establish a physical connection."""
        self.connections[other_node.node_id] = {
            'node': other_node,
            'length': channel_length_km,
            'loss_db': 0.2 * channel_length_km
        }

    def generate_entanglement(self, target_node_id):
        """Attempt to generate entanglement with connected node."""
        if target_node_id not in self.connections:
            return False, "Not connected"

        conn = self.connections[target_node_id]
        p_success = 10 ** (-conn['loss_db'] / 10)

        if np.random.random() < p_success:
            pair_id = f"{self.node_id}-{target_node_id}-{len(self.entangled_pairs)}"
            self.entangled_pairs[pair_id] = {
                'partner': target_node_id,
                'fidelity': 0.95,
                'created': 'now'
            }
            return True, pair_id
        return False, "Generation failed"
```

## 4. Entanglement Routing

Unlike classical routing that forwards packets, quantum routing establishes entanglement paths:

```python
import numpy as np
from collections import defaultdict

class QuantumRouter:
    """
    Simple quantum network router.
    """

    def __init__(self, topology):
        """
        topology: dict mapping (node1, node2) -> distance
        """
        self.topology = topology
        self.nodes = set()
        for (n1, n2) in topology.keys():
            self.nodes.add(n1)
            self.nodes.add(n2)

    def find_path(self, source, destination):
        """
        Find best path for entanglement distribution.
        Uses modified Dijkstra with entanglement probability metric.
        """
        if source not in self.nodes or destination not in self.nodes:
            return None, 0

        # Build adjacency list
        adj = defaultdict(list)
        for (n1, n2), dist in self.topology.items():
            p = 10 ** (-0.2 * dist / 10)  # Success probability
            adj[n1].append((n2, -np.log(p)))  # Use -log(p) as weight
            adj[n2].append((n1, -np.log(p)))

        # Dijkstra
        import heapq
        dist = {n: float('inf') for n in self.nodes}
        prev = {n: None for n in self.nodes}
        dist[source] = 0

        pq = [(0, source)]
        while pq:
            d, u = heapq.heappop(pq)
            if d > dist[u]:
                continue
            for v, w in adj[u]:
                if dist[u] + w < dist[v]:
                    dist[v] = dist[u] + w
                    prev[v] = u
                    heapq.heappush(pq, (dist[v], v))

        # Reconstruct path
        path = []
        node = destination
        while node is not None:
            path.append(node)
            node = prev[node]
        path.reverse()

        if path[0] != source:
            return None, 0

        # Calculate path probability
        total_prob = np.exp(-dist[destination])

        return path, total_prob


def demonstrate_routing():
    """
    Demonstrate quantum routing in a network.
    """
    # Define network topology (distances in km)
    topology = {
        ('A', 'R1'): 30,
        ('R1', 'R2'): 50,
        ('R2', 'B'): 40,
        ('A', 'R3'): 60,
        ('R3', 'B'): 45,
        ('R1', 'R3'): 35,
        ('R2', 'R3'): 25
    }

    router = QuantumRouter(topology)

    print("Quantum Network Routing")
    print("=" * 55)
    print("Network topology:")
    print("       R1 --- R2")
    print("      / |     | \\")
    print("     A  |     |  B")
    print("      \\ |     | /")
    print("       R3 ----+")
    print()

    # Find best path A to B
    path, prob = router.find_path('A', 'B')
    print(f"Best path A → B: {' → '.join(path)}")
    print(f"Path success probability: {prob:.4f}")

    # Compare alternative paths
    print("\nAlternative paths:")
    paths = [
        ['A', 'R1', 'R2', 'B'],
        ['A', 'R3', 'B'],
        ['A', 'R1', 'R3', 'B']
    ]
    for p in paths:
        total_dist = sum(topology.get((p[i], p[i+1]), topology.get((p[i+1], p[i]), 0))
                         for i in range(len(p)-1))
        prob = 10 ** (-0.2 * total_dist / 10)
        print(f"  {' → '.join(p)}: {total_dist} km, P = {prob:.4f}")

demonstrate_routing()
```

## 7. Network Simulation

```python
import numpy as np

class QuantumInternet:
    """
    Simple quantum internet simulation.
    """

    def __init__(self):
        self.nodes = {}
        self.links = {}
        self.entanglement_registry = {}

    def add_node(self, name, capabilities):
        self.nodes[name] = {
            'capabilities': capabilities,
            'memory_slots': capabilities.get('memory_qubits', 0),
            'stored_entanglement': []
        }

    def add_link(self, node1, node2, distance_km, channel_type='fiber'):
        link_id = f"{node1}-{node2}"
        loss_db = 0.2 * distance_km if channel_type == 'fiber' else 0.05 * distance_km
        self.links[link_id] = {
            'nodes': (node1, node2),
            'distance': distance_km,
            'loss_db': loss_db,
            'p_success': 10 ** (-loss_db / 10)
        }

    def establish_entanglement(self, node1, node2, fidelity_target=0.9):
        """
        Establish entanglement between two nodes.
        """
        # Find path
        path = self._find_path(node1, node2)
        if not path:
            return None

        # Generate entanglement for each segment
        segment_entanglements = []
        for i in range(len(path) - 1):
            link_id = f"{path[i]}-{path[i+1]}"
            if link_id not in self.links:
                link_id = f"{path[i+1]}-{path[i]}"

            link = self.links[link_id]
            attempts = 0
            while np.random.random() > link['p_success']:
                attempts += 1
                if attempts > 1000:
                    return None

            segment_entanglements.append({
                'link': link_id,
                'fidelity': 0.95,
                'attempts': attempts + 1
            })

        # Perform swapping
        final_fidelity = np.prod([s['fidelity'] for s in segment_entanglements])
        final_fidelity *= 0.95 ** (len(path) - 2)  # Swapping loss

        if final_fidelity < fidelity_target:
            # Would need purification
            final_fidelity = min(0.99, final_fidelity * 1.1)

        ent_id = f"ent-{node1}-{node2}-{len(self.entanglement_registry)}"
        self.entanglement_registry[ent_id] = {
            'nodes': (node1, node2),
            'path': path,
            'fidelity': final_fidelity,
            'total_attempts': sum(s['attempts'] for s in segment_entanglements)
        }

        return ent_id

    def _find_path(self, source, dest):
        # Simple BFS for path finding
        from collections import deque

        adj = {}
        for link_id, link in self.links.items():
            n1, n2 = link['nodes']
            adj.setdefault(n1, []).append(n2)
            adj.setdefault(n2, []).append(n1)

        visited = set()
        queue = deque([(source, [source])])

        while queue:
            node, path = queue.popleft()
            if node == dest:
                return path
            if node in visited:
                continue
            visited.add(node)
            for neighbor in adj.get(node, []):
                if neighbor not in visited:
                    queue.append((neighbor, path + [neighbor]))

        return None


def simulate_quantum_internet():

    # Add links
    qnet.add_link('Alice', 'R1', 50)
```
