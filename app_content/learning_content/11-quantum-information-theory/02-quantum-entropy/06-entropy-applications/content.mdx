## 1. Entanglement Quantification

<Definition term="Entanglement Entropy">
For a pure bipartite state $|\psi\rangle_{AB}$, the entanglement entropy is:
$$E(|\psi\rangle_{AB}) = S(\rho_A) = S(\rho_B)$$
This is the unique measure of entanglement for pure states, quantifying the number of ebits (maximally entangled pairs) needed to create the state.
</Definition>

### Case Study: Quantifying Entanglement in Quantum Algorithms

```python
import numpy as np
from scipy.linalg import expm, logm
from typing import List, Tuple, Dict

class EntanglementAnalyzer:
    """Comprehensive tools for entanglement analysis via entropy."""

    def __init__(self):
        pass

    def von_neumann_entropy(self, rho: np.ndarray) -> float:
        """Compute S(rho) = -Tr(rho log2 rho)."""
        eigenvalues = np.linalg.eigvalsh(rho)
        eigenvalues = eigenvalues[eigenvalues > 1e-15]
        if len(eigenvalues) == 0:
            return 0.0
        return -np.sum(eigenvalues * np.log2(eigenvalues))

    def partial_trace(self, rho: np.ndarray, dims: List[int],
                      keep: List[int]) -> np.ndarray:
        """Compute partial trace over specified subsystems."""
        n = len(dims)
        rho_tensor = rho.reshape(dims + dims)

        trace_over = [i for i in range(n) if i not in keep]
        for idx in sorted(trace_over, reverse=True):
            rho_tensor = np.trace(rho_tensor, axis1=idx, axis2=idx + n)
            n -= 1

        kept_dims = [dims[i] for i in keep]
        new_dim = np.prod(kept_dims) if kept_dims else 1
        return rho_tensor.reshape(new_dim, new_dim)

    def entanglement_entropy(self, psi: np.ndarray, dims: List[int],
                              bipartition: Tuple[List[int], List[int]]) -> float:
        """
        Compute entanglement entropy for given bipartition.

        Args:
            psi: Pure state vector
            dims: Dimensions of each subsystem
            bipartition: (subsystemA_indices, subsystemB_indices)
        """
        rho = np.outer(psi, psi.conj())
        rho_A = self.partial_trace(rho, dims, bipartition[0])
        return self.von_neumann_entropy(rho_A)

    def schmidt_decomposition(self, psi: np.ndarray, dim_A: int,
                               dim_B: int) -> Dict:
        """
        Compute Schmidt decomposition of bipartite state.

        |psi> = sum_i lambda_i |i>_A |i>_B
        """
        # Reshape to matrix
        psi_matrix = psi.reshape(dim_A, dim_B)

        # SVD gives Schmidt coefficients
        U, S, Vh = np.linalg.svd(psi_matrix, full_matrices=False)

        # Normalize
        S = S / np.linalg.norm(S)

        # Schmidt number (number of non-zero coefficients)
        schmidt_number = np.sum(S > 1e-10)

        # Entanglement entropy from Schmidt coefficients
        probs = S[S > 1e-10]**2
        entropy = -np.sum(probs * np.log2(probs))

        return {
            'schmidt_coefficients': S,
            'schmidt_number': schmidt_number,
            'entanglement_entropy': entropy,
            'U': U,
            'V': Vh.conj().T
        }

    def concurrence(self, rho: np.ndarray) -> float:
        """
        Compute concurrence for two-qubit mixed state.
        C(rho) = max(0, lambda_1 - lambda_2 - lambda_3 - lambda_4)
        """
        # Pauli Y matrices
        sigma_y = np.array([[0, -1j], [1j, 0]])
        sigma_yy = np.kron(sigma_y, sigma_y)

        # Spin-flipped state
        rho_tilde = sigma_yy @ rho.conj() @ sigma_yy

        # R matrix
        sqrt_rho = self._matrix_sqrt(rho)
        R = sqrt_rho @ rho_tilde @ sqrt_rho
        R_sqrt = self._matrix_sqrt(R)

        # Eigenvalues in decreasing order
        eigenvalues = np.sort(np.real(np.linalg.eigvalsh(R_sqrt)))[::-1]

        concurrence = max(0, eigenvalues[0] - eigenvalues[1] -
                         eigenvalues[2] - eigenvalues[3])
        return concurrence

    def _matrix_sqrt(self, A: np.ndarray) -> np.ndarray:
        """Compute matrix square root."""
        eigenvalues, eigenvectors = np.linalg.eigh(A)
        eigenvalues = np.maximum(eigenvalues, 0)
        sqrt_eigenvalues = np.sqrt(eigenvalues)
        return eigenvectors @ np.diag(sqrt_eigenvalues) @ eigenvectors.conj().T

    def entanglement_of_formation(self, rho: np.ndarray) -> float:
        """
        Compute entanglement of formation for two-qubit state.
        E_F(rho) = h((1 + sqrt(1 - C^2))/2)
        where h is binary entropy.
        """
        C = self.concurrence(rho)

        def binary_entropy(p):
            if p <= 0 or p >= 1:
                return 0
            return -p * np.log2(p) - (1-p) * np.log2(1-p)

        x = (1 + np.sqrt(1 - C**2)) / 2
        return binary_entropy(x)
```

## 2. Quantum Thermodynamics Applications

<Definition term="Quantum Free Energy">
The non-equilibrium free energy of a quantum state $\rho$ with respect to Hamiltonian $H$ at temperature $T$ is:
$$F(\rho) = \langle H \rangle_\rho - T \cdot S(\rho) = \text{Tr}(H\rho) + T \cdot \text{Tr}(\rho \ln \rho)$$
</Definition>

### Landauer's Principle: Information Erasure Cost

<Formula>
**Landauer's Principle**:
Erasing $n$ bits of information requires dissipating at least:
$$Q_{min} = k_B T \ln(2) \cdot S(\rho)$$
where $S(\rho)$ is the von Neumann entropy of the state being erased.
</Formula>

```python
class QuantumThermodynamics:
    """Entropy applications in quantum thermodynamics."""

    def __init__(self, temperature: float = 1.0):
        """Initialize with temperature (k_B = 1 units)."""
        self.T = temperature
        self.beta = 1.0 / temperature if temperature > 0 else np.inf

    def thermal_state(self, H: np.ndarray) -> np.ndarray:
        """Compute Gibbs thermal state rho_th = exp(-beta*H)/Z."""
        exp_H = expm(-self.beta * H)
        Z = np.trace(exp_H)
        return exp_H / Z

    def free_energy(self, rho: np.ndarray, H: np.ndarray) -> float:
        """Compute F(rho) = <H> - T*S(rho)."""
        energy = np.real(np.trace(H @ rho))

        eigenvalues = np.linalg.eigvalsh(rho)
        eigenvalues = eigenvalues[eigenvalues > 1e-15]
        entropy = -np.sum(eigenvalues * np.log(eigenvalues))

        return energy - self.T * entropy

    def extractable_work(self, rho: np.ndarray, H: np.ndarray) -> float:
        """
        Maximum work extractable from state rho.
        W_max = F(rho) - F(rho_th) = T * D(rho || rho_th)
        """
        F_rho = self.free_energy(rho, H)
        rho_th = self.thermal_state(H)
        F_th = self.free_energy(rho_th, H)
        return F_rho - F_th

    def landauer_cost(self, rho: np.ndarray) -> float:
        """Minimum heat dissipation to erase state rho."""
        eigenvalues = np.linalg.eigvalsh(rho)
        eigenvalues = eigenvalues[eigenvalues > 1e-15]
        entropy = -np.sum(eigenvalues * np.log(eigenvalues))
        return self.T * entropy

    def relative_entropy(self, rho: np.ndarray, sigma: np.ndarray) -> float:
        """Compute D(rho||sigma) = Tr(rho(log rho - log sigma))."""
        eigenvalues_rho = np.linalg.eigvalsh(rho)
        eigenvalues_rho = eigenvalues_rho[eigenvalues_rho > 1e-15]
        S_rho = -np.sum(eigenvalues_rho * np.log(eigenvalues_rho))

        # Cross entropy -Tr(rho log sigma)
        log_sigma = logm(sigma + 1e-15 * np.eye(len(sigma)))
        cross_entropy = -np.real(np.trace(rho @ log_sigma))

        return cross_entropy - S_rho

class QuantumHeatEngine:
    """Simulate quantum heat engines using entropic analysis."""

    def __init__(self, omega_1: float, omega_2: float,
                 T_hot: float, T_cold: float):
        """
        Quantum Otto engine parameters.
        omega_1, omega_2: Energy gaps for compression/expansion
        T_hot, T_cold: Bath temperatures
        """
        self.omega_1 = omega_1
        self.omega_2 = omega_2
        self.T_H = T_hot
        self.T_C = T_cold

    def thermal_population(self, omega: float, T: float) -> float:
        """Excited state population at thermal equilibrium."""
        if T < 1e-10:
            return 0
        return 1 / (1 + np.exp(omega / T))

    def run_otto_cycle(self) -> Dict:
        """Execute quantum Otto cycle and compute efficiency."""
        # Stage 1: Thermal contact with hot bath at omega_1
        p1_hot = self.thermal_population(self.omega_1, self.T_H)

        # Stage 2: Adiabatic expansion (omega_1 -> omega_2)
        # Population unchanged, energy changes
        E_after_expansion = self.omega_2 * p1_hot
        E_before_expansion = self.omega_1 * p1_hot
        W_expansion = E_before_expansion - E_after_expansion

        # Stage 3: Thermal contact with cold bath at omega_2
        p1_cold = self.thermal_population(self.omega_2, self.T_C)
        Q_cold = self.omega_2 * (p1_cold - p1_hot)  # Heat absorbed (negative)

        # Stage 4: Adiabatic compression (omega_2 -> omega_1)
        E_after_compression = self.omega_1 * p1_cold
        E_before_compression = self.omega_2 * p1_cold
        W_compression = E_before_compression - E_after_compression

        # Heat absorbed from hot bath
        Q_hot = self.omega_1 * (p1_hot - p1_cold)

        W_net = W_expansion + W_compression
        efficiency = W_net / Q_hot if Q_hot > 0 else 0

        # Theoretical limits
        eta_carnot = 1 - self.T_C / self.T_H
        eta_otto_ideal = 1 - self.omega_2 / self.omega_1

        return {
            'W_net': W_net,
            'Q_hot': Q_hot,
            'Q_cold': Q_cold,
            'efficiency': efficiency,
            'carnot_efficiency': eta_carnot,
            'otto_efficiency': eta_otto_ideal,
            'entropy_production': -Q_cold/self.T_C - Q_hot/self.T_H
        }
```

## 3. Communication Theory: Channel Capacity

<Definition term="Quantum Channel Capacity">
The quantum capacity of a channel $\mathcal{N}$ is the maximum rate of quantum information transmission:
$$Q(\mathcal{N}) = \lim_{n \to \infty} \frac{1}{n} \max_\rho I_c(\rho, \mathcal{N}^{\otimes n})$$
where $I_c(\rho, \mathcal{N}) = S(\mathcal{N}(\rho)) - S(E)$ is the coherent information.
</Definition>

<Formula>
**Holevo Bound**:
The classical capacity of a quantum channel is bounded by:
$$C \leq \chi = S\left(\sum_i p_i \rho_i\right) - \sum_i p_i S(\rho_i)$$
</Formula>

```python
class QuantumChannelCapacity:
    """Analyze quantum channel capacities using entropy."""

    def __init__(self):
        pass

    def von_neumann_entropy(self, rho: np.ndarray) -> float:
        """Compute von Neumann entropy."""
        eigenvalues = np.linalg.eigvalsh(rho)
        eigenvalues = eigenvalues[eigenvalues > 1e-15]
        if len(eigenvalues) == 0:
            return 0.0
        return -np.sum(eigenvalues * np.log2(eigenvalues))

    def depolarizing_channel(self, rho: np.ndarray, p: float) -> np.ndarray:
        """Apply depolarizing channel: rho -> (1-p)*rho + p*I/d."""
        d = len(rho)
        return (1 - p) * rho + p * np.eye(d) / d

    def amplitude_damping_channel(self, rho: np.ndarray, gamma: float) -> np.ndarray:
        """Apply amplitude damping channel."""
        K0 = np.array([[1, 0], [0, np.sqrt(1-gamma)]])
        K1 = np.array([[0, np.sqrt(gamma)], [0, 0]])
        return K0 @ rho @ K0.conj().T + K1 @ rho @ K1.conj().T

    def dephasing_channel(self, rho: np.ndarray, p: float) -> np.ndarray:
        """Apply dephasing channel."""
        d = len(rho)
        rho_dephased = rho.copy()
        for i in range(d):
            for j in range(d):
                if i != j:
                    rho_dephased[i, j] *= (1 - p)
        return rho_dephased

    def holevo_quantity(self, ensemble: List[Tuple[float, np.ndarray]]) -> float:
        """
        Compute Holevo chi for an ensemble {(p_i, rho_i)}.
        chi = S(sum_i p_i rho_i) - sum_i p_i S(rho_i)
        """
        # Average state
        rho_avg = sum(p * rho for p, rho in ensemble)

        # Holevo quantity
        S_avg = self.von_neumann_entropy(rho_avg)
        avg_S = sum(p * self.von_neumann_entropy(rho) for p, rho in ensemble)

        return S_avg - avg_S

    def coherent_information(self, rho_input: np.ndarray,
                              channel_output: np.ndarray,
                              env_state: np.ndarray) -> float:
        """
        Compute coherent information I_c = S(B) - S(E).
        B: channel output, E: environment
        """
        S_B = self.von_neumann_entropy(channel_output)
        S_E = self.von_neumann_entropy(env_state)
        return S_B - S_E

    def classical_capacity_bound(self, channel_fn, noise_param: float,
                                  n_samples: int = 100) -> float:
        """
        Estimate classical capacity via Holevo bound optimization.
        """
        max_holevo = 0

        for _ in range(n_samples):
            # Random ensemble
            n_states = np.random.randint(2, 5)
            probs = np.random.dirichlet(np.ones(n_states))

            ensemble_input = []
            ensemble_output = []

            for i in range(n_states):
                # Random pure state
                psi = np.random.randn(2) + 1j * np.random.randn(2)
                psi /= np.linalg.norm(psi)
                rho_in = np.outer(psi, psi.conj())

                rho_out = channel_fn(rho_in, noise_param)

                ensemble_input.append((probs[i], rho_in))
                ensemble_output.append((probs[i], rho_out))

            chi = self.holevo_quantity(ensemble_output)
            max_holevo = max(max_holevo, chi)

        return max_holevo
```

## 4. Many-Body Physics: Area Laws and Criticality

<Definition term="Entanglement Area Law">
For ground states of local gapped Hamiltonians in $d$ dimensions, the entanglement entropy of a region $A$ scales as:
$$S(A) \propto |\partial A|$$
where $|\partial A|$ is the boundary area, not the volume.
</Definition>

<Formula>
**Logarithmic Correction at Criticality** (1D CFT):
$$S(L) = \frac{c}{3} \log_2\left(\frac{L}{\epsilon}\right) + \text{const}$$
where $c$ is the central charge characterizing the universality class.
</Formula>

```python
class ManyBodyEntropy:
    """Entropy analysis in many-body quantum systems."""

    def __init__(self):
        pass

    def von_neumann_entropy(self, rho: np.ndarray) -> float:
        """Compute von Neumann entropy."""
        eigenvalues = np.linalg.eigvalsh(rho)
        eigenvalues = eigenvalues[eigenvalues > 1e-15]
        if len(eigenvalues) == 0:
            return 0.0
        return -np.sum(eigenvalues * np.log2(eigenvalues))

    def create_transverse_ising_ground_state(self, n_sites: int,
                                              h: float) -> np.ndarray:
        """
        Ground state of transverse field Ising model.
        H = -sum_i sigma_z^i sigma_z^{i+1} - h sum_i sigma_x^i
        """
        # Pauli matrices
        sigma_x = np.array([[0, 1], [1, 0]])
        sigma_z = np.array([[1, 0], [0, -1]])
        I = np.eye(2)

        # Build Hamiltonian
        dim = 2**n_sites
        H = np.zeros((dim, dim))

        for i in range(n_sites - 1):
            # sigma_z^i sigma_z^{i+1}
            term = np.eye(1)
            for j in range(n_sites):
                if j == i or j == i + 1:
                    term = np.kron(term, sigma_z)
                else:
                    term = np.kron(term, I)
            H -= term

        for i in range(n_sites):
            # h * sigma_x^i
            term = np.eye(1)
            for j in range(n_sites):
                if j == i:
                    term = np.kron(term, sigma_x)
                else:
                    term = np.kron(term, I)
            H -= h * term

        # Find ground state
        eigenvalues, eigenvectors = np.linalg.eigh(H)
        ground_state = eigenvectors[:, 0]

        return ground_state

    def compute_half_chain_entropy(self, psi: np.ndarray, n_sites: int) -> float:
        """Compute entanglement entropy for half-chain bipartition."""
        n_A = n_sites // 2
        n_B = n_sites - n_A
        dim_A = 2**n_A
        dim_B = 2**n_B

        # Reshape and compute reduced density matrix
        psi_matrix = psi.reshape(dim_A, dim_B)
        rho_A = psi_matrix @ psi_matrix.conj().T

        return self.von_neumann_entropy(rho_A)

    def area_law_scaling(self, psi: np.ndarray, n_sites: int) -> List[Dict]:
        """
        Compute entropy for various subsystem sizes.
        Returns data for area law analysis.
        """
        results = []

        for n_A in range(1, n_sites):
            n_B = n_sites - n_A
            dim_A = 2**n_A
            dim_B = 2**n_B

            # Compute reduced density matrix
            psi_matrix = psi.reshape(dim_A, dim_B)
            rho_A = psi_matrix @ psi_matrix.conj().T

            S = self.von_neumann_entropy(rho_A)

            results.append({
                'n_A': n_A,
                'boundary_size': 1,  # 1D: boundary is just 2 points
                'entropy': S
            })

        return results

    def fit_cft_scaling(self, sizes: List[int],
                         entropies: List[float]) -> Dict:
        """
        Fit CFT logarithmic scaling: S = (c/3) log(L) + const
        Returns estimated central charge.
        """
        # Linear regression on log scale
        log_sizes = np.log2(sizes)
        A = np.vstack([log_sizes, np.ones(len(sizes))]).T
        slope, intercept = np.linalg.lstsq(A, entropies, rcond=None)[0]

        central_charge = 3 * slope

        return {
            'central_charge': central_charge,
            'slope': slope,
            'intercept': intercept
        }
```

## 5. Quantum Error Correction Applications

<Definition term="Coherent Information and Capacity">
The quantum capacity of a channel is achieved when the coherent information is maximized:
$$I_c(\rho, \mathcal{N}) = S(\mathcal{N}(\rho)) - S(\mathcal{N}^c(\rho))$$
where $\mathcal{N}^c$ is the complementary channel (to environment).
</Definition>

```python
class EntropicQEC:
    """Entropic bounds on quantum error correction."""

    def __init__(self):
        pass

    def von_neumann_entropy(self, rho: np.ndarray) -> float:
        """Compute von Neumann entropy."""
        eigenvalues = np.linalg.eigvalsh(rho)
        eigenvalues = eigenvalues[eigenvalues > 1e-15]
        if len(eigenvalues) == 0:
            return 0.0
        return -np.sum(eigenvalues * np.log2(eigenvalues))

    def quantum_singleton_bound(self, n: int, d: int) -> int:
        """Maximum k for [[n, k, d]] code."""
        return max(0, n - 2 * (d - 1))

    def check_knill_laflamme(self, code_projector: np.ndarray,
                              errors: List[np.ndarray]) -> Dict:
        """
        Check Knill-Laflamme conditions for error correction.

        P E_a^dag E_b P = c_{ab} P
        where P is code projector, E_a are errors.
        """
        results = []
        d = len(errors)

        for i in range(d):
            for j in range(d):
                E_i, E_j = errors[i], errors[j]
                # P E_i^dag E_j P
                term = code_projector @ E_i.conj().T @ E_j @ code_projector

                # Check if proportional to P
                if np.allclose(term, np.zeros_like(term)):
                    c_ij = 0
                    proportional = True
                else:
                    # Find coefficient
                    # term = c * P implies c = Tr(term)/Tr(P)
                    c_ij = np.trace(term) / np.trace(code_projector)
                    proportional = np.allclose(term, c_ij * code_projector)

                results.append({
                    'i': i, 'j': j,
                    'c_ij': c_ij,
                    'satisfied': proportional
                })

        all_satisfied = all(r['satisfied'] for r in results)
        return {'conditions': results, 'correctable': all_satisfied}

    def approximate_qec_bound(self, rho_code: np.ndarray,
                               channel_output: np.ndarray,
                               fidelity: float) -> Dict:
        """
        Bounds on approximate QEC from entropy.

        For epsilon-approximate correction:
        1 - F >= f(I_c) for some function f
        """
        S_out = self.von_neumann_entropy(channel_output)
        S_in = self.von_neumann_entropy(rho_code)

        # Fannes inequality bound
        epsilon = 1 - fidelity
        entropy_change = abs(S_out - S_in)

        # Bound from continuity
        fannes_bound = 2 * epsilon * np.log2(len(rho_code)) + 2 * epsilon

        return {
            'entropy_input': S_in,
            'entropy_output': S_out,
            'entropy_change': entropy_change,
            'fannes_bound': fannes_bound,
            'fidelity': fidelity
        }
```

```python
class PracticalEntropyTools:
    """Practical entropy calculations for quantum computing."""

    def __init__(self):
        pass

    def von_neumann_entropy(self, rho: np.ndarray) -> float:
        """Compute von Neumann entropy."""
        eigenvalues = np.linalg.eigvalsh(rho)
        eigenvalues = eigenvalues[eigenvalues > 1e-15]
        if len(eigenvalues) == 0:
            return 0.0
        return -np.sum(eigenvalues * np.log2(eigenvalues))

    def linear_entropy(self, rho: np.ndarray) -> float:
        """Compute linear entropy S_L = 1 - Tr(rho^2)."""
        return 1 - np.real(np.trace(rho @ rho))

    def renyi_entropy(self, rho: np.ndarray, alpha: float) -> float:
        """
        Compute Renyi entropy S_alpha = (1/(1-alpha)) log Tr(rho^alpha).
        """
        if abs(alpha - 1) < 1e-10:
            return self.von_neumann_entropy(rho)

        eigenvalues = np.linalg.eigvalsh(rho)
        eigenvalues = eigenvalues[eigenvalues > 1e-15]

        trace_rho_alpha = np.sum(eigenvalues**alpha)
        return np.log2(trace_rho_alpha) / (1 - alpha)

    def min_entropy(self, rho: np.ndarray) -> float:
        """Compute min-entropy S_min = -log max(eigenvalue)."""
        eigenvalues = np.linalg.eigvalsh(rho)
        return -np.log2(np.max(eigenvalues))

    def mutual_information(self, rho_AB: np.ndarray, dim_A: int) -> float:
        """Compute I(A:B) = S(A) + S(B) - S(AB)."""
        dim_B = len(rho_AB) // dim_A

        # Partial traces
        rho_reshaped = rho_AB.reshape(dim_A, dim_B, dim_A, dim_B)
        rho_A = np.trace(rho_reshaped, axis1=1, axis2=3)
        rho_B = np.trace(rho_reshaped, axis1=0, axis2=2)

        S_A = self.von_neumann_entropy(rho_A)
        S_B = self.von_neumann_entropy(rho_B)
        S_AB = self.von_neumann_entropy(rho_AB)

        return S_A + S_B - S_AB

    def conditional_entropy(self, rho_AB: np.ndarray, dim_A: int) -> float:
        """Compute S(A|B) = S(AB) - S(B)."""
        dim_B = len(rho_AB) // dim_A

        rho_reshaped = rho_AB.reshape(dim_A, dim_B, dim_A, dim_B)
        rho_B = np.trace(rho_reshaped, axis1=0, axis2=2)

        S_AB = self.von_neumann_entropy(rho_AB)
        S_B = self.von_neumann_entropy(rho_B)

        return S_AB - S_B

    def quantum_discord(self, rho_AB: np.ndarray, dim_A: int = 2) -> float:
        """
        Estimate quantum discord (simplified measurement optimization).
        D(A|B) = I(A:B) - J(A|B) where J is classical correlation.
        """
        I_AB = self.mutual_information(rho_AB, dim_A)

        # Classical correlation: optimize over measurements on B
        dim_B = len(rho_AB) // dim_A

        # Try computational basis measurement
        J_classical = 0
        for b in range(dim_B):
            # Projector onto |b>
            proj_b = np.zeros((dim_B, dim_B))
            proj_b[b, b] = 1
            proj_AB = np.kron(np.eye(dim_A), proj_b)

            # Post-measurement state
            rho_post = proj_AB @ rho_AB @ proj_AB
            prob_b = np.real(np.trace(rho_post))

            if prob_b > 1e-10:
                rho_A_given_b = np.trace(
                    rho_post.reshape(dim_A, dim_B, dim_A, dim_B),
                    axis1=1, axis2=3
                ) / prob_b
                J_classical += prob_b * self.von_neumann_entropy(rho_A_given_b)

        # S(A) - J
        rho_A = np.trace(rho_AB.reshape(dim_A, dim_B, dim_A, dim_B), axis1=1, axis2=3)
        S_A = self.von_neumann_entropy(rho_A)
        J = S_A - J_classical

        return I_AB - J
```
