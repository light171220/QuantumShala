## 1. The Depolarizing Channel

The depolarizing channel is the simplest model of random noise that treats all errors symmetrically.

<Definition term="Depolarizing Channel">
The depolarizing channel with parameter $p \in [0, 1]$ acts as:

$$\mathcal{D}_p(\rho) = (1-p)\rho + p\frac{I}{d}$$

For qubits ($d=2$), equivalently:

$$\mathcal{D}_p(\rho) = (1-p)\rho + \frac{p}{3}(X\rho X + Y\rho Y + Z\rho Z)$$
</Definition>

The depolarizing channel randomly replaces the state with the maximally mixed state with probability $p$.

```python
import numpy as np
from typing import List, Tuple

# Pauli matrices
I = np.eye(2, dtype=complex)
X = np.array([[0, 1], [1, 0]], dtype=complex)
Y = np.array([[0, -1j], [1j, 0]], dtype=complex)
Z = np.array([[1, 0], [0, -1]], dtype=complex)

def depolarizing_kraus(p: float) -> List[np.ndarray]:
    """
    Kraus operators for qubit depolarizing channel.

    Args:
        p: Depolarizing probability (0 to 1)

    Returns:
        List of 4 Kraus operators
    """
    return [
        np.sqrt(1 - 3*p/4) * I,
        np.sqrt(p/4) * X,
        np.sqrt(p/4) * Y,
        np.sqrt(p/4) * Z
    ]

def apply_channel(rho: np.ndarray, kraus_ops: List[np.ndarray]) -> np.ndarray:
    """Apply quantum channel using Kraus operators."""
    return sum(K @ rho @ K.conj().T for K in kraus_ops)

def depolarizing_direct(rho: np.ndarray, p: float) -> np.ndarray:
    """Direct formula: (1-p)ρ + p I/2"""
    return (1 - p) * rho + p * I / 2

# Test depolarizing channel
rho = np.array([[0.8, 0.4], [0.4, 0.2]], dtype=complex)
p = 0.2

kraus = depolarizing_kraus(p)
output_kraus = apply_channel(rho, kraus)
output_direct = depolarizing_direct(rho, p)

print("Depolarizing Channel (p = 0.2)")
print("="*40)
print(f"Input state:\n{rho}")
print(f"\nKraus output:\n{output_kraus.round(4)}")
print(f"Direct formula:\n{output_direct.round(4)}")
print(f"Match: {np.allclose(output_kraus, output_direct)}")
```

### Bloch Sphere Visualization

The depolarizing channel shrinks the Bloch vector uniformly:

<Formula>
$$\vec{r} \to (1-p)\vec{r}$$
</Formula>

```python
def bloch_vector(rho: np.ndarray) -> np.ndarray:
    """Extract Bloch vector from qubit density matrix."""
    rx = np.real(np.trace(rho @ X))
    ry = np.real(np.trace(rho @ Y))
    rz = np.real(np.trace(rho @ Z))
    return np.array([rx, ry, rz])

def depolarizing_bloch_effect(p: float):
    """Analyze depolarizing effect on Bloch sphere."""
    # Test on pure states
    states = {
        "|0>": np.array([[1, 0], [0, 0]], dtype=complex),
        "|1>": np.array([[0, 0], [0, 1]], dtype=complex),
        "|+>": np.array([[0.5, 0.5], [0.5, 0.5]], dtype=complex),
        "|->": np.array([[0.5, -0.5], [-0.5, 0.5]], dtype=complex),
    }

    print(f"Depolarizing (p={p}) Bloch vector contraction:")
    print("-" * 50)
    for name, rho in states.items():
        r_in = bloch_vector(rho)
        rho_out = depolarizing_direct(rho, p)
        r_out = bloch_vector(rho_out)

        len_in = np.linalg.norm(r_in)
        len_out = np.linalg.norm(r_out)
        ratio = len_out / len_in if len_in > 0 else 0

        print(f"{name}: |r| = {len_in:.3f} -> {len_out:.3f} (ratio: {ratio:.3f})")

depolarizing_bloch_effect(0.3)
```

## 2. Amplitude Damping Channel

Amplitude damping models spontaneous emission and T1 relaxation.

<Definition term="Amplitude Damping">
The amplitude damping channel with parameter $\gamma \in [0, 1]$ has Kraus operators:

$$K_0 = \begin{pmatrix} 1 & 0 \\ 0 & \sqrt{1-\gamma} \end{pmatrix}, \quad K_1 = \begin{pmatrix} 0 & \sqrt{\gamma} \\ 0 & 0 \end{pmatrix}$$

This describes decay from $|1\rangle$ to $|0\rangle$ with probability $\gamma$.
</Definition>

The physical interpretation: an excited atom spontaneously emits a photon and decays to the ground state.

```python
def amplitude_damping_kraus(gamma: float) -> List[np.ndarray]:
    """
    Kraus operators for amplitude damping.

    Args:
        gamma: Decay probability (0 to 1)
    """
    K0 = np.array([[1, 0], [0, np.sqrt(1 - gamma)]], dtype=complex)
    K1 = np.array([[0, np.sqrt(gamma)], [0, 0]], dtype=complex)
    return [K0, K1]

def amplitude_damping_analysis():
    """Analyze amplitude damping on various states."""
    gamma = 0.3
    kraus = amplitude_damping_kraus(gamma)

    states = {
        "|0>": np.array([[1, 0], [0, 0]], dtype=complex),
        "|1>": np.array([[0, 0], [0, 1]], dtype=complex),
        "|+>": np.array([[0.5, 0.5], [0.5, 0.5]], dtype=complex),
    }

    print(f"Amplitude Damping (gamma={gamma})")
    print("=" * 50)

    for name, rho in states.items():
        rho_out = apply_channel(rho, kraus)
        p0_in = np.real(rho[0, 0])
        p0_out = np.real(rho_out[0, 0])
        coh_in = np.abs(rho[0, 1])
        coh_out = np.abs(rho_out[0, 1])

        print(f"\n{name}:")
        print(f"  P(|0>): {p0_in:.3f} -> {p0_out:.3f}")
        print(f"  Coherence: {coh_in:.3f} -> {coh_out:.3f}")

amplitude_damping_analysis()
```

### T1 Relaxation Time

<Formula>
$$\gamma = 1 - e^{-t/T_1}$$
</Formula>

For short times: $\gamma \approx t/T_1$.

```python
def gamma_from_t1(t: float, T1: float) -> float:
    """Calculate damping parameter from time and T1."""
    return 1 - np.exp(-t / T1)

def t1_from_gamma(gamma: float, t: float) -> float:
    """Calculate T1 from damping parameter and time."""
    if gamma >= 1:
        return 0
    return -t / np.log(1 - gamma)

# Example: T1 = 100 microseconds, gate time = 100 ns
T1 = 100e-6  # 100 us
t_gate = 100e-9  # 100 ns

gamma = gamma_from_t1(t_gate, T1)
print(f"T1 = {T1*1e6:.1f} us, gate time = {t_gate*1e9:.1f} ns")
print(f"Damping per gate: gamma = {gamma:.6f}")
print(f"Gates before significant decay (~0.37): {int(1/gamma)}")
```

### Bloch Sphere Effect

Amplitude damping contracts the Bloch sphere asymmetrically toward $|0\rangle$:

```python
def amplitude_damping_bloch(gamma: float):
    """Visualize amplitude damping on Bloch sphere."""
    kraus = amplitude_damping_kraus(gamma)

    # Transform Bloch vector
    # r' = (sqrt(1-gamma)*rx, sqrt(1-gamma)*ry, gamma + (1-gamma)*rz)
    print(f"Amplitude Damping Bloch Transformation (gamma={gamma}):")
    print(f"  rx' = sqrt(1-gamma) * rx = {np.sqrt(1-gamma):.3f} * rx")
    print(f"  ry' = sqrt(1-gamma) * ry = {np.sqrt(1-gamma):.3f} * ry")
    print(f"  rz' = gamma + (1-gamma) * rz")
    print(f"  Fixed point: (0, 0, 1) = |0>")

amplitude_damping_bloch(0.3)
```

## 3. Phase Damping Channel

Phase damping (or dephasing) models loss of quantum coherence without energy loss.

<Definition term="Phase Damping">
The phase damping channel with parameter $\lambda \in [0, 1]$ has Kraus operators:

$$K_0 = \begin{pmatrix} 1 & 0 \\ 0 & \sqrt{1-\lambda} \end{pmatrix}, \quad K_1 = \begin{pmatrix} 0 & 0 \\ 0 & \sqrt{\lambda} \end{pmatrix}$$

This destroys off-diagonal elements without changing populations.
</Definition>

```python
def phase_damping_kraus(lambda_: float) -> List[np.ndarray]:
    """
    Kraus operators for phase damping.

    Args:
        lambda_: Dephasing probability (0 to 1)
    """
    K0 = np.array([[1, 0], [0, np.sqrt(1 - lambda_)]], dtype=complex)
    K1 = np.array([[0, 0], [0, np.sqrt(lambda_)]], dtype=complex)
    return [K0, K1]

def phase_damping_analysis():
    """Analyze phase damping effects."""
    lambda_ = 0.4
    kraus = phase_damping_kraus(lambda_)

    # The |+> state is most affected
    rho_plus = np.array([[0.5, 0.5], [0.5, 0.5]], dtype=complex)
    rho_out = apply_channel(rho_plus, kraus)

    print(f"Phase Damping (lambda={lambda_})")
    print("=" * 50)
    print(f"Input |+> state:\n{rho_plus}")
    print(f"\nOutput state:\n{rho_out.round(4)}")
    print(f"\nCoherence: {np.abs(rho_plus[0,1]):.3f} -> {np.abs(rho_out[0,1]):.3f}")
    print(f"Populations unchanged: {np.isclose(rho_plus[0,0], rho_out[0,0])}")

phase_damping_analysis()
```

### T2 Dephasing Time

<Formula>
$$\lambda = 1 - e^{-t/T_\phi}$$

where $\frac{1}{T_2} = \frac{1}{2T_1} + \frac{1}{T_\phi}$
</Formula>

<Warning>
Pure dephasing ($T_\phi$) and amplitude damping ($T_1$) both contribute to the total coherence time $T_2$. Always $T_2 \leq 2T_1$.
</Warning>

```python
def t2_from_t1_tphi(T1: float, T_phi: float) -> float:
    """Calculate T2 from T1 and pure dephasing time."""
    return 1 / (1/(2*T1) + 1/T_phi)

def analyze_coherence_times():
    """Analyze relationship between T1, T2, and T_phi."""
    T1 = 100e-6  # 100 us
    T_phi_values = [1000e-6, 200e-6, 100e-6, 50e-6]  # Various pure dephasing times

    print("Coherence Time Analysis")
    print("=" * 50)
    print(f"T1 = {T1*1e6:.1f} us")
    print(f"Maximum T2 (T_phi -> inf): {2*T1*1e6:.1f} us")
    print()

    for T_phi in T_phi_values:
        T2 = t2_from_t1_tphi(T1, T_phi)
        print(f"T_phi = {T_phi*1e6:.0f} us -> T2 = {T2*1e6:.1f} us")

analyze_coherence_times()
```

## 4. Combined Noise Model

Real qubits experience multiple noise sources simultaneously.

<Definition term="Generalized Amplitude Damping">
At finite temperature, both decay ($|1\rangle \to |0\rangle$) and excitation ($|0\rangle \to |1\rangle$) occur:

$$K_0 = \sqrt{p}\begin{pmatrix} 1 & 0 \\ 0 & \sqrt{1-\gamma} \end{pmatrix}, \quad K_1 = \sqrt{p}\begin{pmatrix} 0 & \sqrt{\gamma} \\ 0 & 0 \end{pmatrix}$$
$$K_2 = \sqrt{1-p}\begin{pmatrix} \sqrt{1-\gamma} & 0 \\ 0 & 1 \end{pmatrix}, \quad K_3 = \sqrt{1-p}\begin{pmatrix} 0 & 0 \\ \sqrt{\gamma} & 0 \end{pmatrix}$$

where $p = 1/(1 + e^{-\hbar\omega/k_B T})$ is the thermal population.
</Definition>

```python
def generalized_amplitude_damping_kraus(gamma: float, p: float) -> List[np.ndarray]:
    """
    Kraus operators for generalized amplitude damping.

    Args:
        gamma: Damping probability
        p: Probability of decay (vs excitation), depends on temperature
    """
    K0 = np.sqrt(p) * np.array([[1, 0], [0, np.sqrt(1-gamma)]], dtype=complex)
    K1 = np.sqrt(p) * np.array([[0, np.sqrt(gamma)], [0, 0]], dtype=complex)
    K2 = np.sqrt(1-p) * np.array([[np.sqrt(1-gamma), 0], [0, 1]], dtype=complex)
    K3 = np.sqrt(1-p) * np.array([[0, 0], [np.sqrt(gamma), 0]], dtype=complex)
    return [K0, K1, K2, K3]

def combined_noise_model(T1: float, T2: float, t: float):
    """
    Create combined amplitude and phase damping model.

    Args:
        T1: Amplitude damping time
        T2: Total dephasing time
        t: Evolution time
    """
    # Calculate parameters
    gamma_ad = 1 - np.exp(-t / T1)

    # Pure dephasing: 1/T_phi = 1/T2 - 1/(2*T1)
    T_phi = 1 / (1/T2 - 1/(2*T1))
    gamma_pd = 1 - np.exp(-t / T_phi)

    # Compose channels (approximately)
    kraus_ad = amplitude_damping_kraus(gamma_ad)
    kraus_pd = phase_damping_kraus(gamma_pd)

    return kraus_ad, kraus_pd, gamma_ad, gamma_pd

# Example: Superconducting qubit parameters
T1 = 50e-6  # 50 us
T2 = 70e-6  # 70 us
t_gate = 50e-9  # 50 ns

kraus_ad, kraus_pd, g_ad, g_pd = combined_noise_model(T1, T2, t_gate)
print(f"Combined Noise Model")
print(f"=" * 50)
print(f"T1 = {T1*1e6:.0f} us, T2 = {T2*1e6:.0f} us")
print(f"Gate time = {t_gate*1e9:.0f} ns")
print(f"Amplitude damping per gate: {g_ad:.2e}")
print(f"Pure dephasing per gate: {g_pd:.2e}")
```

## 5. Channel Composition and Repeated Application

```python
def compose_kraus(kraus1: List[np.ndarray], kraus2: List[np.ndarray]) -> List[np.ndarray]:
    """Compose channels: E2 o E1"""
    return [L @ K for L in kraus2 for K in kraus1]

def repeated_channel(kraus_ops: List[np.ndarray], n: int) -> List[np.ndarray]:
    """Apply channel n times."""
    result = kraus_ops
    for _ in range(n - 1):
        result = compose_kraus(result, kraus_ops)
    return result

def simulate_decay(rho_init: np.ndarray, kraus_ops: List[np.ndarray],
                   n_steps: int) -> List[Tuple[float, float]]:
    """
    Simulate state evolution under repeated channel application.

    Returns:
        List of (population, coherence) at each step
    """
    results = []
    rho = rho_init.copy()

    for _ in range(n_steps + 1):
        pop = np.real(rho[1, 1])  # Excited state population
        coh = np.abs(rho[0, 1])   # Coherence magnitude
        results.append((pop, coh))
        rho = apply_channel(rho, kraus_ops)

    return results

# Simulate T1 and T2 decay
gamma_per_step = 0.02
lambda_per_step = 0.03

kraus_ad = amplitude_damping_kraus(gamma_per_step)
kraus_pd = phase_damping_kraus(lambda_per_step)

# Combined: AD then PD
kraus_combined = compose_kraus(kraus_ad, kraus_pd)

# Start in |+> state
rho_plus = np.array([[0.5, 0.5], [0.5, 0.5]], dtype=complex)
results = simulate_decay(rho_plus, kraus_combined, 100)

print("Decay simulation (starting from |+>):")
print(f"Step | Population | Coherence")
print("-" * 35)
for i in [0, 25, 50, 75, 100]:
    pop, coh = results[i]
    print(f"{i:4d} | {pop:10.4f} | {coh:9.4f}")
```

## Summary

In this lesson, you learned about three fundamental quantum channels:

| Channel | Effect | Kraus Operators | Time Constant |
|---------|--------|-----------------|---------------|
| Depolarizing | Uniform shrinking | $\sqrt{1-3p/4}I$, $\sqrt{p/4}X,Y,Z$ | - |
| Amplitude Damping | Decay to $\|0\rangle$ | Asymmetric contraction | $T_1$ |
| Phase Damping | Coherence loss | Diagonal contraction | $T_\phi$ |

Key relationships:
- $\gamma_{AD} = 1 - e^{-t/T_1}$
- $\lambda_{PD} = 1 - e^{-t/T_\phi}$
- $\frac{1}{T_2} = \frac{1}{2T_1} + \frac{1}{T_\phi}$

These channels form the foundation for understanding noise in real quantum devices.
