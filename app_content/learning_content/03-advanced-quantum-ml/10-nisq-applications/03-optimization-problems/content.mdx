## 1. QAOA Overview

The Quantum Approximate Optimization Algorithm (QAOA) is designed for combinatorial optimization.

<Definition term="QAOA">
A variational quantum algorithm that alternates between a problem Hamiltonian and a mixer Hamiltonian to find approximate solutions to combinatorial optimization problems.
</Definition>

<Formula>
QAOA ansatz with p layers:
$$|\gamma, \beta\rangle = \prod_{l=1}^{p} e^{-i\beta_l H_M} e^{-i\gamma_l H_C} |+\rangle^{\otimes n}$$
where $H_C$ encodes the cost function and $H_M$ is the mixer.
</Formula>

## 2. Problem Encoding: MaxCut

MaxCut is the canonical QAOA benchmark problem.

```python
import numpy as np

class MaxCutQAOA:
    """
    QAOA implementation for MaxCut problem.
    """

    def __init__(self, edges, n_nodes):
        """
        Args:
            edges: List of (i, j) tuples representing graph edges
            n_nodes: Number of nodes in graph
        """
        self.edges = edges
        self.n_nodes = n_nodes
        self.cost_hamiltonian = self._build_cost_hamiltonian()

    def _build_cost_hamiltonian(self):
        """
        Build MaxCut cost Hamiltonian.

        For each edge (i,j): C_ij = (1 - Z_i Z_j) / 2
        Total: H_C = Σ_{ij} (1 - Z_i Z_j) / 2
        """
        hamiltonian = []

        for i, j in self.edges:
            # Constant term: 1/2 per edge
            # ZZ term: -1/2 Z_i Z_j

            # Build Pauli string
            pauli = ['I'] * self.n_nodes
            pauli[i] = 'Z'
            pauli[j] = 'Z'

            hamiltonian.append((-0.5, ''.join(pauli)))

        # Constant offset
        hamiltonian.append((len(self.edges) / 2, 'I' * self.n_nodes))

        return hamiltonian

    def classical_cost(self, bitstring):
        """
        Compute MaxCut cost for a bitstring solution.
        """
        cut = 0
        for i, j in self.edges:
            if bitstring[i] != bitstring[j]:
                cut += 1
        return cut

    def evaluate_solution(self, bitstring):
        """
        Evaluate and display solution quality.
        """
        cost = self.classical_cost(bitstring)
        max_possible = len(self.edges)
        ratio = cost / max_possible if max_possible > 0 else 0

        return {
            'bitstring': bitstring,
            'cut_value': cost,
            'max_edges': max_possible,
            'approximation_ratio': ratio
        }

          f"(ratio = {result['approximation_ratio']:.2f})")
```

## 3. QAOA Circuit Simulation

Simulate the QAOA circuit and optimization.

```python
import numpy as np

class QAOASimulator:
    """
    Simulate QAOA circuit.
    """

    def __init__(self, cost_hamiltonian, n_qubits, p=1):
        self.hamiltonian = cost_hamiltonian
        self.n_qubits = n_qubits
        self.p = p
        self.dim = 2 ** n_qubits

    def apply_cost_unitary(self, state, gamma):
        """
        Apply exp(-i γ H_C).
        Simplified: apply phase based on cost of each basis state.
        """
        new_state = state.copy()
        for idx in range(self.dim):
            bitstring = format(idx, f'0{self.n_qubits}b')
            cost = self._compute_cost(bitstring)
            phase = np.exp(-1j * gamma * cost)
            new_state[idx] *= phase
        return new_state

    def apply_mixer_unitary(self, state, beta):
        """
        Apply exp(-i β H_M) where H_M = Σ X_i.
        For each qubit, apply Rx(2β).
        """
        # Rx(2β) = exp(-i β X)
        cos_b = np.cos(beta)
        sin_b = np.sin(beta)

        for q in range(self.n_qubits):
            new_state = np.zeros(self.dim, dtype=complex)
            for idx in range(self.dim):
                # Bit flip on qubit q
                flipped_idx = idx ^ (1 << q)

                # Apply Rx
                bit_val = (idx >> q) & 1
                if bit_val == 0:
                    new_state[idx] += cos_b * state[idx] - 1j * sin_b * state[flipped_idx]
                else:
                    new_state[idx] += -1j * sin_b * state[flipped_idx] + cos_b * state[idx]

            state = new_state
        return state

    def _compute_cost(self, bitstring):
        """Compute cost for a bitstring."""
        cost = 0.0
        for coeff, pauli in self.hamiltonian:
            term = coeff
            for q, p in enumerate(pauli):
                if p == 'Z':
                    bit = int(bitstring[q])
                    term *= (1 - 2 * bit)  # Z eigenvalue
            cost += term
        return cost

    def run_qaoa(self, gammas, betas):
        """
        Run QAOA circuit with given parameters.
        """
        # Initialize to |+⟩^n
        state = np.ones(self.dim, dtype=complex) / np.sqrt(self.dim)

        # Apply p layers
        for gamma, beta in zip(gammas, betas):
            state = self.apply_cost_unitary(state, gamma)
            state = self.apply_mixer_unitary(state, beta)

        return state

    def compute_expectation(self, gammas, betas):
        """Compute ⟨H_C⟩."""
        state = self.run_qaoa(gammas, betas)
        expectation = 0.0
        for idx in range(self.dim):
            bitstring = format(idx, f'0{self.n_qubits}b')
            prob = np.abs(state[idx]) ** 2
            cost = self._compute_cost(bitstring)
            expectation += prob * cost
        return expectation

    def sample(self, gammas, betas, n_samples=100):
        """Sample solutions from QAOA state."""
        state = self.run_qaoa(gammas, betas)
        probs = np.abs(state) ** 2
        samples = np.random.choice(self.dim, size=n_samples, p=probs)
        return [format(s, f'0{self.n_qubits}b') for s in samples]

# Demonstration
print("QAOA Simulation:")
print("=" * 50)

np.random.seed(42)

# MaxCut on 4-node graph
edges = [(0, 1), (1, 2), (2, 3), (0, 3), (0, 2)]
maxcut = MaxCutQAOA(edges, 4)

qaoa = QAOASimulator(maxcut.cost_hamiltonian, n_qubits=4, p=1)

# Test different parameters
print("\nParameter sweep (p=1):")
best_cost = float('inf')
best_params = None

for gamma in np.linspace(0, np.pi, 5):
    for beta in np.linspace(0, np.pi/2, 5):
        cost = qaoa.compute_expectation([gamma], [beta])
        if cost < best_cost:
            best_cost = cost
            best_params = (gamma, beta)

print(f"Best γ={best_params[0]:.3f}, β={best_params[1]:.3f}")
print(f"Expected cost: {best_cost:.4f}")

# Sample solutions
samples = qaoa.sample([best_params[0]], [best_params[1]], n_samples=20)
print(f"\nSampled solutions:")
for s in set(samples):
    cut = maxcut.classical_cost(s)
    count = samples.count(s)
    print(f"  {s}: cut={cut}, count={count}")
```

<Warning>
QAOA performance depends heavily on parameter choice. Grid search works for small p, but advanced methods are needed for larger p.
</Warning>

## Summary

QAOA offers a quantum approach to combinatorial optimization:

1. **Structure**: Alternate cost and mixer unitaries with tunable parameters
2. **Encoding**: Express optimization objectives as Hamiltonians
3. **Applications**: MaxCut, SAT, TSP, portfolio optimization, etc.
4. **Current Status**: Competitive with classical for some small instances
5. **Limitations**: Circuit depth, parameter optimization, noise
6. **Improvements**: Warm-starting, recursive methods, custom mixers

While practical advantage is not yet demonstrated, QAOA remains an active research area.
