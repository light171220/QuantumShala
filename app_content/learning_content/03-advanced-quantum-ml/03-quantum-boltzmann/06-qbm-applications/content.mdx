## 1. Generative Modeling with QBMs

Quantum Boltzmann machines excel at generative modeling by learning probability distributions over data.

<Definition term="QBM Generative Model">
A QBM generative model learns to represent P(x) where x is the data. Samples are generated by:
1. Preparing the trained Gibbs state
2. Measuring visible qubits
3. Post-processing measurements to obtain data samples

Quantum coherence enables representing distributions impossible for classical RBMs.
</Definition>

```python
import numpy as np
from scipy.linalg import expm
from typing import Tuple, Dict, List

class QBMGenerativeModel:
    """
    Quantum Boltzmann Machine for generative modeling.
    """

    def __init__(
        self,
        n_visible: int,
        n_hidden: int,
        beta: float = 1.0
    ):
        self.n_visible = n_visible
        self.n_hidden = n_hidden
        self.n_total = n_visible + n_hidden
        self.dim = 2 ** self.n_total
        self.beta = beta

        # Pauli matrices
        self.I = np.eye(2, dtype=complex)
        self.X = np.array([[0, 1], [1, 0]], dtype=complex)
        self.Z = np.array([[1, 0], [0, -1]], dtype=complex)

        # Model parameters
        self.params = self._initialize_params()

    def _initialize_params(self) -> Dict[str, np.ndarray]:
        """Initialize model parameters."""
        return {
            'a': np.random.randn(self.n_visible) * 0.1,
            'b': np.random.randn(self.n_hidden) * 0.1,
            'W': np.random.randn(self.n_visible, self.n_hidden) * 0.1,
            'gamma': np.abs(np.random.randn()) * 0.1
        }

    def _single_op(self, op: np.ndarray, qubit: int) -> np.ndarray:
        """Create single-qubit operator."""
        result = np.array([[1]], dtype=complex)
        for i in range(self.n_total):
            result = np.kron(result, op if i == qubit else self.I)
        return result

    def build_hamiltonian(self) -> np.ndarray:
        """Build QBM Hamiltonian."""
        H = np.zeros((self.dim, self.dim), dtype=complex)

        # Visible biases
        for i in range(self.n_visible):
            H -= self.params['a'][i] * self._single_op(self.Z, i)

        # Hidden biases
        for j in range(self.n_hidden):
            H -= self.params['b'][j] * self._single_op(self.Z, self.n_visible + j)

        # Couplings
        for i in range(self.n_visible):
            for j in range(self.n_hidden):
                ZZ = self._single_op(self.Z, i) @ self._single_op(
                    self.Z, self.n_visible + j
                )
                H -= self.params['W'][i, j] * ZZ

        # Transverse field
        for k in range(self.n_total):
            H -= self.params['gamma'] * self._single_op(self.X, k)

        return H

    def compute_gibbs_state(self) -> np.ndarray:
        """Compute Gibbs state density matrix."""
        H = self.build_hamiltonian()
        exp_H = expm(-self.beta * H)
        return exp_H / np.trace(exp_H)

    def visible_distribution(self) -> np.ndarray:
        """Get probability distribution over visible configurations."""
        rho = self.compute_gibbs_state()

        dim_v = 2 ** self.n_visible
        dim_h = 2 ** self.n_hidden

        rho_reshaped = rho.reshape(dim_v, dim_h, dim_v, dim_h)
        rho_v = np.einsum('ijik->jk', rho_reshaped)

        return np.real(np.diag(rho_v))

    def generate_samples(self, n_samples: int) -> np.ndarray:
        """Generate samples from the model."""
        probs = self.visible_distribution()
        probs = probs / np.sum(probs)

        indices = np.random.choice(len(probs), size=n_samples, p=probs)

        samples = np.array([
            [(idx >> i) & 1 for i in range(self.n_visible)]
            for idx in indices
        ])

        return samples

    def train(
        self,
        data: np.ndarray,
        n_epochs: int = 100,
        learning_rate: float = 0.1
    ):
        """Train the generative model."""
        for epoch in range(n_epochs):
            # Compute gradients
            grads = self._compute_gradients(data)

            # Update parameters
            for key in ['a', 'b', 'W']:
                self.params[key] -= learning_rate * grads[key]

            # Log progress
            if epoch % 20 == 0:
                nll = self._compute_nll(data)
                print(f"Epoch {epoch}: NLL = {nll:.4f}")

    def _compute_gradients(self, data: np.ndarray) -> Dict[str, np.ndarray]:
        """Compute training gradients."""
        H = self.build_hamiltonian()
        rho = self.compute_gibbs_state()

        grad_a = np.zeros(self.n_visible)
        grad_b = np.zeros(self.n_hidden)
        grad_W = np.zeros((self.n_visible, self.n_hidden))

        # Model expectations (negative phase)
        for i in range(self.n_visible):
            grad_a[i] = -self.beta * np.real(
                np.trace(self._single_op(self.Z, i) @ rho)
            )

        for j in range(self.n_hidden):
            grad_b[j] = -self.beta * np.real(
                np.trace(self._single_op(self.Z, self.n_visible + j) @ rho)
            )

        for i in range(self.n_visible):
            for j in range(self.n_hidden):
                ZZ = self._single_op(self.Z, i) @ self._single_op(
                    self.Z, self.n_visible + j
                )
                grad_W[i, j] = -self.beta * np.real(np.trace(ZZ @ rho))

        # Data expectations (positive phase)
        for v in data:
            for i in range(self.n_visible):
                grad_a[i] += self.beta * (1 - 2 * v[i]) / len(data)

        return {'a': grad_a, 'b': grad_b, 'W': grad_W}

    def _compute_nll(self, data: np.ndarray) -> float:
        """Compute negative log-likelihood."""
        probs = self.visible_distribution()

        nll = 0.0
        for v in data:
            idx = int(sum(v[i] * 2**i for i in range(self.n_visible)))
            nll -= np.log(max(probs[idx], 1e-10))

        return nll / len(data)
```

## 2. Combinatorial Optimization

QBMs can solve optimization problems by encoding cost functions as Hamiltonians.

<Definition term="QBM for Optimization">
A combinatorial optimization problem min_x f(x) can be solved by:
1. Encoding f(x) as a QBM Hamiltonian H
2. Sampling from low-temperature Gibbs states
3. Returning the lowest-energy sample as the solution

Quantum tunneling helps escape local minima that trap classical optimizers.
</Definition>

<Tip>
Low temperature (high β) concentrates the Gibbs distribution on low-energy configurations, while quantum fluctuations (high γ) enable exploration.
</Tip>

```python
class QBMOptimizer:
    """
    QBM-based combinatorial optimizer.
    """

    def __init__(
        self,
        n_variables: int,
        beta: float = 10.0,
        gamma: float = 0.3
    ):
        self.n_variables = n_variables
        self.dim = 2 ** n_variables
        self.beta = beta
        self.gamma = gamma

        self.I = np.eye(2, dtype=complex)
        self.X = np.array([[0, 1], [1, 0]], dtype=complex)
        self.Z = np.array([[1, 0], [0, -1]], dtype=complex)

    def _single_op(self, op: np.ndarray, qubit: int) -> np.ndarray:
        result = np.array([[1]], dtype=complex)
        for i in range(self.n_variables):
            result = np.kron(result, op if i == qubit else self.I)
        return result

    def encode_qubo(
        self,
        Q: np.ndarray,
        linear: np.ndarray = None
    ) -> np.ndarray:
        """
        Encode QUBO problem as Hamiltonian.

        QUBO: min x^T Q x + c^T x
        Mapping: x_i ∈ {0,1} → (1-Z_i)/2

        Args:
            Q: Quadratic coefficients (n x n)
            linear: Linear coefficients (n,)

        Returns:
            Hamiltonian matrix
        """
        H = np.zeros((self.dim, self.dim), dtype=complex)

        if linear is None:
            linear = np.zeros(self.n_variables)

        # Linear terms
        for i in range(self.n_variables):
            coeff = linear[i] + np.sum(Q[i, :]) / 2
            H += coeff / 2 * self._single_op(self.Z, i)
            H -= coeff / 2 * np.eye(self.dim)

        # Quadratic terms
        for i in range(self.n_variables):
            for j in range(i + 1, self.n_variables):
                coeff = (Q[i, j] + Q[j, i]) / 4
                ZZ = self._single_op(self.Z, i) @ self._single_op(self.Z, j)
                H += coeff * ZZ

        # Transverse field for quantum fluctuations
        for k in range(self.n_variables):
            H -= self.gamma * self._single_op(self.X, k)

        return H

    def encode_maxcut(self, adjacency: np.ndarray) -> np.ndarray:
        """
        Encode MaxCut problem.

        MaxCut: max Σ_{ij} w_ij * (1 - x_i * x_j) / 2
        """
        H = np.zeros((self.dim, self.dim), dtype=complex)

        for i in range(self.n_variables):
            for j in range(i + 1, self.n_variables):
                if adjacency[i, j] != 0:
                    ZZ = self._single_op(self.Z, i) @ self._single_op(self.Z, j)
                    H += adjacency[i, j] / 2 * ZZ

        # Transverse field
        for k in range(self.n_variables):
            H -= self.gamma * self._single_op(self.X, k)

        return H

    def solve(
        self,
        H: np.ndarray,
        n_samples: int = 100,
        annealing_schedule: bool = True
    ) -> Tuple[np.ndarray, float]:
        """
        Find minimum energy configuration.

        Returns:
            (best_configuration, best_energy)
        """
        if annealing_schedule:
            return self._solve_annealing(H, n_samples)
        else:
            return self._solve_direct(H, n_samples)

    def _solve_direct(
        self,
        H: np.ndarray,
        n_samples: int
    ) -> Tuple[np.ndarray, float]:
        """Direct sampling from Gibbs state."""
        exp_H = expm(-self.beta * H)
        rho = exp_H / np.trace(exp_H)

        probs = np.real(np.diag(rho))
        probs = probs / np.sum(probs)

        # Sample configurations
        indices = np.random.choice(self.dim, size=n_samples, p=probs)

        # Find best
        energies = np.real(np.diag(H))
        best_idx = min(indices, key=lambda x: energies[x])
        best_config = np.array([
            (best_idx >> i) & 1 for i in range(self.n_variables)
        ])

        return best_config, energies[best_idx]

    def _solve_annealing(
        self,
        H: np.ndarray,
        n_samples: int,
        n_steps: int = 50
    ) -> Tuple[np.ndarray, float]:
        """Solve using simulated quantum annealing."""
        # Build driver
        H0 = np.zeros((self.dim, self.dim), dtype=complex)
        for k in range(self.n_variables):
            H0 -= self._single_op(self.X, k)

        # Remove transverse field from problem Hamiltonian
        H_problem = H.copy()
        for k in range(self.n_variables):
            H_problem += self.gamma * self._single_op(self.X, k)

        # Annealing
        psi = np.ones(self.dim, dtype=complex) / np.sqrt(self.dim)

        for step in range(n_steps):
            s = step / n_steps
            H_t = (1 - s) * H0 + s * H_problem

            # Evolution step
            dt = 0.1
            U = expm(-1j * dt * H_t)
            psi = U @ psi

            # Add thermal noise
            if s > 0.5:
                noise = np.random.randn(self.dim) * 0.01
                psi = psi + noise * (1j * noise)
                psi = psi / np.linalg.norm(psi)

        # Sample from final state
        probs = np.abs(psi) ** 2
        indices = np.random.choice(self.dim, size=n_samples, p=probs)

        # Find best
        energies = np.real(np.diag(H_problem))
        best_idx = min(indices, key=lambda x: energies[x])
        best_config = np.array([
            (best_idx >> i) & 1 for i in range(self.n_variables)
        ])

        return best_config, energies[best_idx]
```

## 3. Quantum Chemistry Simulations

QBMs can simulate molecular systems by representing electronic states.

<Definition term="QBM for Molecular Simulation">
Molecular Hamiltonians in second quantization can be mapped to qubit Hamiltonians:

H = Σ_{pq} h_pq a†_p a_q + Σ_{pqrs} g_pqrs a†_p a†_q a_r a_s

Using Jordan-Wigner or Bravyi-Kitaev transformations, this becomes a qubit Hamiltonian suitable for QBM representation.
</Definition>

```python
class QBMMolecularSimulator:
    """
    QBM for molecular quantum chemistry simulations.
    """

    def __init__(self, n_orbitals: int, n_electrons: int):
        self.n_orbitals = n_orbitals
        self.n_electrons = n_electrons
        self.n_qubits = 2 * n_orbitals  # Spin orbitals
        self.dim = 2 ** self.n_qubits

        self.I = np.eye(2, dtype=complex)
        self.X = np.array([[0, 1], [1, 0]], dtype=complex)
        self.Y = np.array([[0, -1j], [1j, 0]], dtype=complex)
        self.Z = np.array([[1, 0], [0, -1]], dtype=complex)

    def _single_op(self, op: np.ndarray, qubit: int) -> np.ndarray:
        result = np.array([[1]], dtype=complex)
        for i in range(self.n_qubits):
            result = np.kron(result, op if i == qubit else self.I)
        return result

    def jordan_wigner_transform(
        self,
        one_body: np.ndarray,
        two_body: np.ndarray
    ) -> np.ndarray:
        """
        Apply Jordan-Wigner transformation to molecular integrals.

        Args:
            one_body: One-electron integrals (n_orbitals, n_orbitals)
            two_body: Two-electron integrals (n_orbitals, n_orbitals, n_orbitals, n_orbitals)

        Returns:
            Qubit Hamiltonian
        """
        H = np.zeros((self.dim, self.dim), dtype=complex)

        # One-body terms: h_pq a†_p a_q
        for p in range(self.n_orbitals):
            for q in range(self.n_orbitals):
                for spin in range(2):  # Alpha and beta
                    i = 2 * p + spin
                    j = 2 * q + spin

                    if i == j:
                        # Number operator
                        H += one_body[p, q] * 0.5 * (
                            np.eye(self.dim) - self._single_op(self.Z, i)
                        )
                    else:
                        # Hopping term (simplified)
                        H += one_body[p, q] * 0.25 * self._hopping_term(i, j)

        # Two-body terms (simplified for demonstration)
        for p in range(self.n_orbitals):
            for q in range(self.n_orbitals):
                for spin1 in range(2):
                    for spin2 in range(2):
                        i = 2 * p + spin1
                        j = 2 * q + spin2
                        if i != j and two_body[p, p, q, q] != 0:
                            ni = 0.5 * (np.eye(self.dim) - self._single_op(self.Z, i))
                            nj = 0.5 * (np.eye(self.dim) - self._single_op(self.Z, j))
                            H += 0.5 * two_body[p, p, q, q] * ni @ nj

        return H

    def _hopping_term(self, i: int, j: int) -> np.ndarray:
        """Create hopping term a†_i a_j + h.c. using JW."""
        if i > j:
            i, j = j, i

        # Jordan-Wigner string
        term = self._single_op(self.X, i) @ self._single_op(self.X, j)
        term += self._single_op(self.Y, i) @ self._single_op(self.Y, j)

        # Include Z string for i < k < j
        for k in range(i + 1, j):
            term = term @ self._single_op(self.Z, k)

        return term

    def prepare_hf_state(self) -> np.ndarray:
        """Prepare Hartree-Fock initial state."""
        psi = np.zeros(self.dim, dtype=complex)

        # Occupy lowest n_electrons spin orbitals
        hf_config = 0
        for i in range(self.n_electrons):
            hf_config |= (1 << i)

        psi[hf_config] = 1.0
        return psi

    def compute_ground_state_energy(
        self,
        H: np.ndarray,
        beta: float = 10.0,
        gamma: float = 0.2
    ) -> Tuple[float, np.ndarray]:
        """
        Estimate ground state energy using QBM.

        Returns:
            (energy, ground_state_approximation)
        """
        # Add transverse field
        H_qbm = H.copy()
        for k in range(self.n_qubits):
            H_qbm -= gamma * self._single_op(self.X, k)

        # Compute Gibbs state at low temperature
        exp_H = expm(-beta * H_qbm)
        rho = exp_H / np.trace(exp_H)

        # Energy expectation
        energy = np.real(np.trace(H @ rho))

        # Approximate ground state from largest probability
        probs = np.real(np.diag(rho))
        gs_idx = np.argmax(probs)
        gs = np.zeros(self.dim, dtype=complex)
        gs[gs_idx] = 1.0

        return energy, gs

    def compute_correlation_functions(
        self,
        H: np.ndarray,
        beta: float = 5.0
    ) -> Dict[str, np.ndarray]:
        """
        Compute electronic correlation functions.
        """
        exp_H = expm(-beta * H)
        rho = exp_H / np.trace(exp_H)

        # Occupation numbers ⟨n_i⟩
        occupations = np.zeros(self.n_qubits)
        for i in range(self.n_qubits):
            n_i = 0.5 * (np.eye(self.dim) - self._single_op(self.Z, i))
            occupations[i] = np.real(np.trace(n_i @ rho))

        # Pair correlations ⟨n_i n_j⟩
        pair_corr = np.zeros((self.n_qubits, self.n_qubits))
        for i in range(self.n_qubits):
            n_i = 0.5 * (np.eye(self.dim) - self._single_op(self.Z, i))
            for j in range(self.n_qubits):
                n_j = 0.5 * (np.eye(self.dim) - self._single_op(self.Z, j))
                pair_corr[i, j] = np.real(np.trace(n_i @ n_j @ rho))

        return {
            'occupations': occupations,
            'pair_correlations': pair_corr
        }
```

## 4. QBM Classifiers

QBMs can be used for discriminative tasks like classification.

<Definition term="QBM Classifier">
A QBM classifier uses:
1. Visible units for input features
2. Hidden units for learned representations
3. Label units for class prediction

Classification: P(y|x) ∝ exp(-βE(x, y))
</Definition>

```python
class QBMClassifier:
    """
    QBM-based classifier for supervised learning.
    """

    def __init__(
        self,
        n_features: int,
        n_hidden: int,
        n_classes: int,
        beta: float = 1.0
    ):
        self.n_features = n_features
        self.n_hidden = n_hidden
        self.n_classes = n_classes

        # Binary encoding of classes
        self.n_label_bits = int(np.ceil(np.log2(max(n_classes, 2))))

        self.n_visible = n_features + self.n_label_bits
        self.n_total = self.n_visible + n_hidden
        self.dim = 2 ** self.n_total
        self.beta = beta

        self.I = np.eye(2, dtype=complex)
        self.X = np.array([[0, 1], [1, 0]], dtype=complex)
        self.Z = np.array([[1, 0], [0, -1]], dtype=complex)

        self.params = self._initialize_params()

    def _initialize_params(self) -> Dict[str, np.ndarray]:
        return {
            'a': np.random.randn(self.n_visible) * 0.1,
            'b': np.random.randn(self.n_hidden) * 0.1,
            'W': np.random.randn(self.n_visible, self.n_hidden) * 0.1,
            'gamma': 0.1
        }

    def _single_op(self, op: np.ndarray, qubit: int) -> np.ndarray:
        result = np.array([[1]], dtype=complex)
        for i in range(self.n_total):
            result = np.kron(result, op if i == qubit else self.I)
        return result

    def _encode_label(self, label: int) -> np.ndarray:
        """Binary encode class label."""
        return np.array([(label >> i) & 1 for i in range(self.n_label_bits)])

    def _decode_label(self, bits: np.ndarray) -> int:
        """Decode binary label to class."""
        return int(sum(bits[i] * 2**i for i in range(len(bits))))

    def build_hamiltonian(self) -> np.ndarray:
        """Build classifier Hamiltonian."""
        H = np.zeros((self.dim, self.dim), dtype=complex)

        for i in range(self.n_visible):
            H -= self.params['a'][i] * self._single_op(self.Z, i)

        for j in range(self.n_hidden):
            H -= self.params['b'][j] * self._single_op(
                self.Z, self.n_visible + j
            )

        for i in range(self.n_visible):
            for j in range(self.n_hidden):
                ZZ = self._single_op(self.Z, i) @ self._single_op(
                    self.Z, self.n_visible + j
                )
                H -= self.params['W'][i, j] * ZZ

        for k in range(self.n_total):
            H -= self.params['gamma'] * self._single_op(self.X, k)

        return H

    def predict_proba(self, x: np.ndarray) -> np.ndarray:
        """
        Predict class probabilities for input x.

        Args:
            x: Binary input features

        Returns:
            Class probabilities
        """
        H = self.build_hamiltonian()
        exp_H = expm(-self.beta * H)
        rho = exp_H / np.trace(exp_H)

        class_probs = np.zeros(self.n_classes)

        for c in range(self.n_classes):
            label_bits = self._encode_label(c)

            # Create projector onto (x, label)
            observed = {}
            for i in range(self.n_features):
                observed[i] = int(x[i])
            for i in range(self.n_label_bits):
                observed[self.n_features + i] = int(label_bits[i])

            proj = self._create_projector(observed)
            class_probs[c] = np.real(np.trace(proj @ rho @ proj))

        # Normalize
        class_probs = class_probs / (np.sum(class_probs) + 1e-10)

        return class_probs

    def _create_projector(self, observed: Dict[int, int]) -> np.ndarray:
        """Create projector onto observed values."""
        proj = np.eye(self.dim, dtype=complex)

        for qubit, value in observed.items():
            P_single = np.zeros((2, 2), dtype=complex)
            P_single[value, value] = 1.0

            P_full = np.array([[1]], dtype=complex)
            for i in range(self.n_total):
                if i == qubit:
                    P_full = np.kron(P_full, P_single)
                else:
                    P_full = np.kron(P_full, self.I)

            proj = proj @ P_full

        return proj

    def predict(self, x: np.ndarray) -> int:
        """Predict class for input x."""
        probs = self.predict_proba(x)
        return np.argmax(probs)

    def train(
        self,
        X: np.ndarray,
        y: np.ndarray,
        n_epochs: int = 50,
        learning_rate: float = 0.1
    ):
        """Train the classifier."""
        for epoch in range(n_epochs):
            total_loss = 0.0

            for x, label in zip(X, y):
                # Compute gradients for this example
                grads = self._compute_gradients(x, label)

                # Update parameters
                for key in ['a', 'b', 'W']:
                    self.params[key] -= learning_rate * grads[key]

                # Track loss
                probs = self.predict_proba(x)
                total_loss -= np.log(max(probs[label], 1e-10))

            if epoch % 10 == 0:
                print(f"Epoch {epoch}: Loss = {total_loss / len(X):.4f}")

    def _compute_gradients(
        self,
        x: np.ndarray,
        label: int
    ) -> Dict[str, np.ndarray]:
        """Compute gradients for single example."""
        # Simplified gradient computation
        grad_a = np.zeros(self.n_visible)
        grad_b = np.zeros(self.n_hidden)
        grad_W = np.zeros((self.n_visible, self.n_hidden))

        # Finite difference approximation
        eps = 1e-4
        base_prob = self.predict_proba(x)[label]

        for i in range(self.n_visible):
            self.params['a'][i] += eps
            new_prob = self.predict_proba(x)[label]
            grad_a[i] = -(new_prob - base_prob) / eps
            self.params['a'][i] -= eps

        return {'a': grad_a, 'b': grad_b, 'W': grad_W}

    def evaluate(self, X: np.ndarray, y: np.ndarray) -> float:
        """Compute classification accuracy."""
        correct = 0
        for x, label in zip(X, y):
            if self.predict(x) == label:
                correct += 1
        return correct / len(X)
```

## 5. Deep Boltzmann Machines with Quantum Layers

QBMs can be combined with classical deep learning architectures.

<Definition term="Deep Quantum Boltzmann Machine">
A deep QBM stacks multiple layers of QBMs:
- Input layer: Classical visible units
- Quantum layers: QBM hidden layers with quantum connections
- Output layer: Task-specific (classification, generation)

This combines depth for representation power with quantum effects.
</Definition>

```python
class DeepQBM:
    """
    Deep Boltzmann Machine with quantum layers.
    """

    def __init__(
        self,
        layer_sizes: List[int],
        quantum_layers: List[bool],
        beta: float = 1.0
    ):
        """
        Args:
            layer_sizes: Number of units in each layer
            quantum_layers: Whether each inter-layer connection is quantum
        """
        self.layer_sizes = layer_sizes
        self.quantum_layers = quantum_layers
        self.n_layers = len(layer_sizes)
        self.beta = beta

        self.I = np.eye(2, dtype=complex)
        self.X = np.array([[0, 1], [1, 0]], dtype=complex)
        self.Z = np.array([[1, 0], [0, -1]], dtype=complex)

        # Initialize parameters for each layer connection
        self.params = self._initialize_params()

    def _initialize_params(self) -> List[Dict[str, np.ndarray]]:
        """Initialize parameters for all layers."""
        params = []

        for l in range(self.n_layers - 1):
            n_in = self.layer_sizes[l]
            n_out = self.layer_sizes[l + 1]

            layer_params = {
                'W': np.random.randn(n_in, n_out) * 0.1,
                'b_in': np.random.randn(n_in) * 0.1,
                'b_out': np.random.randn(n_out) * 0.1,
            }

            if self.quantum_layers[l]:
                layer_params['gamma'] = 0.1

            params.append(layer_params)

        return params

    def forward_pass(
        self,
        x: np.ndarray,
        sample: bool = True
    ) -> List[np.ndarray]:
        """
        Forward pass through all layers.

        Args:
            x: Input visible units
            sample: Whether to sample (True) or use probabilities (False)

        Returns:
            Activations for all layers
        """
        activations = [x]
        current = x

        for l in range(self.n_layers - 1):
            if self.quantum_layers[l]:
                next_layer = self._quantum_layer_forward(
                    current, self.params[l]
                )
            else:
                next_layer = self._classical_layer_forward(
                    current, self.params[l]
                )

            if sample:
                next_layer = (np.random.random(len(next_layer)) < next_layer).astype(float)

            activations.append(next_layer)
            current = next_layer

        return activations

    def _classical_layer_forward(
        self,
        h: np.ndarray,
        params: Dict[str, np.ndarray]
    ) -> np.ndarray:
        """Classical RBM-style forward pass."""
        activation = params['b_out'] + h @ params['W']
        return 1 / (1 + np.exp(-self.beta * activation))

    def _quantum_layer_forward(
        self,
        h: np.ndarray,
        params: Dict[str, np.ndarray]
    ) -> np.ndarray:
        """Quantum layer forward pass."""
        n_in = len(h)
        n_out = len(params['b_out'])
        n_total = n_in + n_out
        dim = 2 ** n_total

        # Build layer Hamiltonian
        H = self._build_layer_hamiltonian(h, params)

        # Compute Gibbs state
        exp_H = expm(-self.beta * H)
        rho = exp_H / np.trace(exp_H)

        # Trace out input layer to get output probabilities
        dim_in = 2 ** n_in
        dim_out = 2 ** n_out

        rho_reshaped = rho.reshape(dim_in, dim_out, dim_in, dim_out)
        rho_out = np.einsum('ijik->jk', rho_reshaped)

        # Compute marginal probabilities for each output unit
        output_probs = np.zeros(n_out)
        for i in range(n_out):
            # Probability that output qubit i is 1
            output_probs[i] = self._marginal_prob(rho_out, i, n_out)

        return output_probs

    def _build_layer_hamiltonian(
        self,
        h_in: np.ndarray,
        params: Dict[str, np.ndarray]
    ) -> np.ndarray:
        """Build Hamiltonian for one layer connection."""
        n_in = len(h_in)
        n_out = len(params['b_out'])
        n_total = n_in + n_out
        dim = 2 ** n_total

        H = np.zeros((dim, dim), dtype=complex)

        # Input biases (clamped)
        for i in range(n_in):
            val = 1 - 2 * h_in[i]  # Map 0/1 to +1/-1
            H -= params['b_in'][i] * val * np.eye(dim)

        # Output biases
        for j in range(n_out):
            H -= params['b_out'][j] * self._single_op_layer(
                self.Z, n_in + j, n_total
            )

        # Couplings
        for i in range(n_in):
            for j in range(n_out):
                val = 1 - 2 * h_in[i]
                H -= params['W'][i, j] * val * self._single_op_layer(
                    self.Z, n_in + j, n_total
                )

        # Quantum fluctuations
        if 'gamma' in params:
            for k in range(n_in, n_total):  # Only on output layer
                H -= params['gamma'] * self._single_op_layer(self.X, k, n_total)

        return H

    def _single_op_layer(
        self,
        op: np.ndarray,
        qubit: int,
        n_qubits: int
    ) -> np.ndarray:
        """Create single-qubit operator for layer."""
        result = np.array([[1]], dtype=complex)
        for i in range(n_qubits):
            result = np.kron(result, op if i == qubit else self.I)
        return result

    def _marginal_prob(
        self,
        rho: np.ndarray,
        qubit: int,
        n_qubits: int
    ) -> float:
        """Compute probability that qubit is in state 1."""
        dim = 2 ** n_qubits
        prob = 0.0

        for idx in range(dim):
            if (idx >> qubit) & 1:
                prob += np.real(rho[idx, idx])

        return prob

    def extract_features(self, x: np.ndarray) -> np.ndarray:
        """Extract features from top hidden layer."""
        activations = self.forward_pass(x, sample=False)
        return activations[-1]
```

## 6. Benchmark Evaluation

Evaluating QBM performance requires careful benchmarking.

```python
class QBMBenchmark:
    """
    Benchmark evaluation for QBMs.
    """

    def __init__(self, n_visible: int, n_hidden: int):
        self.n_visible = n_visible
        self.n_hidden = n_hidden

    def generate_synthetic_data(
        self,
        n_samples: int,
        pattern: str = 'bars'
    ) -> np.ndarray:
        """
        Generate synthetic benchmark data.

        Patterns:
        - 'bars': Horizontal or vertical bars
        - 'corners': Corner patterns
        - 'random': Random binary patterns
        """
        side = int(np.sqrt(self.n_visible))

        if pattern == 'bars':
            data = []
            for _ in range(n_samples):
                img = np.zeros((side, side))
                if np.random.random() > 0.5:
                    # Horizontal bar
                    row = np.random.randint(side)
                    img[row, :] = 1
                else:
                    # Vertical bar
                    col = np.random.randint(side)
                    img[:, col] = 1
                data.append(img.flatten())

        elif pattern == 'corners':
            data = []
            corners = [
                (0, 0), (0, side-2),
                (side-2, 0), (side-2, side-2)
            ]
            for _ in range(n_samples):
                img = np.zeros((side, side))
                r, c = corners[np.random.randint(4)]
                img[r:r+2, c:c+2] = 1
                data.append(img.flatten())

        else:  # random
            data = np.random.randint(0, 2, size=(n_samples, self.n_visible))

        return np.array(data)

    def compute_metrics(
        self,
        model,
        test_data: np.ndarray,
        n_generated: int = 1000
    ) -> Dict[str, float]:
        """
        Compute benchmark metrics.
        """
        # Generate samples
        generated = model.generate_samples(n_generated)

        # Compute metrics
        metrics = {}

        # 1. Log-likelihood (on test data)
        probs = model.visible_distribution()
        ll = 0.0
        for v in test_data:
            idx = int(sum(v[i] * 2**i for i in range(self.n_visible)))
            ll += np.log(max(probs[idx], 1e-10))
        metrics['log_likelihood'] = ll / len(test_data)

        # 2. Mode coverage
        test_modes = set(tuple(v) for v in test_data)
        gen_modes = set(tuple(v) for v in generated)
        metrics['mode_coverage'] = len(test_modes & gen_modes) / len(test_modes)

        # 3. Sample diversity
        unique_samples = len(set(tuple(v) for v in generated))
        metrics['diversity'] = unique_samples / n_generated

        # 4. Quality score (fraction matching training patterns)
        matches = sum(1 for v in generated if tuple(v) in test_modes)
        metrics['quality'] = matches / n_generated

        return metrics

    def compare_classical_quantum(
        self,
        data: np.ndarray,
        n_trials: int = 5
    ) -> Dict[str, Dict[str, float]]:
        """
        Compare classical RBM vs QBM.
        """
        from scipy.linalg import expm

        results = {
            'classical': {'ll': [], 'coverage': [], 'diversity': []},
            'quantum': {'ll': [], 'coverage': [], 'diversity': []}
        }

        for _ in range(n_trials):
            # Train classical (γ = 0)
            classical_model = QBMGenerativeModel(
                self.n_visible, self.n_hidden
            )
            classical_model.params['gamma'] = 0.0
            classical_model.train(data, n_epochs=50)
            classical_metrics = self.compute_metrics(classical_model, data)

            results['classical']['ll'].append(classical_metrics['log_likelihood'])
            results['classical']['coverage'].append(classical_metrics['mode_coverage'])
            results['classical']['diversity'].append(classical_metrics['diversity'])

            # Train quantum (γ > 0)
            quantum_model = QBMGenerativeModel(
                self.n_visible, self.n_hidden
            )
            quantum_model.params['gamma'] = 0.3
            quantum_model.train(data, n_epochs=50)
            quantum_metrics = self.compute_metrics(quantum_model, data)

            results['quantum']['ll'].append(quantum_metrics['log_likelihood'])
            results['quantum']['coverage'].append(quantum_metrics['mode_coverage'])
            results['quantum']['diversity'].append(quantum_metrics['diversity'])

        # Average results
        summary = {}
        for model_type in ['classical', 'quantum']:
            summary[model_type] = {
                metric: np.mean(values)
                for metric, values in results[model_type].items()
            }

        return summary
```
