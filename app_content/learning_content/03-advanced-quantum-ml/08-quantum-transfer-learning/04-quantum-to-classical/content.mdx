# Quantum to Classical Transfer

Quantum models may capture patterns that classical models miss, but quantum hardware is expensive. This lesson covers transferring quantum knowledge to efficient classical models.

## Learning Objectives

After completing this lesson, you will be able to:

- Extract knowledge from quantum models
- Implement knowledge distillation to classical
- Create classical surrogates for quantum circuits
- Optimize classical approximations
- Balance fidelity and efficiency
- Deploy quantum-informed classical models

## 1. Motivation for Q→C Transfer

Why transfer from quantum to classical?

<Definition term="Quantum to Classical Transfer">
Q→C transfer extracts knowledge from a quantum model (teacher) and transfers it to a classical model (student) for:

- Faster inference without quantum hardware
- Lower deployment costs
- Scalability to larger problems
- Preservation of quantum-discovered patterns
</Definition>

```python
import numpy as np

class QuantumToClassicalMotivation:
    """
    Analyze motivation for Q→C transfer.
    """

    def __init__(self):
        self.metrics = {}

    def compare_costs(self, n_inferences, quantum_cost_per_inference=0.10,
                      classical_cost_per_inference=0.0001):
        """
        Compare inference costs.
        """
        quantum_total = n_inferences * quantum_cost_per_inference
        classical_total = n_inferences * classical_cost_per_inference

        savings = quantum_total - classical_total
        speedup = quantum_cost_per_inference / classical_cost_per_inference

        return {
            'quantum_cost': quantum_total,
            'classical_cost': classical_total,
            'savings': savings,
            'cost_ratio': speedup
        }

    def compare_latency(self, quantum_latency_ms=100, classical_latency_ms=1):
        """
        Compare inference latency.
        """
        return {
            'quantum_latency': quantum_latency_ms,
            'classical_latency': classical_latency_ms,
            'speedup': quantum_latency_ms / classical_latency_ms
        }

    def assess_transfer_value(self, quantum_accuracy, classical_baseline_accuracy,
                              quantum_cost_ratio):
        """
        Assess if Q→C transfer is valuable.
        """
        accuracy_gain = quantum_accuracy - classical_baseline_accuracy

        # Transfer is valuable if quantum learns something classical didn't
        value_score = accuracy_gain * 100 / np.log(quantum_cost_ratio + 1)

        return {
            'quantum_accuracy': quantum_accuracy,
            'classical_baseline': classical_baseline_accuracy,
            'accuracy_gain': accuracy_gain,
            'value_score': value_score,
            'recommendation': 'Transfer valuable' if value_score > 5 else 'Limited benefit'
        }
```

## 2. Knowledge Distillation

Transfer quantum knowledge via soft labels.

<Formula>
Knowledge distillation minimizes:

L = α·L_hard(y_student, y_true) + (1-α)·L_soft(y_student, y_teacher)

where y_teacher are softened outputs from the quantum model.
</Formula>

```python
class QuantumDistillation:
    """
    Distill knowledge from quantum to classical model.
    """

    def __init__(self, quantum_model_fn, classical_model):
        self.quantum_model = quantum_model_fn
        self.classical_model = classical_model

    def generate_soft_labels(self, x_data, temperature=2.0):
        """
        Generate soft labels from quantum teacher.

        Higher temperature = softer labels.
        """
        soft_labels = []

        for x in x_data:
            # Quantum model output
            quantum_output = self.quantum_model(x)

            # Soften with temperature (for classification)
            # Here we use continuous outputs directly
            soft_labels.append(quantum_output)

        return np.array(soft_labels)

    def distillation_loss(self, student_output, soft_label, hard_label, alpha=0.5):
        """
        Combined distillation loss.
        """
        # Hard label loss (if available)
        hard_loss = (student_output - hard_label) ** 2

        # Soft label loss (mimic teacher)
        soft_loss = (student_output - soft_label) ** 2

        return alpha * hard_loss + (1 - alpha) * soft_loss

    def distill(self, x_train, y_train, epochs=100, lr=0.1, alpha=0.5):
        """
        Train classical student to mimic quantum teacher.
        """
        # Generate soft labels from teacher
        soft_labels = self.generate_soft_labels(x_train)

        print(f"Distilling knowledge (α={alpha})")
        print(f"Samples: {len(x_train)}")

        for epoch in range(epochs):
            total_loss = 0

            for i, (x, y_hard) in enumerate(zip(x_train, y_train)):
                # Student prediction
                student_pred = self.classical_model.forward(x)

                # Distillation loss
                loss = self.distillation_loss(
                    student_pred, soft_labels[i], y_hard, alpha
                )
                total_loss += loss

                # Update student
                self.classical_model.backward(x, soft_labels[i], y_hard, alpha, lr)

            if epoch % 20 == 0:
                print(f"  Epoch {epoch}: Loss = {total_loss/len(x_train):.4f}")

        return self.classical_model

class SimpleClassicalModel:
    """Simple classical neural network."""

    def __init__(self, n_params=4):
        self.params = np.random.randn(n_params) * 0.1

    def forward(self, x):
        result = x
        for p in self.params:
            result = np.tanh(p * result)
        return result

    def backward(self, x, soft_label, hard_label, alpha, lr, eps=0.01):
        grad = np.zeros_like(self.params)

        for i in range(len(self.params)):
            self.params[i] += eps
            pred_plus = self.forward(x)
            loss_plus = alpha * (pred_plus - hard_label)**2 + \
                       (1 - alpha) * (pred_plus - soft_label)**2

            self.params[i] -= 2 * eps
            pred_minus = self.forward(x)
            loss_minus = alpha * (pred_minus - hard_label)**2 + \
                        (1 - alpha) * (pred_minus - soft_label)**2

            self.params[i] += eps
            grad[i] = (loss_plus - loss_minus) / (2 * eps)

        self.params -= lr * grad

# Demonstrate distillation
print("\nKnowledge Distillation:")
print("=" * 50)

np.random.seed(42)

# Quantum teacher (simulated)
def quantum_teacher(x):
    # Simulated quantum model - captures nonlinear patterns
    return np.sin(2 * x) * np.exp(-0.1 * x**2)

# Classical student
student = SimpleClassicalModel(n_params=6)

# Training data
x_train = np.linspace(-3, 3, 30)
y_train = quantum_teacher(x_train)

# Distill
distiller = QuantumDistillation(quantum_teacher, student)
trained_student = distiller.distill(x_train, y_train, epochs=80, alpha=0.3)

# Evaluate
x_test = np.linspace(-3, 3, 50)
teacher_preds = [quantum_teacher(x) for x in x_test]
student_preds = [trained_student.forward(x) for x in x_test]

mse = np.mean((np.array(student_preds) - np.array(teacher_preds))**2)
print(f"\nStudent-Teacher MSE: {mse:.4f}")
```

<Tip>
Lower α emphasizes matching the teacher; higher α emphasizes true labels. Use lower α when the quantum model is trusted.
</Tip>

## 3. Surrogate Model Training

Train classical surrogates that approximate quantum circuits.

```python
class QuantumSurrogate:
    """
    Classical surrogate for quantum circuit.
    """

    def __init__(self, input_dim, hidden_dim=16):
        self.input_dim = input_dim
        self.hidden_dim = hidden_dim

        # Simple MLP: input -> hidden -> output
        self.W1 = np.random.randn(input_dim, hidden_dim) * 0.1
        self.b1 = np.zeros(hidden_dim)
        self.W2 = np.random.randn(hidden_dim, 1) * 0.1
        self.b2 = np.zeros(1)

    def forward(self, x):
        """Forward pass."""
        x = np.atleast_1d(x)
        hidden = np.tanh(x @ self.W1 + self.b1)
        output = hidden @ self.W2 + self.b2
        return output[0]

    def train_on_quantum_data(self, quantum_fn, x_samples, epochs=100, lr=0.01):
        """
        Train surrogate on quantum circuit outputs.
        """
        # Generate quantum data
        y_samples = np.array([quantum_fn(x) for x in x_samples])

        print(f"Training surrogate on {len(x_samples)} quantum samples")

        for epoch in range(epochs):
            total_loss = 0

            for x, y in zip(x_samples, y_samples):
                x = np.atleast_1d(x)

                # Forward
                pred = self.forward(x)
                loss = (pred - y) ** 2
                total_loss += loss

                # Backward (simplified gradient descent)
                hidden = np.tanh(x @ self.W1 + self.b1)

                # Output layer gradient
                d_out = 2 * (pred - y)
                d_W2 = hidden.reshape(-1, 1) * d_out
                d_b2 = d_out

                # Hidden layer gradient
                d_hidden = d_out * self.W2.flatten() * (1 - hidden**2)
                d_W1 = np.outer(x, d_hidden)
                d_b1 = d_hidden

                # Update
                self.W2 -= lr * d_W2
                self.b2 -= lr * d_b2
                self.W1 -= lr * d_W1
                self.b1 -= lr * d_b1

            if epoch % 20 == 0:
                print(f"  Epoch {epoch}: MSE = {total_loss/len(x_samples):.4f}")

    def evaluate_fidelity(self, quantum_fn, x_test):
        """
        Evaluate surrogate fidelity to quantum model.
        """
        quantum_outputs = np.array([quantum_fn(x) for x in x_test])
        surrogate_outputs = np.array([self.forward(np.atleast_1d(x)) for x in x_test])

        mse = np.mean((quantum_outputs - surrogate_outputs)**2)
        correlation = np.corrcoef(quantum_outputs, surrogate_outputs)[0, 1]

        return {
            'mse': mse,
            'correlation': correlation,
            'fidelity_score': correlation ** 2
        }

# Demonstrate surrogate training
print("\nSurrogate Model Training:")
print("=" * 50)

np.random.seed(42)

# Simulated quantum circuit function
def quantum_circuit(x):
    x = np.atleast_1d(x)
    # Complex nonlinear function representing quantum behavior
    return np.sin(x[0]) * np.cos(0.5 * x[0]) + 0.2 * np.sin(3 * x[0])

# Create and train surrogate
surrogate = QuantumSurrogate(input_dim=1, hidden_dim=8)

# Training samples (expensive quantum evaluations)
x_train = np.linspace(-np.pi, np.pi, 40).reshape(-1, 1)
surrogate.train_on_quantum_data(quantum_circuit, x_train, epochs=100)

# Evaluate
x_test = np.linspace(-np.pi, np.pi, 100).reshape(-1, 1)
fidelity = surrogate.evaluate_fidelity(quantum_circuit, x_test)

print(f"\nSurrogate Fidelity:")
print(f"  MSE: {fidelity['mse']:.4f}")
print(f"  Correlation: {fidelity['correlation']:.4f}")
print(f"  Fidelity score: {fidelity['fidelity_score']:.4f}")
```

## 4. Feature Transfer

Transfer quantum feature representations.

```python
class QuantumFeatureTransfer:
    """
    Transfer quantum feature representations to classical model.
    """

    def __init__(self, quantum_encoder):
        self.quantum_encoder = quantum_encoder

    def extract_quantum_features(self, x_data):
        """
        Extract features using quantum encoder.
        """
        features = []
        for x in x_data:
            # Quantum feature extraction
            qf = self.quantum_encoder(x)
            features.append(qf)
        return np.array(features)

    def train_classical_on_quantum_features(self, x_train, y_train):
        """
        Train classical model on quantum-extracted features.
        """
        # Extract quantum features
        quantum_features = self.extract_quantum_features(x_train)

        print(f"Extracted {len(quantum_features)} quantum feature vectors")
        print(f"Feature dimension: {quantum_features[0].shape if hasattr(quantum_features[0], 'shape') else 1}")

        # Train simple linear classifier on features
        # Add bias term
        X = np.column_stack([quantum_features, np.ones(len(quantum_features))])
        self.classical_weights = np.linalg.lstsq(X, y_train, rcond=None)[0]

        return self.classical_weights

    def hybrid_predict(self, x):
        """
        Predict using quantum features + classical classifier.
        """
        qf = self.quantum_encoder(x)
        features = np.append(qf, 1.0)
        return np.dot(features, self.classical_weights)

    def create_classical_approximation(self, x_train):
        """
        Create purely classical model that approximates quantum features.
        """
        quantum_features = self.extract_quantum_features(x_train)

        # Train classical feature extractor to mimic quantum
        # Using simple nonlinear transformation
        self.classical_feature_params = np.random.randn(3) * 0.5

        # Optimize to match quantum features
        for _ in range(100):
            grad = np.zeros(3)
            for x, qf in zip(x_train, quantum_features):
                # Classical approximation
                cf = self._classical_features(x)
                error = cf - qf

                # Gradient
                grad[0] += 2 * error * np.sin(self.classical_feature_params[0] * x)
                grad[1] += 2 * error * np.cos(self.classical_feature_params[1] * x)
                grad[2] += 2 * error

            self.classical_feature_params -= 0.01 * grad / len(x_train)

        return self.classical_feature_params

    def _classical_features(self, x):
        """Classical feature extraction."""
        p = self.classical_feature_params
        return np.sin(p[0] * x) * np.cos(p[1] * x) + p[2]
```

## 5. Ensemble Methods

Combine quantum and classical models.

```python
class QuantumClassicalEnsemble:
    """
    Ensemble of quantum and classical models.
    """

    def __init__(self, quantum_model, classical_model):
        self.quantum_model = quantum_model
        self.classical_model = classical_model
        self.weights = [0.5, 0.5]  # Equal weighting initially

    def predict(self, x):
        """Weighted ensemble prediction."""
        q_pred = self.quantum_model(x)
        c_pred = self.classical_model(x)
        return self.weights[0] * q_pred + self.weights[1] * c_pred

    def learn_weights(self, x_val, y_val):
        """
        Learn optimal ensemble weights on validation data.
        """
        q_preds = np.array([self.quantum_model(x) for x in x_val])
        c_preds = np.array([self.classical_model(x) for x in x_val])

        # Grid search for optimal weights
        best_mse = float('inf')
        best_weights = [0.5, 0.5]

        for w_q in np.linspace(0, 1, 21):
            w_c = 1 - w_q
            ensemble_preds = w_q * q_preds + w_c * c_preds
            mse = np.mean((ensemble_preds - y_val)**2)

            if mse < best_mse:
                best_mse = mse
                best_weights = [w_q, w_c]

        self.weights = best_weights
        return best_weights, best_mse

    def progressive_distillation(self, x_train, y_train, stages=3):
        """
        Progressively shift from quantum to classical.
        """
        print("Progressive Distillation:")

        for stage in range(stages):
            # Stage ratio: more classical as stages progress
            ratio = (stage + 1) / stages

            # Blend labels
            q_labels = np.array([self.quantum_model(x) for x in x_train])
            blended = (1 - ratio) * q_labels + ratio * y_train

            # Train classical on blended labels
            self._train_classical(x_train, blended)

            # Evaluate
            c_preds = np.array([self.classical_model(x) for x in x_train])
            mse = np.mean((c_preds - y_train)**2)
            print(f"  Stage {stage+1}: ratio={ratio:.2f}, MSE={mse:.4f}")

    def _train_classical(self, x_data, y_data, epochs=20, lr=0.1):
        """Simple training of classical model."""
        if hasattr(self.classical_model, 'params'):
            for _ in range(epochs):
                for x, y in zip(x_data, y_data):
                    pred = self.classical_model(x)
                    error = pred - y
                    # Simple gradient update
                    for i in range(len(self.classical_model.params)):
                        self.classical_model.params[i] -= lr * error * 0.1
```

## 6. Deployment Strategies

Deploy quantum-informed classical models.

```python
class QuantumInformedDeployment:
    """
    Deploy classical models with quantum-derived knowledge.
    """

    def __init__(self, distilled_model):
        self.model = distilled_model
        self.metrics = {}

    def export_model(self, format='dict'):
        """
        Export model for deployment.
        """
        if format == 'dict':
            if hasattr(self.model, 'params'):
                return {
                    'type': 'classical_surrogate',
                    'params': self.model.params.tolist() if hasattr(self.model.params, 'tolist')
                             else list(self.model.params),
                    'origin': 'quantum_distillation'
                }
        return None

    def create_inference_pipeline(self, preprocessing_fn=None):
        """
        Create inference pipeline.
        """
        def pipeline(x):
            # Preprocess
            if preprocessing_fn:
                x = preprocessing_fn(x)

            # Model inference
            return self.model(x)

        return pipeline

    def monitor_drift(self, x_new, reference_distribution):
        """
        Monitor for distribution drift in production.
        """
        # Compare new data to reference
        new_mean = np.mean(x_new)
        new_std = np.std(x_new)

        ref_mean = reference_distribution['mean']
        ref_std = reference_distribution['std']

        # Z-score for mean shift
        z_score = abs(new_mean - ref_mean) / (ref_std / np.sqrt(len(x_new)))

        drift_detected = z_score > 2.0

        return {
            'z_score': z_score,
            'drift_detected': drift_detected,
            'action': 'Re-calibrate with quantum' if drift_detected else 'Continue'
        }

    def calibration_schedule(self, performance_threshold=0.9):
        """
        Determine when to re-run quantum calibration.
        """
        return {
            'strategy': 'performance_based',
            'threshold': performance_threshold,
            'check_interval': 'daily',
            'actions': [
                'Monitor prediction confidence',
                'Track distribution drift',
                'Re-distill if performance drops below threshold'
            ]
        }

# Demonstrate deployment
print("\nQuantum-Informed Deployment:")
print("=" * 50)

# Create distilled model
class DistilledModel:
    def __init__(self):
        self.params = np.array([1.8, 0.3, -0.1])

    def __call__(self, x):
        return np.sin(self.params[0] * x) + self.params[1] * x + self.params[2]

model = DistilledModel()

# Export

export = deployer.export_model()
print(f"   Type: {export['type']}")
print(f"   Params: {export['params']}")
print(f"   Origin: {export['origin']}")

# Create pipeline
print("\n2. Inference pipeline created")
pipeline = deployer.create_inference_pipeline(
    preprocessing_fn=lambda x: x / np.max(np.abs(x)) if np.max(np.abs(x)) > 0 else x
)

# Test inference
for x in [-1, 0, 1]:
    pred = pipeline(np.array([x]))
    print(f"   Input: {x} -> Output: {pred:.3f}")

# Monitor drift
print("\n3. Drift monitoring:")
np.random.seed(42)
reference = {'mean': 0.0, 'std': 1.0}
new_data_ok = np.random.randn(50)
new_data_drift = np.random.randn(50) + 1.5

print(f"   Normal data: {deployer.monitor_drift(new_data_ok, reference)}")
print(f"   Drifted data: {deployer.monitor_drift(new_data_drift, reference)}")

# Calibration schedule
print("\n4. Calibration schedule:")
schedule = deployer.calibration_schedule()
for action in schedule['actions']:
    print(f"   - {action}")
```

<Tip>
Monitor deployed classical models for drift. When performance degrades, consider re-distilling from the quantum model with updated calibration.
</Tip>
