# Pre-trained Quantum Models

Pre-trained models are foundation for transfer learning. This lesson covers how to create, manage, and deploy pre-trained quantum models for reuse across tasks.

## Learning Objectives

After completing this lesson, you will be able to:

- Design pre-trainable quantum architectures
- Save and load quantum model parameters
- Create quantum model repositories
- Evaluate pre-trained model quality
- Select appropriate pre-trained models
- Adapt pre-trained models to new tasks

## 1. Quantum Model Architecture

Design architectures suitable for pre-training and transfer.

<Definition term="Pre-trainable Architecture">
A pre-trainable quantum architecture has:
1. Modular structure with separable encoder and task-specific layers
2. Sufficient expressibility for the pre-training task
3. Transferable representations in early/middle layers
</Definition>

```python
import numpy as np
import json

class PretrainableQuantumModel:
    """
    Quantum model designed for pre-training and transfer.
    """

    def __init__(self, n_qubits, n_encoder_layers, n_task_layers):
        self.n_qubits = n_qubits
        self.n_encoder_layers = n_encoder_layers
        self.n_task_layers = n_task_layers

        # Parameter counts
        self.encoder_params = n_qubits * n_encoder_layers * 2  # RY, RZ per qubit
        self.task_params = n_qubits * n_task_layers * 2

        self.total_params = self.encoder_params + self.task_params

        # Initialize parameters
        self.params = None
        self.metadata = {}

    def initialize(self, seed=None):
        """Initialize parameters randomly."""
        if seed is not None:
            np.random.seed(seed)
        self.params = np.random.randn(self.total_params) * 0.1
        return self.params

    def get_encoder_params(self):
        """Get encoder (transferable) parameters."""
        return self.params[:self.encoder_params]

    def get_task_params(self):
        """Get task-specific parameters."""
        return self.params[self.encoder_params:]

    def set_encoder_params(self, encoder_params):
        """Set encoder parameters (e.g., from pre-trained model)."""
        self.params[:self.encoder_params] = encoder_params

    def forward(self, x):
        """
        Forward pass through the model.

        Simulates: encoding -> encoder layers -> task layers -> measurement
        """
        # Data encoding
        state = np.array([np.cos(x/2), np.sin(x/2)])

        # Encoder layers (transferable)
        for i in range(self.n_encoder_layers):
            idx = i * self.n_qubits * 2
            for q in range(self.n_qubits):
                theta = self.params[idx + q*2]
                phi = self.params[idx + q*2 + 1]
                state = self._apply_rotation(state, theta, phi)

        # Task-specific layers
        for i in range(self.n_task_layers):
            idx = self.encoder_params + i * self.n_qubits * 2
            for q in range(self.n_qubits):
                theta = self.params[idx + q*2]
                phi = self.params[idx + q*2 + 1]
                state = self._apply_rotation(state, theta, phi)

        # Measurement (simplified)
        return np.real(state[0] * np.conj(state[0]) - state[1] * np.conj(state[1]))

    def _apply_rotation(self, state, theta, phi):
        """Apply parameterized rotation."""
        c, s = np.cos(theta/2), np.sin(theta/2)
        rot = np.array([[c, -s], [s, c]])
        return rot @ state * np.exp(1j * phi)

    def describe(self):
        """Describe the model architecture."""
        return {
            'n_qubits': self.n_qubits,
            'n_encoder_layers': self.n_encoder_layers,
            'n_task_layers': self.n_task_layers,
            'encoder_params': self.encoder_params,
            'task_params': self.task_params,
            'total_params': self.total_params
        }

# Demonstrate architecture
print("Pre-trainable Quantum Model Architecture:")
print("=" * 50)

model = PretrainableQuantumModel(n_qubits=4, n_encoder_layers=3, n_task_layers=2)
model.initialize(seed=42)

desc = model.describe()
for key, value in desc.items():
    print(f"  {key}: {value}")

print(f"\nEncoder params (transferable): {model.get_encoder_params()[:6]}...")
print(f"Task params (task-specific): {model.get_task_params()[:6]}...")

# Test forward pass
x = 0.5
output = model.forward(x)
print(f"\nForward pass: input={x} -> output={output:.4f}")
```

## 2. Model Checkpointing

Save and load model parameters for reuse.

```python
class QuantumModelCheckpoint:
    """
    Save and load quantum model checkpoints.
    """

    def __init__(self, save_dir="./checkpoints"):
        self.save_dir = save_dir

    def save(self, model, name, metadata=None):
        """
        Save model checkpoint.

        Args:
            model: PretrainableQuantumModel instance
            name: Checkpoint name
            metadata: Additional metadata (training info, etc.)
        """
        checkpoint = {
            'architecture': model.describe(),
            'params': model.params.tolist(),
            'encoder_params': model.get_encoder_params().tolist(),
            'metadata': metadata or {}
        }

        # In practice, save to file
        # with open(f"{self.save_dir}/{name}.json", 'w') as f:
        #     json.dump(checkpoint, f)

        print(f"Saved checkpoint: {name}")
        print(f"  Parameters: {len(checkpoint['params'])}")
        print(f"  Encoder params: {len(checkpoint['encoder_params'])}")
        if metadata:
            print(f"  Metadata: {list(metadata.keys())}")

        return checkpoint

    def load(self, checkpoint, target_model=None):
        """
        Load model from checkpoint.

        Args:
            checkpoint: Checkpoint dict
            target_model: Optional model to load into
        """
        if target_model is not None:
            # Verify architecture compatibility
            target_arch = target_model.describe()
            source_arch = checkpoint['architecture']

            if target_arch['encoder_params'] != source_arch['encoder_params']:
                raise ValueError("Incompatible encoder architecture")

            # Load encoder params (for transfer)
            target_model.set_encoder_params(np.array(checkpoint['encoder_params']))
            print("Loaded encoder params into target model")

            return target_model

        # Create new model from checkpoint
        arch = checkpoint['architecture']
        model = PretrainableQuantumModel(
            n_qubits=arch['n_qubits'],
            n_encoder_layers=arch['n_encoder_layers'],
            n_task_layers=arch['n_task_layers']
        )
        model.params = np.array(checkpoint['params'])

        return model

    def load_encoder_only(self, checkpoint, target_model):
        """Load only encoder params for transfer learning."""
        encoder_params = np.array(checkpoint['encoder_params'])
        target_model.set_encoder_params(encoder_params)
        print(f"Transferred {len(encoder_params)} encoder parameters")
        return target_model

# Demonstrate checkpointing
print("\nModel Checkpointing:")
print("=" * 50)

# Create and train a model (simulated)

source_model.initialize(seed=42)

# Simulate training by modifying params

# Save checkpoint

    'training_task': 'classification',
    'epochs': 100,
    'final_loss': 0.05,
    'training_samples': 1000
}

checkpoint = checkpoint_mgr.save(source_model, "pretrained_classifier_v1", training_metadata)

# Load for transfer
print("\n" + "-" * 50)
print("Loading for transfer learning:")

target_model = PretrainableQuantumModel(n_qubits=4, n_encoder_layers=3, n_task_layers=2)
target_model.initialize(seed=123)  # Random initialization

print(f"Before transfer, task params: {target_model.get_task_params()[:4]}")

checkpoint_mgr.load_encoder_only(checkpoint, target_model)

print(f"After transfer, encoder matches: {np.allclose(target_model.get_encoder_params(), source_model.get_encoder_params())}")
```

## 3. Quantum Model Zoo

Create a collection of pre-trained models for different tasks.

```python
class QuantumModelZoo:
    """
    Repository of pre-trained quantum models.
    """

    def __init__(self):
        self.models = {}

    def register(self, name, model, checkpoint, tags=None):
        """
        Register a pre-trained model.

        Args:
            name: Unique model name
            model: Model instance
            checkpoint: Saved checkpoint
            tags: List of tags for search
        """
        self.models[name] = {
            'model': model,
            'checkpoint': checkpoint,
            'tags': tags or [],
            'architecture': model.describe()
        }
        print(f"Registered: {name}")

    def list_models(self, tag=None):
        """List available models, optionally filtered by tag."""
        if tag is None:
            return list(self.models.keys())

        return [name for name, info in self.models.items()
                if tag in info['tags']]

    def get_model(self, name):
        """Get a model by name."""
        if name not in self.models:
            raise ValueError(f"Model {name} not found")
        return self.models[name]

    def search(self, n_qubits=None, task_type=None, min_performance=None):
        """
        Search for compatible models.

        Args:
            n_qubits: Required number of qubits
            task_type: Task type tag
            min_performance: Minimum performance threshold
        """
        results = []

        for name, info in self.models.items():
            # Check qubit count
            if n_qubits and info['architecture']['n_qubits'] != n_qubits:
                continue

            # Check task type
            if task_type and task_type not in info['tags']:
                continue

            # Check performance
            if min_performance:
                perf = info['checkpoint'].get('metadata', {}).get('final_loss', float('inf'))
                if perf > (1 - min_performance):
                    continue

            results.append({
                'name': name,
                'architecture': info['architecture'],
                'tags': info['tags'],
                'metadata': info['checkpoint'].get('metadata', {})
            })

        return results

    def recommend(self, target_task, n_qubits):
        """
        Recommend best model for target task.
        """
        candidates = self.search(n_qubits=n_qubits)

        if not candidates:
            return None

        # Score candidates based on relevance
        scored = []
        for cand in candidates:
            score = 0

            # Task similarity
            if target_task in cand['tags']:
                score += 5
            elif any(t in ['general', 'feature_extraction'] for t in cand['tags']):
                score += 2

            # Performance
            loss = cand['metadata'].get('final_loss', 1.0)
            score += (1 - loss) * 3

            scored.append((score, cand))

        scored.sort(key=lambda x: -x[0])
        return scored[0][1] if scored else None

# Create model zoo

print("=" * 50)

    }
```

## 4. Model Quality Evaluation

Evaluate pre-trained model quality before transfer.

```python
class PretrainedModelEvaluator:
    """
    Evaluate quality and transferability of pre-trained models.
    """

    def __init__(self):
        pass

    def evaluate_expressibility(self, model, n_samples=100):
        """
        Evaluate model's expressibility.

        Higher expressibility = more diverse outputs.
        """
        outputs = []

        for _ in range(n_samples):
            # Random parameters
            model.params = np.random.randn(model.total_params)
            x = np.random.uniform(-np.pi, np.pi)
            outputs.append(model.forward(x))

        # Expressibility measured by output variance
        expressibility = np.std(outputs)

        return {
            'expressibility': expressibility,
            'output_range': (min(outputs), max(outputs)),
            'mean_output': np.mean(outputs)
        }

    def evaluate_transferability(self, model, source_task_data, target_task_data):
        """
        Estimate transferability by comparing representations.
        """
        # Get encoder representations for both tasks
        source_reps = []
        target_reps = []

        for x in source_task_data:
            rep = self._get_encoder_representation(model, x)
            source_reps.append(rep)

        for x in target_task_data:
            rep = self._get_encoder_representation(model, x)
            target_reps.append(rep)

        # Measure representation overlap
        source_reps = np.array(source_reps)
        target_reps = np.array(target_reps)

        # Simple overlap metric: correlation of representations
        overlap = np.abs(np.corrcoef(
            np.mean(source_reps, axis=0).flatten() if source_reps.ndim > 1 else source_reps,
            np.mean(target_reps, axis=0).flatten() if target_reps.ndim > 1 else target_reps
        )[0, 1])

        return {
            'transferability_score': overlap if not np.isnan(overlap) else 0.0,
            'source_rep_std': np.std(source_reps),
            'target_rep_std': np.std(target_reps)
        }

    def _get_encoder_representation(self, model, x):
        """Get intermediate representation after encoder."""
        # Simplified: run encoder layers only
        state = np.array([np.cos(x/2), np.sin(x/2)])

        for i in range(model.n_encoder_layers):
            idx = i * model.n_qubits * 2
            for q in range(model.n_qubits):
                theta = model.params[idx + q*2]
                state = np.array([
                    np.cos(theta/2) * state[0] - np.sin(theta/2) * state[1],
                    np.sin(theta/2) * state[0] + np.cos(theta/2) * state[1]
                ])

        return state

    def full_evaluation(self, model, source_data, target_data):
        """Complete model evaluation."""
        expr = self.evaluate_expressibility(model)
        trans = self.evaluate_transferability(model, source_data, target_data)

        overall_score = 0.5 * expr['expressibility'] + 0.5 * trans['transferability_score']

        return {
            'expressibility': expr,
            'transferability': trans,
            'overall_score': overall_score,
            'recommendation': 'good' if overall_score > 0.3 else 'marginal'
        }

# Evaluate pre-trained models
print("\nPre-trained Model Evaluation:")
print("=" * 50)

evaluator = PretrainedModelEvaluator()

# Create and evaluate a model
model = PretrainableQuantumModel(n_qubits=4, n_encoder_layers=3, n_task_layers=2)
model.initialize(seed=42)
model.params = model.params + 0.5  # Simulate training

# Source and target data
source_data = np.linspace(-np.pi, np.pi, 20)
target_data = np.linspace(-np.pi/2, np.pi/2, 15)

# Full evaluation
eval_result = evaluator.full_evaluation(model, source_data, target_data)

print("\nEvaluation Results:")
print(f"  Expressibility: {eval_result['expressibility']['expressibility']:.4f}")
print(f"  Transferability: {eval_result['transferability']['transferability_score']:.4f}")
print(f"  Overall Score: {eval_result['overall_score']:.4f}")
print(f"  Recommendation: {eval_result['recommendation']}")
```

<Warning>
High expressibility doesn't guarantee good transfer. Evaluate both expressibility and task-specific performance on validation data.
</Warning>

## 5. Adapting Pre-trained Models

Strategies for adapting pre-trained models to new tasks.

```python
class PretrainedAdapter:
    """
    Adapt pre-trained models to new tasks.
    """

    def __init__(self, pretrained_model):
        self.pretrained = pretrained_model

    def freeze_and_extend(self, new_task_layers):
        """
        Freeze encoder, add new task layers.
        """
        # Create new model with same encoder
        new_model = PretrainableQuantumModel(
            n_qubits=self.pretrained.n_qubits,
            n_encoder_layers=self.pretrained.n_encoder_layers,
            n_task_layers=new_task_layers
        )
        new_model.initialize()

        # Copy encoder params
        new_model.set_encoder_params(self.pretrained.get_encoder_params())

        return new_model, 'encoder_frozen'

    def full_model_with_scaling(self, lr_scale=0.1):
        """
        Use full model with scaled learning rate for encoder.
        """
        return {
            'model': self.pretrained,
            'encoder_lr': lr_scale,
            'task_lr': 1.0,
            'strategy': 'differential_lr'
        }

    def adapter_layers(self, adapter_size=4):
        """
        Add small adapter layers between encoder and task layers.
        """
        # In practice, add small trainable layers
        adapter = {
            'adapter_params': np.random.randn(adapter_size) * 0.01,
            'frozen_encoder': self.pretrained.get_encoder_params(),
            'frozen_task': self.pretrained.get_task_params(),
            'strategy': 'adapters'
        }
        return adapter

    def knowledge_distillation_setup(self, student_model):
        """
        Setup for distilling knowledge to a smaller student.
        """
        return {
            'teacher': self.pretrained,
            'student': student_model,
            'temperature': 2.0,
            'alpha': 0.5,  # Balance between hard and soft labels
            'strategy': 'distillation'
        }

# Demonstrate adaptation strategies
print("\nPre-trained Model Adaptation:")
print("=" * 50)

# Pre-trained model
pretrained = PretrainableQuantumModel(n_qubits=4, n_encoder_layers=3, n_task_layers=2)
pretrained.initialize(seed=42)
pretrained.params += 0.5

adapter = PretrainedAdapter(pretrained)

# Strategy 1: Freeze and extend
print("\n1. Freeze and Extend:")
new_model, strategy = adapter.freeze_and_extend(new_task_layers=3)
print(f"   New model has {new_model.task_params} task params")
print(f"   Encoder frozen: {new_model.encoder_params} params")

# Strategy 2: Differential learning rate
print("\n2. Differential Learning Rate:")

print(f"   Encoder LR: {config['encoder_lr']}x")
print(f"   Task LR: {config['task_lr']}x")

# Strategy 3: Adapter layers
print("\n3. Adapter Layers:")
adapter_config = adapter.adapter_layers(adapter_size=8)
print(f"   Trainable adapter params: {len(adapter_config['adapter_params'])}")
print(f"   Frozen encoder: {len(adapter_config['frozen_encoder'])} params")

# Strategy 4: Knowledge distillation
print("\n4. Knowledge Distillation:")
student = PretrainableQuantumModel(n_qubits=4, n_encoder_layers=2, n_task_layers=1)
student.initialize()
distill_config = adapter.knowledge_distillation_setup(student)
print(f"   Teacher params: {pretrained.total_params}")
print(f"   Student params: {student.total_params}")
print(f"   Temperature: {distill_config['temperature']}")
```

## 6. Best Practices

Guidelines for using pre-trained quantum models effectively.

```python
class TransferBestPractices:
    """
    Best practices for quantum transfer learning.
    """

    @staticmethod
    def checklist():
        """Pre-transfer checklist."""
        return {
            'architecture_check': [
                'Verify qubit count matches or is compatible',
                'Check encoder/task layer structure',
                'Confirm gate set compatibility with target hardware'
            ],
            'data_check': [
                'Ensure source and target domains are related',
                'Verify data encoding is consistent',
                'Check for distribution shift'
            ],
            'training_check': [
                'Use appropriate freeze/unfreeze strategy',
                'Start with lower learning rate',
                'Monitor for negative transfer'
            ],
            'validation': [
                'Compare against training from scratch',
                'Evaluate on held-out test set',
                'Check for overfitting to small target set'
            ]
        }

    @staticmethod
    def common_pitfalls():
        """Common mistakes to avoid."""
        return [
            {
                'pitfall': 'Mismatched architectures',
                'solution': 'Use compatible encoder structure, adapt task layers'
            },
            {
                'pitfall': 'Too different domains',
                'solution': 'Evaluate domain distance before transfer'
            },
            {
                'pitfall': 'Overfreezing',
                'solution': 'Try progressive unfreezing if performance plateaus'
            },
            {
                'pitfall': 'Wrong learning rate',
                'solution': 'Use lower LR for transferred layers'
            },
            {
                'pitfall': 'No validation',
                'solution': 'Always compare to from-scratch baseline'
            }
        ]

    @staticmethod
    def hardware_considerations():
        """Hardware-specific considerations."""
        return {
            'noise_adaptation': 'Pre-trained on simulator may need noise adaptation',
            'gate_decomposition': 'Different hardware may decompose gates differently',
            'qubit_mapping': 'Optimal qubit layout may differ between devices',
            'calibration_drift': 'Hardware calibration changes over time'
        }
```
