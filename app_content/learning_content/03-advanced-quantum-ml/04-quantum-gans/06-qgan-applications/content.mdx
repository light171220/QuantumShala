## 1. Quantum State Generation

<Definition term="Quantum State QGAN">
A QGAN specifically designed to generate quantum states that approximate a target state or distribution of states, useful for state preparation, quantum simulation, and variational algorithms.
</Definition>

<Formula>
\text{Target: Learn to prepare } |\psi_{target}\rangle \text{ from } |0\rangle^{\otimes n}
\\[8pt]
\text{Generator: } G(\theta): |0\rangle^{\otimes n} \mapsto |\psi_G(\theta)\rangle
\\[8pt]
\text{Fidelity metric: } F = |\langle\psi_{target}|\psi_G(\theta)\rangle|^2
</Formula>

```python
import numpy as np
from typing import List, Tuple, Dict

class QuantumStateQGAN:
    """QGAN for learning to generate quantum states."""

    def __init__(self, n_qubits: int, n_layers: int = 3):
        self.n_qubits = n_qubits
        self.n_layers = n_layers
        self.dim = 2**n_qubits

        # Generator parameters
        params_per_layer = 3 * n_qubits  # RX, RY, RZ per qubit
        self.gen_params = np.random.randn(n_layers * params_per_layer) * 0.1

        # Discriminator parameters (for SWAP test or direct comparison)
        self.disc_params = np.random.randn(n_layers * 2 * n_qubits) * 0.1

    def _ry_gate(self, theta: float) -> np.ndarray:
        c, s = np.cos(theta/2), np.sin(theta/2)
        return np.array([[c, -s], [s, c]])

    def _rz_gate(self, theta: float) -> np.ndarray:
        return np.array([[np.exp(-1j*theta/2), 0],
                        [0, np.exp(1j*theta/2)]])

    def _apply_gate(self, state: np.ndarray, gate: np.ndarray,
                    qubit: int) -> np.ndarray:
        n_qubits = int(np.log2(len(state)))
        full_gate = np.eye(1)
        for q in range(n_qubits):
            full_gate = np.kron(full_gate,
                               gate if q == qubit else np.eye(2))
        return full_gate @ state

    def generate_state(self, z: np.ndarray = None) -> np.ndarray:
        """Generate a quantum state from the circuit."""
        state = np.zeros(self.dim, dtype=complex)
        state[0] = 1.0

        # Optional noise input for diversity
        if z is not None:
            for i, zi in enumerate(z[:self.n_qubits]):
                ry = self._ry_gate(zi * 0.1)
                state = self._apply_gate(state, ry, i)

        # Variational layers
        params_per_layer = 3 * self.n_qubits
        for layer in range(self.n_layers):
            layer_params = self.gen_params[layer*params_per_layer:
                                           (layer+1)*params_per_layer]

            for q in range(self.n_qubits):
                # RY and RZ rotations
                ry = self._ry_gate(layer_params[q])
                rz = self._rz_gate(layer_params[self.n_qubits + q])
                state = self._apply_gate(state, ry, q)
                state = self._apply_gate(state, rz, q)

        return state / np.linalg.norm(state)

    def compute_fidelity(self, target_state: np.ndarray,
                         generated_state: np.ndarray) -> float:
        """Compute fidelity between target and generated states."""
        inner_product = np.vdot(target_state, generated_state)
        return np.abs(inner_product)**2

    def swap_test_probability(self, state1: np.ndarray,
                               state2: np.ndarray) -> float:
        """Compute probability from SWAP test (estimates |⟨ψ1|ψ2⟩|²)."""
        inner_product = np.vdot(state1, state2)
        # P(0) = (1 + |⟨ψ1|ψ2⟩|²) / 2
        return (1 + np.abs(inner_product)**2) / 2


class QuantumStateGenerator:
    """Generator for specific quantum states."""

    def __init__(self, n_qubits: int):
        self.n_qubits = n_qubits
        self.dim = 2**n_qubits

    def ghz_state(self) -> np.ndarray:
        """Generate GHZ state: (|00...0⟩ + |11...1⟩)/√2."""
        state = np.zeros(self.dim, dtype=complex)
        state[0] = 1/np.sqrt(2)
        state[-1] = 1/np.sqrt(2)
        return state

    def w_state(self) -> np.ndarray:
        """Generate W state: (|100...⟩ + |010...⟩ + ...)/√n."""
        state = np.zeros(self.dim, dtype=complex)
        for i in range(self.n_qubits):
            basis = 1 << (self.n_qubits - 1 - i)
            state[basis] = 1/np.sqrt(self.n_qubits)
        return state

    def random_state(self) -> np.ndarray:
        """Generate random pure state (Haar random)."""
        real = np.random.randn(self.dim)
        imag = np.random.randn(self.dim)
        state = real + 1j * imag
        return state / np.linalg.norm(state)

    def cluster_state(self) -> np.ndarray:
        """Generate 1D cluster state."""
        # Start with |+⟩^n
        state = np.ones(self.dim, dtype=complex) / np.sqrt(self.dim)

        # Apply CZ between adjacent qubits
        for i in range(self.n_qubits - 1):
            for basis in range(self.dim):
                bits = format(basis, f'0{self.n_qubits}b')
                if bits[i] == '1' and bits[i+1] == '1':
                    state[basis] *= -1

        return state
```

## 2. Probability Distribution Learning

<Definition term="Born Machine">
A quantum generative model that outputs probability distributions through measurement of quantum states. QGANs can be viewed as adversarially trained Born machines.
</Definition>

```python
class DistributionQGAN:
    """QGAN for learning classical probability distributions."""

    def __init__(self, n_qubits: int, n_layers: int = 3):
        self.n_qubits = n_qubits
        self.n_layers = n_layers
        self.n_outcomes = 2**n_qubits

        # Generator parameters
        self.gen_params = np.random.randn(n_layers * 2 * n_qubits) * 0.1

        # Discriminator (classical for distribution comparison)
        self.disc_hidden = 32
        self.disc_W1 = np.random.randn(self.disc_hidden, self.n_outcomes) * 0.1
        self.disc_b1 = np.zeros(self.disc_hidden)
        self.disc_W2 = np.random.randn(1, self.disc_hidden) * 0.1
        self.disc_b2 = np.zeros(1)

    def _ry_gate(self, theta: float) -> np.ndarray:
        c, s = np.cos(theta/2), np.sin(theta/2)
        return np.array([[c, -s], [s, c]])

    def _apply_gate(self, state: np.ndarray, gate: np.ndarray,
                    qubit: int) -> np.ndarray:
        n_qubits = int(np.log2(len(state)))
        full_gate = np.eye(1)
        for q in range(n_qubits):
            full_gate = np.kron(full_gate,
                               gate if q == qubit else np.eye(2))
        return full_gate @ state

    def generate_distribution(self, z: np.ndarray = None) -> np.ndarray:
        """Generate probability distribution via Born rule."""
        dim = 2**self.n_qubits
        state = np.zeros(dim, dtype=complex)
        state[0] = 1.0

        # Noise encoding
        if z is not None:
            for i, zi in enumerate(z[:self.n_qubits]):
                ry = self._ry_gate(np.pi * np.tanh(zi))
                state = self._apply_gate(state, ry, i)

        # Variational circuit
        params_per_layer = 2 * self.n_qubits
        for layer in range(self.n_layers):
            layer_params = self.gen_params[layer*params_per_layer:
                                           (layer+1)*params_per_layer]

            for q in range(self.n_qubits):
                ry1 = self._ry_gate(layer_params[q])
                ry2 = self._ry_gate(layer_params[self.n_qubits + q])
                state = self._apply_gate(state, ry1, q)
                state = self._apply_gate(state, ry2, q)

        # Born rule: |ψ|²
        probs = np.abs(state)**2
        return probs

    def sample(self, n_samples: int, z: np.ndarray = None) -> np.ndarray:
        """Sample from the generated distribution."""
        probs = self.generate_distribution(z)
        samples = np.random.choice(self.n_outcomes, size=n_samples, p=probs)
        return samples

    def discriminator_forward(self, probs: np.ndarray) -> float:
        """Classical discriminator on probability vector."""
        h = np.maximum(0, self.disc_W1 @ probs + self.disc_b1)
        out = self.disc_W2 @ h + self.disc_b2
        return 1 / (1 + np.exp(-out[0]))


class DistributionMetrics:
    """Metrics for comparing probability distributions."""

    @staticmethod
    def kl_divergence(p: np.ndarray, q: np.ndarray,
                      epsilon: float = 1e-10) -> float:
        """Compute KL divergence D_KL(P || Q)."""
        p = np.clip(p, epsilon, 1)
        q = np.clip(q, epsilon, 1)
        return np.sum(p * np.log(p / q))

    @staticmethod
    def total_variation(p: np.ndarray, q: np.ndarray) -> float:
        """Compute total variation distance."""
        return 0.5 * np.sum(np.abs(p - q))

    @staticmethod
    def hellinger_distance(p: np.ndarray, q: np.ndarray) -> float:
        """Compute Hellinger distance."""
        return np.sqrt(0.5 * np.sum((np.sqrt(p) - np.sqrt(q))**2))

    @staticmethod
    def wasserstein_distance_1d(p: np.ndarray, q: np.ndarray) -> float:
        """Compute 1D Wasserstein distance (EMD)."""
        # CDF-based computation for 1D
        cdf_p = np.cumsum(p)
        cdf_q = np.cumsum(q)
        return np.sum(np.abs(cdf_p - cdf_q))
```

## 3. Financial Modeling Applications

<Tip>
QGANs show promise for financial applications where distributions have complex structures, correlations, or tail behaviors that are difficult to model classically.
</Tip>

```python
class FinancialQGAN:
    """QGAN for financial distribution modeling."""

    def __init__(self, n_qubits: int, n_assets: int = 1):
        self.n_qubits = n_qubits
        self.n_assets = n_assets
        self.n_layers = 3

        # Generator for price/return distributions
        self.gen_params = np.random.randn(self.n_layers * 2 * n_qubits) * 0.1

    def _ry_gate(self, theta: float) -> np.ndarray:
        c, s = np.cos(theta/2), np.sin(theta/2)
        return np.array([[c, -s], [s, c]])

    def _apply_gate(self, state: np.ndarray, gate: np.ndarray,
                    qubit: int) -> np.ndarray:
        n_qubits = int(np.log2(len(state)))
        full_gate = np.eye(1)
        for q in range(n_qubits):
            full_gate = np.kron(full_gate,
                               gate if q == qubit else np.eye(2))
        return full_gate @ state

    def generate_return_distribution(self, market_condition: np.ndarray = None
                                      ) -> np.ndarray:
        """Generate distribution of asset returns."""
        dim = 2**self.n_qubits
        state = np.zeros(dim, dtype=complex)
        state[0] = 1.0

        # Encode market conditions
        if market_condition is not None:
            for i, mc in enumerate(market_condition[:self.n_qubits]):
                ry = self._ry_gate(np.pi * np.tanh(mc))
                state = self._apply_gate(state, ry, i)

        # Variational circuit
        params_per_layer = 2 * self.n_qubits
        for layer in range(self.n_layers):
            layer_params = self.gen_params[layer*params_per_layer:
                                           (layer+1)*params_per_layer]
            for q in range(self.n_qubits):
                ry = self._ry_gate(layer_params[q])
                state = self._apply_gate(state, ry, q)

        return np.abs(state)**2

    def sample_returns(self, n_samples: int,
                       return_range: Tuple[float, float] = (-0.1, 0.1)
                       ) -> np.ndarray:
        """Sample returns from the distribution."""
        probs = self.generate_return_distribution()
        n_bins = len(probs)

        # Map bin indices to return values
        bins = np.linspace(return_range[0], return_range[1], n_bins)
        indices = np.random.choice(n_bins, size=n_samples, p=probs)
        returns = bins[indices]

        return returns

    def compute_var(self, confidence: float = 0.95,
                    return_range: Tuple[float, float] = (-0.1, 0.1)
                    ) -> float:
        """Compute Value at Risk from the distribution."""
        probs = self.generate_return_distribution()
        n_bins = len(probs)
        bins = np.linspace(return_range[0], return_range[1], n_bins)

        # Find quantile
        cumulative = np.cumsum(probs)
        var_idx = np.searchsorted(cumulative, 1 - confidence)
        var_idx = min(var_idx, n_bins - 1)

        return bins[var_idx]

    def compute_expected_shortfall(self, confidence: float = 0.95,
                                    return_range: Tuple[float, float] = (-0.1, 0.1)
                                    ) -> float:
        """Compute Expected Shortfall (CVaR)."""
        probs = self.generate_return_distribution()
        n_bins = len(probs)
        bins = np.linspace(return_range[0], return_range[1], n_bins)

        # Find VaR first
        cumulative = np.cumsum(probs)
        var_idx = np.searchsorted(cumulative, 1 - confidence)

        # Average of losses beyond VaR
        tail_probs = probs[:var_idx]
        if tail_probs.sum() > 0:
            tail_probs = tail_probs / tail_probs.sum()
            es = np.sum(bins[:var_idx] * tail_probs)
        else:
            es = bins[0]

        return es


class OptionPricingQGAN:
    """QGAN for option pricing via distribution learning."""

    def __init__(self, n_qubits: int):
        self.n_qubits = n_qubits
        self.financial_qgan = FinancialQGAN(n_qubits)

    def price_european_call(self, strike: float, spot: float,
                            n_samples: int = 10000) -> float:
        """Price European call option using generated distribution."""
        returns = self.financial_qgan.sample_returns(n_samples)
        final_prices = spot * (1 + returns)

        payoffs = np.maximum(final_prices - strike, 0)
        # Assume risk-free rate is 0 for simplicity
        return np.mean(payoffs)

    def price_european_put(self, strike: float, spot: float,
                           n_samples: int = 10000) -> float:
        """Price European put option using generated distribution."""
        returns = self.financial_qgan.sample_returns(n_samples)
        final_prices = spot * (1 + returns)

        payoffs = np.maximum(strike - final_prices, 0)
        return np.mean(payoffs)
```

```python
class MolecularPropertyQGAN:
    """QGAN for molecular property generation."""

    def __init__(self, n_qubits: int, property_dim: int = 4):

        self.n_layers = 4

        # Classical post-processing to property space
        self.W_out = np.random.randn(property_dim, 2**n_qubits) * 0.1
        self.b_out = np.zeros(property_dim)

    def _ry_gate(self, theta: float) -> np.ndarray:
        c, s = np.cos(theta/2), np.sin(theta/2)
        return np.array([[c, -s], [s, c]])

    def _apply_gate(self, state: np.ndarray, gate: np.ndarray,
                    qubit: int) -> np.ndarray:
        n_qubits = int(np.log2(len(state)))
        full_gate = np.eye(1)
        for q in range(n_qubits):
            full_gate = np.kron(full_gate,
                               gate if q == qubit else np.eye(2))
        return full_gate @ state

    def generate_properties(self, z: np.ndarray) -> np.ndarray:
        """Generate molecular property vector."""
        dim = 2**self.n_qubits
        state = np.zeros(dim, dtype=complex)
        state[0] = 1.0

        # Encode noise
        for i, zi in enumerate(z[:self.n_qubits]):
            ry = self._ry_gate(np.pi * np.tanh(zi))
            state = self._apply_gate(state, ry, i)

        # Variational circuit
        params_per_layer = 2 * self.n_qubits
        for layer in range(self.n_layers):
            layer_params = self.gen_params[layer*params_per_layer:
                                           (layer+1)*params_per_layer]
            for q in range(self.n_qubits):
                ry = self._ry_gate(layer_params[q])
                state = self._apply_gate(state, ry, q)

        # Quantum features
        quantum_features = np.abs(state)**2

        # Map to property space
        properties = self.W_out @ quantum_features + self.b_out
        return properties

    def generate_molecule_batch(self, n_molecules: int) -> np.ndarray:
        """Generate batch of molecular properties."""
        molecules = []
        for _ in range(n_molecules):
            z = np.random.randn(self.n_qubits)
            props = self.generate_properties(z)
            molecules.append(props)
        return np.array(molecules)
```

## 5. Systematic Benchmarking

<Warning>
Fair benchmarking of QGANs requires careful attention to resource counts (qubits, gates, parameters), noise models, and comparison against properly tuned classical baselines.
</Warning>

```python
class QGANBenchmarker:
    """Systematic benchmarking framework for QGANs."""

    def __init__(self):
        self.results = {}

    def benchmark_state_generation(self, qgan: QuantumStateQGAN,
                                    target_states: List[np.ndarray],
                                    n_trials: int = 10) -> Dict:
        """Benchmark state generation fidelity."""
        fidelities = []

        for target in target_states:
            trial_fidelities = []
            for _ in range(n_trials):
                z = np.random.randn(qgan.n_qubits)
                generated = qgan.generate_state(z)
                fidelity = qgan.compute_fidelity(target, generated)
                trial_fidelities.append(fidelity)

            fidelities.append({
                'mean': np.mean(trial_fidelities),
                'std': np.std(trial_fidelities),
                'max': np.max(trial_fidelities)
            })

        return {
            'fidelities': fidelities,
            'overall_mean': np.mean([f['mean'] for f in fidelities]),
            'n_params': len(qgan.gen_params)
        }

    def benchmark_distribution_learning(self, qgan: DistributionQGAN,
                                          target_dist: np.ndarray,
                                          n_iterations: int = 100) -> Dict:
        """Benchmark distribution learning quality."""
        metrics = DistributionMetrics()
        history = {'kl': [], 'tv': [], 'hellinger': []}

        for _ in range(n_iterations):
            z = np.random.randn(qgan.n_qubits)
            generated = qgan.generate_distribution(z)

            history['kl'].append(metrics.kl_divergence(target_dist, generated))
            history['tv'].append(metrics.total_variation(target_dist, generated))
            history['hellinger'].append(metrics.hellinger_distance(target_dist, generated))

        return {
            'kl_divergence': {'mean': np.mean(history['kl']),
                             'std': np.std(history['kl'])},
            'total_variation': {'mean': np.mean(history['tv']),
                               'std': np.std(history['tv'])},
            'hellinger': {'mean': np.mean(history['hellinger']),
                         'std': np.std(history['hellinger'])},
            'n_params': len(qgan.gen_params)
        }

    def benchmark_training_convergence(self, trainer, real_data: np.ndarray,
                                        n_epochs: int = 100) -> Dict:
        """Benchmark training convergence speed and stability."""
        history = trainer.train(real_data, n_epochs, verbose=False)

        # Compute convergence metrics
        d_real = history.get('d_real', [0.5] * n_epochs)
        d_fake = history.get('d_fake', [0.5] * n_epochs)

        convergence_epoch = n_epochs
        for i in range(len(d_real)):
            if abs(d_real[i] - 0.5) < 0.1 and abs(d_fake[i] - 0.5) < 0.1:
                convergence_epoch = i
                break

        # Training stability
        if len(d_real) > 20:
            stability = np.std(d_real[-20:]) + np.std(d_fake[-20:])
        else:
            stability = np.std(d_real) + np.std(d_fake)

        return {
            'convergence_epoch': convergence_epoch,
            'final_d_real': d_real[-1] if d_real else 0.5,
            'final_d_fake': d_fake[-1] if d_fake else 0.5,
            'stability_score': 1 / (1 + stability),
            'total_epochs': n_epochs
        }

    def compare_architectures(self, architectures: Dict,
                               task: str, **task_params) -> Dict:
        """Compare multiple QGAN architectures on a task."""
        comparisons = {}

        for name, qgan in architectures.items():
            if task == 'state_generation':
                result = self.benchmark_state_generation(qgan, **task_params)
            elif task == 'distribution':
                result = self.benchmark_distribution_learning(qgan, **task_params)
            else:
                result = {}

            comparisons[name] = result

        return comparisons


class PerformanceMetrics:
    """Compute comprehensive performance metrics."""

    @staticmethod
    def compute_all_metrics(generated_samples: np.ndarray,
                            real_samples: np.ndarray) -> Dict:
        """Compute all relevant metrics."""
        metrics = {}

        # Sample-level metrics
        gen_mean = np.mean(generated_samples, axis=0)
        real_mean = np.mean(real_samples, axis=0)
        metrics['mean_diff'] = np.linalg.norm(gen_mean - real_mean)

        gen_cov = np.cov(generated_samples.T) if generated_samples.shape[0] > 1 else 0
        real_cov = np.cov(real_samples.T) if real_samples.shape[0] > 1 else 0
        if isinstance(gen_cov, np.ndarray) and isinstance(real_cov, np.ndarray):
            metrics['cov_diff'] = np.linalg.norm(gen_cov - real_cov)
        else:
            metrics['cov_diff'] = abs(gen_cov - real_cov)

        # Diversity metrics
        if generated_samples.shape[0] > 1:
            metrics['gen_variance'] = np.mean(np.var(generated_samples, axis=0))
        else:
            metrics['gen_variance'] = 0

        return metrics

    @staticmethod
    def quantum_resource_count(n_qubits: int, n_layers: int,
                                gates_per_layer: int = 2) -> Dict:
        """Count quantum resources used."""
        return {
            'n_qubits': n_qubits,
            'n_layers': n_layers,
            'total_gates': n_layers * n_qubits * gates_per_layer,
            'total_params': n_layers * n_qubits * gates_per_layer,
            'circuit_depth': n_layers * gates_per_layer
        }
```

```python
    """Run complete QGAN benchmark experiments."""

    def __init__(self):
        self.benchmarker = QGANBenchmarker()

    def run_state_generation_experiment(self, n_qubits_range: List[int],
                                         target_states: List[str]
                                         ) -> Dict:
        """Run state generation experiments across qubit counts."""
        results = {}
        state_gen = QuantumStateGenerator(max(n_qubits_range))

        for n_qubits in n_qubits_range:
            qgan = QuantumStateQGAN(n_qubits)
            state_gen_local = QuantumStateGenerator(n_qubits)

            targets = []
            for state_name in target_states:
                if state_name == 'ghz':
                    targets.append(state_gen_local.ghz_state())
                elif state_name == 'w':
                    targets.append(state_gen_local.w_state())
                elif state_name == 'random':
                    targets.append(state_gen_local.random_state())

            benchmark = self.benchmarker.benchmark_state_generation(
                qgan, targets, n_trials=5)

            results[n_qubits] = {
                'fidelity': benchmark['overall_mean'],
                'n_params': benchmark['n_params'],
                'resources': PerformanceMetrics.quantum_resource_count(
                    n_qubits, qgan.n_layers)
            }

        return results

    def run_distribution_experiment(self, n_qubits_range: List[int],
                                     target_type: str = 'gaussian'
                                     ) -> Dict:
        """Run distribution learning experiments."""
        results = {}

        for n_qubits in n_qubits_range:
            qgan = DistributionQGAN(n_qubits)
            n_outcomes = 2**n_qubits

            # Create target distribution
            if target_type == 'gaussian':
                x = np.linspace(-3, 3, n_outcomes)
                target = np.exp(-x**2 / 2)
                target /= target.sum()
            elif target_type == 'bimodal':
                x = np.linspace(-3, 3, n_outcomes)
                target = np.exp(-(x-1)**2/0.5) + np.exp(-(x+1)**2/0.5)
                target /= target.sum()
            else:
                target = np.ones(n_outcomes) / n_outcomes

            benchmark = self.benchmarker.benchmark_distribution_learning(
                qgan, target)

            results[n_qubits] = {
                'kl_divergence': benchmark['kl_divergence']['mean'],
                'total_variation': benchmark['total_variation']['mean'],
                'n_params': benchmark['n_params']
            }

        return results

    def generate_report(self, experiment_results: Dict) -> str:
        """Generate human-readable benchmark report."""
        report = "QGAN Benchmark Report\n"
        report += "=" * 50 + "\n\n"

        for exp_name, results in experiment_results.items():
            report += f"Experiment: {exp_name}\n"
            report += "-" * 30 + "\n"

            for config, metrics in results.items():
                report += f"  Configuration: {config}\n"
                for metric, value in metrics.items():
                    if isinstance(value, float):
                        report += f"    {metric}: {value:.4f}\n"
                    elif isinstance(value, dict):
                        report += f"    {metric}:\n"
                        for k, v in value.items():
                            report += f"      {k}: {v}\n"
                    else:
                        report += f"    {metric}: {value}\n"
                report += "\n"

        return report
```
