## 1. Why Second Quantization?

<Definition term="Second Quantization">
A formulation of quantum mechanics where the wavefunction itself becomes an operator, and particle creation/annihilation are the fundamental operations. It naturally handles variable particle numbers and identical particle statistics.
</Definition>

```python
def motivation():
    """Explain why second quantization is useful."""

    print("Why Second Quantization?")
    print("=" * 60)

    print("\nFirst Quantization Problems:")
    print("  • Wavefunctions for N particles: Ψ(r₁, r₂, ..., rₙ)")
    print("  • Antisymmetrization is cumbersome")
    print("  • Particle number is fixed")
    print("  • Hard to describe excitations")

    print("\nSecond Quantization Advantages:")
    print("  • States labeled by occupation numbers")
    print("  • Antisymmetry built into operators")
    print("  • Variable particle number natural")
    print("  • Clean notation for many-body physics")
    print("  • Direct connection to quantum computing")

    print("\nKey Idea:")
    print("  Instead of: 'where is particle i?'")
    print("  Ask: 'how many particles in state i?'")

motivation()
```

## 2. The Vacuum State

<Definition term="Vacuum State">
The state with no particles, denoted |0⟩ or |vac⟩. It serves as the reference state from which all other states are built by applying creation operators.
</Definition>

```python
def vacuum_state():
    """Define the vacuum state."""

    print("The Vacuum State |0⟩")
    print("=" * 60)

    print("\nDefinition:")
    print("  |0⟩ = state with zero particles")
    print("  Also written as |vac⟩ or |Ω⟩")

    print("\nProperties:")
    print("  • Normalized: ⟨0|0⟩ = 1")
    print("  • Annihilated by all annihilation operators:")
    print("    aᵢ|0⟩ = 0  for all i")
    print("  • Not the zero vector! |0⟩ ≠ 0")

    print("\nOccupation Number Representation:")
    print("  |0⟩ = |0, 0, 0, ...⟩")
    print("  (zero particles in every orbital)")

    print("\nBuilding States:")
    print("  |n₁, n₂, ...⟩ = (a₁†)^n₁ (a₂†)^n₂ ... |0⟩")
    print("  Creation operators add particles to vacuum")

vacuum_state()
```

## 3. Creation Operators

<Formula>
$$\hat{a}_i^\dagger |n_1, n_2, \ldots, n_i, \ldots\rangle = \sqrt{n_i + 1} |n_1, n_2, \ldots, n_i + 1, \ldots\rangle$$
</Formula>

```python
import numpy as np

def creation_operators():
    """Explain creation operators."""

    print("Creation Operators a†")
    print("=" * 60)

    print("\nNotation:")
    print("  a†ᵢ = creation operator for state i")
    print("  'a-dagger-i' or 'a-i-dagger'")

    print("\nAction (Bosons):")
    print("  a†ᵢ |..., nᵢ, ...⟩ = √(nᵢ+1) |..., nᵢ+1, ...⟩")
    print()
    print("  Adds one particle to state i")
    print("  Factor √(nᵢ+1) ensures proper normalization")

    print("\nExamples:")
    print("-" * 45)

    print("\n  a†₁ |0,0,0⟩ = |1,0,0⟩")
    print("  (Create particle in state 1)")

    print("\n  a†₂ |1,0,0⟩ = |1,1,0⟩")
    print("  (Create particle in state 2)")

    print("\n  a†₁ |1,0,0⟩ = √2 |2,0,0⟩")
    print("  (Add second particle to state 1, bosons)")

    print("\nHermitian Conjugate:")
    print("  (a†ᵢ)† = aᵢ")
    print("  Creation and annihilation are adjoints")

creation_operators()
```

## 4. Annihilation Operators

<Formula>
$$\hat{a}_i |n_1, n_2, \ldots, n_i, \ldots\rangle = \sqrt{n_i} |n_1, n_2, \ldots, n_i - 1, \ldots\rangle$$
</Formula>

```python
def annihilation_operators():
    """Explain annihilation operators."""

    print("Annihilation Operators a")
    print("=" * 60)

    print("\nAction (Bosons):")
    print("  aᵢ |..., nᵢ, ...⟩ = √nᵢ |..., nᵢ-1, ...⟩")
    print()
    print("  Removes one particle from state i")
    print("  Factor √nᵢ ensures proper normalization")

    print("\nSpecial Case - Empty State:")
    print("  aᵢ |..., 0, ...⟩ = 0  (the zero vector)")
    print("  Cannot remove particle that isn't there!")

    print("\nExamples:")
    print("-" * 45)

    print("\n  a₁ |1,0,0⟩ = |0,0,0⟩")
    print("  (Remove particle from state 1)")

    print("\n  a₁ |2,0,0⟩ = √2 |1,0,0⟩")
    print("  (Remove one of two particles)")

    print("\n  a₁ |0,1,0⟩ = 0")
    print("  (State 1 empty, get zero vector)")

    print("\nMatrix Representation (truncated basis):")
    # For a 3-level system
    n_max = 3
    a = np.zeros((n_max, n_max))
    for n in range(1, n_max):
        a[n-1, n] = np.sqrt(n)
    print(f"\n  a (bosonic, up to n={n_max-1}):")
    print(a)

annihilation_operators()
```

## 5. Bosonic Commutation Relations

<Definition term="Commutation Relations">
Algebraic rules defining how operators combine. For bosons, creation and annihilation operators satisfy canonical commutation relations: [a, a†] = 1.
</Definition>

```python
def bosonic_commutation():
    """Explain bosonic commutation relations."""

    print("Bosonic Commutation Relations")
    print("=" * 60)

    print("\nCanonical Commutation Relations:")
    print("  [aᵢ, a†ⱼ] = δᵢⱼ")
    print("  [aᵢ, aⱼ] = 0")
    print("  [a†ᵢ, a†ⱼ] = 0")
    print()
    print("  where [A, B] = AB - BA")

    print("\nFor Same State (i = j):")
    print("  [a, a†] = aa† - a†a = 1")
    print("  This means: aa† = a†a + 1")

    print("\nPhysical Meaning:")
    print("  • Bosons can occupy same state")
    print("  • Order of creation matters (slightly)")
    print("  • aa† ≠ a†a (don't commute)")

    print("\nProof for Single Mode:")
    print("-" * 45)
    print("  Acting on |n⟩:")
    print("  aa†|n⟩ = a(√(n+1)|n+1⟩) = √(n+1)·√(n+1)|n⟩ = (n+1)|n⟩")
    print("  a†a|n⟩ = a†(√n|n-1⟩) = √n·√n|n⟩ = n|n⟩")
    print("  (aa† - a†a)|n⟩ = (n+1-n)|n⟩ = |n⟩ ✓")

bosonic_commutation()
```

## 6. Fermionic Anti-commutation Relations

<Formula>
$$\{c_i, c_j^\dagger\} = \delta_{ij}, \quad \{c_i, c_j\} = 0, \quad \{c_i^\dagger, c_j^\dagger\} = 0$$
</Formula>

```python
def fermionic_anticommutation():
    """Explain fermionic anti-commutation relations."""

    print("Fermionic Anti-commutation Relations")
    print("=" * 60)

    print("\nFermion Operators: c, c†")
    print("  (Different letter to distinguish from bosons)")

    print("\nCanonical Anti-commutation Relations:")
    print("  {cᵢ, c†ⱼ} = δᵢⱼ")
    print("  {cᵢ, cⱼ} = 0")
    print("  {c†ᵢ, c†ⱼ} = 0")
    print()
    print("  where {A, B} = AB + BA (anticommutator)")

    print("\nKey Consequences:")
    print("-" * 45)

    print("\n1. Pauli Exclusion:")
    print("   (c†ᵢ)² = 0")
    print("   Cannot create two fermions in same state!")
    print("   Proof: {c†ᵢ, c†ᵢ} = 2(c†ᵢ)² = 0")

    print("\n2. Occupation Numbers 0 or 1 only:")
    print("   nᵢ ∈ {0, 1} for fermions")

    print("\n3. Sign from Ordering:")
    print("   c†ᵢc†ⱼ = -c†ⱼc†ᵢ  (i ≠ j)")
    print("   Order of creation matters (sign flip)!")

    print("\nFermionic States:")
    print("  |1,0,1,0,...⟩ means:")
    print("  Orbital 1: occupied")
    print("  Orbital 2: empty")
    print("  Orbital 3: occupied")
    print("  etc.")

fermionic_anticommutation()
```

<Warning>
The sign conventions in fermionic operators are crucial! The order in which you apply operators matters and affects the sign of the result. Always track the ordering carefully.
</Warning>

## 7. Building Fermionic States

```python
def fermionic_states():
    """Show how to build fermionic states."""

    print("Building Fermionic Many-Body States")
    print("=" * 60)

    print("\nSingle Particle States:")
    print("  |1ᵢ⟩ = c†ᵢ |0⟩")
    print("  One fermion in orbital i")

    print("\nTwo Particle States:")
    print("  |1ᵢ, 1ⱼ⟩ = c†ᵢc†ⱼ |0⟩  (i < j by convention)")
    print()
    print("  Important: c†ᵢc†ⱼ = -c†ⱼc†ᵢ")
    print("  Order defines the phase!")

    print("\nSlater Determinant Connection:")
    print("-" * 45)
    print("  The state c†₁c†₂...c†ₙ|0⟩ corresponds to:")
    print("  Ψ = (1/√N!) det|χ₁(r₁) χ₂(r₁) ...|")
    print("                  |χ₁(r₂) χ₂(r₂) ...|")
    print("                  |  ...    ...   ...|")
    print()
    print("  Antisymmetry is automatic from {c†,c†} = 0!")

    print("\nExample: He Ground State")
    print("  |He⟩ = c†₁ₛ↑ c†₁ₛ↓ |0⟩")
    print("  Two electrons: 1s spin-up and 1s spin-down")

    print("\nExample: Li Ground State")
    print("  |Li⟩ = c†₁ₛ↑ c†₁ₛ↓ c†₂ₛ↑ |0⟩")
    print("  Three electrons in 1s², 2s¹ configuration")

fermionic_states()
```

## 8. Matrix Representations

```python
import numpy as np

def matrix_representations():
    """Show matrix forms of creation/annihilation operators."""

    print("Matrix Representations")
    print("=" * 60)

    print("\nSingle Fermionic Mode (2×2 matrices):")
    print("-" * 45)

    c = np.array([[0, 1],
                  [0, 0]])
    c_dag = np.array([[0, 0],
                      [1, 0]])

    print("  Basis: |0⟩, |1⟩")
    print("\n  c (annihilation):")
    print(c)
    print("\n  c† (creation):")
    print(c_dag)

    print("\n  Verification:")
    anticomm = c @ c_dag + c_dag @ c
    print(f"  {{c, c†}} = cc† + c†c =")
    print(anticomm)
    print("  = I ✓")

    print("\nTwo Fermionic Modes (4×4 matrices):")
    print("-" * 45)
    print("  Basis: |00⟩, |01⟩, |10⟩, |11⟩")
    print()
    print("  Need Jordan-Wigner transformation:")
    print("  c₁ = σ⁻ ⊗ I")
    print("  c₂ = σᶻ ⊗ σ⁻")
    print()
    print("  The σᶻ (Z operator) tracks fermion parity!")

    # Build 2-mode operators
    I2 = np.eye(2)
    sigma_z = np.array([[1, 0], [0, -1]])
    sigma_minus = np.array([[0, 1], [0, 0]])

    c1 = np.kron(sigma_minus, I2)
    c2 = np.kron(sigma_z, sigma_minus)

    print("\n  c₁:")
    print(c1)
    print("\n  c₂:")
    print(c2)

matrix_representations()
```

<Tip>
The Jordan-Wigner transformation is essential for mapping fermionic operators to qubit operators. The Z gates ("Jordan-Wigner strings") enforce the correct fermionic anticommutation.
</Tip>

## Summary

In this lesson, you learned:

1. **Second quantization** uses occupation numbers instead of coordinates
2. **Vacuum state** |0⟩ is the reference with no particles
3. **Creation operators** a† add particles to states
4. **Annihilation operators** a remove particles from states
5. **Bosons** satisfy commutation relations [a, a†] = 1
6. **Fermions** satisfy anticommutation relations {c, c†} = 1
7. **Pauli exclusion** follows from (c†)² = 0
8. **Matrix representations** connect to quantum computing
