## 1. The Number Operator

<Definition term="Number Operator">
The operator n̂ = a†a that counts the number of particles in a given quantum state. Its eigenvalues are the occupation numbers (0, 1, 2, ... for bosons; 0 or 1 for fermions).
</Definition>

```python
import numpy as np

def number_operator():
    """Explain the number operator."""

    print("The Number Operator n̂ = a†a")
    print("=" * 60)

    print("\nDefinition:")
    print("  n̂ = a†a  (bosons)")
    print("  n̂ = c†c  (fermions)")

    print("\nEigenvalue Equation:")
    print("  n̂|n⟩ = n|n⟩")
    print()
    print("  The occupation number n is the eigenvalue")

    print("\nProof (using bosonic operators):")
    print("  n̂|n⟩ = a†a|n⟩")
    print("       = a†(√n |n-1⟩)")
    print("       = √n · √n |n⟩")
    print("       = n|n⟩ ✓")

    print("\nFor Fermions:")
    print("  n̂|0⟩ = c†c|0⟩ = 0")
    print("  n̂|1⟩ = c†c|1⟩ = c†|0⟩ = |1⟩")
    print()
    print("  Eigenvalues: 0 or 1 only!")

    print("\nMatrix Form (bosonic, truncated):")
    n_max = 5
    n_op = np.diag(range(n_max))
    print(f"  n̂ (up to n={n_max-1}):")
    print(n_op)

number_operator()
```

## 2. Properties of the Number Operator

<Formula>
$$[\hat{n}, \hat{a}] = -\hat{a}, \quad [\hat{n}, \hat{a}^\dagger] = \hat{a}^\dagger$$
</Formula>

```python
def number_operator_properties():
    """Properties and commutation relations of n̂."""

    print("Properties of the Number Operator")
    print("=" * 60)

    print("\nHermiticity:")
    print("  n̂† = (a†a)† = a†a = n̂")
    print("  Number operator is Hermitian (observable!)")

    print("\nCommutation with a and a†:")
    print("-" * 45)
    print("  [n̂, a] = [a†a, a] = a†[a, a] + [a†, a]a")
    print("         = 0 + (-1)a = -a")
    print()
    print("  [n̂, a†] = [a†a, a†] = a†[a, a†] + [a†, a†]a")
    print("          = a†(1) + 0 = a†")

    print("\nPhysical Interpretation:")
    print("  [n̂, a] = -a  means:")
    print("    n̂(a|n⟩) = (n-1)(a|n⟩)")
    print("    After annihilation, particle count decreases by 1")
    print()
    print("  [n̂, a†] = a† means:")
    print("    n̂(a†|n⟩) = (n+1)(a†|n⟩)")
    print("    After creation, particle count increases by 1")

    print("\nLadder Operator Interpretation:")
    print("  a  = lowering operator (n → n-1)")
    print("  a† = raising operator (n → n+1)")

number_operator_properties()
```

## 3. Multi-Mode Number Operators

```python
def multi_mode_number():
    """Number operators for multiple modes."""

    print("Multi-Mode Systems")
    print("=" * 60)

    print("\nFor M Modes/Orbitals:")
    print("  Each mode i has its own number operator:")
    print("  n̂ᵢ = aᵢ†aᵢ  (bosons)")
    print("  n̂ᵢ = cᵢ†cᵢ  (fermions)")

    print("\nTotal Number Operator:")
    print("  N̂ = Σᵢ n̂ᵢ = Σᵢ aᵢ†aᵢ")
    print()
    print("  Counts total particles in the system")

    print("\nEigenvalues:")
    print("  N̂|n₁, n₂, ...⟩ = (n₁ + n₂ + ...)|n₁, n₂, ...⟩")
    print("                 = N|n₁, n₂, ...⟩")

    print("\nExample: 3-Orbital System")
    print("-" * 45)
    print("  State: |2, 0, 1⟩")
    print("    n̂₁|2,0,1⟩ = 2|2,0,1⟩")
    print("    n̂₂|2,0,1⟩ = 0|2,0,1⟩")
    print("    n̂₃|2,0,1⟩ = 1|2,0,1⟩")
    print("    N̂|2,0,1⟩ = 3|2,0,1⟩")

    print("\nCommutation (different modes):")
    print("  [n̂ᵢ, n̂ⱼ] = 0  (all modes commute)")
    print("  [n̂ᵢ, aⱼ] = -δᵢⱼ aⱼ")
    print("  [n̂ᵢ, aⱼ†] = δᵢⱼ aⱼ†")

multi_mode_number()
```

## 4. Occupation Number Representation

<Definition term="Occupation Number Representation">
A basis for Fock space where states are labeled by the occupation numbers of each single-particle state: |n₁, n₂, n₃, ...⟩.
</Definition>

```python
def occupation_representation():
    """The occupation number basis."""

    print("Occupation Number Representation")
    print("=" * 60)

    print("\nBasis States:")
    print("  |n₁, n₂, n₃, ...⟩")
    print("  nᵢ = number of particles in orbital i")

    print("\nFor Bosons:")
    print("  nᵢ ∈ {0, 1, 2, 3, ...}")
    print("  Any number of particles per orbital")

    print("\nFor Fermions:")
    print("  nᵢ ∈ {0, 1}")
    print("  At most one particle per orbital (Pauli)")

    print("\nOrthonormality:")
    print("  ⟨n₁', n₂', ...|n₁, n₂, ...⟩ = δₙ₁'ₙ₁ δₙ₂'ₙ₂ ...")
    print("  = 1 if all occupation numbers match, else 0")

    print("\nCompleteness (fermions, M orbitals):")
    print("  Σₙ₁∈{0,1} Σₙ₂∈{0,1} ... |n₁,n₂,...⟩⟨n₁,n₂,...| = I")
    print()
    print("  Total dimension: 2^M states")

    print("\nExample States (2 orbitals, fermions):")
    print("-" * 45)
    states = [
        ('|0,0⟩', 'Vacuum', 0),
        ('|1,0⟩', 'One particle in orbital 1', 1),
        ('|0,1⟩', 'One particle in orbital 2', 1),
        ('|1,1⟩', 'Both orbitals occupied', 2),
    ]
    for state, desc, n in states:
        print(f"  {state}: {desc} (N={n})")

occupation_representation()
```

## 5. Fock Space

<Definition term="Fock Space">
The Hilbert space of variable particle number, constructed as the direct sum of N-particle Hilbert spaces: F = H₀ ⊕ H₁ ⊕ H₂ ⊕ ...
</Definition>

```python
def fock_space():
    """Structure of Fock space."""

    print("Fock Space Structure")
    print("=" * 60)

    print("\nDefinition:")
    print("  F = H₀ ⊕ H₁ ⊕ H₂ ⊕ ...")
    print()
    print("  H₀ = vacuum sector (1-dimensional)")
    print("  H₁ = 1-particle sector")
    print("  H₂ = 2-particle sector")
    print("  etc.")

    print("\nDimensions (M orbitals):")
    print("-" * 45)

    print("\nBosons:")
    print("  dim(Hₙ) = C(n+M-1, n) = (n+M-1)! / (n!(M-1)!)")
    print("  Total: infinite-dimensional")

    print("\nFermions:")
    print("  dim(Hₙ) = C(M, n) = M! / (n!(M-n)!)")
    print("  Total: 2^M dimensional")

    print("\nExample: 4 Fermionic Orbitals")
    print("-" * 45)
    M = 4
    from math import comb
    print(f"  M = {M} orbitals")
    for n in range(M + 1):
        dim = comb(M, n)
        print(f"  H_{n}: {dim} states with {n} particles")
    print(f"  Total Fock space: {2**M} dimensions")

fock_space()
```

## 6. Number Operator Matrix Representation

```python
import numpy as np

def number_operator_matrix():
    """Build number operator matrices."""

    print("Number Operator Matrices")
    print("=" * 60)

    print("\nSingle Fermionic Mode:")
    print("-" * 45)

    # Basis: |0⟩, |1⟩
    n_single = np.array([[0, 0],
                         [0, 1]])
    print("  n̂ (2×2):")
    print(n_single)

    print("\nTwo Fermionic Modes:")
    print("-" * 45)
    print("  Basis: |00⟩, |01⟩, |10⟩, |11⟩")

    # Number operators for 2 modes
    I = np.eye(2)
    n1_local = np.array([[0, 0], [0, 1]])

    n1 = np.kron(n1_local, I)  # n̂₁ = n₁ ⊗ I
    n2 = np.kron(I, n1_local)  # n̂₂ = I ⊗ n₂
    N_total = n1 + n2

    print("\n  n̂₁ (counts particles in mode 1):")
    print(n1)
    print("\n  n̂₂ (counts particles in mode 2):")
    print(n2)
    print("\n  N̂ = n̂₁ + n̂₂ (total number):")
    print(N_total)

    print("\nVerification:")
    basis_labels = ['|00⟩', '|01⟩', '|10⟩', '|11⟩']
    for i, label in enumerate(basis_labels):
        print(f"  N̂{label} = {int(N_total[i,i])}{label}")

number_operator_matrix()
```

## 7. Number Conservation

<Tip>
Many Hamiltonians conserve particle number: [Ĥ, N̂] = 0. This means we can work in subspaces of fixed particle number, dramatically reducing computational cost.
</Tip>

```python
def number_conservation():
    """Number conservation in quantum systems."""

    print("Particle Number Conservation")
    print("=" * 60)

    print("\nConservation Condition:")
    print("  [Ĥ, N̂] = 0")
    print()
    print("  If Hamiltonian commutes with number operator,")
    print("  particle number is conserved.")

    print("\nPhysical Meaning:")
    print("  • Energy eigenstates have definite particle number")
    print("  • N is a good quantum number")
    print("  • Fock space splits into independent sectors")

    print("\nOperators that Conserve Number:")
    print("-" * 45)
    print("  n̂ᵢ = aᵢ†aᵢ       ✓ (diagonal in occupation)")
    print("  aᵢ†aⱼ            ✓ (hopping: removes j, adds i)")
    print("  aᵢ†aⱼ†aₖaₗ       ✓ (two-body: balanced creation/annihilation)")

    print("\nOperators that Don't Conserve Number:")
    print("-" * 45)
    print("  aᵢ               ✗ (removes particle)")
    print("  aᵢ†              ✗ (adds particle)")
    print("  aᵢ†aⱼ†           ✗ (pair creation)")
    print("  aᵢaⱼ             ✗ (pair annihilation)")

    print("\nComputational Advantage:")
    print("  • 8 orbitals, 4 electrons: C(8,4) = 70 states")
    print("  • Full Fock space: 2⁸ = 256 states")
    print("  • Factor of ~4 reduction!")

number_conservation()
```

## 8. Expectation Values

```python
def expectation_values():
    """Calculate expectation values of number operators."""

    print("Expectation Values of Number Operators")
    print("=" * 60)

    print("\nFor Occupation Number States:")
    print("  ⟨n₁, n₂, ...|n̂ᵢ|n₁, n₂, ...⟩ = nᵢ")
    print()
    print("  The occupation number is the expectation value")

    print("\nFor Superposition States:")
    print("-" * 45)
    print("  |ψ⟩ = Σₙ cₙ|n⟩")
    print("  ⟨n̂⟩ = Σₙ |cₙ|² n")
    print()
    print("  Weighted average of occupation numbers")

    print("\nExample:")
    print("  |ψ⟩ = (|0⟩ + √2|1⟩ + |2⟩)/2")
    print()
    print("  Probabilities: P(0)=1/4, P(1)=2/4, P(2)=1/4")
    print("  ⟨n̂⟩ = 0×(1/4) + 1×(2/4) + 2×(1/4)")
    print("      = 0 + 0.5 + 0.5 = 1.0")

    print("\nVariance:")
    print("  ⟨Δn̂²⟩ = ⟨n̂²⟩ - ⟨n̂⟩²")
    print()
    print("  For number states: ⟨Δn̂²⟩ = 0 (sharp)")
    print("  For coherent states: ⟨Δn̂²⟩ = ⟨n̂⟩ (Poissonian)")

expectation_values()
```

## Summary

In this lesson, you learned:

1. **Number operator** n̂ = a†a counts particles in a state
2. **Eigenvalues** are occupation numbers (0,1,2,... or 0,1)
3. **Multi-mode** systems have n̂ᵢ for each orbital
4. **Total number** N̂ = Σᵢ n̂ᵢ counts all particles
5. **Occupation basis** |n₁, n₂, ...⟩ spans Fock space
6. **Fock space** has dimension 2^M for M fermionic orbitals
7. **Number conservation** [Ĥ, N̂] = 0 enables simplifications
8. **Expectation values** give average occupation
