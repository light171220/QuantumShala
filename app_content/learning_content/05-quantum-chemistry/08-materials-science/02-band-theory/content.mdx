## 1. Tight Binding Model

<Definition term="Tight Binding Model">
A method for calculating band structure by expanding crystal wavefunctions as linear combinations of atomic orbitals, treating the overlap between neighboring atoms as a perturbation.
</Definition>

```python
import numpy as np

def tight_binding():
    """Tight binding model for band structure."""

    print("Tight Binding Model")
    print("=" * 60)

    print("\nBasic Idea:")
    print("-" * 45)
    print("  Start with atomic orbitals φ_n(r - R)")
    print("  Construct Bloch states:")
    print("  ψ_k(r) = Σ_R e^(ik·R) φ(r - R)")
    print()
    print("  Overlap between neighbors → band formation")

    print("\n1D Chain Example:")
    print("-" * 45)
    print("  One s-orbital per atom, spacing a")
    print()
    print("  H_ij = ε₀δᵢⱼ - t(δᵢ,ⱼ₊₁ + δᵢ,ⱼ₋₁)")
    print()
    print("  ε₀ = on-site energy")
    print("  t = hopping integral (overlap)")
    print()
    print("  Band dispersion:")
    print("  E(k) = ε₀ - 2t cos(ka)")

    # Plot band structure
    print("\nBand Parameters:")
    print("-" * 45)
    print("  Band width: W = 4t")
    print("  Bottom of band: E = ε₀ - 2t (k = 0)")
    print("  Top of band: E = ε₀ + 2t (k = π/a)")
    print()
    print("  Effective mass at band bottom:")
    print("  m* = ℏ²/(∂²E/∂k²) = ℏ²/(2ta²)")

    print("\n3D Generalization:")
    print("-" * 45)
    print("  Simple cubic with s-orbitals:")
    print("  E(k) = ε₀ - 2t[cos(kₓa) + cos(kᵧa) + cos(kᵤa)]")
    print()
    print("  Band width: W = 12t (3D)")
    print("  6 nearest neighbors → 6 terms")

tight_binding()
```

<Formula name="Tight Binding Band Dispersion (1D)">
$$E(k) = \epsilon_0 - 2t\cos(ka)$$

Where $\epsilon_0$ is the on-site energy, $t$ is the hopping integral between neighbors, $k$ is the wavevector, and $a$ is the lattice constant.
</Formula>

## 2. Nearly Free Electron Model

```python
def nearly_free_electron():
    """Nearly free electron model."""

    print("Nearly Free Electron Model")
    print("=" * 60)

    print("\nFree Electron Dispersion:")
    print("-" * 45)
    print("  E(k) = ℏ²k²/(2m)")
    print("  Parabolic, no gaps")

    print("\nWeak Periodic Potential:")
    print("-" * 45)
    print("  V(x) = Σ_G V_G e^(iGx)")
    print()
    print("  Treat as perturbation")
    print("  Gaps open at zone boundaries!")

    print("\nBand Gap Formation:")
    print("-" * 45)
    print("  At k = G/2 (zone boundary):")
    print()
    print("  E_± = E₀ ± |V_G|")
    print()
    print("  Gap = 2|V_G|")
    print()
    print("  Physical picture:")
    print("    • Standing waves at zone boundary")
    print("    • cos(Gx/2): concentrated on atoms (lower E)")
    print("    • sin(Gx/2): concentrated between (higher E)")

    print("\nExtended Zone Scheme:")
    print("-" * 45)
    print("  Plot E(k) for all k")
    print("  Free electron parabolas + gaps at G/2")
    print()
    print("  Reduced Zone Scheme:")
    print("  Fold all bands into 1st BZ")
    print("  Multiple bands, single zone")

    # Example calculation
    print("\nExample: Gap at Zone Boundary")
    print("-" * 45)
    m = 9.11e-31  # kg
    a = 5e-10  # m
    hbar = 1.055e-34
    k_zone = np.pi / a
    E_0 = (hbar * k_zone)**2 / (2 * m)
    print(f"  a = 5 Å")
    print(f"  k at zone edge = π/a = {k_zone:.2e} m⁻¹")
    print(f"  E₀ (free electron) = {E_0/1.6e-19:.2f} eV")
    print(f"  If |V_G| = 2 eV, gap = 4 eV")

nearly_free_electron()
```

## 4. Effective Mass and DOS

```python
def effective_mass_dos():
    """Effective mass and density of states."""

    print("Effective Mass and Density of States")
    print("=" * 60)

    print("\nEffective Mass:")
    print("-" * 45)
    print("  Definition:")
    print("  (m*)⁻¹ = (1/ℏ²) ∂²E/∂k²")
    print()
    print("  For parabolic band near minimum:")
    print("  E(k) ≈ E₀ + ℏ²k²/(2m*)")
    print()
    print("  Physical meaning:")
    print("    • Response to external field")
    print("    • Can be < m₀ (light carriers)")
    print("    • Can be > m₀ (heavy carriers)")
    print("    • Can be negative (holes)")

    print("\nMaterial Examples:")
    print("-" * 45)
    materials = [
        ("GaAs", 0.067, 0.45, "Light electrons"),
        ("Si", 0.26, 0.49, "Moderate"),
        ("Ge", 0.12, 0.28, "Light"),
        ("InSb", 0.014, 0.43, "Very light e⁻"),
    ]
    print(f"  {'Material':<10} {'m*_e/m₀':<10} {'m*_h/m₀':<10} {'Notes'}")
    print("  " + "-" * 45)
    for mat, me, mh, notes in materials:
        print(f"  {mat:<10} {me:<10} {mh:<10} {notes}")

    print("\nDensity of States:")
    print("-" * 45)
    print("  g(E) = number of states per unit energy")
    print()
    print("  3D parabolic band:")
    print("  g(E) = (1/2π²)(2m*/ℏ²)^(3/2) √(E-E_c)")
    print()
    print("  At band edge: g(E) → 0")
    print("  Van Hove singularities at critical points")

    print("\nDOS Features:")
    print("-" * 45)
    print("  1D: g(E) ∝ 1/√E (diverges at edges)")
    print("  2D: g(E) = const (step function)")
    print("  3D: g(E) ∝ √E (smooth onset)")

effective_mass_dos()
```

<Tip>
A small effective mass means high carrier mobility. GaAs has much lighter electrons than Si, making it attractive for high-speed electronics despite being more expensive.
</Tip>

## 5. Direct vs Indirect Gaps

```python
def direct_indirect():
    """Direct vs indirect band gaps."""

    print("Direct vs Indirect Band Gaps")
    print("=" * 60)

    print("\nDefinitions:")
    print("-" * 45)
    print("  Direct gap: VB max and CB min at same k")
    print("    E.g., GaAs, ZnO, GaN")
    print()
    print("  Indirect gap: VB max and CB min at different k")
    print("    E.g., Si, Ge, Diamond")

    print("\nOptical Transitions:")
    print("-" * 45)
    print("  Direct:")
    print("    hν = E_g (photon only)")
    print("    Strong absorption/emission")
    print("    → Good for LEDs, lasers")
    print()
    print("  Indirect:")
    print("    hν = E_g ± ℏω_phonon")
    print("    Requires phonon assistance")
    print("    Weak absorption/emission")
    print("    → Si cannot make efficient LEDs")

    print("\nBand Structure Examples:")
    print("-" * 45)
    print("  Si: VB max at Γ, CB min near X (0.85×Γ-X)")
    print("      Direct gap: 3.4 eV")
    print("      Indirect gap: 1.1 eV")
    print()
    print("  GaAs: Both at Γ point")
    print("      Direct gap: 1.42 eV")
    print("      → Efficient light emission!")

    print("\nImplications:")
    print("-" * 45)
    implications = [
        ("Solar cells", "Indirect OK (thick absorber)", "Si dominates"),
        ("LEDs/Lasers", "Need direct gap", "III-V compounds"),
        ("Detectors", "Direct preferred", "Faster response"),
        ("Transistors", "Either works", "Si for cost"),
    ]
    print(f"  {'Application':<15} {'Gap Type':<25} {'Material'}")
    print("  " + "-" * 55)
    for app, gap, mat in implications:
        print(f"  {app:<15} {gap:<25} {mat}")

direct_indirect()
```
