## 1. Errors in NISQ Devices

<Definition term="NISQ">
Noisy Intermediate-Scale Quantum devices have 50-1000 qubits with error rates of 10⁻³ to 10⁻², insufficient for full error correction but capable of useful computations with error mitigation.
</Definition>


## 2. Zero-Noise Extrapolation (ZNE)

<Formula>
$$E_{\text{mitigated}} = \lim_{\lambda \to 0} E(\lambda) \approx \sum_k c_k E(\lambda_k)$$
</Formula>

```python
import numpy as np

def zero_noise_extrapolation():
    """Zero-noise extrapolation technique."""

    print("Zero-Noise Extrapolation (ZNE)")
    print("=" * 60)

    print("\nKey Idea:")
    print("  Run circuits at multiple noise levels")
    print("  Extrapolate to zero-noise limit")

    print("\nNoise Amplification Methods:")
    print("-" * 45)
    methods = [
        ("Pulse stretching", "Stretch gate pulses by factor λ"),
        ("Unitary folding", "Replace U with U·U†·U (factor 3)"),
        ("Identity insertion", "Insert I = U·U† pairs"),
    ]
    for method, desc in methods:
        print(f"  {method}:")
        print(f"    {desc}")
        print()

    print("Unitary Folding:")
    print("-" * 45)
    print("  Original:  U")
    print("  λ = 1:     U")
    print("  λ = 3:     U · U† · U")
    print("  λ = 5:     U · U† · U · U† · U")

    print("\nExtrapolation Methods:")
    print("-" * 45)

    # Example data
    lambdas = np.array([1.0, 3.0, 5.0])
    energies = np.array([-1.10, -1.05, -1.00])

    print("  Linear extrapolation:")
    print("    E(λ) = a + bλ → E(0) = a")
    coeffs_lin = np.polyfit(lambdas, energies, 1)
    e_linear = coeffs_lin[1]
    print(f"    Example: E(0) = {e_linear:.3f}")

    print("\n  Richardson extrapolation:")
    print("    Polynomial fit through data points")
    coeffs_quad = np.polyfit(lambdas, energies, 2)
    e_richardson = np.polyval(coeffs_quad, 0)
    print(f"    Example: E(0) = {e_richardson:.3f}")

    print("\n  Exponential extrapolation:")
    print("    E(λ) = E₀ + A·exp(-λ/B)")
    print("    Better for large noise scaling")

    print("\nExample Results:")
    print("-" * 45)
    print(f"  {'λ':<8} {'Measured E':<15} {'True E'}")
    print("  " + "-" * 35)
    for l, e in zip(lambdas, energies):
        print(f"  {l:<8} {e:<15.3f}")
    print("  " + "-" * 35)
    print(f"  {'0 (extrap)':<8} {e_richardson:<15.3f} {-1.137:.3f}")

zero_noise_extrapolation()
```

<Tip>
Linear extrapolation is simplest but may underestimate the zero-noise limit. Richardson extrapolation with multiple noise levels often gives better results but requires more circuit evaluations.
</Tip>

## 4. Measurement Error Mitigation

```python
def measurement_error_mitigation():
    """Measurement error mitigation techniques."""

    print("Measurement Error Mitigation")
    print("=" * 60)

    print("\nMeasurement Error Model:")
    print("-" * 45)
    print("  Confusion matrix M:")
    print("    M[i,j] = P(measure i | prepared j)")
    print()
    print("  Example (2 qubits):")
    confusion = np.array([
        [0.95, 0.03, 0.02, 0.01],
        [0.02, 0.94, 0.01, 0.02],
        [0.02, 0.02, 0.93, 0.03],
        [0.01, 0.01, 0.04, 0.94]
    ])
    print("    States: |00⟩, |01⟩, |10⟩, |11⟩")
    print(f"    M = ")
    for row in confusion:
        print(f"        [{', '.join(f'{x:.2f}' for x in row)}]")

    print("\nCalibration Procedure:")
    print("-" * 45)
    print("  1. Prepare each computational basis state")
    print("  2. Measure many times")
    print("  3. Build confusion matrix from statistics")

    print("\nMitigation by Matrix Inversion:")
    print("-" * 45)
    print("  measured = M @ ideal")
    print("  ideal = M⁻¹ @ measured")
    print()
    print("  Problem: M⁻¹ may give negative probabilities")
    print("  Solution: Use constrained optimization (NNLS)")

    print("\nTensor Product Assumption:")
    print("-" * 45)
    print("  For n qubits, full M is 2ⁿ × 2ⁿ")
    print("  Too large to calibrate!")
    print()
    print("  Approximation: M = M₁ ⊗ M₂ ⊗ ... ⊗ Mₙ")
    print("  Each Mᵢ is 2×2 → only 2n parameters")

    print("\nExample Correction:")
    print("-" * 45)
    measured = np.array([0.42, 0.08, 0.07, 0.43])
    M_inv = np.linalg.inv(confusion)
    corrected = M_inv @ measured
    # Clip to valid probabilities
    corrected = np.clip(corrected, 0, 1)
    corrected = corrected / np.sum(corrected)

    print("  Measured:  ", [f"{p:.2f}" for p in measured])
    print("  Corrected: ", [f"{p:.2f}" for p in corrected])

measurement_error_mitigation()
```

```python
    print()

    print("     ⟨O⟩_sym = ⟨P_sym O P_sym⟩ / ⟨P_sym⟩")
    print()
    print("  3. Constrained VQE:")
    print("     Add symmetry penalty to cost function")
    print("     E_total = E + λ(⟨S²⟩ - s(s+1))²")

    print("\nExample: Particle Number Conservation")
    print("-" * 45)
    print("  H₂: should have exactly 2 electrons")
    print("  Measure particle number after VQE circuit")
    print()
    print("  Results:")
    print("    N=2 (correct): 85% of shots")
    print("    N=1 or N=3:    12% (1-bit error)")
    print("    N=0 or N=4:    3%  (2-bit errors)")
    print()
    print("  Keep only N=2 results → effective error mitigation")

symmetry_verification()
```

<Warning>
Post-selection on symmetries reduces the effective number of measurements. If too many shots are discarded, statistical uncertainty increases. Balance error mitigation with measurement overhead.
</Warning>

## 6. Clifford Data Regression (CDR)

```python
def clifford_data_regression():
    """Clifford data regression for error mitigation."""

    print("Clifford Data Regression (CDR)")
    print("=" * 60)

    print("\nKey Insight:")
    print("  Clifford circuits can be efficiently simulated")
    print("  Use them to learn the noise model")

    print("\nClifford Circuits:")
    print("-" * 45)
    print("  Gates: H, S, CNOT (and combinations)")
    print("  Property: map Paulis to Paulis")
    print("  Simulation: O(n²) classical cost")

    print("\nCDR Algorithm:")
    print("-" * 45)
    steps = [
        "1. Generate 'near-Clifford' training circuits",
        "2. Compute exact results classically",
        "3. Run training circuits on noisy hardware",
        "4. Learn noise model: noisy = f(ideal)",
        "5. Run target circuit on hardware",
        "6. Apply learned correction: ideal ≈ f⁻¹(noisy)",
    ]
    for step in steps:
        print(f"  {step}")

    print("\nNear-Clifford Construction:")
    print("-" * 45)
    print("  Original: U(θ) with non-Clifford rotations")
    print("  Training: Replace θ → nearest Clifford angle")
    print("    Clifford angles: 0, π/2, π, 3π/2")

    print("\nLearning the Correction:")
    print("-" * 45)
    print("  Simple model: E_ideal = a × E_noisy + b")
    print("  Fit a, b from training data")
    print()
    print("  Example training data:")
    training = [
        (1.2, 1.0),
        (0.8, 0.6),
        (0.4, 0.2),
        (-0.2, -0.3),
    ]
    print(f"    {'E_noisy':<12} {'E_exact'}")
    for noisy, exact in training:
        print(f"    {noisy:<12} {exact}")

    # Fit linear model
    noisy_vals = np.array([t[0] for t in training])
    exact_vals = np.array([t[1] for t in training])
    a, b = np.polyfit(noisy_vals, exact_vals, 1)
    print(f"\n  Fitted model: E_ideal = {a:.2f} × E_noisy + {b:.2f}")

clifford_data_regression()
```

## 7. Virtual Distillation

```python
def virtual_distillation():
    """Virtual distillation error mitigation."""

    print("Virtual Distillation")
    print("=" * 60)

    print("\nKey Idea:")
    print("  Use multiple copies of noisy state")
    print("  'Distill' a purer estimate")

    print("\nMathematical Basis:")
    print("-" * 45)
    print("  Noisy state: ρ = (1-p)|ψ⟩⟨ψ| + p·noise")
    print("  With 2 copies:")
    print("    Tr(ρ²O) / Tr(ρ²) → purer estimate")

    print("\nPurity Estimation:")
    print("-" * 45)
    print("  Tr(ρ²) ≤ 1 for mixed states")
    print("  Tr(ρ²) = 1 for pure states")
    print()
    print("  Error-free: ρ = |ψ⟩⟨ψ|, Tr(ρ²) = 1")
    print("  With errors: Tr(ρ²) < 1")

    print("\nCircuit Implementation:")
    print("-" * 45)
    print("  Prepare 2 copies: ρ ⊗ ρ")
    print("  Apply SWAP test variants")
    print("  Measure collective observables")
    print()
    print("  Cost: 2× qubits, modified measurements")

    print("\nExponential Error Suppression:")
    print("-" * 45)
    print("  With M copies:")
    print("    Error suppression: ~ (error)^M")
    print("    Cost: M× qubits")
    print()
    print("  Example:")
    print("    1 copy:  10% error")
    print("    2 copies: 1% error")
    print("    3 copies: 0.1% error")

    print("\nLimitations:")
    print("-" * 45)
```
