## 1. What is an Ansatz?

<Definition term="Ansatz">
A parameterized quantum circuit U(θ) that prepares trial wavefunctions |ψ(θ)⟩ = U(θ)|ψ₀⟩. The choice of ansatz determines which states are accessible during VQE optimization.
</Definition>

```python
def ansatz_introduction():
    """Introduction to quantum circuit ansätze."""

    print("What is an Ansatz?")
    print("=" * 60)

    print("\nDefinition:")
    print("  |ψ(θ)⟩ = U(θ)|ψ₀⟩")
    print()
    print("  U(θ) = parameterized unitary circuit")
    print("  |ψ₀⟩ = reference state (often |00...0⟩ or |HF⟩)")
    print("  θ = variational parameters")

    print("\nKey Properties:")
    print("-" * 45)
    properties = [
        ("Expressibility", "Can represent target state"),
        ("Trainability", "Gradients remain accessible"),
        ("Efficiency", "Polynomial circuit depth"),
        ("Hardware-friendly", "Uses native gate set"),
    ]
    for prop, desc in properties:
        print(f"  {prop:<18} {desc}")

    print("\nAnsatz Categories:")
    print("-" * 45)
    print("  1. Hardware-Efficient Ansätze (HEA)")
    print("     • Generic structure, native gates")
    print("     • Easy to implement, may lack physics")
    print()
    print("  2. Chemistry-Inspired Ansätze")
    print("     • Based on excitation operators")
    print("     • Physical, may need deep circuits")
    print()
    print("  3. Adaptive Ansätze")
    print("     • Grow circuit during optimization")
    print("     • ADAPT-VQE approach")
```

## 2. Hardware-Efficient Ansatz

```python
def hardware_efficient_ansatz():
    """Hardware-efficient ansatz design."""

    print("Hardware-Efficient Ansatz (HEA)")
    print("=" * 60)

    print("\nDesign Philosophy:")
    print("  • Use native gates of quantum hardware")
    print("  • Minimize circuit depth")
    print("  • Generic structure, problem-agnostic")

    print("\nTypical Structure:")
    print("-" * 45)
    print("  |0⟩ ─[Ry(θ₁)]─[Rz(θ₂)]─●───────[Ry(θ₅)]─[Rz(θ₆)]─")
    print("                         │")
    print("  |0⟩ ─[Ry(θ₃)]─[Rz(θ₄)]─X───────[Ry(θ₇)]─[Rz(θ₈)]─")
    print()
    print("  Layer = Single-qubit rotations + Entangling gates")

    print("\nCommon Gate Choices:")
    print("-" * 45)
    single_qubit = [
        ("Ry(θ)", "exp(-iθY/2)", "Real amplitudes"),
        ("Rz(θ)", "exp(-iθZ/2)", "Phases"),
        ("Rx(θ)", "exp(-iθX/2)", "Alternative"),
    ]
    print("  Single-qubit rotations:")
    for gate, formula, note in single_qubit:
        print(f"    {gate:<10} {formula:<18} {note}")

    print("\n  Entangling gates:")
    entangling = [
        ("CNOT", "Native on many platforms"),
        ("CZ", "Symmetric, easier compilation"),
        ("iSWAP", "Native on some superconducting"),
    ]
    for gate, note in entangling:
        print(f"    {gate:<10} {note}")

    print("\nEntanglement Patterns:")
    print("-" * 45)
    print("  Linear:    q₀-q₁-q₂-q₃-q₄")
    print("  Circular:  q₀-q₁-q₂-q₃-q₄-q₀")
    print("  Full:      All pairs connected")
    print("  Pairwise:  (q₀,q₁), (q₂,q₃), then (q₁,q₂), (q₃,q₄)")

    print("\nParameter Count:")
    print("  N_params = n_qubits × n_layers × params_per_qubit")
    print("  Example: 4 qubits, 3 layers, 2 rotations")
    print("           → 4 × 3 × 2 = 24 parameters")

hardware_efficient_ansatz()
```

<Warning>
Hardware-efficient ansätze may suffer from barren plateaus - exponentially vanishing gradients that make optimization difficult for deep circuits or many qubits.
</Warning>

## 3. Chemistry-Inspired Ansätze

<Formula>
$$|\psi_{UCCSD}\rangle = e^{\hat{T} - \hat{T}^\dagger}|HF\rangle, \quad \hat{T} = \hat{T}_1 + \hat{T}_2$$
</Formula>

```python
def chemistry_inspired_ansatz():
    """Chemistry-inspired ansatz designs."""

    print("Chemistry-Inspired Ansätze")
    print("=" * 60)

    print("\nMotivation:")
    print("  • Incorporate physical structure")
    print("  • Based on coupled cluster theory")
    print("  • Excitation operators respect chemistry")

    print("\nUnitary Coupled Cluster (UCC):")
    print("-" * 45)
    print("  |ψ_UCC⟩ = exp(T - T†)|HF⟩")
    print()
    print("  T = cluster operator")
    print("  T† = de-excitation operator")
    print("  exp(T - T†) is unitary!")

    print("\nExcitation Operators:")
    print("-" * 45)
    print("  Singles (T₁):")
    print("    T₁ = Σᵢₐ tᵢᵃ a†ₐaᵢ")
    print("    Excite one electron: occupied → virtual")
    print()
    print("  Doubles (T₂):")
    print("    T₂ = Σᵢⱼₐᵦ tᵢⱼᵃᵇ a†ₐa†ᵦaⱼaᵢ")
    print("    Excite two electrons simultaneously")

    print("\nUCCSD (Singles and Doubles):")
    print("-" * 45)
    print("  T = T₁ + T₂")
    print()
    print("  Number of parameters:")
    print("    Singles: n_occ × n_virt")
    print("    Doubles: n_occ² × n_virt²")
    print()
    print("  Example (H₂O, minimal basis):")
    print("    5 occupied, 2 virtual")
    print("    Singles: 5 × 2 = 10")
    print("    Doubles: 25 × 4 = 100")
    print("    Total: ~110 parameters")

    print("\nTrotterized UCCSD:")
    print("-" * 45)
    print("  exp(T - T†) ≈ Πₖ exp(θₖ(τₖ - τₖ†))")
    print()
    print("  Each term → quantum circuit")
    print("  Trotter error controlled by ordering")

chemistry_inspired_ansatz()
```

```python
    print("                   │             │")
    print("  .: ──────────────X─●─────────●─X──────────────")
```

## 5. Ansatz Expressibility

```python
def ansatz_expressibility():
    """Analyze ansatz expressibility."""

    print("Ansatz Expressibility")
    print("=" * 60)

    print("\nWhat is Expressibility?")
    print("  The ability of ansatz to represent target states")
    print("  Measured by coverage of Hilbert space")

    print("\nExpressibility Metric:")
    print("-" * 45)
    print("  Compare distribution of fidelities to Haar random")
    print()
    print("  Expr = D_KL(P_ansatz || P_Haar)")
    print()
    print("  Lower = more expressive (closer to Haar)")

    print("\nExpressibility vs Depth:")
    print("-" * 45)
    depths = [1, 2, 4, 8, 16]
    expr_hea = [2.1, 1.4, 0.8, 0.3, 0.1]
    print(f"  {'Depth':<10} {'Expressibility':<15} {'Quality'}")
    print("  " + "-" * 40)
    for d, e in zip(depths, expr_hea):
        quality = "Low" if e > 1.5 else "Medium" if e > 0.5 else "High"
        print(f"  {d:<10} {e:<15.2f} {quality}")

    print("\nThe Expressibility Dilemma:")
    print("-" * 45)
    print("  Too low expressibility:")
    print("    • Cannot represent ground state")
    print("    • Systematic error (ansatz error)")
    print()
    print("  Too high expressibility:")
    print("    • Barren plateaus (vanishing gradients)")
    print("    • Optimization becomes exponentially hard")

    print("\nRecommendation:")
    print("  • Match expressibility to problem complexity")
    print("  • Start simple, increase if needed")
    print("  • Use chemistry-inspired for molecules")

ansatz_expressibility()
```

<Tip>
A good ansatz is expressive enough to represent the ground state but not so expressive that it suffers from barren plateaus. Chemistry-inspired ansätze often achieve this balance by encoding physical constraints.
</Tip>

## 6. Barren Plateaus

```python
def barren_plateaus():
    """Understanding barren plateaus."""

    print("Barren Plateaus in VQE")
    print("=" * 60)

    print("\nWhat are Barren Plateaus?")
    print("  Regions where gradients vanish exponentially")
    print("  ∂E/∂θ → 0 as n → ∞")

    print("\nMathematical Statement:")
    print("-" * 45)
    print("  Var[∂E/∂θ] ∝ 1/2ⁿ")
    print()
    print("  For n qubits with random initialization:")
    print("  Gradients exponentially small!")

    print("\nCauses of Barren Plateaus:")
    print("-" * 45)
    causes = [
        ("Deep circuits", "Too many layers → random unitaries"),
        ("Global cost functions", "Non-local operators spread info"),
        ("High expressibility", "Too much of Hilbert space covered"),
        ("Noise", "Hardware noise can induce plateaus"),
    ]
    for cause, explanation in causes:
        print(f"  {cause}:")
        print(f"    {explanation}")
        print()

    print("Mitigation Strategies:")
    print("-" * 45)
    strategies = [
        "Layer-wise training (grow circuit gradually)",
        "Local cost functions when possible",
        "Problem-specific initialization",
        "Shallow circuits with structure",
        "Parameter correlation (fewer independent params)",
    ]
    for i, s in enumerate(strategies, 1):
        print(f"  {i}. {s}")

    print("\nGradient Magnitude vs Qubits:")
    print("-" * 45)
    qubits = [4, 8, 12, 16, 20]
    grad_hea = [0.1, 0.006, 0.0004, 2e-5, 1e-6]
    grad_uccsd = [0.1, 0.05, 0.02, 0.01, 0.005]
    print(f"  {'Qubits':<10} {'HEA |∇|':<15} {'UCCSD |∇|'}")
    print("  " + "-" * 40)
    for n, g_hea, g_ucc in zip(qubits, grad_hea, grad_uccsd):
        print(f"  {n:<10} {g_hea:<15.1e} {g_ucc:.1e}")

barren_plateaus()
```

## 7. Adaptive Ansätze

```python
def adaptive_ansatz():
    """Adaptive ansatz design (ADAPT-VQE)."""

    print("Adaptive Ansätze: ADAPT-VQE")
    print("=" * 60)

    print("\nMotivation:")
    print("  • Avoid fixed ansatz limitations")
    print("  • Grow circuit based on gradients")
    print("  • Problem-tailored structure")

    print("\nADAPT-VQE Algorithm:")
    print("-" * 45)
    steps = [
        "1. Start with reference state |ψ₀⟩",
        "2. Define operator pool {Aₖ} (excitations)",
        "3. Compute gradient ∂E/∂θₖ for each Aₖ",
        "4. Add operator with largest |gradient|",
        "5. Optimize all parameters",
        "6. Repeat until gradients below threshold",
    ]
    for step in steps:
        print(f"  {step}")

    print("\nOperator Pool:")
    print("-" * 45)
    print("  Fermionic pool:")
    print("    • Single excitations: a†ₚaᵩ - h.c.")
    print("    • Double excitations: a†ₚa†ᵩaᵣaₛ - h.c.")
    print()
    print("  Qubit pool:")
    print("    • Individual Pauli strings from decomposition")
    print("    • More flexible, hardware-efficient")

    print("\nAdvantages:")
    print("-" * 45)
    print("  • Compact circuits (only needed operators)")
    print("  • Systematic convergence")
    print("  • Avoids barren plateaus (gradual growth)")

    print("\nExample Convergence:")
    print("-" * 45)
    iterations = [0, 1, 2, 3, 4, 5]
    n_params = [0, 1, 2, 4, 6, 7]
    energies = [-1.05, -1.10, -1.12, -1.135, -1.137, -1.137]
    print(f"  {'Iter':<8} {'Params':<10} {'Energy':<12}")
    print("  " + "-" * 30)
    for it, n, e in zip(iterations, n_params, energies):
        print(f"  {it:<8} {n:<10} {e:<12.3f}")

adaptive_ansatz()
```
