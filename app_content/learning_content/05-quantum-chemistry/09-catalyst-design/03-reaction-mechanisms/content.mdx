## 1. Reaction Pathway Mapping

<Definition term="Reaction Mechanism">
A sequence of elementary steps that converts reactants to products, each step involving the making or breaking of chemical bonds at specific sites on the catalyst surface.
</Definition>

```python
import numpy as np

def reaction_pathway_overview():
    """Overview of reaction mechanism concepts."""

    print("Reaction Mechanism Analysis")
    print("=" * 60)

    print("\nElementary Steps:")
    print("-" * 45)
    steps = [
        ("Adsorption", "A(g) + * → A*", "Molecule binds"),
        ("Dissociation", "A₂* → 2A*", "Bond breaking"),
        ("Diffusion", "A* → A*'", "Site hopping"),
        ("Reaction", "A* + B* → C*", "Bond formation"),
        ("Desorption", "C* → C(g) + *", "Product release"),
    ]
    for step, reaction, desc in steps:
        print(f"  {step:<12} {reaction:<18} {desc}")

    print("\nReaction Network:")
    print("-" * 45)
    print("  CO Oxidation on Pt:")
    print("    1. CO(g) + * → CO*")
    print("    2. O₂(g) + 2* → 2O*")
    print("    3. CO* + O* → CO₂* + *")
    print("    4. CO₂* → CO₂(g) + *")

    print("\nPotential Energy Surface (PES):")
    print("-" * 45)
    print("  Maps energy vs. reaction coordinate")
    print("  Minima: Stable intermediates")
    print("  Saddle points: Transition states")
    print("  Barriers: Activation energies")

    print("\nMechanistic Complexity:")
    print("-" * 45)
    print("  • Parallel pathways (selectivity)")
    print("  • Branching points")
    print("  • Coverage-dependent barriers")
    print("  • Competing reactions")

reaction_pathway_overview()
```

## 2. Transition State Search

```python
def transition_state_methods():
    """Methods for finding transition states."""

    print("Transition State Search Methods")
    print("=" * 60)

    print("\nNudged Elastic Band (NEB):")
    print("-" * 45)
    print("  • Chain of images between reactant/product")
    print("  • Spring forces keep images evenly spaced")
    print("  • Climb to saddle point")
    print()
    print("  Algorithm:")
    print("    1. Interpolate initial path")
    print("    2. Optimize with spring + true forces")
    print("    3. CI-NEB: Highest image climbs")
    print()
    print("  Parameters:")
    print("    N_images: 5-9 typical")
    print("    Spring constant: 5 eV/Å²")
    print("    Convergence: |F| < 0.03 eV/Å")

    print("\nDimer Method:")
    print("-" * 45)
    print("  • Start from guessed TS geometry")
    print("  • Rotate dimer to find lowest mode")
    print("  • Move uphill along mode")
    print()
    print("  Advantages:")
    print("    - No need for product state")
    print("    - Good for complex mechanisms")
    print("    - Can escape wrong basin")

    print("\nOther Methods:")
    print("-" * 45)
    methods = [
        ("Growing String", "Adaptive path length"),
        ("Synchronous Transit", "Initial guess for NEB"),
        ("Eigenvector Following", "Follow imaginary mode"),
        ("Metadynamics", "Free energy barriers"),
    ]
    for method, desc in methods:
        print(f"  {method:<20} {desc}")

    print("\nTS Verification:")
    print("-" * 45)
    print("  1. Single imaginary frequency")
    print("  2. Mode connects reactant/product")
    print("  3. IRC confirms path")

transition_state_methods()
```

<Tip>
Always verify transition states by checking for exactly one imaginary frequency. The eigenvector of this mode should point toward both reactant and product states.
</Tip>

## 4. Free Energy Diagrams

```python
def free_energy_diagrams():
    """Constructing free energy diagrams."""

    print("Free Energy Diagrams")
    print("=" * 60)

    print("\nFrom DFT to Free Energy:")
    print("-" * 45)
    print("  G = E_DFT + ZPE + ∫Cp dT - TS")
    print()
    print("  Contributions:")
    print("    E_DFT: Electronic energy from DFT")
    print("    ZPE: Zero-point vibrational energy")
    print("    ∫Cp dT: Thermal enthalpy")
    print("    -TS: Entropy term")

    print("\nVibrational Corrections:")
    print("-" * 45)
    print("  For adsorbed species:")
    print("    • Vibrations only (no rotation/translation)")
    print("    • ZPE = ½Σℏω")
    print("    • Use harmonic approximation")
    print()
    print("  For gas phase:")
    print("    • Include rotation and translation")
    print("    • Ideal gas approximation")

    print("\nEntropy Considerations:")
    print("-" * 45)
    print("  Adsorption: Large entropy loss")
    print("    S_gas(CO) ≈ 200 J/mol·K")
    print("    S_ads(CO) ≈ 50 J/mol·K")
    print()
    print("  At T = 500K:")
    print("    -TΔS ≈ +0.78 eV (unfavorable)")
    print()
    print("  → High T favors desorption")

    print("\nReference States:")
    print("-" * 45)
    print("  Gas-phase molecules + clean surface = 0")
    print("  Plot relative energies along pathway")
    print()
    print("  Example (CO oxidation):")
    print("    State         ΔG (eV)")
    print("    CO + ½O₂ + *   0.0")
    print("    CO* + ½O₂     -0.5")
    print("    CO* + O*      -0.8")
    print("    TS            -0.2")
    print("    CO₂*          -1.5")
    print("    CO₂ + *       -2.0")

free_energy_diagrams()
```

<Formula name="Gibbs Free Energy">
$$G = E_{DFT} + E_{ZPE} + \int_0^T C_p dT - TS$$

Where $E_{ZPE} = \frac{1}{2}\sum_i \hbar\omega_i$ for all vibrational modes.
</Formula>

## 5. Rate-Determining Steps

```python
def rate_determining_analysis():
    """Identifying rate-determining steps."""

    print("Rate-Determining Step Analysis")
    print("=" * 60)

    print("\nDefinition:")
    print("-" * 45)
    print("  RDS: Step that most controls overall rate")
    print("  Not always highest barrier!")
    print()
    print("  Degree of Rate Control (XRC):")
    print("    XRC,i = (kT/r) × (∂r/∂G_TS,i)")
    print()
    print("    Sum of all XRC = 1")
    print("    XRC → 1 means rate-determining")

    print("\nApparent Activation Energy:")
    print("-" * 45)
    print("  E_app = E_TS - E_most_stable_intermediate")
    print()
    print("  Not just intrinsic barrier!")
    print("  Includes coverage effects")

    print("\nExample - NH₃ Synthesis:")
    print("-" * 45)
    print("  N₂ + 3H₂ → 2NH₃")
    print()
    print("  On Fe:")
    print("    Step           E_a (eV)")
    print("    N₂ → 2N*       1.0  ← RDS")
    print("    N* + H* → NH*  0.5")
    print("    NH* + H* → NH₂* 0.3")
    print("    NH₂* + H* → NH₃ 0.2")

    print("  N₂ dissociation is rate-limiting")

    print("\nCoverage Effects:")
```

## 6. Selectivity Analysis

```python
def selectivity_analysis():
    """Analyzing reaction selectivity."""

    print("Selectivity in Heterogeneous Catalysis")
    print("=" * 60)

    print("\nTypes of Selectivity:")
    print("-" * 45)
    print("  Chemoselectivity: Which functional group reacts")
    print("  Regioselectivity: Which position on molecule")
    print("  Stereoselectivity: Which stereoisomer forms")

    print("\nKinetic vs Thermodynamic Control:")
    print("-" * 45)
    print("  Kinetic: Lowest barrier wins")
    print("    Selectivity = exp(-ΔΔE‡/kT)")
    print()
    print("  Thermodynamic: Most stable product wins")
    print("    At equilibrium, reversible steps")

    print("\nBranching Ratio:")
    print("-" * 45)
    print("  For A → B and A → C:")
    print()
    print("  r_B/r_C = (A_B/A_C) × exp(-(E_B-E_C)/kT)")
    print()
    print("  At 500K, ΔE = 0.1 eV:")
    print("    Selectivity = exp(0.1/0.043) = 10:1")

    print("\nExample - Acetylene Hydrogenation:")
    print("-" * 45)
    print("  C₂H₂ + H₂ → C₂H₄ (desired)")
    print("  C₂H₄ + H₂ → C₂H₆ (over-hydrogenation)")
    print()
    print("  Strategy:")
    print("    • Weaker H binding (Pd-Ag alloy)")
    print("    • C₂H₄ desorbs before further H₂")
    print("    • E_des(C₂H₄) < E_a(hydrogenation)")

    print("\nVolcano for Selectivity:")
    print("-" * 45)
    print("  Two products have different optima")
    print("  Choose binding energy for desired product")

selectivity_analysis()
```

<Warning>
Selectivity calculations require high accuracy. A 0.05 eV error in barrier difference translates to ~3× error in selectivity at typical operating temperatures.
</Warning>
