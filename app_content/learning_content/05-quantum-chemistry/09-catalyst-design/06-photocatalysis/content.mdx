## 1. Photocatalysis Fundamentals

<Definition term="Photocatalysis">
A catalytic process driven by light absorption, where photoexcited electrons and holes drive chemical reactions at the catalyst surface.
</Definition>

```python
import numpy as np

def photocatalysis_intro():
    """Introduction to photocatalytic processes."""

    print("Photocatalysis Fundamentals")
    print("=" * 60)

    print("\nKey Steps:")
    print("-" * 45)
    steps = [
        ("1. Light absorption", "Generate electron-hole pairs"),
        ("2. Charge separation", "Prevent recombination"),
        ("3. Charge transport", "Move to surface"),
        ("4. Surface reactions", "Oxidation/reduction"),
    ]
    for step, desc in steps:
        print(f"  {step}: {desc}")

    print("\nEnergy Requirements:")
    print("-" * 45)
    print("  Photon energy: E = hν = hc/λ")
    print()
    print("  λ = 400 nm (violet): E = 3.1 eV")
    print("  λ = 550 nm (green):  E = 2.3 eV")
    print("  λ = 700 nm (red):    E = 1.8 eV")
    print()
    print("  Solar spectrum peaks ~500 nm")

    print("\nBand Gap Requirements:")
    print("-" * 45)
    print("  Eg > reaction potential difference")
    print()
    print("  Water splitting: H₂O → H₂ + ½O₂")
    print("    ΔG = 1.23 eV (minimum)")
    print("    Practical: Eg > 1.8 eV (overpotentials)")
    print("    But Eg < 3.0 eV (visible light)")
    print()
    print("  Optimal: 1.8 < Eg < 2.5 eV")

    print("\nBand Edge Positions:")
    print("-" * 45)
    print("  Must straddle reaction potentials")
    print()
    print("  For water splitting:")
    print("    CB < 0 V vs NHE (H⁺/H₂)")
    print("    VB > 1.23 V vs NHE (O₂/H₂O)")
    print()
    print("  Both conditions must be satisfied!")

photocatalysis_intro()

```

<Formula name="Optical Band Gap">
$$E_g^{opt} = \frac{hc}{\lambda_{onset}} = \frac{1240}{\lambda_{onset}(nm)} \text{ eV}$$

Where $\lambda_{onset}$ is the absorption onset wavelength.
</Formula>

## 3. Optical Properties

```python
def optical_properties():
    """Calculating optical absorption."""

    print("Optical Properties Calculations")
    print("=" * 60)

    print("\nDielectric Function:")
    print("-" * 45)
    print("  ε(ω) = ε₁(ω) + iε₂(ω)")
    print()
    print("  ε₂: Imaginary part (absorption)")
    print("  ε₁: Real part (refraction)")
    print()
    print("  From DFT: Sum over transitions")
    print("    ε₂ ∝ Σ |⟨ψ_v|r|ψ_c⟩|² δ(E_c - E_v - ℏω)")

    print("\nAbsorption Coefficient:")
    print("-" * 45)
    print("  α(ω) = (ω/c) × Im[√ε(ω)]")
    print()
    print("  Determines light penetration depth")
    print("  1/α: characteristic depth")

    print("\nDirect vs Indirect Gaps:")
    print("-" * 45)
    print("  Direct: Vertical transition in k-space")
    print("    Strong absorption at band edge")
    print("    α ∝ √(E - Eg)")
    print()
    print("  Indirect: Requires phonon assistance")
    print("    Weak absorption, longer lifetime")
    print("    α ∝ (E - Eg)²")
    print()
    print("  TiO₂: Indirect → long carrier lifetime")
    print("  GaAs: Direct → strong absorption")

    print("\nExciton Effects:")
    print("-" * 45)
    print("  Optical gap ≠ electronic gap")
    print("  Exciton binding: E_b = Eg - Eg_opt")
    print()
    print("  Wannier (delocalized): E_b ~ 10 meV")
    print("  Frenkel (localized): E_b ~ 0.5 eV")
    print()
    print("  BSE calculation for accurate excitons")

optical_properties()
```

## 4. Charge Carrier Dynamics

```python
def charge_dynamics():
    """Modeling charge carrier behavior."""

    print("Charge Carrier Dynamics")
    print("=" * 60)

    print("\nRecombination Mechanisms:")
    print("-" * 45)
    mechanisms = [
        ("Radiative", "Photon emission", "GW-BSE"),
        ("Auger", "Energy to third carrier", "Many-body"),
        ("Trap-assisted", "Via defect states", "DFT+defects"),
        ("Surface", "At surface states", "Slab calcs"),
    ]
    print(f"  {'Mechanism':<16} {'Description':<22} {'Theory'}")
    print("  " + "-" * 50)
    for mech, desc, theory in mechanisms:
        print(f"  {mech:<16} {desc:<22} {theory}")

    print("\nCarrier Mobility:")
    print("-" * 45)
    print("  μ = eτ/m*")
    print()
    print("  τ: Scattering time")
    print("  m*: Effective mass")
    print()
    print("  Calculate m* from band curvature:")
    print("    m* = ℏ² / (d²E/dk²)")
    print()
    print("  Flat bands → heavy m* → poor transport")

    print("\nDiffusion Length:")
    print("-" * 45)
    print("  L = √(D × τ)")
    print()
    print("  D: Diffusion coefficient = μ × kT/e")
    print("  τ: Carrier lifetime")
    print()
    print("  L must exceed particle size for efficiency")
    print()
    print("  Example values:")
    print("    TiO₂: L_e ~ 10 μm, L_h ~ 10 nm")
    print("    Fe₂O₃: L ~ 2-4 nm (major limitation!)")

    print("\nDefect Trapping:")
    print("-" * 45)
    print("  Defects create mid-gap states")
    print("  Can act as recombination centers")
    print()
    print("  DFT calculation:")
    print("    • Create supercell with defect")
    print("    • Calculate defect energy levels")
    print("    • Determine if trap is deep or shallow")

charge_dynamics()
```

<Warning>
Fe₂O₃ has excellent optical properties but very short hole diffusion lengths (~2 nm). Nanostructuring is essential to overcome this transport limitation.
</Warning>

## 6. Photocatalyst Design

```python
def photocatalyst_design():
    """Computational design of photocatalysts."""

    print("Computational Photocatalyst Design")
    print("=" * 60)

    print("\nDesign Criteria:")
    print("-" * 45)
    print("  1. Band gap: 1.8-2.5 eV (visible light)")
    print("  2. Band edges: Straddle reaction potentials")
    print("  3. Low effective mass: Good transport")
    print("  4. Defect tolerance: Few deep traps")
    print("  5. Stability: In aqueous, under illumination")

    print("\nBand Engineering Strategies:")
    print("-" * 45)
    print("  Narrowing the gap:")
    print("    • N-doping in oxides (VB up)")
    print("    • Reduced oxides (CB states)")
    print("    • Solid solutions (continuous tuning)")
    print()
    print("  Shifting band edges:")
    print("    • Cation substitution (CB position)")
    print("    • Anion substitution (VB position)")
    print("    • Surface modification")

    print("\nCo-Catalysts:")
    print("-" * 45)
    print("  Separate e⁻/h⁺ to different sites")
    print()
    print("  For H₂ evolution:")
    print("    Pt, Pd, Ni, CoP, MoS₂")
    print()
    print("  For O₂ evolution:")
    print("    IrO₂, RuO₂, CoOx, NiFeOx")
    print()
    print("  Compute: Interface band alignment")

    print("\nHeterojunctions:")
    print("-" * 45)
    print("  Type II: Staggered bands")
    print("    e⁻ → material A, h⁺ → material B")
    print("    Charge separation enhanced")
    print()
    print("  Z-scheme: Direct electron transfer")
    print("    Maintains strong redox potentials")
    print()
    print("  Compute: Band offsets at interface")

    print("\nScreening Workflow:")
    print("-" * 45)
    print("  1. Database search (ICSD, MP)")
    print("  2. Filter by Eg range")
    print("  3. Calculate band alignment (HSE)")
    print("  4. Check stability (Pourbaix, ΔG_hull)")
    print("  5. Calculate effective masses")
    print("  6. Evaluate surface properties")

photocatalyst_design()
```

<Tip>
When screening photocatalysts, first filter by band gap and stability using fast DFT (GGA), then refine promising candidates with accurate hybrid functionals (HSE06) for reliable band edge positions.
</Tip>
