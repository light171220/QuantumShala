## 1. Nuclear Quantum Effects

<Definition term="Nuclear Quantum Effects">
Deviations from classical nuclear behavior due to quantum mechanics, including zero-point energy, tunneling, and delocalization. Most significant for light atoms like hydrogen.
</Definition>

```python
def nuclear_quantum_effects():
    """Introduction to nuclear quantum effects."""

    print("Nuclear Quantum Effects (NQE)")
    print("=" * 60)

    print("\nWhen Classical Nuclei Fail:")
    print("-" * 45)
    print("  • Light atoms (H, D, He)")
    print("  • Low temperatures")
    print("  • Narrow barriers (tunneling)")
    print("  • Zero-point energy contributions")

    print("\nManifestations:")
    print("-" * 45)
    effects = [
        ("Zero-point energy", "Quantum ground state > classical minimum"),
        ("Tunneling", "Barrier crossing without thermal activation"),
        ("Delocalization", "Smeared nuclear positions"),
        ("Isotope effects", "H/D rate ratios")
    ]
    for effect, desc in effects:
        print(f"  {effect}:")
        print(f"    {desc}")

    print("\nDe Broglie Wavelength:")
    print("-" * 45)
    print("  λ_dB = h/√(2πmkT)")
    print()
    print("  At 300 K:")
    print("    H: λ_dB ≈ 1.0 Å")
    print("    D: λ_dB ≈ 0.7 Å")
    print("    C: λ_dB ≈ 0.3 Å")
    print()
    print("  When λ_dB ~ bond length: NQE important!")

nuclear_quantum_effects()
```

## 2. Feynman Path Integrals

<Formula>
$$Z = \oint \mathcal{D}[r(\tau)] e^{-S[r(\tau)]/\hbar}$$
</Formula>

```python
def path_integral_formulation():
    """Feynman path integral formulation."""

    print("Feynman Path Integral Formulation")
    print("=" * 60)

    print("\nQuantum Partition Function:")
    print("-" * 45)
    print("  Z = Tr[e^{-βĤ}]")
    print("  β = 1/kT (inverse temperature)")

    print("\nPath Integral Representation:")
    print("-" * 45)
    print("  Z = ∫ dr₁...drₚ exp(-βₚ V_eff)")
    print()
    print("  P = number of 'beads' (imaginary time slices)")
    print("  βₚ = β/P")

    print("\nRing Polymer Mapping:")
    print("-" * 45)
    print("  Each quantum particle → P classical beads")
    print("  Connected by harmonic springs")
    print()
    print("  Spring constant: k = P(mkT)²/ℏ²")
    print("  'Ring' because periodic: rₚ₊₁ = r₁")

    print("\nEffective Potential:")
    print("-" * 45)
    print("  V_eff = Σⱼ[½mωₚ²(rⱼ-rⱼ₊₁)² + V(rⱼ)/P]")
    print()
    print("  ωₚ = √P kT/ℏ (ring polymer frequency)")

    print("\nClassical Isomorphism:")
    print("-" * 45)
    print("  Quantum statistics ↔ Classical ring polymer")
    print("  Sample with MD or MC!")

path_integral_formulation()
```

## 3. Path Integral Molecular Dynamics

```python
def pimd_method():
    """Path integral molecular dynamics."""

    print("Path Integral Molecular Dynamics (PIMD)")
    print("=" * 60)

    print("\nPIMD Setup:")
    print("-" * 45)
    print("  Each atom → P replicas ('beads')")
    print("  Beads connected by springs")
    print("  External potential at each bead")

    print("\nEquations of Motion:")
    print("-" * 45)
    print("  mⱼ r̈ⱼ = -∂V(rⱼ)/∂rⱼ - mωₚ²(2rⱼ - rⱼ₋₁ - rⱼ₊₁)")
    print()
    print("  External force + Spring forces")

    print("\nNormal Mode Transformation:")
    print("-" * 45)
    print("  Transform to ring polymer normal modes")
    print("  Centroid mode (k=0): average position")
    print("  Other modes: internal ring dynamics")

    print("\nThermostatting:")
    print("-" * 45)
    print("  Each mode needs its own thermostat!")
    print("  Massive Nosé-Hoover chains common")
    print("  PILE: Path Integral Langevin Equation")

    print("\nConvergence with P:")
    print("-" * 45)
    beads = [4, 8, 16, 32, 64]
    errors = [25, 10, 3, 1, 0.3]
    print(f"  {'Beads':<10} {'Relative Error %'}")
    print("  " + "-" * 25)
    for p, e in zip(beads, errors):
        print(f"  {p:<10} {e}")

pimd_method()
```

<Tip>
The number of beads P should be chosen such that ℏωₚ >> kT, typically P ~ 20-64 for room temperature hydrogen simulations.
</Tip>

## 5. Centroid Molecular Dynamics

```python
def centroid_md():
    """Centroid molecular dynamics method."""

    print("Centroid Molecular Dynamics (CMD)")
    print("=" * 60)

    print("\nCentroid Definition:")
    print("-" * 45)
    print("  r̄ = (1/P) Σⱼ rⱼ")
    print("  Average position of ring polymer")
    print("  Classical-like quantity")

    print("\nCentroid Dynamics:")
    print("-" * 45)
    print("  Propagate centroid on mean-field potential")
    print("  m r̄̈ = -∂⟨V⟩/∂r̄")
    print()
    print("  ⟨V⟩ = average over ring polymer configs")

    print("\nCentroid Mean Force:")
    print("-" * 45)
    print("  F_c(r̄) = -⟨∂V/∂r⟩|_{centroid = r̄}")
    print()
    print("  Requires constrained sampling")
    print("  Computationally demanding")

    print("\nAdiabatic CMD:")
    print("-" * 45)
    print("  Separate centroid and non-centroid modes")
    print("  Non-centroid modes equilibrated 'instantaneously'")
    print("  Similar to Born-Oppenheimer approximation")

    print("\nComparison:")
    print("-" * 45)
    methods = [
        ("PIMD", "Thermodynamics", "Exact quantum stat."),
        ("RPMD", "Rates", "Approx. dynamics"),
        ("CMD", "Spectra", "Infrared, correlation"),
    ]
    print(f"  {'Method':<8} {'Best for':<15} {'Note'}")
    print("  " + "-" * 40)
    for m, use, note in methods:
        print(f"  {m:<8} {use:<15} {note}")

centroid_md()
```
