## 1. Introduction to ADMET

<Definition term="ADMET">
Absorption, Distribution, Metabolism, Excretion, and Toxicity - the pharmacokinetic and safety properties that determine whether a compound can become a successful drug.
</Definition>

```python
def admet_overview():
    """Overview of ADMET properties in drug discovery."""

    print("ADMET Properties Overview")
    print("=" * 60)

    print("\nWhy ADMET Matters:")
    print("-" * 45)

    print("\nADMET Components:")
```

## 2. Absorption Prediction

```python
def absorption_prediction():
    """Computational prediction of drug absorption."""

    print("Absorption Prediction")
    print("=" * 60)

    print("\nLipinski's Rule of Five:")
    print("-" * 45)
    print("  For oral bioavailability:")
    print("    • MW ≤ 500 Da")
    print("    • LogP ≤ 5")
    print("    • H-bond donors ≤ 5")
    print("    • H-bond acceptors ≤ 10")
    print()
    print("  Note: 0-1 violations usually acceptable")

    print("\nVeber Rules:")
    print("-" * 45)
    print("  Additional oral bioavailability criteria:")
    print("    • Rotatable bonds ≤ 10")
    print("    • Polar surface area ≤ 140 Å²")

    print("\nPermeability Prediction:")
    print("-" * 45)
    print("  Caco-2 cell permeability:")
    print("    • Model of intestinal absorption")
    print("    • Papp > 1×10⁻⁶ cm/s: Good permeability")
    print()
    print("  PAMPA (Parallel Artificial Membrane):")
    print("    • Passive diffusion model")
    print("    • Faster, cheaper than Caco-2")

    print("\nSolubility:")
    print("-" * 45)
    print("  Minimum solubility for oral drugs:")
    print("    • Low dose: > 1 μg/mL")
    print("    • High dose: > 100 μg/mL")
    print()
    print("  Prediction methods:")
    print("    • LogS from molecular descriptors")
    print("    • Group contribution methods")
    print("    • COSMO-RS (quantum-based)")

    # Lipinski checker implementation
    def check_lipinski(mw, logp, hbd, hba):
        violations = []
        if mw > 500: violations.append("MW > 500")
        if logp > 5: violations.append("LogP > 5")
        if hbd > 5: violations.append("HBD > 5")
        if hba > 10: violations.append("HBA > 10")
        return len(violations) <= 1, violations

    print("\nExample Check:")
    print("-" * 45)
    test_compounds = [
        ("Aspirin", 180, 1.2, 1, 4),
        ("Imatinib", 493, 3.5, 2, 7),
        ("Cyclosporin", 1202, 2.9, 5, 23),
    ]
    for name, mw, logp, hbd, hba in test_compounds:
        passes, violations = check_lipinski(mw, logp, hbd, hba)
        status = "PASS" if passes else "FAIL"
        print(f"  {name}: {status} ({len(violations)} violations)")

absorption_prediction()
```

```python
        ("CYP3A4", "~50%", "Large, lipophilic"),
        ("CYP2D6", "~25%", "Basic nitrogen"),
        ("CYP2C9", "~15%", "Acidic, lipophilic"),
        ("CYP2C19", "~5%", "Various"),
        ("CYP1A2", "~5%", "Planar, aromatic"),
    ]
    print(f"  {'Enzyme':<10} {'% Drugs':<10} {'Substrate Features'}")

    for cyp, pct, features in cyps:

    print("\nMetabolism Issues:")

    issues = [
        ("Too rapid", "Low exposure, need high doses"),

        ("Induction", "Reduced efficacy over time"),
        ("Reactive metabolites", "Toxicity"),
    ]
    for issue, consequence in issues:
        print(f"  • {issue}: {consequence}")

    print("\nSite of Metabolism (SoM) Prediction:")
    print("-" * 45)
    print("  Methods:")
    print("    1. Reactivity-based (bond energies)")
    print("    2. Docking to CYP structure")
    print("    3. ML models (SMARTCyp, XenoSite)")
    print("    4. QM/MM for accurate barriers")

    print("\nMetabolic Soft Spots:")
    print("-" * 45)
    soft_spots = [
        "Benzylic positions (C-H activation)",
        "Aliphatic chains (ω and ω-1 oxidation)",
        "Heteroatom neighbors (N-dealkylation)",
        "Aromatic rings (epoxidation)",
        "Thioethers (S-oxidation)",
    ]
    for spot in soft_spots:
        print(f"  • {spot}")

metabolism_prediction()
```

<Tip>
Identify metabolic soft spots early and consider blocking them with fluorine substitution or deuterium incorporation to improve metabolic stability.
</Tip>

```python
        ("Mitochondrial", "Energy disruption"),

toxicity_prediction()
```

<Warning>
Structural alerts don't automatically disqualify compounds - many marketed drugs contain them. They indicate areas requiring careful evaluation and additional testing.
</Warning>

```python
        ("Tier 4 (In vivo)", "PK studies, toxicology"),

    properties = [
```

5. **Toxicity alerts** flag potential safety issues early
6. **Multi-parameter optimization** balances competing properties
