## 1. Thermodynamics of Binding

<Definition term="Binding Affinity">
The strength of interaction between a ligand and its target, typically quantified by the dissociation constant Kd or binding free energy ΔG. Related by ΔG = -RT ln(Ka) = RT ln(Kd).
</Definition>

```python
import numpy as np

def binding_thermodynamics():
    """Thermodynamics of protein-ligand binding."""

    print("Binding Thermodynamics")
    print("=" * 60)

    print("\nFundamental Equation:")
    print("-" * 45)
    print("  ΔG = ΔH - TΔS")
    print()
    print("  Where:")
    print("    ΔG = Binding free energy (determines Kd)")
    print("    ΔH = Enthalpy (interaction strength)")
    print("    ΔS = Entropy (disorder change)")

    print("\nRelation to Kd:")
    print("-" * 45)
    print("  ΔG = RT ln(Kd)")
    print("  Kd = exp(ΔG/RT)")
    print()
    # Convert between ΔG and Kd
    R = 1.987e-3  # kcal/(mol·K)
    T = 300  # K

    print(f"  At T = {T} K:")
    for dG in [-6, -8, -10, -12]:
        Kd = np.exp(dG / (R * T))
        if Kd > 1e-6:
            print(f"    ΔG = {dG:3d} kcal/mol → Kd = {Kd*1e6:.0f} μM")
        elif Kd > 1e-9:
            print(f"    ΔG = {dG:3d} kcal/mol → Kd = {Kd*1e9:.0f} nM")
        else:
            print(f"    ΔG = {dG:3d} kcal/mol → Kd = {Kd*1e12:.0f} pM")

    print("\nBinding Process Contributions:")
    print("-" * 45)
    contributions = [
        ("Interactions", "ΔH < 0", "Favorable H-bonds, vdW"),
        ("Desolvation", "ΔH > 0", "Breaking water interactions"),
        ("Conformational", "ΔH varies", "Strain in bound state"),
        ("Translational", "ΔS < 0", "Loss of 3 DOF"),
        ("Rotational", "ΔS < 0", "Loss of 3 DOF"),
        ("Vibrational", "ΔS varies", "New modes in complex"),
        ("Solvent", "ΔS > 0", "Release of ordered water"),
    ]
    print(f"  {'Contribution':<15} {'Sign':<10} {'Description'}")
    print("  " + "-" * 50)
    for contrib, sign, desc in contributions:
        print(f"  {contrib:<15} {sign:<10} {desc}")

binding_thermodynamics()
```

<Formula name="Free Energy of Binding">
$$\Delta G_{bind} = G_{complex} - G_{protein} - G_{ligand}$$

Each term includes contributions from potential energy, solvation, and entropy:
$$G = E_{MM} + G_{solv} - TS_{conf}$$
</Formula>

## 2. End-Point Methods: MM-GBSA/MM-PBSA

```python
def mm_gbsa_method():
    """MM-GBSA and MM-PBSA binding energy estimation."""

    print("MM-GBSA/MM-PBSA Methods")
    print("=" * 60)

    print("\nBasic Equation:")
    print("-" * 45)
    print("  ΔG_bind = <G_complex> - <G_protein> - <G_ligand>")
    print()
    print("  G = E_MM + G_solv - TS")
    print()
    print("  E_MM = E_bond + E_angle + E_dihedral + E_vdW + E_elec")
    print("  G_solv = G_polar + G_nonpolar")

    print("\nSolvation Models:")
    print("-" * 45)
    print("  GB (Generalized Born):")
    print("    • Fast (analytical)")
    print("    • Born radii from geometry")
    print("    • Less accurate for buried sites")
    print()
    print("  PB (Poisson-Boltzmann):")
    print("    • More accurate")
    print("    • Numerical solution (grid)")
    print("    • Slower (10-100x)")

    print("\nEntropy Estimation:")
    print("-" * 45)
    print("  Normal Mode Analysis (NMA):")
    print("    S = k Σ[hν/(exp(hν/kT)-1) - ln(1-exp(-hν/kT))]")
    print("    • Requires minimization")
    print("    • Very slow (minutes per frame)")
    print()
    print("  Quasi-harmonic Analysis:")
    print("    • From MD covariance matrix")
    print("    • Faster but less accurate")
    print()
    print("  Often entropy is ignored (ranking only)")

    print("\nProtocol:")
    print("-" * 45)
    steps = [
        "1. Run MD simulation of complex",
        "2. Extract snapshots (50-500 frames)",
        "3. Compute E_MM for each state",
        "4. Compute G_solv (GB or PB)",
        "5. (Optional) Compute entropy",
        "6. Average over snapshots",
        "7. Calculate ΔΔG for ranking",
    ]
    for step in steps:
        print(f"  {step}")

mm_gbsa_method()
```

```python
def mm_gbsa_example():
    """Example MM-GBSA calculation."""

    print("MM-GBSA Example Calculation")
    print("=" * 60)

    # Simulated energy components (kcal/mol)
    complex_data = {
        'E_vdW': -2450.3,
        'E_elec': -18234.5,
        'E_internal': 3421.2,
        'G_GB': -3245.6,
        'G_SA': 45.2,
    }

    protein_data = {
        'E_vdW': -2235.1,
        'E_elec': -17892.3,
        'E_internal': 3398.5,
        'G_GB': -3312.4,
        'G_SA': 42.8,
    }

    ligand_data = {
        'E_vdW': -15.2,
        'E_elec': -45.8,
        'E_internal': 18.2,
        'G_GB': -22.4,
        'G_SA': 3.1,
    }

    def calc_total(data):
        return sum(data.values())

    print("\nEnergy Components (kcal/mol):")
    print("-" * 45)
    print(f"  {'Component':<12} {'Complex':>12} {'Protein':>12} {'Ligand':>12}")
    print("  " + "-" * 50)

    for key in complex_data:
        print(f"  {key:<12} {complex_data[key]:>12.1f} {protein_data[key]:>12.1f} {ligand_data[key]:>12.1f}")

    G_complex = calc_total(complex_data)
    G_protein = calc_total(protein_data)
    G_ligand = calc_total(ligand_data)

    print("  " + "-" * 50)
    print(f"  {'Total':<12} {G_complex:>12.1f} {G_protein:>12.1f} {G_ligand:>12.1f}")

    dG = G_complex - G_protein - G_ligand
    print(f"\n  ΔG_bind = {G_complex:.1f} - ({G_protein:.1f}) - ({G_ligand:.1f})")
    print(f"  ΔG_bind = {dG:.1f} kcal/mol")

    print("\nDecomposition:")
    print("-" * 45)
    dE_vdW = complex_data['E_vdW'] - protein_data['E_vdW'] - ligand_data['E_vdW']
    dE_elec = complex_data['E_elec'] - protein_data['E_elec'] - ligand_data['E_elec']
    dG_solv = (complex_data['G_GB'] + complex_data['G_SA']) - \
              (protein_data['G_GB'] + protein_data['G_SA']) - \
              (ligand_data['G_GB'] + ligand_data['G_SA'])

    print(f"  ΔE_vdW  = {dE_vdW:.1f} kcal/mol")
    print(f"  ΔE_elec = {dE_elec:.1f} kcal/mol")
    print(f"  ΔG_solv = {dG_solv:.1f} kcal/mol")

mm_gbsa_example()
```

<Tip>
MM-GBSA is best for ranking congeneric series (same scaffold, different substituents). Don't trust absolute ΔG values; focus on ΔΔG differences.
</Tip>

## 3. Free Energy Perturbation (FEP)

```python
def fep_method():
    """Free Energy Perturbation theory and methods."""

    print("Free Energy Perturbation (FEP)")
    print("=" * 60)

    print("\nZwanzig Equation:")
    print("-" * 45)
    print("  ΔG(A→B) = -kT ln<exp(-(E_B - E_A)/kT)>_A")
    print()
    print("  Compute ensemble average of energy difference")
    print("  Sample state A, evaluate with Hamiltonian B")

    print("\nAlchemical Transformation:")
    print("-" * 45)
    print("  Transform ligand A → ligand B 'alchemically'")
    print("  H(λ) = (1-λ)H_A + λH_B")
    print()
    print("  λ windows: 0.0 → 0.1 → 0.2 → ... → 1.0")
    print("  Run MD at each λ, compute ΔG for each step")
    print("  Sum: ΔG_total = Σ ΔG(λ_i → λ_{i+1})")

    print("\nThermodynamic Cycle:")
    print("-" * 45)
    print("  For relative binding affinity:")
    print()
    print("       P + A -----> P·A")
    print("         |           |")
    print("      ΔG_1         ΔG_2")
    print("         ↓           ↓")
    print("       P + B -----> P·B")
    print()
    print("  ΔΔG_bind = ΔG_bind(B) - ΔG_bind(A)")
    print("          = ΔG_2 - ΔG_1")
    print()
    print("  Only compute alchemical legs (ΔG_1, ΔG_2)")

    print("\nSoft-Core Potentials:")
    print("-" * 45)
    print("  Problem: Atoms appearing/disappearing")
    print("  Solution: Modified LJ potential")
    print()
    print("  V_sc = 4ε[(α(1-λ)² + (r/σ)⁶)⁻² - (α(1-λ)² + (r/σ)⁶)⁻¹]")
    print()
    print("  Prevents singularities as atoms vanish")

fep_method()
```

<Formula name="Zwanzig Equation">
$$\Delta G_{A \to B} = -k_BT \ln \langle e^{-(E_B - E_A)/k_BT} \rangle_A$$

Where the average is computed over configurations sampled from state A.
</Formula>

```python
    print("\nMBAR (Multistate BAR):")

    print("\nConvergence Checks:")
```

## Summary

In this lesson, you learned:

1. **Binding free energy** determines affinity through ΔG = RT ln(Kd)

3. **FEP** achieves ~1 kcal/mol accuracy for relative affinities
4. **Alchemical methods** transform ligands computationally
5. **QM enhancements** improve accuracy for challenging systems
6. **Method choice** depends on throughput vs accuracy needs
