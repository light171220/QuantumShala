## 1. The Max-Cut Problem

Max-Cut is a fundamental combinatorial optimization problem with applications in physics, circuit design, and machine learning.

<Definition term="Max-Cut Problem">
Given an undirected graph $G = (V, E)$ with edge weights $w_{ij}$, find a partition of vertices into two sets $S$ and $\bar{S}$ that maximizes the total weight of edges crossing the partition:
$$\text{Max-Cut}(G) = \max_{S \subseteq V} \sum_{(i,j) \in E: i \in S, j \notin S} w_{ij}$$
</Definition>

<Formula>
$$\text{Cut value: } C(z) = \sum_{(i,j) \in E} w_{ij} \frac{1 - z_i z_j}{2}$$
</Formula>

where $z_i \in \{-1, +1\}$ indicates which partition vertex $i$ belongs to.

```python
import numpy as np
from typing import List, Tuple, Dict
import networkx as nx

def create_random_graph(n_vertices: int, edge_prob: float = 0.5,
                         weighted: bool = True) -> Tuple[np.ndarray, nx.Graph]:
    """
    Create a random graph for Max-Cut testing.

    Args:
        n_vertices: Number of vertices
        edge_prob: Probability of edge between any two vertices
        weighted: Whether to assign random weights

    Returns:
        (adjacency_matrix, networkx_graph)
    """
    G = nx.Graph()
    G.add_nodes_from(range(n_vertices))

    adj = np.zeros((n_vertices, n_vertices))

    for i in range(n_vertices):
        for j in range(i + 1, n_vertices):
            if np.random.random() < edge_prob:
                weight = np.random.uniform(0.5, 2.0) if weighted else 1.0
                G.add_edge(i, j, weight=weight)
                adj[i, j] = weight
                adj[j, i] = weight

    return adj, G

def evaluate_cut(adj: np.ndarray, partition: np.ndarray) -> float:
    """
    Evaluate the cut value for a given partition.

    Args:
        adj: Adjacency matrix with edge weights
        partition: Array of +1/-1 indicating partition assignment

    Returns:
        Total weight of edges cut
    """
    n = len(partition)
    cut_value = 0.0

    for i in range(n):
        for j in range(i + 1, n):
            if partition[i] != partition[j]:
                cut_value += adj[i, j]

    return cut_value

def brute_force_max_cut(adj: np.ndarray) -> Tuple[float, np.ndarray]:
    """
    Find optimal Max-Cut by brute force enumeration.
    """
    n = adj.shape[0]
    best_cut = 0
    best_partition = np.ones(n)

    for bits in range(2**n):
        partition = np.array([1 if (bits >> i) & 1 else -1 for i in range(n)])
        cut = evaluate_cut(adj, partition)
        if cut > best_cut:
            best_cut = cut
            best_partition = partition.copy()

    return best_cut, best_partition

# Example: Create and analyze a small graph
np.random.seed(42)
n = 6
adj, G = create_random_graph(n, edge_prob=0.6)

optimal_cut, optimal_partition = brute_force_max_cut(adj)

print(f"Graph with {n} vertices, {len(G.edges())} edges")
print(f"Optimal Max-Cut value: {optimal_cut:.3f}")
print(f"Optimal partition: {optimal_partition}")
```

## 2. QAOA: Quantum Approximate Optimization Algorithm

QAOA is a variational quantum algorithm designed for combinatorial optimization problems.

<Definition term="QAOA">
The Quantum Approximate Optimization Algorithm uses a parametrized quantum circuit with alternating cost and mixer layers to prepare states that approximate optimal solutions:
$$|\psi_p(\gamma, \beta)\rangle = \prod_{l=1}^{p} e^{-i\beta_l H_M} e^{-i\gamma_l H_C} |+\rangle^{\otimes n}$$
</Definition>

<Formula>
$$H_C = \sum_{(i,j) \in E} w_{ij} \frac{1 - Z_i Z_j}{2} \quad \text{(Cost Hamiltonian)}$$
$$H_M = \sum_{i=1}^{n} X_i \quad \text{(Mixer Hamiltonian)}$$
</Formula>

```python
from qiskit import QuantumCircuit, QuantumRegister, ClassicalRegister
from qiskit_aer import AerSimulator
from qiskit.quantum_info import Statevector
from scipy.optimize import minimize

def qaoa_circuit(adj: np.ndarray, gamma: List[float], beta: List[float]) -> QuantumCircuit:
    """
    Build QAOA circuit for Max-Cut.

    Args:
        adj: Adjacency matrix
        gamma: Cost layer parameters
        beta: Mixer layer parameters

    Returns:
        QAOA quantum circuit
    """
    n = adj.shape[0]
    p = len(gamma)  # Number of layers

    qc = QuantumCircuit(n)

    # Initial state: uniform superposition
    qc.h(range(n))

    # Alternating layers
    for layer in range(p):
        # Cost layer: exp(-i * gamma * H_C)
        for i in range(n):
            for j in range(i + 1, n):
                if adj[i, j] > 0:
                    # ZZ interaction: exp(-i * gamma * w_ij * (1 - Z_i Z_j) / 2)
                    # This equals exp(-i * gamma * w_ij / 2) * exp(i * gamma * w_ij * Z_i Z_j / 2)
                    angle = gamma[layer] * adj[i, j]
                    qc.cx(i, j)
                    qc.rz(angle, j)
                    qc.cx(i, j)

        # Mixer layer: exp(-i * beta * H_M)
        for i in range(n):
            qc.rx(2 * beta[layer], i)

    return qc

def compute_expectation(adj: np.ndarray, statevector: np.ndarray) -> float:
    """
    Compute expectation value of Max-Cut cost from statevector.
    """
    n = adj.shape[0]
    expectation = 0.0

    # Sum over all basis states
    for bits in range(2**n):
        prob = np.abs(statevector[bits])**2
        # Convert to +1/-1 partition
        partition = np.array([1 if (bits >> i) & 1 else -1 for i in range(n)])
        cut_value = evaluate_cut(adj, partition)
        expectation += prob * cut_value

    return expectation

def qaoa_objective(params: np.ndarray, adj: np.ndarray, p: int) -> float:
    """
    QAOA objective function (negative expectation for minimization).
    """
    gamma = params[:p]
    beta = params[p:]

    qc = qaoa_circuit(adj, list(gamma), list(beta))

    # Get statevector
    sv = Statevector.from_instruction(qc)
    statevector = sv.data

    # Return negative because we want to maximize
    return -compute_expectation(adj, statevector)

def run_qaoa(adj: np.ndarray, p: int = 1, n_trials: int = 10) -> Dict:
    """
    Run QAOA optimization for Max-Cut.

    Args:
        adj: Adjacency matrix
        p: Number of QAOA layers
        n_trials: Number of random initializations

    Returns:
        Dictionary with results
    """
    best_result = None
    best_value = -np.inf

    for trial in range(n_trials):
        # Random initialization
        gamma_init = np.random.uniform(0, 2*np.pi, p)
        beta_init = np.random.uniform(0, np.pi, p)
        params_init = np.concatenate([gamma_init, beta_init])

        # Optimize
        result = minimize(
            qaoa_objective,
            params_init,
            args=(adj, p),
            method='COBYLA',
            options={'maxiter': 500}
        )

        if -result.fun > best_value:
            best_value = -result.fun
            best_result = result

    # Get final circuit and probabilities
    gamma_opt = best_result.x[:p]
    beta_opt = best_result.x[p:]
    qc = qaoa_circuit(adj, list(gamma_opt), list(beta_opt))

    sv = Statevector.from_instruction(qc)
    probs = np.abs(sv.data)**2

    return {
        'optimal_gamma': gamma_opt,
        'optimal_beta': beta_opt,
        'expected_cut': best_value,
        'probabilities': probs
    }

# Run QAOA on our example graph
result_p1 = run_qaoa(adj, p=1, n_trials=5)

print(f"\nQAOA Results (p=1):")
print(f"Expected cut value: {result_p1['expected_cut']:.3f}")
print(f"Optimal cut (brute force): {optimal_cut:.3f}")
print(f"Approximation ratio: {result_p1['expected_cut'] / optimal_cut:.3f}")
```

## 3. QAOA Cost Hamiltonian for Max-Cut

The cost Hamiltonian encodes the Max-Cut objective in terms of Pauli operators.

<Formula>
$$H_C = \sum_{(i,j) \in E} w_{ij} \frac{I - Z_i Z_j}{2}$$
$$= \frac{1}{2}\sum_{(i,j) \in E} w_{ij} I - \frac{1}{2}\sum_{(i,j) \in E} w_{ij} Z_i Z_j$$
</Formula>

<Tip>
The constant term $\frac{1}{2}\sum w_{ij}$ can be ignored in optimization since it doesn't affect the optimal solution. The circuit implements $e^{-i\gamma H_C}$ using ZZ rotations.
</Tip>

```python
def maxcut_hamiltonian(adj: np.ndarray) -> np.ndarray:
    """
    Build the Max-Cut cost Hamiltonian as a matrix.
    """
    n = adj.shape[0]
    dim = 2**n

    # Pauli matrices
    I = np.eye(2)
    Z = np.array([[1, 0], [0, -1]])

    H = np.zeros((dim, dim), dtype=complex)

    for i in range(n):
        for j in range(i + 1, n):
            if adj[i, j] > 0:
                # Build Z_i Z_j operator
                ops = [I] * n
                ops[i] = Z
                ops[j] = Z

                ZZ = ops[0]
                for op in ops[1:]:
                    ZZ = np.kron(ZZ, op)

                # Add (I - ZZ)/2 term
                H += adj[i, j] * (np.eye(dim) - ZZ) / 2

    return H

def verify_cost_encoding(adj: np.ndarray):
    """
    Verify that the Hamiltonian correctly encodes the cut value.
    """
    n = adj.shape[0]
    H = maxcut_hamiltonian(adj)

    print("Verifying cost Hamiltonian encoding:")
    print("-" * 40)

    for bits in range(min(8, 2**n)):  # Check first few states
        # Computational basis state
        state = np.zeros(2**n)
        state[bits] = 1.0

        # Energy from Hamiltonian
        energy = np.real(state @ H @ state)

        # Cut value from direct calculation
        partition = np.array([1 if (bits >> i) & 1 else -1 for i in range(n)])
        cut_value = evaluate_cut(adj, partition)

        match = "OK" if np.isclose(energy, cut_value) else "FAIL"
        print(f"State {bits:0{n}b}: H={energy:.3f}, Cut={cut_value:.3f} [{match}]")

# Verify encoding
verify_cost_encoding(adj)
```

## 4. QAOA Mixer Hamiltonian

The mixer Hamiltonian creates transitions between computational basis states.

<Definition term="Mixer Hamiltonian">
The standard QAOA mixer is $H_M = \sum_i X_i$, which generates transitions that flip individual qubits. The initial state $|+\rangle^{\otimes n}$ is the ground state of $-H_M$.
</Definition>

<Formula>
$$e^{-i\beta H_M} = \prod_{i=1}^{n} e^{-i\beta X_i} = \prod_{i=1}^{n} R_X(2\beta)$$
</Formula>

```python
def mixer_hamiltonian(n: int) -> np.ndarray:
    """
    Build the mixer Hamiltonian: H_M = sum_i X_i
    """
    dim = 2**n

    I = np.eye(2)
    X = np.array([[0, 1], [1, 0]])

    H = np.zeros((dim, dim), dtype=complex)

    for i in range(n):
        ops = [I] * n
        ops[i] = X

        X_i = ops[0]
        for op in ops[1:]:
            X_i = np.kron(X_i, op)

        H += X_i

    return H

def initial_state(n: int) -> np.ndarray:
    """
    Create the initial state |+>^n (ground state of -H_M).
    """
    return np.ones(2**n) / np.sqrt(2**n)

def verify_mixer_ground_state(n: int):
    """
    Verify that |+>^n is the ground state of -H_M.
    """
    H_M = mixer_hamiltonian(n)
    psi_plus = initial_state(n)

    eigenvalues, eigenvectors = np.linalg.eigh(-H_M)

    print(f"Mixer Hamiltonian Analysis (n={n}):")
    print(f"  Ground state energy of -H_M: {eigenvalues[0]:.3f}")
    print(f"  <+|(-H_M)|+> = {np.real(psi_plus @ (-H_M) @ psi_plus):.3f}")
    print(f"  |+> is ground state: {np.allclose(np.abs(eigenvectors[:, 0]), np.abs(psi_plus))}")

verify_mixer_ground_state(4)
```

## 5. QAOA Performance Analysis

QAOA performance depends on the number of layers $p$ and approaches optimal as $p \to \infty$.

<Formula>
$$\lim_{p \to \infty} \max_{\gamma, \beta} \frac{\langle \psi_p | H_C | \psi_p \rangle}{\text{Max-Cut}(G)} = 1$$
</Formula>

```python
def analyze_qaoa_performance(adj: np.ndarray, max_p: int = 4) -> Dict:
    """
    Analyze QAOA performance across different depths.
    """
    optimal_cut, _ = brute_force_max_cut(adj)

    results = []

    for p in range(1, max_p + 1):
        result = run_qaoa(adj, p=p, n_trials=5)

        approximation_ratio = result['expected_cut'] / optimal_cut

        # Find most probable solution
        probs = result['probabilities']
        best_state = np.argmax(probs)
        n = adj.shape[0]
        best_partition = np.array([1 if (best_state >> i) & 1 else -1 for i in range(n)])
        best_state_cut = evaluate_cut(adj, best_partition)

        results.append({
            'p': p,
            'expected_cut': result['expected_cut'],
            'best_state_cut': best_state_cut,
            'approximation_ratio': approximation_ratio,
            'best_state_prob': probs[best_state]
        })

    return {'optimal_cut': optimal_cut, 'results': results}

# Analyze performance
performance = analyze_qaoa_performance(adj, max_p=3)

print(f"\nQAOA Performance Analysis")
print(f"Optimal Cut: {performance['optimal_cut']:.3f}")
print("=" * 60)
print(f"{'p':>3} {'E[Cut]':>10} {'Best Cut':>10} {'Ratio':>8} {'P(best)':>10}")
print("-" * 60)

for r in performance['results']:
    print(f"{r['p']:3d} {r['expected_cut']:10.3f} {r['best_state_cut']:10.3f} "
          f"{r['approximation_ratio']:8.3f} {r['best_state_prob']:10.4f}")
```

## 6. Approximation Ratio Guarantees

For the standard QAOA at $p=1$, there are provable approximation ratio bounds.

<Definition term="Approximation Ratio">
The approximation ratio for a Max-Cut algorithm is $\alpha = \frac{\text{Algorithm's cut}}{\text{Optimal cut}}$. The Goemans-Williamson SDP algorithm achieves $\alpha \geq 0.878$.
</Definition>

<Warning>
For $p=1$ QAOA on 3-regular graphs, the worst-case approximation ratio is approximately $0.6924$. This is worse than the classical GW algorithm but improves with larger $p$.
</Warning>

```python
def goemans_williamson_sdp(adj: np.ndarray) -> Tuple[float, np.ndarray]:
    """
    Simplified Goemans-Williamson style random hyperplane rounding.
    (Not the full SDP, but demonstrates the approach)
    """
    n = adj.shape[0]

    # Use spectral method as approximation
    # Compute Laplacian
    D = np.diag(np.sum(adj, axis=1))
    L = D - adj

    # Get eigenvector corresponding to second smallest eigenvalue
    eigenvalues, eigenvectors = np.linalg.eigh(L)
    fiedler_vector = eigenvectors[:, 1]

    # Random hyperplane rounding
    best_cut = 0
    best_partition = None

    for _ in range(100):
        # Random direction
        direction = np.random.randn(1)

        # Partition based on sign
        partition = np.sign(fiedler_vector - direction[0] * np.std(fiedler_vector))
        partition[partition == 0] = 1

        cut = evaluate_cut(adj, partition)
        if cut > best_cut:
            best_cut = cut
            best_partition = partition.copy()

    return best_cut, best_partition

def compare_algorithms(adj: np.ndarray) -> Dict:
    """
    Compare QAOA with classical algorithms.
    """
    optimal_cut, optimal_partition = brute_force_max_cut(adj)
    gw_cut, gw_partition = goemans_williamson_sdp(adj)
    qaoa_result = run_qaoa(adj, p=2, n_trials=5)

    # Random baseline
    n = adj.shape[0]
    random_cuts = [evaluate_cut(adj, np.random.choice([-1, 1], n))
                   for _ in range(1000)]
    random_avg = np.mean(random_cuts)

    return {
        'optimal': optimal_cut,
        'gw_approx': gw_cut,
        'gw_ratio': gw_cut / optimal_cut,
        'qaoa_expected': qaoa_result['expected_cut'],
        'qaoa_ratio': qaoa_result['expected_cut'] / optimal_cut,
        'random_avg': random_avg,
        'random_ratio': random_avg / optimal_cut
    }

# Compare algorithms
comparison = compare_algorithms(adj)

print("\nAlgorithm Comparison:")
print("=" * 50)
print(f"{'Algorithm':>15} {'Cut Value':>12} {'Ratio':>10}")
print("-" * 50)
print(f"{'Optimal':>15} {comparison['optimal']:12.3f} {1.000:10.3f}")
print(f"{'GW Approx':>15} {comparison['gw_approx']:12.3f} {comparison['gw_ratio']:10.3f}")
print(f"{'QAOA (p=2)':>15} {comparison['qaoa_expected']:12.3f} {comparison['qaoa_ratio']:10.3f}")
print(f"{'Random':>15} {comparison['random_avg']:12.3f} {comparison['random_ratio']:10.3f}")
```

## 7. Parameter Optimization Strategies

Optimizing QAOA parameters is crucial and several strategies exist.

```python
def linear_ramp_parameters(p: int, total_time: float = 1.0) -> Tuple[np.ndarray, np.ndarray]:
    """
    Initialize parameters using linear ramp schedule.
    gamma: 0 -> pi/2, beta: pi/4 -> 0
    """
    gamma = np.linspace(0.1, np.pi/2, p) * total_time
    beta = np.linspace(np.pi/4, 0.1, p) * total_time
    return gamma, beta

def interp_parameters(p: int, gamma_prev: np.ndarray, beta_prev: np.ndarray) -> Tuple[np.ndarray, np.ndarray]:
    """
    Initialize p-layer parameters by interpolating from (p-1)-layer optimal.
    """
    p_prev = len(gamma_prev)

    if p_prev == 0:
        return linear_ramp_parameters(p)

    # Interpolation indices
    indices = np.linspace(0, p_prev - 1, p)

    gamma = np.interp(indices, np.arange(p_prev), gamma_prev)
    beta = np.interp(indices, np.arange(p_prev), beta_prev)

    return gamma, beta

def layerwise_training(adj: np.ndarray, max_p: int = 3) -> Dict:
    """
    Train QAOA layer by layer with parameter transfer.
    """
    n = adj.shape[0]
    gamma_opt = np.array([])
    beta_opt = np.array([])

    results = []

    for p in range(1, max_p + 1):
        # Initialize from previous layer
        gamma_init, beta_init = interp_parameters(p, gamma_opt, beta_opt)
        params_init = np.concatenate([gamma_init, beta_init])

        # Optimize
        result = minimize(
            qaoa_objective,
            params_init,
            args=(adj, p),
            method='COBYLA',
            options={'maxiter': 300}
        )

        gamma_opt = result.x[:p]
        beta_opt = result.x[p:]

        results.append({
            'p': p,
            'expected_cut': -result.fun,
            'gamma': gamma_opt.copy(),
            'beta': beta_opt.copy()
        })

    return results

# Run layerwise training
layerwise_results = layerwise_training(adj, max_p=3)

print("\nLayerwise Training Results:")
print("=" * 50)
optimal_cut, _ = brute_force_max_cut(adj)

for r in layerwise_results:
    print(f"\np={r['p']}:")
    print(f"  Expected cut: {r['expected_cut']:.3f} (ratio: {r['expected_cut']/optimal_cut:.3f})")
    print(f"  gamma: {np.round(r['gamma'], 3)}")
    print(f"  beta: {np.round(r['beta'], 3)}")
```

## 8. Warm-Starting QAOA

Warm-starting uses classical solutions to improve QAOA performance.

<Definition term="Warm-Start QAOA">
Instead of starting from $|+\rangle^{\otimes n}$, warm-start QAOA initializes from a state biased toward a classical solution, potentially reaching better optima faster.
</Definition>

```python
def warm_start_state_circuit(n: int, classical_solution: np.ndarray,
                              bias: float = 0.3) -> QuantumCircuit:
    """
    Create initial state biased toward a classical solution.

    Args:
        n: Number of qubits
        classical_solution: Array of +1/-1
        bias: How strongly to bias toward solution (0 = uniform, 1 = deterministic)
    """
    qc = QuantumCircuit(n)

    # Rotation angle for bias
    theta = np.pi/2 - bias * np.pi/2

    for i in range(n):
        if classical_solution[i] == 1:
            qc.ry(theta, i)
        else:
            qc.ry(np.pi - theta, i)

    return qc

def warm_start_qaoa(adj: np.ndarray, p: int, warm_solution: np.ndarray,
                     bias: float = 0.3) -> Dict:
    """
    Run warm-started QAOA.
    """
    n = adj.shape[0]

    def warm_qaoa_objective(params):
        gamma = params[:p]
        beta = params[p:]

        # Create circuit
        qc = warm_start_state_circuit(n, warm_solution, bias)

        # Add QAOA layers
        for layer in range(p):
            # Cost layer
            for i in range(n):
                for j in range(i + 1, n):
                    if adj[i, j] > 0:
                        angle = gamma[layer] * adj[i, j]
                        qc.cx(i, j)
                        qc.rz(angle, j)
                        qc.cx(i, j)

            # Mixer layer
            for i in range(n):
                qc.rx(2 * beta[layer], i)

        # Get expectation
        sv = Statevector.from_instruction(qc)
        return -compute_expectation(adj, sv.data)

    # Get classical solution from GW
    gw_cut, gw_partition = goemans_williamson_sdp(adj)

    # Optimize
    params_init = np.random.uniform(0, np.pi, 2 * p)
    result = minimize(
        warm_qaoa_objective,
        params_init,
        method='COBYLA',
        options={'maxiter': 300}
    )

    return {
        'expected_cut': -result.fun,
        'warm_start_cut': gw_cut,
        'optimal_params': result.x
    }

# Compare standard vs warm-start QAOA
standard_result = run_qaoa(adj, p=2, n_trials=3)

_, gw_partition = goemans_williamson_sdp(adj)
warm_result = warm_start_qaoa(adj, p=2, gw_partition, bias=0.3)

print("\nStandard vs Warm-Start QAOA:")
print(f"  Standard QAOA (p=2): {standard_result['expected_cut']:.3f}")
print(f"  Warm-start QAOA (p=2): {warm_result['expected_cut']:.3f}")
print(f"  Warm-start initial: {warm_result['warm_start_cut']:.3f}")
```

## Summary

In this lesson, you learned:

1. **Max-Cut problem**: Finding a partition that maximizes edges cut is NP-hard
2. **QAOA structure**: Alternating cost ($e^{-i\gamma H_C}$) and mixer ($e^{-i\beta H_M}$) layers
3. **Cost Hamiltonian**: Encoded as $\sum w_{ij}(I - Z_i Z_j)/2$ using ZZ gates
4. **Mixer Hamiltonian**: $\sum X_i$ creates transitions between solutions
5. **Performance scaling**: Approximation ratio improves with depth $p$
6. **Parameter strategies**: Layerwise training, interpolation, and warm-starting improve optimization
7. **Classical comparison**: QAOA competes with GW algorithm, especially at higher $p$

QAOA represents a promising near-term approach to combinatorial optimization, with ongoing research into its quantum advantages and optimal implementation strategies.
