## 2. Block Encodings

Block encodings are a fundamental technique for embedding non-unitary matrices into larger unitary operations.

<Definition term="Block Encoding">
A $(\\alpha, m, \\epsilon)$-block encoding of an $n$-qubit operator $A$ is a unitary $U$ acting on $n + m$ qubits such that $\\|A - \\alpha(\\langle 0|^{\\otimes m} \\otimes I)U(|0\\rangle^{\\otimes m} \\otimes I)\\| \\leq \\epsilon$, where $\\alpha$ is a normalization factor.
</Definition>

<Formula>
$$U = \\begin{pmatrix} A/\\alpha & \\cdot \\\\ \\cdot & \\cdot \\end{pmatrix}$$
</Formula>

### Constructing Block Encodings

```python
import numpy as np
from scipy.linalg import block_diag

class BlockEncoding:
    """
    Implements block encoding of matrices.
    """

    def __init__(self, matrix: np.ndarray, alpha: float = None):
        """
        Create a block encoding of a matrix.

        Args:
            matrix: The matrix to encode
            alpha: Normalization factor (default: spectral norm)
        """
        self.matrix = matrix
        self.n = int(np.log2(matrix.shape[0]))
        self.alpha = alpha if alpha else np.linalg.norm(matrix, 2)
        self.normalized = matrix / self.alpha

    def construct_unitary(self) -> np.ndarray:
        """
        Construct the block-encoded unitary.
        Uses the standard construction via SVD.
        """
        A = self.normalized

        # SVD: A = U @ Sigma @ Vh
        U, sigma, Vh = np.linalg.svd(A, full_matrices=True)

        # Build the block encoding using the SVD
        # The unitary is constructed as a direct sum and controlled operations
        n = A.shape[0]

        # Create the full unitary
        # Top-left block is A, completed to be unitary
        sqrt_complement = np.sqrt(np.maximum(0, 1 - sigma**2))

        # Construct W such that [A; B] forms the left columns of a unitary
        B = U @ np.diag(sqrt_complement) @ Vh

        # Build full unitary (simplified 2x2 block structure)
        full_unitary = np.block([
            [A, np.sqrt(np.eye(n) - A @ A.conj().T)],
            [np.sqrt(np.eye(n) - A.conj().T @ A), -A.conj().T]
        ])

        # Ensure unitarity through QR decomposition cleanup
        Q, R = np.linalg.qr(full_unitary)
        return Q

    def verify_encoding(self, unitary: np.ndarray) -> dict:
        """Verify the block encoding is correct."""
        n = self.matrix.shape[0]

        # Extract top-left block
        extracted = unitary[:n, :n] * self.alpha

        # Compute error
        error = np.linalg.norm(extracted - self.matrix)

        # Check unitarity
        unitarity_error = np.linalg.norm(
            unitary @ unitary.conj().T - np.eye(unitary.shape[0])
        )

        return {
            'encoding_error': error,
            'unitarity_error': unitarity_error,
            'alpha': self.alpha,
            'ancilla_qubits': int(np.log2(unitary.shape[0])) - self.n
        }

# Example: Block encode a Hamiltonian
def example_block_encoding():
    """Demonstrate block encoding of a Hamiltonian."""
    # Simple 2-qubit Hamiltonian
    H = np.array([
        [1, 0, 0, 0.5],
        [0, -1, 0.5, 0],
        [0, 0.5, -1, 0],
        [0.5, 0, 0, 1]
    ], dtype=complex)

    be = BlockEncoding(H)
    U = be.construct_unitary()
    verification = be.verify_encoding(U)

    print("Block Encoding Example")
    print("=" * 40)
    print(f"Original matrix shape: {H.shape}")
    print(f"Block-encoded unitary shape: {U.shape}")
    print(f"Normalization factor alpha: {verification['alpha']:.4f}")
    print(f"Encoding error: {verification['encoding_error']:.2e}")
    print(f"Unitarity error: {verification['unitarity_error']:.2e}")

    return be, U
```

### Types of Block Encodings

<Tip>
Different types of block encodings are suited for different applications. Sparse-access block encodings are efficient for sparse matrices, while LCU-based encodings work well for Hamiltonians expressed as linear combinations of unitaries.
</Tip>

```python
from enum import Enum
from abc import ABC, abstractmethod

class BlockEncodingType(Enum):
    """Types of block encoding constructions."""
    SPARSE_ACCESS = "sparse_access"
    LCU = "linear_combination_of_unitaries"
    PURIFICATION = "purification"
    GRAM_MATRIX = "gram_matrix"

class BlockEncodingMethod(ABC):
    """Abstract base class for block encoding methods."""

    @abstractmethod
    def encode(self, operator) -> np.ndarray:
        """Create the block-encoded unitary."""
        pass

    @abstractmethod
    def get_complexity(self) -> dict:
        """Return complexity metrics."""
        pass

class LCUBlockEncoding(BlockEncodingMethod):
    """
    Block encoding via Linear Combination of Unitaries.

    For H = sum_j alpha_j U_j, creates block encoding with
    normalization factor sum_j |alpha_j|.
    """

    def __init__(self, coefficients: List[complex], unitaries: List[np.ndarray]):
        self.coefficients = np.array(coefficients)
        self.unitaries = unitaries
        self.alpha = np.sum(np.abs(self.coefficients))

    def encode(self, operator=None) -> np.ndarray:
        """Create LCU block encoding."""
        L = len(self.coefficients)
        n = self.unitaries[0].shape[0]

        # Number of ancilla qubits for coefficient selection
        m = int(np.ceil(np.log2(L)))
        ancilla_dim = 2**m

        # Prepare coefficient state |psi> = sum sqrt(|alpha_j|/alpha) |j>
        coeff_state = np.sqrt(np.abs(self.coefficients) / self.alpha)
        coeff_state = np.pad(coeff_state, (0, ancilla_dim - L))

        # The full construction requires:
        # 1. PREPARE: |0> -> |psi>
        # 2. SELECT: |j>|x> -> |j>U_j|x>
        # 3. PREPARE^dagger

        return self._construct_full_unitary(coeff_state, n, ancilla_dim)

    def _construct_full_unitary(self, coeff_state, n, ancilla_dim):
        """Construct the full block-encoded unitary."""
        # Simplified construction for demonstration
        total_dim = ancilla_dim * n
        U = np.zeros((total_dim, total_dim), dtype=complex)

        # Build block structure
        for j, (c, Uj) in enumerate(zip(self.coefficients, self.unitaries)):
            weight = np.sqrt(np.abs(c) / self.alpha)
            # Place weighted unitaries in appropriate blocks
            U[j*n:(j+1)*n, :n] = weight * Uj

        # Complete to unitary via QR
        Q, _ = np.linalg.qr(U)
        return Q

    def get_complexity(self) -> dict:
        """Return complexity of LCU block encoding."""
        L = len(self.coefficients)
        return {
            'ancilla_qubits': int(np.ceil(np.log2(L))),
            'normalization': self.alpha,
            'num_unitaries': L,
            'select_complexity': 'O(L * unitary_cost)',
            'prepare_complexity': 'O(L)'
        }

class SparseAccessBlockEncoding(BlockEncodingMethod):
    """
    Block encoding for sparse matrices.

    Requires oracles for:
    - O_F: |i,k> -> |i, f(i,k)> (k-th nonzero in row i)
    - O_A: |i,j,z> -> |i,j,z XOR A_ij> (matrix elements)
    """

    def __init__(self, matrix: np.ndarray, sparsity: int = None):
        self.matrix = matrix
        self.n = matrix.shape[0]
        self.sparsity = sparsity or self._compute_sparsity()
        self.alpha = self.sparsity * np.max(np.abs(matrix))

    def _compute_sparsity(self) -> int:
        """Compute the maximum row/column sparsity."""
        row_sparsity = np.max(np.sum(self.matrix != 0, axis=1))
        col_sparsity = np.max(np.sum(self.matrix != 0, axis=0))
        return max(row_sparsity, col_sparsity)

    def encode(self, operator=None) -> np.ndarray:
        """Create sparse-access block encoding."""
        # Simplified construction
        A = self.matrix / self.alpha

        # Complete to unitary
        dim = 2 * self.n
        U = np.zeros((dim, dim), dtype=complex)
        U[:self.n, :self.n] = A

        # Fill remaining blocks
        sqrt_comp = np.sqrt(np.maximum(0, np.eye(self.n) - A @ A.conj().T))
        U[:self.n, self.n:] = sqrt_comp
        U[self.n:, :self.n] = sqrt_comp
        U[self.n:, self.n:] = -A.conj().T

        return U

    def get_complexity(self) -> dict:
        """Return complexity of sparse-access block encoding."""
        return {
            'ancilla_qubits': int(np.ceil(np.log2(self.sparsity))) + 1,
            'normalization': self.alpha,
            'sparsity': self.sparsity,
            'oracle_queries': 2,  # O_F and O_A
            'total_complexity': f'O({self.sparsity})'
        }
```

## 3. Quantum Signal Processing

Quantum Signal Processing (QSP) provides a systematic way to apply polynomial transformations to quantum operations.

<Definition term="Quantum Signal Processing">
QSP is a technique for implementing polynomial transformations of eigenvalues. Given access to a single-qubit rotation $e^{i\\theta Z}$, QSP can implement $P(\\cos\\theta)$ for any polynomial $P$ satisfying certain parity and boundedness conditions.
</Definition>

<Formula>
$$U_{\\text{QSP}}(\\theta) = e^{i\\phi_0 Z} \\prod_{k=1}^{d} \\left[ e^{i\\theta X} e^{i\\phi_k Z} \\right] = \\begin{pmatrix} P(\\cos\\theta) & iQ(\\cos\\theta)\\sin\\theta \\\\ iQ^*(\\cos\\theta)\\sin\\theta & P^*(\\cos\\theta) \\end{pmatrix}$$
</Formula>

```python
class QuantumSignalProcessing:
    """
    Implements Quantum Signal Processing for polynomial transformations.
    """

    def __init__(self, target_polynomial: Callable, degree: int):
        """
        Initialize QSP for a target polynomial.

        Args:
            target_polynomial: Function P(x) to implement
            degree: Degree of the polynomial
        """
        self.target_poly = target_polynomial
        self.degree = degree
        self.phases = None

    def find_qsp_phases(self, precision: float = 1e-10) -> np.ndarray:
        """
        Find QSP phase angles to implement the target polynomial.

        Uses optimization to find phases phi_0, ..., phi_d.
        """
        from scipy.optimize import minimize

        d = self.degree

        def qsp_matrix(theta: float, phases: np.ndarray) -> np.ndarray:
            """Compute QSP unitary for given theta and phases."""
            # Signal rotation
            W = np.array([
                [np.cos(theta), 1j * np.sin(theta)],
                [1j * np.sin(theta), np.cos(theta)]
            ])

            # Initialize with first phase
            result = self._phase_rotation(phases[0])

            # Interleave signal and phase rotations
            for k in range(1, len(phases)):
                result = result @ W @ self._phase_rotation(phases[k])

            return result

        def objective(phases: np.ndarray) -> float:
            """Objective: match target polynomial at sample points."""
            thetas = np.linspace(0, np.pi, 50)
            error = 0

            for theta in thetas:
                U = qsp_matrix(theta, phases)
                achieved = np.real(U[0, 0])
                target = self.target_poly(np.cos(theta))
                error += (achieved - target)**2

            return error

        # Initial guess
        initial_phases = np.zeros(d + 1)

        # Optimize
        result = minimize(
            objective,
            initial_phases,
            method='BFGS',
            options={'gtol': precision}
        )

        self.phases = result.x
        return self.phases

    def _phase_rotation(self, phi: float) -> np.ndarray:
        """Z-rotation by angle phi."""
        return np.array([
            [np.exp(1j * phi), 0],
            [0, np.exp(-1j * phi)]
        ])

    def evaluate(self, theta: float) -> complex:
        """Evaluate the QSP transformation at angle theta."""
        if self.phases is None:
            raise ValueError("Must find phases first")

        W = np.array([
            [np.cos(theta), 1j * np.sin(theta)],
            [1j * np.sin(theta), np.cos(theta)]
        ])

        result = self._phase_rotation(self.phases[0])
        for phi in self.phases[1:]:
            result = result @ W @ self._phase_rotation(phi)

        return result[0, 0]

    def verify(self, num_points: int = 100) -> dict:
        """Verify QSP implementation matches target polynomial."""
        thetas = np.linspace(0, np.pi, num_points)

        achieved = [np.real(self.evaluate(t)) for t in thetas]
        target = [self.target_poly(np.cos(t)) for t in thetas]

        max_error = np.max(np.abs(np.array(achieved) - np.array(target)))
        avg_error = np.mean(np.abs(np.array(achieved) - np.array(target)))

        return {
            'max_error': max_error,
            'avg_error': avg_error,
            'num_phases': len(self.phases),
            'phases': self.phases
        }

# Example: Implement threshold function approximation
def qsp_threshold_example():
    """Demonstrate QSP for threshold function."""
    # Approximate sign function using Chebyshev polynomial
    def chebyshev_sign(x, degree=10):

        result = 0
        for k in range(degree):
            # Simplified approximation
            result += (2 * (k % 2) - 1) * x ** (2*k + 1) / (2*k + 1)

    degree = 15
    qsp = QuantumSignalProcessing(

    print("QSP Threshold Function Example")
```

## 4. Quantum Singular Value Transformation (QSVT)

QSVT is a unifying framework that generalizes quantum signal processing to matrix functions.

<Definition term="Quantum Singular Value Transformation">
QSVT applies polynomial transformations to the singular values of a block-encoded matrix. Given an $(\\alpha, m, 0)$-block encoding of $A = \\sum_j \\sigma_j |u_j\\rangle\\langle v_j|$, QSVT implements block encoding of $\\sum_j P(\\sigma_j) |u_j\\rangle\\langle v_j|$ for polynomial $P$.
</Definition>

<Formula>
$$\\text{QSVT}[A] = \\sum_j P(\\sigma_j/\\alpha) |u_j\\rangle\\langle v_j|$$
</Formula>

```python
class QSVT:
    """
    Quantum Singular Value Transformation implementation.

    Unifies quantum algorithms through polynomial transformations
    of singular values.
    """

    def __init__(self, block_encoding: BlockEncoding):
        """
        Initialize QSVT with a block encoding.

        Args:
            block_encoding: Block encoding of target matrix
        """
        self.be = block_encoding
        self.U = block_encoding.construct_unitary()

    def apply_polynomial(
        self,
        polynomial_coeffs: np.ndarray,
        parity: str = 'even'
    ) -> np.ndarray:
        """
        Apply polynomial transformation to singular values.

        Args:
            polynomial_coeffs: Coefficients of the polynomial
            parity: 'even' or 'odd' polynomial parity

        Returns:
            Transformed matrix
        """
        # Get singular value decomposition of original matrix
        A = self.be.normalized
        U, sigma, Vh = np.linalg.svd(A, full_matrices=False)

        # Apply polynomial to singular values
        def eval_polynomial(x):
            return sum(c * x**k for k, c in enumerate(polynomial_coeffs))

        transformed_sigma = np.array([eval_polynomial(s) for s in sigma])

        # Reconstruct matrix
        return U @ np.diag(transformed_sigma) @ Vh

    def implement_matrix_function(
        self,
        func: Callable,
        degree: int,
        interval: Tuple[float, float] = (-1, 1)
    ) -> Tuple[np.ndarray, dict]:
        """
        Implement a matrix function via QSVT.

        Uses polynomial approximation of the function.
        """
        # Approximate function with Chebyshev polynomial
        coeffs = self._chebyshev_approximation(func, degree, interval)

        # Apply the polynomial
        result = self.apply_polynomial(coeffs)

        # Compute true result for comparison
        A = self.be.matrix
        U, sigma, Vh = np.linalg.svd(A, full_matrices=False)
        true_result = U @ np.diag([func(s) for s in sigma]) @ Vh

        error = np.linalg.norm(result * self.be.alpha - true_result)

        return result * self.be.alpha, {
            'approximation_degree': degree,
            'error': error,
            'polynomial_coeffs': coeffs[:5]  # First 5 coefficients
        }

    def _chebyshev_approximation(
        self,
        func: Callable,
        degree: int,
        interval: Tuple[float, float]
    ) -> np.ndarray:
        """Compute Chebyshev polynomial approximation."""
        a, b = interval

        # Chebyshev nodes
        k = np.arange(degree + 1)
        nodes = np.cos((2*k + 1) * np.pi / (2 * (degree + 1)))

        # Scale to interval
        scaled_nodes = 0.5 * (b - a) * nodes + 0.5 * (a + b)

        # Function values at nodes
        fvals = np.array([func(x) for x in scaled_nodes])

        # Compute Chebyshev coefficients
        coeffs = np.zeros(degree + 1)
        for j in range(degree + 1):
            Tj = np.cos(j * np.arccos(nodes))
            coeffs[j] = 2 / (degree + 1) * np.sum(fvals * Tj)
        coeffs[0] /= 2

        return coeffs

# Unified algorithm implementations via QSVT
class QSVTAlgorithms:
    """
    Standard quantum algorithms implemented via QSVT framework.
    """

    @staticmethod
    def matrix_inversion(
        qsvt: QSVT,
        threshold: float = 0.1,
        degree: int = 50
    ) -> np.ndarray:
        """
        HHL-style matrix inversion via QSVT.

        Implements f(x) = 1/x for |x| >= threshold.
        """
        def regularized_inverse(x):
            if abs(x) < threshold:
                return 0
            return 1 / x

        result, info = qsvt.implement_matrix_function(
            regularized_inverse, degree
        )
        return result

    @staticmethod
    def hamiltonian_simulation(
        qsvt: QSVT,
        time: float,
        degree: int = None
    ) -> np.ndarray:
        """
        Hamiltonian simulation e^{-iHt} via QSVT.
        """
        # Degree needed for precision epsilon
        if degree is None:
            degree = int(2 * np.abs(time) * qsvt.be.alpha + 10)

        result, info = qsvt.implement_matrix_function(
            lambda x: np.exp(-1j * time * x),
            degree
        )
        return result

    @staticmethod
    def amplitude_amplification(
        qsvt: QSVT,
        target_amplitude: float,
        num_iterations: int
    ) -> np.ndarray:
        """
        Amplitude amplification via QSVT.

        Implements polynomial that amplifies eigenvalue target_amplitude.
        """
        # Chebyshev polynomial of the first kind
        def amplification_poly(x):
            theta = np.arcsin(target_amplitude)
            return np.sin((2 * num_iterations + 1) * np.arcsin(x))

        result, info = qsvt.implement_matrix_function(
            amplification_poly,
            2 * num_iterations + 1
        )
        return result
```

### Eigenvalue Filtering

```python
class EigenvalueFilter:
    """
    Eigenvalue filtering using QSVT.

    Projects onto eigenspaces with eigenvalues in specified ranges.
    """

    def __init__(self, qsvt: QSVT):
        self.qsvt = qsvt

    def threshold_projector(
        self,
        threshold: float,
        above: bool = True,
        degree: int = 50
    ) -> np.ndarray:
        """
        Project onto eigenspace with eigenvalues above/below threshold.

        Uses polynomial approximation to step function.
        """
        # Smooth step function approximation
        def smooth_step(x, kappa=20):
            return 0.5 * (1 + np.tanh(kappa * (x - threshold)))

        if not above:
            orig_step = smooth_step
            smooth_step = lambda x: 1 - orig_step(x)

        result, info = self.qsvt.implement_matrix_function(
            smooth_step, degree
        )
        return result

    def band_filter(
        self,
        low: float,
        high: float,
        degree: int = 50
    ) -> np.ndarray:
        """
        Project onto eigenspace with eigenvalues in [low, high].
        """
        def band(x, kappa=20):
            above_low = 0.5 * (1 + np.tanh(kappa * (x - low)))
            below_high = 0.5 * (1 + np.tanh(kappa * (high - x)))
            return above_low * below_high

        result, _ = self.qsvt.implement_matrix_function(band, degree)
        return result
```

```python
class GroundStatePreparation:
    """
    Prepare ground state of Hamiltonian using QSVT.
    """

    def __init__(self, hamiltonian: np.ndarray):

    def prepare(self, initial_state: np.ndarray, degree: int = 100) -> np.ndarray:
        """
        Prepare ground state from initial state.

        Uses eigenvalue filtering to project onto ground state.
        """
        # Normalize energies
        E_max = np.max(np.abs(np.linalg.eigvalsh(self.H)))

        # Filter function: peaks at ground state energy
        def ground_filter(E):
            # Gaussian-like filter centered at E0
            width = self.gap / 4
            return np.exp(-((E - self.E0) / width)**2)

        # Apply filter via QSVT
        filter_result, _ = self.qsvt.implement_matrix_function(
            ground_filter, degree
        )

        # Apply to initial state
        ground_state = filter_result @ initial_state

        # Normalize
        norm = np.linalg.norm(ground_state)
        if norm > 1e-10:
            ground_state /= norm

        return ground_state

    def verify(self, prepared_state: np.ndarray) -> dict:
        """Verify ground state preparation quality."""
        # True ground state
        eigenvalues, eigenvectors = np.linalg.eigh(self.H)
        true_ground = eigenvectors[:, 0]

        # Overlap
        overlap = np.abs(np.vdot(prepared_state, true_ground))**2

        # Energy
        energy = np.real(np.vdot(prepared_state, self.H @ prepared_state))

        return {
            'ground_state_overlap': overlap,
            'energy': energy,
            'true_ground_energy': self.E0,
            'energy_error': np.abs(energy - self.E0)
        }
```

## 6. Complexity and Resource Analysis

<Formula>
$$\\text{QSVT Query Complexity: } O\\left(\\frac{\\alpha}{\\epsilon} \\cdot \\text{poly}(d)\\right)$$
</Formula>

where $\\alpha$ is the block encoding normalization, $\\epsilon$ is the approximation error, and $d$ is the polynomial degree.

```python
@dataclass
class QSVTResources:
    """Resource estimation for QSVT algorithms."""
    block_encoding_queries: int
    ancilla_qubits: int
    total_gates: int
    polynomial_degree: int

def estimate_qsvt_resources(
    matrix_size: int,
    target_precision: float,
    normalization: float,
    application: str
) -> QSVTResources:
    """
    Estimate resources needed for QSVT application.

    Args:
        matrix_size: Dimension of the matrix
        target_precision: Desired precision epsilon
        normalization: Block encoding normalization alpha
        application: Type of application
    """
    # Polynomial degree depends on application
    if application == 'matrix_inversion':
        # HHL: degree ~ kappa * log(1/epsilon)
        kappa = normalization  # Condition number estimate
        degree = int(kappa * np.log(1/target_precision))
    elif application == 'hamiltonian_simulation':
        # Simulation: degree ~ alpha * t
        degree = int(2 * normalization + np.log(1/target_precision))
    elif application == 'eigenvalue_filtering':
        # Filtering: degree ~ log(1/epsilon) / gap
        degree = int(10 * np.log(1/target_precision))
    else:
        degree = int(np.log(1/target_precision) * 10)

    # Block encoding queries = 2 * degree (controlled-U and controlled-U^dagger)
    be_queries = 2 * degree

    # Ancilla qubits: 1 for QSVT + block encoding ancillas
    n_qubits = int(np.log2(matrix_size))
    ancilla = 1 + n_qubits  # Simplified estimate

    # Total gates: queries * gates_per_query
    total_gates = be_queries * matrix_size * int(np.log2(matrix_size))

    return QSVTResources(
        block_encoding_queries=be_queries,
        ancilla_qubits=ancilla,
        total_gates=total_gates,
        polynomial_degree=degree
    )

def compare_algorithm_resources():
    """Compare resource requirements across QSVT applications."""
    applications = ['matrix_inversion', 'hamiltonian_simulation', 'eigenvalue_filtering']

    print("QSVT Resource Comparison")
    print("=" * 60)
    print(f"{'Application':<25} {'Queries':<12} {'Ancillas':<10} {'Degree':<10}")
    print("-" * 60)

    for app in applications:
        resources = estimate_qsvt_resources(
            matrix_size=256,  # 8 qubits
            target_precision=1e-3,
            normalization=10.0,
            application=app
        )
        print(f"{app:<25} {resources.block_encoding_queries:<12} "
              f"{resources.ancilla_qubits:<10} {resources.polynomial_degree:<10}")
```
