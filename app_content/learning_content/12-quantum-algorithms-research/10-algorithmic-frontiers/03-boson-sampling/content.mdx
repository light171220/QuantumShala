## 1. Introduction to Boson Sampling

**Boson sampling** was proposed by Aaronson and Arkhipov (2011) as a non-universal model of quantum computation that is nevertheless hard to simulate classically.

<Definition term="Boson Sampling Problem">
Given n identical photons input to an m-mode linear optical network described by unitary U, sample from the output distribution where the probability of detecting photons in modes S is proportional to |Perm(U_S)|^2.
</Definition>

<Formula>
$$P(S) = \frac{|\text{Perm}(U_S)|^2}{\prod_{i} s_i!}$$
</Formula>

where U_S is the n x n submatrix of U corresponding to input and output modes.

```python
import numpy as np
from typing import List, Tuple, Optional
from itertools import combinations
from functools import lru_cache

class BosonSamplingSimulator:
    """Simulates boson sampling with linear optical networks."""

    def __init__(self, n_modes: int, n_photons: int, seed: Optional[int] = None):
        self.m = n_modes
        self.n = n_photons
        self.rng = np.random.default_rng(seed)
        self.unitary = None

    def random_unitary(self) -> np.ndarray:
        """Generate Haar-random unitary matrix using QR decomposition."""
        z = self.rng.standard_normal((self.m, self.m)) + \
            1j * self.rng.standard_normal((self.m, self.m))
        q, r = np.linalg.qr(z)
        d = np.diag(r)
        self.unitary = q @ np.diag(d / np.abs(d))
        return self.unitary

    def permanent_naive(self, matrix: np.ndarray) -> complex:
        """Compute permanent using Ryser's formula (O(2^n * n))."""
        n = matrix.shape[0]
        if n == 0:
            return 1.0

        result = 0.0
        for k in range(1, 2**n):
            bits = bin(k)[2:].zfill(n)
            cols = [j for j, b in enumerate(bits) if b == '1']
            sign = (-1) ** (n - len(cols))
            col_sums = matrix[:, cols].sum(axis=1)
            result += sign * np.prod(col_sums)

        return ((-1) ** n) * result
```

## 2. The Permanent and Computational Hardness

<Definition term="Matrix Permanent">
For an n x n matrix A, the permanent is Perm(A) = sum over all permutations sigma of product A[i, sigma(i)].
</Definition>

<Insight>
Computing the permanent is #P-hard (Valiant, 1979), meaning it's at least as hard as counting satisfying assignments to Boolean formulas.
</Insight>

<Formula>
$$\text{Perm}(A) = \sum_{\sigma \in S_n} \prod_{i=1}^{n} A_{i,\sigma(i)}$$
</Formula>

```python
def ryser_permanent(matrix: np.ndarray) -> complex:
    """
    Ryser's algorithm for permanent computation.
    Complexity: O(2^n * n) - the fastest known exact algorithm.
    """
    n = matrix.shape[0]
    if n == 0:
        return 1.0 + 0j

    # Gray code optimization
    row_sums = np.zeros(n, dtype=complex)
    result = 0.0
    sign = (-1) ** n

    for k in range(2**n):
        if k == 0:
            continue

        # Find the bit that changed (Gray code)
        gray_k = k ^ (k >> 1)
        gray_prev = (k - 1) ^ ((k - 1) >> 1)
        changed_bit = (gray_k ^ gray_prev).bit_length() - 1

        if gray_k & (1 << changed_bit):
            row_sums += matrix[:, changed_bit]
        else:
            row_sums -= matrix[:, changed_bit]

        # Count set bits for sign
        popcount = bin(gray_k).count('1')
        contrib_sign = (-1) ** (n - popcount)
        result += contrib_sign * np.prod(row_sums)

    return sign * result


def approximate_permanent_gurvits(matrix: np.ndarray,
                                   epsilon: float = 0.1,
                                   n_samples: int = 10000) -> Tuple[float, float]:
    """
    Gurvits' polynomial-time approximation for non-negative matrices.
    Returns (estimate, relative_error).
    """
    n = matrix.shape[0]
    samples = []

    for _ in range(n_samples):
        # Random doubly stochastic scaling
        perm = np.random.permutation(n)
        estimate = np.prod([matrix[i, perm[i]] for i in range(n)]) * np.math.factorial(n)
        samples.append(estimate)

    mean_est = np.mean(samples)
    std_est = np.std(samples) / np.sqrt(n_samples)

    return mean_est, std_est / (mean_est + 1e-10)
```

## 3. Gaussian Boson Sampling

<Definition term="Gaussian Boson Sampling (GBS)">
Instead of single photons, use squeezed states as input. Output probabilities are related to the hafnian of a matrix derived from the Gaussian state covariance.
</Definition>

<Formula>
$$P(S) \propto |\text{Haf}(A_S)|^2$$
</Formula>

where Haf denotes the hafnian and A_S is derived from the adjacency matrix of the output modes.

```python
def hafnian_recursive(matrix: np.ndarray) -> complex:
    """
    Compute hafnian recursively.
    Hafnian counts perfect matchings in a graph.
    """
    n = matrix.shape[0]
    if n == 0:
        return 1.0
    if n == 2:
        return matrix[0, 1]
    if n % 2 == 1:
        return 0.0

    # Sum over all ways to pair first element
    result = 0.0
    for j in range(1, n):
        # Remove rows/cols 0 and j
        remaining = [i for i in range(n) if i not in (0, j)]
        submatrix = matrix[np.ix_(remaining, remaining)]
        result += matrix[0, j] * hafnian_recursive(submatrix)

    return result


class GaussianBosonSampling:
    """Gaussian Boson Sampling simulator."""

    def __init__(self, n_modes: int, squeezing: float = 1.0):
        self.n_modes = n_modes
        self.squeezing = squeezing

    def create_adjacency_matrix(self, unitary: np.ndarray) -> np.ndarray:
        """Create the adjacency matrix A from interferometer unitary."""
        # For squeezed vacuum input
        r = self.squeezing
        tanh_r = np.tanh(r)

        # Adjacency matrix: A = U @ diag(tanh(r)) @ U^T
        squeezed_diag = tanh_r * np.eye(self.n_modes)
        A = unitary @ squeezed_diag @ unitary.T

        return A

    def sample_probability(self, adjacency: np.ndarray,
                           output_pattern: List[int]) -> float:
        """
        Compute probability of detecting photons in given pattern.
        output_pattern[i] = number of photons in mode i
        """
        # Build the repeated adjacency matrix
        total_photons = sum(output_pattern)
        if total_photons % 2 == 1:
            return 0.0

        # Construct submatrix with repeated rows/cols
        indices = []
        for mode, count in enumerate(output_pattern):
            indices.extend([mode] * count)

        if len(indices) == 0:
            # Vacuum probability
            det = np.linalg.det(np.eye(self.n_modes) - adjacency @ adjacency.conj())
            return np.sqrt(np.abs(det))

        submatrix = adjacency[np.ix_(indices, indices)]
        haf = hafnian_recursive(submatrix)

        # Normalization
        factorial_prod = np.prod([np.math.factorial(c) for c in output_pattern])

        return np.abs(haf) ** 2 / factorial_prod
```

## 4. Experimental Implementations

<KeyConcept title="Photonic Implementation Challenges">
1. **Photon Loss**: Main source of error; probability decreases exponentially with photon number
2. **Mode Mismatch**: Distinguishability reduces quantum interference
3. **Detection Efficiency**: Imperfect photon counting affects statistics
4. **Photon Source Quality**: Single-photon sources must be indistinguishable
</KeyConcept>

```python
class NoisyBosonSampling:
    """Model noisy boson sampling with experimental imperfections."""

    def __init__(self, n_modes: int, n_photons: int):
        self.m = n_modes
        self.n = n_photons

    def apply_loss(self, unitary: np.ndarray, loss_rate: float) -> np.ndarray:
        """
        Model photon loss as beam splitter with vacuum mode.
        Loss transforms m modes to 2m modes (signal + loss).
        """
        m = unitary.shape[0]
        eta = 1 - loss_rate  # transmission coefficient

        # Extended unitary: each mode couples to a loss mode
        extended = np.zeros((2*m, 2*m), dtype=complex)

        # Signal-signal block (scaled original unitary)
        extended[:m, :m] = np.sqrt(eta) * unitary

        # Loss block (identity with loss coefficient)
        for i in range(m):
            extended[m+i, i] = np.sqrt(loss_rate)
            extended[i, m+i] = np.sqrt(loss_rate)
            extended[m+i, m+i] = np.sqrt(eta)

        return extended

    def partial_distinguishability(self, unitary: np.ndarray,
                                   distinguishability: float) -> dict:
        """
        Model partial photon distinguishability.
        Returns sampling distribution interpolating quantum/classical.
        """
        # At distinguishability = 0: full quantum interference
        # At distinguishability = 1: classical (no interference)

        return {
            'quantum_weight': 1 - distinguishability,
            'classical_weight': distinguishability,
            'effective_unitary': unitary,
            'note': 'Sample from mixture of quantum and classical distributions'
        }

    def expected_fidelity(self, loss_rate: float,
                          n_photons: int,
                          distinguishability: float) -> float:
        """Estimate experimental fidelity with noise."""
        # Loss fidelity: probability all photons survive
        loss_fidelity = (1 - loss_rate) ** n_photons

        # Distinguishability reduces quantum interference
        # Fidelity ~ (1-d)^(n*(n-1)/2) for pairwise interference
        dist_fidelity = (1 - distinguishability) ** (n_photons * (n_photons - 1) / 2)

        return loss_fidelity * dist_fidelity
```

## 5. Validation and Spoofing

<Warning>
Validating boson sampling is itself computationally hard. Various statistical tests can detect certain types of classical spoofing.
</Warning>

```python
from scipy import stats

class BosonSamplingValidator:
    """Tools for validating boson sampling experiments."""

    def __init__(self, n_modes: int, n_photons: int):
        self.m = n_modes
        self.n = n_photons

    def bunching_test(self, samples: List[Tuple[int, ...]],
                      unitary: np.ndarray) -> dict:
        """
        Test photon bunching behavior.
        Bosons tend to bunch compared to distinguishable particles.
        """
        collision_counts = []
        for sample in samples:
            # Count modes with multiple photons
            collisions = sum(1 for count in sample if count > 1)
            collision_counts.append(collisions)

        mean_collisions = np.mean(collision_counts)

        # Expected for distinguishable particles
        classical_expected = self.n * (self.n - 1) / (2 * self.m)

        # Bosons should show MORE bunching
        bunching_ratio = mean_collisions / (classical_expected + 1e-10)

        return {
            'mean_collisions': mean_collisions,
            'classical_expected': classical_expected,
            'bunching_ratio': bunching_ratio,
            'is_bosonic': bunching_ratio > 1.2  # Heuristic threshold
        }

    def row_norm_test(self, samples: List[Tuple[int, ...]],
                      unitary: np.ndarray) -> dict:
        """
        Test correlation with row norms of unitary submatrices.
        Spoofing based on row norms can be detected.
        """
        # Compute row norms for each output mode
        row_norms = np.sum(np.abs(unitary[:self.n, :]) ** 2, axis=1)

        # Count photons per mode in samples
        mode_counts = np.zeros(self.m)
        for sample in samples:
            for mode, count in enumerate(sample):
                mode_counts[mode] += count

        mode_frequencies = mode_counts / (len(samples) * self.n)

        # Correlation between frequencies and row norms
        correlation = np.corrcoef(mode_frequencies[:len(row_norms)],
                                  row_norms)[0, 1]

        return {
            'row_norm_correlation': correlation,
            'is_suspicious': correlation > 0.9,  # Too correlated suggests spoofing
            'mode_frequencies': mode_frequencies
        }

    def likelihood_ratio_test(self, samples: List[Tuple[int, ...]],
                               unitary: np.ndarray,
                               n_reference: int = 1000) -> dict:
        """
        Compare sample likelihood against uniform distribution.
        """
        # This is expensive - simplified version
        sample_set = set(tuple(s) for s in samples)
        unique_fraction = len(sample_set) / len(samples)

        # Boson sampling should have high collision probability
        # due to bunching - fewer unique samples
        expected_unique_uniform = 1 - np.exp(-len(samples) /
                                              np.math.comb(self.m + self.n - 1, self.n))

        return {
            'unique_fraction': unique_fraction,
            'expected_uniform': expected_unique_uniform,
            'collision_evidence': unique_fraction < expected_unique_uniform * 0.9
        }
```

## 6. Comparison with Other Supremacy Proposals

<Tabs>
<Tab label="Boson Sampling">
**Advantages:**
- Natural physical implementation with photons
- Clear mathematical hardness (permanents)
- Room-temperature operation possible

**Challenges:**
- Photon loss scales exponentially
- Limited programmability
- Validation is computationally hard
</Tab>
<Tab label="Random Circuit Sampling">
**Advantages:**
- Programmable gate-based architecture
- XEB provides verifiable benchmark
- Scales with improving qubit quality

**Challenges:**
- Requires error correction for scaling
- Cryogenic operation for superconducting
- Classical tensor network competition
</Tab>
<Tab label="IQP Circuits">
**Advantages:**
- Simpler circuit structure (commuting gates)
- Strong complexity-theoretic evidence

**Challenges:**
- No known experimental advantage
- Similar spoofing concerns
</Tab>
</Tabs>

```python
class SupremacyComparison:
    """Compare different quantum supremacy approaches."""

    def classical_simulation_cost(self, approach: str,
                                   n: int,
                                   depth_or_photons: int) -> dict:
        """Estimate classical simulation cost for different approaches."""

        if approach == 'boson_sampling':
            # Permanent computation: O(n * 2^n)
            exact_cost = depth_or_photons * (2 ** depth_or_photons)
            # Best approximate algorithms
            approx_cost = depth_or_photons ** 3 * np.log(1/0.01)  # For specific matrices

        elif approach == 'gaussian_boson_sampling':
            # Hafnian: O(n^3 * 2^(n/2))
            exact_cost = (depth_or_photons ** 3) * (2 ** (depth_or_photons / 2))
            approx_cost = exact_cost  # No known good approximation

        elif approach == 'random_circuit_sampling':
            # Tensor network: O(2^tw * n * d)
            tw = int(np.sqrt(n) * min(np.sqrt(n), depth_or_photons))
            exact_cost = (2 ** tw) * n * depth_or_photons
            approx_cost = exact_cost * 0.1  # With approximations

        elif approach == 'iqp':
            # Similar to RCS
            exact_cost = 2 ** n
            approx_cost = 2 ** (n * 0.8)  # Slightly easier

        else:
            raise ValueError(f"Unknown approach: {approach}")

        return {
            'approach': approach,
            'exact_cost_flops': exact_cost,
            'approx_cost_flops': approx_cost,
            'log10_exact': np.log10(exact_cost + 1),
            'log10_approx': np.log10(approx_cost + 1)
        }

    def experimental_requirements(self, approach: str,
                                   target_advantage: str = 'clear') -> dict:
        """Estimate experimental requirements for quantum advantage."""

        requirements = {
            'boson_sampling': {
                'photons_needed': 50 if target_advantage == 'clear' else 30,
                'loss_tolerance': 0.05,
                'distinguishability_tolerance': 0.01,
                'modes_needed': 100,
                'technology': 'Integrated photonics + SNSPD detectors'
            },
            'gaussian_boson_sampling': {
                'photons_needed': 100 if target_advantage == 'clear' else 50,
                'squeezing_db': 12,
                'loss_tolerance': 0.10,
                'modes_needed': 144,
                'technology': 'Squeezed light + Time-bin multiplexing'
            },
            'random_circuit_sampling': {
                'qubits_needed': 70 if target_advantage == 'clear' else 53,
                'two_qubit_fidelity': 0.995,
                'depth': 20,
                'technology': 'Superconducting transmons or trapped ions'
            }
        }

        return requirements.get(approach, {'error': 'Unknown approach'})
```
