## 1. What is a Quantum Neural Network?

A Quantum Neural Network (QNN) is a parameterized quantum circuit that can be trained to perform machine learning tasks. Unlike classical neural networks with discrete layers of neurons, QNNs operate on quantum states through unitary transformations.

<Definition term="Quantum Neural Network">
A QNN is a parameterized unitary $U(\theta)$ that transforms input quantum states to output states, where $\theta$ are trainable parameters. The output is extracted through measurements.
</Definition>

<Formula>
$$|\psi_{out}\rangle = U(\theta)|\psi_{in}\rangle = \prod_{l=1}^{L} U_l(\theta_l)|\psi_{in}\rangle$$
</Formula>

```python
import numpy as np
from typing import List, Tuple, Callable, Optional
from dataclasses import dataclass
from enum import Enum

class GateType(Enum):
    """Types of parameterized gates."""
    RX = "rx"
    RY = "ry"
    RZ = "rz"
    CRX = "crx"
    CRY = "cry"
    CRZ = "crz"
    CNOT = "cnot"
    CZ = "cz"

@dataclass
class QNNLayer:
    """Definition of a QNN layer."""
    rotation_gates: List[Tuple[GateType, int]]  # (gate_type, qubit)
    entangling_gates: List[Tuple[GateType, int, int]]  # (gate_type, control, target)

class QuantumNeuralNetwork:
    """
    Flexible quantum neural network implementation.
    """

    def __init__(self, num_qubits: int, layers: List[QNNLayer]):
        self.num_qubits = num_qubits
        self.layers = layers
        self.dim = 2 ** num_qubits
        self.num_params = self._count_parameters()

    def _count_parameters(self) -> int:
        """Count total trainable parameters."""
        count = 0
        for layer in self.layers:
            # Each rotation gate has 1 parameter
            count += len(layer.rotation_gates)
            # Controlled rotations also have parameters
            for gate, _, _ in layer.entangling_gates:
                if gate in [GateType.CRX, GateType.CRY, GateType.CRZ]:
                    count += 1
        return count

    def _rx(self, theta: float) -> np.ndarray:
        return np.array([
            [np.cos(theta/2), -1j * np.sin(theta/2)],
            [-1j * np.sin(theta/2), np.cos(theta/2)]
        ])

    def _ry(self, theta: float) -> np.ndarray:
        return np.array([
            [np.cos(theta/2), -np.sin(theta/2)],
            [np.sin(theta/2), np.cos(theta/2)]
        ])

    def _rz(self, theta: float) -> np.ndarray:
        return np.array([
            [np.exp(-1j * theta/2), 0],
            [0, np.exp(1j * theta/2)]
        ])

    def _tensor_product(self, ops: List[np.ndarray]) -> np.ndarray:
        result = ops[0]
        for op in ops[1:]:
            result = np.kron(result, op)
        return result

    def _apply_single_gate(
        self,
        state: np.ndarray,
        gate: np.ndarray,
        qubit: int
    ) -> np.ndarray:
        ops = [np.eye(2)] * self.num_qubits
        ops[qubit] = gate
        return self._tensor_product(ops) @ state

    def _apply_cnot(self, state: np.ndarray, ctrl: int, tgt: int) -> np.ndarray:
        new_state = state.copy()
        n = self.num_qubits
        for i in range(self.dim):
            if (i >> (n - 1 - ctrl)) & 1:
                j = i ^ (1 << (n - 1 - tgt))
                new_state[i], new_state[j] = state[j], state[i]
        return new_state

    def _apply_cz(self, state: np.ndarray, ctrl: int, tgt: int) -> np.ndarray:
        new_state = state.copy()
        n = self.num_qubits
        for i in range(self.dim):
            if ((i >> (n - 1 - ctrl)) & 1) and ((i >> (n - 1 - tgt)) & 1):
                new_state[i] = -state[i]
        return new_state

    def _apply_controlled_rotation(
        self,
        state: np.ndarray,
        gate_type: GateType,
        ctrl: int,
        tgt: int,
        theta: float
    ) -> np.ndarray:
        """Apply controlled rotation gate."""
        if gate_type == GateType.CRX:
            gate = self._rx(theta)
        elif gate_type == GateType.CRY:
            gate = self._ry(theta)
        else:  # CRZ
            gate = self._rz(theta)

        new_state = state.copy()
        n = self.num_qubits

        for i in range(self.dim):
            if (i >> (n - 1 - ctrl)) & 1:  # Control is |1>
                # Apply rotation to target
                tgt_bit = (i >> (n - 1 - tgt)) & 1
                j = i ^ (1 << (n - 1 - tgt))  # Flipped target state

                # Get amplitudes
                if tgt_bit == 0:
                    amp0, amp1 = state[i], state[j]
                else:
                    amp0, amp1 = state[j], state[i]

                # Apply gate
                new_amp0 = gate[0, 0] * amp0 + gate[0, 1] * amp1
                new_amp1 = gate[1, 0] * amp0 + gate[1, 1] * amp1

                if tgt_bit == 0:
                    new_state[i] = new_amp0
                    new_state[j] = new_amp1
                else:
                    new_state[j] = new_amp0
                    new_state[i] = new_amp1

        return new_state

    def forward(self, state: np.ndarray, params: np.ndarray) -> np.ndarray:
        """Apply QNN to input state."""
        param_idx = 0

        for layer in self.layers:
            # Apply rotation gates
            for gate_type, qubit in layer.rotation_gates:
                theta = params[param_idx]
                param_idx += 1

                if gate_type == GateType.RX:
                    gate = self._rx(theta)
                elif gate_type == GateType.RY:
                    gate = self._ry(theta)
                else:
                    gate = self._rz(theta)

                state = self._apply_single_gate(state, gate, qubit)

            # Apply entangling gates
            for gate_type, ctrl, tgt in layer.entangling_gates:
                if gate_type == GateType.CNOT:
                    state = self._apply_cnot(state, ctrl, tgt)
                elif gate_type == GateType.CZ:
                    state = self._apply_cz(state, ctrl, tgt)
                elif gate_type in [GateType.CRX, GateType.CRY, GateType.CRZ]:
                    theta = params[param_idx]
                    param_idx += 1
                    state = self._apply_controlled_rotation(
                        state, gate_type, ctrl, tgt, theta
                    )

        return state

    def measure_expectation(
        self,
        state: np.ndarray,
        observable: np.ndarray
    ) -> float:
        """Compute expectation value of observable."""
        return np.real(np.vdot(state, observable @ state))
```

## 2. Standard QNN Architectures

### Hardware-Efficient Ansatz

The most common architecture, designed for NISQ devices:

```python
def hardware_efficient_ansatz(
    num_qubits: int,
    num_layers: int,
    rotation_gates: List[GateType] = [GateType.RY, GateType.RZ],
    entanglement: str = "linear"
) -> List[QNNLayer]:
    """
    Create hardware-efficient ansatz layers.

    Args:
        num_qubits: Number of qubits
        num_layers: Number of variational layers
        rotation_gates: Gates to use for rotations
        entanglement: 'linear', 'circular', or 'full'
    """
    layers = []

    for _ in range(num_layers):
        # Rotation layer
        rotations = []
        for q in range(num_qubits):
            for gate_type in rotation_gates:
                rotations.append((gate_type, q))

        # Entangling layer
        entangling = []
        if entanglement == "linear":
            for q in range(num_qubits - 1):
                entangling.append((GateType.CNOT, q, q + 1))
        elif entanglement == "circular":
            for q in range(num_qubits - 1):
                entangling.append((GateType.CNOT, q, q + 1))
            if num_qubits > 2:
                entangling.append((GateType.CNOT, num_qubits - 1, 0))
        elif entanglement == "full":
            for q1 in range(num_qubits):
                for q2 in range(q1 + 1, num_qubits):
                    entangling.append((GateType.CNOT, q1, q2))

        layers.append(QNNLayer(rotations, entangling))

    # Final rotation layer (no entanglement)
    final_rotations = []
    for q in range(num_qubits):
        for gate_type in rotation_gates:
            final_rotations.append((gate_type, q))
    layers.append(QNNLayer(final_rotations, []))

    return layers

# Example usage
layers = hardware_efficient_ansatz(4, 3, entanglement="circular")
qnn = QuantumNeuralNetwork(4, layers)
print(f"QNN with {qnn.num_params} parameters")
```

### Strongly Entangling Layers

A more expressive architecture with all-to-all connectivity per layer:

```python
def strongly_entangling_layers(
    num_qubits: int,
    num_layers: int
) -> List[QNNLayer]:
    """
    Create strongly entangling layers.

    Each layer has:
    - Full rotation (Rx, Ry, Rz) on each qubit
    - CNOT connections that shift each layer
    """
    layers = []

    for layer_idx in range(num_layers):
        # Full rotations on each qubit
        rotations = []
        for q in range(num_qubits):
            rotations.extend([
                (GateType.RX, q),
                (GateType.RY, q),
                (GateType.RZ, q)
            ])

        # Shifted entanglement pattern
        entangling = []
        for q in range(num_qubits):
            target = (q + 1 + layer_idx) % num_qubits
            if target != q:
                entangling.append((GateType.CNOT, q, target))

        layers.append(QNNLayer(rotations, entangling))

    return layers
```

### Tree Tensor Network Ansatz

Inspired by tensor network theory for efficient computation:

```python
def tree_tensor_ansatz(
    num_qubits: int
) -> List[QNNLayer]:
    """
    Create tree tensor network ansatz.

    Hierarchical structure that reduces to single qubit output.
    Requires num_qubits to be a power of 2.
    """
    assert num_qubits & (num_qubits - 1) == 0, "num_qubits must be power of 2"

    layers = []
    active_qubits = list(range(num_qubits))

    while len(active_qubits) > 1:
        rotations = []
        entangling = []

        # Pair adjacent qubits
        new_active = []
        for i in range(0, len(active_qubits), 2):
            q1, q2 = active_qubits[i], active_qubits[i + 1]

            # Rotations before entangling
            rotations.extend([
                (GateType.RY, q1),
                (GateType.RZ, q1),
                (GateType.RY, q2),
                (GateType.RZ, q2)
            ])

            # Entangle pair
            entangling.append((GateType.CNOT, q1, q2))

            # q1 becomes the "surviving" qubit for next level
            new_active.append(q1)

        layers.append(QNNLayer(rotations, entangling))
        active_qubits = new_active

    # Final rotation on output qubit
    layers.append(QNNLayer([(GateType.RY, active_qubits[0])], []))

    return layers
```

## 3. Expressibility Analysis

Expressibility measures how well a parameterized circuit can explore the Hilbert space.

<Definition term="Expressibility">
The expressibility of a circuit is quantified by how closely the distribution of states it generates matches the Haar-random distribution. Lower expressibility distance means higher expressibility.
</Definition>

<Formula>
$$\text{Expr} = D_{KL}(\hat{P}_{PQC} \| P_{Haar})$$
</Formula>

```python
def compute_expressibility(
    qnn: QuantumNeuralNetwork,
    num_samples: int = 1000,
    num_bins: int = 75
) -> dict:
    """
    Estimate expressibility using fidelity distribution.

    Computes distribution of |<psi(theta)|psi(phi)>|^2 for random
    parameter pairs and compares to Haar-random distribution.
    """
    dim = qnn.dim
    fidelities = []

    # Sample random parameter pairs
    for _ in range(num_samples):
        # Two random parameter sets
        theta = np.random.uniform(0, 2 * np.pi, qnn.num_params)
        phi = np.random.uniform(0, 2 * np.pi, qnn.num_params)

        # Initial state |0...0>
        initial = np.zeros(dim, dtype=complex)
        initial[0] = 1.0

        # Generate states
        state1 = qnn.forward(initial.copy(), theta)
        state2 = qnn.forward(initial.copy(), phi)

        # Compute fidelity
        fidelity = np.abs(np.vdot(state1, state2)) ** 2
        fidelities.append(fidelity)

    fidelities = np.array(fidelities)

    # Haar-random fidelity distribution: P(F) = (d-1)(1-F)^(d-2)
    # For comparison, compute histogram
    hist_pqc, bin_edges = np.histogram(fidelities, bins=num_bins, range=(0, 1), density=True)

    # Theoretical Haar distribution
    bin_centers = (bin_edges[:-1] + bin_edges[1:]) / 2
    haar_dist = (dim - 1) * (1 - bin_centers) ** (dim - 2)
    haar_dist = haar_dist / np.sum(haar_dist) * num_bins  # Normalize

    # KL divergence (expressibility measure)
    # Add small epsilon to avoid log(0)
    eps = 1e-10
    hist_pqc_norm = hist_pqc / (np.sum(hist_pqc) + eps) + eps
    haar_norm = haar_dist / (np.sum(haar_dist) + eps) + eps

    kl_divergence = np.sum(hist_pqc_norm * np.log(hist_pqc_norm / haar_norm))

    return {
        'expressibility': kl_divergence,
        'mean_fidelity': np.mean(fidelities),
        'fidelity_std': np.std(fidelities),
        'fidelity_distribution': fidelities,
        'histogram': hist_pqc,
        'bin_centers': bin_centers
    }

def compare_architectures_expressibility(
    num_qubits: int,
    num_layers: int
) -> dict:
    """Compare expressibility of different architectures."""
    results = {}

    # Hardware-efficient with different entanglements
    for ent in ["linear", "circular", "full"]:
        layers = hardware_efficient_ansatz(num_qubits, num_layers, entanglement=ent)
        qnn = QuantumNeuralNetwork(num_qubits, layers)
        expr = compute_expressibility(qnn, num_samples=500)
        results[f"HEA_{ent}"] = {
            'expressibility': expr['expressibility'],
            'num_params': qnn.num_params
        }

    # Strongly entangling
    layers = strongly_entangling_layers(num_qubits, num_layers)
    qnn = QuantumNeuralNetwork(num_qubits, layers)
    expr = compute_expressibility(qnn, num_samples=500)
    results["StronglyEntangling"] = {
        'expressibility': expr['expressibility'],
        'num_params': qnn.num_params
    }

    return results
```

## 4. Entangling Capability

Beyond expressibility, we need circuits that can generate entanglement:

```python
def compute_entangling_capability(
    qnn: QuantumNeuralNetwork,
    num_samples: int = 500
) -> dict:
    """
    Measure entangling capability using Meyer-Wallach entanglement.

    The Meyer-Wallach measure Q ranges from 0 (product states) to 1 (maximally entangled).
    """
    n = qnn.num_qubits
    dim = qnn.dim

    entanglement_values = []

    for _ in range(num_samples):
        # Random parameters
        params = np.random.uniform(0, 2 * np.pi, qnn.num_params)

        # Generate state
        initial = np.zeros(dim, dtype=complex)
        initial[0] = 1.0
        state = qnn.forward(initial, params)

        # Compute Meyer-Wallach entanglement
        Q = meyer_wallach_measure(state, n)
        entanglement_values.append(Q)

    entanglement_values = np.array(entanglement_values)

    return {
        'mean_entanglement': np.mean(entanglement_values),
        'std_entanglement': np.std(entanglement_values),
        'max_entanglement': np.max(entanglement_values),
        'entanglement_distribution': entanglement_values
    }

def meyer_wallach_measure(state: np.ndarray, num_qubits: int) -> float:
    """
    Compute Meyer-Wallach entanglement measure.

    Q = (2/n) * sum_k (1 - Tr(rho_k^2))

    where rho_k is the reduced density matrix of qubit k.
    """
    n = num_qubits
    dim = 2 ** n

    Q = 0.0

    for k in range(n):
        # Compute reduced density matrix for qubit k
        rho_k = partial_trace_single_qubit(state, k, n)

        # Compute purity
        purity = np.real(np.trace(rho_k @ rho_k))

        # Add linear entropy contribution
        Q += 1 - purity

    Q = (2 / n) * Q
    return Q

def partial_trace_single_qubit(
    state: np.ndarray,
    keep_qubit: int,
    num_qubits: int
) -> np.ndarray:
    """
    Compute reduced density matrix for a single qubit.
    """
    n = num_qubits
    dim = 2 ** n

    # Reshape state into tensor
    # state[i] where i = i_{n-1} ... i_1 i_0
    state_tensor = state.reshape([2] * n)

    # Density matrix
    rho = np.outer(state, state.conj())
    rho_tensor = rho.reshape([2] * (2 * n))

    # Trace out all qubits except keep_qubit
    # Contract over pairs of indices
    trace_qubits = [q for q in range(n) if q != keep_qubit]

    # Simplified: compute directly from state
    rho_k = np.zeros((2, 2), dtype=complex)

    for i in range(dim):
        for j in range(dim):
            # Extract bit values for keep_qubit
            bit_i = (i >> (n - 1 - keep_qubit)) & 1
            bit_j = (j >> (n - 1 - keep_qubit)) & 1

            # Check if other qubits match
            mask = ~(1 << (n - 1 - keep_qubit)) & (dim - 1)
            if (i & mask) == (j & mask):
                rho_k[bit_i, bit_j] += state[i] * state[j].conj()

    return rho_k
```

## 5. Relationship to Classical Neural Networks

QNNs and classical NNs have interesting connections:

```python
def qnn_to_classical_features(
    qnn: QuantumNeuralNetwork,
    input_state: np.ndarray,
    params: np.ndarray,
    observables: List[np.ndarray]
) -> np.ndarray:
    """
    Extract classical features from QNN output.

    This shows how QNN can be viewed as a feature extractor.
    """
    output_state = qnn.forward(input_state, params)

    features = []
    for obs in observables:
        exp_val = np.real(np.vdot(output_state, obs @ output_state))
        features.append(exp_val)

    return np.array(features)

def classical_nn_analogy(num_qubits: int, depth: int) -> dict:
    """
    Analyze QNN in terms of classical NN concepts.
    """
    # A QNN layer is analogous to a classical layer but:
    # - Weights are angles (bounded)
    # - Activations are unitary operations
    # - "Hidden layer width" is exponential (2^n)

    # Effective parameters
    qnn_params = 2 * num_qubits * depth  # Typical HEA

    # Equivalent classical network (very rough)
    # 2^n-dim space, but constrained to unit sphere
    classical_equiv_params = num_qubits * depth * 2  # Similar scaling

    # Effective dimension analysis
    # QNN operates on 2^n complex amplitudes (2^{n+1} real)
    # But normalized: 2^{n+1} - 1 degrees of freedom
    effective_dim = 2 ** (num_qubits + 1) - 1

    return {
        'num_qnn_params': qnn_params,
        'effective_hilbert_dim': effective_dim,
        'param_efficiency': effective_dim / qnn_params,
        'entanglement_advantage': "QNN can create correlations classical NNs cannot"
    }

class HybridQNN:
    """
    Hybrid quantum-classical neural network.

    Classical preprocessing -> Quantum circuit -> Classical postprocessing
    """

    def __init__(
        self,
        input_dim: int,
        num_qubits: int,
        qnn_layers: int,
        output_dim: int
    ):
        self.input_dim = input_dim
        self.num_qubits = num_qubits
        self.output_dim = output_dim

        # Classical preprocessing weights
        self.W_pre = np.random.randn(num_qubits, input_dim) * 0.1
        self.b_pre = np.zeros(num_qubits)

        # QNN
        layers = hardware_efficient_ansatz(num_qubits, qnn_layers)
        self.qnn = QuantumNeuralNetwork(num_qubits, layers)

        # Classical postprocessing
        self.W_post = np.random.randn(output_dim, num_qubits) * 0.1
        self.b_post = np.zeros(output_dim)

    def forward(
        self,
        x: np.ndarray,
        qnn_params: np.ndarray
    ) -> np.ndarray:
        """
        Forward pass through hybrid network.
        """
        # Classical preprocessing (encode to angles)
        encoding_angles = np.tanh(self.W_pre @ x + self.b_pre) * np.pi

        # Prepare quantum state with encoding
        state = np.zeros(self.qnn.dim, dtype=complex)
        state[0] = 1.0

        # Apply encoding rotations
        for q in range(self.num_qubits):
            ry = np.array([
                [np.cos(encoding_angles[q]/2), -np.sin(encoding_angles[q]/2)],
                [np.sin(encoding_angles[q]/2), np.cos(encoding_angles[q]/2)]
            ])
            ops = [np.eye(2)] * self.num_qubits
            ops[q] = ry
            gate = ops[0]
            for op in ops[1:]:
                gate = np.kron(gate, op)
            state = gate @ state

        # Apply QNN
        state = self.qnn.forward(state, qnn_params)

        # Extract features (expectation values of Z on each qubit)
        features = []
        Z = np.array([[1, 0], [0, -1]])
        for q in range(self.num_qubits):
            ops = [np.eye(2)] * self.num_qubits
            ops[q] = Z
            obs = ops[0]
            for op in ops[1:]:
                obs = np.kron(obs, op)
            exp_val = np.real(np.vdot(state, obs @ state))
            features.append(exp_val)

        features = np.array(features)

        # Classical postprocessing
        output = self.W_post @ features + self.b_post

        return output
```

## 6. Designing Custom Architectures

Guidelines for architecture design:

```python
def design_qnn_for_task(
    task: str,
    num_qubits: int,
    constraints: dict = None
) -> List[QNNLayer]:
    """
    Design QNN architecture based on task requirements.

    Args:
        task: 'classification', 'regression', 'generative'
        num_qubits: Available qubits
        constraints: {'max_depth': int, 'max_params': int, 'connectivity': str}
    """
    constraints = constraints or {}
    max_depth = constraints.get('max_depth', 10)
    connectivity = constraints.get('connectivity', 'linear')

    if task == 'classification':
        # Moderate depth, focus on expressibility
        num_layers = min(max_depth, num_qubits)
        return hardware_efficient_ansatz(
            num_qubits, num_layers,
            rotation_gates=[GateType.RY, GateType.RZ],
            entanglement=connectivity
        )

    elif task == 'regression':
        # Deeper circuits for continuous output
        num_layers = min(max_depth, 2 * num_qubits)
        return strongly_entangling_layers(num_qubits, num_layers)

    elif task == 'generative':
        # High entanglement for correlations
        layers = []
        for l in range(min(max_depth, num_qubits)):
            rotations = []
            for q in range(num_qubits):
                rotations.extend([
                    (GateType.RX, q),
                    (GateType.RY, q),
                    (GateType.RZ, q)
                ])

            # Dense entanglement
            entangling = []
            for q1 in range(num_qubits):
                for q2 in range(q1 + 1, num_qubits):
                    entangling.append((GateType.CZ, q1, q2))

            layers.append(QNNLayer(rotations, entangling))

        return layers

    else:
        raise ValueError(f"Unknown task: {task}")

def analyze_architecture(qnn: QuantumNeuralNetwork) -> dict:
    """
    Comprehensive architecture analysis.
    """
    # Parameter analysis
    total_params = qnn.num_params

    # Gate counts
    single_qubit_gates = 0
    two_qubit_gates = 0
    for layer in qnn.layers:
        single_qubit_gates += len(layer.rotation_gates)
        two_qubit_gates += len(layer.entangling_gates)

    # Depth estimation (simplified)
    depth = len(qnn.layers) * 2  # Rough estimate

    # Expressibility
    expr = compute_expressibility(qnn, num_samples=300)

    # Entangling capability
    ent = compute_entangling_capability(qnn, num_samples=300)

    return {
        'total_parameters': total_params,
        'single_qubit_gates': single_qubit_gates,
        'two_qubit_gates': two_qubit_gates,
        'estimated_depth': depth,
        'expressibility': expr['expressibility'],
        'mean_entanglement': ent['mean_entanglement'],
        'gate_ratio': two_qubit_gates / (single_qubit_gates + 1e-10)
    }
```
