## 1. PAC Learning Framework

Probably Approximately Correct (PAC) learning provides the theoretical foundation for understanding when learning is possible.

<Definition term="PAC Learning">
A concept class C is PAC learnable if there exists an algorithm that, for any distribution D and any epsilon, delta > 0, outputs a hypothesis h with P[error(h) > epsilon] < delta using poly(1/epsilon, 1/delta, n) samples.
</Definition>

<Formula>
$$P_{(x,y)\sim D}[h(x) \neq y] \leq \epsilon \text{ with probability } \geq 1 - \delta$$
</Formula>

```python
import numpy as np
from typing import Callable, List, Tuple, Optional
from dataclasses import dataclass

@dataclass
class PACBounds:
    """PAC learning bounds for a hypothesis class."""
    sample_complexity: int
    generalization_gap: float
    confidence: float
    epsilon: float
    delta: float

def compute_pac_sample_complexity(
    hypothesis_class_size: int,
    epsilon: float,
    delta: float
) -> int:
    """
    Compute sample complexity for finite hypothesis class.

    For finite |H|, sample complexity is O(log(|H|/delta) / epsilon^2)
    """
    import math
    return int(math.ceil(
        (math.log(hypothesis_class_size) + math.log(1/delta)) / (epsilon**2)
    ))

def compute_vc_sample_complexity(
    vc_dimension: int,
    epsilon: float,
    delta: float
) -> int:
    """
    Compute sample complexity using VC dimension.

    Sample complexity: O((d + log(1/delta)) / epsilon^2)
    """
    import math
    return int(math.ceil(
        (vc_dimension + math.log(1/delta)) / (epsilon**2)
    ))

class QuantumPACLearner:
    """
    PAC learning framework for quantum classifiers.
    """

    def __init__(
        self,
        model: Callable,
        num_params: int
    ):
        self.model = model
        self.num_params = num_params

    def empirical_risk(
        self,
        params: np.ndarray,
        X: np.ndarray,
        y: np.ndarray
    ) -> float:
        """Compute training error."""
        predictions = np.array([
            1 if self.model(x, params) > 0.5 else 0
            for x in X
        ])
        return np.mean(predictions != y)

    def estimate_generalization_gap(
        self,
        params: np.ndarray,
        X_train: np.ndarray,
        y_train: np.ndarray,
        X_test: np.ndarray,
        y_test: np.ndarray
    ) -> dict:
        """
        Estimate gap between training and test error.
        """
        train_error = self.empirical_risk(params, X_train, y_train)
        test_error = self.empirical_risk(params, X_test, y_test)

        return {
            'train_error': train_error,
            'test_error': test_error,
            'generalization_gap': test_error - train_error
        }

    def pac_bound(
        self,
        n_samples: int,
        delta: float = 0.05
    ) -> float:
        """
        Compute PAC generalization bound.

        Using parameter counting: effective VC dim ~ num_params
        """
        import math

        # Simplified bound based on parameter count
        vc_dim = self.num_params

        # Generalization bound
        bound = math.sqrt((vc_dim * math.log(n_samples) + math.log(1/delta)) / n_samples)

        return bound
```

## 2. Rademacher Complexity for Quantum Models

Rademacher complexity measures the richness of a function class by its ability to fit random labels.

<Definition term="Rademacher Complexity">
The empirical Rademacher complexity of function class F on sample S is:
$$\hat{R}_S(F) = \mathbb{E}_\sigma\left[\sup_{f \in F} \frac{1}{n}\sum_{i=1}^n \sigma_i f(x_i)\right]$$
where sigma_i are uniform random +/-1 variables.
</Definition>

```python
def estimate_rademacher_complexity(
    model_class: Callable,
    X: np.ndarray,
    num_samples: int = 100,
    param_range: Tuple[float, float] = (0, 2*np.pi)
) -> float:
    """
    Estimate Rademacher complexity of a quantum model class.

    Args:
        model_class: Function (x, params) -> prediction in [-1, 1]
        X: Data samples
        num_samples: Number of Rademacher samples
        param_range: Range for parameter optimization
    """
    n = len(X)
    suprema = []

    for _ in range(num_samples):
        # Random Rademacher variables
        sigma = 2 * np.random.randint(0, 2, n) - 1

        # Find supremum over hypothesis class
        # In practice, we sample parameters
        best_correlation = -np.inf

        for _ in range(50):  # Sample parameters
            params = np.random.uniform(param_range[0], param_range[1], 10)

            try:
                predictions = np.array([model_class(x, params) for x in X])
                correlation = np.mean(sigma * predictions)
                best_correlation = max(best_correlation, correlation)
            except:
                continue

        suprema.append(best_correlation)

    return np.mean(suprema)

def rademacher_generalization_bound(
    rademacher_complexity: float,
    n_samples: int,
    delta: float = 0.05
) -> float:
    """
    Compute generalization bound from Rademacher complexity.

    P[R(h) - R_emp(h) <= 2*Rad + sqrt(log(1/delta)/(2n))] >= 1 - delta
    """
    import math
    return 2 * rademacher_complexity + math.sqrt(math.log(1/delta) / (2*n_samples))

class QuantumRademacherAnalysis:
    """
    Rademacher complexity analysis for quantum circuits.
    """

    def __init__(self, circuit_fn: Callable, num_qubits: int, num_params: int):
        self.circuit = circuit_fn
        self.num_qubits = num_qubits
        self.num_params = num_params

    def compute_complexity(
        self,
        X: np.ndarray,
        num_rademacher_samples: int = 100,
        num_param_samples: int = 50
    ) -> dict:
        """
        Estimate Rademacher complexity with detailed analysis.
        """
        n = len(X)
        correlations = []

        for _ in range(num_rademacher_samples):
            sigma = 2 * np.random.randint(0, 2, n) - 1

            max_corr = -np.inf
            for _ in range(num_param_samples):
                params = np.random.uniform(0, 2*np.pi, self.num_params)

                try:
                    preds = np.array([
                        2 * self.circuit(x, params) - 1  # Map [0,1] to [-1,1]
                        for x in X
                    ])
                    corr = np.mean(sigma * preds)
                    max_corr = max(max_corr, corr)
                except:
                    continue

            correlations.append(max_corr)

        rad_complexity = np.mean(correlations)

        # Theoretical bound based on circuit properties
        # For quantum circuits with L layers and n qubits:
        # Rad <= O(sqrt(L * n / m)) where m is sample size
        theoretical_bound = np.sqrt(self.num_params / n)

        return {
            'empirical_rademacher': rad_complexity,
            'theoretical_bound': theoretical_bound,
            'std_rademacher': np.std(correlations),
            'generalization_bound': rademacher_generalization_bound(rad_complexity, n)
        }
```

## 3. Sample Complexity of Quantum Models

Understanding how many samples are needed for successful quantum learning:

<Formula>
$$m \geq O\left(\frac{d_{VC} + \log(1/\delta)}{\epsilon^2}\right)$$
</Formula>

```python
def quantum_sample_complexity(
    num_qubits: int,
    num_layers: int,
    epsilon: float,
    delta: float
) -> dict:
    """
    Estimate sample complexity for quantum classifier.

    Args:
        num_qubits: Number of qubits
        num_layers: Circuit depth
        epsilon: Target error
        delta: Failure probability
    """
    import math

    # Effective number of parameters
    num_params = 2 * num_qubits * num_layers

    # VC dimension upper bound (roughly proportional to params)
    vc_dim_upper = 2 * num_params

    # Sample complexity bounds
    pac_bound = math.ceil(
        (vc_dim_upper * math.log(num_params) + math.log(1/delta)) / (epsilon**2)
    )

    # Quantum-specific bound (from literature)
    # Takes into account measurement noise
    quantum_bound = math.ceil(
        num_params * (math.log(num_params) + math.log(1/delta)) / (epsilon**2)
    )

    # With shot noise (need more samples per point)
    shots_per_sample = math.ceil(1 / (epsilon**2))
    total_measurements = quantum_bound * shots_per_sample

    return {
        'num_params': num_params,
        'vc_dimension_upper': vc_dim_upper,
        'sample_complexity_pac': pac_bound,
        'sample_complexity_quantum': quantum_bound,
        'shots_per_sample': shots_per_sample,
        'total_measurements': total_measurements
    }

def compare_quantum_classical_complexity(
    input_dimension: int,
    num_qubits: int,
    circuit_depth: int,
    classical_hidden_units: int,
    epsilon: float = 0.1,
    delta: float = 0.05
) -> dict:
    """
    Compare sample complexity of quantum vs classical models.
    """
    import math

    # Quantum model
    quantum_params = 2 * num_qubits * circuit_depth
    quantum_vc = 2 * quantum_params
    quantum_samples = math.ceil(
        (quantum_vc + math.log(1/delta)) / (epsilon**2)
    )

    # Classical neural network (one hidden layer)
    classical_params = (input_dimension + 1) * classical_hidden_units + (classical_hidden_units + 1)
    classical_vc = classical_params  # Rough approximation
    classical_samples = math.ceil(
        (classical_vc + math.log(1/delta)) / (epsilon**2)
    )

    # Expressiveness comparison
    # Quantum: 2^n dimensional Hilbert space
    # Classical: classical_hidden_units dimensional
    quantum_expressiveness = 2 ** num_qubits
    classical_expressiveness = classical_hidden_units

    return {
        'quantum': {
            'params': quantum_params,
            'vc_dimension': quantum_vc,
            'sample_complexity': quantum_samples,
            'expressiveness': quantum_expressiveness
        },
        'classical': {
            'params': classical_params,
            'vc_dimension': classical_vc,
            'sample_complexity': classical_samples,
            'expressiveness': classical_expressiveness
        },
        'quantum_advantage_ratio': classical_samples / quantum_samples,
        'expressiveness_ratio': quantum_expressiveness / classical_expressiveness
    }
```

## 4. Covering Numbers and Metric Entropy

```python
def estimate_covering_number(
    model_class: Callable,
    X: np.ndarray,
    epsilon: float,
    num_param_samples: int = 1000
) -> int:
    """
    Estimate epsilon-covering number of model class.

    Covering number N(epsilon, F, d) is minimum number of balls
    of radius epsilon needed to cover F.
    """
    # Sample predictions for many parameter values
    predictions_list = []

    for _ in range(num_param_samples):
        params = np.random.uniform(0, 2*np.pi, 10)
        try:
            preds = np.array([model_class(x, params) for x in X])
            predictions_list.append(preds)
        except:
            continue

    predictions_array = np.array(predictions_list)

    # Greedy covering
    covered = np.zeros(len(predictions_list), dtype=bool)
    centers = []

    for i in range(len(predictions_list)):
        if covered[i]:
            continue

        centers.append(i)

        # Mark all points within epsilon as covered
        for j in range(len(predictions_list)):
            dist = np.linalg.norm(predictions_array[i] - predictions_array[j])
            if dist <= epsilon:
                covered[j] = True

    return len(centers)

def metric_entropy_bound(
    covering_number: int,
    n_samples: int,
    delta: float = 0.05
) -> float:
    """
    Generalization bound from covering numbers.

    R(h) <= R_emp(h) + 2*sqrt(2*log(N)/n) + sqrt(log(1/delta)/(2n))
    """
    import math
    return 2 * math.sqrt(2 * math.log(covering_number) / n_samples) + \
           math.sqrt(math.log(1/delta) / (2*n_samples))
```

## 5. VC Dimension of Quantum Circuits

<Definition term="VC Dimension">
The VC dimension of hypothesis class H is the largest set S that can be shattered by H, meaning H can realize all 2^|S| labelings of S.
</Definition>

```python
def estimate_vc_dimension(
    model: Callable,
    num_params: int,
    max_check: int = 20,
    num_trials: int = 100
) -> int:
    """
    Estimate VC dimension by checking shattering.

    Args:
        model: Binary classifier (x, params) -> {0, 1}
        num_params: Number of model parameters
        max_check: Maximum dimension to check
        num_trials: Trials per dimension
    """
    def can_shatter_set(X: np.ndarray) -> bool:
        """Check if all labelings of X can be realized."""
        n = len(X)
        num_labelings = 2 ** n

        realized_labelings = set()

        # Sample many parameter configurations
        for _ in range(num_trials * 10):
            params = np.random.uniform(0, 2*np.pi, num_params)

            try:
                predictions = tuple(
                    1 if model(x, params) > 0.5 else 0
                    for x in X
                )
                realized_labelings.add(predictions)
            except:
                continue

            if len(realized_labelings) == num_labelings:
                return True

        return len(realized_labelings) == num_labelings

    estimated_vc = 0

    for d in range(1, max_check + 1):
        can_shatter = False

        for _ in range(num_trials):
            # Random point set of size d
            X = np.random.randn(d, 2)  # 2D inputs

            if can_shatter_set(X):
                can_shatter = True
                break

        if can_shatter:
            estimated_vc = d
        else:
            break

    return estimated_vc

def vc_dimension_bounds_quantum(
    num_qubits: int,
    num_layers: int
) -> dict:
    """
    Theoretical VC dimension bounds for quantum circuits.
    """
    num_params = 2 * num_qubits * num_layers

    # Lower bound: at least linear in params
    lower_bound = num_params

    # Upper bound: can't exceed 2^(params) but typically polynomial
    upper_bound = num_params * (num_params + 1)

    # Expected (empirical observation)
    expected = int(1.5 * num_params)

    return {
        'num_params': num_params,
        'vc_lower_bound': lower_bound,
        'vc_upper_bound': upper_bound,
        'vc_expected': expected,
        'note': 'VC dimension typically scales linearly with parameters'
    }
```

## 6. Generalization and Expressibility Trade-off

```python
def analyze_expressibility_generalization_tradeoff(
    model_fn: Callable,
    X_train: np.ndarray,
    y_train: np.ndarray,
    X_test: np.ndarray,
    y_test: np.ndarray,
    param_counts: List[int]
) -> dict:
    """
    Analyze trade-off between model complexity and generalization.
    """
    results = []

    for num_params in param_counts:
        # Train model
        best_train_error = 1.0
        best_test_error = 1.0

        for _ in range(10):  # Multiple random initializations
            params = np.random.uniform(0, 2*np.pi, num_params)

            # Simple training (gradient descent)
            lr = 0.1
            for _ in range(50):
                grad = np.zeros(num_params)
                for i in range(num_params):
                    p_plus = params.copy()
                    p_plus[i] += np.pi/2
                    p_minus = params.copy()
                    p_minus[i] -= np.pi/2

                    train_loss_plus = compute_loss(model_fn, p_plus, X_train, y_train)
                    train_loss_minus = compute_loss(model_fn, p_minus, X_train, y_train)
                    grad[i] = (train_loss_plus - train_loss_minus) / 2

                params -= lr * grad

            # Evaluate
            train_error = compute_error(model_fn, params, X_train, y_train)
            test_error = compute_error(model_fn, params, X_test, y_test)

            if train_error < best_train_error:
                best_train_error = train_error
                best_test_error = test_error

        results.append({
            'num_params': num_params,
            'train_error': best_train_error,
            'test_error': best_test_error,
            'generalization_gap': best_test_error - best_train_error
        })

    return {
        'results': results,
        'optimal_params': min(results, key=lambda x: x['test_error'])['num_params']
    }

def compute_loss(model, params, X, y):
    predictions = np.array([model(x, params) for x in X])
    return np.mean((predictions - y) ** 2)

def compute_error(model, params, X, y):
    predictions = np.array([1 if model(x, params) > 0.5 else 0 for x in X])
    return np.mean(predictions != y)
```

## 7. Information-Theoretic Bounds

```python
def information_theoretic_bounds(
    num_qubits: int,
    num_measurements: int,
    precision_bits: int = 10
) -> dict:
    """
    Information-theoretic limits on quantum learning.

    Args:
        num_qubits: Number of qubits in circuit
        num_measurements: Total measurement budget
        precision_bits: Bits of precision per measurement
    """
    import math

    # Holevo bound: max classical info from n qubits
    holevo_bound = num_qubits  # bits per copy

    # Total information extractable
    total_info = num_measurements * precision_bits

    # Parameters that can be learned
    max_learnable_params = total_info // 10  # Rough estimate

    # Shadow tomography bounds
    # Can learn m observables to epsilon precision with
    # O(log(m) / epsilon^2) measurements
    shadow_efficiency = "O(log(m) / eps^2)"

    return {
        'holevo_bound_per_state': holevo_bound,
        'total_classical_info': total_info,
        'max_learnable_params': max_learnable_params,
        'shadow_tomography_scaling': shadow_efficiency,
        'measurement_efficiency': total_info / num_measurements
    }
```
