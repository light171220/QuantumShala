## 1. Logical Qubits and Operations

In fault-tolerant quantum computing, quantum information is encoded in logical qubits that are protected by quantum error correction codes.

<Definition term="Logical Qubit">
A logical qubit is encoded in multiple physical qubits using a quantum error correction code. Operations on logical qubits must preserve the code space and propagate errors in a controlled manner.
</Definition>

<Formula>
$$|\overline{0}\rangle = \frac{1}{\sqrt{2^{n-k}}} \sum_{s \in S} s|0^n\rangle$$
</Formula>

where S is the stabilizer group of the code.

```python
import numpy as np
from typing import List, Tuple, Dict, Optional
from dataclasses import dataclass
from enum import Enum

class GateType(Enum):
    """Classification of gates by fault-tolerance properties."""
    TRANSVERSAL = "transversal"
    MAGIC_STATE = "magic_state"
    CODE_DEFORMATION = "code_deformation"
    LATTICE_SURGERY = "lattice_surgery"

@dataclass
class LogicalGate:
    """Specification of a logical gate."""
    name: str
    gate_type: GateType
    physical_operations: int
    logical_error_rate: float
    time_steps: int

class StabilizerCode:
    """
    Base class for stabilizer codes with logical operations.
    """

    def __init__(self, n: int, k: int, d: int):
        """
        Initialize stabilizer code [[n, k, d]].

        Args:
            n: Number of physical qubits
            k: Number of logical qubits
            d: Code distance
        """
        self.n = n
        self.k = k
        self.d = d
        self.stabilizers = []
        self.logical_x = []
        self.logical_z = []

    def transversal_gates(self) -> List[str]:
        """Return list of transversal gates for this code."""
        raise NotImplementedError

    def logical_error_rate(self, physical_error_rate: float) -> float:
        """
        Estimate logical error rate.

        For code with distance d, logical error ~ (p/p_th)^(d/2)
        """
        p_th = 0.01  # Typical threshold
        if physical_error_rate < p_th:
            return (physical_error_rate / p_th) ** (self.d // 2)
        else:
            return 1.0

class SteaneCode(StabilizerCode):
    """
    The [[7,1,3]] Steane code - smallest code with transversal CNOT.
    """

    def __init__(self):
        super().__init__(n=7, k=1, d=3)
        self._initialize_stabilizers()

    def _initialize_stabilizers(self):
        # X stabilizers
        self.x_stabilizers = [
            [1, 1, 1, 1, 0, 0, 0],  # X1 X2 X3 X4
            [1, 1, 0, 0, 1, 1, 0],  # X1 X2 X5 X6
            [1, 0, 1, 0, 1, 0, 1]   # X1 X3 X5 X7
        ]

        # Z stabilizers
        self.z_stabilizers = [
            [1, 1, 1, 1, 0, 0, 0],  # Z1 Z2 Z3 Z4
            [1, 1, 0, 0, 1, 1, 0],  # Z1 Z2 Z5 Z6
            [1, 0, 1, 0, 1, 0, 1]   # Z1 Z3 Z5 Z7
        ]

        # Logical operators
        self.logical_x = [1, 1, 1, 1, 1, 1, 1]  # X on all qubits
        self.logical_z = [1, 1, 1, 1, 1, 1, 1]  # Z on all qubits

    def transversal_gates(self) -> List[str]:
        """Steane code has transversal H, S, CNOT."""
        return ['H', 'S', 'CNOT', 'X', 'Z']

    def encode_logical_zero(self) -> np.ndarray:
        """
        Return the logical |0> state for Steane code.
        """
        # |0_L> is superposition of even-weight codewords
        dim = 2 ** 7
        state = np.zeros(dim, dtype=complex)

        # Codewords: apply stabilizers to |0000000>
        codewords = [
            0b0000000,
            0b1010101,
            0b0110011,
            0b1100110,
            0b0001111,
            0b1011010,
            0b0111100,
            0b1101001
        ]

        for cw in codewords:
            state[cw] = 1.0 / np.sqrt(8)

        return state
```

## 2. Transversal Gates

Transversal gates apply the same single-qubit gate to each physical qubit, automatically preserving fault-tolerance.

<Definition term="Transversal Gate">
A gate is transversal if it can be implemented by applying operations to corresponding qubits of code blocks independently. Errors cannot spread between qubits within a block.
</Definition>

```python
def apply_transversal_gate(
    state: np.ndarray,
    gate: np.ndarray,
    n_qubits: int
) -> np.ndarray:
    """
    Apply transversal gate (same single-qubit gate to all qubits).

    Args:
        state: Encoded state vector
        gate: 2x2 single-qubit gate matrix
        n_qubits: Number of physical qubits
    """
    result = state.copy()

    for qubit in range(n_qubits):
        result = apply_single_qubit_gate(result, gate, qubit, n_qubits)

    return result

def apply_single_qubit_gate(
    state: np.ndarray,
    gate: np.ndarray,
    qubit: int,
    n_qubits: int
) -> np.ndarray:
    """Apply single-qubit gate to specified qubit."""
    dim = 2 ** n_qubits
    new_state = np.zeros(dim, dtype=complex)

    for i in range(dim):
        bit = (i >> (n_qubits - 1 - qubit)) & 1
        j = i ^ (1 << (n_qubits - 1 - qubit))

        if bit == 0:
            new_state[i] += gate[0, 0] * state[i]
            new_state[i] += gate[0, 1] * state[j]
        else:
            new_state[i] += gate[1, 0] * state[j]
            new_state[i] += gate[1, 1] * state[i]

    return new_state

class TransversalGateSet:
    """
    Collection of transversal gates for a code.
    """

    def __init__(self, code: StabilizerCode):
        self.code = code
        self._build_gates()

    def _build_gates(self):
        """Build transversal gate matrices."""
        self.H = np.array([[1, 1], [1, -1]]) / np.sqrt(2)
        self.S = np.array([[1, 0], [0, 1j]])
        self.X = np.array([[0, 1], [1, 0]])
        self.Z = np.array([[1, 0], [0, -1]])

    def apply_logical_H(self, state: np.ndarray) -> np.ndarray:
        """Apply logical Hadamard via transversal H."""
        return apply_transversal_gate(state, self.H, self.code.n)

    def apply_logical_S(self, state: np.ndarray) -> np.ndarray:
        """Apply logical S (phase) gate."""
        return apply_transversal_gate(state, self.S, self.code.n)

    def apply_logical_cnot(
        self,
        state: np.ndarray,
        control_block: int,
        target_block: int
    ) -> np.ndarray:
        """
        Apply logical CNOT between two code blocks.

        For transversal CNOT: apply CNOT_i,j for each pair of
        corresponding physical qubits.
        """
        n = self.code.n
        # State has 2 * n qubits total (two code blocks)
        total_qubits = 2 * n
        result = state.copy()

        for i in range(n):
            control_qubit = control_block * n + i
            target_qubit = target_block * n + i
            result = apply_cnot(result, control_qubit, target_qubit, total_qubits)

        return result

def apply_cnot(
    state: np.ndarray,
    control: int,
    target: int,
    n_qubits: int
) -> np.ndarray:
    """Apply CNOT gate."""
    dim = 2 ** n_qubits
    new_state = state.copy()

    for i in range(dim):
        if (i >> (n_qubits - 1 - control)) & 1:
            j = i ^ (1 << (n_qubits - 1 - target))
            new_state[i], new_state[j] = state[j], state[i]

    return new_state
```

## 3. The Eastin-Knill Theorem

<Definition term="Eastin-Knill Theorem">
No quantum error-correcting code can have a universal set of transversal gates. At least one gate must be implemented non-transversally (e.g., via magic state injection).
</Definition>

This fundamental result means we need additional techniques for universal fault-tolerant computation:

```python
def eastin_knill_implications() -> Dict:
    """
    Explain implications of Eastin-Knill theorem.
    """
    return {
        'theorem': 'No code has universal transversal gates',
        'codes_and_transversal_gates': {
            'Steane [[7,1,3]]': ['H', 'S', 'CNOT'],  # Missing T
            'Surface code': ['CNOT', 'H'],  # Missing S, T
            'Color code': ['H', 'S', 'CNOT'],  # Missing T
            'Reed-Muller [[15,1,3]]': ['T', 'CNOT']  # Missing H!

        'solutions': [
            'Magic state distillation for missing gates',
            'Code switching between complementary codes',
            'Lattice surgery for some operations',
            'Gate teleportation with magic states'
        ],
        'cost': 'Non-transversal gates are orders of magnitude more expensive'
    }
```

## 4. Magic State Distillation

Magic states enable implementation of non-transversal gates like T:

<Formula>
$$|T\rangle = \frac{1}{\sqrt{2}}(|0\rangle + e^{i\pi/4}|1\rangle)$$
</Formula>

```python
@dataclass
class MagicState:
    """Specification of a magic state."""
    name: str
    state_vector: np.ndarray
    target_gate: str
    fidelity: float

def create_magic_states() -> Dict[str, MagicState]:
    """Create common magic states."""
    # T state: |T> = (|0> + e^{i*pi/4}|1>)/sqrt(2)
    T_state = np.array([1, np.exp(1j * np.pi / 4)]) / np.sqrt(2)

    # H state: |H> = cos(pi/8)|0> + sin(pi/8)|1>
    H_state = np.array([np.cos(np.pi/8), np.sin(np.pi/8)])

    return {
        'T': MagicState('T', T_state, 'T gate', 1.0),
        'H': MagicState('H', H_state, 'H magic', 1.0)
    }

class MagicStateDistillation:
    """
    Magic state distillation protocols.
    """

    def __init__(self, protocol: str = "15-to-1"):
        self.protocol = protocol
        self._configure_protocol()

    def _configure_protocol(self):
        """Configure distillation parameters."""
        if self.protocol == "15-to-1":
            # Standard 15-to-1 T state distillation
            self.input_states = 15
            self.output_states = 1
            self.input_error = None  # Set during distillation
            self.output_error_factor = 35  # Output error ~ 35 * input_error^3
            self.success_probability = 1 - 15 * 0.01  # ~85% at p=0.01

        elif self.protocol == "20-to-4":
            # Efficient 20-to-4 protocol
            self.input_states = 20
            self.output_states = 4
            self.output_error_factor = 20
            self.success_probability = 0.9

    def distill(
        self,
        input_fidelity: float,
        target_fidelity: float
    ) -> Dict:
        """
        Calculate distillation requirements.

        Args:
            input_fidelity: Fidelity of input magic states
            target_fidelity: Required output fidelity
        """
        input_error = 1 - input_fidelity
        target_error = 1 - target_fidelity

        # Number of rounds needed
        rounds = 0
        current_error = input_error

        while current_error > target_error:
            # Each round: error -> factor * error^3
            current_error = self.output_error_factor * (current_error ** 3)
            rounds += 1

            if rounds > 10:  # Safety limit
                break

        # Total input states needed
        total_inputs = (self.input_states ** rounds) * self.output_states

        # Total physical qubits (each magic state needs code block)
        # For [[7,1,3]] code: 7 qubits per state
        physical_qubits_per_state = 7  # Code dependent
        total_physical_qubits = total_inputs * physical_qubits_per_state

        return {
            'rounds': rounds,
            'total_input_states': total_inputs,
            'output_states': self.output_states,
            'output_error': current_error,
            'output_fidelity': 1 - current_error,
            'physical_qubits': total_physical_qubits,
            'success_probability': self.success_probability ** rounds
        }

    def t_gate_via_injection(
        self,
        magic_state_error: float
    ) -> Dict:
        """
        Implement T gate via magic state injection.

        Circuit:
        |psi> ---[controlled-S]--- T|psi>
        |T>  ---[measure X]-------

        If measurement = +1: done
        If measurement = -1: apply S^dag correction
        """
        return {
            'circuit': 'Controlled-S + X measurement + correction',
            'magic_states_consumed': 1,
            'success_probability': 1.0,  # Deterministic with correction
            'output_error': magic_state_error,
            'clifford_gates': 3,  # Controlled-S (2) + correction (1)
            'measurements': 1
        }
```

## 5. Fault-Tolerant Gate Sequences

For a universal gate set, we need to combine transversal gates with magic states:

```python
class FaultTolerantGateSet:
    """
    Complete fault-tolerant universal gate set.
    """

    def __init__(self, code: StabilizerCode):
        self.code = code
        self.transversal = TransversalGateSet(code)
        self.distillation = MagicStateDistillation()

    def available_gates(self) -> Dict[str, LogicalGate]:
        """Return all available logical gates with costs."""
        # Physical error rate (typical)
        p = 0.001

        gates = {
            'X': LogicalGate('X', GateType.TRANSVERSAL, self.code.n, p, 1),
            'Z': LogicalGate('Z', GateType.TRANSVERSAL, self.code.n, p, 1),
            'H': LogicalGate('H', GateType.TRANSVERSAL, self.code.n, p, 1),
            'S': LogicalGate('S', GateType.TRANSVERSAL, self.code.n, p, 1),
            'CNOT': LogicalGate('CNOT', GateType.TRANSVERSAL, 2*self.code.n, p, 1),
            'T': LogicalGate('T', GateType.MAGIC_STATE,
                           self._t_gate_cost(), self._t_gate_error(p), 10)
        }

        return gates

    def _t_gate_cost(self) -> int:
        """Physical operations for T gate."""
        # Approximate: distillation + injection
        return 100 * self.code.n  # Order of magnitude

    def _t_gate_error(self, p: float) -> float:
        """Logical error rate for T gate."""
        # After distillation
        distill_result = self.distillation.distill(1 - p, 1 - 1e-10)
        return distill_result['output_error']

    def decompose_arbitrary_rotation(
        self,
        angle: float,
        precision: float
    ) -> List[str]:
        """
        Decompose arbitrary Z-rotation into H, S, T gates.

        Uses Solovay-Kitaev or grid-based synthesis.
        """
        # Number of T gates needed scales as O(log(1/precision))
        num_t_gates = int(np.ceil(3 * np.log2(1 / precision)))

        # Simplified decomposition (actual requires SK algorithm)
        sequence = []
        remaining_angle = angle

        # Approximate with T, T^dag, S, S^dag
        while abs(remaining_angle) > precision:
            if remaining_angle > np.pi / 4:
                sequence.append('T')
                remaining_angle -= np.pi / 4
            elif remaining_angle < -np.pi / 4:
                sequence.append('Tdag')
                remaining_angle += np.pi / 4
            else:
                # Finer approximation needed
                sequence.append('T')
                sequence.append('S')
                break

        return sequence

def analyze_circuit_cost(
    circuit: List[str],
    gate_set: FaultTolerantGateSet
) -> Dict:
    """
    Analyze total cost of a fault-tolerant circuit.
    """
    gates = gate_set.available_gates()

    total_physical_ops = 0
    total_time_steps = 0
    total_t_gates = 0
    total_clifford = 0

    for gate_name in circuit:
        if gate_name in gates:
            gate = gates[gate_name]
            total_physical_ops += gate.physical_operations
            total_time_steps += gate.time_steps

            if gate.gate_type == GateType.MAGIC_STATE:
                total_t_gates += 1
            else:
                total_clifford += 1

    # Magic state factory requirements
    magic_states_needed = total_t_gates
    distill_info = gate_set.distillation.distill(0.99, 0.9999999)

    return {
        'clifford_gates': total_clifford,
        't_gates': total_t_gates,
        'total_physical_operations': total_physical_ops,
        'time_steps': total_time_steps,
        'magic_states_needed': magic_states_needed,
        'magic_state_factory_qubits': distill_info['physical_qubits'],
        'bottleneck': 'T gate production' if total_t_gates > 0 else 'Circuit depth'
    }
```

## 6. Error Analysis

```python
def logical_error_analysis(
    code: StabilizerCode,
    circuit_depth: int,
    physical_error_rate: float
) -> Dict:
    """
    Analyze logical error rates for a fault-tolerant circuit.
    """
    # Logical error per round of error correction
    logical_error_per_round = code.logical_error_rate(physical_error_rate)

    # Total logical error (accumulates with depth)
    total_logical_error = 1 - (1 - logical_error_per_round) ** circuit_depth

    # Threshold behavior
    p_th = 0.01  # Typical threshold
    below_threshold = physical_error_rate < p_th

    # Break-even analysis
    if below_threshold:
        effective_error_reduction = physical_error_rate / logical_error_per_round
    else:
        effective_error_reduction = 0  # No benefit above threshold

    return {
        'physical_error_rate': physical_error_rate,
        'logical_error_per_round': logical_error_per_round,
        'circuit_depth': circuit_depth,
        'total_logical_error': total_logical_error,
        'below_threshold': below_threshold,
        'error_reduction_factor': effective_error_reduction,
        'recommendation': (
            'Good operating point' if effective_error_reduction > 10 else
            'Marginal benefit' if effective_error_reduction > 1 else
            'Above threshold - no benefit'
        )
    }
```
