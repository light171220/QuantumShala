## 1. The Optimization Landscape

The VQE energy landscape is the function $E(\theta): \mathbb{R}^p \to \mathbb{R}$:

<Formula>
$$E(\theta) = \langle\psi(\theta)|H|\psi(\theta)\rangle$$
</Formula>

```python
import numpy as np
from typing import Callable, Tuple

def compute_gradient(
    energy_fn: Callable,
    params: np.ndarray,
    epsilon: float = 1e-5
) -> np.ndarray:
    """Numerical gradient computation."""
    grad = np.zeros_like(params)
    for i in range(len(params)):
        params_plus = params.copy()
        params_plus[i] += epsilon
        params_minus = params.copy()
        params_minus[i] -= epsilon
        grad[i] = (energy_fn(params_plus) - energy_fn(params_minus)) / (2 * epsilon)
    return grad

def compute_hessian(
    energy_fn: Callable,
    params: np.ndarray,
    epsilon: float = 1e-4
) -> np.ndarray:
    """Numerical Hessian computation."""
    n = len(params)
    hessian = np.zeros((n, n))

    for i in range(n):
        for j in range(n):
            params_pp = params.copy()
            params_pp[i] += epsilon
            params_pp[j] += epsilon

            params_pm = params.copy()
            params_pm[i] += epsilon
            params_pm[j] -= epsilon

            params_mp = params.copy()
            params_mp[i] -= epsilon
            params_mp[j] += epsilon

            params_mm = params.copy()
            params_mm[i] -= epsilon
            params_mm[j] -= epsilon

            hessian[i, j] = (energy_fn(params_pp) - energy_fn(params_pm)
                           - energy_fn(params_mp) + energy_fn(params_mm)) / (4 * epsilon**2)

    return hessian

def classify_critical_point(hessian: np.ndarray) -> str:
    """Classify critical point using Hessian eigenvalues."""
    eigenvalues = np.linalg.eigvalsh(hessian)

    if np.all(eigenvalues > 0):
        return "local_minimum"
    elif np.all(eigenvalues < 0):
        return "local_maximum"
    elif np.any(eigenvalues > 0) and np.any(eigenvalues < 0):
        return "saddle_point"
    else:
        return "degenerate"
```

## 2. Local Minima Analysis

```python
def find_local_minima(
    energy_fn: Callable,
    num_params: int,
    num_starts: int = 50,
    tolerance: float = 1e-6
) -> list:
    """Find local minima via multiple random starts."""
    from scipy.optimize import minimize

    minima = []

    for _ in range(num_starts):
        x0 = np.random.uniform(-np.pi, np.pi, num_params)
        result = minimize(energy_fn, x0, method='BFGS',
                         options={'gtol': tolerance})

        if result.success:
            # Check if this is a new minimum
            is_new = True
            for existing in minima:
                if np.allclose(result.x % (2*np.pi), existing['params'] % (2*np.pi), atol=0.1):
                    is_new = False
                    break

            if is_new:
                hessian = compute_hessian(energy_fn, result.x)
                minima.append({
                    'params': result.x,
                    'energy': result.fun,
                    'hessian_eigenvalues': np.linalg.eigvalsh(hessian),
                    'type': classify_critical_point(hessian)
                })

    return sorted(minima, key=lambda x: x['energy'])
```

## 3. Landscape Visualization

```python
def visualize_2d_slice(
    energy_fn: Callable,
    base_params: np.ndarray,
    directions: Tuple[int, int],
    range_val: float = np.pi,
    resolution: int = 50
) -> dict:
    """Create 2D slice of energy landscape."""
    i, j = directions
    vals_i = np.linspace(-range_val, range_val, resolution)
    vals_j = np.linspace(-range_val, range_val, resolution)

    landscape = np.zeros((resolution, resolution))

    for ii, vi in enumerate(vals_i):
        for jj, vj in enumerate(vals_j):
            params = base_params.copy()
            params[i] = vi
            params[j] = vj
            landscape[ii, jj] = energy_fn(params)

    return {
        'landscape': landscape,
        'axis_i': vals_i,
        'axis_j': vals_j,
        'directions': directions
    }
```
