## 1. Key Hyperparameters

1. **Learning rate**: Step size for parameter updates
2. **Ansatz depth**: Number of variational layers
3. **Shot count**: Measurements per evaluation
4. **Optimizer settings**: Momentum, decay rates

```python
import numpy as np
from typing import Dict, List

def grid_search_lr(energy_fn, num_params, lr_values, num_trials=3):
    """Grid search over learning rates."""
    from scipy.optimize import minimize

    results = {}
    for lr in lr_values:
        trial_energies = []
        for _ in range(num_trials):
            x0 = np.random.uniform(-np.pi, np.pi, num_params)
            # Simulate gradient descent with given learning rate
            x = x0.copy()
            for _ in range(100):
                grad = numerical_gradient(energy_fn, x)
                x = x - lr * grad
            trial_energies.append(energy_fn(x))
        results[lr] = {'mean': np.mean(trial_energies), 'std': np.std(trial_energies)}
    return results

def numerical_gradient(fn, x, eps=1e-5):
    grad = np.zeros_like(x)
    for i in range(len(x)):
        x_p = x.copy(); x_p[i] += eps
        x_m = x.copy(); x_m[i] -= eps
        grad[i] = (fn(x_p) - fn(x_m)) / (2*eps)
    return grad
```

## 2. Depth Selection

```python
def select_optimal_depth(energy_fn, max_depth, target_accuracy):
    """Find minimal depth achieving target accuracy."""
    for depth in range(1, max_depth + 1):
        # Run optimization at this depth
        energy, _ = optimize_at_depth(energy_fn, depth)
        if energy <= target_accuracy:
            return depth, energy
    return max_depth, energy

def optimize_at_depth(energy_fn, depth, trials=5):
    """Optimize at given depth with multiple restarts."""
    from scipy.optimize import minimize
    best = float('inf')
    best_params = None
    for _ in range(trials):
        x0 = np.random.uniform(-np.pi, np.pi, depth * 2)
        result = minimize(lambda p: energy_fn(p, depth), x0, method='COBYLA')
        if result.fun < best:
            best = result.fun
            best_params = result.x
    return best, best_params
```

## 3. Shot Budget Allocation

```python
def optimal_shot_distribution(total_shots, num_iterations):
    """Distribute shots across iterations."""
    # More shots early for exploration, fewer later
    weights = np.exp(-np.arange(num_iterations) / num_iterations)
    weights = weights / weights.sum()
    return (weights * total_shots).astype(int)
```
