## 1. The Noise Challenge in NISQ Algorithms

Current quantum computers are noisy, with error rates that can significantly corrupt VQE results:

<Definition term="NISQ">
Noisy Intermediate-Scale Quantum devices are quantum computers with 50-1000 qubits that lack full error correction. They must use error mitigation rather than error correction to produce useful results.
</Definition>

### Sources of Noise in VQE

| Noise Type | Source | Effect on VQE |
|------------|--------|---------------|
| **Gate errors** | Imperfect control pulses | Biased energy estimates |
| **Decoherence** | T1, T2 relaxation | Loss of quantum correlations |
| **Readout errors** | Measurement imperfections | Biased measurement statistics |
| **Crosstalk** | Unwanted qubit interactions | Correlated errors |
| **Leakage** | Population in non-computational states | Non-physical results |

<Warning>
Without error mitigation, VQE on noisy hardware typically produces energies biased toward zero (or the maximally mixed state expectation value). This bias can completely obscure the quantum advantage.
</Warning>

## 2. Zero-Noise Extrapolation (ZNE)

ZNE extrapolates results to the zero-noise limit by artificially amplifying noise and fitting the trend.

<Definition term="Zero-Noise Extrapolation">
A technique that runs circuits at multiple noise levels and uses extrapolation to estimate the zero-noise result. Noise is typically amplified by stretching or folding gates.
</Definition>

<Formula>
$$E(\lambda) = E_0 + a_1\lambda + a_2\lambda^2 + \ldots$$
</Formula>

where $\lambda$ is the noise scale factor and $E_0$ is the ideal (zero-noise) expectation value.

```python
import numpy as np
from typing import Callable, List, Tuple
from dataclasses import dataclass

@dataclass
class ZNEResult:
    """Results from zero-noise extrapolation."""
    extrapolated_value: float
    noise_levels: List[float]
    measured_values: List[float]
    fit_coefficients: np.ndarray
    extrapolation_uncertainty: float

class ZeroNoiseExtrapolation:
    """
    Zero-noise extrapolation for error mitigation.
    """

    def __init__(
        self,
        noise_factors: List[float] = [1, 2, 3],
        extrapolation_order: int = 2
    ):
        """
        Args:
            noise_factors: Noise scaling factors (1 = nominal noise)
            extrapolation_order: Polynomial order for extrapolation
        """
        self.noise_factors = noise_factors
        self.extrapolation_order = extrapolation_order

    def amplify_noise_by_folding(
        self,
        circuit: List[dict],
        scale_factor: int
    ) -> List[dict]:
        """
        Amplify noise by circuit folding: G -> G G^dag G.

        Each fold approximately doubles the noise for that gate.

        Args:
            circuit: Original circuit as list of gate operations
            scale_factor: How many times to fold (1 = no folding)

        Returns:
            Folded circuit
        """
        if scale_factor == 1:
            return circuit

        folded_circuit = []
        folds_needed = scale_factor - 1

        for gate in circuit:
            # Add original gate
            folded_circuit.append(gate)

            # Add folds: G G^dag G G^dag ... G
            for _ in range(folds_needed):
                # Add G^dag (inverse)
                folded_circuit.append(self._inverse_gate(gate))
                # Add G again
                folded_circuit.append(gate)

        return folded_circuit

    def _inverse_gate(self, gate: dict) -> dict:
        """Return the inverse of a gate."""
        inv_gate = gate.copy()
        if 'angle' in gate:
            inv_gate['angle'] = -gate['angle']
        inv_gate['is_inverse'] = True
        return inv_gate

    def extrapolate(
        self,
        executor: Callable[[float], float],
    ) -> ZNEResult:
        """
        Run ZNE and extrapolate to zero noise.

        Args:
            executor: Function that runs circuit at given noise level
                     and returns expectation value

        Returns:
            ZNEResult with extrapolated value
        """
        # Collect data at different noise levels
        measured_values = []
        for factor in self.noise_factors:
            value = executor(factor)
            measured_values.append(value)

        # Fit polynomial to noise factor vs measured value
        coeffs = np.polyfit(
            self.noise_factors,
            measured_values,
            self.extrapolation_order
        )

        # Extrapolate to noise_factor = 0
        extrapolated = np.polyval(coeffs, 0)

        # Estimate uncertainty from fit residuals
        fit_values = np.polyval(coeffs, self.noise_factors)
        residuals = np.array(measured_values) - fit_values
        uncertainty = np.std(residuals)

        return ZNEResult(
            extrapolated_value=extrapolated,
            noise_levels=self.noise_factors,
            measured_values=measured_values,
            fit_coefficients=coeffs,
            extrapolation_uncertainty=uncertainty
        )

# Example usage
def noisy_vqe_executor(noise_factor: float, base_noise: float = 0.05) -> float:
    """Simulated noisy VQE execution."""
    true_energy = -1.5
    # Noise pushes energy toward zero
    effective_noise = base_noise * noise_factor
    noisy_energy = true_energy * np.exp(-effective_noise * 10)
    return noisy_energy + 0.01 * np.random.randn()

zne = ZeroNoiseExtrapolation(noise_factors=[1, 1.5, 2, 3])
result = zne.extrapolate(noisy_vqe_executor)

print(f"True energy: -1.5")
print(f"Noisy (factor=1): {noisy_vqe_executor(1):.4f}")
print(f"ZNE extrapolated: {result.extrapolated_value:.4f}")
```

### Richardson Extrapolation

For more accurate extrapolation, use Richardson's method:

```python
def richardson_extrapolation(
    noise_factors: List[float],
    values: List[float]
) -> float:
    """
    Apply Richardson extrapolation for ZNE.

    Uses Lagrange interpolation to extrapolate to zero noise.
    """
    n = len(noise_factors)
    extrapolated = 0.0

    for i in range(n):
        # Compute Lagrange basis polynomial at x=0
        basis = 1.0
        for j in range(n):
            if i != j:
                basis *= (0 - noise_factors[j]) / (noise_factors[i] - noise_factors[j])
        extrapolated += values[i] * basis

    return extrapolated
```

## 3. Probabilistic Error Cancellation (PEC)

PEC exactly inverts noise channels using a quasi-probability decomposition.

<Definition term="Probabilistic Error Cancellation">
A technique that expresses the inverse of a noise channel as a linear combination of physically implementable operations. The ideal result is recovered as a weighted average over sampled circuits.
</Definition>

<Formula>
$$\mathcal{N}^{-1} = \sum_i \eta_i \mathcal{O}_i, \quad \text{where } \sum_i |\eta_i| = \gamma \geq 1$$
</Formula>

```python
from typing import Dict

class ProbabilisticErrorCancellation:
    """
    Probabilistic error cancellation for gate-level noise.
    """

    def __init__(self, noise_model: Dict[str, np.ndarray]):
        """
        Args:
            noise_model: Dictionary mapping gate names to their
                        noise channel (as Kraus operators or process matrix)
        """
        self.noise_model = noise_model
        self.quasi_probabilities = {}
        self._compute_quasi_probabilities()

    def _compute_quasi_probabilities(self):
        """
        Compute quasi-probability decomposition for each noisy gate.

        For depolarizing channel: N = (1-p)I + p*D
        Inverse: N^-1 = (1/(1-p))I - (p/(1-p))D (approximately)
        """
        for gate_name, noise_params in self.noise_model.items():
            if isinstance(noise_params, float):
                # Assume depolarizing noise with error rate p
                p = noise_params
                # Quasi-probability decomposition
                self.quasi_probabilities[gate_name] = {
                    'identity': 1 / (1 - p),
                    'X': -p / (3 * (1 - p)),
                    'Y': -p / (3 * (1 - p)),
                    'Z': -p / (3 * (1 - p))
                }

    def get_sampling_cost(self) -> float:
        """
        Compute the sampling overhead factor gamma.

        gamma = product over gates of sum of |quasi-prob|
        """
        gamma = 1.0
        for gate_name, qp in self.quasi_probabilities.items():
            gate_gamma = sum(abs(v) for v in qp.values())
            gamma *= gate_gamma
        return gamma

    def sample_circuit_variant(
        self,
        circuit: List[dict]
    ) -> Tuple[List[dict], float]:
        """
        Sample a circuit variant according to quasi-probabilities.

        Returns:
            (modified_circuit, sign) where sign is +1 or -1
        """
        modified_circuit = []
        total_sign = 1.0

        for gate in circuit:
            gate_name = gate.get('name', 'generic')
            if gate_name not in self.quasi_probabilities:
                modified_circuit.append(gate)
                continue

            qp = self.quasi_probabilities[gate_name]

            # Sample operation according to |quasi-prob|
            ops = list(qp.keys())
            probs = np.array([abs(qp[op]) for op in ops])
            probs /= probs.sum()

            chosen_op = np.random.choice(ops, p=probs)

            # Update sign based on whether quasi-prob is positive/negative
            if qp[chosen_op] < 0:
                total_sign *= -1

            # Add modified gate
            if chosen_op == 'identity':
                modified_circuit.append(gate)
            else:
                # Insert Pauli after gate
                modified_circuit.append(gate)
                modified_circuit.append({
                    'name': chosen_op,
                    'qubit': gate['qubit']
                })

        return modified_circuit, total_sign

    def mitigate(
        self,
        executor: Callable[[List[dict]], float],
        circuit: List[dict],
        num_samples: int = 1000
    ) -> Tuple[float, float]:
        """
        Apply PEC to estimate error-mitigated expectation value.

        Returns:
            (mitigated_value, statistical_uncertainty)
        """
        gamma = self.get_sampling_cost()
        results = []

        for _ in range(num_samples):
            variant_circuit, sign = self.sample_circuit_variant(circuit)
            value = executor(variant_circuit)
            results.append(sign * gamma * value)

        mitigated = np.mean(results)
        uncertainty = np.std(results) / np.sqrt(num_samples)

        return mitigated, uncertainty
```

<Warning>
PEC's sampling cost grows exponentially with circuit depth: $\gamma = \prod_{gates} \gamma_{gate}$. This limits PEC to relatively shallow circuits.
</Warning>

## 4. Measurement Error Mitigation

Readout errors can be characterized and corrected using calibration measurements.

<Definition term="Measurement Error Mitigation">
Correction of systematic biases in qubit readout by characterizing the confusion matrix through calibration measurements and applying its inverse to experimental results.
</Definition>

```python
class MeasurementErrorMitigation:
    """
    Mitigate measurement readout errors.
    """

    def __init__(self, num_qubits: int):
        self.num_qubits = num_qubits
        self.calibration_matrix = None

    def calibrate(
        self,
        executor: Callable[[str], Dict[str, int]],
        shots_per_state: int = 10000
    ):
        """
        Calibrate measurement error by preparing computational basis states.

        Args:
            executor: Function that prepares state and returns counts
            shots_per_state: Measurement shots per basis state
        """
        n = self.num_qubits
        num_states = 2 ** n

        # Initialize calibration matrix
        # M[i,j] = P(measure i | prepared j)
        M = np.zeros((num_states, num_states))

        for j in range(num_states):
            # Prepare computational basis state |j>
            prep_state = format(j, f'0{n}b')
            counts = executor(prep_state)
            total_counts = sum(counts.values())

            for measured_state, count in counts.items():
                i = int(measured_state, 2)
                M[i, j] = count / total_counts

        self.calibration_matrix = M

    def get_inverse_matrix(self) -> np.ndarray:
        """Return pseudo-inverse of calibration matrix."""
        return np.linalg.pinv(self.calibration_matrix)

    def mitigate_counts(
        self,
        counts: Dict[str, int]
    ) -> Dict[str, float]:
        """
        Apply measurement error mitigation to raw counts.

        Args:
            counts: Raw measurement counts

        Returns:
            Mitigated probability distribution
        """
        if self.calibration_matrix is None:
            raise ValueError("Must calibrate before mitigation")

        n = self.num_qubits
        num_states = 2 ** n

        # Convert counts to probability vector
        total = sum(counts.values())
        raw_probs = np.zeros(num_states)
        for state, count in counts.items():
            idx = int(state, 2)
            raw_probs[idx] = count / total

        # Apply inverse calibration matrix
        M_inv = self.get_inverse_matrix()
        mitigated_probs = M_inv @ raw_probs

        # Clip negative probabilities (can occur due to noise in calibration)
        mitigated_probs = np.clip(mitigated_probs, 0, None)
        mitigated_probs /= mitigated_probs.sum()

        # Convert back to dictionary
        result = {}
        for i in range(num_states):
            if mitigated_probs[i] > 1e-10:
                state = format(i, f'0{n}b')
                result[state] = mitigated_probs[i]

        return result

    def mitigate_expectation(
        self,
        counts: Dict[str, int],
        observable: np.ndarray
    ) -> float:
        """
        Compute mitigated expectation value of diagonal observable.

        Args:
            counts: Raw measurement counts
            observable: Diagonal elements of observable

        Returns:
            Mitigated expectation value
        """
        mitigated_probs = self.mitigate_counts(counts)

        expectation = 0.0
        for state, prob in mitigated_probs.items():
            idx = int(state, 2)
            expectation += prob * observable[idx]

        return expectation
```

## 6. Integrated Error Mitigation Strategy

Combine multiple techniques for best results:

```python
class IntegratedErrorMitigation:
    """
    Combine multiple error mitigation techniques.
    """

    def __init__(
        self,
        use_zne: bool = True,
        use_pec: bool = False,
        use_measurement_mitigation: bool = True
    ):
        self.use_zne = use_zne
        self.use_pec = use_pec
        self.use_measurement_mitigation = use_measurement_mitigation

        self.zne = ZeroNoiseExtrapolation() if use_zne else None
        self.meas_mit = None  # Set up during calibration

    def estimate_overhead(
        self,
        circuit_depth: int,
        num_qubits: int
    ) -> dict:
        """
        Estimate the sampling overhead for error mitigation.
        """
        overhead = 1

        if self.use_zne:
            # ZNE overhead: number of noise levels
            zne_overhead = len(self.zne.noise_factors)
            overhead *= zne_overhead

        if self.use_pec:
            # PEC overhead: exponential in circuit depth
            # Assume gamma_per_gate ~ 1.1 for low noise
            gamma_per_gate = 1.1
            pec_overhead = gamma_per_gate ** circuit_depth
            overhead *= pec_overhead ** 2  # Variance scaling

        return {
            'total_overhead': overhead,
            'recommended_shots': int(1000 * overhead),
            'feasible': overhead < 1000,
            'warning': 'Consider circuit simplification' if overhead > 100 else None
        }

    def run_mitigated_vqe(
        self,
        circuit_executor,
        circuit,
        hamiltonian,
        base_shots: int = 1000
    ) -> dict:
        """
        Run VQE with integrated error mitigation.
        """
        results = {}

        # 1. Measurement error mitigation (always first)
        if self.use_measurement_mitigation and self.meas_mit:
            # Apply to raw counts
            pass

        # 2. Zero-noise extrapolation
        if self.use_zne:
            def zne_executor(noise_factor):
                return circuit_executor(circuit, noise_factor, base_shots)

            zne_result = self.zne.extrapolate(zne_executor)
            results['zne'] = zne_result.extrapolated_value

        # 3. Combine results
        if 'zne' in results:
            results['final'] = results['zne']
        else:
            results['final'] = circuit_executor(circuit, 1.0, base_shots)

        return results
```
