## 1. The Problem

Given a function f: {0,1} → {0,1}, determine if f is **constant** (f(0)=f(1)) or **balanced** (f(0)≠f(1)).

<Definition term="Deutsch's Problem">
Classically requires 2 queries to f. Quantum requires only 1 query!
</Definition>

## 2. The Oracle

The oracle computes f in a reversible way:

<Formula>
$$U_f|x\rangle|y\rangle = |x\rangle|y \oplus f(x)\rangle$$
</Formula>

```python
import numpy as np

def make_oracle(f):
    """Create oracle for f: {0,1} -> {0,1}"""
    oracle = np.zeros((4, 4))
    for x in [0, 1]:
        for y in [0, 1]:
            input_idx = 2*x + y
            output_y = y ^ f(x)
            output_idx = 2*x + output_y
            oracle[output_idx, input_idx] = 1
    return oracle

# Example: f(x) = x (balanced)
f_balanced = lambda x: x
Uf = make_oracle(f_balanced)
print("Oracle for f(x)=x:")
print(Uf)
```

## 3. The Circuit

```python
import numpy as np

H = np.array([[1, 1], [1, -1]]) / np.sqrt(2)
I = np.eye(2)

def deutsch_algorithm(f):
    """Run Deutsch's algorithm on function f."""
    def make_oracle(f):
        oracle = np.zeros((4, 4))
        for x in [0, 1]:
            for y in [0, 1]:
                input_idx = 2*x + y
                output_y = y ^ f(x)
                output_idx = 2*x + output_y
                oracle[output_idx, input_idx] = 1
        return oracle
    
    Uf = make_oracle(f)
    state = np.array([0, 1, 0, 0])  # |01⟩
    
    HH = np.kron(H, H)
    state = HH @ state
    state = Uf @ state
    
    H_I = np.kron(H, I)
    state = H_I @ state
    
    prob_0 = np.abs(state[0])**2 + np.abs(state[1])**2
    return "constant" if prob_0 > 0.5 else "balanced"

# Test
functions = [
    (lambda x: 0, "f(x)=0", "constant"),
    (lambda x: 1, "f(x)=1", "constant"),
    (lambda x: x, "f(x)=x", "balanced"),
    (lambda x: 1-x, "f(x)=1-x", "balanced"),
]

for f, name, expected in functions:
    result = deutsch_algorithm(f)
    print(f"{name}: {result} {'✓' if result == expected else '✗'}")
```

## 4. Phase Kickback

<Formula>
$$U_f|x\rangle|-\rangle = (-1)^{f(x)}|x\rangle|-\rangle$$
</Formula>
