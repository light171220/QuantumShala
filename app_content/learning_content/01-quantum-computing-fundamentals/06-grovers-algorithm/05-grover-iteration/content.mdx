## 1. The Grover Operator

The Grover operator G combines oracle O and diffusion D:

<Formula>
$$G = D \cdot O = (2|s\rangle\langle s| - I) \cdot O_f$$
</Formula>

```python
import numpy as np

def grover_operator(oracle, n_qubits):
    """Create full Grover operator G = D · O"""
    N = 2**n_qubits
    s = np.ones(N) / np.sqrt(N)
    D = 2 * np.outer(s, s) - np.eye(N)
    return D @ oracle

# Example
n = 3
marked = 5
N = 2**n

oracle = np.eye(N)
oracle[marked, marked] = -1

G = grover_operator(oracle, n)
print("Grover operator G = D · O created")
print(f"Dimension: {G.shape}")
```

## 2. Complete Algorithm

```python
import numpy as np

def grovers_algorithm(n_qubits, marked_item):
    """
    Complete Grover's algorithm.
    Returns final state and number of iterations.
    """
    N = 2**n_qubits
    
    # Optimal number of iterations
    iterations = int(np.round(np.pi/4 * np.sqrt(N)))
    
    # Initial state: uniform superposition
    state = np.ones(N) / np.sqrt(N)
    
    # Oracle and diffusion
    oracle = np.eye(N)
    oracle[marked_item, marked_item] = -1
    
    s = np.ones(N) / np.sqrt(N)
    diffusion = 2 * np.outer(s, s) - np.eye(N)
    
    # Grover iterations
    for _ in range(iterations):
        state = oracle @ state      # Oracle
        state = diffusion @ state   # Diffusion
    
    return state, iterations

# Run
state, iters = grovers_algorithm(4, 7)
print(f"Iterations: {iters}")
print(f"Probability of finding 7: {np.abs(state[7])**2:.4f}")
```

## 3. Tracking State Evolution

```python
import numpy as np

def track_evolution(n_qubits, marked_item, max_iter=10):
    """Track how amplitude evolves."""
    N = 2**n_qubits
    
    state = np.ones(N) / np.sqrt(N)
    
    oracle = np.eye(N)
    oracle[marked_item, marked_item] = -1
    
    s = np.ones(N) / np.sqrt(N)
    D = 2 * np.outer(s, s) - np.eye(N)
    
    print(f"{'Iter':>4} {'Amplitude':>10} {'Probability':>12}")
    print("-" * 30)
    
    for i in range(max_iter + 1):
        amp = state[marked_item]
        prob = np.abs(amp)**2
        print(f"{i:>4} {amp:>10.4f} {prob:>12.4f}")
        
        if i < max_iter:
            state = oracle @ state
            state = D @ state

track_evolution(4, 7)
```

## Summary

- G = D · O (diffusion after oracle)
- Repeat ~(π/4)√N times
- Each iteration amplifies marked state
- Probability approaches 1 at optimal iteration
