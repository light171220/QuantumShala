## 1. Cryptographic Applications

Grover's algorithm has profound implications for symmetric cryptography.

<Definition term="Grover's Cryptographic Impact">
A symmetric cipher with n-bit keys has only n/2 bits of effective security against a quantum adversary using Grover's algorithm.
</Definition>

```python
import numpy as np

def database_oracle_example():
    """
    Construct an oracle for database search.
    The oracle checks if record x matches our criteria.
    """
    # Simulated database: records are just integers
    # Search criterion: find records divisible by 7

    n_qubits = 4

    print(f"Criterion: x divisible by 7 (and x > 0)")
```

## 3. SAT Solving with Grover's Algorithm

Satisfiability (SAT) is a fundamental problem in computer science.

### SAT Problem Overview

<Definition term="Boolean Satisfiability (SAT)">
Given a Boolean formula in CNF (Conjunctive Normal Form), find an assignment of variables that makes the formula TRUE, or prove none exists.
</Definition>

```python
import numpy as np

def sat_example():
    """Example SAT problem and oracle construction."""
    print("SAT Problem Example")
    print("=" * 50)

    # Formula: (x0 OR x1) AND (NOT x1 OR x2) AND (NOT x0 OR NOT x2)
    print("Formula: (x0 v x1) ^ (~x1 v x2) ^ (~x0 v ~x2)")
    print()
    print("Variables: x0, x1, x2 (3 bits = 8 possible assignments)")
    print()

    def evaluate_formula(x0, x1, x2):
        """Evaluate the SAT formula."""
        clause1 = x0 or x1
        clause2 = (not x1) or x2
        clause3 = (not x0) or (not x2)
        return clause1 and clause2 and clause3

    print("Truth table:")
    print(f"{'x2':>3} {'x1':>3} {'x0':>3} | {'C1':>3} {'C2':>3} {'C3':>3} | {'SAT':>4}")
    print("-" * 35)

    solutions = []
    for x in range(8):
        x0 = (x >> 0) & 1
        x1 = (x >> 1) & 1
        x2 = (x >> 2) & 1

        c1 = int(x0 or x1)
        c2 = int((not x1) or x2)
        c3 = int((not x0) or (not x2))
        sat = evaluate_formula(x0, x1, x2)

        if sat:
            solutions.append(x)

        print(f"{x2:>3} {x1:>3} {x0:>3} | {c1:>3} {c2:>3} {c3:>3} | {'YES' if sat else 'no':>4}")

    print(f"\nSolutions: {solutions} ({len(solutions)} total)")

    return solutions

solutions = sat_example()
```

### Grover for SAT

```python
import numpy as np

def grover_sat(n_vars, clause_function, verbose=True):
    """
    Use Grover's algorithm to solve a SAT problem.

    Args:
        n_vars: Number of Boolean variables
        clause_function: Function that returns True if assignment satisfies formula
        verbose: Print progress

    Returns:
        Solutions found and their probabilities
    """
    N = 2**n_vars

    # Build oracle from SAT formula
    oracle = np.eye(N)
    solutions = []

    for x in range(N):
        # Convert integer to bit assignment
        bits = [(x >> i) & 1 for i in range(n_vars)]
        if clause_function(*bits):
            oracle[x, x] = -1
            solutions.append(x)

    k = len(solutions)

    if k == 0:
        if verbose:
            print("No solutions exist (UNSAT)")
        return [], None

    # Calculate optimal iterations
    theta = np.arcsin(np.sqrt(k/N))
    r_opt = max(0, int(round((np.pi/2 - theta) / (2*theta))))

    # Run Grover's
    state = np.ones(N) / np.sqrt(N)
    s = np.ones(N) / np.sqrt(N)
    diffusion = 2 * np.outer(s, s) - np.eye(N)

    for _ in range(r_opt):
        state = diffusion @ (oracle @ state)

    probs = np.abs(state)**2

    if verbose:
        print(f"SAT Solutions Found via Grover's Algorithm")
        print(f"Variables: {n_vars}, Search space: {N}")
        print(f"Solutions: {k}, Iterations: {r_opt}")
        print()
        print("High-probability assignments:")
        for x in range(N):
            if probs[x] > 0.01:
                bits = [(x >> i) & 1 for i in range(n_vars)]
                is_sol = "SOLUTION" if x in solutions else ""
                print(f"  {x}: {bits} prob={probs[x]:.4f} {is_sol}")

    return solutions, probs

# Test with our example formula
def example_sat(x0, x1, x2):
    c1 = x0 or x1
    c2 = (not x1) or x2
    c3 = (not x0) or (not x2)
    return c1 and c2 and c3
```

### Minimum Finding

```python
import numpy as np

def minimum_finding():
    """
    Use Grover's for minimum finding.
    Algorithm by Durr and Hoyer (1996).
    """
    print("Durr-Hoyer Algorithm: Quantum Minimum Finding")
    print("=" * 55)

    print("""
Algorithm:
1. Pick random index y, set threshold T = f(y)
2. Use Grover to find x with f(x) < T
3. If found: y = x, T = f(x), repeat
4. If not found: y is likely the minimum

Complexity: O(sqrt(N)) expected queries

Key insight: Each iteration has expected sqrt(N/k) cost
where k = number of elements below threshold
    """)

    # Simulation
    N = 64
    values = np.random.permutation(N)  # Random permutation

    def simulate_durr_hoyer(values, verbose=True):
        N = len(values)
        y = np.random.randint(N)
        T = values[y]
        iterations = 0

        if verbose:
            print(f"\nSimulating Durr-Hoyer on {N} elements:")
            print(f"Initial: y={y}, T=f(y)={T}")

        while True:
            # Count elements below threshold
            below = np.sum(values < T)
            if below == 0:
                break

            # Grover would find one in sqrt(N/below) expected
            grover_cost = np.sqrt(N / below)
            iterations += int(grover_cost)

            # Simulate finding element below threshold
            candidates = np.where(values < T)[0]
            y = np.random.choice(candidates)
            T = values[y]

            if verbose:
                print(f"Found y={y}, T={T}, below={below}, cost={grover_cost:.1f}")

        if verbose:
            print(f"\nResult: minimum at index {y}, value {T}")
            print(f"Actual minimum: {np.min(values)}")
            print(f"Total iterations: {iterations} (classical would need ~{N})")

        return y, T, iterations

    np.random.seed(42)
    simulate_durr_hoyer(values)

minimum_finding()
```

## 6. Implementation Summary

```python
import numpy as np

def full_grover_demo(n_qubits, oracle_function, name="Search"):
    """
    Complete Grover implementation with analysis.

    Args:
        n_qubits: Number of qubits
        oracle_function: f(x) -> True if x is solution
        name: Name for display

    Returns:
        Dictionary with results
    """
    N = 2**n_qubits

    # Build oracle and find solutions
    oracle = np.eye(N)
    solutions = []
    for x in range(N):
        if oracle_function(x):
            oracle[x, x] = -1
            solutions.append(x)

    k = len(solutions)

    if k == 0:
        print(f"{name}: No solutions found")
        return None

    # Calculate optimal iterations
    theta = np.arcsin(np.sqrt(k/N))
    r_opt = max(0, int(round((np.pi/2 - theta) / (2*theta))))

    # Run algorithm
    state = np.ones(N) / np.sqrt(N)
    s = np.ones(N) / np.sqrt(N)
    diffusion = 2 * np.outer(s, s) - np.eye(N)

    for _ in range(r_opt):
        state = diffusion @ (oracle @ state)

    probs = np.abs(state)**2
    total_prob = sum(probs[x] for x in solutions)

    # Display results
    print(f"\n{'='*50}")
    print(f"Grover's Algorithm: {name}")
    print(f"{'='*50}")
    print(f"Search space: N = 2^{n_qubits} = {N}")
    print(f"Solutions found: k = {k}")
    print(f"Optimal iterations: r = {r_opt}")
    print(f"Success probability: {total_prob:.4f}")
    print(f"Classical queries needed: {N}")
    print(f"Quantum queries needed: {r_opt}")
    print(f"Speedup: {N // max(1, r_opt):.1f}x")

    return {
        'solutions': solutions,
        'iterations': r_opt,
        'success_prob': total_prob,
        'speedup': N / max(1, r_opt)
    }
```
