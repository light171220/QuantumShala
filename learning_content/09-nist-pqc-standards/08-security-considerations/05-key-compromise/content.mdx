# Key Compromise Scenarios in PQC

Key compromise is inevitable given enough time. Understanding compromise scenarios and having recovery procedures is essential for maintaining security.

## Impact by Key Type

```python
from dataclasses import dataclass

                f"  Recovery: {self.recovery_difficulty}\n"

# Define impact for each key type
```

## Forward Secrecy Considerations

```
Forward Secrecy in PQC:
═══════════════════════════════════════════════════════════════════

Without Forward Secrecy:
┌─────────────────────────────────────────────────────────────────┐
│  Long-term key K used for all sessions                          │
│                                                                  │
│  Time ─────────────────────────────────────────────────────▶    │
│                                                                  │
│  Session 1 ────┬──── Session 2 ────┬──── Session 3              │
│  (uses K)      │     (uses K)      │     (uses K)               │
│                │                   │                             │
│                │   K compromised ──┼──▶ ALL sessions exposed!   │
└─────────────────────────────────────────────────────────────────┘

With Forward Secrecy (Ephemeral Keys):
┌─────────────────────────────────────────────────────────────────┐
│  Each session uses fresh ephemeral key                          │
│                                                                  │
│  Time ─────────────────────────────────────────────────────▶    │
│                                                                  │
│  Session 1 ────┬──── Session 2 ────┬──── Session 3              │
│  (K1, deleted) │   (K2, deleted)   │   (K3, deleted)            │
│                │                   │                             │
│  Long-term compromised ───────────▶ Past sessions SAFE!         │
└─────────────────────────────────────────────────────────────────┘

PQC Forward Secrecy Implementation:
├─ Generate fresh ML-KEM key pair per session
├─ Derive session keys from ephemeral KEM
├─ Securely delete ephemeral private key after use
└─ Long-term keys used only for authentication, not encryption
```

```python
from typing import Tuple, Optional
from dataclasses import dataclass
import os
import hashlib

@dataclass
class Session:
    """Represents a communication session."""
    session_id: str
    ephemeral_pk: bytes
    ephemeral_sk: Optional[bytes]  # None after key deletion
    session_key: Optional[bytes]
    is_active: bool
    forward_secure: bool

class ForwardSecureProtocol:
    """
    Forward-secure key exchange protocol.

    Demonstrates proper ephemeral key handling for PQC.
    """

    def __init__(self, long_term_signing_key: bytes):
        self.signing_key = long_term_signing_key
        self.sessions: dict = {}

    def initiate_session(self, peer_id: str) -> Tuple[str, bytes, bytes]:
        """
        Initiate a forward-secure session.

        Returns (session_id, ephemeral_pk, signed_pk).
        """
        # Generate ephemeral ML-KEM key pair
        session_id = os.urandom(16).hex()
        ephemeral_pk, ephemeral_sk = self._generate_kem_keypair()

        # Sign ephemeral public key with long-term key
        signature = self._sign(ephemeral_pk)

        # Store session (sk stored only temporarily)
        self.sessions[session_id] = Session(
            session_id=session_id,
            ephemeral_pk=ephemeral_pk,
            ephemeral_sk=ephemeral_sk,
            session_key=None,
            is_active=True,
            forward_secure=True
        )

        return session_id, ephemeral_pk, signature

    def complete_session(
        self,
        session_id: str,
        peer_ciphertext: bytes
    ) -> bytes:
        """
        Complete session by decapsulating and deriving session key.

        CRITICAL: Ephemeral secret key is deleted after use.
        """
        session = self.sessions.get(session_id)
        if not session or not session.ephemeral_sk:
            raise ValueError("Invalid or expired session")

        # Decapsulate using ephemeral secret key
        shared_secret = self._decapsulate(
            session.ephemeral_sk,
            peer_ciphertext
        )

        # Derive session key
        session_key = hashlib.sha3_256(
            b"session_key" + shared_secret
        ).digest()

        # CRITICAL: Securely delete ephemeral secret key
        self._secure_delete_key(session.ephemeral_sk)
        session.ephemeral_sk = None  # Mark as deleted
        session.session_key = session_key

        return session_key

    def _secure_delete_key(self, key: bytes):
        """
        Securely delete key material.

        In real implementation, use secure memory wiping.
        """
        # Overwrite with random data multiple times
        if isinstance(key, bytearray):
            for _ in range(3):
                for i in range(len(key)):
                    key[i] = os.urandom(1)[0]
            for i in range(len(key)):
                key[i] = 0

    def _generate_kem_keypair(self) -> Tuple[bytes, bytes]:
        """Generate ephemeral KEM key pair (mock)."""
        pk = os.urandom(1568)  # ML-KEM-1024 public key size
        sk = bytearray(os.urandom(3168))  # Mutable for secure deletion
        return pk, sk

    def _decapsulate(self, sk: bytes, ct: bytes) -> bytes:
        """Mock decapsulation."""
        return hashlib.sha3_256(sk + ct).digest()

    def _sign(self, message: bytes) -> bytes:
        """Mock signing."""
        return hashlib.sha3_256(self.signing_key + message).digest()

    def get_session_status(self, session_id: str) -> dict:
        """Get session security status."""
        session = self.sessions.get(session_id)
        if not session:
            return {"error": "Session not found"}

        return {
            "session_id": session_id,
            "forward_secure": session.forward_secure,
            "ephemeral_sk_deleted": session.ephemeral_sk is None,
            "has_session_key": session.session_key is not None,
            "is_active": session.is_active
        }
```

## Key Revocation

```
Key Revocation Mechanisms:
═══════════════════════════════════════════════════════════════════

Certificate Revocation List (CRL):
┌─────────────────────────────────────────────────────────────────┐
│  CA publishes list of revoked certificates                      │
│                                                                  │
│  ┌────────────────────────────────────────┐                     │
│  │ CRL                                    │                     │
│  ├────────────────────────────────────────┤                     │
│  │ Serial: 1234  Revoked: 2024-01-15     │                     │
│  │ Serial: 5678  Revoked: 2024-01-16     │                     │
│  │ Serial: 9012  Revoked: 2024-01-17     │                     │
│  └────────────────────────────────────────┘                     │
│                                                                  │
│  Pros: Simple, widely supported                                 │
│  Cons: Can grow large, update latency                           │
└─────────────────────────────────────────────────────────────────┘

Online Certificate Status Protocol (OCSP):
┌─────────────────────────────────────────────────────────────────┐
│  Real-time certificate status queries                           │
│                                                                  │
│  Client ─── "Is cert X valid?" ───▶ OCSP Responder             │
│         ◀── "Valid/Revoked/Unknown" ────                        │
│                                                                  │
│  Pros: Real-time, smaller responses                             │
│  Cons: Privacy (responder sees all queries), availability       │
└─────────────────────────────────────────────────────────────────┘

OCSP Stapling:
┌─────────────────────────────────────────────────────────────────┐
│  Server fetches and caches OCSP response                        │
│                                                                  │
│  Server ◀── OCSP Response ─── OCSP Responder                   │
│         │                                                        │
│         └── TLS + Stapled OCSP ───▶ Client                      │
│                                                                  │
│  Pros: Privacy preserved, reduced latency                       │
│  Cons: Server must refresh responses                            │
└─────────────────────────────────────────────────────────────────┘
```

```python
from dataclasses import dataclass, field
from datetime import datetime, timedelta
from typing import Dict, List, Optional, Set
from enum import Enum

class RevocationReason(Enum):
    """Standard revocation reasons (RFC 5280)."""
    UNSPECIFIED = 0
    KEY_COMPROMISE = 1
    CA_COMPROMISE = 2
    AFFILIATION_CHANGED = 3
    SUPERSEDED = 4
    CESSATION_OF_OPERATION = 5
    CERTIFICATE_HOLD = 6
    REMOVE_FROM_CRL = 8
    PRIVILEGE_WITHDRAWN = 9
    AA_COMPROMISE = 10

@dataclass
class RevocationEntry:
    """Entry in a revocation list."""
    serial_number: str
    revocation_date: datetime
    reason: RevocationReason
    invalidity_date: Optional[datetime] = None  # When compromise occurred

@dataclass
class CRL:
    """Certificate Revocation List."""
    issuer: str
    this_update: datetime
    next_update: datetime
    entries: List[RevocationEntry] = field(default_factory=list)

    def is_revoked(self, serial: str) -> tuple:
        """Check if certificate is revoked."""
        for entry in self.entries:
            if entry.serial_number == serial:
                return True, entry
        return False, None

    def add_revocation(
        self,
        serial: str,
        reason: RevocationReason,
        invalidity_date: Optional[datetime] = None
    ):
        """Add certificate to revocation list."""
        entry = RevocationEntry(
            serial_number=serial,
            revocation_date=datetime.now(),
            reason=reason,
            invalidity_date=invalidity_date
        )
        self.entries.append(entry)

class RevocationManager:
    """
    Manage key and certificate revocation.
    """

    def __init__(self):
        self.crls: Dict[str, CRL] = {}
        self.revoked_keys: Set[str] = set()
        self.ocsp_cache: Dict[str, tuple] = {}

    def revoke_key(
        self,
        key_id: str,
        reason: RevocationReason,
        compromise_time: Optional[datetime] = None
    ) -> dict:
        """
        Revoke a key and return impact assessment.
        """
        self.revoked_keys.add(key_id)

        # Assess impact
        impact = self._assess_revocation_impact(
            key_id, reason, compromise_time
        )

        # Generate revocation notification
        notification = self._generate_notification(
            key_id, reason, impact
        )

        return {
            "key_id": key_id,
            "revoked_at": datetime.now().isoformat(),
            "reason": reason.name,
            "impact": impact,
            "notification": notification
        }

    def _assess_revocation_impact(
        self,
        key_id: str,
        reason: RevocationReason,
        compromise_time: Optional[datetime]
    ) -> dict:
        """Assess the impact of key revocation."""
        impact = {
            "affected_signatures": [],
            "affected_sessions": [],
            "requires_rekey": False,
            "historical_data_risk": False
        }

        if reason == RevocationReason.KEY_COMPROMISE:
            impact["requires_rekey"] = True
            if compromise_time:
                # All signatures after compromise are suspect
                impact["historical_data_risk"] = True
                impact["suspect_window"] = {
                    "start": compromise_time.isoformat(),
                    "end": datetime.now().isoformat()
                }

        return impact

    def _generate_notification(
        self,
        key_id: str,
        reason: RevocationReason,
        impact: dict
    ) -> str:
        """Generate revocation notification."""
        lines = [
            "KEY REVOCATION NOTICE",
            "=" * 40,
            f"Key ID: {key_id}",
            f"Reason: {reason.name}",
            f"Time: {datetime.now().isoformat()}",
            "",
            "REQUIRED ACTIONS:",
        ]

        if impact["requires_rekey"]:
            lines.append("- Generate new key pair immediately")
            lines.append("- Update all systems using this key")

        if impact["historical_data_risk"]:
            lines.append("- Review all signatures in suspect window")
            lines.append("- Notify affected parties")

        return "\n".join(lines)

    def is_key_valid(self, key_id: str) -> bool:
        """Check if key is currently valid (not revoked)."""
        return key_id not in self.revoked_keys
```

## Compromise Recovery Procedures

```python
from dataclasses import dataclass
from typing import List, Callable
from enum import Enum
from datetime import datetime

class RecoveryPhase(Enum):
    DETECTION = 1
    CONTAINMENT = 2
    ERADICATION = 3
    RECOVERY = 4
    LESSONS_LEARNED = 5

@dataclass
class RecoveryStep:
    """Individual recovery action."""
    phase: RecoveryPhase
    action: str
    responsible: str
    deadline_hours: int
    completed: bool = False

class CompromiseRecoveryPlan:
    """
    Incident response plan for key compromise.
    """

    def __init__(self, incident_id: str, key_type: str):
        self.incident_id = incident_id
        self.key_type = key_type
        self.start_time = datetime.now()
        self.steps: List[RecoveryStep] = []
        self._build_plan()

    def _build_plan(self):
        """Build recovery plan based on key type."""

        # Detection phase
        self.steps.extend([
            RecoveryStep(
                RecoveryPhase.DETECTION,
                "Confirm compromise through forensic analysis",
                "Security Team",
                2
            ),
            RecoveryStep(
                RecoveryPhase.DETECTION,
                "Determine scope: which keys, what access",
                "Security Team",
                4
            ),
            RecoveryStep(
                RecoveryPhase.DETECTION,
                "Identify compromise timeline",
                "Security Team",
                4
            ),
        ])

        # Containment phase
        self.steps.extend([
            RecoveryStep(
                RecoveryPhase.CONTAINMENT,
                "Revoke compromised key immediately",
                "PKI Admin",
                1
            ),
            RecoveryStep(
                RecoveryPhase.CONTAINMENT,
                "Publish emergency CRL/OCSP update",
                "PKI Admin",
                1
            ),
            RecoveryStep(
                RecoveryPhase.CONTAINMENT,
                "Notify affected parties",
                "Communications",
                2
            ),
            RecoveryStep(
                RecoveryPhase.CONTAINMENT,
                "Isolate affected systems",
                "IT Operations",
                2
            ),
        ])

        # Eradication phase
        self.steps.extend([
            RecoveryStep(
                RecoveryPhase.ERADICATION,
                "Remove compromised key from all systems",
                "IT Operations",
                8
            ),
            RecoveryStep(
                RecoveryPhase.ERADICATION,
                "Audit systems for unauthorized access",
                "Security Team",
                24
            ),
            RecoveryStep(
                RecoveryPhase.ERADICATION,
                "Patch vulnerability that led to compromise",
                "Development",
                48
            ),
        ])

        # Recovery phase
        self.steps.extend([
            RecoveryStep(
                RecoveryPhase.RECOVERY,
                "Generate new key pair with secure procedure",
                "Security Team",
                4
            ),
            RecoveryStep(
                RecoveryPhase.RECOVERY,
                "Issue new certificate",
                "PKI Admin",
                2
            ),
            RecoveryStep(
                RecoveryPhase.RECOVERY,
                "Deploy new key to all systems",
                "IT Operations",
                24
            ),
            RecoveryStep(
                RecoveryPhase.RECOVERY,
                "Verify system functionality",
                "QA",
                8
            ),
            RecoveryStep(
                RecoveryPhase.RECOVERY,
                "Resign affected code/documents if needed",
                "Development",
                48
            ),
        ])

        # Lessons learned
        self.steps.extend([
            RecoveryStep(
                RecoveryPhase.LESSONS_LEARNED,
                "Document incident timeline",
                "Security Team",
                72
            ),
            RecoveryStep(
                RecoveryPhase.LESSONS_LEARNED,
                "Identify root cause",
                "Security Team",
                72
            ),
            RecoveryStep(
                RecoveryPhase.LESSONS_LEARNED,
                "Implement preventive measures",
                "Development",
                168
            ),
            RecoveryStep(
                RecoveryPhase.LESSONS_LEARNED,
                "Update security policies",
                "Security Team",
                168
            ),
        ])

    def get_current_phase(self) -> RecoveryPhase:
        """Get current recovery phase."""
        for step in self.steps:
            if not step.completed:
                return step.phase
        return RecoveryPhase.LESSONS_LEARNED

    def get_pending_steps(self) -> List[RecoveryStep]:
        """Get all pending steps."""
        return [s for s in self.steps if not s.completed]

    def complete_step(self, step_index: int):
        """Mark step as completed."""
        if 0 <= step_index < len(self.steps):
            self.steps[step_index].completed = True

    def generate_status_report(self) -> str:
        """Generate current status report."""
        current_phase = self.get_current_phase()
        completed = sum(1 for s in self.steps if s.completed)
        total = len(self.steps)

        lines = [
            f"INCIDENT RECOVERY STATUS: {self.incident_id}",
            "=" * 50,
            f"Key Type: {self.key_type}",
            f"Started: {self.start_time.isoformat()}",
            f"Current Phase: {current_phase.name}",
            f"Progress: {completed}/{total} steps completed",
            "",
            "PENDING ACTIONS:",
        ]

        for step in self.get_pending_steps()[:5]:
            lines.append(
                f"  [{step.phase.name}] {step.action} "
                f"({step.responsible}, {step.deadline_hours}h)"
            )

        return "\n".join(lines)

# Example usage
```

## Summary

Key compromise management requires:

1. **Impact assessment**: Understanding scope based on key type and usage
2. **Forward secrecy**: Using ephemeral keys to limit compromise impact
3. **Revocation mechanisms**: CRL, OCSP, and immediate notification
4. **Recovery procedures**: Structured incident response with clear steps

Preparation before compromise occurs dramatically improves recovery outcomes.
