# Fault Injection Attacks on PQC

Fault injection attacks actively induce errors in cryptographic computations to extract secret information. PQC algorithms have unique fault vulnerabilities.

## Differential Fault Analysis (DFA)

```python
from typing import List, Tuple, Optional
from dataclasses import dataclass
import random

@dataclass
class FaultedOutput:
    """Output from a faulted computation."""
    correct: bytes
    faulted: bytes
    fault_location: str
    fault_type: str

class DifferentialFaultAnalysis:
    """
    Differential Fault Analysis framework.

    DFA compares correct and faulted outputs to extract
    information about secret values.
    """

    def __init__(self):
        self.fault_pairs: List[FaultedOutput] = []

    def analyze_difference(
        self,
        correct: List[int],
        faulted: List[int]
    ) -> dict:
        """
        Analyze difference between correct and faulted outputs.

        Returns information about the fault and potential key leakage.
        """
        if len(correct) != len(faulted):
            return {"error": "Length mismatch"}

        differences = []
        for i, (c, f) in enumerate(zip(correct, faulted)):
            if c != f:
                diff = c ^ f  # XOR shows which bits changed
                differences.append({
                    "position": i,
                    "correct": c,
                    "faulted": f,
                    "xor_diff": diff,
                    "bit_flips": bin(diff).count('1')
                })

        return {
            "total_differences": len(differences),
            "differences": differences,
            "fault_pattern": self._classify_fault(differences)
        }

    def _classify_fault(self, diffs: List[dict]) -> str:
        """Classify the type of fault based on differences."""
        if len(diffs) == 0:
            return "no_fault"
        elif len(diffs) == 1:
            if diffs[0]["bit_flips"] == 1:
                return "single_bit_flip"
            return "single_byte_fault"
        elif all(d["bit_flips"] == 1 for d in diffs):
            return "multiple_single_bit_flips"
        return "random_fault"

        Simulate a fault injection attempt.

        Returns (success, leaked_info).
```

```python
from typing import List, Tuple, Callable, Any
import hashlib
import os

class RedundantComputation:
    """
    Redundant computation for fault detection.

    Compute operations multiple times and compare results.
    """

    def __init__(self, redundancy: int = 2):
        self.redundancy = redundancy

    def compute_with_check(
        self,
        func: Callable,
        *args,
        **kwargs
    ) -> Tuple[Any, bool]:
        """
        Execute function with redundancy check.

        Returns (result, verified).
        """
        results = []
        for _ in range(self.redundancy):
            result = func(*args, **kwargs)
            results.append(result)

        # Check all results match
        if all(r == results[0] for r in results):
            return results[0], True
        else:
            # Fault detected!
            return None, False

    def triple_modular_redundancy(
        self,
        func: Callable,
        *args,
        **kwargs
    ) -> Tuple[Any, bool]:
        """
        Triple Modular Redundancy (TMR).

        Execute three times and take majority vote.
        """
        results = [func(*args, **kwargs) for _ in range(3)]

        # Count occurrences
        from collections import Counter
        counts = Counter(map(str, results))
        majority = counts.most_common(1)[0]

        if majority[1] >= 2:
            # Have majority - return it
            return results[0] if str(results[0]) == majority[0] else results[1], True
        else:
            # No majority - all different
            return None, False

class AlgorithmicCountermeasures:
    """
    Algorithm-level fault countermeasures for PQC.
    """

    @staticmethod
    def kem_decapsulation_protection(
        decrypt_fn: Callable,
        reencrypt_fn: Callable,
        sk: bytes,
        pk: bytes,
        ct: bytes,
        implicit_rejection_seed: bytes
    ) -> Tuple[bytes, bool]:
        """
        Fault-resistant KEM decapsulation.

        Uses multiple checks and implicit rejection.
        """
        # First decryption
        m1 = decrypt_fn(sk, ct)

        # Redundant decryption
        m2 = decrypt_fn(sk, ct)

        # Check consistency
        if m1 != m2:
            # Fault detected in decryption
            return hashlib.sha3_256(
                implicit_rejection_seed + ct
            ).digest(), False

        # Derive key and re-encryption randomness
        g_output = hashlib.sha3_512(m1).digest()
        K1 = g_output[:32]
        r = g_output[32:]

        # Re-encrypt to verify
        ct_check = reencrypt_fn(pk, m1, r)

        # Redundant re-encryption
        ct_check2 = reencrypt_fn(pk, m1, r)

        # Check re-encryption consistency
        if ct_check != ct_check2:
            return hashlib.sha3_256(
                implicit_rejection_seed + ct
            ).digest(), False

        # Constant-time comparison of ciphertexts
        valid = ct == ct_check

        # Return appropriate key (implicit rejection if invalid)
        if valid:
            return K1, True
        else:
            return hashlib.sha3_256(
                implicit_rejection_seed + ct
            ).digest(), True  # Still valid from protocol perspective

    @staticmethod
    def signature_verification_protection(
        verify_fn: Callable,
        pk: bytes,
        msg: bytes,
        sig: bytes
    ) -> Tuple[bool, bool]:
        """
        Fault-resistant signature verification.

        Returns (valid, no_fault_detected).
        """
        # Multiple verification rounds
        results = []
        for _ in range(3):
            try:
                result = verify_fn(pk, msg, sig)
                results.append(result)
            except Exception:
                results.append(None)

        # Check consistency
        valid_results = [r for r in results if r is not None]

        if len(valid_results) < 2:
            # Too many faults
            return False, False

        # Take majority
        if sum(valid_results) >= 2:
            return True, True
        elif sum(r == False for r in valid_results) >= 2:
            return False, True
        else:
            # Inconsistent - possible fault
            return False, False

class PolynomialChecksums:
    """
    Checksum verification for polynomial operations.

    Uses algebraic checksums to detect faults in
    polynomial arithmetic.
    """

    def __init__(self, q: int):
        self.q = q
        # Random checksum coefficient (fixed at init)
        self.alpha = int.from_bytes(os.urandom(4), 'little') % q

    def compute_checksum(self, poly: List[int]) -> int:
        """
        Compute algebraic checksum of polynomial.

        checksum = sum(alpha^i * poly[i]) mod q
        """
        result = 0
        power = 1
        for coeff in poly:
            result = (result + power * coeff) % self.q
            power = (power * self.alpha) % self.q
        return result

    def verify_addition(
        self,
        a: List[int],
        b: List[int],
        result: List[int]
    ) -> bool:
        """
        Verify polynomial addition using checksums.

        Check: checksum(a) + checksum(b) == checksum(result)
        """
        cs_a = self.compute_checksum(a)
        cs_b = self.compute_checksum(b)
        cs_result = self.compute_checksum(result)

        expected = (cs_a + cs_b) % self.q
        return cs_result == expected

    def verify_ntt(
        self,
        input_poly: List[int],
        output_poly: List[int],
        ntt_fn: Callable
    ) -> bool:
        """
        Verify NTT computation using evaluation check.

        The checksum should transform consistently.
        """
        # Compute expected checksum transformation
        # For NTT, this involves the specific structure
        # Simplified check: verify inverse NTT recovers input
        recovered = ntt_fn(output_poly, inverse=True)

        # Compare with original
        return all(
            input_poly[i] == recovered[i]
            for i in range(len(input_poly))
        )

class FaultResponseHandler:
    """
    Handle fault detection events securely.
    """

    def __init__(self):
        self.fault_count = 0
        self.fault_threshold = 3

    def on_fault_detected(self, context: str):
        """
        Handle a detected fault.

        Secure response to fault detection.
        """
        self.fault_count += 1

        # Log the event (secure audit log)
        print(f"SECURITY: Fault detected in {context}")
        print(f"  Fault count: {self.fault_count}")

        if self.fault_count >= self.fault_threshold:
            self._lockdown()

    def _lockdown(self):
        """
        Initiate security lockdown.

        In practice, this would:
        - Zeroize all secret keys
        - Disable cryptographic operations
        - Trigger security alert
        """
        print("SECURITY: Fault threshold exceeded!")
        print("  Initiating lockdown...")
        print("  - Secrets zeroized")
        print("  - Operations disabled")
        print("  - Alert triggered")

    def safe_operation(
        self,
        operation: Callable,
        context: str,
        *args,
        **kwargs
    ):
        """
        Execute operation with fault detection wrapper.
        """
        redundant = RedundantComputation(redundancy=2)
        result, verified = redundant.compute_with_check(
            operation, *args, **kwargs
        )

        if not verified:
            self.on_fault_detected(context)
            raise SecurityError(f"Fault detected in {context}")

        return result

class SecurityError(Exception):
    """Raised when a security violation is detected."""
    pass
```

## Summary

Fault injection poses serious threats to PQC implementations:

1. **Attack techniques**: Voltage glitching, clock manipulation, EM pulses, laser injection
2. **PQC vulnerabilities**: Validity check bypass, sampling manipulation, NTT faults
3. **Software countermeasures**: Redundant computation, checksums, TMR
4. **Hardware countermeasures**: Voltage/clock monitoring, shield mesh, sensors

Defense requires multiple layers combining detection, prevention, and secure response.
