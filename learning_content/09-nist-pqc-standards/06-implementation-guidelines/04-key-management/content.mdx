# PQC Key Management

Post-quantum cryptographic keys are significantly larger than classical keys, creating unique challenges for key management. This lesson covers best practices for the entire key lifecycle.

```python
Secure encrypted key storage implementation
"""

import os
import json
import hashlib
from base64 import b64encode, b64decode
from typing import Optional
from datetime import datetime

# This is simplified for illustration

class EncryptedKeyStore:
    """
    Encrypted key storage using a master key.
    """

    def __init__(self, master_key: bytes, storage_path: str):
        """
        Initialize key store.

        Args:
            master_key: 32-byte master encryption key
            storage_path: Path to key storage file
        """
        if len(master_key) != 32:
            raise ValueError("Master key must be 32 bytes")

        self.master_key = master_key
        self.storage_path = storage_path
        self._keys = {}
        self._load()

    def _derive_key(self, key_id: str) -> bytes:
        """Derive encryption key for specific key ID."""
        return hashlib.sha256(
            self.master_key + key_id.encode()
        ).digest()

    def _encrypt(self, data: bytes, key_id: str) -> bytes:
        """Encrypt data using derived key."""
        # In production, use AES-GCM or ChaCha20-Poly1305
        derived_key = self._derive_key(key_id)
        nonce = os.urandom(12)

        # Placeholder - use proper AEAD in production
        # encrypted = aead_encrypt(derived_key, nonce, data)
        encrypted = b"[encrypted data]"

        return nonce + encrypted

    def _decrypt(self, data: bytes, key_id: str) -> bytes:
        """Decrypt data using derived key."""
        derived_key = self._derive_key(key_id)
        nonce = data[:12]
        ciphertext = data[12:]

        # Placeholder - use proper AEAD in production
        # plaintext = aead_decrypt(derived_key, nonce, ciphertext)
        plaintext = b"[decrypted data]"

        return plaintext

    def store_key(self, key_id: str, private_key: bytes,
                  public_key: bytes, metadata: dict):
        """
        Store a key pair securely.
        """
        # Encrypt private key
        encrypted_sk = self._encrypt(private_key, key_id)

        self._keys[key_id] = {
            "public_key": b64encode(public_key).decode(),
            "encrypted_private_key": b64encode(encrypted_sk).decode(),
            "metadata": metadata,
            "stored_at": datetime.utcnow().isoformat()
        }

        self._save()
        print(f"Stored key: {key_id}")

    def get_private_key(self, key_id: str) -> Optional[bytes]:
        """
        Retrieve and decrypt private key.
        """
        if key_id not in self._keys:
            return None

        encrypted = b64decode(self._keys[key_id]["encrypted_private_key"])
        return self._decrypt(encrypted, key_id)

    def get_public_key(self, key_id: str) -> Optional[bytes]:
        """Retrieve public key."""
        if key_id not in self._keys:
            return None
        return b64decode(self._keys[key_id]["public_key"])

    def delete_key(self, key_id: str):
        """
        Securely delete a key.
        """
        if key_id in self._keys:
            # In production, also securely wipe from storage
            del self._keys[key_id]
            self._save()
            print(f"Deleted key: {key_id}")

    def _load(self):
        """Load keys from storage."""
        try:
            with open(self.storage_path, 'r') as f:
                self._keys = json.load(f)
        except FileNotFoundError:
            self._keys = {}

    def _save(self):
        """Save keys to storage."""
        with open(self.storage_path, 'w') as f:
            json.dump(self._keys, f, indent=2)
```

### Key Lifecycle Management

```python
"""
Key lifecycle management for PQC
"""

from enum import Enum
from dataclasses import dataclass
from datetime import datetime, timedelta
from typing import List, Optional

class KeyState(Enum):
    PENDING = "pending"          # Generated but not yet active
    ACTIVE = "active"            # Currently in use
    DEPRECATED = "deprecated"    # Being phased out
    COMPROMISED = "compromised"  # Suspected or confirmed compromise
    DESTROYED = "destroyed"      # Securely deleted

@dataclass
class KeyLifecycle:
    key_id: str
    algorithm: str
    state: KeyState
    created_at: datetime
    activated_at: Optional[datetime]
    deprecated_at: Optional[datetime]
    destroyed_at: Optional[datetime]
    max_lifetime_days: int
    max_operations: int
    current_operations: int

class KeyLifecycleManager:
    """Manage key states and transitions."""

    def __init__(self):
        self.keys: dict[str, KeyLifecycle] = {}

    def create_key(self, key_id: str, algorithm: str,
                   max_lifetime_days: int = 365,
                   max_operations: int = 10_000_000) -> KeyLifecycle:
        """Create a new key in pending state."""
        lifecycle = KeyLifecycle(
            key_id=key_id,
            algorithm=algorithm,
            state=KeyState.PENDING,
            created_at=datetime.utcnow(),
            activated_at=None,
            deprecated_at=None,
            destroyed_at=None,
            max_lifetime_days=max_lifetime_days,
            max_operations=max_operations,
            current_operations=0
        )
        self.keys[key_id] = lifecycle
        return lifecycle

    def activate_key(self, key_id: str):
        """Move key from pending to active."""
        key = self.keys.get(key_id)
        if not key or key.state != KeyState.PENDING:
            raise ValueError("Key not in pending state")

        key.state = KeyState.ACTIVE
        key.activated_at = datetime.utcnow()

    def record_operation(self, key_id: str):
        """Record a key operation."""
        key = self.keys.get(key_id)
        if not key or key.state != KeyState.ACTIVE:
            raise ValueError("Key not active")

        key.current_operations += 1

        # Check if rotation needed
        if key.current_operations >= key.max_operations:
            self._trigger_rotation(key_id, "max_operations_reached")

    def check_rotation_needed(self, key_id: str) -> List[str]:
        """Check if key needs rotation and return reasons."""
        key = self.keys.get(key_id)
        if not key or key.state != KeyState.ACTIVE:
            return []

        reasons = []

        # Check age
        if key.activated_at:
            age = datetime.utcnow() - key.activated_at
            if age > timedelta(days=key.max_lifetime_days):
                reasons.append(f"Key age ({age.days} days) exceeds limit")

        # Check usage
        if key.current_operations >= key.max_operations * 0.9:
            reasons.append(f"Operations ({key.current_operations}) near limit")

        return reasons

    def deprecate_key(self, key_id: str, reason: str):
        """Mark key as deprecated."""
        key = self.keys.get(key_id)
        if not key or key.state not in [KeyState.ACTIVE, KeyState.PENDING]:
            raise ValueError("Key cannot be deprecated")

        key.state = KeyState.DEPRECATED
        key.deprecated_at = datetime.utcnow()
        print(f"Key {key_id} deprecated: {reason}")
```

## Summary

PQC key management requires attention to:

1. **Size**: Larger keys impact storage, transmission, and processing
2. **Generation**: Use proper entropy, follow checklist
3. **Storage**: Choose appropriate solution (HSM, TPM, software, cloud)
4. **Rotation**: Plan for time-based, usage-based, and event-based rotation
5. **Agility**: Design for algorithm migration from day one

Plan for change - the PQC landscape will continue to evolve.
