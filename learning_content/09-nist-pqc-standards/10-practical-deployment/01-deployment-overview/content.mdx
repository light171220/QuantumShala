# Deployment Overview for PQC

Moving from testing to production with post-quantum cryptography requires careful planning. This chapter covers practical aspects of PQC deployment.

## Deployment Strategy Options

```
PQC Deployment Strategies:
═══════════════════════════════════════════════════════════════════

Strategy 1: Big Bang
┌─────────────────────────────────────────────────────────────────┐
│ All systems migrate simultaneously                              │
│                                                                  │
│ Pros: Clean cutover, minimal hybrid duration                   │
│ Cons: High risk, requires extensive coordination                │
│ Best for: Small, tightly coupled environments                   │
└─────────────────────────────────────────────────────────────────┘

Strategy 2: Phased Rollout
┌─────────────────────────────────────────────────────────────────┐
│ Systems migrate in prioritized waves                            │
│                                                                  │
│ Wave 1 ──▶ Wave 2 ──▶ Wave 3 ──▶ Wave 4                        │
│ (Critical) (High)    (Medium)   (Low)                          │
│                                                                  │
│ Pros: Manageable risk, lessons learned between waves           │
│ Cons: Extended hybrid period, more complex coordination        │
│ Best for: Large enterprises, heterogeneous environments        │
└─────────────────────────────────────────────────────────────────┘

Strategy 3: Feature Flag / Gradual
┌─────────────────────────────────────────────────────────────────┐
│ PQC enabled incrementally via configuration                     │
│                                                                  │
│    0% ──▶ 10% ──▶ 50% ──▶ 100% traffic using PQC              │
│                                                                  │
│ Pros: Granular control, easy rollback                          │
│ Cons: Requires sophisticated traffic management                │
│ Best for: SaaS platforms, high-availability systems            │
└─────────────────────────────────────────────────────────────────┘

Strategy 4: Hybrid First
┌─────────────────────────────────────────────────────────────────┐
│ Deploy hybrid mode everywhere, then remove classical           │
│                                                                  │
│ Classical ──▶ Hybrid (Classical+PQC) ──▶ PQC Only             │
│                                                                  │
│ Pros: Security during transition, gradual confidence building  │
│ Cons: Performance overhead, complexity                         │
│ Best for: Risk-averse organizations, regulatory environments   │
└─────────────────────────────────────────────────────────────────┘
```

## Deployment Prerequisites

```python
from dataclasses import dataclass, field
from typing import List, Dict
from enum import Enum

class ReadinessLevel(Enum):
    NOT_READY = 0
    PARTIAL = 1
    READY = 2

@dataclass
class Prerequisite:
    """Deployment prerequisite."""
    name: str
    category: str
    description: str
    verification: str
    readiness: ReadinessLevel = ReadinessLevel.NOT_READY

@dataclass
class DeploymentReadiness:
    """Assess deployment readiness."""
    prerequisites: List[Prerequisite] = field(default_factory=list)

    def add_prerequisite(self, prereq: Prerequisite):
        self.prerequisites.append(prereq)

    def calculate_readiness(self) -> Dict:
        """Calculate overall readiness."""
        total = len(self.prerequisites)
        if total == 0:
            return {"score": 0, "status": "NO_DATA"}

        ready = sum(1 for p in self.prerequisites
                   if p.readiness == ReadinessLevel.READY)
        partial = sum(1 for p in self.prerequisites
                     if p.readiness == ReadinessLevel.PARTIAL)

        score = (ready + partial * 0.5) / total * 100

        if score >= 90:
            status = "READY"
        elif score >= 70:
            status = "MOSTLY_READY"
        elif score >= 50:
            status = "PARTIAL"
        else:
            status = "NOT_READY"

        return {
            "score": score,
            "status": status,
            "ready": ready,
            "partial": partial,
            "not_ready": total - ready - partial
        }

    def generate_checklist(self) -> str:
        """Generate readiness checklist."""
        lines = [
            "DEPLOYMENT READINESS CHECKLIST",
            "=" * 50,
        ]

        # Group by category
        by_category: Dict[str, List[Prerequisite]] = {}
        for p in self.prerequisites:
            by_category.setdefault(p.category, []).append(p)

        for category, prereqs in by_category.items():
            lines.append(f"\n{category}:")
            for p in prereqs:
                icon = {
                    ReadinessLevel.READY: "✓",
                    ReadinessLevel.PARTIAL: "◐",
                    ReadinessLevel.NOT_READY: "○"
                }[p.readiness]
                lines.append(f"  {icon} {p.name}")
                lines.append(f"      {p.description}")
                lines.append(f"      Verify: {p.verification}")

        # Summary
        readiness = self.calculate_readiness()
        lines.extend([
            "",
            f"OVERALL: {readiness['status']} ({readiness['score']:.0f}%)",
            f"  Ready: {readiness['ready']}",
            f"  Partial: {readiness['partial']}",
            f"  Not Ready: {readiness['not_ready']}",
        ])

        return "\n".join(lines)

# Create readiness assessment
readiness = DeploymentReadiness()

# Infrastructure prerequisites
readiness.add_prerequisite(Prerequisite(
    "PQC library installed",
    "Infrastructure",
    "FIPS-validated PQC library available on all target systems",
    "Verify library version and FIPS certificate on each system",
    ReadinessLevel.READY
))

readiness.add_prerequisite(Prerequisite(
    "Certificate infrastructure updated",
    "Infrastructure",
    "PKI supports PQC certificates and hybrid certificates",
    "Issue test certificate with ML-DSA, verify chain validation",
    ReadinessLevel.PARTIAL
))

readiness.add_prerequisite(Prerequisite(
    "Performance validated",
    "Infrastructure",
    "Performance testing shows acceptable latency with PQC",
    "Run load tests with PQC enabled, compare to baseline",
    ReadinessLevel.READY
))

# Application prerequisites
readiness.add_prerequisite(Prerequisite(
    "Application code updated",
    "Application",
    "Applications call PQC algorithms correctly",
    "Review code changes, run integration tests",
    ReadinessLevel.READY
))

readiness.add_prerequisite(Prerequisite(
    "Feature flags configured",
    "Application",
    "PQC can be enabled/disabled via configuration",
    "Toggle flag, verify behavior change",
    ReadinessLevel.READY
))

# Operational prerequisites
readiness.add_prerequisite(Prerequisite(
    "Monitoring updated",
    "Operations",
    "Dashboards show PQC-specific metrics",
    "Review dashboards for algorithm, handshake, error metrics",
    ReadinessLevel.PARTIAL
))

readiness.add_prerequisite(Prerequisite(
    "Runbooks created",
    "Operations",
    "Operational procedures documented for PQC",
    "Review runbooks for key rotation, incident response",
    ReadinessLevel.NOT_READY
))

readiness.add_prerequisite(Prerequisite(
    "Team trained",
    "Operations",
    "Operations team trained on PQC specifics",
    "Training completion records for all team members",
    ReadinessLevel.PARTIAL
))

print(readiness.generate_checklist())
```

## Risk Assessment

```python
from dataclasses import dataclass
from typing import List
from enum import Enum

class Likelihood(Enum):
    LOW = 1
    MEDIUM = 2
    HIGH = 3

class Impact(Enum):
    LOW = 1
    MEDIUM = 2
    HIGH = 3
    CRITICAL = 4

@dataclass
class DeploymentRisk:
    """Deployment risk assessment."""
    name: str
    description: str
    likelihood: Likelihood
    impact: Impact
    mitigation: str
    contingency: str

    @property
    def risk_score(self) -> int:
        return self.likelihood.value * self.impact.value

    @property
    def risk_level(self) -> str:
        score = self.risk_score
        if score >= 9:
            return "CRITICAL"
        elif score >= 6:
            return "HIGH"
        elif score >= 3:
            return "MEDIUM"
        return "LOW"

class DeploymentRiskRegistry:
    """Registry of deployment risks."""

    def __init__(self):
        self.risks: List[DeploymentRisk] = []
        self._load_common_risks()

    def _load_common_risks(self):
        """Load common PQC deployment risks."""
        self.risks = [
            DeploymentRisk(
                "Performance degradation",
                "PQC operations cause unacceptable latency in production",
                Likelihood.MEDIUM,
                Impact.HIGH,
                "Thorough performance testing in staging; capacity planning",
                "Rollback to classical; tune timeouts; add capacity"
            ),
            DeploymentRisk(
                "Interoperability failure",
                "Clients/partners cannot connect with PQC enabled",
                Likelihood.MEDIUM,
                Impact.HIGH,
                "Test with all major client types; maintain hybrid mode",
                "Keep classical fallback; work with partners on upgrades"
            ),
            DeploymentRisk(
                "Certificate chain issues",
                "PQC certificates not trusted by clients",
                Likelihood.MEDIUM,
                Impact.CRITICAL,
                "Pre-deploy certificates; test chain validation",
                "Switch to hybrid certs; update client trust stores"
            ),
            DeploymentRisk(
                "Library bugs",
                "Undiscovered bugs in PQC library cause crashes",
                Likelihood.LOW,
                Impact.HIGH,
                "Use well-tested, validated libraries; thorough testing",
                "Rollback; patch library; escalate to vendor"
            ),
            DeploymentRisk(
                "Key management failures",
                "Key generation or rotation fails in production",
                Likelihood.LOW,
                Impact.CRITICAL,
                "Test key lifecycle thoroughly; have backup procedures",
                "Manual key operations; emergency key rotation"
            ),
            DeploymentRisk(
                "Configuration errors",
                "Incorrect configuration leads to security or functionality issues",
                Likelihood.MEDIUM,
                Impact.MEDIUM,
                "Configuration review; automated validation",
                "Correct configuration; redeploy"
            ),
        ]

    def get_high_risks(self) -> List[DeploymentRisk]:
        return [r for r in self.risks if r.risk_level in ("HIGH", "CRITICAL")]

    def generate_report(self) -> str:
        """Generate risk assessment report."""
        lines = [
            "DEPLOYMENT RISK ASSESSMENT",
            "=" * 50,
        ]

        # Sort by risk score
        sorted_risks = sorted(self.risks, key=lambda r: -r.risk_score)

        for risk in sorted_risks:
            lines.extend([
                f"\n[{risk.risk_level}] {risk.name}",
                f"  Description: {risk.description}",
                f"  Likelihood: {risk.likelihood.name}, Impact: {risk.impact.name}",
                f"  Mitigation: {risk.mitigation}",
                f"  Contingency: {risk.contingency}",
            ])

        high_risks = self.get_high_risks()
        lines.extend([
            "",
            f"Summary: {len(high_risks)} high/critical risks identified",
        ])

        return "\n".join(lines)

registry = DeploymentRiskRegistry()
print(registry.generate_report())
```

## Summary

Successful PQC deployment requires:

1. **Strategy selection**: Choose approach based on environment and risk tolerance
2. **Prerequisites**: Complete readiness checklist before deployment
3. **Risk management**: Identify, mitigate, and plan contingencies for risks
4. **Success criteria**: Define measurable criteria for deployment success

Careful planning and preparation are key to smooth production deployment.
