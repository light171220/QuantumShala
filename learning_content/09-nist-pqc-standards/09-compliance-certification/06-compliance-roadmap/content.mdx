## Building the Roadmap

```python
from dataclasses import dataclass, field
from typing import List, Dict, Optional
from datetime import date, timedelta
from enum import Enum

class Phase(Enum):
    ASSESSMENT = "Assessment"
    PLANNING = "Planning"
    PILOT = "Pilot"
    MIGRATION = "Migration"
    SUSTAINMENT = "Sustainment"

class Priority(Enum):
    CRITICAL = 1
    HIGH = 2
    MEDIUM = 3
    LOW = 4

class Status(Enum):
    NOT_STARTED = "Not Started"
    IN_PROGRESS = "In Progress"
    COMPLETED = "Completed"
    BLOCKED = "Blocked"

@dataclass
class Milestone:
    """Roadmap milestone."""
    id: str
    name: str
    phase: Phase
    target_date: date
    owner: str
    dependencies: List[str] = field(default_factory=list)
    status: Status = Status.NOT_STARTED
    completion_date: Optional[date] = None

@dataclass
class SystemMigration:
    """System migration tracking."""
    system_name: str
    priority: Priority
    data_sensitivity: str
    current_algorithms: List[str]
    target_algorithms: List[str]
    target_date: date
    status: Status = Status.NOT_STARTED

class PQCComplianceRoadmap:
    """
    PQC compliance roadmap builder and tracker.
    """

    def __init__(self, organization: str, start_date: date):
        self.organization = organization
        self.start_date = start_date
        self.milestones: List[Milestone] = []
        self.systems: List[SystemMigration] = []
        self.risks: List[Dict] = []

    def add_default_milestones(self):
        """Add standard PQC compliance milestones."""
        milestones = [
            # Assessment Phase
            Milestone("A1", "Complete cryptographic inventory",
                     Phase.ASSESSMENT, self.start_date + timedelta(days=30),
                     "Security Team"),
            Milestone("A2", "Map regulatory requirements",
                     Phase.ASSESSMENT, self.start_date + timedelta(days=45),
                     "Compliance", ["A1"]),
            Milestone("A3", "Complete risk assessment",
                     Phase.ASSESSMENT, self.start_date + timedelta(days=60),
                     "Security Team", ["A1", "A2"]),
            Milestone("A4", "Document current state",
                     Phase.ASSESSMENT, self.start_date + timedelta(days=90),
                     "Security Team", ["A3"]),

            # Planning Phase
            Milestone("P1", "Prioritize systems for migration",
                     Phase.PLANNING, self.start_date + timedelta(days=120),
                     "Security Team", ["A4"]),
            Milestone("P2", "Select algorithms and libraries",
                     Phase.PLANNING, self.start_date + timedelta(days=135),
                     "Architecture", ["P1"]),
            Milestone("P3", "Develop detailed migration plan",
                     Phase.PLANNING, self.start_date + timedelta(days=165),
                     "Project Management", ["P2"]),
            Milestone("P4", "Secure budget approval",
                     Phase.PLANNING, self.start_date + timedelta(days=180),
                     "Management", ["P3"]),

            # Pilot Phase
            Milestone("T1", "Set up test environment",
                     Phase.PILOT, self.start_date + timedelta(days=210),
                     "Infrastructure", ["P4"]),
            Milestone("T2", "Complete pilot migration",
                     Phase.PILOT, self.start_date + timedelta(days=270),
                     "Development", ["T1"]),
            Milestone("T3", "Validate interoperability",
                     Phase.PILOT, self.start_date + timedelta(days=300),
                     "QA", ["T2"]),
            Milestone("T4", "Document lessons learned",
                     Phase.PILOT, self.start_date + timedelta(days=330),
                     "Project Management", ["T3"]),

            # Migration Phase
            Milestone("M1", "Wave 1: Critical systems",
                     Phase.MIGRATION, self.start_date + timedelta(days=400),
                     "Development", ["T4"]),
            Milestone("M2", "Wave 2: High priority systems",
                     Phase.MIGRATION, self.start_date + timedelta(days=520),
                     "Development", ["M1"]),
            Milestone("M3", "Wave 3: Medium priority systems",
                     Phase.MIGRATION, self.start_date + timedelta(days=700),
                     "Development", ["M2"]),
            Milestone("M4", "Complete FIPS validation",
                     Phase.MIGRATION, self.start_date + timedelta(days=800),
                     "Security Team", ["M1"]),
            Milestone("M5", "Retire legacy algorithms",
                     Phase.MIGRATION, self.start_date + timedelta(days=1000),
                     "Security Team", ["M3"]),
        ]

        self.milestones.extend(milestones)

    def add_system(self, system: SystemMigration):
        """Add system for migration tracking."""
        self.systems.append(system)

    def add_risk(self, risk: Dict):
        """Add risk to tracking."""
        self.risks.append(risk)

    def get_phase_progress(self, phase: Phase) -> Dict:
        """Calculate progress for a phase."""
        phase_milestones = [m for m in self.milestones if m.phase == phase]
        if not phase_milestones:
            return {"total": 0, "completed": 0, "percent": 0}

        completed = sum(1 for m in phase_milestones
                       if m.status == Status.COMPLETED)
        total = len(phase_milestones)

        return {
            "total": total,
            "completed": completed,
            "percent": (completed / total * 100) if total > 0 else 0
        }

    def get_overdue_milestones(self) -> List[Milestone]:
        """Get overdue milestones."""
        today = date.today()
        return [
            m for m in self.milestones
            if m.target_date < today
            and m.status not in (Status.COMPLETED,)
        ]

    def generate_executive_summary(self) -> str:
        """Generate executive summary of roadmap status."""
        lines = [
            f"PQC COMPLIANCE ROADMAP: {self.organization}",
            f"Start Date: {self.start_date}",
            f"Report Date: {date.today()}",
            "=" * 60,
            "",
            "PHASE PROGRESS:",
        ]

        for phase in Phase:
            progress = self.get_phase_progress(phase)
            bar = "█" * int(progress['percent'] / 5)
            bar += "░" * (20 - len(bar))
            lines.append(
                f"  {phase.value:15} [{bar}] "
                f"{progress['completed']}/{progress['total']} "
                f"({progress['percent']:.0f}%)"
            )

        # Overdue items
        overdue = self.get_overdue_milestones()
        if overdue:
            lines.extend([
                "",
                f"⚠ OVERDUE MILESTONES ({len(overdue)}):",
            ])
            for m in overdue:
                days_overdue = (date.today() - m.target_date).days
                lines.append(f"  • {m.name} ({days_overdue} days overdue)")

        # Upcoming milestones
        upcoming = [
            m for m in self.milestones
            if m.status == Status.NOT_STARTED
            and m.target_date >= date.today()
            and m.target_date <= date.today() + timedelta(days=90)
        ]
        if upcoming:
            lines.extend([
                "",
                f"UPCOMING (90 days):",
            ])
            for m in sorted(upcoming, key=lambda x: x.target_date)[:5]:
                lines.append(f"  • {m.name} (Due: {m.target_date})")

        # Risk summary
        if self.risks:
            high_risks = [r for r in self.risks if r.get('severity') == 'HIGH']
            lines.extend([
                "",
                f"RISK SUMMARY: {len(self.risks)} total, {len(high_risks)} high",
            ])

        return "\n".join(lines)

# Create roadmap
roadmap = PQCComplianceRoadmap(
    "Example Corporation",
    date(2024, 1, 1)
)
roadmap.add_default_milestones()

# Add some systems
roadmap.add_system(SystemMigration(
    "TLS Gateway",
    Priority.CRITICAL,
    "High",
    ["RSA-2048", "ECDHE-P256"],
    ["ML-KEM-768", "X25519-ML-KEM-768"],
    date(2025, 6, 30)
))

roadmap.add_system(SystemMigration(
    "Document Signing Service",
    Priority.HIGH,
    "High",
    ["RSA-2048"],
    ["ML-DSA-65"],
    date(2025, 12, 31)
))

# Add risks
roadmap.add_risk({
    "id": "R1",
    "description": "FIPS validation delays",
    "severity": "HIGH",
    "mitigation": "Early engagement with lab"
})

print(roadmap.generate_executive_summary())
```

## Progress Tracking

```python
from dataclasses import dataclass
from typing import List, Dict
from datetime import date
import json

@dataclass
class ProgressUpdate:
    """Weekly progress update."""
    update_date: date
    milestone_updates: Dict[str, str]  # milestone_id -> status update
    blockers: List[str]
    decisions_needed: List[str]
    next_week_focus: List[str]

class ProgressTracker:
    """
    Track and report roadmap progress.
    """

    def __init__(self, roadmap: PQCComplianceRoadmap):
        self.roadmap = roadmap
        self.updates: List[ProgressUpdate] = []

    def add_update(self, update: ProgressUpdate):
        """Add progress update."""
        self.updates.append(update)

    def calculate_schedule_variance(self) -> Dict:
        """Calculate schedule variance."""
        today = date.today()
        total_milestones = len(self.roadmap.milestones)
        if total_milestones == 0:
            return {"variance_days": 0, "status": "ON_TRACK"}

        # Calculate expected vs actual completion
        expected_complete = sum(
            1 for m in self.roadmap.milestones
            if m.target_date <= today
        )

        actual_complete = sum(
            1 for m in self.roadmap.milestones
            if m.status == Status.COMPLETED
        )

        variance = actual_complete - expected_complete

        if variance >= 0:
            status = "ON_TRACK"
        elif variance >= -2:
            status = "MINOR_DELAY"
        else:
            status = "SIGNIFICANT_DELAY"

        return {
            "expected_complete": expected_complete,
            "actual_complete": actual_complete,
            "variance": variance,
            "status": status
        }

    def generate_status_report(self) -> str:
        """Generate status report."""
        variance = self.calculate_schedule_variance()

        lines = [
            "PQC COMPLIANCE STATUS REPORT",
            f"Date: {date.today()}",
            "=" * 50,
            "",
            f"Schedule Status: {variance['status']}",
            f"Milestones Expected: {variance['expected_complete']}",
            f"Milestones Complete: {variance['actual_complete']}",
            f"Variance: {variance['variance']:+d}",
            "",
        ]

        # Recent updates
        if self.updates:
            latest = self.updates[-1]
            lines.extend([
                f"Latest Update ({latest.update_date}):",
            ])
            if latest.blockers:
                lines.append("  Blockers:")
                for b in latest.blockers:
                    lines.append(f"    ⚠ {b}")
            if latest.decisions_needed:
                lines.append("  Decisions Needed:")
                for d in latest.decisions_needed:
                    lines.append(f"    ? {d}")

        return "\n".join(lines)

# Example tracking
tracker = ProgressTracker(roadmap)
tracker.add_update(ProgressUpdate(
    update_date=date.today(),
    milestone_updates={"A1": "90% complete"},
    blockers=["Waiting for asset discovery tool license"],
    decisions_needed=["Select primary PQC library vendor"],
    next_week_focus=["Complete application inventory", "Begin vendor evaluation"]
))

print(tracker.generate_status_report())
```

## Summary

Building and executing a PQC compliance roadmap requires:

1. **Phased approach**: Assessment → Planning → Pilot → Migration → Sustainment
2. **Clear milestones**: Trackable deliverables with owners and dates
3. **Resource planning**: Budget and staffing for each phase
4. **Progress tracking**: Regular status updates and variance analysis

Start planning now—compliance deadlines are approaching and the migration is complex.
