## Entanglement-Based QKD

Distributing entanglement enables information-theoretically secure key exchange:

```python
import numpy as np
from dataclasses import dataclass
from typing import Tuple, List, Optional, Dict
from enum import Enum

class E91Protocol:
    """
    E91 (Ekert 1991) quantum key distribution protocol.

    Security based on Bell inequality violation.
    Uses shared entanglement for key generation.
    """

    def __init__(
        self,
        entanglement_rate: float = 1000,  # pairs/s
        entanglement_fidelity: float = 0.98,
        detector_efficiency: float = 0.9
    ):
        self.entanglement_rate = entanglement_rate
        self.entanglement_fidelity = entanglement_fidelity
        self.detector_efficiency = detector_efficiency

        # Measurement angles for E91
        # Alice: 0, π/8, π/4
        # Bob: π/8, π/4, 3π/8
        self.alice_angles = [0, np.pi/8, np.pi/4]
        self.bob_angles = [np.pi/8, np.pi/4, 3*np.pi/8]

    def measure_pair(
        self,
        fidelity: float
    ) -> Tuple[int, int, float, float]:
        """
        Simulate measurement of one entangled pair.

        Returns (alice_result, bob_result, alice_angle, bob_angle)
        """
        # Random angle selection
        alice_angle = np.random.choice(self.alice_angles)
        bob_angle = np.random.choice(self.bob_angles)

        # For |Φ⁺⟩, correlation is cos(θ_A - θ_B)
        # Result correlation based on angle difference
        angle_diff = alice_angle - bob_angle
        correlation = fidelity * np.cos(2 * angle_diff)

        # Sample results
        if np.random.random() < (1 + correlation) / 2:
            alice_result = np.random.choice([0, 1])
            bob_result = alice_result  # Correlated
        else:
            alice_result = np.random.choice([0, 1])
            bob_result = 1 - alice_result  # Anti-correlated

        return alice_result, bob_result, alice_angle, bob_angle

    def run_protocol(
        self,
        n_pairs: int = 10000
    ) -> Dict:
        """Run complete E91 protocol."""
        # Measurement results
        key_bits_alice = []
        key_bits_bob = []
        bell_test_data = []

        for _ in range(n_pairs):
            # Generate pair with some fidelity
            if np.random.random() > self.detector_efficiency**2:
                continue  # Lost pair

            fidelity = self.entanglement_fidelity

            a_result, b_result, a_angle, b_angle = self.measure_pair(fidelity)

            # Check if angles match for key generation
            # Key bits when Alice=0 and Bob=π/8 or Alice=π/4 and Bob=3π/8
            if (np.isclose(a_angle, 0) and np.isclose(b_angle, np.pi/8)) or \
               (np.isclose(a_angle, np.pi/4) and np.isclose(b_angle, 3*np.pi/8)):
                # Wait - E91 uses non-matching for key
                # Actually for key: both measure in same basis
                pass

            # For Bell test: specific angle combinations
            if np.isclose(a_angle, 0) and np.isclose(b_angle, np.pi/8):
                bell_test_data.append((a_result, b_result, "a1b1"))
            elif np.isclose(a_angle, 0) and np.isclose(b_angle, 3*np.pi/8):
                bell_test_data.append((a_result, b_result, "a1b3"))
            elif np.isclose(a_angle, np.pi/4) and np.isclose(b_angle, np.pi/8):
                bell_test_data.append((a_result, b_result, "a3b1"))
            elif np.isclose(a_angle, np.pi/4) and np.isclose(b_angle, 3*np.pi/8):
                bell_test_data.append((a_result, b_result, "a3b3"))

            # Key generation when angles match (after sifting)
            if np.isclose(a_angle, np.pi/8) and np.isclose(b_angle, np.pi/8):
                key_bits_alice.append(a_result)
                key_bits_bob.append(b_result)

        # Calculate Bell parameter
        S = self._calculate_bell_parameter(bell_test_data)

        # Calculate QBER
        matching = sum(a == b for a, b in zip(key_bits_alice, key_bits_bob))
        qber = 1 - matching / len(key_bits_alice) if key_bits_alice else 0.5

        return {
            "total_pairs": n_pairs,
            "detected_pairs": len(key_bits_alice) + len(bell_test_data),
            "key_bits": len(key_bits_alice),
            "bell_parameter": S,
            "bell_violation": S > 2,
            "qber": qber,
            "secure": S > 2 and qber < 0.11
        }

    def _calculate_bell_parameter(self, data: List) -> float:
        """Calculate CHSH Bell parameter from measurement data."""
        correlations = {"a1b1": [], "a1b3": [], "a3b1": [], "a3b3": []}

        for a, b, label in data:
            # Correlation = +1 if same, -1 if different
            corr = 1 if a == b else -1
            correlations[label].append(corr)

        E = {}
        for label, values in correlations.items():
            E[label] = np.mean(values) if values else 0

        # S = E(a1,b1) - E(a1,b3) + E(a3,b1) + E(a3,b3)
        S = E["a1b1"] - E["a1b3"] + E["a3b1"] + E["a3b3"]
        return abs(S)

# Demonstrate E91
e91 = E91Protocol(entanglement_fidelity=0.98)
result = e91.run_protocol(n_pairs=10000)

print("E91 QKD Protocol")
print("=" * 50)
print(f"Total pairs generated: {result['total_pairs']}")
print(f"Key bits generated: {result['key_bits']}")
print(f"Bell parameter S: {result['bell_parameter']:.3f}")
print(f"Bell violation (S > 2): {result['bell_violation']}")
print(f"QBER: {result['qber']*100:.2f}%")
print(f"Key is secure: {result['secure']}")
```

## Quantum Teleportation Networks

Entanglement enables teleporting quantum states across networks:

```python
class TeleportationNetwork:
    """
    Quantum teleportation network using distributed entanglement.

    Enables transferring quantum states between nodes
    without physically sending quantum particles.
    """

    def __init__(self, n_nodes: int = 4):
        self.n_nodes = n_nodes
        self.entanglement_links = {}
        self.teleportation_fidelities = {}

    def establish_link(
        self,
        node_a: int,
        node_b: int,
        fidelity: float
    ):
        """Establish entanglement link between nodes."""
        self.entanglement_links[(node_a, node_b)] = fidelity
        self.entanglement_links[(node_b, node_a)] = fidelity

    def teleport_state(
        self,
        source: int,
        target: int,
        input_fidelity: float = 1.0
    ) -> Tuple[bool, float]:
        """
        Teleport quantum state from source to target.

        Returns (success, output_fidelity).
        """
        # Find path (simple direct or one hop)
        if (source, target) in self.entanglement_links:
            # Direct link
            link_fidelity = self.entanglement_links[(source, target)]
            output_fidelity = self._teleportation_fidelity(
                input_fidelity, link_fidelity
            )
            return True, output_fidelity

        # Try one-hop path
        for intermediate in range(self.n_nodes):
            if intermediate in [source, target]:
                continue

            if (source, intermediate) in self.entanglement_links and \
               (intermediate, target) in self.entanglement_links:
                # Two-hop teleportation (teleport to intermediate, then to target)
                F1 = self.entanglement_links[(source, intermediate)]
                F2 = self.entanglement_links[(intermediate, target)]

                # First teleportation
                F_mid = self._teleportation_fidelity(input_fidelity, F1)
                # Second teleportation
                F_out = self._teleportation_fidelity(F_mid, F2)

                return True, F_out

        return False, 0.0

    def _teleportation_fidelity(
        self,
        state_fidelity: float,
        link_fidelity: float
    ) -> float:
        """
        Calculate output fidelity of teleportation.

        F_out ≈ F_state × F_link (for high fidelities)
        """
        # More accurate formula including errors
        # For Werner states: F_out = (3F_link + 1)/4 × F_state
        return ((3 * link_fidelity + 1) / 4) * state_fidelity

    def build_complete_network(self, base_fidelity: float = 0.95):
        """Build fully connected network."""
        for i in range(self.n_nodes):
            for j in range(i + 1, self.n_nodes):
                # Fidelity decreases with "distance"
                fidelity = base_fidelity ** (1 + 0.1 * abs(i - j))
                self.establish_link(i, j, fidelity)

    def network_analysis(self) -> Dict:
        """Analyze network connectivity and fidelities."""
        analysis = {
            "nodes": self.n_nodes,
            "links": len(self.entanglement_links) // 2,
            "min_fidelity": min(self.entanglement_links.values()),
            "max_fidelity": max(self.entanglement_links.values()),
            "avg_fidelity": np.mean(list(self.entanglement_links.values()))
        }
        return analysis

# Build and analyze teleportation network
network = TeleportationNetwork(n_nodes=5)
network.build_complete_network(base_fidelity=0.96)

print("\nQuantum Teleportation Network")
print("=" * 50)

analysis = network.network_analysis()
print(f"Nodes: {analysis['nodes']}")
print(f"Links: {analysis['links']}")
print(f"Link fidelity range: {analysis['min_fidelity']:.3f} - {analysis['max_fidelity']:.3f}")

# Test teleportation
print("\nTeleportation tests (input F=1.0):")
for source, target in [(0, 1), (0, 4), (1, 3)]:
    success, fidelity = network.teleport_state(source, target)
    print(f"  {source} → {target}: F_out = {fidelity:.4f}")
```

## Distributed Quantum Computing

Entanglement connects separate quantum processors:

```python
class DistributedQuantumComputer:
    """
    Distributed quantum computing across networked processors.

    Uses entanglement to perform non-local gates between
    qubits on different processors.
    """

    def __init__(self, n_processors: int = 2, qubits_per_processor: int = 5):
        self.n_processors = n_processors
        self.qubits_per_processor = qubits_per_processor
        self.total_qubits = n_processors * qubits_per_processor

        # Entanglement pool between processors
        self.entanglement_pool = {}

    def prepare_entanglement(
        self,
        proc_a: int,
        proc_b: int,
        n_pairs: int,
        fidelity: float = 0.95
    ):
        """Prepare entanglement pool between processor pairs."""
        key = (min(proc_a, proc_b), max(proc_a, proc_b))
        self.entanglement_pool[key] = {
            "n_pairs": n_pairs,
            "fidelity": fidelity
        }

    def non_local_cnot(
        self,
        control_proc: int,
        control_qubit: int,
        target_proc: int,
        target_qubit: int
    ) -> Tuple[bool, float]:
        """
        Perform non-local CNOT using teleportation.

        Uses one entangled pair and 4 classical bits.
        """
        key = (min(control_proc, target_proc), max(control_proc, target_proc))

        if key not in self.entanglement_pool:
            return False, 0.0

        pool = self.entanglement_pool[key]
        if pool["n_pairs"] < 1:
            return False, 0.0

        # Consume one entangled pair
        pool["n_pairs"] -= 1

        # Gate fidelity depends on entanglement fidelity
        # F_gate ≈ (3F + 1)/4 for teleported gate
        gate_fidelity = (3 * pool["fidelity"] + 1) / 4

        return True, gate_fidelity

    def execute_circuit(
        self,
        circuit: List[Tuple]
    ) -> Dict:
        """
        Execute a distributed circuit.

        Circuit format: [(gate, control_proc, control_q, target_proc, target_q), ...]
        """
        results = {
            "gates_attempted": 0,
            "gates_succeeded": 0,
            "entanglement_consumed": 0,
            "avg_gate_fidelity": []
        }

        for gate in circuit:
            gate_type = gate[0]
            results["gates_attempted"] += 1

            if gate_type == "CNOT":
                _, control_proc, control_q, target_proc, target_q = gate

                if control_proc == target_proc:
                    # Local CNOT - perfect
                    results["gates_succeeded"] += 1
                    results["avg_gate_fidelity"].append(0.999)  # Local gate fidelity
                else:
                    # Non-local CNOT
                    success, fidelity = self.non_local_cnot(
                        control_proc, control_q, target_proc, target_q
                    )
                    if success:
                        results["gates_succeeded"] += 1
                        results["entanglement_consumed"] += 1
                        results["avg_gate_fidelity"].append(fidelity)

        if results["avg_gate_fidelity"]:
            results["avg_gate_fidelity"] = np.mean(results["avg_gate_fidelity"])
        else:
            results["avg_gate_fidelity"] = 0

        return results

# Demonstrate distributed computing
dqc = DistributedQuantumComputer(n_processors=3, qubits_per_processor=4)

# Prepare entanglement pools
dqc.prepare_entanglement(0, 1, n_pairs=10, fidelity=0.95)
dqc.prepare_entanglement(1, 2, n_pairs=10, fidelity=0.95)
dqc.prepare_entanglement(0, 2, n_pairs=10, fidelity=0.92)

print("\nDistributed Quantum Computing")
print("=" * 50)

# Define a distributed circuit
circuit = [
    ("CNOT", 0, 0, 0, 1),  # Local CNOT on processor 0
    ("CNOT", 0, 0, 1, 0),  # Non-local: proc 0 → proc 1
    ("CNOT", 1, 0, 2, 0),  # Non-local: proc 1 → proc 2
    ("CNOT", 0, 1, 2, 1),  # Non-local: proc 0 → proc 2
    ("CNOT", 1, 1, 1, 2),  # Local CNOT on processor 1
]

result = dqc.execute_circuit(circuit)
print(f"Gates attempted: {result['gates_attempted']}")
print(f"Gates succeeded: {result['gates_succeeded']}")
print(f"Entanglement consumed: {result['entanglement_consumed']} pairs")
print(f"Average gate fidelity: {result['avg_gate_fidelity']:.4f}")
```

## Quantum Sensing Networks

Entanglement enhances distributed sensing beyond classical limits:

```python
class QuantumSensorNetwork:
    """
    Network of quantum sensors using distributed entanglement.

    Achieves Heisenberg-limited precision for global measurements.
    """

    def __init__(self, n_sensors: int = 4):
        self.n_sensors = n_sensors
        self.sensor_locations = np.linspace(0, 100, n_sensors)  # km

    def classical_sensing(
        self,
        true_signal: float,
        noise_std: float = 0.1,
        n_measurements: int = 1000
    ) -> Dict:
        """
        Classical independent sensing.

        Precision scales as 1/√(N×M) (SQL - Standard Quantum Limit)
        """
        measurements = []

        for _ in range(n_measurements):
            # Each sensor measures independently
            sensor_readings = []
            for loc in self.sensor_locations:
                # Signal with local variation
                local_signal = true_signal + 0.01 * loc
                reading = local_signal + np.random.normal(0, noise_std)
                sensor_readings.append(reading)

            # Average of sensors
            measurements.append(np.mean(sensor_readings))

        estimate = np.mean(measurements)
        uncertainty = np.std(measurements) / np.sqrt(n_measurements)

        return {
            "estimate": estimate,
            "uncertainty": uncertainty,
            "true_value": true_signal,
            "error": abs(estimate - true_signal),
            "scaling": "1/√(NM)"
        }

    def entangled_sensing(
        self,
        true_signal: float,
        entanglement_fidelity: float = 0.95,
        noise_std: float = 0.1,
        n_measurements: int = 1000
    ) -> Dict:
        """
        Entanglement-enhanced sensing.

        Approaches Heisenberg limit: precision ∝ 1/N
        """
        measurements = []

        for _ in range(n_measurements):
            # GHZ-like entangled state across sensors
            # Collective phase accumulation

            # Total phase from all sensors
            total_phase = 0
            for loc in self.sensor_locations:
                local_signal = true_signal + 0.01 * loc
                total_phase += local_signal

            # Enhanced sensitivity but also enhanced noise
            # Effective measurement with entanglement
            effective_signal = total_phase / self.n_sensors

            # Entanglement enhancement factor
            # Perfect GHZ: √N improvement
            enhancement = np.sqrt(self.n_sensors) * entanglement_fidelity

            # Noise reduced by entanglement
            effective_noise = noise_std / enhancement
            reading = effective_signal + np.random.normal(0, effective_noise)

            measurements.append(reading)

        estimate = np.mean(measurements)
        uncertainty = np.std(measurements) / np.sqrt(n_measurements)

        return {
            "estimate": estimate,
            "uncertainty": uncertainty,
            "true_value": true_signal,
            "error": abs(estimate - true_signal),
            "scaling": "1/(√M × N^α)",  # α depends on entanglement quality
            "enhancement_factor": np.sqrt(self.n_sensors) * entanglement_fidelity
        }

    def compare_methods(
        self,
        true_signal: float = 1.0,
        n_measurements: int = 1000
    ) -> Dict:
        """Compare classical vs entangled sensing."""
        classical = self.classical_sensing(true_signal, n_measurements=n_measurements)
        entangled = self.entangled_sensing(true_signal, n_measurements=n_measurements)

        improvement = classical["uncertainty"] / entangled["uncertainty"]

        return {
            "classical": classical,
            "entangled": entangled,
            "uncertainty_improvement": improvement,
            "n_sensors": self.n_sensors
        }

# Demonstrate quantum sensing
print("\nQuantum Sensor Network")
print("=" * 50)

for n_sensors in [2, 4, 8, 16]:

    comparison = network.compare_methods(true_signal=1.0, n_measurements=1000)

    print(f"  Classical uncertainty: {comparison['classical']['uncertainty']:.6f}")
    print(f"  Entangled uncertainty: {comparison['entangled']['uncertainty']:.6f}")
    print(f"  Improvement factor: {comparison['uncertainty_improvement']:.2f}×")
```

## Clock Synchronization

Entanglement enables ultra-precise time distribution:

```python
class QuantumClockNetwork:
    """
    Quantum clock synchronization using entanglement.

    Achieves synchronization beyond classical limits.
    """

    def __init__(self):
        self.clock_offsets = {}

    def classical_sync(
        self,
        distance: float,  # km
        n_exchanges: int = 100
    ) -> float:
        """
        Classical clock synchronization (like NTP).

        Limited by speed of light and channel noise.
        """
        # Round-trip time
        c = 3e5  # km/s
        rtt = 2 * distance / c

        # Timing uncertainty from channel
        timing_jitter = 1e-12 * distance  # 1 ps/km

        # Multiple exchanges average down noise
        uncertainty = timing_jitter / np.sqrt(n_exchanges)

        return uncertainty

    def quantum_sync(
        self,
        distance: float,
        entanglement_fidelity: float = 0.95,
        n_pairs: int = 100
    ) -> float:
        """
        Quantum clock synchronization using entanglement.

        Uses correlated photon arrival times.
        """
        # Entanglement provides correlated detection events
        # Timing correlation from simultaneous emission

        # Base timing resolution (detector + source)
        base_resolution = 10e-12  # 10 ps

        # Enhancement from entanglement
        # Perfect entanglement: √N improvement possible
        enhancement = np.sqrt(n_pairs) * entanglement_fidelity

        # Distance doesn't directly affect precision
        # (unlike classical where path asymmetry matters)
        distance_factor = 1 + 0.001 * distance  # Small correction

        uncertainty = base_resolution / enhancement * distance_factor

        return uncertainty

    def compare_sync_methods(
        self,
        distances: List[float]
    ) -> Dict:
        """Compare synchronization methods across distances."""
        results = {
            "distance": distances,
            "classical": [],
            "quantum": [],
            "improvement": []
        }

        for d in distances:
            classical = self.classical_sync(d)
            quantum = self.quantum_sync(d)

            results["classical"].append(classical)
            results["quantum"].append(quantum)
            results["improvement"].append(classical / quantum)

        return results
```
