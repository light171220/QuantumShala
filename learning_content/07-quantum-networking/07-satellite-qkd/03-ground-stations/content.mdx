# Ground Station Systems

## Introduction

While satellites carry the quantum payload through space, ground stations provide the critical interface between space and terrestrial networks. This lesson examines the design, components, and operational requirements of quantum ground stations.

## Key Components

```python
import numpy as np
from dataclasses import dataclass
from typing import List, Dict
from enum import Enum

class DetectorType(Enum):
    SPAD = "Single-Photon Avalanche Diode"
    SNSPD = "Superconducting Nanowire"
    PMT = "Photomultiplier Tube"

@dataclass
class TelescopeSpec:
    """Ground station telescope specifications."""
    aperture_m: float
    focal_length_m: float
    field_of_view_deg: float
    tracking_accuracy_arcsec: float
    slew_rate_deg_s: float
    mount_type: str  # alt-az, equatorial

@dataclass
class DetectorSpec:
    """Single-photon detector specifications."""
    detector_type: DetectorType
    efficiency: float
    dark_count_hz: float
    timing_jitter_ps: float
    dead_time_ns: float
    operating_temp_k: float

class GroundStationComponents:
    """
    Ground station component specifications.
    """

    def __init__(self):
        self.telescope = self._define_telescope()
        self.detectors = self._define_detectors()
        self.filtering = self._define_filtering()

    def _define_telescope(self) -> TelescopeSpec:
        """Define typical telescope specifications."""
        return TelescopeSpec(
            aperture_m=1.0,
            focal_length_m=10.0,
            field_of_view_deg=0.01,
            tracking_accuracy_arcsec=1.0,
            slew_rate_deg_s=5.0,
            mount_type="alt-azimuth"
        )

    def _define_detectors(self) -> Dict[str, DetectorSpec]:
        """Define detector options."""
        return {
            "spad": DetectorSpec(
                detector_type=DetectorType.SPAD,
                efficiency=0.6,
                dark_count_hz=100,
                timing_jitter_ps=350,
                dead_time_ns=50,
                operating_temp_k=250  # Cooled
            ),
            "snspd": DetectorSpec(
                detector_type=DetectorType.SNSPD,
                efficiency=0.9,
                dark_count_hz=10,
                timing_jitter_ps=50,
                dead_time_ns=10,
                operating_temp_k=2.5  # Cryogenic
            )
        }

    def _define_filtering(self) -> Dict:
        """Define filtering systems."""
        return {
            "spectral": {
                "type": "Narrowband interference filter",
                "bandwidth_nm": 1.0,
                "transmission": 0.9,
                "purpose": "Reduce background light"
            },
            "spatial": {
                "type": "Single-mode fiber coupling",
                "efficiency": 0.5,
                "purpose": "Spatial mode filtering"
            },
            "temporal": {
                "type": "Time gating",
                "gate_width_ns": 1.0,
                "purpose": "Coincidence with satellite pulses"
            }
        }

    def calculate_collection_efficiency(self) -> float:
        """Calculate total collection efficiency."""
        telescope_eff = 0.8  # Mirror reflectivity etc.
        spectral_eff = self.filtering["spectral"]["transmission"]
        spatial_eff = self.filtering["spatial"]["efficiency"]
        detector_eff = self.detectors["snspd"].efficiency

        total = telescope_eff * spectral_eff * spatial_eff * detector_eff
        return total

    def print_components(self):
        """Print component specifications."""
        print("Ground Station Components")
        print("=" * 70)

        print("\nTELESCOPE:")
        t = self.telescope
        print(f"  Aperture: {t.aperture_m} m")
        print(f"  Focal length: {t.focal_length_m} m")
        print(f"  Tracking: {t.tracking_accuracy_arcsec} arcsec")
        print(f"  Slew rate: {t.slew_rate_deg_s}°/s")
        print(f"  Mount: {t.mount_type}")

        print("\nDETECTORS:")
        for name, d in self.detectors.items():
            print(f"\n  {d.detector_type.value}:")
            print(f"    Efficiency: {d.efficiency*100:.0f}%")
            print(f"    Dark counts: {d.dark_count_hz} Hz")
            print(f"    Timing jitter: {d.timing_jitter_ps} ps")
            print(f"    Operating temp: {d.operating_temp_k} K")

        print("\nFILTERING:")
        for name, f in self.filtering.items():
            print(f"\n  {name.title()}: {f['type']}")
            print(f"    Purpose: {f['purpose']}")

        print(f"\nTotal collection efficiency: {self.calculate_collection_efficiency()*100:.1f}%")

# Display components
components = GroundStationComponents()
components.print_components()
```

## Site Selection

```python
@dataclass
class SiteParameters:
    """Ground station site parameters."""
    latitude: float
    longitude: float
    altitude_m: float
    clear_nights_per_year: int
    seeing_arcsec: float
    light_pollution: str  # low, medium, high
    infrastructure: str

class SiteSelection:
    """
    Ground station site selection criteria.
    """

    def __init__(self):
        self.criteria = self._define_criteria()
        self.example_sites = self._define_sites()

    def _define_criteria(self) -> Dict:
        """Define selection criteria."""
        return {
            "weather": {
                "weight": 0.25,
                "description": "Clear sky availability",
                "ideal": ">250 clear nights/year"
            },
            "seeing": {
                "weight": 0.20,
                "description": "Atmospheric turbulence",
                "ideal": "<1 arcsec seeing"
            },
            "altitude": {
                "weight": 0.15,
                "description": "Reduce atmospheric path",
                "ideal": ">2000m elevation"
            },
            "light_pollution": {
                "weight": 0.15,
                "description": "Background light level",
                "ideal": "Dark site, no nearby cities"
            },
            "infrastructure": {
                "weight": 0.15,
                "description": "Power, network, access",
                "ideal": "Good infrastructure access"
            },
            "latitude": {
                "weight": 0.10,
                "description": "Satellite visibility",
                "ideal": "Depends on satellite orbit"
            }
        }

    def _define_sites(self) -> List[SiteParameters]:
        """Define example sites."""
        return [
            SiteParameters(
                latitude=43.07,
                longitude=141.35,
                altitude_m=100,
                clear_nights_per_year=120,
                seeing_arcsec=2.0,
                light_pollution="medium",
                infrastructure="Excellent (urban)"
            ),
            SiteParameters(
                latitude=28.76,
                longitude=-17.89,
                altitude_m=2400,
                clear_nights_per_year=280,
                seeing_arcsec=0.7,
                light_pollution="low",
                infrastructure="Good (observatory)"
            ),
            SiteParameters(
                latitude=37.22,
                longitude=100.15,
                altitude_m=3200,
                clear_nights_per_year=260,
                seeing_arcsec=1.0,
                light_pollution="low",
                infrastructure="Moderate"
            )
        ]

    def score_site(self, site: SiteParameters) -> float:
        """Score a site based on criteria."""
        score = 0

        # Weather score (more clear nights = better)
        score += self.criteria["weather"]["weight"] * min(site.clear_nights_per_year / 300, 1)

        # Seeing score (lower = better)
        score += self.criteria["seeing"]["weight"] * max(0, 1 - site.seeing_arcsec / 3)

        # Altitude score
        score += self.criteria["altitude"]["weight"] * min(site.altitude_m / 3000, 1)

        # Light pollution
        lp_scores = {"low": 1.0, "medium": 0.5, "high": 0.2}
        score += self.criteria["light_pollution"]["weight"] * lp_scores.get(site.light_pollution, 0.5)

        return score

    def print_analysis(self):
        """Print site analysis."""
        print("\nGround Station Site Analysis")
        print("=" * 70)

        print("\nSelection Criteria:")
        for crit, details in self.criteria.items():
            print(f"  {crit.title()}: {details['description']}")
            print(f"    Weight: {details['weight']*100:.0f}%")
            print(f"    Ideal: {details['ideal']}")

        print("\n\nExample Site Comparison:")
        print("-" * 60)
        for i, site in enumerate(self.example_sites):
            score = self.score_site(site)
            print(f"\nSite {i+1}: ({site.latitude:.1f}°, {site.longitude:.1f}°)")
            print(f"  Altitude: {site.altitude_m} m")
            print(f"  Clear nights: {site.clear_nights_per_year}/year")
            print(f"  Seeing: {site.seeing_arcsec} arcsec")
            print(f"  Light pollution: {site.light_pollution}")
            print(f"  SCORE: {score*100:.1f}%")

# Analyze sites
selection = SiteSelection()
selection.print_analysis()
```

## Detection System Performance

```python
class DetectionPerformance:
    """
    Analyze detection system performance.
    """

    def __init__(
        self,
        detector: DetectorSpec,
        background_rate_hz: float = 1000,
        signal_rate_hz: float = 10000
    ):
        self.detector = detector
        self.background = background_rate_hz
        self.signal = signal_rate_hz

    def calculate_snr(self) -> float:
        """Calculate signal-to-noise ratio."""
        # Total noise
        noise = np.sqrt(self.background + self.detector.dark_count_hz)

        # Signal
        signal = self.signal * self.detector.efficiency

        return signal / noise

    def calculate_qber_contribution(self) -> float:
        """Estimate QBER from dark counts and background."""
        total_counts = self.signal * self.detector.efficiency + self.background + self.detector.dark_count_hz
        error_counts = (self.background + self.detector.dark_count_hz) / 2  # Half are errors

        return error_counts / total_counts if total_counts > 0 else 0

    def timing_requirements(self) -> Dict:
        """Calculate timing requirements."""
        return {
            "coincidence_window_ns": self.detector.timing_jitter_ps / 1000 * 3,  # 3σ
            "max_count_rate_mhz": 1e3 / self.detector.dead_time_ns,
            "time_tagging_resolution_ps": self.detector.timing_jitter_ps / 2
        }

    def print_analysis(self):
        """Print performance analysis."""
        print(f"\nDetection Performance Analysis")
        print("=" * 60)
        print(f"\nDetector: {self.detector.detector_type.value}")
        print(f"Signal rate: {self.signal} Hz")
        print(f"Background: {self.background} Hz")

        print(f"\nPerformance Metrics:")
        print(f"  SNR: {self.calculate_snr():.1f}")
        print(f"  QBER contribution: {self.calculate_qber_contribution()*100:.2f}%")

        timing = self.timing_requirements()
        print(f"\nTiming Requirements:")
        print(f"  Coincidence window: {timing['coincidence_window_ns']:.1f} ns")
        print(f"  Max count rate: {timing['max_count_rate_mhz']:.1f} MHz")
        print(f"  Time resolution: {timing['time_tagging_resolution_ps']:.0f} ps")

# Analyze SNSPD performance
snspd_spec = DetectorSpec(
    detector_type=DetectorType.SNSPD,
    efficiency=0.9,
    dark_count_hz=10,
    timing_jitter_ps=50,
    dead_time_ns=10,
    operating_temp_k=2.5
)

performance = DetectionPerformance(snspd_spec, background_rate_hz=500, signal_rate_hz=5000)
performance.print_analysis()
```
