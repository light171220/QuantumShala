## Technical Specifications

```python
import numpy as np
from dataclasses import dataclass
from typing import List, Dict

@dataclass
class MiciusSpecs:
    """Micius satellite specifications."""
    # Orbit
    altitude_km: float = 500
    inclination_deg: float = 97.4
    orbital_period_min: float = 95

    # Optical system
    telescope_diameter_m: float = 0.3
    pointing_accuracy_urad: float = 0.5
    wavelength_nm: float = 850

    # Entanglement source
    source_type: str = "SPDC in BBO crystal"
    pair_rate_hz: float = 5.9e6
    pair_fidelity: float = 0.907

    # QKD system
    qkd_protocol: str = "Decoy-state BB84"
    encoding: str = "Polarization"

class MiciusAnalysis:
    """
    Analyze Micius satellite performance.
    """

    def __init__(self):
        self.specs = MiciusSpecs()
        self.experiments = self._define_experiments()

    def _define_experiments(self) -> Dict:
        """Define key Micius experiments."""
        return {
            "qkd_downlink": {
                "name": "Satellite-to-Ground QKD",
                "date": "2017",
                "distance_km": 1200,
                "key_rate_bps": 1000,  # Peak during pass
                "qber": 0.011,
                "significance": "First satellite QKD demonstration"
            },
            "entanglement_distribution": {
                "name": "Entanglement Distribution",
                "date": "2017",
                "distance_km": 1203,
                "fidelity": 0.869,
                "bell_violation": 2.37,  # S value
                "significance": "Longest entanglement distribution"
            },
            "teleportation": {
                "name": "Ground-to-Satellite Teleportation",
                "date": "2017",
                "distance_km": 1400,
                "fidelity": 0.80,
                "significance": "First ground-to-space teleportation"
            },
            "intercontinental_qkd": {
                "name": "Intercontinental QKD",
                "date": "2017",
                "endpoints": ["Xinglong (China)", "Graz (Austria)"],
                "distance_km": 7600,
                "method": "Entanglement swapping relay",
                "significance": "First intercontinental quantum-secured videoconference"
            }
        }

    def calculate_link_budget(self, zenith_angle: float = 0) -> Dict:
        """Calculate Micius link budget."""
        specs = self.specs

        # Slant range
        slant_range = specs.altitude_km * 1000 / np.cos(np.radians(zenith_angle))

        # Beam divergence (diffraction limited)
        wavelength_m = specs.wavelength_nm * 1e-9
        beam_div = 1.22 * wavelength_m / specs.telescope_diameter_m

        # Geometric loss
        spot_radius = slant_range * beam_div
        receive_area = np.pi * (1.0) ** 2  # 1m ground telescope
        spot_area = np.pi * spot_radius ** 2
        geo_loss_db = -10 * np.log10(receive_area / spot_area)

        # Atmospheric
        atm_loss_db = 3 * (1 / np.cos(np.radians(min(zenith_angle, 70))))

        # Total
        total_loss_db = geo_loss_db + atm_loss_db

        return {
            "slant_range_km": slant_range / 1000,
            "geometric_loss_db": geo_loss_db,
            "atmospheric_loss_db": atm_loss_db,
            "total_loss_db": total_loss_db,
            "transmission": 10 ** (-total_loss_db / 10)
        }

    def print_experiments(self):
        """Print experiment summary."""
        print("Micius Key Experiments")
        print("=" * 70)

        for exp_id, exp in self.experiments.items():
            print(f"\n{exp['name']} ({exp['date']})")
            print("-" * 50)
            if 'distance_km' in exp:
                print(f"  Distance: {exp['distance_km']} km")
            if 'key_rate_bps' in exp:
                print(f"  Key rate: {exp['key_rate_bps']} bps")
            if 'fidelity' in exp:
                print(f"  Fidelity: {exp['fidelity']}")
            if 'bell_violation' in exp:
                print(f"  Bell S value: {exp['bell_violation']} (>2 = violation)")
            print(f"  Significance: {exp['significance']}")

# Analyze Micius
micius = MiciusAnalysis()
micius.print_experiments()

print("\n\nLink Budget Analysis:")
print("-" * 50)
for angle in [0, 30, 53]:  # 53° was typical operational angle
    budget = micius.calculate_link_budget(angle)
    print(f"\nZenith angle {angle}°:")
    print(f"  Slant range: {budget['slant_range_km']:.0f} km")
    print(f"  Total loss: {budget['total_loss_db']:.1f} dB")
```

class RateComparison:
    """
    Compare Micius rates to fiber systems.
    """

    def __init__(self):
        self.systems = self._define_systems()

    def _define_systems(self) -> Dict:
        """Define comparison systems."""
        return {
            "micius_1200km": {
                "name": "Micius (1,200 km)",
                "distance_km": 1200,
                "key_rate_bps": 1000,
                "availability": "~10 min/pass, nighttime, clear weather"
            },
            "fiber_100km": {
                "name": "Fiber QKD (100 km)",
                "distance_km": 100,
                "key_rate_bps": 10000,
                "availability": "24/7"
            },
            "fiber_500km": {
                "name": "Fiber QKD (500 km, trusted nodes)",
                "distance_km": 500,
                "key_rate_bps": 100,
                "availability": "24/7 (with trusted nodes)"
            },
            "theoretical_fiber_1200km": {
                "name": "Theoretical fiber (1,200 km, no repeaters)",
                "distance_km": 1200,
                "key_rate_bps": 0.0001,  # Essentially zero
                "availability": "Impractical"
            }
        }

    def print_comparison(self):
        """Print rate comparison."""
        print("\nKey Rate Comparison")
        print("=" * 70)
        print(f"{'System':<35} {'Distance':<12} {'Rate (bps)':<12} {'Notes'}")
        print("-" * 70)

        for sys_id, sys in self.systems.items():
            rate_str = f"{sys['key_rate_bps']:.1f}" if sys['key_rate_bps'] >= 0.1 else "<0.001"
            print(f"{sys['name']:<35} {sys['distance_km']:<12} {rate_str:<12}")

        print("\n\nKey Insight: At 1,200 km, Micius achieves rates that are")
        print("            10,000,000× higher than theoretical fiber!")
