# Quantum Key Distribution Overview

Quantum Key Distribution (QKD) enables two parties to establish a shared secret key with security guaranteed by the laws of physics.

## What is QKD?

```

═══════════════════════════════════════════════════════════════════

The Cryptographic Stack:
┌─────────────────────────────────────────────────────────────────┐
│                      Application Layer                          │
│                  (Encrypted messaging, VPN, etc.)               │
├─────────────────────────────────────────────────────────────────┤
│                    Encryption Layer                              │

│                           │                                      │
│                           │ Requires                             │
│                           │ shared key                           │
│                           ▼                                      │
├─────────────────────────────────────────────────────────────────┤

│  ┌─────────────────────┐    ┌─────────────────────┐             │
│  │    Classical        │ OR │     Quantum         │             │
│  │  (RSA, ECDH, PQC)   │    │      (QKD)          │             │
│  │                     │    │                     │             │
```

```python
class QKDStep(Enum):
    """Steps in QKD protocol."""
    QUANTUM_TRANSMISSION = "quantum_transmission"
    BASIS_ANNOUNCEMENT = "basis_announcement"
    SIFTING = "sifting"
    ERROR_ESTIMATION = "error_estimation"
    ERROR_CORRECTION = "error_correction"
    PRIVACY_AMPLIFICATION = "privacy_amplification"

@dataclass
class QKDState:
    """State of QKD protocol execution."""
    step: QKDStep
    alice_bits: List[int]
    bob_bits: List[int]
    sifted_alice: List[int]
    sifted_bob: List[int]
    error_rate: float
    final_key_length: int

class QKDProtocolSimulator:
    """
    Simulate a generic QKD protocol flow.
    """

    def __init__(self, num_bits: int = 1000):
        self.num_bits = num_bits
        self.qber = 0.02  # 2% error rate

    def quantum_transmission(self) -> Tuple[List[int], List[int], List[int], List[int]]:
        """
        Simulate quantum bit transmission.

        Returns:
            alice_bits, alice_bases, bob_bases, bob_bits
        """
        # Alice generates random bits and bases
        alice_bits = np.random.randint(0, 2, self.num_bits).tolist()
        alice_bases = np.random.randint(0, 2, self.num_bits).tolist()

        # Bob chooses random bases
        bob_bases = np.random.randint(0, 2, self.num_bits).tolist()

        # Bob measures - gets correct result only if bases match
        bob_bits = []
        for i in range(self.num_bits):
            if alice_bases[i] == bob_bases[i]:
                # Same basis - get correct bit with noise
                if np.random.random() < self.qber:
                    bob_bits.append(1 - alice_bits[i])  # Error
                else:
                    bob_bits.append(alice_bits[i])  # Correct
            else:
                # Different basis - random result
                bob_bits.append(np.random.randint(0, 2))

        return alice_bits, alice_bases, bob_bases, bob_bits

    def sifting(
        self,
        alice_bits: List[int],
        alice_bases: List[int],
        bob_bases: List[int],
        bob_bits: List[int]
    ) -> Tuple[List[int], List[int]]:
        """Keep only bits where bases matched."""
        sifted_alice = []
        sifted_bob = []

        for i in range(len(alice_bits)):
            if alice_bases[i] == bob_bases[i]:
                sifted_alice.append(alice_bits[i])
                sifted_bob.append(bob_bits[i])

        return sifted_alice, sifted_bob

    def error_estimation(
        self,
        alice_bits: List[int],
        bob_bits: List[int],
        sample_fraction: float = 0.1
    ) -> Tuple[float, List[int], List[int]]:
        """
        Estimate error rate by sacrificing some bits.

        Returns:
            (error_rate, remaining_alice, remaining_bob)
        """
        n = len(alice_bits)
        sample_size = int(n * sample_fraction)

        # Random sample for estimation
        indices = np.random.choice(n, sample_size, replace=False)
        indices.sort()

        errors = sum(1 for i in indices if alice_bits[i] != bob_bits[i])
        error_rate = errors / sample_size

        # Remove sampled bits
        remaining_indices = set(range(n)) - set(indices)
        remaining_alice = [alice_bits[i] for i in remaining_indices]
        remaining_bob = [bob_bits[i] for i in remaining_indices]

        return error_rate, remaining_alice, remaining_bob

    def calculate_secret_key_length(
        self,
        sifted_length: int,
        error_rate: float
    ) -> int:
        """Calculate final secret key length."""
        if error_rate >= 0.11:
            return 0  # Above threshold

        # Binary entropy function
        def h(p):
            if p <= 0 or p >= 1:
                return 0
            return -p * np.log2(p) - (1 - p) * np.log2(1 - p)

        # Secret fraction: 1 - 2h(e) for BB84-like
        secret_fraction = max(0, 1 - 2 * h(error_rate))

        # Account for error correction overhead (1.16 efficiency)
        correction_overhead = 1.16 * h(error_rate)
        net_fraction = secret_fraction - correction_overhead + h(error_rate)
        net_fraction = max(0, net_fraction)

        return int(sifted_length * net_fraction)

    def run_protocol(self) -> QKDState:
        """Run complete QKD protocol simulation."""
        # Step 1: Quantum transmission
        alice_bits, alice_bases, bob_bases, bob_bits = self.quantum_transmission()

        # Step 2: Sifting
        sifted_alice, sifted_bob = self.sifting(
            alice_bits, alice_bases, bob_bases, bob_bits
        )

        # Step 3: Error estimation
        error_rate, final_alice, final_bob = self.error_estimation(
            sifted_alice, sifted_bob
        )

        # Step 4: Calculate final key length
        final_key_length = self.calculate_secret_key_length(
            len(final_alice), error_rate
        )

        return QKDState(
            step=QKDStep.PRIVACY_AMPLIFICATION,
            alice_bits=alice_bits,
            bob_bits=bob_bits,
            sifted_alice=sifted_alice,
            sifted_bob=sifted_bob,
            error_rate=error_rate,
            final_key_length=final_key_length
        )

# Run simulation
sim = QKDProtocolSimulator(num_bits=10000)
result = sim.run_protocol()

print("QKD PROTOCOL SIMULATION")
print("=" * 50)
print(f"Initial bits transmitted: {len(result.alice_bits)}")
print(f"After sifting: {len(result.sifted_alice)} ({len(result.sifted_alice)/len(result.alice_bits)*100:.1f}%)")
print(f"Estimated QBER: {result.error_rate*100:.2f}%")
print(f"Final secret key length: {result.final_key_length} bits")
print(f"Efficiency: {result.final_key_length/len(result.alice_bits)*100:.2f}%")
```

## Security Foundations

```
Why QKD is Secure:
═══════════════════════════════════════════════════════════════════

Three Quantum Principles:
┌─────────────────────────────────────────────────────────────────┐
│                                                                  │
│ 1. NO-CLONING THEOREM                                           │
│    ════════════════════                                         │
│    Eve cannot copy quantum states without disturbing them       │
│    → She cannot store copies for later analysis                │
│                                                                  │
│ 2. MEASUREMENT DISTURBANCE                                      │
│    ═══════════════════════                                      │
│    Measuring a quantum state changes it                         │
│    → Any eavesdropping introduces detectable errors            │
│                                                                  │
│ 3. HEISENBERG UNCERTAINTY                                       │
│    ═══════════════════════                                      │
│    Cannot precisely measure conjugate observables              │
│    → Wrong basis measurement destroys information              │
│                                                                  │
└─────────────────────────────────────────────────────────────────┘

Eve's Dilemma:
┌─────────────────────────────────────────────────────────────────┐
│                                                                  │
│  Alice ──────────── Eve ──────────── Bob                        │
│                      │                                          │
│         If Eve intercepts:                                      │
│         ┌────────────┴────────────┐                             │
│         │                         │                             │
│         ▼                         ▼                             │
│   Measures in     │        Measures in                          │
│   correct basis   │        wrong basis                          │
│         │         │              │                              │
│         │         │              ▼                              │
│         │         │     Destroys quantum                        │
│         │         │     info (50% random)                       │
│         │         │              │                              │
│         ▼         ▼              ▼                              │
│   Gets bit    Introduces     Introduces                         │
│   value        25% QBER       error                             │

│   Either way: Eavesdropping creates detectable errors!         │
```

```python
class SecurityAnalysis:
    """
    Analyze QKD security properties.
    """

    def eavesdropper_impact(self, intercept_fraction: float) -> dict:
        """
        Calculate impact of intercept-resend attack.

        Args:
            intercept_fraction: Fraction of qubits Eve intercepts
        """
        # Eve guesses basis correctly 50% of time
        # When wrong, introduces 50% error rate on those bits
        # Net error from eavesdropping: fraction × 0.5 × 0.5 = fraction × 0.25

        error_introduced = intercept_fraction * 0.25
        detection_probability = 1 - (1 - error_introduced) ** 100  # 100 sample bits

        return {
            "intercept_fraction": intercept_fraction,
            "qber_introduced": error_introduced,
            "detection_probability": detection_probability,
            "eve_information_bits": intercept_fraction * 0.5  # Relative to key
        }

    def security_bound(self, qber: float) -> dict:
        """
        Calculate security bounds for given QBER.
        """
        def h(p):
            if p <= 0 or p >= 1:
                return 0
            return -p * np.log2(p) - (1 - p) * np.log2(1 - p)

        # Maximum information Eve could have
        eve_info_bound = h(qber)

        # Secret key rate (Devetak-Winter bound)
        secret_rate = 1 - 2 * h(qber)

        return {
            "qber": qber,
            "max_eve_information": eve_info_bound,
            "secret_key_rate": max(0, secret_rate),
            "secure": qber < 0.11,
            "threshold_qber": 0.11
        }

    def compare_with_classical(self) -> str:
        """Compare QKD with classical key exchange."""
        return """
QKD vs CLASSICAL KEY EXCHANGE
═══════════════════════════════════════════════════════════════

Property               │ Classical (ECDH)    │ Quantum (QKD)
═══════════════════════╪═════════════════════╪═══════════════════

Quantum computer safe  │ No (Shor's alg)     │ Yes

Distance               │ Global              │ ~100km (direct)
Infrastructure         │ Software only       │ Specialized hardware
Cost                   │ Low                 │ High
Maturity               │ Decades of use      │ Emerging
"""

# Security analysis
security = SecurityAnalysis()

    impact = security.eavesdropper_impact(fraction)
    print(f"\nEve intercepts {fraction*100:.0f}% of qubits:")
    print(f"  QBER introduced: {impact['qber_introduced']*100:.2f}%")
    print(f"  Detection probability: {impact['detection_probability']*100:.1f}%")

print("\n" + security.compare_with_classical())
```

## Summary

QKD fundamentals:

1. **Purpose**: Distribute cryptographic keys with physics-based security
2. **Protocol types**: Prepare-measure, entanglement-based, continuous-variable
3. **Structure**: Quantum transmission → sifting → error correction → privacy amplification
4. **Security**: Based on no-cloning, measurement disturbance, uncertainty principle

QKD provides the key; classical symmetric encryption provides the confidentiality.
