## Hardware Platforms

import numpy as np
from dataclasses import dataclass
from typing import Dict, List

@dataclass
class HardwarePlatform:
    """Quantum repeater hardware platform specifications."""
    name: str
    memory_coherence: float    # seconds
    gate_fidelity: float
    photon_efficiency: float
    clock_rate: float          # Hz
    operating_temp: float      # K
    scalability: str           # low/medium/high
    maturity: str              # research/development/deployment

# Define major platforms
platforms = {
    "NV Centers": HardwarePlatform(
        name="Nitrogen-Vacancy Centers in Diamond",
        memory_coherence=1.0,

        clock_rate=1e6,
        operating_temp=4,
        scalability="medium",
        maturity="development"

    "Trapped Ions": HardwarePlatform(

        photon_efficiency=0.01,
        clock_rate=1e4,
        operating_temp=0.001,
        scalability="medium",
        maturity="research"
    ),
    "Neutral Atoms": HardwarePlatform(
        name="Neutral Atom Arrays",
        memory_coherence=10,
        gate_fidelity=0.995,

        memory_coherence=0.01,
        gate_fidelity=0.95,
        photon_efficiency=0.3,
        clock_rate=1e7,
        operating_temp=0.1,
        scalability="high",
        maturity="research"
    )
}

class PlatformAnalyzer:
    """Analyze and compare hardware platforms."""

    def __init__(self, platforms: Dict[str, HardwarePlatform]):
        self.platforms = platforms

    def rate_for_distance(
        self,
        platform: HardwarePlatform,
        distance: float,
        n_segments: int = 10
    ) -> float:
        """Estimate achievable rate for given distance."""
        L_seg = distance / n_segments

        # Fiber transmission
        eta_fiber = 10**(-0.2 * L_seg / 10)

        # Total link efficiency
        eta_link = eta_fiber * platform.photon_efficiency

        # Time per segment attempt
        t_attempt = 1 / platform.clock_rate

        # Average time for segment (DLCZ-like)
        p_seg = 0.01 * eta_link  # DLCZ probability
        t_segment = t_attempt / p_seg if p_seg > 0 else float('inf')

        # Check coherence constraint
        total_wait = n_segments * t_segment
        if total_wait > platform.memory_coherence * 5:
            return 0.0

        # Swapping success
        p_swap = 0.5 ** (n_segments - 1)

        # End-to-end rate
        rate = p_seg * platform.clock_rate * p_swap / n_segments

        return rate

    def compare_platforms(self, distance: float) -> Dict:
        """Compare all platforms at given distance."""
        results = {}

        for name, platform in self.platforms.items():
            # Find optimal segment count
            best_rate = 0
            best_n = 2

            for n in range(2, 50):
                rate = self.rate_for_distance(platform, distance, n)
                if rate > best_rate:
                    best_rate = rate
                    best_n = n

            results[name] = {
                "rate": best_rate,
                "optimal_segments": best_n,
                "coherence": platform.memory_coherence,
                "maturity": platform.maturity
            }

        return results

# Compare platforms
analyzer = PlatformAnalyzer(platforms)

print("Hardware Platform Comparison")
print("=" * 70)

for distance in [100, 500]:
    print(f"\nDistance: {distance} km")
    comparison = analyzer.compare_platforms(distance)

    print(f"{'Platform':<20} {'Rate (Hz)':<15} {'Segments':<10} {'Maturity':<12}")
    print("-" * 60)

    for name, data in sorted(comparison.items(), key=lambda x: -x[1]["rate"]):
        rate_str = f"{data['rate']:.2e}" if data['rate'] > 0 else "0"
        print(f"{name:<20} {rate_str:<15} {data['optimal_segments']:<10} {data['maturity']:<12}")

## System Architecture

```python
@dataclass
class RepeaterNode:
    """Complete repeater node specification."""
    platform: str
    n_qubits: int
    memory_time: float
    gate_fidelity: float
    photon_rate: float
    detection_efficiency: float
    operating_temp: float
    power_consumption: float  # kW
    footprint: float  # m²

class NetworkArchitect:
    """Design repeater network architecture."""

    def __init__(self):
        pass

    def design_link(
        self,
        distance: float,
        target_rate: float,
        target_fidelity: float
    ) -> Dict:
        """Design a point-to-point repeater link."""
        # Determine number of segments
        n_segments = int(np.ceil(distance / 50))  # ~50 km segments

        # Node specifications
        node = RepeaterNode(
            platform="NV Centers",
            n_qubits=4,
            memory_time=1.0,
            gate_fidelity=0.99,
            photon_rate=1e6,
            detection_efficiency=0.9,
            operating_temp=4,
            power_consumption=5,
            footprint=2
        )

        # Estimate performance
        L_seg = distance / n_segments
        eta = 10**(-0.2 * L_seg / 10)

        achievable_rate = node.photon_rate * 0.01 * eta * (0.5 ** (n_segments - 1)) / n_segments

        return {
            "distance": distance,
            "n_segments": n_segments,
            "n_nodes": n_segments - 1,
            "node_spec": node,
            "estimated_rate": achievable_rate,
            "meets_target": achievable_rate >= target_rate,
            "total_power": node.power_consumption * (n_segments - 1),
            "deployment_cost_estimate": "high"
        }

    def network_topology(
        self,
        cities: List[str],
        distances: Dict
    ) -> Dict:
        """Design multi-city network topology."""
        links = []
        total_nodes = 0

        for (city1, city2), dist in distances.items():
            link = self.design_link(dist, target_rate=1, target_fidelity=0.9)
            links.append({
                "endpoints": (city1, city2),
                **link
            })
            total_nodes += link["n_nodes"]

        return {
            "cities": cities,
            "n_links": len(links),
            "total_nodes": total_nodes,
            "links": links
        }

# Design example network
architect = NetworkArchitect()

print("\nRepeater Network Design")
print("=" * 60)

link = architect.design_link(distance=500, target_rate=1, target_fidelity=0.9)

print(f"Link: 500 km")
print(f"  Segments: {link['n_segments']}")
print(f"  Repeater nodes: {link['n_nodes']}")
print(f"  Estimated rate: {link['estimated_rate']:.2e} Hz")
print(f"  Total power: {link['total_power']} kW")
print(f"  Platform: {link['node_spec'].platform}")
```
