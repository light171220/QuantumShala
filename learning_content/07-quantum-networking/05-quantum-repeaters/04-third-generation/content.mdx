# Third-Generation Repeaters

## Introduction

Third-generation quantum repeaters represent the ultimate goal of quantum networking: fully fault-tolerant quantum communication. By implementing complete quantum error correction at each node, these repeaters can achieve near-classical communication rates while maintaining quantum coherence over arbitrary distances.

## Fault-Tolerant Operations

```python
import numpy as np
from dataclasses import dataclass
from typing import Tuple, List, Dict

@dataclass
class SurfaceCode:
    """Surface code parameters for fault-tolerant operations."""
    distance: int
    physical_error_rate: float

    def logical_error_rate(self) -> float:
        """Calculate logical error rate per round."""
        p = self.physical_error_rate
        d = self.distance
        # Logical error scales as (p/p_th)^((d+1)/2)
        p_th = 0.01  # Approximate threshold
        if p >= p_th:
            return 0.5
        return ((p / p_th) ** ((d + 1) / 2))

    def physical_qubits(self) -> int:
        """Physical qubits needed for this code."""
        # 2d² for data and syndrome qubits
        return 2 * self.distance ** 2

    def gate_time(self) -> float:
        """Time for one syndrome measurement round (seconds)."""
        # Typical superconducting: ~1 μs per round
        return 1e-6

class ThirdGenRepeater:
    """
    Third-generation fault-tolerant quantum repeater.

    Each node is a small fault-tolerant quantum processor.
    """

    def __init__(
        self,
        n_segments: int = 10,
        total_distance: float = 1000,  # km
        code_distance: int = 17,
        physical_error_rate: float = 0.001
    ):
        self.n_segments = n_segments
        self.total_distance = total_distance
        self.segment_length = total_distance / n_segments

        self.code = SurfaceCode(code_distance, physical_error_rate)

        # Fiber parameters
        self.fiber_loss_db_km = 0.2

    def logical_bell_generation(self) -> Tuple[float, float]:
        """
        Generate fault-tolerant logical Bell pair.

        Uses transversal operations and magic state distillation.
        """
        # Physical transmission probability
        eta = 10**(-self.fiber_loss_db_km * self.segment_length / 10)

        # With fault tolerance, can retry until success
        # Average attempts needed
        n_attempts = 1 / eta

        # Time per attempt (code cycles)
        cycles_per_attempt = 10 * self.code.distance  # Approximate
        time_per_attempt = cycles_per_attempt * self.code.gate_time()

        # Total time
        generation_time = n_attempts * time_per_attempt

        # Logical fidelity (error corrected)
        p_L = self.code.logical_error_rate()
        fidelity = 1 - p_L  # Per logical qubit

        return 1.0, fidelity  # Always succeeds (fault-tolerant)

    def fault_tolerant_swapping(self, F1: float, F2: float) -> Tuple[float, float]:
        """
        Fault-tolerant Bell measurement and swapping.

        Deterministic with error correction.
        """
        # Deterministic success
        p_success = 1.0

        # Fidelity: logical gate errors only
        gate_error = self.code.logical_error_rate() * 10  # Multiple gates
        F_out = F1 * F2 * (1 - gate_error)

        return p_success, F_out

    def calculate_rate(self) -> float:
        """Calculate end-to-end rate."""
        # Segment generation time dominates
        eta = 10**(-self.fiber_loss_db_km * self.segment_length / 10)
        n_attempts = 1 / eta

        gate_time = self.code.gate_time()
        cycles_per_attempt = 10 * self.code.distance
        time_per_attempt = cycles_per_attempt * gate_time

        segment_time = n_attempts * time_per_attempt

        # Light travel time
        travel_time = self.segment_length / 2e5  # seconds

        total_time = segment_time + travel_time

        # Segments can be generated in parallel
        # But swapping is sequential: log2(N) levels
        n_swap_levels = int(np.ceil(np.log2(self.n_segments)))
        swap_time = n_swap_levels * 100 * gate_time  # Approximate

        end_to_end_time = total_time + swap_time
        return 1 / end_to_end_time

    def calculate_fidelity(self) -> float:
        """Calculate end-to-end fidelity."""
        p_L = self.code.logical_error_rate()

        # Fidelity per segment
        F_seg = 1 - p_L

        # After N-1 swaps, each with gate errors
        n_swaps = self.n_segments - 1
        gate_error_per_swap = p_L * 10

        F_final = (F_seg ** self.n_segments) * ((1 - gate_error_per_swap) ** n_swaps)

        return F_final

    def resource_requirements(self) -> Dict:
        """Calculate resources needed per node."""
        qubits_per_logical = self.code.physical_qubits()

        # Need multiple logical qubits per node
        # Communication, processing, ancillas
        logical_qubits_per_node = 10

        return {
            "physical_qubits_per_node": qubits_per_logical * logical_qubits_per_node,
            "code_distance": self.code.distance,
            "logical_qubits_per_node": logical_qubits_per_node,
            "total_physical_qubits": qubits_per_logical * logical_qubits_per_node * self.n_segments
        }

    def analyze(self) -> Dict:
        """Complete analysis."""
        return {
            "n_segments": self.n_segments,
            "total_distance": self.total_distance,
            "rate": self.calculate_rate(),
            "fidelity": self.calculate_fidelity(),
            "deterministic": True,
            "fault_tolerant": True,
            **self.resource_requirements()
        }

# Demonstrate third-generation repeater
print("Third-Generation Repeater Analysis")
print("=" * 60)

for distance in [500, 1000, 2000]:
    repeater = ThirdGenRepeater(
        n_segments=distance // 100,  # 100 km segments
        total_distance=distance,

    analysis = repeater.analyze()

    print(f"\n{distance} km:")

    print(f"  Fidelity: {analysis['fidelity']:.6f}")
    print(f"  Physical qubits/node: {analysis['physical_qubits_per_node']}")
```

## Comparison of All Generations

```python
class GenerationComparison:
    """Compare all three repeater generations."""

    def __init__(self, total_distance: float = 1000):
        self.total_distance = total_distance

    def first_gen(self, n_segments: int) -> Dict:
        """First-generation metrics."""
        L_seg = self.total_distance / n_segments
        eta = 10**(-0.2 * L_seg / 10)

        p_seg = 0.01 * eta
        p_swap = 0.5 ** (n_segments - 1)

        rate = 1e6 * p_seg * p_swap / n_segments
        fidelity = 0.95 ** n_segments * 0.99 ** (n_segments - 1)

        return {
            "generation": 1,
            "rate": rate,
            "fidelity": fidelity,
            "qubits_per_node": 2,  # One qubit each end
            "deterministic": False
        }

    def second_gen(self, n_segments: int, code_n: int = 7) -> Dict:
        """Second-generation metrics."""
        L_seg = self.total_distance / n_segments
        eta = 10**(-0.2 * L_seg / 10)

        p_seg = eta ** (code_n // 2 + 1)
        # Deterministic swapping
        rate = 1e6 * p_seg / n_segments
        fidelity = 0.999 ** n_segments * 0.999 ** (n_segments - 1)

        return {
            "generation": 2,
            "rate": rate,
            "fidelity": fidelity,
            "qubits_per_node": code_n * 4,
            "deterministic": True
        }

    def third_gen(self, n_segments: int, code_d: int = 17) -> Dict:
        """Third-generation metrics."""
        L_seg = self.total_distance / n_segments
        eta = 10**(-0.2 * L_seg / 10)

        # Always succeeds (fault tolerant retries)
        segment_time = (1/eta) * 1e-5  # ~10 μs per attempt
        rate = 1 / (segment_time + n_segments * 1e-6)

        # High fidelity from error correction
        p_L = 1e-10  # Very low logical error with d=17
        fidelity = (1 - p_L) ** (2 * n_segments)

        return {
            "generation": 3,
            "rate": rate,
            "fidelity": fidelity,
            "qubits_per_node": 2 * code_d**2 * 10,
            "deterministic": True
        }

    def compare_all(self, n_segments: int = 10) -> List[Dict]:
        """Compare all three generations."""
        return [
            self.first_gen(n_segments),
            self.second_gen(n_segments),
            self.third_gen(n_segments)
        ]

# Compare generations
comp = GenerationComparison(total_distance=1000)

print("\nAll Generations Comparison (1000 km)")
print("=" * 70)

results = comp.compare_all(n_segments=10)

print("\nGen | Rate (Hz)    | Fidelity     | Qubits/Node | Deterministic")
print("-" * 70)

for r in results:
    det = "Yes" if r["deterministic"] else "No"
    print(f" {r['generation']}  | {r['rate']:.2e}    | {r['fidelity']:.6f}   |"
          f"    {r['qubits_per_node']:5d}    |     {det}")
```
