# Memory Requirements

## Introduction

Quantum memories are essential components of quantum repeaters, enabling asynchronous operations and storing entanglement while waiting for other protocol steps. This lesson explores the requirements for quantum memories across different repeater generations and surveys current technology platforms.

## Role of Quantum Memories

```

├─────────────────────────────────────────────────────────────────────┤
│                                                                     │
```

## Coherence Time Requirements

```python
import numpy as np
from dataclasses import dataclass
from typing import Tuple, List, Dict

@dataclass
class MemoryPlatform:
    """Quantum memory platform specifications."""
    name: str
    coherence_time: float   # seconds
    efficiency: float       # write × read
    fidelity: float        # storage fidelity
    wavelength: float      # nm, interface wavelength
    operating_temp: float  # Kelvin

class MemoryRequirementsAnalyzer:
    """
    Analyze quantum memory requirements for repeaters.
    """

    def __init__(
        self,
        total_distance: float = 500,  # km
        fiber_loss_db_km: float = 0.2,
        source_rate: float = 1e6      # Hz
    ):
        self.total_distance = total_distance
        self.fiber_loss = fiber_loss_db_km
        self.source_rate = source_rate

    def segment_generation_time(self, n_segments: int) -> float:
        """Average time to generate one segment."""
        L_seg = self.total_distance / n_segments
        eta = 10**(-self.fiber_loss * L_seg / 10)

        # DLCZ-type probability
        p_success = 0.01 * eta

        # Average attempts needed
        n_attempts = 1 / p_success

        # Time per attempt
        attempt_time = 1 / self.source_rate

        return n_attempts * attempt_time

    def minimum_coherence_time(self, n_segments: int) -> float:
        """
        Minimum memory coherence time for protocol to work.

        Must store first segment until last segment ready.
        """
        t_segment = self.segment_generation_time(n_segments)

        # Need to store for up to n_segments generations
        # With some safety margin
        return n_segments * t_segment * 2

    def fidelity_after_storage(
        self,
        initial_fidelity: float,
        storage_time: float,
        coherence_time: float
    ) -> float:
        """Calculate fidelity after storage with decoherence."""
        # Exponential decay model
        decay = np.exp(-storage_time / coherence_time)

        # Fidelity approaches 0.5 (maximally mixed) as decay → 0
        F_out = initial_fidelity * decay + 0.5 * (1 - decay)

        return F_out

    def analyze_requirements(self, n_segments: int) -> Dict:
        """Complete memory requirements analysis."""
        t_gen = self.segment_generation_time(n_segments)
        t_min = self.minimum_coherence_time(n_segments)

        # Calculate fidelity for different coherence times
        fidelities = {}
        for T2 in [0.01, 0.1, 1.0, 10.0]:
            F = self.fidelity_after_storage(0.95, t_min / 2, T2)
            fidelities[f"T2={T2}s"] = F

        return {
            "n_segments": n_segments,
            "segment_gen_time": t_gen,
            "min_coherence_time": t_min,
            "fidelities": fidelities
        }

# Analyze requirements
analyzer = MemoryRequirementsAnalyzer(total_distance=500)

print("Quantum Memory Requirements Analysis")
print("=" * 60)

for n_seg in [4, 8, 16]:
    analysis = analyzer.analyze_requirements(n_seg)

    print(f"\n{n_seg} segments:")
    print(f"  Segment generation time: {analysis['segment_gen_time']*1000:.1f} ms")
    print(f"  Minimum coherence time: {analysis['min_coherence_time']*1000:.0f} ms")
    print(f"  Fidelity after storage (at T_min/2):")
    for key, F in analysis["fidelities"].items():
        print(f"    {key}: {F:.4f}")
```

## Memory Multiplexing

Multiplexing dramatically improves repeater performance:

```python
class MultiplexedMemory:
    """
    Quantum memory with multiplexing capability.

    Multiple temporal, spectral, or spatial modes
    can be used in parallel.
    """

    def __init__(
        self,
        n_modes: int = 100,
        coherence_time: float = 0.1,
        mode_efficiency: float = 0.9
    ):
        self.n_modes = n_modes
        self.coherence_time = coherence_time
        self.mode_efficiency = mode_efficiency

    def effective_success_probability(
        self,
        single_mode_prob: float
    ) -> float:
        """
        Probability at least one mode succeeds.

        P(≥1) = 1 - (1-p)^N
        """
        p_all_fail = (1 - single_mode_prob) ** self.n_modes
        p_success = 1 - p_all_fail

        # Efficiency of retrieving successful mode
        return p_success * self.mode_efficiency

    def rate_improvement(self, single_mode_prob: float) -> float:
        """Rate improvement factor from multiplexing."""
        p_single = single_mode_prob
        p_mux = self.effective_success_probability(single_mode_prob)
        return p_mux / p_single if p_single > 0 else 0

    def optimal_modes_vs_coherence(
        self,
        single_mode_prob: float,
        attempt_rate: float
    ) -> int:
        """Find optimal number of modes given coherence constraint."""
        best_n = 1
        best_rate = single_mode_prob * attempt_rate

        for n in range(1, 1000):
            # Time to try n modes
            time_per_round = n / attempt_rate

            if time_per_round > self.coherence_time:
                break

            # Success probability
            p_success = self.effective_success_probability(single_mode_prob)

            # Effective rate
            rate = p_success * attempt_rate / n

            if rate > best_rate:
                best_rate = rate
                best_n = n

        return best_n

# Demonstrate multiplexing advantage
print("\nMemory Multiplexing Analysis")
print("=" * 60)

for n_modes in [1, 10, 100, 1000]:
    mux = MultiplexedMemory(n_modes=n_modes)

    p_single = 0.01  # 1% single-mode success
    improvement = mux.rate_improvement(p_single)

    print(f"\n{n_modes} modes:")
    print(f"  Single-mode prob: {p_single*100:.1f}%")
    print(f"  Multiplexed prob: {mux.effective_success_probability(p_single)*100:.1f}%")
    print(f"  Rate improvement: {improvement:.1f}×")
```

## Requirements by Generation

```python
class GenerationMemoryRequirements:
    """Memory requirements for each repeater generation."""

    def first_generation(self, distance: float) -> Dict:
        """First-gen memory requirements."""
        # Need long coherence for synchronization
        n_segments = int(distance / 50)  # ~50 km segments
        L_seg = distance / n_segments
        eta = 10**(-0.2 * L_seg / 10)

        # Time to generate all segments
        p_seg = 0.01 * eta
        t_gen = n_segments / (1e6 * p_seg)

        return {
            "min_coherence": t_gen * 3,
            "multiplexing_benefit": "High",
            "efficiency_threshold": 0.5,
            "notes": "Long coherence critical; multiplexing helps"
        }

    def second_generation(self, distance: float) -> Dict:
        """Second-gen memory requirements."""
        return {
            "min_coherence": 0.001,  # Much shorter (deterministic)
            "multiplexing_benefit": "Low",
            "efficiency_threshold": 0.8,
            "notes": "Need encoded memory states; shorter coherence OK"
        }

    def third_generation(self, distance: float) -> Dict:
        """Third-gen memory requirements."""
        return {
            "min_coherence": 1e-6,  # Very short (continuous QEC)
            "multiplexing_benefit": "None",
            "efficiency_threshold": 0.99,
            "notes": "Memory integrated with processor; QEC protects"
        }

# Compare requirements
gen_reqs = GenerationMemoryRequirements()

print("\nMemory Requirements by Generation (500 km)")
print("=" * 60)

for gen, func in [("1st", gen_reqs.first_generation),
                   ("2nd", gen_reqs.second_generation),
                   ("3rd", gen_reqs.third_generation)]:
    reqs = func(500)
    print(f"\n{gen} Generation:")
    print(f"  Min coherence: {reqs['min_coherence']*1000:.2f} ms")
    print(f"  Multiplexing benefit: {reqs['multiplexing_benefit']}")
    print(f"  Efficiency threshold: {reqs['efficiency_threshold']}")
    print(f"  Notes: {reqs['notes']}")
```
