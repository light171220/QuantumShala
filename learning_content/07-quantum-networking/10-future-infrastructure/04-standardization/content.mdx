## Standards Categories

```python
import numpy as np
from dataclasses import dataclass
from typing import List, Dict
from enum import Enum

class StandardCategory(Enum):
    """Categories of quantum network standards."""
    SECURITY = "security"
    INTERFACE = "interface"
    PROTOCOL = "protocol"
    PHYSICAL = "physical"
    MANAGEMENT = "management"

class StandardStatus(Enum):
    """Status of standards work."""
    PUBLISHED = "published"
    IN_PROGRESS = "in_progress"
    PLANNED = "planned"

@dataclass
class QuantumStandard:
    """Quantum network standard."""
    identifier: str
    title: str
    body: str
    category: StandardCategory
    status: StandardStatus
    scope: str

class StandardsLandscape:
    """
    Overview of quantum network standards.
    """

    def __init__(self):
        self.standards = self._define_standards()

    def _define_standards(self) -> List[QuantumStandard]:
        """Define key standards."""
        return [
            QuantumStandard(
                "ETSI GS QKD 004",
                "Application Interface",
                "ETSI",
                StandardCategory.INTERFACE,
                StandardStatus.PUBLISHED,
                "Interface for applications to request keys from QKD"
            ),
            QuantumStandard(
                "ETSI GS QKD 018",
                "Orchestration Interface",
                "ETSI",
                StandardCategory.MANAGEMENT,
                StandardStatus.PUBLISHED,
                "Interface for SDN control of QKD networks"
            ),
            QuantumStandard(
                "ITU-T X.1710",
                "Security Framework",
                "ITU",
                StandardCategory.SECURITY,
                StandardStatus.PUBLISHED,
                "Security framework for QKD networks"
            ),
            QuantumStandard(
                "ITU-T X.1712",
                "Quantum Network Architecture",
                "ITU",
                StandardCategory.PROTOCOL,
                StandardStatus.PUBLISHED,
                "Functional architecture for QKD networks"
            ),
            QuantumStandard(
                "ISO/IEC 23837",
                "QKD Security Requirements",
                "ISO",
                StandardCategory.SECURITY,
                StandardStatus.PUBLISHED,
                "Security requirements for QKD modules"
            ),
            QuantumStandard(
                "IEEE P1913",
                "Software-Defined QC",
                "IEEE",
                StandardCategory.INTERFACE,
                StandardStatus.IN_PROGRESS,
                "SDN interfaces for quantum communication"
            ),
            QuantumStandard(
                "IETF QIRG",
                "Quantum Internet Architecture",
                "IETF",
                StandardCategory.PROTOCOL,
                StandardStatus.IN_PROGRESS,
                "Architectural principles for quantum internet"
            )
        ]

    def by_category(self) -> Dict[StandardCategory, List[QuantumStandard]]:
        """Group standards by category."""
        result = {}
        for std in self.standards:
            if std.category not in result:
                result[std.category] = []
            result[std.category].append(std)
        return result

    def gaps_analysis(self) -> List[str]:
        """Identify standardization gaps."""
        gaps = [
            "Quantum memory interfaces",
            "Repeater-to-repeater protocols",
            "Multi-vendor entanglement distribution",
            "Quantum network simulation test specs",
            "Long-distance fidelity requirements",
            "Quantum routing protocol standards"
        ]
        return gaps

    def print_landscape(self):
        """Print standards landscape."""
        print("Quantum Network Standards Landscape")
        print("=" * 50)

        by_cat = self.by_category()
        for cat, stds in by_cat.items():
            print(f"\n{cat.value.upper()}:")
            for std in stds:
                status = "✓" if std.status == StandardStatus.PUBLISHED else "◐"
                print(f"  {status} {std.identifier}")
                print(f"    {std.title} ({std.body})")

# Display standards
landscape = StandardsLandscape()
landscape.print_landscape()

print("\n\nStandardization Gaps:")
for gap in landscape.gaps_analysis():
    print(f"  • {gap}")
```

```python
@dataclass
class InteropRequirement:
    """Interoperability requirement."""
    layer: str
    requirement: str
    current_status: str
    priority: int  # 1-5

class InteroperabilityFramework:
    """
    Interoperability requirements framework.
    """

    def __init__(self):
        self.requirements = self._define_requirements()

    def _define_requirements(self) -> List[InteropRequirement]:
        """Define interop requirements."""
        return [
            InteropRequirement(
                "Application",
                "Standard key request/delivery API",
                "ETSI GS QKD 004 published",
                5
            ),
            InteropRequirement(
                "Application",
                "Common key format and metadata",
                "Partially standardized",
                4
            ),
            InteropRequirement(
                "Network",
                "Quantum routing protocol standard",
                "Research phase",
                3
            ),
            InteropRequirement(
                "Network",
                "Multi-domain path setup",
                "Not standardized",
                3
            ),
            InteropRequirement(
                "Link",
                "Entanglement distribution protocol",
                "Vendor-specific currently",
                4
            ),
            InteropRequirement(
                "Link",
                "Link-layer framing format",
                "Not standardized",
                3
            ),
            InteropRequirement(
                "Physical",
                "Standard wavelength plan",
                "Emerging (O-band for quantum)",
                4
            ),
            InteropRequirement(
                "Physical",
                "Timing sync requirements",
                "Partially addressed",
                5
            ),
            InteropRequirement(
                "Management",
                "Common monitoring MIB",
                "ETSI work in progress",
                4
            ),
            InteropRequirement(
                "Management",
                "SDN control interface",
                "ETSI GS QKD 018 published",
                5
            )
        ]

    def readiness_score(self) -> Dict[str, float]:
        """Calculate readiness score by layer."""
        scores = {}
        by_layer = {}

        for req in self.requirements:
            if req.layer not in by_layer:
                by_layer[req.layer] = []
            by_layer[req.layer].append(req)

        for layer, reqs in by_layer.items():
            standardized = sum(1 for r in reqs if "published" in r.current_status.lower())
            scores[layer] = standardized / len(reqs) if reqs else 0

        return scores

    def print_requirements(self):
        """Print requirements summary."""
        print("\n\nInteroperability Requirements")
        print("=" * 50)

        by_layer = {}
        for req in self.requirements:
            if req.layer not in by_layer:
                by_layer[req.layer] = []
            by_layer[req.layer].append(req)

        for layer, reqs in by_layer.items():
            print(f"\n{layer} Layer:")
            for req in reqs:
                priority = "★" * req.priority
                print(f"  [{priority}] {req.requirement}")
                print(f"      Status: {req.current_status}")

# Display interoperability
interop = InteroperabilityFramework()
interop.print_requirements()

print("\n\nReadiness by Layer:")
scores = interop.readiness_score()
for layer, score in scores.items():
    bar = "█" * int(score * 10) + "░" * (10 - int(score * 10))
    print(f"  {layer}: {bar} {score:.0%}")
```
