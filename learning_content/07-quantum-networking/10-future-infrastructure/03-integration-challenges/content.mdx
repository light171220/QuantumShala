## Quantum-Classical Coexistence

```python
import numpy as np
from dataclasses import dataclass
from typing import List, Dict, Tuple
from enum import Enum

class CoexistenceMode(Enum):
    """Modes for quantum-classical coexistence."""
    SEPARATE_FIBER = "separate_fiber"
    WAVELENGTH_MUX = "wavelength_multiplexing"
    TIME_MUX = "time_multiplexing"
    DEDICATED_DARK = "dedicated_dark_fiber"

@dataclass
class IntegrationChallenge:
    """Integration challenge specification."""
    name: str
    category: str
    severity: int  # 1-5
    description: str
    mitigation: List[str]
    cost_impact: str

class FiberCoexistence:
    """
    Model quantum-classical fiber coexistence.
    """

    def __init__(self):
        self.classical_power_dbm = 0  # dBm per channel
        self.quantum_sensitivity_dbm = -90  # dBm noise threshold
        self.fiber_raman_coefficient = 0.3e-3  # per km per mW

    def calculate_raman_noise(
        self,
        classical_channels: int,
        fiber_length_km: float
    ) -> float:
        """Calculate Raman scattering noise from classical channels."""
        classical_power_mw = 10 ** (self.classical_power_dbm / 10) * classical_channels
        raman_noise_mw = self.fiber_raman_coefficient * classical_power_mw * fiber_length_km
        raman_noise_dbm = 10 * np.log10(raman_noise_mw + 1e-15)
        return raman_noise_dbm

    def check_coexistence(
        self,
        classical_channels: int,
        quantum_wavelength_nm: float,
        classical_band: str,
        fiber_length_km: float
    ) -> Dict:
        """Check if quantum and classical can coexist."""
        raman_noise = self.calculate_raman_noise(classical_channels, fiber_length_km)

        # Spectral separation benefit
        if classical_band == "C-band" and quantum_wavelength_nm > 1560:
            separation_benefit = 10  # dB isolation from filters
        else:
            separation_benefit = 0

        effective_noise = raman_noise - separation_benefit

        feasible = effective_noise < self.quantum_sensitivity_dbm

        return {
            "feasible": feasible,
            "raman_noise_dbm": raman_noise,
            "effective_noise_dbm": effective_noise,
            "margin_db": self.quantum_sensitivity_dbm - effective_noise,
            "recommendation": self._recommend_mode(
                feasible, classical_channels, fiber_length_km
            )
        }

    def _recommend_mode(
        self,
        feasible: bool,
        channels: int,
        length: float
    ) -> CoexistenceMode:
        """Recommend coexistence mode."""
        if feasible and channels < 10 and length < 50:
            return CoexistenceMode.WAVELENGTH_MUX
        elif feasible and length < 20:
            return CoexistenceMode.TIME_MUX
        elif channels > 40:
            return CoexistenceMode.SEPARATE_FIBER
        else:
            return CoexistenceMode.DEDICATED_DARK

# Demonstrate coexistence analysis
print("Quantum-Classical Fiber Coexistence")
print("=" * 50)

coexistence = FiberCoexistence()

# Test scenarios
scenarios = [
    (8, 1550, "C-band", 30),
    (40, 1550, "C-band", 50),
    (80, 1310, "O-band", 20),
]

for channels, qwl, cband, length in scenarios:
    result = coexistence.check_coexistence(channels, qwl, cband, length)
    print(f"\n{channels} classical channels, {length}km:")
    print(f"  Feasible: {result['feasible']}")
    print(f"  Noise margin: {result['margin_db']:.1f} dB")
    print(f"  Recommendation: {result['recommendation'].value}")
```

```python
class IntegrationAssessment:
    """
    Assess integration challenges for deployment.
    """

    def __init__(self):
        self.challenges = self._define_challenges()

    def _define_challenges(self) -> List[IntegrationChallenge]:
        """Define common integration challenges."""
        return [
            IntegrationChallenge(
                name="Fiber Availability",
                category="Physical",
                severity=4,
                description="Obtaining dedicated or shared fiber routes",
                mitigation=[
                    "Use existing dark fiber",
                    "Co-locate with existing deployments",
                    "Wavelength multiplexing"
                ],
                cost_impact="High"
            ),
            IntegrationChallenge(
                name="Skills Gap",
                category="Operational",
                severity=4,
                description="Lack of quantum-trained personnel",
                mitigation=[
                    "Training programs",
                    "Vendor support contracts",
                    "Hire specialists"
                ],
                cost_impact="Medium"
            ),
            IntegrationChallenge(
                name="Timing Synchronization",
                category="Technical",
                severity=3,
                description="Synchronizing quantum operations across network",
                mitigation=[
                    "GPS disciplined clocks",
                    "Network time protocols",
                    "Dedicated timing channels"
                ],
                cost_impact="Medium"
            ),
            IntegrationChallenge(
                name="Classical Channel Interference",
                category="Technical",
                severity=3,
                description="Raman noise from co-propagating classical signals",
                mitigation=[
                    "Spectral filtering",
                    "Power management",
                    "Separate fibers"
                ],
                cost_impact="Medium"
            ),
            IntegrationChallenge(
                name="Security Compliance",
                category="Organizational",
                severity=4,
                description="Meeting regulatory and certification requirements",
                mitigation=[
                    "Early engagement with regulators",
                    "Third-party audits",
                    "Standards compliance"
                ],
                cost_impact="Medium"
            ),
            IntegrationChallenge(
                name="Cooling Requirements",
                category="Physical",
                severity=2,
                description="Cryogenic cooling for some components",
                mitigation=[
                    "Consolidated cooling systems",
                    "Room-temp alternatives where possible",
                    "Efficient cryo systems"
                ],
                cost_impact="Medium"
            )
        ]

    def assess_readiness(
        self,
        site_conditions: Dict[str, bool]
    ) -> Dict:
        """Assess integration readiness for a site."""
        challenges_remaining = []
        mitigations_needed = []

        for challenge in self.challenges:
            key = challenge.name.lower().replace(" ", "_")
            if not site_conditions.get(key, False):
                challenges_remaining.append(challenge)
                mitigations_needed.extend(challenge.mitigation[:1])

        return {
            "ready": len(challenges_remaining) == 0,
            "challenges_remaining": len(challenges_remaining),
            "high_severity": sum(1 for c in challenges_remaining if c.severity >= 4),
            "top_priorities": [c.name for c in challenges_remaining[:3]],
            "needed_mitigations": mitigations_needed[:5]
        }

    def print_challenges(self):
        """Print challenge summary."""
        print("\n\nIntegration Challenges")
        print("=" * 50)

        by_cat = {}
        for c in self.challenges:
            if c.category not in by_cat:
                by_cat[c.category] = []
            by_cat[c.category].append(c)

        for cat, challenges in by_cat.items():
            print(f"\n{cat}:")
            for c in challenges:
                severity_str = "★" * c.severity
                print(f"  • {c.name} [{severity_str}]")
                print(f"    {c.description}")

# Display challenges
assessment = IntegrationAssessment()
assessment.print_challenges()

# Example readiness assessment
print("\n\nSite Readiness Assessment:")
site = {
    "fiber_availability": True,
    "skills_gap": False,
    "timing_synchronization": True,
    "classical_channel_interference": False
}
result = assessment.assess_readiness(site)
print(f"  Ready: {result['ready']}")
print(f"  Challenges remaining: {result['challenges_remaining']}")
print(f"  High severity: {result['high_severity']}")
print(f"  Priorities: {result['top_priorities']}")
```
