## Research Areas

```python
import numpy as np
from dataclasses import dataclass
from typing import List, Dict
from enum import Enum

class ResearchArea(Enum):
    """Research areas in quantum networking."""
    MEMORY = "quantum_memory"
    REPEATER = "quantum_repeater"
    ENTANGLEMENT = "entanglement"
    PROTOCOLS = "protocols"
    HARDWARE = "hardware"
    THEORY = "theory"

class Difficulty(Enum):
    """Challenge difficulty level."""
    HARD = "hard"
    VERY_HARD = "very_hard"
    UNSOLVED = "unsolved"

@dataclass
class ResearchChallenge:
    """Open research challenge."""
    title: str
    area: ResearchArea
    difficulty: Difficulty
    description: str
    current_state: str
    breakthrough_needed: str
    potential_impact: str

class ResearchFrontiers:
    """
    Map of quantum networking research frontiers.
    """

    def __init__(self):
        self.challenges = self._define_challenges()

    def _define_challenges(self) -> List[ResearchChallenge]:
        """Define key research challenges."""
        return [
            ResearchChallenge(
                "Room-Temperature Quantum Memory",
                ResearchArea.MEMORY,
                Difficulty.VERY_HARD,
                "Quantum memory operating at room temperature",
                "Best results: warm vapor cells with short coherence",
                "New materials or protection schemes",
                "Eliminates cryogenic requirement, enables widespread deployment"
            ),
            ResearchChallenge(
                "Hour-Scale Memory with High Efficiency",
                ResearchArea.MEMORY,
                Difficulty.VERY_HARD,
                "Memories with hours of coherence and >90% efficiency",
                "Best: rare-earth ions (6h coherence, 50% efficiency)",
                "Overcome efficiency-coherence tradeoff",
                "Enables asynchronous global quantum communication"
            ),
            ResearchChallenge(
                "Practical Quantum Repeater",
                ResearchArea.REPEATER,
                Difficulty.VERY_HARD,
                "Quantum repeater that outperforms direct transmission",
                "Lab demonstrations, not yet beating direct link",
                "Improved memory + faster entanglement generation",
                "Extends quantum communication beyond ~100km"
            ),
            ResearchChallenge(
                "Error-Corrected Quantum Communication",
                ResearchArea.REPEATER,
                Difficulty.UNSOLVED,
                "Fault-tolerant repeaters using quantum error correction",
                "Theoretical frameworks, early experiments",
                "Low-overhead quantum error correction",
                "Unlimited-distance quantum communication"
            ),
            ResearchChallenge(
                "1000km Entanglement with High Fidelity",
                ResearchArea.ENTANGLEMENT,
                Difficulty.HARD,
                "Continental-scale entanglement >99% fidelity",
                "Satellite: 1200km, fiber limited to ~100km",
                "Repeater chain or improved satellite links",
                "Continental quantum networks"
            ),
            ResearchChallenge(
                "Multi-Party Entanglement Distribution",
                ResearchArea.ENTANGLEMENT,
                Difficulty.HARD,
                "Distribute GHZ/cluster states to many parties",
                "Limited lab demonstrations with few parties",
                "Scalable multi-party protocols",
                "Enables multiparty quantum protocols at scale"
            ),
            ResearchChallenge(
                "Practical Device-Independent QKD",
                ResearchArea.PROTOCOLS,
                Difficulty.VERY_HARD,
                "DI-QKD at usable key rates",
                "Very low rates, lab demonstrations only",
                "Overcome detection efficiency loophole at distance",
                "Minimal trust in equipment vendors"
            ),
            ResearchChallenge(
                "Quantum Network Stack",
                ResearchArea.PROTOCOLS,
                Difficulty.HARD,
                "Complete protocol stack for quantum internet",
                "Research protocols, not standardized",
                "Design and validate full stack",
                "Interoperable quantum internet"
            ),
            ResearchChallenge(
                "Scalable Integrated Quantum Nodes",
                ResearchArea.HARDWARE,
                Difficulty.HARD,
                "Chip-scale quantum network nodes",
                "Individual components integrated, not full nodes",
                "Heterogeneous integration of all components",
                "Mass-producible quantum network equipment"
            ),
            ResearchChallenge(
                "Quantum Network Security Proofs",
                ResearchArea.THEORY,
                Difficulty.HARD,
                "Composable security for full protocol stack",
                "QKD proven, higher layers incomplete",
                "Security frameworks for all protocols",
                "Provably secure quantum internet"
            )
        ]

    def by_area(self) -> Dict[ResearchArea, List[ResearchChallenge]]:
        """Group by research area."""
        result = {}
        for ch in self.challenges:
            if ch.area not in result:
                result[ch.area] = []
            result[ch.area].append(ch)
        return result

    def print_frontiers(self):
        """Print research frontiers."""
        print("Research Frontiers in Quantum Networking")
        print("=" * 60)

        by_area = self.by_area()
        for area, challenges in by_area.items():
            print(f"\n{area.value.upper().replace('_', ' ')}:")
            for ch in challenges:
                diff_icon = {"hard": "★", "very_hard": "★★", "unsolved": "★★★"}
                print(f"  [{diff_icon[ch.difficulty.value]}] {ch.title}")
                print(f"      Current: {ch.current_state[:50]}...")
                print(f"      Needed: {ch.breakthrough_needed}")

# Display research frontiers
frontiers = ResearchFrontiers()
frontiers.print_frontiers()
```

@dataclass
class Breakthrough:
    """Recent research breakthrough."""
    title: str
    year: int
    team: str
    significance: str
    next_steps: str

class RecentBreakthroughs:
    """
    Catalog of recent breakthroughs.

            )
        ]

    def print_breakthroughs(self):
        """Print recent breakthroughs."""
        print("\n\nRecent Breakthroughs")
        print("=" * 50)

        for b in sorted(self.breakthroughs, key=lambda x: x.year, reverse=True):
            print(f"\n[{b.year}] {b.title}")
            print(f"  Team: {b.team}")
            print(f"  Significance: {b.significance}")
            print(f"  Next: {b.next_steps}")

# Display breakthroughs
