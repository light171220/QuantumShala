# Photon Transmission for Quantum Networks

Photons are the primary carriers of quantum information in networks. Their properties make them ideal for long-distance quantum communication.

```python
import numpy as np
from dataclasses import dataclass
from typing import Tuple, Optional, List
from enum import Enum

class PhotonDOF(Enum):
    """Degrees of freedom for photonic qubits."""
    POLARIZATION = "polarization"
    TIME_BIN = "time_bin"
    FREQUENCY = "frequency"
    PATH = "path"
    OAM = "orbital_angular_momentum"

@dataclass
class Photon:
    """
    Model of a single photon for quantum communication.
    """
    wavelength_nm: float
    encoding: PhotonDOF
    state: np.ndarray  # 2D state vector

    # Physical properties
    energy_ev: float = None

    def __post_init__(self):
        # E = hc/λ
        h = 6.626e-34  # Planck constant (J·s)
        c = 3e8        # Speed of light (m/s)
        wavelength_m = self.wavelength_nm * 1e-9
        self.energy_ev = (h * c / wavelength_m) / 1.602e-19

        # Normalize state
        norm = np.linalg.norm(self.state)
        self.state = self.state / norm

    @property
    def frequency_thz(self) -> float:
        """Photon frequency in THz."""
        c = 3e8
        return c / (self.wavelength_nm * 1e-9) / 1e12

class PolarizationEncoding:
    """
    Polarization-based qubit encoding.
    """

    def __init__(self):
        # Basis states
        self.H = np.array([1, 0], dtype=complex)  # Horizontal |H⟩ = |0⟩
        self.V = np.array([0, 1], dtype=complex)  # Vertical |V⟩ = |1⟩

        # Diagonal basis
        self.D = (self.H + self.V) / np.sqrt(2)   # |D⟩ = |+⟩
        self.A = (self.H - self.V) / np.sqrt(2)   # |A⟩ = |-⟩

        # Circular basis
        self.R = (self.H + 1j * self.V) / np.sqrt(2)  # Right circular
        self.L = (self.H - 1j * self.V) / np.sqrt(2)  # Left circular

    def create_photon(self, state: str, wavelength: float = 1550) -> Photon:
        """Create polarization-encoded photon."""
        states = {
            'H': self.H, 'V': self.V,
            'D': self.D, 'A': self.A,
            'R': self.R, 'L': self.L
        }
        return Photon(
            wavelength_nm=wavelength,
            encoding=PhotonDOF.POLARIZATION,
            state=states[state]
        )

    def measure(self, photon: Photon, basis: str = 'HV') -> Tuple[int, Photon]:
        """
        Measure photon in specified basis.

        Args:
            photon: Photon to measure
            basis: 'HV' (rectilinear), 'DA' (diagonal), or 'RL' (circular)
        """
        if basis == 'HV':
            basis_states = [self.H, self.V]
        elif basis == 'DA':
            basis_states = [self.D, self.A]
        elif basis == 'RL':
            basis_states = [self.R, self.L]
        else:
            raise ValueError(f"Unknown basis: {basis}")

        # Calculate probabilities
        prob_0 = abs(np.vdot(basis_states[0], photon.state))**2

        # Collapse
        if np.random.random() < prob_0:
            outcome = 0
            new_state = basis_states[0]
        else:
            outcome = 1
            new_state = basis_states[1]

        collapsed_photon = Photon(
            wavelength_nm=photon.wavelength_nm,
            encoding=photon.encoding,
            state=new_state
        )

        return outcome, collapsed_photon

class TimeBinEncoding:
    """
    Time-bin qubit encoding for fiber transmission.
    """

    def __init__(self, time_separation_ps: float = 100):
        """
        Initialize time-bin encoding.

        Args:
            time_separation_ps: Time between early/late bins (picoseconds)
        """
        self.delta_t = time_separation_ps

        # |early⟩ = |0⟩, |late⟩ = |1⟩
        self.early = np.array([1, 0], dtype=complex)
        self.late = np.array([0, 1], dtype=complex)

    def create_superposition(self, phase: float = 0) -> np.ndarray:
        """Create (|early⟩ + e^{iφ}|late⟩)/√2."""
        return (self.early + np.exp(1j * phase) * self.late) / np.sqrt(2)

    def advantage_over_polarization(self) -> str:
        """Explain advantage for fiber transmission."""
        return """
TIME-BIN vs POLARIZATION IN FIBER:
═══════════════════════════════════════════════════════════════

Polarization issues in fiber:
• Birefringence causes polarization rotation
• Stress and bending change polarization
• Requires active polarization compensation

Time-bin advantages:
• Robust against polarization drift
• Self-compensating with Mach-Zehnder interferometer
• Better for long-distance fiber QKD

Time-bin challenges:
• Requires interferometric stability
• More complex detection setup
• Phase stability over path length
"""

# Demonstrate encodings
pol = PolarizationEncoding()
photon = pol.create_photon('D', wavelength=1550)
print(f"Created photon: λ={photon.wavelength_nm}nm, E={photon.energy_ev:.3f}eV")
print(f"State: {photon.state}")

# Measure in different bases
for basis in ['HV', 'DA']:
    outcome, _ = pol.measure(photon, basis)
    print(f"Measurement in {basis} basis: {outcome}")

tb = TimeBinEncoding()
print(tb.advantage_over_polarization())
```

```python
@dataclass
class OpticalFiber:
    """
    Model of optical fiber for quantum communication.
    """
    length_km: float
    core_diameter_um: float = 9.0      # Single-mode
    wavelength_nm: float = 1550.0
    attenuation_db_per_km: float = 0.2

    # Dispersion parameters
    dispersion_ps_per_nm_km: float = 17.0  # at 1550 nm
    pmd_ps_per_sqrt_km: float = 0.1        # Polarization mode dispersion

    def transmission_efficiency(self) -> float:
        """Calculate photon transmission probability."""
        loss_db = self.attenuation_db_per_km * self.length_km
        return 10 ** (-loss_db / 10)

    def photon_survival_probability(self) -> float:
        """Probability a single photon reaches the end."""
        return self.transmission_efficiency()

    def propagation_delay_us(self) -> float:
        """Calculate propagation delay in microseconds."""
        # Light speed in fiber ≈ c/n ≈ 2×10^8 m/s
        c_fiber = 2e8  # m/s
        distance_m = self.length_km * 1000
        return (distance_m / c_fiber) * 1e6  # μs

    def pulse_broadening_ps(self, bandwidth_nm: float = 0.1) -> float:
        """
        Calculate pulse broadening due to chromatic dispersion.

        Args:
            bandwidth_nm: Source spectral width
        """
        return abs(self.dispersion_ps_per_nm_km * bandwidth_nm * self.length_km)

    def pmd_broadening_ps(self) -> float:
        """Calculate PMD-induced pulse broadening."""
        return self.pmd_ps_per_sqrt_km * np.sqrt(self.length_km)

class FiberLink:
    """
    Complete fiber link for quantum communication.
    """

    def __init__(self, fiber: OpticalFiber):
        self.fiber = fiber
        self.connector_loss_db = 0.3  # Per connector
        self.splice_loss_db = 0.1     # Per splice
        self.num_connectors = 2
        self.num_splices = int(fiber.length_km / 5)  # Splice every 5 km

    def total_loss_db(self) -> float:
        """Calculate total link loss."""
        fiber_loss = self.fiber.attenuation_db_per_km * self.fiber.length_km
        connector_loss = self.connector_loss_db * self.num_connectors
        splice_loss = self.splice_loss_db * self.num_splices
        return fiber_loss + connector_loss + splice_loss

    def effective_transmission(self) -> float:
        """Calculate effective transmission including all losses."""
        return 10 ** (-self.total_loss_db() / 10)

    def max_qkd_distance(self, min_detection_rate: float = 10) -> float:
        """
        Calculate maximum QKD distance for given detection rate.

        Args:
            min_detection_rate: Minimum acceptable detections per second
        """
        # Assume 1 MHz source, 10% detector efficiency
        source_rate = 1e6
        detector_efficiency = 0.1

        # Rate = source × transmission × detector_eff
        # Solve for transmission
        required_transmission = min_detection_rate / (source_rate * detector_efficiency)

        # Convert to distance
        if required_transmission >= 1:
            return float('inf')

        loss_db = -10 * np.log10(required_transmission)
        max_distance = loss_db / self.fiber.attenuation_db_per_km

        return max_distance

    def link_budget_report(self) -> str:
        """Generate detailed link budget report."""
        return f"""
FIBER LINK BUDGET
═══════════════════════════════════════════════════════════════

Link Parameters:
  Length: {self.fiber.length_km} km
  Wavelength: {self.fiber.wavelength_nm} nm

Loss Budget:
  Fiber attenuation: {self.fiber.attenuation_db_per_km * self.fiber.length_km:.2f} dB
  Connector losses:  {self.connector_loss_db * self.num_connectors:.2f} dB ({self.num_connectors} connectors)
  Splice losses:     {self.splice_loss_db * self.num_splices:.2f} dB ({self.num_splices} splices)
  ─────────────────────────────────────
  Total loss:        {self.total_loss_db():.2f} dB

Transmission:
  Efficiency: {self.effective_transmission():.6f} ({self.effective_transmission()*100:.4f}%)

Timing:
  Propagation delay: {self.fiber.propagation_delay_us():.2f} μs
  Pulse broadening:  {self.fiber.pulse_broadening_ps():.2f} ps

QKD Estimate:
  Max distance (10 Hz min): {self.max_qkd_distance():.1f} km
"""

# Example link analysis
fiber = OpticalFiber(length_km=50, wavelength_nm=1550)
link = FiberLink(fiber)
print(link.link_budget_report())
```

## Single-Photon Sources and Detectors

```
Single-Photon Source Technologies:
═══════════════════════════════════════════════════════════════════

Weak Coherent Pulse (WCP):
┌─────────────────────────────────────────────────────────────────┐
│ • Attenuated laser pulses                                       │
│ • Mean photon number μ ≈ 0.1-0.5                               │
│ • Poisson distribution: P(n) = μⁿe^(-μ)/n!                     │
│ • Some pulses have 0, 2, 3+ photons                            │
│ • Decoy-state protocol mitigates multi-photon attacks          │
│                                                                  │
│ Pros: Simple, high rate, room temperature                      │
│ Cons: Non-deterministic, multi-photon vulnerability            │
└─────────────────────────────────────────────────────────────────┘

Spontaneous Parametric Down-Conversion (SPDC):
┌─────────────────────────────────────────────────────────────────┐
│ • Nonlinear crystal converts pump photon to signal+idler pair  │
│ • Heralded single photons: detect one, know other exists       │
│ • Creates entangled photon pairs                               │
│                                                                  │
│     Pump (405 nm)                                               │
│         │                                                        │
│         ▼                                                        │
│    ┌─────────┐                                                  │
│    │   BBO   │──▶ Signal (810 nm)                               │
│    │ Crystal │                                                   │
│    └─────────┘──▶ Idler (810 nm)                                │
│                                                                  │
│ Pros: Heralded, entanglement source                            │
│ Cons: Probabilistic, limited rate                              │
└─────────────────────────────────────────────────────────────────┘

Quantum Dots:
┌─────────────────────────────────────────────────────────────────┐
│ • Semiconductor nanostructures                                  │
│ • True single-photon emission                                   │
│ • Deterministic (on-demand)                                     │
│ • Requires cryogenic cooling                                    │
│                                                                  │
│ Pros: On-demand, high purity                                   │
│ Cons: Cryogenic, wavelength matching challenges                │
└─────────────────────────────────────────────────────────────────┘
```

```python
class SinglePhotonSource:
    """
    Model of single-photon sources for QKD.
    """

    def __init__(self, source_type: str, mean_photon_number: float = 0.1):
        """
        Initialize source.

        Args:
            source_type: 'wcp', 'spdc', or 'quantum_dot'
            mean_photon_number: μ for WCP sources
        """
        self.source_type = source_type
        self.mu = mean_photon_number

    def photon_number_distribution(self, max_n: int = 5) -> dict:
        """
        Calculate photon number probabilities.
        """
        if self.source_type == 'wcp':
            # Poisson distribution
            probs = {}
            for n in range(max_n + 1):
                probs[n] = (self.mu ** n * np.exp(-self.mu)) / np.math.factorial(n)
            return probs

        elif self.source_type == 'quantum_dot':
            # Ideal single-photon (Fock state |1⟩)
            return {0: 0.0, 1: 1.0, 2: 0.0}

        elif self.source_type == 'spdc':
            # Thermal-like distribution for unheralded
            # Approximately Poisson when heralded
            return {0: 0.1, 1: 0.85, 2: 0.05}

    def g2_zero(self) -> float:
        """
        Calculate second-order correlation g²(0).

        g²(0) = 0: Perfect single-photon source
        g²(0) = 1: Coherent state (laser)
        g²(0) = 2: Thermal light
        """
        if self.source_type == 'wcp':
            return 1.0  # Coherent state
        elif self.source_type == 'quantum_dot':
            return 0.02  # Near-ideal (small multi-photon)
        elif self.source_type == 'spdc':
            return 0.01  # Heralded: very low

class SinglePhotonDetector:
    """
    Model of single-photon detectors.
    """

    def __init__(self, detector_type: str):
        """
        Initialize detector.

        Args:
            detector_type: 'apd', 'snspd', or 'pmt'
        """
        self.detector_type = detector_type
        self._set_parameters()

    def _set_parameters(self):
        """Set detector parameters based on type."""
        if self.detector_type == 'apd':
            # Avalanche Photodiode (InGaAs)
            self.efficiency = 0.25        # 25% at 1550 nm
            self.dark_count_rate = 1e3    # 1 kHz
            self.timing_jitter_ps = 200
            self.dead_time_ns = 50
            self.operating_temp = 223     # -50°C (cooled)

        elif self.detector_type == 'snspd':
            # Superconducting Nanowire
            self.efficiency = 0.93        # 93% at 1550 nm
            self.dark_count_rate = 10     # 10 Hz
            self.timing_jitter_ps = 20
            self.dead_time_ns = 10
            self.operating_temp = 2.5     # 2.5 K (cryo)

        elif self.detector_type == 'pmt':
            # Photomultiplier Tube
            self.efficiency = 0.40        # At visible wavelengths
            self.dark_count_rate = 100
            self.timing_jitter_ps = 500
            self.dead_time_ns = 20
            self.operating_temp = 300     # Room temp

    def detect(self, photon_present: bool) -> bool:
        """
        Simulate detection attempt.

        Args:
            photon_present: Whether a photon arrived

        Returns:
            Whether detector clicked
        """
        if photon_present:
            # True positive with probability = efficiency
            return np.random.random() < self.efficiency
        else:
            # False positive (dark count)
            # Approximate: dark count rate × time window
            dark_prob = self.dark_count_rate * 1e-9  # 1 ns window
            return np.random.random() < dark_prob

    def comparison_table(self) -> str:
        """Generate detector comparison."""
        return f"""
SINGLE-PHOTON DETECTOR: {self.detector_type.upper()}
═══════════════════════════════════════════════════════════════

Efficiency:        {self.efficiency*100:.1f}%
Dark count rate:   {self.dark_count_rate:.0f} Hz
Timing jitter:     {self.timing_jitter_ps:.0f} ps
Dead time:         {self.dead_time_ns:.0f} ns
Operating temp:    {self.operating_temp:.1f} K
"""

# Compare sources and detectors
print("=== Single-Photon Sources ===")
for src_type in ['wcp', 'quantum_dot', 'spdc']:
    source = SinglePhotonSource(src_type)
    dist = source.photon_number_distribution()
    print(f"\n{src_type.upper()}: g²(0) = {source.g2_zero():.2f}")
    print(f"  P(n=0) = {dist[0]:.3f}, P(n=1) = {dist[1]:.3f}, P(n≥2) = {1-dist[0]-dist[1]:.3f}")

print("\n=== Single-Photon Detectors ===")
for det_type in ['apd', 'snspd']:
    det = SinglePhotonDetector(det_type)
    print(det.comparison_table())
```

## Summary

Photon transmission fundamentals:

1. **Photon properties**: Ideal carriers due to speed and low decoherence
2. **Encoding schemes**: Polarization, time-bin, frequency each with trade-offs
3. **Fiber characteristics**: 0.2 dB/km loss at 1550 nm limits direct transmission to ~100 km
4. **Sources**: WCP with decoy states most practical; quantum dots for future
5. **Detectors**: SNSPDs offer best performance but require cryogenics

Understanding photon transmission is essential for designing practical quantum networks.
