# Classical Networks Overview

Before diving into quantum networking, we must understand classical networks—their architecture, protocols, and fundamental limitations that quantum networks aim to address.

```python
import numpy as np
from dataclasses import dataclass
from typing import List, Dict, Optional
from enum import Enum

class OSILayer(Enum):
    """OSI Model layers."""
    PHYSICAL = 1
    DATA_LINK = 2
    NETWORK = 3
    TRANSPORT = 4
    SESSION = 5
    PRESENTATION = 6
    APPLICATION = 7

@dataclass
class NetworkPacket:
    """Classical network packet representation."""
    source: str
    destination: str
    payload: bytes
    protocol: str
    layer: OSILayer

    def get_header_size(self) -> int:
        """Return typical header size in bytes."""
        header_sizes = {
            OSILayer.PHYSICAL: 8,      # Preamble
            OSILayer.DATA_LINK: 18,    # Ethernet header
            OSILayer.NETWORK: 20,      # IP header (minimum)
            OSILayer.TRANSPORT: 20,    # TCP header (minimum)
        }
        return header_sizes.get(self.layer, 0)

class ClassicalNetwork:
    """
    Model of a classical network for comparison with quantum networks.
    """

    def __init__(self, name: str):
        self.name = name
        self.nodes: Dict[str, dict] = {}
        self.links: List[tuple] = []
        self.routing_table: Dict[str, str] = {}

    def add_node(self, node_id: str, node_type: str = "router"):
        """Add a network node."""
        self.nodes[node_id] = {
            "type": node_type,
            "neighbors": [],
            "ip_address": f"192.168.1.{len(self.nodes) + 1}"
        }

    def add_link(self, node1: str, node2: str,
                 bandwidth_mbps: float, latency_ms: float):
        """Add a network link between nodes."""
        self.links.append({
            "endpoints": (node1, node2),
            "bandwidth": bandwidth_mbps,
            "latency": latency_ms,
            "type": "classical"
        })
        self.nodes[node1]["neighbors"].append(node2)
        self.nodes[node2]["neighbors"].append(node1)

    def calculate_path_latency(self, path: List[str]) -> float:
        """Calculate total latency along a path."""
        total_latency = 0
        for i in range(len(path) - 1):
            for link in self.links:
                if set(link["endpoints"]) == {path[i], path[i+1]}:
                    total_latency += link["latency"]
                    break
        return total_latency

    def transmit_data(self, source: str, dest: str,
                      data_size_mb: float) -> dict:
        """Simulate classical data transmission."""
        # Find path (simplified - direct neighbor or via routing)
        if dest in self.nodes[source]["neighbors"]:
            path = [source, dest]
        else:
            # In practice, use routing algorithms
            path = self._find_path(source, dest)

        latency = self.calculate_path_latency(path)

        # Calculate transmission time
        min_bandwidth = float('inf')
        for i in range(len(path) - 1):
            for link in self.links:
                if set(link["endpoints"]) == {path[i], path[i+1]}:
                    min_bandwidth = min(min_bandwidth, link["bandwidth"])

        transmission_time = (data_size_mb * 8) / min_bandwidth * 1000  # ms

        return {
            "path": path,
            "latency_ms": latency,
            "transmission_time_ms": transmission_time,
            "total_time_ms": latency + transmission_time,
            "security": "encryption_dependent"
        }

    def _find_path(self, source: str, dest: str) -> List[str]:
        """Simple BFS path finding."""
        from collections import deque

        visited = {source}
        queue = deque([(source, [source])])

        while queue:
            node, path = queue.popleft()
            if node == dest:
                return path

            for neighbor in self.nodes[node]["neighbors"]:
                if neighbor not in visited:
                    visited.add(neighbor)
                    queue.append((neighbor, path + [neighbor]))

        return []

# Create example classical network
network = ClassicalNetwork("Enterprise Network")
network.add_node("HQ", "datacenter")
network.add_node("Branch1", "router")
network.add_node("Branch2", "router")
network.add_node("ISP", "router")

network.add_link("HQ", "ISP", bandwidth_mbps=1000, latency_ms=5)
network.add_link("ISP", "Branch1", bandwidth_mbps=100, latency_ms=20)
network.add_link("ISP", "Branch2", bandwidth_mbps=100, latency_ms=25)

result = network.transmit_data("HQ", "Branch1", data_size_mb=1.0)
print(f"Transmission result: {result}")
```

```python
@dataclass
class TCPSegment:
    """TCP segment structure."""
    source_port: int
    dest_port: int
    sequence_number: int
    ack_number: int
    flags: Dict[str, bool]
    window_size: int
    payload: bytes

    def __post_init__(self):
        if self.flags is None:
            self.flags = {
                "SYN": False, "ACK": False, "FIN": False,
                "RST": False, "PSH": False, "URG": False
            }

class TCPConnection:
    """
    Simulates TCP connection establishment and data transfer.
    """

    def __init__(self, client_port: int, server_port: int):
        self.client_port = client_port
        self.server_port = server_port
        self.state = "CLOSED"
        self.sequence_number = np.random.randint(0, 2**32)
        self.ack_number = 0

    def three_way_handshake(self) -> List[TCPSegment]:
        """Simulate TCP 3-way handshake."""
        segments = []

        # Step 1: Client sends SYN
        syn = TCPSegment(
            source_port=self.client_port,
            dest_port=self.server_port,
            sequence_number=self.sequence_number,
            ack_number=0,
            flags={"SYN": True, "ACK": False, "FIN": False,
                   "RST": False, "PSH": False, "URG": False},
            window_size=65535,
            payload=b""
        )
        segments.append(("Client → Server: SYN", syn))
        self.state = "SYN_SENT"

        # Step 2: Server responds with SYN-ACK
        server_seq = np.random.randint(0, 2**32)
        syn_ack = TCPSegment(
            source_port=self.server_port,
            dest_port=self.client_port,
            sequence_number=server_seq,
            ack_number=self.sequence_number + 1,
            flags={"SYN": True, "ACK": True, "FIN": False,
                   "RST": False, "PSH": False, "URG": False},
            window_size=65535,
            payload=b""
        )
        segments.append(("Server → Client: SYN-ACK", syn_ack))
        self.state = "SYN_RECEIVED"

        # Step 3: Client sends ACK
        ack = TCPSegment(
            source_port=self.client_port,
            dest_port=self.server_port,
            sequence_number=self.sequence_number + 1,
            ack_number=server_seq + 1,
            flags={"SYN": False, "ACK": True, "FIN": False,
                   "RST": False, "PSH": False, "URG": False},
            window_size=65535,
            payload=b""
        )
        segments.append(("Client → Server: ACK", ack))
        self.state = "ESTABLISHED"

        return segments

# Demonstrate TCP handshake
tcp = TCPConnection(client_port=54321, server_port=443)
handshake = tcp.three_way_handshake()

for description, segment in handshake:
    print(f"{description}")
    print(f"  Seq: {segment.sequence_number}, Ack: {segment.ack_number}")
    print(f"  Flags: SYN={segment.flags['SYN']}, ACK={segment.flags['ACK']}")
    print()

print(f"Connection state: {tcp.state}")
```

## Network Metrics and Performance

```python
@dataclass
class NetworkMetrics:
    """Network performance metrics."""
    latency_ms: float
    bandwidth_mbps: float
    packet_loss_rate: float
    jitter_ms: float

    def calculate_throughput(self, packet_size_bytes: int = 1500) -> float:
        """Calculate effective throughput considering packet loss."""
        bits_per_packet = packet_size_bytes * 8
        packets_per_second = (self.bandwidth_mbps * 1e6) / bits_per_packet
        effective_pps = packets_per_second * (1 - self.packet_loss_rate)
        return (effective_pps * bits_per_packet) / 1e6  # Mbps

    def calculate_rtt(self, distance_km: float,
                      propagation_speed: float = 200000) -> float:
        """
        Calculate round-trip time.

        Args:
            distance_km: Distance in kilometers
            propagation_speed: Speed in km/s (fiber ≈ 200,000 km/s)
        """
        propagation_delay = (distance_km / propagation_speed) * 1000  # ms
        return 2 * propagation_delay + 2 * self.latency_ms

class ClassicalNetworkAnalysis:
    """
    Comprehensive classical network analysis.
    """

    def __init__(self):
        self.fiber_attenuation_db_per_km = 0.2  # Typical single-mode fiber
        self.amplifier_spacing_km = 80

    def calculate_signal_attenuation(self, distance_km: float) -> float:
        """Calculate signal attenuation over fiber."""
        return self.fiber_attenuation_db_per_km * distance_km

    def amplifiers_needed(self, distance_km: float) -> int:
        """Calculate number of optical amplifiers needed."""
        return max(0, int(distance_km / self.amplifier_spacing_km) - 1)

    def compare_with_quantum(self, distance_km: float) -> dict:
        """
        Compare classical and quantum network capabilities.
        """
        # Classical network
        classical_attenuation = self.calculate_signal_attenuation(distance_km)
        amplifiers = self.amplifiers_needed(distance_km)

        # Quantum network (no amplification possible!)
        quantum_attenuation = classical_attenuation  # Same fiber loss
        max_qkd_distance = 100  # km without repeaters
        repeaters_needed = max(0, int(distance_km / max_qkd_distance) - 1)

        return {
            "distance_km": distance_km,
            "classical": {
                "attenuation_db": classical_attenuation,
                "amplifiers_needed": amplifiers,
                "feasible": True,  # Always feasible with amplifiers
                "security": "computational"
            },
            "quantum": {
                "attenuation_db": quantum_attenuation,
                "repeaters_needed": repeaters_needed,
                "feasible": repeaters_needed == 0 or "with_repeaters",
                "security": "information_theoretic"
            }
        }

# Analysis example
analysis = ClassicalNetworkAnalysis()
for distance in [50, 100, 500, 1000]:
    result = analysis.compare_with_quantum(distance)
    print(f"\nDistance: {distance} km")
    print(f"  Classical: {result['classical']['amplifiers_needed']} amplifiers")
    print(f"  Quantum: {result['quantum']['repeaters_needed']} repeaters needed")
```

## Summary

Classical networks provide the foundation for understanding quantum networks:

1. **Architecture**: OSI/TCP-IP layers provide structured communication
2. **Protocols**: TCP/IP enables reliable global connectivity
3. **Security limitations**: Eavesdropping is undetectable; key distribution relies on computational hardness
4. **Quantum advantage**: QKD offers information-theoretic security unavailable classically

The limitations of classical networks—particularly in security—motivate the development of quantum networks.
