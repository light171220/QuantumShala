# Noise and Loss in Quantum Channels

Quantum information is fragile. Understanding noise and loss mechanisms is essential for designing robust quantum networks.

```python
import numpy as np
from dataclasses import dataclass
from typing import Tuple, List

@dataclass
class NoiseSource:
    """Model of a noise source in quantum channel."""
    name: str
    error_type: str  # 'loss', 'bit_flip', 'phase_flip', 'depolarizing'
    probability: float
    description: str

class QuantumChannelNoise:
    """
    Comprehensive noise model for quantum channels.
    """

    def __init__(self):
        self.noise_sources: List[NoiseSource] = []
        self._initialize_sources()

    def _initialize_sources(self):
        """Initialize common noise sources."""
        self.noise_sources = [
            NoiseSource(
                "Fiber attenuation",
                "loss",
                0.0,  # Calculated per km
                "Absorption and scattering in optical fiber"
            ),
            NoiseSource(
                "Detector dark counts",
                "depolarizing",
                1e-6,
                "Spontaneous avalanche events in APD/SNSPD"
            ),
            NoiseSource(
                "Polarization rotation",
                "bit_flip",
                0.01,
                "Birefringence in fiber causing polarization drift"
            ),
            NoiseSource(
                "Phase noise",
                "phase_flip",
                0.005,
                "Thermal fluctuations causing phase drift"
            ),
            NoiseSource(
                "Timing jitter",
                "depolarizing",
                0.001,
                "Electronic noise causing detection timing errors"
            ),
        ]

    def calculate_fiber_loss(
        self,
        distance_km: float,
        wavelength_nm: float = 1550
    ) -> float:
        """Calculate loss probability for fiber."""
        # Attenuation coefficient (dB/km)
        if wavelength_nm == 1550:
            alpha = 0.2
        elif wavelength_nm == 1310:
            alpha = 0.35
        else:
            alpha = 0.5  # Conservative estimate

        loss_db = alpha * distance_km
        survival_prob = 10 ** (-loss_db / 10)
        return 1 - survival_prob  # Loss probability

    def total_error_rate(self, distance_km: float) -> float:
        """Calculate total error rate from all sources."""
        # Start with base errors
        total_error = 0

        for source in self.noise_sources:
            if source.error_type != 'loss':
                total_error += source.probability

        # Loss contributes to effective error through dark counts
        loss_prob = self.calculate_fiber_loss(distance_km)
        signal_prob = 1 - loss_prob

        # Dark count contribution to QBER
        dark_rate = 1e-6  # Per detection window
        dark_qber = 0.5 * dark_rate / (signal_prob + dark_rate)

        return total_error + dark_qber

class DecoherenceModel:
    """
    Model decoherence in quantum systems.
    """

    def __init__(self, T1_us: float, T2_us: float):
        """
        Initialize decoherence model.

        Args:
            T1_us: Relaxation time in microseconds
            T2_us: Dephasing time in microseconds
        """
        self.T1 = T1_us
        self.T2 = T2_us

    def amplitude_damping_probability(self, time_us: float) -> float:
        """Calculate amplitude damping probability."""
        return 1 - np.exp(-time_us / self.T1)

    def dephasing_probability(self, time_us: float) -> float:
        """Calculate dephasing probability."""
        # T2 includes both T1 and pure dephasing contributions
        # 1/T2 = 1/(2*T1) + 1/T_phi
        return 1 - np.exp(-time_us / self.T2)

    def apply_decoherence(
        self,
        rho: np.ndarray,
        time_us: float
    ) -> np.ndarray:
        """
        Apply decoherence to density matrix.

        Args:
            rho: 2x2 density matrix
            time_us: Evolution time

        Returns:
            Decohered density matrix
        """
        gamma1 = self.amplitude_damping_probability(time_us)
        gamma2 = self.dephasing_probability(time_us)

        # Apply amplitude damping
        rho_00 = rho[0, 0] + gamma1 * rho[1, 1]
        rho_11 = (1 - gamma1) * rho[1, 1]
        rho_01 = np.sqrt(1 - gamma1) * (1 - gamma2) * rho[0, 1]
        rho_10 = np.sqrt(1 - gamma1) * (1 - gamma2) * rho[1, 0]

        return np.array([
            [rho_00, rho_01],
            [rho_10, rho_11]
        ], dtype=complex)

    def fidelity_decay(self, time_us: float, initial_state: str = 'plus') -> float:
        """
        Calculate fidelity decay for common initial states.

        Args:
            time_us: Evolution time
            initial_state: 'zero', 'one', 'plus', 'minus'
        """
        if initial_state == 'zero':
            # |0⟩ is stable under amplitude damping
            return 1.0

        elif initial_state == 'one':
            # |1⟩ decays to |0⟩
            gamma = self.amplitude_damping_probability(time_us)
            return 1 - gamma

        elif initial_state in ['plus', 'minus']:
            # Superposition affected by both T1 and T2
            gamma1 = self.amplitude_damping_probability(time_us)
            gamma2 = self.dephasing_probability(time_us)

            # Fidelity with initial |+⟩ state
            coherence = np.sqrt(1 - gamma1) * (1 - gamma2)
            pop_0 = 0.5 + 0.5 * gamma1
            pop_1 = 0.5 - 0.5 * gamma1

            return 0.5 * (pop_0 + pop_1 + 2 * coherence * 0.5)

        return 0.5
```

## Error Models and QBER

```
Quantum Bit Error Rate (QBER) Components:
═══════════════════════════════════════════════════════════════════

QBER = e_opt + e_det + e_dark + e_timing

where:
┌─────────────────────────────────────────────────────────────────┐
│ Component    │ Formula              │ Typical Value            │
├──────────────┼──────────────────────┼──────────────────────────┤
│ e_opt        │ Optical misalignment │ 0.5% - 2%                │
│ e_det        │ Detector imperfection│ 0.1% - 1%                │
│ e_dark       │ Dark count errors    │ Distance-dependent       │
│ e_timing     │ Timing window errors │ 0.1% - 0.5%              │
└──────────────┴──────────────────────┴──────────────────────────┘

Dark Count Contribution:
┌─────────────────────────────────────────────────────────────────┐
│                                                                  │
│  e_dark = 0.5 × (D × Δt) / (η × μ + D × Δt)                    │
│                                                                  │
│  where:                                                          │
│    D = dark count rate (Hz)                                     │
│    Δt = detection window (s)                                    │
│    η = channel transmission                                     │
│    μ = mean photon number                                       │
│                                                                  │
│  At long distances (η → 0): e_dark → 0.5 (maximum)              │
│                                                                  │
└─────────────────────────────────────────────────────────────────┘
```

```python
@dataclass
class QBERAnalysis:
    """
    Analyze QBER components in QKD.
    """

    optical_error: float = 0.01      # 1%
    detector_error: float = 0.005    # 0.5%
    timing_error: float = 0.002      # 0.2%
    dark_count_rate: float = 1e3     # 1 kHz
    detection_window: float = 1e-9   # 1 ns
    mean_photon_number: float = 0.1  # WCP source

    def dark_count_error(self, transmission: float) -> float:
        """Calculate QBER contribution from dark counts."""
        if transmission <= 0:
            return 0.5

        signal = transmission * self.mean_photon_number
        dark = self.dark_count_rate * self.detection_window

        return 0.5 * dark / (signal + dark)

    def total_qber(self, transmission: float) -> float:
        """Calculate total QBER."""
        e_dark = self.dark_count_error(transmission)
        return self.optical_error + self.detector_error + self.timing_error + e_dark

    def qber_vs_distance(
        self,
        max_distance: float = 200,
        attenuation: float = 0.2
    ) -> List[Tuple[float, float]]:
        """Calculate QBER vs distance."""
        results = []
        for d in np.linspace(0, max_distance, 100):
            loss_db = attenuation * d
            transmission = 10 ** (-loss_db / 10)
            qber = self.total_qber(transmission)
            results.append((d, qber))
        return results

    def max_secure_distance(self, max_qber: float = 0.11) -> float:
        """Find maximum distance below QBER threshold."""
        for d in np.linspace(0, 500, 1000):
            loss_db = 0.2 * d
            transmission = 10 ** (-loss_db / 10)
            if self.total_qber(transmission) >= max_qber:
                return d
        return 500

class ErrorCorrection:
    """
    Error correction impact on secure key rate.
    """

    def __init__(self, efficiency: float = 1.16):
        """
        Initialize error correction model.

        Args:
            efficiency: Efficiency factor (1.0 = Shannon limit)
        """
        self.efficiency = efficiency

    def information_leaked(self, qber: float) -> float:
        """
        Calculate information leaked to eavesdropper.

        Uses binary entropy function.
        """
        if qber <= 0:
            return 0
        if qber >= 0.5:
            return 1
        return -qber * np.log2(qber) - (1 - qber) * np.log2(1 - qber)

    def error_correction_cost(self, qber: float) -> float:
        """
        Calculate bits sacrificed for error correction.
        """
        return self.efficiency * self.information_leaked(qber)

    def privacy_amplification_cost(self, qber: float) -> float:
        """
        Calculate bits sacrificed for privacy amplification.

        Assumes symmetric channel.
        """
        return self.information_leaked(qber)

    def secret_fraction(self, qber: float) -> float:
        """
        Calculate fraction of raw bits that become secret key.
        """
        ec_cost = self.error_correction_cost(qber)
        pa_cost = self.privacy_amplification_cost(qber)

        return max(0, 1 - ec_cost - pa_cost)

print("QBER ANALYSIS")
print("=" * 50)

ec = ErrorCorrection()

print("\nQBER vs Distance:")

    trans = 10 ** (-loss_db / 10)
    qber = qber_analysis.total_qber(trans)
    secret_frac = ec.secret_fraction(qber)
    print(f"  {dist} km: QBER={qber*100:.2f}%, Secret fraction={secret_frac:.4f}")

max_dist = qber_analysis.max_secure_distance()
print(f"\nMaximum secure distance (QBER < 11%): {max_dist:.1f} km")
```

```python
class NoiseMitigation:
    """
    Noise mitigation strategies for quantum networks.
    """

    def decoy_state_gain(
        self,
        mu: float = 0.5,
        nu: float = 0.1,
        transmission: float = 0.1
    ) -> dict:
        """
        Calculate gain from decoy state protocol.

        Args:
            mu: Signal state intensity
            nu: Decoy state intensity
            transmission: Channel transmission
        """
        # Simplified model
        # Without decoy: must assume all multi-photon attacked
        # With decoy: can estimate single-photon contribution

        # Yield estimates
        Q_mu = 1 - np.exp(-mu * transmission)  # Detection prob
        Q_nu = 1 - np.exp(-nu * transmission)

        # Single photon yield estimate
        Y_1 = (Q_mu * np.exp(mu) - Q_nu * np.exp(nu)) / (mu - nu)

        # Key rate improvement
        improvement = Y_1 / Q_mu if Q_mu > 0 else 0

        return {
            "Q_signal": Q_mu,
            "Q_decoy": Q_nu,
            "Y_1_estimate": Y_1,
            "improvement_factor": improvement,
            "explanation": "Decoy states allow estimation of single-photon yield"
        }

    def error_correction_efficiency(
        self,
        code: str = 'LDPC'
    ) -> dict:
        """
        Error correction code efficiency.
        """
        efficiencies = {
            'LDPC': 1.05,      # Very close to Shannon
            'CASCADE': 1.16,  # Original QKD code
            'Polar': 1.02,    # Near-optimal
        }

        return {
            "code": code,
            "efficiency": efficiencies.get(code, 1.2),
            "overhead": f"{(efficiencies.get(code, 1.2) - 1) * 100:.1f}% above Shannon limit"
        }

    def privacy_amplification_methods(self) -> dict:
        """
        Privacy amplification techniques.
        """
        return {
            "hash_functions": [
                "Universal hashing (Toeplitz matrices)",
                "2-universal hash families",
                "Leftover hash lemma guarantee"
            ],
            "output_length": "n × (1 - h(e) - leakage)",
            "security": "Exponentially small in number of bits sacrificed"
        }
```

## Summary

Noise and loss fundamentals:

1. **Sources**: Fiber attenuation, scattering, dark counts, polarization drift
2. **Decoherence**: T1 (relaxation) and T2 (dephasing) govern state decay
3. **QBER**: Multiple contributions; threshold ~11% for secure key
4. **Mitigation**: Decoy states, error correction, privacy amplification

Managing noise is critical for practical quantum network deployment.
