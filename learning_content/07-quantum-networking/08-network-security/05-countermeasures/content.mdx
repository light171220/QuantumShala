```python
import numpy as np
from dataclasses import dataclass, field
from typing import List, Dict, Tuple, Optional
from enum import Enum

class CountermeasureType(Enum):
    """Types of countermeasures."""
    OPTICAL = "optical"
    ELECTRICAL = "electrical"
    PROTOCOL = "protocol"
    MONITORING = "monitoring"

@dataclass
class Countermeasure:
    """Countermeasure specification."""
    name: str
    cm_type: CountermeasureType
    target_attacks: List[str]
    description: str
    implementation_cost: str
    effectiveness: float  # 0-1

class DeviceHardening:
    """
    Device hardening countermeasures.
    """

    def __init__(self):
        self.countermeasures = self._define_countermeasures()

    def _define_countermeasures(self) -> Dict[str, Countermeasure]:
        """Define available countermeasures."""
        return {
            "optical_isolator": Countermeasure(
                name="Optical Isolator",
                cm_type=CountermeasureType.OPTICAL,
                target_attacks=["trojan_horse", "backflash"],
                description="One-way optical element prevents light "
                           "from entering through output port",
                implementation_cost="Low",
                effectiveness=0.95
            ),
            "power_limiter": Countermeasure(
                name="Optical Power Limiter",
                cm_type=CountermeasureType.OPTICAL,
                target_attacks=["detector_blinding", "laser_damage"],
                description="Limits incoming optical power to safe levels, "
                           "prevents bright light attacks",
                implementation_cost="Medium",
                effectiveness=0.90
            ),
            "wavelength_filter": Countermeasure(
                name="Narrow Wavelength Filter",
                cm_type=CountermeasureType.OPTICAL,
                target_attacks=["trojan_horse", "wavelength_attack"],
                description="Passes only expected wavelengths, blocks "
                           "probe light at other wavelengths",
                implementation_cost="Low",
                effectiveness=0.85
            ),
            "current_monitor": Countermeasure(
                name="Detector Current Monitor",
                cm_type=CountermeasureType.ELECTRICAL,
                target_attacks=["detector_blinding"],
                description="Monitors APD bias current; elevated current "
                           "indicates blinding attack",
                implementation_cost="Low",
                effectiveness=0.80
            ),
            "random_efficiency_test": Countermeasure(
                name="Random Efficiency Testing",
                cm_type=CountermeasureType.PROTOCOL,
                target_attacks=["detector_blinding", "time_shift"],
                description="Randomly test detector with known signals "
                           "to verify normal operation",
                implementation_cost="Medium",
                effectiveness=0.85
            ),
            "tamper_enclosure": Countermeasure(
                name="Tamper-Evident Enclosure",
                cm_type=CountermeasureType.OPTICAL,
                target_attacks=["physical_access", "component_swap"],
                description="Physical enclosure that shows evidence "
                           "of opening or tampering",
                implementation_cost="Medium",
                effectiveness=0.90
            )
        }

    def get_countermeasures_for_attack(self, attack: str) -> List[Countermeasure]:
        """Get countermeasures effective against specific attack."""
        return [cm for cm in self.countermeasures.values()
                if attack in cm.target_attacks]

    def calculate_combined_effectiveness(
        self,
        applied_countermeasures: List[str]
    ) -> float:
        """Calculate combined effectiveness of multiple countermeasures."""
        # Probability of attack succeeding = product of bypass probabilities
        bypass_prob = 1.0
        for cm_name in applied_countermeasures:
            cm = self.countermeasures.get(cm_name)
            if cm:
                bypass_prob *= (1 - cm.effectiveness)
        return 1 - bypass_prob

    def print_countermeasures(self):
        """Print countermeasure catalog."""
        print("Device Hardening Countermeasures")
        print("=" * 60)

        by_type = {}
        for cm in self.countermeasures.values():
            t = cm.cm_type.value
            if t not in by_type:
                by_type[t] = []
            by_type[t].append(cm)

        for cm_type, cms in by_type.items():
            print(f"\n{cm_type.upper()}:")
            for cm in cms:
                targets = ", ".join(cm.target_attacks[:2])
                print(f"  • {cm.name} ({cm.effectiveness:.0%}) - {targets}")

# Display countermeasures
hardening = DeviceHardening()
hardening.print_countermeasures()

# Example: countermeasures for blinding attack
print("\n\nCountermeasures for Detector Blinding:")
for cm in hardening.get_countermeasures_for_attack("detector_blinding"):
    print(f"  • {cm.name}: {cm.description[:50]}...")
```

```python
class SecurityMonitor:
    """
    Real-time security monitoring system.
    """

    def __init__(self):
        self.baselines = {}
        self.current_readings = {}
        self.alerts = []
        self.thresholds = {
            "qber": 0.11,  # Above this, abort
            "detection_rate_drop": 0.5,  # 50% drop is suspicious
            "power_spike": 10.0,  # 10x normal power
            "timing_deviation": 3.0  # 3 sigma
        }

    def set_baseline(self, parameter: str, value: float, std: float = 0.1):
        """Set normal operating baseline."""
        self.baselines[parameter] = {
            "mean": value,
            "std": std,
            "min": value - 3 * std,
            "max": value + 3 * std
        }

    def update_reading(self, parameter: str, value: float) -> Optional[Dict]:
        """Update current reading and check for anomalies."""
        self.current_readings[parameter] = value

        if parameter in self.baselines:
            baseline = self.baselines[parameter]
            if value < baseline["min"] or value > baseline["max"]:
                alert = self._generate_alert(parameter, value, baseline)
                self.alerts.append(alert)
                return alert
        return None

    def _generate_alert(
        self,
        parameter: str,
        value: float,
        baseline: Dict
    ) -> Dict:
        """Generate security alert."""
        deviation = abs(value - baseline["mean"]) / baseline["std"]
        severity = "WARNING" if deviation < 5 else "CRITICAL"

        return {
            "type": "ANOMALY",
            "parameter": parameter,
            "value": value,
            "expected": baseline["mean"],
            "deviation_sigma": deviation,
            "severity": severity,
            "action": self._recommend_action(parameter, severity)
        }

    def _recommend_action(self, parameter: str, severity: str) -> str:
        """Recommend response action."""
        if severity == "CRITICAL":
            return "ABORT_SESSION"
        elif parameter == "qber":
            return "INCREASE_PRIVACY_AMP"
        elif parameter == "power":
            return "CHECK_OPTICAL_PATH"
        return "MONITOR"

    def check_attack_signatures(self) -> List[Dict]:
        """Check for known attack signatures."""
        signatures = []

        # Blinding attack signature: high power + low QBER
        power = self.current_readings.get("input_power", 0)
        qber = self.current_readings.get("qber", 0)

        if power > 5.0 and qber < 0.01:
            signatures.append({
                "attack": "Detector Blinding",
                "confidence": 0.9,
                "indicators": ["High power", "Suspiciously low QBER"],
                "action": "ABORT_SESSION"
            })

        # Time-shift attack: asymmetric detection
        det_0 = self.current_readings.get("detector_0_rate", 0.5)
        det_1 = self.current_readings.get("detector_1_rate", 0.5)

        if abs(det_0 - det_1) > 0.3:
            signatures.append({
                "attack": "Time-Shift Attack",
                "confidence": 0.7,
                "indicators": ["Detector imbalance"],
                "action": "INVESTIGATE"
            })

        return signatures

    def get_security_status(self) -> Dict:
        """Get overall security status."""
        active_alerts = [a for a in self.alerts[-10:]
                        if a["severity"] == "CRITICAL"]

        return {
            "status": "COMPROMISED" if active_alerts else "SECURE",
            "active_alerts": len(active_alerts),
            "recent_alerts": self.alerts[-5:],
            "parameters_monitored": len(self.baselines)
        }

# Demonstrate monitoring
print("\n\nSecurity Monitoring System")

monitor.set_baseline("input_power", 1.0, 0.2)

# Normal operation
```

```python
                    "detector_efficiency_mismatch"

                "overhead": "~50% key rate reduction",

            },
            "di_qkd": {
                "name": "Device-Independent QKD",
                "defense_against": [
                    "all_implementation_attacks",
                    "trojan_horse",
                    "side_channels"
                ],
                "mechanism": "Security from Bell inequality violation only",
                "overhead": "Very low key rates currently",
                "security_gain": "Minimal device trust required"
            },
            "continuous_variable": {
                "name": "CV-QKD with Local Oscillator",
                "defense_against": ["detector_blinding"],
                "mechanism": "Coherent detection immune to blinding",
                "overhead": "Different hardware requirements",
                "security_gain": "Natural blinding immunity"
            }
        }

    def recommend_protocol(
        self,
        threat_profile: List[str],
        rate_requirements: str
    ) -> str:
        """Recommend protocol based on threats and requirements."""
        scores = {}

        for proto_id, proto in self.defense_protocols.items():
            # Score based on threat coverage
            coverage = sum(1 for t in threat_profile
                          if t in proto["defense_against"]
                          or "all_implementation_attacks" in proto["defense_against"])
            scores[proto_id] = coverage

        # Filter by rate requirements
        if rate_requirements == "high":
            scores.pop("di_qkd", None)  # DI-QKD has low rates

        # Return highest scoring
        if scores:
            return max(scores, key=scores.get)
        return "standard_bb84"

    def print_protocols(self):
        """Print protocol defenses."""
        print("\n\nProtocol-Level Defenses")
        print("=" * 50)

        for proto_id, proto in self.defense_protocols.items():
            print(f"\n{proto['name']}:")
            print(f"  Defends against: {', '.join(proto['defense_against'][:2])}")
            print(f"  Mechanism: {proto['mechanism'][:50]}...")
            print(f"  Overhead: {proto['overhead']}")
```

## MDI-QKD Architecture

```
│  │      │      │ Charlie │      │      │                    │

│                     ▼                                               │
│         Bell state measurement                                      │

│                     │                                               │
│                     ▼                                               │
│         Alice & Bob correlate to get key                           │
│                                                                     │

│  Reason: Charlie only sees interference, not individual states     │
```

```python
class MDIQKDSimulator:
    """
    Simplified MDI-QKD demonstration.
    """

    def __init__(self):
        self.alice_bits = []
        self.bob_bits = []
        self.successful_rounds = []

    def alice_prepare(self, bit: int, basis: int) -> Dict:
        """Alice prepares and sends state."""
        return {"bit": bit, "basis": basis, "source": "alice"}

    def bob_prepare(self, bit: int, basis: int) -> Dict:
        """Bob prepares and sends state."""
        return {"bit": bit, "basis": basis, "source": "bob"}

    def charlie_measure(
        self,
        alice_state: Dict,
        bob_state: Dict
    ) -> Optional[str]:
        """
        Charlie performs Bell state measurement.
        Even if Charlie is Eve, he can't learn the key!
        """
        # Only succeeds when bases match
        if alice_state["basis"] != bob_state["basis"]:
            return None  # No correlation

        # Simplified: BSM succeeds 50% of time
        if np.random.random() > 0.5:
            return None

        # Determine which Bell state
        bit_sum = (alice_state["bit"] + bob_state["bit"]) % 2
        return "psi_plus" if bit_sum == 0 else "psi_minus"

    def simulate_round(self) -> Optional[Tuple[int, int]]:
        """Simulate one MDI-QKD round."""
        # Alice and Bob randomly choose bits and bases
        alice_bit = np.random.randint(0, 2)
        alice_basis = np.random.randint(0, 2)
        bob_bit = np.random.randint(0, 2)
        bob_basis = np.random.randint(0, 2)

        alice_state = self.alice_prepare(alice_bit, alice_basis)
        bob_state = self.bob_prepare(bob_bit, bob_basis)

        result = self.charlie_measure(alice_state, bob_state)

        if result:
            # Successful BSM - Alice and Bob can correlate
            # Their key bits are correlated based on BSM result
            if result == "psi_plus":
                key_bit = alice_bit  # They have same bit
            else:
                key_bit = alice_bit  # They have opposite (Bob flips)
            return (key_bit, alice_basis)

        return None

    def simulate_protocol(self, n_rounds: int) -> Dict:
        """Simulate complete MDI-QKD protocol."""
        successful = 0
        key_bits = []

        for _ in range(n_rounds):
            result = self.simulate_round()
            if result:
                successful += 1
                key_bits.append(result[0])

        return {
            "total_rounds": n_rounds,
            "successful": successful,
            "success_rate": successful / n_rounds,
            "key_bits": len(key_bits),
            "security": "Detector attacks impossible"
        }
```
