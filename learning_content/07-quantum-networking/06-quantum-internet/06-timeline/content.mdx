@dataclass
class Milestone:
    """Development milestone."""
    name: str
    optimistic_year: int
    expected_year: int
    conservative_year: int
    dependencies: List[str]

class TimelineScenarios:
    """
    Different scenarios for quantum internet development.
    """

    def __init__(self):
        self.milestones = self._define_milestones()

    def _define_milestones(self) -> List[Milestone]:
        """Define key milestones with timeline ranges."""
        return [
            Milestone(
                name="First practical quantum repeater",
                optimistic_year=2026,
                expected_year=2028,
                conservative_year=2032,
                dependencies=["Quantum memory >100ms", "High-efficiency BSM"]
            ),
            Milestone(
                name="10-node entanglement network",
                optimistic_year=2027,
                expected_year=2030,
                conservative_year=2035,
                dependencies=["Repeater technology", "Routing protocols"]
            ),
            Milestone(
                name="Metropolitan quantum network",
                optimistic_year=2028,
                expected_year=2032,
                conservative_year=2038,
                dependencies=["Repeater deployment", "Commercial systems"]
            ),
            Milestone(
                name="Quantum-computer networking",
                optimistic_year=2030,
                expected_year=2035,
                conservative_year=2042,
                dependencies=["Fault-tolerant QC", "High-fidelity links"]
            ),
            Milestone(
                name="Continental quantum network",
                optimistic_year=2035,
                expected_year=2040,
                conservative_year=2050,
                dependencies=["Repeater chains", "Satellite integration"]
            ),
            Milestone(
                name="Global quantum internet",
                optimistic_year=2040,
                expected_year=2050,
                conservative_year=2060,
                dependencies=["All previous milestones"]
            )
        ]

    def plot_timeline(self):
        """Display timeline scenarios."""
        print("\nQuantum Internet Timeline Scenarios")
        print("=" * 70)

        current_year = 2024

        print(f"\n{'Milestone':<35} {'Optimistic':<12} {'Expected':<12} {'Conservative'}")
        print("-" * 70)

        for m in self.milestones:
            print(f"{m.name:<35} {m.optimistic_year:<12} "
                  f"{m.expected_year:<12} {m.conservative_year}")

    def analyze_scenario(self, scenario: str = "expected") -> Dict:
        """Analyze timeline for given scenario."""
        if scenario == "optimistic":
            get_year = lambda m: m.optimistic_year
        elif scenario == "conservative":
            get_year = lambda m: m.conservative_year
        else:
            get_year = lambda m: m.expected_year

        years_to_milestone = {}
        for m in self.milestones:
            years_to_milestone[m.name] = get_year(m) - 2024

        return {
            "scenario": scenario,
            "years_to_global_internet": years_to_milestone.get("Global quantum internet"),
            "milestones": years_to_milestone
        }

# Display scenarios
scenarios = TimelineScenarios()
scenarios.plot_timeline()

print("\n\nScenario Analysis:")
for scenario in ["optimistic", "expected", "conservative"]:
    analysis = scenarios.analyze_scenario(scenario)
    print(f"\n{scenario.upper()} scenario:")
    print(f"  Global quantum internet in: {analysis['years_to_global_internet']} years")
