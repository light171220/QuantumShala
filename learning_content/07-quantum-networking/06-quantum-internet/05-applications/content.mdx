```python
                qubit_requirement="Many qubits per node",
                network_requirement="High-rate, high-fidelity entanglement"
            ),
            ComputingApplication(
                name="Blind Quantum Computation",
                description="User runs computation on server without revealing data or algorithm",
                privacy_level="Complete user privacy",
                qubit_requirement="User: minimal; Server: full QC",
                network_requirement="Single-qubit transmission"
            ),
            ComputingApplication(
                name="Verifiable Delegation",
                description="Verify that remote QC performed correct computation",
                privacy_level="Verification possible",
                qubit_requirement="User: minimal verification qubits",
                network_requirement="Quantum and classical channels"
            ),
            ComputingApplication(
                name="Secure Multi-Party Computation",
                description="Multiple parties compute on joint data without revealing inputs",
                privacy_level="Input privacy for all parties",
                qubit_requirement="Distributed across parties",
                network_requirement="Multi-party entanglement"
            )
        ]

    def blind_computation_protocol(self) -> Dict:
        """
        Describe blind quantum computation protocol.

        The user can:
        1. Prepare single qubits with random rotations
        2. Send to server (who cannot determine rotation)
        3. Instruct server on gates (adjusted for rotations)
        4. Server returns results (meaningless without rotation key)
        """
        return {
            "user_operations": [
                "Generate random rotation angles",
                "Prepare rotated qubits |+θ⟩",
                "Send qubits to server",
                "Compute adjusted measurement angles",
                "Send measurement instructions",
                "Receive measurement results",
                "Decode using rotation keys"
            ],
            "server_operations": [
                "Receive encoded qubits",
                "Entangle according to computation graph",
                "Measure as instructed",
                "Return results"
            ],
            "security_guarantee": "Server learns nothing about computation",
            "requirements": "Single-qubit quantum channel"
        }

    def print_applications(self):
        """Print distributed computing applications."""
        print("\n\nDistributed Quantum Computing Applications")
        print("=" * 70)

        for app in self.applications:
            print(f"\n{app.name}")
            print(f"  {app.description}")
            print(f"  Privacy: {app.privacy_level}")
            print(f"  Qubits: {app.qubit_requirement}")
            print(f"  Network: {app.network_requirement}")

# Display applications
computing_apps = DistributedComputingApps()
computing_apps.print_applications()

print("\n\nBlind Quantum Computation Protocol:")
protocol = computing_apps.blind_computation_protocol()
print(f"\nUser operations:")
for op in protocol["user_operations"]:
    print(f"  {op}")
print(f"\nSecurity: {protocol['security_guarantee']}")
```

```python
@dataclass
class SensingApplication:
    """Quantum sensing application."""
    name: str
    physical_quantity: str
    quantum_advantage: str
    network_size: str
    precision_gain: str

class QuantumSensingApps:
    """
    Quantum sensing network applications.
    """

    def __init__(self):
        self.applications = self._define_applications()

    def _define_applications(self) -> List[SensingApplication]:
        """Define sensing applications."""
        return [
            SensingApplication(
                name="Clock Synchronization",
                physical_quantity="Time/frequency",
                quantum_advantage="Entanglement-enhanced correlation",
                network_size="2+ nodes",
                precision_gain="Beyond classical limits"
            ),
            SensingApplication(
                name="Gravitational Wave Detection",
                physical_quantity="Spacetime strain",
                quantum_advantage="Squeezed states + distributed sensing",
                network_size="Multiple detectors globally",
                precision_gain="√N improvement with N detectors"
            ),
            SensingApplication(
                name="Magnetic Field Mapping",
                physical_quantity="Magnetic field",
                quantum_advantage="Entangled sensor array",
                network_size="Array of magnetometers",
                precision_gain="Heisenberg scaling"
            ),
            SensingApplication(
                name="Radio Telescope Networks",
                physical_quantity="Radio waves",
                quantum_advantage="Quantum-enhanced correlation",
                network_size="Global telescope array",
                precision_gain="Improved baseline resolution"
            ),
            SensingApplication(
                name="Navigation Enhancement",
                physical_quantity="Position/orientation",
                quantum_advantage="Entangled inertial sensors",
                network_size="Vehicle sensor network",
                precision_gain="GPS-free navigation"
            )
        ]

    def calculate_sensitivity_gain(
        self,
        n_sensors: int,
        is_entangled: bool = True
    ) -> Dict:
        """Calculate sensitivity improvement."""
        if is_entangled:
            # Heisenberg scaling
            gain = n_sensors
            scaling = "Heisenberg (1/N)"
        else:
            # Classical (shot noise) scaling
            gain = np.sqrt(n_sensors)
            scaling = "Standard (1/√N)"

        return {
            "n_sensors": n_sensors,
            "entangled": is_entangled,
            "sensitivity_gain": gain,
            "scaling": scaling,
            "vs_single_sensor": f"{gain:.1f}x improvement"
        }

    def print_applications(self):
        """Print sensing applications."""
        print("\n\nQuantum Sensing Network Applications")
        print("=" * 70)

        for app in self.applications:
            print(f"\n{app.name}")
            print(f"  Measures: {app.physical_quantity}")
            print(f"  Advantage: {app.quantum_advantage}")
            print(f"  Network: {app.network_size}")
            print(f"  Precision: {app.precision_gain}")

# Display applications
sensing_apps = QuantumSensingApps()
sensing_apps.print_applications()

print("\n\nSensitivity Comparison:")
print("-" * 50)
for n in [4, 16, 64]:
    result = sensing_apps.calculate_sensitivity_gain(n, True)
    print(f"\n{n} entangled sensors:")
    print(f"  Scaling: {result['scaling']}")
    print(f"  Improvement: {result['vs_single_sensor']}")
```
