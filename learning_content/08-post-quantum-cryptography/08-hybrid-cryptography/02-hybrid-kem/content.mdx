## 2. Implementation Deep Dive

```python
import hashlib
import secrets
import struct

class HybridKEM:
    """
    Production-quality Hybrid KEM: X25519 + ML-KEM-768
    
    This implementation demonstrates proper hybrid key exchange
    with defense in depth against both classical and quantum attackers.
    """
    
    # Algorithm identifiers for domain separation
    ALGORITHM_ID = b"X25519-MLKEM768-Hybrid-v1"
    
    # Key sizes
    X25519_PUBLIC_KEY_SIZE = 32
    X25519_SECRET_KEY_SIZE = 32
    MLKEM768_PUBLIC_KEY_SIZE = 1184
    MLKEM768_SECRET_KEY_SIZE = 2400
    MLKEM768_CIPHERTEXT_SIZE = 1088
    SHARED_SECRET_SIZE = 32
    
    def __init__(self):
        """Initialize hybrid KEM."""
        self.x25519_sk = None
        self.x25519_pk = None
        self.mlkem_sk = None
        self.mlkem_pk = None
    
    def keygen(self) -> tuple:
        """
        Generate hybrid key pair.
        
        Returns:
            (public_key, secret_key) where each is concatenation of
            classical and PQC components.
        """
        # Generate X25519 key pair
        self.x25519_sk = secrets.token_bytes(self.X25519_SECRET_KEY_SIZE)
        self.x25519_pk = self._x25519_public_key(self.x25519_sk)
        
        # Generate ML-KEM-768 key pair
        self.mlkem_pk, self.mlkem_sk = self._mlkem_keygen()
        
        # Concatenate public keys
        public_key = self.x25519_pk + self.mlkem_pk
        
        # Concatenate secret keys
        secret_key = self.x25519_sk + self.mlkem_sk
        
        return public_key, secret_key
    
    def encapsulate(self, public_key: bytes) -> tuple:
        """
        Encapsulate to create shared secret and ciphertext.
        
        Args:
            public_key: Recipient's hybrid public key
            
        Returns:
            (ciphertext, shared_secret)
        """
        # Parse public key
        x25519_pk = public_key[:self.X25519_PUBLIC_KEY_SIZE]
        mlkem_pk = public_key[self.X25519_PUBLIC_KEY_SIZE:]
        
        # X25519 key exchange
        ephemeral_sk = secrets.token_bytes(self.X25519_SECRET_KEY_SIZE)
        ephemeral_pk = self._x25519_public_key(ephemeral_sk)
        k_classical = self._x25519_shared_secret(ephemeral_sk, x25519_pk)
        
        # ML-KEM encapsulation
        mlkem_ct, k_pqc = self._mlkem_encapsulate(mlkem_pk)
        
        # Combine ciphertexts
        ciphertext = ephemeral_pk + mlkem_ct
        
        # Combine secrets using KDF with domain separation
        shared_secret = self._combine_secrets(
            k_classical, k_pqc, ciphertext
        )
        
        return ciphertext, shared_secret
    
    def decapsulate(self, secret_key: bytes, ciphertext: bytes) -> bytes:
        """
        Decapsulate to recover shared secret.
        
        Args:
            secret_key: Own hybrid secret key
            ciphertext: Received hybrid ciphertext
            
        Returns:
            shared_secret
        """
        # Parse secret key
        x25519_sk = secret_key[:self.X25519_SECRET_KEY_SIZE]
        mlkem_sk = secret_key[self.X25519_SECRET_KEY_SIZE:]
        
        # Parse ciphertext
        ephemeral_pk = ciphertext[:self.X25519_PUBLIC_KEY_SIZE]
        mlkem_ct = ciphertext[self.X25519_PUBLIC_KEY_SIZE:]
        
        # X25519 shared secret
        k_classical = self._x25519_shared_secret(x25519_sk, ephemeral_pk)
        
        # ML-KEM decapsulation
        k_pqc = self._mlkem_decapsulate(mlkem_sk, mlkem_ct)
        
        # Combine secrets
        shared_secret = self._combine_secrets(
            k_classical, k_pqc, ciphertext
        )
        
        return shared_secret
    
    def _combine_secrets(self, k_classical: bytes, k_pqc: bytes,
                        ciphertext: bytes) -> bytes:
        """
        Combine classical and PQC secrets using HKDF.
        
        CRITICAL: This is where hybrid security is achieved.
        The combination must be done correctly to ensure
        security if either component is broken.
        """
        # Concatenate inputs with domain separation
        ikm = k_classical + k_pqc
        
        # Use ciphertext as salt for binding
        salt = hashlib.sha256(ciphertext).digest()
        
        # Info includes algorithm identifier
        info = self.ALGORITHM_ID
        
        # HKDF-Extract
        prk = hashlib.sha256(salt + ikm).digest()
        
        # HKDF-Expand (simplified - use proper HKDF in production)
        okm = hashlib.sha256(prk + info + b'\x01').digest()
        
        return okm[:self.SHARED_SECRET_SIZE]
    
    # Placeholder methods (use real crypto libraries in production)
    def _x25519_public_key(self, sk):
        return hashlib.sha256(b"x25519_pk" + sk).digest()
    
    def _x25519_shared_secret(self, sk, pk):
        return hashlib.sha256(b"x25519_ss" + sk + pk).digest()
    
    def _mlkem_keygen(self):
        pk = secrets.token_bytes(self.MLKEM768_PUBLIC_KEY_SIZE)
        sk = secrets.token_bytes(self.MLKEM768_SECRET_KEY_SIZE)
        return pk, sk
    
    def _mlkem_encapsulate(self, pk):
        ct = secrets.token_bytes(self.MLKEM768_CIPHERTEXT_SIZE)
        ss = secrets.token_bytes(32)
        return ct, ss
    
    def _mlkem_decapsulate(self, sk, ct):
        return hashlib.sha256(b"mlkem_ss" + sk[:32] + ct[:32]).digest()


# Demonstrate usage
def demonstrate_hybrid_kem():
    print("\nHYBRID KEM DEMONSTRATION")
    print("-" * 50)
    
    kem = HybridKEM()
    
    # Alice generates key pair
    alice_pk, alice_sk = kem.keygen()
    print(f"Alice's public key: {len(alice_pk)} bytes")
    print(f"  X25519 component: {kem.X25519_PUBLIC_KEY_SIZE} bytes")
    print(f"  ML-KEM component: {kem.MLKEM768_PUBLIC_KEY_SIZE} bytes")
    
    # Bob encapsulates to Alice
    ciphertext, bob_secret = kem.encapsulate(alice_pk)
    print(f"\nCiphertext: {len(ciphertext)} bytes")
    print(f"  X25519 component: {kem.X25519_PUBLIC_KEY_SIZE} bytes")
    print(f"  ML-KEM component: {kem.MLKEM768_CIPHERTEXT_SIZE} bytes")
    
    # Alice decapsulates
    alice_secret = kem.decapsulate(alice_sk, ciphertext)
    
    print(f"\nBob's shared secret:   {bob_secret.hex()[:32]}...")
    print(f"Alice's shared secret: {alice_secret.hex()[:32]}...")

demonstrate_hybrid_kem()
```

## Summary

Hybrid KEMs provide essential defense in depth:

1. **Architecture**: Run classical and PQC KEMs in parallel
2. **Security**: Secure if EITHER component is secure
3. **Overhead**: Only +64 bytes vs PQC-only
4. **Implementation**: Use validated libraries, proper error handling
5. **Deployment**: Already available in TLS via major browsers

Always use hybrid during the transition period.
