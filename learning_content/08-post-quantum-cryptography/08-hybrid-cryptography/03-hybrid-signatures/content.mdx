```python
    │                                                                 │

    │ Risk:     Attacker creates fake "historical" certificates      │
    │                                                                 │
    │ SOLUTION: Sign with BOTH classical AND post-quantum            │
    │           Future verification requires both to be valid        │
    └─────────────────────────────────────────────────────────────────┘

    ━━━━━━━━━━━━━━━━━━━━━━━━━━━
    
    A hybrid signature is valid only if BOTH components verify:
    
    Verify(Hybrid) = Verify(Classical) AND Verify(PQC)
    
    This means:
    • Classical attacker must break PQC (hard classically)
    • Quantum attacker must break classical (easy) AND PQC (hard)

    ┌─────────────────────────────────────────────────────────────────┐
    │ Use Case                       │ Hybrid Needed?                │
    ├────────────────────────────────┼───────────────────────────────┤
    │ Long-lived certificates        │ YES - validity spans years   │
    │ Code signing                   │ YES - software lives long    │
    │ Document signing               │ YES - legal validity matters │
    │ Session authentication         │ MAYBE - short-lived tokens   │
    │ Real-time protocols            │ OPTIONAL - PQC-only may work │
    └─────────────────────────────────────────────────────────────────┘
    """)
```

## 3. Implementation

```python
import hashlib
import secrets

class HybridSignature:
    """
    Hybrid Signature: Ed25519 + ML-DSA-65
    
    Provides defense in depth against both classical and quantum attackers.
    """
    
    # Algorithm identifier for domain separation
    ALGORITHM_ID = b"Ed25519-MLDSA65-Hybrid-v1"
    
    # Signature sizes
    ED25519_SIG_SIZE = 64
    MLDSA65_SIG_SIZE = 3309
    ED25519_PK_SIZE = 32
    MLDSA65_PK_SIZE = 1952
    
    def __init__(self):
        """Initialize hybrid signature scheme."""
        self.ed25519_sk = None
        self.ed25519_pk = None
        self.mldsa_sk = None
        self.mldsa_pk = None
    
    def keygen(self) -> tuple:
        """
        Generate hybrid key pair.
        
        Returns:
            (public_key, secret_key) - concatenated components
        """
        # Generate Ed25519 key pair
        self.ed25519_sk = secrets.token_bytes(32)
        self.ed25519_pk = self._ed25519_public_key(self.ed25519_sk)
        
        # Generate ML-DSA-65 key pair
        self.mldsa_pk, self.mldsa_sk = self._mldsa_keygen()
        
        # Concatenate keys
        public_key = self.ed25519_pk + self.mldsa_pk
        secret_key = self.ed25519_sk + self.mldsa_sk
        
        return public_key, secret_key
    
    def sign(self, secret_key: bytes, message: bytes) -> bytes:
        """
        Create hybrid signature (concatenated approach).
        
        Args:
            secret_key: Hybrid secret key
            message: Message to sign
            
        Returns:
            Hybrid signature (σ_ed25519 || σ_mldsa)
        """
        # Parse secret key
        ed25519_sk = secret_key[:32]
        mldsa_sk = secret_key[32:]
        
        # Add domain separation to message
        tagged_message = self.ALGORITHM_ID + message
        
        # Sign with Ed25519
        sig_classical = self._ed25519_sign(ed25519_sk, tagged_message)
        
        # Sign with ML-DSA-65
        sig_pqc = self._mldsa_sign(mldsa_sk, tagged_message)
        
        # Concatenate signatures
        return sig_classical + sig_pqc
    
    def verify(self, public_key: bytes, message: bytes, 
               signature: bytes) -> bool:
        """
        Verify hybrid signature.
        
        BOTH components must verify for the signature to be valid.
        
        Args:
            public_key: Hybrid public key
            message: Original message
            signature: Hybrid signature
            
        Returns:
            True if BOTH components verify, False otherwise
        """
        # Parse public key
        ed25519_pk = public_key[:self.ED25519_PK_SIZE]
        mldsa_pk = public_key[self.ED25519_PK_SIZE:]
        
        # Parse signature
        sig_classical = signature[:self.ED25519_SIG_SIZE]
        sig_pqc = signature[self.ED25519_SIG_SIZE:]
        
        # Reconstruct tagged message
        tagged_message = self.ALGORITHM_ID + message
        
        # Verify BOTH signatures
        classical_valid = self._ed25519_verify(
            ed25519_pk, tagged_message, sig_classical
        )
        
        pqc_valid = self._mldsa_verify(
            mldsa_pk, tagged_message, sig_pqc
        )
        
        # BOTH must be valid
        return classical_valid and pqc_valid
    
    # Placeholder implementations (use real crypto in production)
    def _ed25519_public_key(self, sk):
        return hashlib.sha256(b"ed25519_pk" + sk).digest()
    
    def _ed25519_sign(self, sk, msg):
        return hashlib.sha512(b"ed25519_sig" + sk + msg).digest()
    
    def _ed25519_verify(self, pk, msg, sig):
        expected = hashlib.sha512(b"ed25519_sig" + pk[:32] + msg).digest()
        return sig == expected  # Simplified
    
    def _mldsa_keygen(self):
        pk = secrets.token_bytes(self.MLDSA65_PK_SIZE)
        sk = secrets.token_bytes(4032)  # ML-DSA-65 secret key size
        return pk, sk
    
    def _mldsa_sign(self, sk, msg):
        return secrets.token_bytes(self.MLDSA65_SIG_SIZE)
    
    def _mldsa_verify(self, pk, msg, sig):
        return True  # Simplified


def demonstrate_hybrid_signature():
    """Demonstrate hybrid signature usage."""
    print("\nHYBRID SIGNATURE DEMONSTRATION")

    hybrid = HybridSignature()
```

## Summary

Hybrid signatures provide essential protection for long-lived signatures:

1. **Approaches**: Concatenated (simple) or Nested (stronger binding)
2. **Security**: Both components must verify for valid signature
3. **Size**: Only +64 bytes overhead vs PQC-only signatures
4. **Verification**: Use strict mode (both must verify)
5. **Use Cases**: Certificates, code signing, legal documents

For any signature that must remain valid for years, use hybrid.
