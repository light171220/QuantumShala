After completing this lesson, you will be able to:

## 5. Recommendation Tool

```python
class PQCRecommender:
    """Interactive PQC parameter recommendation tool."""
    
    @staticmethod
    def recommend_kem(
        security_years: int = 15,
        size_critical: bool = False,
        max_confidence: bool = False
    ) -> dict:
        """Recommend KEM parameters."""
        
        if max_confidence:
            return {
                'primary': 'Classic McEliece',
                'alternative': 'ML-KEM-1024 + X25519 hybrid',
                'note': 'Maximum confidence, very large keys'
            }
        
        if size_critical:
            return {
                'primary': 'ML-KEM-512',
                'alternative': 'ML-KEM-768 if size allows',
                'note': 'Smallest PQC option'
            }
        
        if security_years > 20:
            return {
                'primary': 'ML-KEM-1024',
                'alternative': 'ML-KEM-768 + X25519 hybrid',
                'note': 'Category 5 for long-term'
            }
        
        return {
            'primary': 'ML-KEM-768',
            'alternative': 'ML-KEM-768 + X25519 hybrid',
            'note': 'RECOMMENDED for most applications'
        }
    
    @staticmethod
    def recommend_sig(
        security_years: int = 15,
        size_critical: bool = False,
        hash_only_security: bool = False
    ) -> dict:
        """Recommend signature parameters."""
        
        if hash_only_security:
            if size_critical:
                return {
                    'primary': 'SLH-DSA-SHA2-128s',
                    'note': 'Smallest hash-based option'
                }
            return {
                'primary': 'SLH-DSA-SHA2-256s',
                'note': 'Maximum confidence, large signatures'
            }
        
        if size_critical:
            return {
                'primary': 'ML-DSA-44',
                'alternative': 'Wait for FALCON',
                'note': 'Smallest lattice signature'
            }
        
        if security_years > 20:
            return {
                'primary': 'ML-DSA-87',
                'note': 'Category 5 for long-term'
            }
        
        return {
            'primary': 'ML-DSA-65',
            'note': 'RECOMMENDED for most applications'
        }


# Demo
rec = PQCRecommender()

print("\nRECOMMENDATION EXAMPLES:")
print("-" * 50)

print("\n1. General web application (15-year security):")
kem = rec.recommend_kem(15)
sig = rec.recommend_sig(15)
print(f"   KEM: {kem['primary']} - {kem['note']}")
print(f"   Sig: {sig['primary']} - {sig['note']}")

print("\n2. IoT device (size-constrained):")
kem = rec.recommend_kem(10, size_critical=True)
sig = rec.recommend_sig(10, size_critical=True)
print(f"   KEM: {kem['primary']} - {kem['note']}")
print(f"   Sig: {sig['primary']} - {sig['note']}")

print("\n3. Root CA certificate (maximum confidence):")
sig = rec.recommend_sig(50, hash_only_security=True)
print(f"   Sig: {sig['primary']} - {sig['note']}")
```

## Summary

Choosing PQC parameters:

1. **Default choice**: ML-KEM-768 + ML-DSA-65 (Category 3)
2. **Size-constrained**: ML-KEM-512 + ML-DSA-44 (Category 1)
3. **Long-term/high-security**: ML-KEM-1024 + ML-DSA-87 (Category 5)
4. **Maximum confidence**: SLH-DSA for signatures
5. **Hybrid**: X25519 + ML-KEM-768 for key exchange
6. **Always consider**: Hybrid approaches during transition

When in doubt, choose Category 3 - it provides excellent security margin with reasonable sizes.
