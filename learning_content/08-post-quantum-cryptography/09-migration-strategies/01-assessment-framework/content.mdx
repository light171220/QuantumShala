```python
class RiskCalculator:
    """Calculate PQC migration risk."""
    
    @staticmethod
    def mosca_risk(migration_years: int, secrecy_years: int,
                   crqc_years: int = 15) -> dict:
        """Calculate risk using Mosca inequality."""
        at_risk = migration_years + secrecy_years > crqc_years
        
        if at_risk:
            urgency = migration_years + secrecy_years - crqc_years
            if urgency > 20:
                level = "CRITICAL"
            elif urgency > 10:
                level = "HIGH"
            else:
                level = "MEDIUM"
        else:
            level = "LOW"
            urgency = 0
        
        return {
            'at_risk': at_risk,
            'risk_level': level,
            'urgency_years': urgency,
            'recommendation': "Start migration now" if at_risk else "Plan migration"
        }

# Example usage
calc = RiskCalculator()
print("\nRISK CALCULATIONS:")
print("-" * 40)
print(f"Healthcare: {calc.mosca_risk(5, 50)}")
print(f"Banking: {calc.mosca_risk(3, 7)}")
print(f"Government: {calc.mosca_risk(10, 75)}")
```

## 4. Gap Analysis

```python
def gap_analysis():
    """Identifying gaps in PQC readiness."""
    print("\nGAP ANALYSIS")
    print("=" * 70)
    
    print("""
    CURRENT STATE VS TARGET STATE:
    ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
    
    ┌─────────────────────────────────────────────────────────────────┐
    │ Area              │ Current State     │ Target State           │
    ├───────────────────┼───────────────────┼────────────────────────┤
    │ Key Exchange      │ ECDH only         │ Hybrid X25519+ML-KEM   │
    │ Signatures        │ ECDSA/RSA         │ Hybrid ECDSA+ML-DSA    │
    │ Certificates      │ Classical PKI     │ PQC-capable PKI        │
    │ HSMs              │ No PQC support    │ PQC-capable HSMs       │
    │ Key Management    │ Classical only    │ PQC key support        │
    │ Libraries         │ OpenSSL 1.x       │ OpenSSL 3.x + oqs      │
    │ Staff Skills      │ Classical crypto  │ PQC knowledge          │
    │ Vendor Support    │ Unknown           │ Confirmed PQC roadmap  │
    └─────────────────────────────────────────────────────────────────┘
    
    
    COMMON GAPS DISCOVERED:
    ━━━━━━━━━━━━━━━━━━━━━━━━
    
    TECHNOLOGY GAPS:
    • Legacy systems that can't be upgraded
    • HSMs without PQC support
    • Hardcoded algorithms in applications
    • Outdated cryptographic libraries
    
    PROCESS GAPS:
    • No cryptographic inventory process
    • No algorithm deprecation procedure
    • No crypto agility in development
    • No PQC in security requirements
    
    PEOPLE GAPS:
    • No PQC expertise on staff
    • No training programs
    • No clear ownership
    • No executive awareness
    
    VENDOR GAPS:
    • Vendors without PQC roadmaps
    • End-of-life products still in use
    • No PQC in procurement requirements
    """)

gap_analysis()
```

## Summary

The PQC assessment framework covers:

1. **Five Pillars**: Inventory, sensitivity, systems, vendors, readiness
2. **Risk Evaluation**: Mosca inequality to determine urgency
3. **Gap Analysis**: Current vs target state comparison
4. **Assessment Report**: Structured findings for stakeholders

Complete assessment before starting migration to ensure success.
