## 3. KEM Construction

```python
class ClassicMcElieceKEM:
    """
    Classic McEliece KEM construction.
    
    Uses Niederreiter variant + Fujisaki-Okamoto transform.
    
    KEM Interface:
    - KeyGen() → (pk, sk)
    - Encaps(pk) → (K, c)  
    - Decaps(sk, c) → K
    """
    
    def __init__(self, m: int = 12, n: int = 3488, t: int = 64):
        """
        Initialize with Goppa code parameters.
        
        Args:
            m: Extension field degree (F_{2^m})
            n: Code length (≤ 2^m)
            t: Error correction capability
        """
        self.m = m
        self.n = n
        self.t = t
        self.k = n - m * t  # Message bits
        
    def _hash_g(self, data: bytes) -> bytes:
        """Hash function G for shared secret derivation."""
        return hashlib.sha3_256(b'G' + data).digest()
    
    def _hash_h(self, data: bytes) -> bytes:
        """Hash function H for randomness generation."""
        return hashlib.sha3_256(b'H' + data).digest()
    
    def _hash_c(self, e_bytes: bytes, one_or_zero: int, c: bytes) -> bytes:
        """Hash function C for confirmation."""
        return hashlib.sha3_256(
            bytes([one_or_zero]) + e_bytes + c
        ).digest()
    
    def keygen(self, seed: bytes = None) -> Tuple[Dict, Dict]:
        """
        Generate key pair.
        
        Real implementation:
        1. Generate random irreducible Goppa polynomial g(x) of degree t
        2. Generate random support L = (α₁, ..., αₙ) from F_{2^m}
        3. Compute parity-check matrix H from g and L
        4. Convert H to systematic form and extract public key
        
        Returns:
            (public_key, secret_key)
        """
        if seed is None:
            seed = secrets.token_bytes(32)
        
        np.random.seed(int.from_bytes(seed[:4], 'big'))
        
        # Simplified: Generate random "Goppa-like" parity-check matrix
        r = self.m * self.t
        H = np.random.randint(0, 2, size=(r, self.n))
        
        # Ensure full rank
        while np.linalg.matrix_rank(H.astype(float)) < r:
            H = np.random.randint(0, 2, size=(r, self.n))
        
        # Convert to systematic form [A | I_r]
        H_sys = self._to_systematic(H)
        
        # Public key: matrix A (first n-r columns)
        pk_matrix = H_sys[:, :self.k]
        
        # Secret key: full H, Goppa polynomial, support
        # (simplified: just store H)
        public_key = {
            'matrix': pk_matrix,
            'n': self.n,
            't': self.t,
            'k': self.k
        }
        
        secret_key = {
            'H': H,
            'H_sys': H_sys,
            's': seed,  # For implicit rejection
            'pk': public_key
        }
        
        return public_key, secret_key
    
    def _to_systematic(self, H: np.ndarray) -> np.ndarray:
        """Convert H to systematic form [A | I_r]."""
        r, n = H.shape
        H_work = H.copy()
        
        # Gaussian elimination on last r columns
        for col in range(n - r, n):
            rel_col = col - (n - r)
            
            # Find pivot
            pivot = None
            for row in range(rel_col, r):
                if H_work[row, col] == 1:
                    pivot = row
                    break
            
            if pivot is None:
                # Swap with earlier column
                for swap_col in range(col):
                    if H_work[rel_col, swap_col] == 1:
                        H_work[:, [col, swap_col]] = H_work[:, [swap_col, col]]
                        break
                pivot = rel_col
            
            H_work[[rel_col, pivot]] = H_work[[pivot, rel_col]]
            
            for row in range(r):
                if row != rel_col and H_work[row, col] == 1:
                    H_work[row] = (H_work[row] + H_work[rel_col]) % 2
        
        return H_work
    
    def encaps(self, public_key: Dict) -> Tuple[bytes, np.ndarray]:
        """
        Encapsulate: generate shared secret and ciphertext.
        
        Steps:
        1. Generate random error e of weight t
        2. Compute ciphertext c = syndrome of e
        3. Compute shared secret K = H(e)
        
        Returns:
            (shared_secret, ciphertext)
        """
        n = public_key['n']
        t = public_key['t']
        k = public_key['k']
        A = public_key['matrix']
        
        # Generate random weight-t error
        e = np.zeros(n, dtype=int)
        positions = np.random.choice(n, t, replace=False)
        e[positions] = 1
        
        # Ciphertext = syndrome = A · e₀ + e₁ (for systematic form)
        # Where e = [e₀ | e₁] with e₀ of length k, e₁ of length r
        e0 = e[:k]
        e1 = e[k:]
        
        r = n - k
        syndrome = (A @ e0 + e1) % 2
        
        # Hash for confirmation
        e_bytes = np.packbits(e).tobytes()
        c_bytes = np.packbits(syndrome).tobytes()
        
        # Shared secret
        shared_secret = self._hash_g(e_bytes)
        
        # Ciphertext includes syndrome and confirmation hash
        ciphertext = syndrome
        
        return shared_secret, ciphertext
    
    def decaps(self, secret_key: Dict, ciphertext: np.ndarray) -> bytes:
        """
        Decapsulate: recover shared secret from ciphertext.
        
        Steps:
        1. Decode syndrome to find error e
        2. If decoding succeeds and weight(e) = t:
           Return H(e)
        3. Else (implicit rejection):
           Return H(s || c) where s is secret random value
        """
        H = secret_key['H']
        s = secret_key['s']
        
        # Decode to find error (simplified brute force for demo)
        e = self._decode(H, ciphertext)
        
        if e is not None and np.sum(e) == self.t:
            # Success
            e_bytes = np.packbits(e).tobytes()
            return self._hash_g(e_bytes)
        else:
            # Implicit rejection (constant-time in real implementation)
            c_bytes = np.packbits(ciphertext).tobytes()
            return self._hash_g(s + c_bytes)
    
    def _decode(self, H: np.ndarray, syndrome: np.ndarray) -> Optional[np.ndarray]:
        """
        Decode syndrome to error (simplified).
        
        Real implementation uses Patterson's algorithm.
        """
        from itertools import combinations
        
        r, n = H.shape
        
        # Brute force for small parameters
        if n <= 100:
            for positions in combinations(range(n), self.t):
                e = np.zeros(n, dtype=int)
                for p in positions:
                    e[p] = 1
                if np.array_equal((H @ e) % 2, syndrome):
                    return e
        
        return None

    print(f"   Ciphertext length: {len(ct)} bits")
    
    # Decapsulation
    print("\n3. DECAPSULATION")
    K_dec = kem.decaps(sk, ct)
    print(f"   Shared secret: {K_dec[:16].hex()}...")
    print(f"   Match: {K_enc == K_dec}")
```

## 4. Implementation Considerations

```python
def implementation_considerations():
    """Discuss implementation considerations."""
    print("\nIMPLEMENTATION CONSIDERATIONS")
    print("=" * 70)
    
    considerations = """
    CONSTANT-TIME REQUIREMENTS:
    ━━━━━━━━━━━━━━━━━━━━━━━━━━━
    
    All operations must be constant-time to prevent timing attacks:
    
    1. DECODING (Patterson's Algorithm)
       • Polynomial operations in F_{2^m}
       • Root finding must be constant-time
       • No early termination on success
    
    2. ERROR WEIGHT CHECK
       • Must check weight = t without revealing error
       • Use constant-time comparison
    
    3. IMPLICIT REJECTION
       • If decoding fails, return H(s || c) not error
       • Execution time must be identical to success case
       • Prevents "decoding failure oracle" attacks
    
    
    KEY GENERATION OPTIMIZATIONS:
    ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
    
    1. SEMI-SYSTEMATIC FORM ('f' variants)
       • Keep some columns non-systematic
       • Faster key generation
       • Same security level
    
    2. SEED-BASED GENERATION
       • Store seed instead of full key
       • Regenerate on demand
       • Reduces storage requirements
    
    
    PERFORMANCE CHARACTERISTICS:
    ┌─────────────────────────────────────────────────────────────────────┐
    │ Operation      │ mceliece348864    │ mceliece6688128               │
    ├─────────────────────────────────────────────────────────────────────┤
    │ KeyGen         │ ~300 ms           │ ~1500 ms                      │
    │ Encaps         │ ~0.05 ms          │ ~0.15 ms                      │
    │ Decaps         │ ~0.2 ms           │ ~0.4 ms                       │
    │ PK Size        │ 255 KB            │ 1,020 KB                      │
    │ CT Size        │ 128 B             │ 240 B                         │
    └─────────────────────────────────────────────────────────────────────┘
    
    MEMORY REQUIREMENTS:
    ━━━━━━━━━━━━━━━━━━━━
    
    • Stack: Patterson decoder needs O(n) space
    • Key storage: Public key is the bottleneck
    • Embedded systems: May need streaming/chunked operations
    
    
    SIDE-CHANNEL PROTECTIONS:
    ━━━━━━━━━━━━━━━━━━━━━━━━━
    
    1. Timing attacks: Constant-time code
    2. Power analysis: Balanced operations
    3. Cache attacks: Memory-access patterns independent of secrets
    4. Fault attacks: Error checking, redundant computation
    """
    print(considerations)

implementation_considerations()
```

## Summary

In this lesson, you learned about Classic McEliece:

1. **Design Philosophy**: Conservative, well-analyzed, maximum confidence
2. **Niederreiter Variant**: Smaller ciphertexts than original McEliece
3. **Parameter Sets**: 10 variants from Category 1 to Category 5
4. **KEM Construction**: Fujisaki-Okamoto for IND-CCA2 security
5. **Key Sizes**: 255 KB - 1.3 MB (the main tradeoff)
6. **Ciphertext Sizes**: Only 128-240 bytes (very compact)
7. **Constant-Time**: Critical for security
8. **Use Cases**: Long-term secrets, hybrid deployments, PKI roots

Classic McEliece is the most conservative choice for post-quantum security.
