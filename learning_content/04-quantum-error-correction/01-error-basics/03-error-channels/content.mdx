## 1. What is a Quantum Channel?

A quantum channel is a mathematical description of any physical process that transforms quantum states, including both ideal operations and noise.

<Definition term="Quantum Channel">
A quantum channel is a completely positive, trace-preserving (CPTP) linear map $\mathcal{E}: \mathcal{B}(\mathcal{H}_A) \rightarrow \mathcal{B}(\mathcal{H}_B)$ that transforms density matrices to density matrices while preserving the statistical interpretation of quantum mechanics.
</Definition>

The key requirements are:

1. **Linearity**: $\mathcal{E}(\alpha\rho + \beta\sigma) = \alpha\mathcal{E}(\rho) + \beta\mathcal{E}(\sigma)$
2. **Trace Preservation**: $\text{Tr}[\mathcal{E}(\rho)] = \text{Tr}[\rho] = 1$
3. **Complete Positivity**: $(\mathcal{E} \otimes \mathcal{I})(\rho_{AB}) \geq 0$ for any extension

```python
import numpy as np

def is_valid_density_matrix(rho, tol=1e-10):
    """
    Check if matrix is a valid density matrix.

    Requirements:
    1. Hermitian
    2. Positive semi-definite
    3. Trace = 1
    """
    # Hermitian check
    is_hermitian = np.allclose(rho, rho.conj().T, atol=tol)

    # Positive semi-definite check
    eigenvalues = np.linalg.eigvalsh(rho)
    is_positive = np.all(eigenvalues >= -tol)

    # Trace check
    trace_one = np.isclose(np.trace(rho), 1, atol=tol)

    return is_hermitian and is_positive and trace_one

def apply_channel(rho, kraus_ops):
    """
    Apply quantum channel defined by Kraus operators.

    E(rho) = sum_k K_k @ rho @ K_k^dagger
    """
    rho_out = np.zeros_like(rho, dtype=complex)
    for K in kraus_ops:
        rho_out += K @ rho @ K.conj().T
    return rho_out

def is_trace_preserving(kraus_ops, tol=1e-10):
    """
    Check if Kraus operators define a trace-preserving map.

    Condition: sum_k K_k^dagger @ K_k = I
    """
    d = kraus_ops[0].shape[0]
    total = sum(K.conj().T @ K for K in kraus_ops)
    return np.allclose(total, np.eye(d), atol=tol)

# Example: Bit-flip channel
def bit_flip_kraus(p):
    """Return Kraus operators for bit-flip channel."""
    K0 = np.sqrt(1 - p) * np.eye(2)
    K1 = np.sqrt(p) * np.array([[0, 1], [1, 0]])
    return [K0, K1]

# Verify properties
p = 0.1
kraus = bit_flip_kraus(p)

print("Bit-flip channel verification:")
print(f"Trace preserving: {is_trace_preserving(kraus)}")

# Apply to various states
test_states = [
    np.array([[1, 0], [0, 0]]),  # |0><0|
    np.array([[0.5, 0.5], [0.5, 0.5]]),  # |+><+|
]

for i, rho in enumerate(test_states):
    rho_out = apply_channel(rho, kraus)
    print(f"State {i+1}: Valid output = {is_valid_density_matrix(rho_out)}")
```

## 2. Common Single-Qubit Channels

### Pauli Channels

The most important class of single-qubit channels applies Pauli operators with given probabilities:

<Formula>
$$\mathcal{E}(\rho) = (1-p_x-p_y-p_z)\rho + p_x X\rho X + p_y Y\rho Y + p_z Z\rho Z$$
</Formula>

```python
import numpy as np

# Pauli matrices
I = np.eye(2, dtype=complex)
X = np.array([[0, 1], [1, 0]], dtype=complex)
Y = np.array([[0, -1j], [1j, 0]], dtype=complex)
Z = np.array([[1, 0], [0, -1]], dtype=complex)

def pauli_channel(rho, px, py, pz):
    """
    Apply general Pauli channel.

    Args:
        rho: Input density matrix
        px, py, pz: Probabilities of X, Y, Z errors

    Returns:
        Output density matrix
    """
    p_no_error = 1 - px - py - pz

    if p_no_error < 0:
        raise ValueError("Probabilities must sum to <= 1")

    rho_out = p_no_error * rho
    rho_out += px * (X @ rho @ X)
    rho_out += py * (Y @ rho @ Y)
    rho_out += pz * (Z @ rho @ Z)

    return rho_out

def pauli_channel_kraus(px, py, pz):
    """
    Return Kraus operators for Pauli channel.
    """
    p0 = 1 - px - py - pz
    return [
        np.sqrt(p0) * I,
        np.sqrt(px) * X,
        np.sqrt(py) * Y,
        np.sqrt(pz) * Z
    ]

# Example: Symmetric depolarizing channel
p = 0.1  # Total error probability
px = py = pz = p / 3

print("Symmetric depolarizing channel (p=0.1):")
plus = np.array([1, 1]) / np.sqrt(2)
rho_plus = np.outer(plus, plus.conj())

rho_out = pauli_channel(rho_plus, px, py, pz)
print(f"Input |+> fidelity: 1.0")
print(f"Output |+> fidelity: {np.real(plus.conj() @ rho_out @ plus):.4f}")
```

### Amplitude Damping Channel

Models energy relaxation (T1 decay):

```python
import numpy as np

def amplitude_damping_kraus(gamma):
    """
    Kraus operators for amplitude damping.

    gamma = 1 - exp(-t/T1)
    """
    K0 = np.array([[1, 0], [0, np.sqrt(1 - gamma)]], dtype=complex)
    K1 = np.array([[0, np.sqrt(gamma)], [0, 0]], dtype=complex)
    return [K0, K1]

def generalized_amplitude_damping_kraus(gamma, p):
    """
    Generalized amplitude damping for finite temperature.

    Args:
        gamma: Damping parameter
        p: Equilibrium excited state population
    """
    K0 = np.sqrt(p) * np.array([[1, 0], [0, np.sqrt(1-gamma)]])
    K1 = np.sqrt(p) * np.array([[0, np.sqrt(gamma)], [0, 0]])
    K2 = np.sqrt(1-p) * np.array([[np.sqrt(1-gamma), 0], [0, 1]])
    K3 = np.sqrt(1-p) * np.array([[0, 0], [np.sqrt(gamma), 0]])

    return [K0, K1, K2, K3]

# Compare zero-temperature and finite-temperature damping
gamma = 0.5

print("Amplitude damping comparison:")
print("-" * 50)

# Zero temperature (decays to |0>)
kraus_zero_T = amplitude_damping_kraus(gamma)
rho_1 = np.array([[0, 0], [0, 1]], dtype=complex)  # |1><1|
rho_out_zero_T = apply_channel(rho_1, kraus_zero_T)

print(f"Zero T: |1> -> P(|0>) = {np.real(rho_out_zero_T[0,0]):.4f}")

# Finite temperature (equilibrium has some |1>)
p_eq = 0.1  # 10% excited at equilibrium
kraus_finite_T = generalized_amplitude_damping_kraus(gamma, p_eq)
rho_out_finite_T = apply_channel(rho_1, kraus_finite_T)

print(f"Finite T: |1> -> P(|0>) = {np.real(rho_out_finite_T[0,0]):.4f}")
```

### Phase Damping Channel

Models pure dephasing (T_phi):

```python
import numpy as np

def phase_damping_kraus(gamma):
    """
    Kraus operators for phase damping.

    gamma = 1 - exp(-t/T_phi)
    """
    K0 = np.array([[1, 0], [0, np.sqrt(1 - gamma)]], dtype=complex)
    K1 = np.array([[0, 0], [0, np.sqrt(gamma)]], dtype=complex)
    return [K0, K1]

# Effect on |+> state
gamma = 0.5
kraus = phase_damping_kraus(gamma)

plus = np.array([1, 1]) / np.sqrt(2)
rho_plus = np.outer(plus, plus.conj())

rho_out = apply_channel(rho_plus, kraus)

print("Phase damping on |+> state:")
print(f"Input: rho_01 = {rho_plus[0,1]:.4f}")
print(f"Output: rho_01 = {rho_out[0,1]:.4f}")
print(f"Off-diagonal decay factor: {np.abs(rho_out[0,1])/np.abs(rho_plus[0,1]):.4f}")
```

## 3. Complete Positivity

Complete positivity is a stronger condition than positivity - it ensures the channel remains physical even when acting on part of an entangled system.

<Definition term="Complete Positivity">
A map $\mathcal{E}$ is completely positive if $(\mathcal{E} \otimes \mathcal{I}_n)(\rho) \geq 0$ for all $n$ and all states $\rho$ on the extended system.
</Definition>

```python
import numpy as np

def check_complete_positivity_choi(kraus_ops):
    """
    Check complete positivity via Choi matrix.

    The channel is CP iff its Choi matrix is positive semi-definite.
    """
    d = kraus_ops[0].shape[0]

    # Construct Choi matrix
    # Choi = sum_k (K_k tensor I) |Phi><Phi| (K_k tensor I)^dag
    # where |Phi> = sum_i |ii> is maximally entangled state

    # Build |Phi><Phi|
    phi = np.zeros((d*d, 1), dtype=complex)
    for i in range(d):
        phi[i*d + i] = 1 / np.sqrt(d)
    phi_phi = phi @ phi.conj().T

    # Build Choi matrix
    choi = np.zeros((d*d, d*d), dtype=complex)
    for K in kraus_ops:
        K_ext = np.kron(K, np.eye(d))
        choi += K_ext @ phi_phi @ K_ext.conj().T

    # Check positive semi-definiteness
    eigenvalues = np.linalg.eigvalsh(choi)
    is_positive = np.all(eigenvalues >= -1e-10)

    return is_positive, eigenvalues

def transpose_map(rho):
    """
    Transpose map - positive but NOT completely positive.
    """
    return rho.T

# Demonstrate CP vs just positive
print("Testing complete positivity:")
print("-" * 50)

# Bit-flip channel (CP)
kraus_bf = bit_flip_kraus(0.1)
is_cp, eigs = check_complete_positivity_choi(kraus_bf)
print(f"Bit-flip channel: CP = {is_cp}")

# Transpose map (not CP)
# For transpose, Kraus representation doesn't exist
# But we can check using Choi matrix directly
d = 2
choi_transpose = np.zeros((d*d, d*d), dtype=complex)
for i in range(d):
    for j in range(d):
        basis_ij = np.zeros((d, d))
        basis_ij[i, j] = 1

        # Apply transpose
        output = basis_ij.T

        # Add to Choi matrix
        choi_transpose += np.kron(output, basis_ij)

eigs_transpose = np.linalg.eigvalsh(choi_transpose)
print(f"Transpose map: CP = {np.all(eigs_transpose >= -1e-10)}")
print(f"  Min eigenvalue: {min(eigs_transpose):.4f}")
```

<Warning>
The transpose map is positive (maps positive matrices to positive matrices) but NOT completely positive. When applied to half of a Bell state, it produces a matrix with negative eigenvalue, which is unphysical!
</Warning>

## 4. Channel Composition

When multiple noise sources act, we compose their channels:

```python
import numpy as np

def compose_channels(kraus1, kraus2):
    """
    Compose two channels: E2(E1(rho)).

    Kraus operators compose as: K'_ij = K2_i @ K1_j
    """
    composed = []
    for K2 in kraus2:
        for K1 in kraus1:
            composed.append(K2 @ K1)
    return composed

def compose_channels_by_application(rho, channel_list):
    """
    Apply multiple channels in sequence.
    """
    for kraus_ops in channel_list:
        rho = apply_channel(rho, kraus_ops)
    return rho

# Example: Amplitude damping followed by dephasing
gamma_ad = 0.1  # Amplitude damping
gamma_pd = 0.2  # Phase damping

kraus_ad = amplitude_damping_kraus(gamma_ad)
kraus_pd = phase_damping_kraus(gamma_pd)

# Compose
kraus_composed = compose_channels(kraus_ad, kraus_pd)

print("Channel composition:")
print(f"Amplitude damping Kraus ops: {len(kraus_ad)}")
print(f"Phase damping Kraus ops: {len(kraus_pd)}")
print(f"Composed Kraus ops: {len(kraus_composed)}")
print(f"Trace preserving: {is_trace_preserving(kraus_composed)}")

# Compare sequential application vs composed channel
rho_test = np.array([[0.3, 0.4], [0.4, 0.7]], dtype=complex)

rho_sequential = apply_channel(rho_test, kraus_ad)
rho_sequential = apply_channel(rho_sequential, kraus_pd)

rho_composed = apply_channel(rho_test, kraus_composed)

print(f"Results match: {np.allclose(rho_sequential, rho_composed)}")
```

## 5. Channel Representations

Quantum channels have multiple equivalent representations:

### Kraus (Operator-Sum) Representation

<Formula>
$$\mathcal{E}(\rho) = \sum_k K_k \rho K_k^\dagger$$
</Formula>

### Choi Matrix Representation

```python
import numpy as np

def channel_to_choi(kraus_ops):
    """
    Convert Kraus operators to Choi matrix.
    """
    d = kraus_ops[0].shape[0]

    # Choi matrix dimension is d^2 x d^2
    choi = np.zeros((d*d, d*d), dtype=complex)

    for i in range(d):
        for j in range(d):
            # |i><j| input state
            basis_ij = np.zeros((d, d), dtype=complex)
            basis_ij[i, j] = 1

            # Apply channel
            output = apply_channel(basis_ij, kraus_ops)

            # Add to Choi matrix
            choi += np.kron(output, basis_ij)

    return choi

def choi_to_kraus(choi):
    """
    Convert Choi matrix to Kraus operators.

    Uses eigendecomposition of Choi matrix.
    """
    d_squared = choi.shape[0]
    d = int(np.sqrt(d_squared))

    # Eigendecompose Choi matrix
    eigenvalues, eigenvectors = np.linalg.eigh(choi)

    kraus_ops = []
    for i, (val, vec) in enumerate(zip(eigenvalues, eigenvectors.T)):
        if val > 1e-10:  # Only positive eigenvalues
            # Reshape eigenvector to d x d matrix
            K = np.sqrt(val) * vec.reshape(d, d)
            kraus_ops.append(K)

    return kraus_ops

# Convert bit-flip to Choi and back
kraus_bf = bit_flip_kraus(0.1)
choi_bf = channel_to_choi(kraus_bf)

print("Choi matrix for bit-flip channel:")
print(np.round(choi_bf, 4))

# Convert back
kraus_recovered = choi_to_kraus(choi_bf)
print(f"\nRecovered {len(kraus_recovered)} Kraus operators")

# Verify same channel
rho_test = np.array([[0.6, 0.3], [0.3, 0.4]], dtype=complex)
rho_original = apply_channel(rho_test, kraus_bf)
rho_recovered = apply_channel(rho_test, kraus_recovered)

print(f"Channels equivalent: {np.allclose(rho_original, rho_recovered)}")
```

### Pauli Transfer Matrix (PTM)

```python
import numpy as np

# Pauli basis (normalized)
paulis = [
    np.eye(2) / np.sqrt(2),
    np.array([[0, 1], [1, 0]]) / np.sqrt(2),
    np.array([[0, -1j], [1j, 0]]) / np.sqrt(2),
    np.array([[1, 0], [0, -1]]) / np.sqrt(2)
]

def channel_to_ptm(kraus_ops):
    """
    Convert channel to Pauli Transfer Matrix.

    PTM_ij = Tr[P_i @ E(P_j)]
    """
    ptm = np.zeros((4, 4), dtype=complex)

    for i, Pi in enumerate(paulis):
        for j, Pj in enumerate(paulis):
            # Apply channel to Pj
            output = apply_channel(Pj, kraus_ops)

            # Compute overlap with Pi
            ptm[i, j] = np.trace(Pi.conj().T @ output)

    return np.real(ptm)  # PTM should be real for physical channels

# PTM for depolarizing channel
px = py = pz = 0.1 / 3
kraus_depol = pauli_channel_kraus(px, py, pz)
ptm_depol = channel_to_ptm(kraus_depol)

print("Pauli Transfer Matrix for depolarizing channel:")
print(np.round(ptm_depol, 4))
print("\nNote: First row/column is always [1,0,0,0] for trace-preserving unital channels")
```

## 6. Channel Fidelity Measures

Different fidelity measures capture different aspects of channel quality:

```python
import numpy as np

def average_gate_fidelity(kraus_ops, target_unitary):
    """
    Calculate average gate fidelity.

    F_avg = integral over Haar |<psi|U^dag E(|psi><psi|) U|psi>| d(psi)

    For single qubit: F_avg = (d*F_ent + 1) / (d + 1)
    where F_ent is entanglement fidelity
    """
    d = kraus_ops[0].shape[0]

    # Entanglement fidelity
    # F_ent = (1/d^2) * |sum_k Tr[U^dag K_k]|^2
    total = 0
    for K in kraus_ops:
        total += np.trace(target_unitary.conj().T @ K)

    F_ent = np.abs(total)**2 / d**2

    # Convert to average gate fidelity
    F_avg = (d * F_ent + 1) / (d + 1)

    return np.real(F_avg)

def process_fidelity(kraus_ops, target_unitary):
    """
    Calculate process (entanglement) fidelity.
    """
    d = kraus_ops[0].shape[0]

    total = 0
    for K in kraus_ops:
        total += np.trace(target_unitary.conj().T @ K)

    return np.abs(total)**2 / d**2

def diamond_distance(kraus1, kraus2):
    """
    Approximate diamond distance between two channels.

    Full calculation requires SDP - this is a simple bound.
    """
    # Use trace distance on Choi matrices as approximation
    choi1 = channel_to_choi(kraus1)
    choi2 = channel_to_choi(kraus2)

    diff = choi1 - choi2
    singular_values = np.linalg.svd(diff, compute_uv=False)
    trace_distance = 0.5 * np.sum(singular_values)

    # Diamond distance >= trace distance of Choi
    return trace_distance

# Example: noisy identity gate
kraus_noisy_id = pauli_channel_kraus(0.01, 0.01, 0.01)
target_I = np.eye(2)

F_avg = average_gate_fidelity(kraus_noisy_id, target_I)
F_proc = process_fidelity(kraus_noisy_id, target_I)

print("Noisy identity gate metrics:")
print(f"Average gate fidelity: {F_avg:.6f}")
print(f"Process fidelity: {F_proc:.6f}")
print(f"Infidelity: {1 - F_avg:.6f}")
```

## 7. Multi-Qubit Channels

For multiple qubits, noise can be independent or correlated:

```python
import numpy as np

def independent_noise(rho, single_qubit_kraus_list):
    """
    Apply independent single-qubit noise to each qubit.

    Args:
        rho: n-qubit density matrix
        single_qubit_kraus_list: List of Kraus operators for each qubit
    """
    n_qubits = len(single_qubit_kraus_list)
    d = 2**n_qubits

    rho_out = np.zeros((d, d), dtype=complex)

    # Generate all combinations of Kraus operators
    from itertools import product

    kraus_indices = [range(len(ks)) for ks in single_qubit_kraus_list]

    for indices in product(*kraus_indices):
        # Build tensor product Kraus operator
        K_total = single_qubit_kraus_list[0][indices[0]]
        for i in range(1, n_qubits):
            K_total = np.kron(K_total, single_qubit_kraus_list[i][indices[i]])

        rho_out += K_total @ rho @ K_total.conj().T

    return rho_out

def correlated_two_qubit_dephasing(rho, gamma_local, gamma_corr):
    """
    Apply two-qubit dephasing with local and correlated components.

    Args:
        rho: 2-qubit density matrix
        gamma_local: Local dephasing probability
        gamma_corr: Correlated dephasing probability (ZZ noise)
    """
    I = np.eye(2)
    Z = np.array([[1, 0], [0, -1]])

    Z1 = np.kron(Z, I)
    Z2 = np.kron(I, Z)
    ZZ = np.kron(Z, Z)

    # Local dephasing
    rho_out = (1 - gamma_local)**2 * rho
    rho_out += gamma_local * (1 - gamma_local) * (Z1 @ rho @ Z1)
    rho_out += gamma_local * (1 - gamma_local) * (Z2 @ rho @ Z2)
    rho_out += gamma_local**2 * (Z1 @ Z2 @ rho @ Z2 @ Z1)

    # Correlated dephasing
    rho_out = (1 - gamma_corr) * rho_out + gamma_corr * (ZZ @ rho_out @ ZZ)

    return rho_out

# Compare independent vs correlated noise on Bell state
bell = np.array([1, 0, 0, 1]) / np.sqrt(2)
rho_bell = np.outer(bell, bell.conj())

# Independent noise
kraus_single = phase_damping_kraus(0.1)
rho_indep = independent_noise(rho_bell, [kraus_single, kraus_single])

# Correlated noise
rho_corr = correlated_two_qubit_dephasing(rho_bell, 0.1, 0.1)

print("Noise on Bell state:")
print(f"Bell fidelity (independent): {np.real(bell.conj() @ rho_indep @ bell):.4f}")
print(f"Bell fidelity (correlated): {np.real(bell.conj() @ rho_corr @ bell):.4f}")
```
