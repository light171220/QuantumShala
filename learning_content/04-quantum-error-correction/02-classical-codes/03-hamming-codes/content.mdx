# Hamming Codes

## Introduction

Hamming codes, invented by Richard Hamming in 1950, are the first class of efficient error-correcting codes. Unlike repetition codes, Hamming codes achieve error correction with much less redundancy, making them practical for real communication systems.

## The [7, 4, 3] Hamming Code

```
┌─────────────────────────────────────────────────────────────────────┐
│                   THE [7, 4, 3] HAMMING CODE                        │
├─────────────────────────────────────────────────────────────────────┤
│                                                                     │
│  PARAMETERS:                                                        │
│  • n = 7: codeword length                                          │
│  • k = 4: information bits                                         │
│  • d = 3: minimum distance                                         │
│  • r = 3: parity (check) bits                                      │
│                                                                     │
│  STRUCTURE:                                                         │
│  Position:  1   2   3   4   5   6   7                              │
│  Type:     [p₁][p₂][d₁][p₃][d₂][d₃][d₄]                           │
│             ↑   ↑       ↑                                          │
│          parity bits  data bits                                    │
│                                                                     │
│  PARITY RULES:                                                      │
│  • p₁ checks positions 1, 3, 5, 7 (binary: ***1)                   │
│  • p₂ checks positions 2, 3, 6, 7 (binary: **1*)                   │
│  • p₃ checks positions 4, 5, 6, 7 (binary: *1**)                   │
│                                                                     │
│  WHY THIS WORKS:                                                    │
│  Each position has unique binary representation 001-111            │
│  Error syndrome = binary representation of error position!         │
│                                                                     │
└─────────────────────────────────────────────────────────────────────┘
```

## Hamming Code Construction

```python
import numpy as np
from typing import List, Tuple

class HammingCode:
    """
    Implementation of the Hamming [2^r - 1, 2^r - r - 1, 3] code.

    The Hamming code is a perfect single-error-correcting code.
    """

    def __init__(self, r: int = 3):
        """
        Initialize Hamming code with r parity bits.

        Args:
            r: Number of parity bits (r >= 2)
        """
        if r < 2:
            raise ValueError("r must be at least 2")

        self.r = r
        self.n = 2**r - 1      # Block length
        self.k = 2**r - r - 1  # Information bits
        self.d = 3             # Minimum distance

        # Build matrices
        self.H = self._build_parity_check_matrix()
        self.G = self._build_generator_matrix()

    def _build_parity_check_matrix(self) -> np.ndarray:
        """
        Build parity check matrix H.

        Columns of H are binary representations of 1 to n.

        For r=3, n=7:
        H = [[0, 0, 0, 1, 1, 1, 1],   (4s place)
             [0, 1, 1, 0, 0, 1, 1],   (2s place)
             [1, 0, 1, 0, 1, 0, 1]]   (1s place)
        """
        H = np.zeros((self.r, self.n), dtype=int)

        for col in range(self.n):
            # Column is binary representation of (col + 1)
            binary = format(col + 1, f'0{self.r}b')
            for row in range(self.r):
                H[row, col] = int(binary[row])

        return H

    def _build_generator_matrix(self) -> np.ndarray:
        """
        Build systematic generator matrix G.

        G = [I_k | P] where H = [P^T | I_r]

        For systematic form, rearrange columns of H.
        """
        # Find columns that are not powers of 2 (data positions)
        data_cols = []
        parity_cols = []

        for i in range(self.n):
            if (i + 1) & i == 0:  # Power of 2
                parity_cols.append(i)
            else:
                data_cols.append(i)

        # Extract P matrix from H
        P = self.H[:, data_cols].T  # k × r

        # G = [I_k | P]
        G = np.hstack([np.eye(self.k, dtype=int), P])

        return G

    def encode(self, message: np.ndarray) -> np.ndarray:
        """
        Encode k-bit message to n-bit codeword.

        Args:
            message: k-bit message as numpy array

        Returns:
            n-bit codeword
        """
        if len(message) != self.k:
            raise ValueError(f"Message must be {self.k} bits")

        codeword = (message @ self.G) % 2
        return codeword

    def syndrome(self, received: np.ndarray) -> np.ndarray:
        """
        Calculate syndrome s = H × r^T.

        Args:
            received: n-bit received word

        Returns:
            r-bit syndrome
        """
        return (self.H @ received) % 2

    def decode(self, received: np.ndarray) -> Tuple[np.ndarray, int]:
        """
        Decode received word using syndrome decoding.

        Args:
            received: n-bit received word

        Returns:
            (corrected_codeword, error_position) where error_position=0 means no error
        """
        s = self.syndrome(received)

        # Syndrome as binary number gives error position
        error_pos = int(''.join(map(str, s)), 2)

        corrected = received.copy()
        if error_pos > 0:
            corrected[error_pos - 1] ^= 1  # Flip the error bit

        return corrected, error_pos

    def extract_message(self, codeword: np.ndarray) -> np.ndarray:
        """Extract original k-bit message from codeword."""
        # For systematic code, message is first k bits
        return codeword[:self.k]

# Demonstrate the [7, 4, 3] Hamming code
hamming = HammingCode(r=3)

print("The [7, 4, 3] Hamming Code")
print("=" * 60)

print(f"\nParameters: n={hamming.n}, k={hamming.k}, d={hamming.d}")
print(f"Rate: {hamming.k}/{hamming.n} = {hamming.k/hamming.n:.2%}")

print("\nParity Check Matrix H:")
print(hamming.H)

print("\nGenerator Matrix G:")
print(hamming.G)

# Verify G × H^T = 0
print("\nVerification: G × H^T mod 2 (should be all zeros):")
print((hamming.G @ hamming.H.T) % 2)

# Encode a message
message = np.array([1, 0, 1, 1])
codeword = hamming.encode(message)
print(f"\nMessage:  {message}")
print(f"Codeword: {codeword}")
```

## Syndrome Decoding

```
┌─────────────────────────────────────────────────────────────────────┐
│                    SYNDROME DECODING                                │
├─────────────────────────────────────────────────────────────────────┤
│                                                                     │
│  SYNDROME: s = H × r^T (mod 2)                                     │
│                                                                     │
│  THE MAGIC:                                                         │
│  • Syndrome of error pattern e = column of H at error position    │
│  • Syndrome as binary number = error position!                     │
│                                                                     │
│  EXAMPLE with [7,4,3] code:                                        │
│                                                                     │
│  Syndrome   Binary   Error Position                                │
│  ────────   ──────   ──────────────                                │
│  [0,0,0]     000        None (no error)                            │
│  [0,0,1]     001        Position 1                                 │
│  [0,1,0]     010        Position 2                                 │
│  [0,1,1]     011        Position 3                                 │
│  [1,0,0]     100        Position 4                                 │
│  [1,0,1]     101        Position 5                                 │
│  [1,1,0]     110        Position 6                                 │
│  [1,1,1]     111        Position 7                                 │
│                                                                     │
│  TO CORRECT: Flip the bit at the position indicated by syndrome   │
│                                                                     │
└─────────────────────────────────────────────────────────────────────┘
```

```python
import numpy as np

def demonstrate_syndrome_decoding():
    """Show syndrome decoding in action."""

    hamming = HammingCode(r=3)

    # Valid codeword
    message = np.array([1, 0, 1, 1])
    codeword = hamming.encode(message)

    print("Syndrome Decoding Demonstration")
    print("=" * 60)
    print(f"\nOriginal message:  {message}")
    print(f"Encoded codeword:  {codeword}")

    # Check syndrome of valid codeword
    s = hamming.syndrome(codeword)
    print(f"\nSyndrome of valid codeword: {s} (should be all zeros)")

    # Introduce errors at each position
    print("\nError at each position:")
    print(f"{'Position':<10} {'Received':<12} {'Syndrome':<10} {'Decoded':<10} {'Corrected?':<10}")
    print("-" * 55)

    for error_pos in range(8):  # 0 = no error, 1-7 = error positions
        received = codeword.copy()
        if error_pos > 0:
            received[error_pos - 1] ^= 1  # Introduce error

        s = hamming.syndrome(received)
        corrected, detected_pos = hamming.decode(received)

        is_correct = np.array_equal(corrected, codeword)
        status = "✓" if is_correct else "✗"

        print(f"{error_pos:<10} {''.join(map(str, received)):<12} {''.join(map(str, s)):<10} {detected_pos:<10} {status}")

demonstrate_syndrome_decoding()
```

## Why Hamming Codes are Perfect

```python
import numpy as np
from math import comb

class HammingCodeAnalysis:
    """
    Analyze properties of Hamming codes.

    ┌─────────────────────────────────────────────────────────────────┐
    │              HAMMING CODES ARE PERFECT                          │
    ├─────────────────────────────────────────────────────────────────┤
    │                                                                  │
    │  SPHERE PACKING BOUND:                                          │
    │  2^k × (1 + n) ≤ 2^n                                            │
    │                                                                  │
    │  For Hamming code [2^r - 1, 2^r - r - 1, 3]:                   │
    │  2^k × (1 + n) = 2^(n-r) × (1 + (2^r - 1))                     │
    │                = 2^(n-r) × 2^r                                  │
    │                = 2^n                                            │
    │                                                                  │
    │  EQUALITY! Hamming codes perfectly fill the space.             │
    │                                                                  │
    │  GEOMETRIC VIEW:                                                │
    │  • Each codeword has a "sphere" of radius 1 around it          │
    │  • Sphere contains 1 + n words (codeword + n single-errors)    │
    │  • Total words in spheres = 2^k × (1 + n) = 2^n                │
    │  • No gaps, no overlaps → perfect!                             │
    │                                                                  │
    └─────────────────────────────────────────────────────────────────┘
    """

    @staticmethod
    def verify_perfect(r: int) -> bool:
        """Verify Hamming code is perfect."""
        n = 2**r - 1
        k = 2**r - r - 1
        t = 1  # Can correct 1 error

        # Sphere volume (1 + n words)
        sphere_volume = sum(comb(n, i) for i in range(t + 1))

        # Number of codewords
        num_codewords = 2**k

        # Total volume needed
        total_needed = num_codewords * sphere_volume

        # Total space
        total_space = 2**n

        print(f"[{n}, {k}, 3] Hamming Code:")
        print(f"  Sphere volume: 1 + {n} = {sphere_volume}")
        print(f"  Number of codewords: 2^{k} = {num_codewords}")
        print(f"  Total needed: {num_codewords} × {sphere_volume} = {total_needed}")
        print(f"  Total space: 2^{n} = {total_space}")
        print(f"  Perfect: {total_needed == total_space}")

        return total_needed == total_space

    @staticmethod
    def compare_with_repetition(p: float = 0.01):
        """Compare efficiency with repetition code."""

        print("\nEfficiency Comparison (p = {:.2%} error rate):".format(p))
        print("=" * 60)

        # 3-bit repetition code
        rep_n, rep_k = 3, 1
        rep_error = 3 * p**2 * (1-p) + p**3  # P(2+ errors)
        rep_rate = rep_k / rep_n

        # [7,4,3] Hamming code
        ham_n, ham_k = 7, 4
        ham_error = sum(comb(7, i) * p**i * (1-p)**(7-i) for i in range(2, 8))
        ham_rate = ham_k / ham_n

        print(f"{'Code':<20} {'Rate':<12} {'P(error)':<15} {'Bits/correct':<15}")
        print("-" * 60)
        print(f"{'Repetition [3,1,3]':<20} {rep_rate:<12.2%} {rep_error:<15.6f} {rep_k/rep_error:<15.1f}")
        print(f"{'Hamming [7,4,3]':<20} {ham_rate:<12.2%} {ham_error:<15.6f} {ham_k/ham_error:<15.1f}")

        print(f"\nHamming is {ham_rate/rep_rate:.1f}× more efficient in rate")
        print(f"Hamming transmits {ham_k} info bits vs {rep_k} for repetition")

# Verify perfectness
analysis = HammingCodeAnalysis()
for r in [3, 4]:
    analysis.verify_perfect(r)
    print()

# Efficiency comparison
analysis.compare_with_repetition(0.01)
```

## Extended Hamming Code

```python
import numpy as np

class ExtendedHammingCode:
    """
    Extended Hamming [2^r, 2^r - r - 1, 4] code.

    Adding an overall parity bit increases distance from 3 to 4.
    Can now DETECT 2 errors (though still only correct 1).
    """

    def __init__(self, r: int = 3):
        self.r = r
        self.n = 2**r        # One more than regular Hamming
        self.k = 2**r - r - 1
        self.d = 4           # Distance increased!

        self.base_hamming = HammingCode(r)

    def encode(self, message: np.ndarray) -> np.ndarray:
        """Encode with additional parity bit."""
        base_codeword = self.base_hamming.encode(message)

        # Add overall parity bit (XOR of all bits)
        parity = np.sum(base_codeword) % 2
        extended = np.append(base_codeword, parity)

        return extended

    def decode(self, received: np.ndarray) -> Tuple[np.ndarray, str]:
        """
        Decode extended Hamming code.

        Returns (corrected, status) where status is one of:
        - 'valid': no error
        - 'corrected': single error corrected
        - 'detected': double error detected (uncorrectable)
        """
        # Check overall parity
        overall_parity = np.sum(received) % 2

        # Check Hamming syndrome
        hamming_received = received[:-1]
        syndrome = self.base_hamming.syndrome(hamming_received)
        syndrome_value = int(''.join(map(str, syndrome)), 2)

        if syndrome_value == 0 and overall_parity == 0:
            return received, 'valid'

        elif syndrome_value != 0 and overall_parity == 1:
            # Single error - correct it
            corrected = received.copy()
            corrected[syndrome_value - 1] ^= 1
            return corrected, 'corrected'

        elif syndrome_value != 0 and overall_parity == 0:
            # Double error - detected but not correctable
            return received, 'detected'

        elif syndrome_value == 0 and overall_parity == 1:
            # Error in the parity bit itself
            corrected = received.copy()
            corrected[-1] ^= 1
            return corrected, 'corrected'

        return received, 'unknown'

# Demonstrate extended Hamming
ext_hamming = ExtendedHammingCode(r=3)

print("\nExtended Hamming [8, 4, 4] Code")
print("=" * 60)
print(f"Parameters: n={ext_hamming.n}, k={ext_hamming.k}, d={ext_hamming.d}")

message = np.array([1, 0, 1, 1])
codeword = ext_hamming.encode(message)
print(f"\nMessage:  {message}")
print(f"Codeword: {codeword}")

# Test error scenarios
print("\nError Detection/Correction:")
print(f"{'Errors':<10} {'Status':<15}")
print("-" * 25)

# No error
_, status = ext_hamming.decode(codeword)
print(f"{'0':<10} {status:<15}")

# 1 error
received_1err = codeword.copy()
received_1err[2] ^= 1
_, status = ext_hamming.decode(received_1err)
print(f"{'1':<10} {status:<15}")

# 2 errors
received_2err = codeword.copy()
received_2err[2] ^= 1
received_2err[5] ^= 1
_, status = ext_hamming.decode(received_2err)
print(f"{'2':<10} {status:<15}")
```
