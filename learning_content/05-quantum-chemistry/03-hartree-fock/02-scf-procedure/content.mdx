## 1. The SCF Algorithm Overview

<Definition term="Self-Consistent Field (SCF)">
An iterative procedure to solve the Hartree-Fock equations. Starting from an initial guess, the algorithm repeatedly builds the Fock matrix and solves for new orbitals until the solution no longer changes.
</Definition>

```python
def scf_algorithm_overview():
    """Outline the SCF procedure."""

    print("Self-Consistent Field (SCF) Algorithm")
    print("=" * 60)

    steps = [
        ('1. Input', 'Molecular geometry, basis set'),
        ('2. Integrals', 'Compute S, h, (μν|λσ)'),
        ('3. Initial Guess', 'Generate starting density P⁽⁰⁾'),
        ('4. Build Fock', 'F = h + G(P) where G = J - ½K'),
        ('5. Solve', 'FC = SCε (generalized eigenvalue)'),
        ('6. Density', 'P = C_occ @ C_occ.T'),
        ('7. Check', 'If converged → done, else → step 4'),
    ]

    print("\nAlgorithm Steps:")
    for step, description in steps:
        print(f"  {step}: {description}")

    print("\nConvergence Criteria:")
    print("  • |E⁽ⁿ⁺¹⁾ - E⁽ⁿ⁾| < E_thresh (~10⁻⁸ Eh)")
    print("  • ||P⁽ⁿ⁺¹⁾ - P⁽ⁿ⁾|| < P_thresh (~10⁻⁶)")

    print("\nTypical Convergence:")
    print("  • Well-behaved molecules: 10-20 iterations")
    print("  • Transition metals: 50-100+ iterations")
    print("  • May not converge at all without tricks!")

scf_algorithm_overview()
```

## 2. The Roothaan-Hall Equations

In a finite basis, the HF equations become matrix equations:

<Formula>
$$\mathbf{FC} = \mathbf{SC}\boldsymbol{\varepsilon}$$
</Formula>

<Definition term="Roothaan-Hall Equations">
The matrix form of the Hartree-Fock equations obtained by expanding molecular orbitals in a finite basis set. F is the Fock matrix, S is the overlap matrix, C contains orbital coefficients, and ε are orbital energies.
</Definition>

```python
import numpy as np

def roothaan_hall_equations():
    """Explain the matrix HF equations."""

    print("Roothaan-Hall Equations")
    print("=" * 60)

    print("\nBasis Set Expansion:")
    print("  φᵢ(r) = Σμ Cμᵢ χμ(r)")
    print()
    print("  φᵢ = molecular orbital")
    print("  χμ = basis function (AO)")
    print("  Cμᵢ = MO coefficient matrix")

    print("\nMatrix Equation:")
    print("  FC = SCε")
    print()
    print("  F = Fock matrix (n_ao × n_ao)")
    print("  S = Overlap matrix (n_ao × n_ao)")
    print("  C = MO coefficients (n_ao × n_mo)")
    print("  ε = Orbital energies (diagonal)")

    print("\nMatrix Elements:")
    print("  Sμν = ⟨χμ|χν⟩  (overlap)")
    print("  hμν = ⟨χμ|ĥ|χν⟩  (one-electron)")
    print("  Fμν = hμν + Σλσ Pλσ[(μν|λσ) - ½(μσ|λν)]")

    print("\nDensity Matrix:")
    print("  P = 2 × C_occ @ C_occ.T  (for RHF)")
    print("  P represents electron density in AO basis")
```

## 3. Solving the Generalized Eigenvalue Problem

<Formula>
$$\mathbf{FC} = \mathbf{SC}\boldsymbol{\varepsilon}$$
</Formula>

This is a generalized eigenvalue problem because S ≠ I:

```python
import numpy as np
from scipy import linalg

def solve_generalized_eigenvalue():
    """Demonstrate solving FC = SCε."""

    print("Solving the Generalized Eigenvalue Problem")
    print("=" * 60)

    print("\nMethod 1: Orthogonalization (Standard)")
    print("-" * 45)
    print("  Transform to orthogonal basis:")
    print("  1. Compute S^(-1/2) from eigendecomposition of S")
    print("  2. F' = S^(-1/2) @ F @ S^(-1/2)")
    print("  3. Solve standard eigenvalue: F'C' = C'ε")
    print("  4. Back-transform: C = S^(-1/2) @ C'")

    print("\nMethod 2: Canonical Orthogonalization")
    print("-" * 45)
    print("  Diagonalize S: S = UsU†")
    print("  Form X = U @ s^(-1/2)")
    print("  F' = X† @ F @ X")
    print("  Solve F'C' = C'ε")
    print("  C = X @ C'")

    print("\nMethod 3: Direct (scipy.linalg.eigh)")
    print("-" * 45)
    print("  ε, C = scipy.linalg.eigh(F, S)")

    # Example
    print("\nExample (2×2):")
    F = np.array([[−1.0, −0.5], [−0.5, −0.8]])
    S = np.array([[1.0, 0.3], [0.3, 1.0]])

    ε, C = linalg.eigh(F, S)
    print(f"  Orbital energies: {ε}")

solve_generalized_eigenvalue()
```

## 4. Building the Fock Matrix

```python
def build_fock_matrix():
    """Show how to construct the Fock matrix."""

    print("Building the Fock Matrix")
    print("=" * 60)

    print("\nFock Matrix Formula:")
    print("  Fμν = hμν + Gμν")
    print()
    print("  where G is the 'two-electron' or 'G-matrix':")
    print("  Gμν = Σλσ Pλσ [(μν|λσ) − ½(μσ|λν)]")
    print("         └── Coulomb   └── Exchange")

    print("\nScaling Analysis:")
    print("  • Naive: O(N⁴) - four nested loops over AOs")
    print("  • With screening: O(N²) to O(N³)")
    print("  • With density fitting: O(N³)")
    print()
    print("  Most integrals (μν|λσ) are negligibly small!")
    print("  Schwarz screening: |(μν|λσ)| ≤ √(μν|μν)√(λσ|λσ)")

    print("\nCode Structure:")
    code = '''
def build_fock(h, eri, P):
    """Build Fock matrix from integrals and density."""
    n = h.shape[0]
    G = np.zeros((n, n))

    for mu in range(n):
        for nu in range(n):
            for lam in range(n):
                for sig in range(n):
                    # Coulomb: (μν|λσ)
                    G[mu, nu] += P[lam, sig] * eri[mu, nu, lam, sig]
                    # Exchange: -½(μσ|λν)
                    G[mu, nu] -= 0.5 * P[lam, sig] * eri[mu, sig, lam, nu]

    return h + G
'''
    print(code)

build_fock_matrix()
```

## 6. DIIS Convergence Acceleration

<Definition term="DIIS (Direct Inversion in the Iterative Subspace)">
A convergence acceleration technique that extrapolates the best Fock matrix from a history of previous iterations by minimizing the error vector in a least-squares sense.
</Definition>

```python
def diis_acceleration():
    """Explain DIIS convergence acceleration."""

    print("DIIS Convergence Acceleration")
    print("=" * 60)

    print("\nThe Problem:")
    print("  Simple SCF often oscillates or converges slowly")
    print("  Need a way to 'damp' or accelerate convergence")

    print("\nDIIS Idea:")
    print("  Keep history of Fock matrices: F₁, F₂, ..., Fₙ")
    print("  Find optimal combination: F_new = Σᵢ cᵢFᵢ")
    print("  Minimize the 'error' in a least-squares sense")

    print("\nError Vector:")
    print("  eᵢ = FᵢPᵢS − SPᵢFᵢ  (commutator)")
    print("  At convergence: [F, P] = 0 (they commute)")
    print("  So eᵢ → 0 as we converge")

    print("\nDIIS Equations:")
    print("  Minimize Σᵢⱼ cᵢcⱼ⟨eᵢ|eⱼ⟩")
    print("  Subject to: Σᵢ cᵢ = 1")
    print()
    print("  Solve linear system:")
    print("  [B  −1] [c]   [0]")
    print("  [−1  0] [λ] = [−1]")
    print()
    print("  where Bᵢⱼ = ⟨eᵢ|eⱼ⟩ = Tr(eᵢ†eⱼ)")

    print("\nPractical Settings:")
    print("  • Keep 6-10 previous Fock matrices")
    print("  • Start DIIS after 2-3 regular iterations")
    print("  • Can reduce iterations from 30+ to 10-15")

diis_acceleration()
```

<Tip>
DIIS is one of the most important algorithmic innovations in quantum chemistry. Without it, many SCF calculations would fail to converge or take hundreds of iterations.
</Tip>

```python
    print("-" * 45)
```

## 9. Complete SCF Implementation

```python
def complete_scf_pseudocode():
    """Show complete SCF implementation."""

    print("Complete SCF Pseudocode")
    print("=" * 60)

    code = '''
def scf(mol, basis, max_iter=100, E_thresh=1e-8, P_thresh=1e-6):
    """Complete SCF procedure."""

    # Step 1: Compute integrals
    S = compute_overlap(mol, basis)
    h = compute_core_hamiltonian(mol, basis)
    eri = compute_two_electron(mol, basis)
    V_nn = nuclear_repulsion(mol)

    # Step 2: Initial guess (SAD)
    P = sad_guess(mol, basis)

    # Step 3: Orthogonalization matrix
    X = orthogonalization_matrix(S)

    # Initialize DIIS
    diis = DIIS(max_vectors=8)
    E_old = 0.0

    # SCF loop
    for iteration in range(max_iter):
        # Build Fock matrix
        F = build_fock(h, eri, P)

        # DIIS extrapolation
        error = F @ P @ S - S @ P @ F
        F = diis.extrapolate(F, error)

        # Solve eigenvalue problem
        F_prime = X.T @ F @ X
        eps, C_prime = np.linalg.eigh(F_prime)
        C = X @ C_prime

        # New density matrix
        n_occ = mol.n_electrons // 2
        P_new = 2.0 * C[:, :n_occ] @ C[:, :n_occ].T

        # Compute energy
        E = 0.5 * np.sum(P_new * (h + F)) + V_nn

        # Check convergence
        dE = abs(E - E_old)
        dP = np.sqrt(np.mean((P_new - P)**2))

        print(f"Iter {iteration}: E = {E:.10f}, dE = {dE:.2e}")

        if dE < E_thresh and dP < P_thresh:
            print("SCF Converged!")
            return E, C, eps

        P = P_new
        E_old = E

    raise RuntimeError("SCF did not converge")
'''
    print(code)

complete_scf_pseudocode()
```

## Summary

In this lesson, you learned:

1. **SCF algorithm** iteratively solves the HF equations
2. **Roothaan-Hall equations** are the matrix form of HF
3. **Generalized eigenvalue** problem requires S⁻¹/² transformation
4. **Fock matrix construction** is the bottleneck (O(N⁴))
5. **Initial guess** quality affects convergence speed
6. **DIIS** dramatically accelerates convergence
7. **Level shifting and damping** help difficult cases

The SCF procedure is the workhorse of quantum chemistry.
