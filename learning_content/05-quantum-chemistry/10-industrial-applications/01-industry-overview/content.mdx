## 5. When to Use Computation

```python
def when_to_compute():
    """Decision framework for computational approaches."""

    print("When to Apply Computational Chemistry")
    print("=" * 60)

    print("\nSTRONGLY RECOMMENDED:")
    print("-" * 45)
    strong = [
        "Understanding binding mechanisms",
        "Predicting relative binding affinities",
        "Screening large virtual libraries",
        "Optimizing lead compounds",
        "Predicting ADMET properties",
        "Catalyst mechanism understanding",
        "Electronic/optical property prediction",
    ]
    for item in strong:
        print(f"  ✓ {item}")

    print("\nUSE WITH CAUTION:")
    print("-" * 45)
    caution = [
        "Absolute binding free energies (±2 kcal/mol)",
        "Reaction barriers in solution (±3 kcal/mol)",
        "pKa predictions (±1 unit typical)",
        "Solubility predictions (order of magnitude)",
        "Crystal structure prediction (challenging)",
    ]
    for item in caution:
        print(f"  ~ {item}")

    print("\nLIMITED APPLICABILITY:")
    print("-" * 45)
    limited = [
        "Toxicity prediction (complex biology)",
        "In vivo efficacy (too many variables)",
        "Clinical outcomes (requires trials)",
        "Exact polymorphic form prediction",
        "Precise formulation optimization",
    ]
    for item in limited:
        print(f"  ✗ {item}")

    print("\nDecision Factors:")
    print("-" * 45)
    print("  Consider computation when:")
    print("    • Experiments are expensive/slow")
    print("    • Chemical space is large")
    print("    • Mechanism understanding is needed")
    print("    • Property trends matter (not absolutes)")
    print()
    print("  Prefer experiments when:")
    print("    • Absolute values are required")
    print("    • Biological complexity dominates")
    print("    • Fast experimental assays exist")
    print("    • Regulatory requirements demand data")

when_to_compute()
```
