## 1. Potential Energy Surfaces

<Definition term="Potential Energy Surface (PES)">
A multidimensional surface showing how molecular energy varies with nuclear coordinates. Key features include equilibrium geometries (minima) and transition states (saddle points).
</Definition>

```python
import numpy as np

def potential_energy_surface():
    """Computing potential energy surfaces with VQE."""

    print("Potential Energy Surfaces with VQE")
    print("=" * 60)

    print("\nWhat is a PES?")
    print("  E = E(R₁, R₂, ..., Rₙ)")
    print("  Energy as function of nuclear coordinates")

    print("\nKey PES Features:")
    print("-" * 45)
    features = [
        ("Minimum", "Equilibrium geometry", "∇E = 0, ∇²E > 0"),
        ("Saddle point", "Transition state", "∇E = 0, one ∇²E < 0"),
        ("Dissociation", "Bond breaking", "E → E_fragments"),
    ]
    for feat, desc, cond in features:
        print(f"  {feat}:")
        print(f"    {desc}")
        print(f"    Condition: {cond}")
        print()

    print("VQE Workflow for PES:")
    print("-" * 45)
    steps = [
        "1. Define geometry scan (e.g., bond lengths)",
        "2. For each geometry:",
        "   a. Compute molecular integrals",
        "   b. Transform to qubit Hamiltonian",
        "   c. Run VQE optimization",
        "   d. Record optimal energy",
        "3. Plot E vs geometry",
    ]
    for step in steps:
        print(f"  {step}")

    print("\nExample: H₂ Dissociation Curve")
    print("-" * 45)
    distances = [0.5, 0.74, 1.0, 1.5, 2.0, 3.0, 5.0]
    energies = [-0.85, -1.137, -1.10, -0.95, -0.85, -0.75, -0.70]
    hf_energies = [-0.85, -1.117, -1.05, -0.75, -0.45, -0.05, 0.10]

    print(f"  {'R (Å)':<10} {'VQE (Eh)':<12} {'HF (Eh)':<12} {'VQE better?'}")
    print("  " + "-" * 45)
    for r, vqe, hf in zip(distances, energies, hf_energies):
        better = "Yes" if vqe < hf else "~Same"
        print(f"  {r:<10.2f} {vqe:<12.3f} {hf:<12.3f} {better}")

    print("\n  Note: VQE correctly dissociates, HF fails at large R")

potential_energy_surface()
```

## 2. Bond Dissociation Energies

```python
def bond_dissociation():
    """Computing bond dissociation energies with VQE."""

    print("Bond Dissociation Energies with VQE")
    print("=" * 60)

    print("\nDefinition:")
    print("  BDE = E(A) + E(B) - E(A-B)")
    print("  Energy required to break bond A-B")

    print("\nWhy VQE for Bond Breaking?")
    print("-" * 45)
    print("  • Stretched bonds have strong correlation")
    print("  • Single-reference methods (HF, DFT) fail")
    print("  • VQE handles multi-reference character")

    print("\nExample: H-H Bond Dissociation")
    print("-" * 45)
    print("  H₂ (eq) → H + H")
    print()
    print("  E(H₂ at 0.74 Å) = -1.137 Eh (VQE)")
    print("  E(H at ∞) × 2  = -1.000 Eh")
    print("  BDE = -1.000 - (-1.137) = 0.137 Eh")
    print("      = 0.137 × 627.5 = 86 kcal/mol")
    print()
    print("  Experimental: 104 kcal/mol")
    print("  (Difference due to minimal basis)")

    print("\nBDE Calculation Protocol:")
    print("-" * 45)
    protocol = [
        "1. Optimize geometry of A-B molecule",
        "2. Compute E(A-B) at equilibrium",
        "3. Compute E(A) for separated fragment",
        "4. Compute E(B) for separated fragment",
        "5. BDE = E(A) + E(B) - E(A-B)",
        "6. Add zero-point energy corrections",
    ]
    for step in protocol:
        print(f"  {step}")

    print("\nMulti-Bond Dissociation:")
    print("-" * 45)
    print("  N₂: Triple bond, very strong correlation")
    print("  O₂: Double bond + spin (triplet ground state)")
    print("  H₂O: Sequential bond breaking")
    print()
    print("  VQE can handle these challenging cases!")

bond_dissociation()
```

<Tip>
When computing bond dissociation energies, always check that both the molecule and fragments converge to correct spin states. Incorrect spin contamination leads to wrong energies.
</Tip>

## 3. Reaction Energies

<Formula>
$$\Delta E = \sum_{\text{products}} E_i - \sum_{\text{reactants}} E_j$$
</Formula>

```python
def reaction_energies():
    """Computing reaction energies with VQE."""

    print("Reaction Energies with VQE")
    print("=" * 60)

    print("\nReaction Energy:")
    print("  ΔE = Σ E(products) - Σ E(reactants)")

    print("\nExample: H₂ + F₂ → 2 HF")
    print("-" * 45)
    print("  Reactants:")
    print("    E(H₂) = -1.137 Eh")
    print("    E(F₂) = -199.12 Eh")
    print()
    print("  Products:")
    print("    E(HF) × 2 = -200.42 Eh")
    print()
    print("  ΔE = -200.42 - (-1.137 - 199.12)")
    print("     = -200.42 + 200.26")
    print("     = -0.16 Eh = -100 kcal/mol")
    print()
    print("  Reaction is exothermic (releases energy)")

    print("\nReaction Types:")
    print("-" * 45)
    reactions = [
        ("Formation", "Elements → Compound", "Standard enthalpies"),
        ("Combustion", "Fuel + O₂ → CO₂ + H₂O", "Energy content"),
        ("Isomerization", "A → A'", "Relative stability"),
        ("Bond breaking", "A-B → A + B", "Bond strength"),
    ]
    for rxn, equation, application in reactions:
        print(f"  {rxn}:")
        print(f"    {equation}")
        print(f"    Application: {application}")
        print()

    print("Error Cancellation:")
    print("-" * 45)
    print("  Isodesmic reactions: Same bond types on both sides")
    print("  Example: CH₃-CH₃ + CH₄ → 2 CH₃-CH₃")
    print()
    print("  Errors in bond descriptions cancel!")
    print("  → More accurate reaction energies")

reaction_energies()
```

## 5. Active Space Methods

```python
def active_space_methods():
    """Active space selection for larger molecules."""

    print("Active Space Methods for VQE")
    print("=" * 60)

    print("\nWhy Active Space?")
    print("  Full VQE: 2n qubits for n orbitals")
    print("  Scales exponentially!")
    print()
    print("  Active space: Select important orbitals")
    print("  Reduce qubit count dramatically")

    print("\nActive Space Definition:")
    print("-" * 45)
    print("  Core orbitals:     Doubly occupied, frozen")
    print("  Active orbitals:   Treated with VQE")
    print("  Virtual orbitals:  Unoccupied, frozen")
    print()
    print("  Notation: CAS(n_elec, n_orb)")
    print("  Example:  CAS(4,4) = 4 electrons in 4 orbitals")

    print("\nSelecting Active Space:")
    print("-" * 45)
    strategies = [
        ("Chemical intuition", "Include bond-breaking orbitals"),
        ("Natural orbital occupations", "Include fractional occupations"),
        ("Energy criteria", "Include orbitals near HOMO-LUMO"),
        ("Automated selection", "Entropy-based methods"),
    ]
    for strategy, desc in strategies:
        print(f"  {strategy}:")
        print(f"    {desc}")
        print()

    print("Example: Transition Metal Complex")
    print("-" * 45)
    print("  [Fe(H₂O)₆]²⁺")
    print()
    print("  Total orbitals: ~100 (depends on basis)")
    print("  Active space: CAS(6,5) - Fe 3d orbitals")
    print("  Qubits needed: 2 × 5 = 10 (vs ~200 for full)")

    print("\nQubit Requirements:")
    print("-" * 45)
    examples = [
        ("H₂", "CAS(2,2)", 4, "Minimal"),
        ("H₂O", "CAS(8,6)", 12, "Full valence"),
        ("N₂", "CAS(6,6)", 12, "σ + π bonds"),
        ("Fe complex", "CAS(6,5)", 10, "d orbitals"),
        ("Benzene", "CAS(6,6)", 12, "π system"),
    ]
    print(f"  {'Molecule':<12} {'Active Space':<12} {'Qubits':<8} {'Notes'}")
    print("  " + "-" * 50)
    for mol, cas, qubits, notes in examples:
        print(f"  {mol:<12} {cas:<12} {qubits:<8} {notes}")

active_space_methods()
```

<Warning>
Active space selection strongly affects results. A poorly chosen active space may miss important correlation effects, leading to qualitatively wrong answers. Validate with larger active spaces when possible.
</Warning>
