## 1. Electrochemistry Basics

<Definition term="Electrochemical Reaction">
A reaction involving electron transfer at an electrode-electrolyte interface, where the reaction free energy depends on the applied electrode potential.
</Definition>

```python
import numpy as np

def electrochemistry_intro():
    """Introduction to electrochemistry for catalysis."""

    print("Electrochemistry Fundamentals")
    print("=" * 60)

    print("\nKey Concepts:")
    print("-" * 45)
    print("  Electrode: Electronic conductor")
    print("  Electrolyte: Ionic conductor")
    print("  Interface: Where chemistry happens")
    print()
    print("  Applied potential U controls:")
    print("    • Direction of electron transfer")
    print("    • Driving force for reaction")

    print("\nThermodynamic Potential:")
    print("-" * 45)
    print("  U⁰ = -ΔG⁰ / (nF)")
    print()
    print("  ΔG⁰: Standard free energy change")
    print("  n: Number of electrons transferred")
    print("  F: Faraday constant (96485 C/mol)")
    print()
    print("  At U = U⁰: Equilibrium (no net reaction)")

    print("\nStandard Potentials (vs RHE):")
    print("-" * 45)
    reactions = [
        ("2H⁺ + 2e⁻ → H₂", 0.0, "HER"),
        ("O₂ + 4H⁺ + 4e⁻ → 2H₂O", 1.23, "ORR"),
        ("2H₂O → O₂ + 4H⁺ + 4e⁻", 1.23, "OER"),
        ("CO₂ + 2H⁺ + 2e⁻ → CO + H₂O", -0.10, "CO₂RR"),
    ]
    print(f"  {'Reaction':<28} {'U⁰ (V)':<8} {'Name'}")
    print("  " + "-" * 45)
    for rxn, u0, name in reactions:
        print(f"  {rxn:<28} {u0:<8.2f} {name}")

    print("\nOverpotential η:")
    print("-" * 45)
    print("  η = U_applied - U⁰")
    print()
    print("  η > 0: Driving force for forward reaction")
    print("  η < 0: Driving force for reverse")
    print()
    print("  Lower |η| = better catalyst!")

electrochemistry_intro()
```

## 2. Computational Hydrogen Electrode

```python
def computational_hydrogen_electrode():
    """The CHE method for electrochemistry."""

    print("Computational Hydrogen Electrode (CHE)")
    print("=" * 60)

    print("\nCore Idea:")
    print("-" * 45)
    print("  At U = 0 V vs RHE:")
    print("    H⁺ + e⁻ ↔ ½H₂(g)    ΔG = 0")
    print()
    print("  This sets the reference for (H⁺ + e⁻)")
    print("  G(H⁺) + G(e⁻) = ½G(H₂)")

    print("\nEffect of Potential:")
    print("-" * 45)
    print("  At potential U:")
    print("    ΔG(U) = ΔG(0) - neU")
    print()
    print("  Each electron becomes more (U > 0)")
    print("  or less (U < 0) energetic")
    print()
    print("  Example: A* + H⁺ + e⁻ → AH*")
    print("    ΔG = G(AH*) - G(A*) - ½G(H₂) - eU")

    print("\nEffect of pH:")
    print("-" * 45)
    print("  G(H⁺) at pH ≠ 0:")
    print("    G(H⁺) = G⁰(H⁺) + kT × ln(aH⁺)")
    print("          = G⁰(H⁺) - 0.059 × pH  (at 298K)")
    print()
    print("  But with RHE reference:")
    print("    U_RHE = U_SHE + 0.059 × pH")
    print("    pH effects cancel! Use RHE scale.")

    print("\nLimitations:")
    print("-" * 45)
    print("  • No explicit solvation")
    print("  • No electric field effects")
    print("  • No potential-dependent barriers")
    print("  • Approximate for kinetics")

computational_hydrogen_electrode()
```

<Formula name="CHE Free Energy">
$$\Delta G(U) = \Delta G(0) - neU$$

Where $n$ is the number of electrons and $U$ is the electrode potential vs RHE.
</Formula>

## 3. Hydrogen Evolution Reaction

```python
def hydrogen_evolution():
    """HER mechanism and catalyst design."""

    print("Hydrogen Evolution Reaction (HER)")
    print("=" * 60)

    print("\nOverall: 2H⁺ + 2e⁻ → H₂")
    print()

    print("Mechanisms:")
    print("-" * 45)
    print("  Volmer: H⁺ + e⁻ + * → H*")
    print("  Heyrovsky: H* + H⁺ + e⁻ → H₂ + *")
    print("  Tafel: 2H* → H₂ + 2*")
    print()
    print("  Volmer-Heyrovsky or Volmer-Tafel pathways")

    print("\nFree Energy Diagram:")
    print("-" * 45)
    print("  At U = 0 V (equilibrium):")
    print("    H⁺ + e⁻ ↔ ½H₂   ΔG = 0")
    print()
    print("  Intermediate: H*")
    print("    ΔG(H*) = G(H*) - G(*) - ½G(H₂)")
    print()
    print("  Optimal: ΔG(H*) ≈ 0")

    print("\nVolcano for HER:")
    print("-" * 45)
    metals = [
        ("Mo", -0.8, -4.0),
        ("W", -0.6, -3.5),
        ("Ni", -0.3, -2.0),
        ("Pt", -0.05, 0.0),  # Near optimum
        ("Cu", +0.2, -2.5),
        ("Au", +0.4, -3.0),
        ("Ag", +0.5, -4.0),
    ]
    print(f"  {'Metal':<6} {'ΔG(H) eV':<10} {'log(j₀)'}")
    print("  " + "-" * 30)
    for m, dg, logj in metals:
        indicator = " ← optimum" if abs(dg) < 0.1 else ""
        print(f"  {m:<6} {dg:<10.2f} {logj:.1f}{indicator}")

    print("\nBeyond Pt:")
    print("-" * 45)
    print("  MoS₂ edge sites: ΔG(H) ≈ 0.08 eV")
    print("  Ni-Mo alloys")
    print("  Transition metal phosphides")
    print("  Single-atom catalysts")

hydrogen_evolution()
```

## 4. Oxygen Reactions (OER/ORR)

```python
def oxygen_reactions():
    """OER and ORR mechanisms."""

    print("Oxygen Evolution and Reduction")
    print("=" * 60)

    print("\nOER: 2H₂O → O₂ + 4H⁺ + 4e⁻ (U⁰ = 1.23 V)")
    print("-" * 45)
    print("  Step 1: * + H₂O → OH* + H⁺ + e⁻")
    print("  Step 2: OH* → O* + H⁺ + e⁻")
    print("  Step 3: O* + H₂O → OOH* + H⁺ + e⁻")
    print("  Step 4: OOH* → O₂ + H⁺ + e⁻ + *")
    print()
    print("  Four coupled proton-electron transfers")

    print("\nScaling Relations Limit OER:")
    print("-" * 45)
    print("  ΔG(OOH) - ΔG(OH) ≈ 3.2 eV (constant)")
    print()
    print("  Ideal: Each step = 1.23 V")
    print("  Reality: Cannot optimize independently")
    print()
    print("  Minimum overpotential:")
    print("    η_min ≈ (3.2 - 2×1.23)/2 ≈ 0.37 V")

    print("\nOER Volcano (descriptor = ΔG(O) - ΔG(OH)):")
    print("-" * 45)
    catalysts = [
        ("RuO₂", 1.5, 0.35),
        ("IrO₂", 1.6, 0.40),
        ("Co₃O₄", 1.7, 0.50),
        ("NiOOH", 1.6, 0.45),
        ("Fe-Ni-OOH", 1.5, 0.30),
    ]
    print(f"  {'Catalyst':<12} {'Descriptor':<12} {'η (V)'}")
    print("  " + "-" * 35)
    for cat, desc, eta in catalysts:
        print(f"  {cat:<12} {desc:<12.1f} {eta:.2f}")

    print("\nORR: O₂ + 4H⁺ + 4e⁻ → 2H₂O (U⁰ = 1.23 V)")
    print("-" * 45)
    print("  Reverse of OER, but can follow:")
    print("    • Associative path (O₂ → OOH* → O* → OH*)")
    print("    • Dissociative path (O₂ → 2O* → ...)")
    print()
    print("  ORR is usually limited by OH* → H₂O step")
    print("  Pt is near optimal but η ≈ 0.3-0.4 V")

oxygen_reactions()
```

<Warning>
The OOH-OH scaling relation sets a fundamental limit on OER/ORR catalysts. Breaking this scaling (e.g., through different active sites for different intermediates) is key to improving efficiency.
</Warning>

## 5. CO₂ Reduction

    """CO₂ electroreduction mechanisms."""

    print("CO₂ Electrochemical Reduction (CO₂RR)")
    print("=" * 60)

    print("\nProducts and Selectivity:")
    print("-" * 45)
    products = [
        ("CO", 2, -0.10, "Easiest"),
        ("HCOOH", 2, -0.12, "Formate"),
        ("CH₃OH", 6, +0.03, "Liquid fuel"),
        ("CH₄", 8, +0.17, "High barrier"),
        ("C₂H₄", 12, +0.08, "C-C coupling!"),
        ("C₂H₅OH", 12, +0.09, "Valuable"),

    print("  " + "-" * 40)

    print("-" * 45)
    print("  CO₂ + * → *CO₂")
    print("  *CO₂ + H⁺ + e⁻ → *COOH")
    print("  *COOH + H⁺ + e⁻ → *CO + H₂O")
    print("  *CO → CO + *")

    print("  Key step: CO₂ activation")

    print("\nCompeting HER:")
    print("-" * 45)
    print("  At negative potentials, HER competes!")
    print()
    print("  H⁺ + e⁻ → ½H₂   (U⁰ = 0 V)")
    print()
    print("  Need catalyst that binds CO₂ > H")
    print("  Cu is unique: makes hydrocarbons!")

    print("  E(CO): CO binding strength")
    print("  E(H): Selectivity vs HER")

    print("\nC-C Coupling on Cu:")
    print("-" * 45)

    print()

    print()

## 6. Electrocatalyst Design

```python
def electrocatalyst_design():
    """Computational design of electrocatalysts."""

    print("Computational Electrocatalyst Design")
    print("=" * 60)

    print("\nDesign Principles:")
    print("-" * 45)
    print("  1. Tune binding to volcanic optimum")
    print("  2. Break scaling relations if possible")
    print("  3. Suppress competing reactions")
    print("  4. Ensure stability under conditions")

    print("\nStrategy - Alloying:")
    print("-" * 45)
    print("  Combine metals to shift binding")
    print()
    print("  Example: Pt₃Ni for ORR")
    print("    Ni dissolution creates Pt-skin")
    print("    Subsurface Ni modifies d-band")
    print("    → Optimal OH binding")

    print("\nStrategy - Single-Atom Catalysts:")
    print("-" * 45)
    print("  M-N-C materials (M = Fe, Co, Ni)")
    print()
    print("  Fe-N₄ for ORR:")
    print("    Different from bulk Fe surface")
    print("    Can have different scaling")
    print()
    print("  Ni-N₄ for CO₂RR:")
    print("    Selective for CO over HER")

    print("\nStrategy - Oxide/Hydroxide Surfaces:")
    print("-" * 45)
    print("  For OER: NiFeOOH catalysts")
    print()
    print("  Surface reconstruction under OER")
    print("  Fe doping modifies Ni sites")
    print("  Synergy between metals")

    print("\nStability Considerations:")
    print("-" * 45)
    print("  Electrochemical stability:")
    print("    • Pourbaix diagram analysis")
    print("    • Dissolution potential")
    print()
    print("  Under operation:")
    print("    • High potentials may oxidize")
    print("    • Low pH may dissolve")
    print("    • Reducing conditions may reduce")

    print("\nComputational Workflow:")
    print("-" * 45)
    print("  1. Screen candidates with CHE")
    print("  2. Calculate full free energy diagram")
    print("  3. Include solvation corrections")
    print("  4. Check stability (Pourbaix)")
    print("  5. Calculate selectivity")

electrocatalyst_design()
```

<Tip>
When modeling electrocatalysis, always include solvation corrections. Water stabilizes intermediates like OH* and OOH* by ~0.3-0.5 eV through hydrogen bonding.
</Tip>
