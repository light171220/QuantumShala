## 1. Catalyst Descriptors

<Definition term="Descriptor">
A computable property that correlates with catalytic activity, allowing rapid screening of catalyst candidates without calculating full reaction mechanisms.
</Definition>

```python
import numpy as np

def descriptor_overview():
    """Overview of descriptors in catalyst design."""

    print("Catalyst Descriptors")
    print("=" * 60)

    print("\nWhat Makes a Good Descriptor?")
    print("-" * 45)
    print("  • Computable from simple calculations")
    print("  • Correlates strongly with activity")

    print("\nElectronic Descriptors:")
    print("-" * 45)
    descriptors = [
        ("d-band center", "ε_d", "Metal reactivity"),
```

## 2. Scaling Relations

```python
def scaling_relations():
    """Linear scaling relations for adsorbates."""

    print("Scaling Relations")
    print("=" * 60)

    print("\nConcept:")
    print("-" * 45)
    print("  Adsorption energies of similar species correlate")
    print()
    print("  E(AHₓ) = γ × E(A) + ξ")
    print()
    print("  γ: Slope (valence-based)")
    print("  ξ: Intercept (constant)")

    print("\nOxygen-Containing Intermediates:")
    print("-" * 45)
    print("  E(OH) = 0.5 × E(O) + const")
    print("  E(OOH) = E(OH) + 3.2 eV (universal)")
    print()
    print("  This sets constraint on ORR catalysts!")

    print("\nNitrogen Intermediates:")
    print("-" * 45)
    print("  E(NH) = 0.75 × E(N) + const")
    print("  E(NH₂) = 0.50 × E(N) + const")
    print("  E(NH₃) = 0.25 × E(N) + const")
    print()
    print("  Slope = (remaining bonds) / (max bonds)")

    print("\nCarbon Intermediates:")
    print("-" * 45)
    print("  CHₓ scales with E(C)")
    print("  E(CHₓ) = (4-x)/4 × E(C) + const")

    print("\nImplications:")
    print("-" * 45)
    print("  Cannot independently tune all intermediates")
    print("  Limits optimal catalyst design")
    print()
    print("  Breaking scaling relations:")
    print("    • Single-atom catalysts")
    print("    • Bifunctional surfaces")
    print("    • Promoters that affect TS selectively")

scaling_relations()
```

<Formula name="Scaling Relation">
$$E(AH_x) = \gamma \cdot E(A) + \xi$$

Where $\gamma = \frac{v_{max} - x}{v_{max}}$ with $v_{max}$ being the maximum valence of atom A.
</Formula>

## 3. Volcano Plots

```python
def volcano_plots():
    """Volcano plot analysis for catalyst screening."""

    print("Volcano Plots")
    print("=" * 60)

    print("\nConstruction:")
    print("-" * 45)
    print("  x-axis: Descriptor (e.g., E_O)")
    print("  y-axis: Activity (TOF, current, rate)")
    print()
    print("  Peak: Sabatier optimum")
    print("  Left: Too weak binding (activation-limited)")
    print("  Right: Too strong binding (desorption-limited)")

    print("\nExample - Oxygen Reduction (ORR):")
    print("-" * 45)
    print("  Metal    E(O) (eV)  log(activity)")
    metals = [
        ("Au", -3.0, -4.0),
        ("Ag", -3.5, -2.5),
        ("Cu", -4.0, -1.5),
        ("Pt", -4.5, 0.0),  # Peak
        ("Pd", -5.0, -0.8),
        ("Ir", -5.2, -1.2),
        ("Ru", -5.5, -2.0),
    ]
    for metal, e_o, activity in metals:
        bar = "█" * int(5 + activity)
        print(f"  {metal:<4} {e_o:>6.1f}   {bar}")

    print("\nTwo-Descriptor Volcanoes:")
    print("-" * 45)
    print("  For reactions with multiple key intermediates")
    print("  E.g., CO₂ reduction: E(CO) and E(H)")
    print()
    print("  Allows more nuanced catalyst design")
    print("  Identify regions of high activity + selectivity")

    print("\nBeyond the Peak:")
    print("-" * 45)
    print("  Pt is near ORR peak - hard to improve!")
    print("  Strategies:")
    print("    • Move along peak (alloying)")
    print("    • Break scaling (unique TS stabilization)")
    print("    • Different mechanism (enzyme-like)")

volcano_plots()
```

## 4. Machine Learning for Catalysis

```python
def ml_catalysis():
    """Machine learning in catalyst design."""

    print("Machine Learning for Catalyst Design")
    print("=" * 60)

    print("\nWhy ML?")
    print("-" * 45)
    print("  • DFT is slow (hours-days per structure)")
    print("  • Millions of possible alloys/structures")
    print("  • Need rapid screening + refinement")

    print("\nML Workflow:")
    print("-" * 45)
    print("  1. Generate DFT training data")
    print("  2. Extract/learn features")
    print("  3. Train surrogate model")
    print("  4. Screen large candidate space")
    print("  5. Validate top candidates with DFT")

    print("\nFeature Types:")
    print("-" * 45)
    features = [
        ("Elemental", "Atomic number, electronegativity, radius"),
        ("Compositional", "Mixing enthalpy, d-band filling"),
        ("Structural", "Coordination, distances, strain"),
        ("Electronic", "d-band center, DOS features"),
        ("Learned", "From graph neural networks"),
    ]
    for ftype, desc in features:
        print(f"  {ftype:<14}: {desc}")

    print("\nModel Types:")
    print("-" * 45)
    models = [
        ("Random Forest", "Robust, interpretable"),
        ("Gradient Boosting", "High accuracy"),
        ("Gaussian Process", "Uncertainty estimates"),
        ("Neural Networks", "Complex patterns"),
        ("Graph NN", "Structure-aware, transferable"),
    ]
    print(f"  {'Model':<18} {'Strengths'}")
    print("  " + "-" * 35)
    for model, strength in models:
        print(f"  {model:<18} {strength}")

    print("\nActive Learning:")
    print("-" * 45)
    print("  Iterate: Train → Predict → Select → Calculate → Add")
    print("  Focus on uncertain regions (exploitation)")
    print("  Or high predicted value (exploration)")

ml_catalysis()
```

<Tip>
Gaussian Process models provide uncertainty estimates, making them ideal for active learning workflows where you want to balance exploration and exploitation.
</Tip>

## 6. Design Strategies

```python
def design_strategies():
    """Strategies for computational catalyst design."""

    print("Computational Catalyst Design Strategies")
    print("=" * 60)

    print("\nAlloying:")
    print("-" * 45)
    print("  Combine metals to tune d-band")
    print()
    print("  d-band shifts:")
    print("    • Ligand effect: Neighbors change ε_d")
    print("    • Strain effect: Lattice mismatch")
    print("    • Ensemble effect: Site geometry")
    print()
    print("  Example: Pt₃Ni for ORR")
    print("    Better than Pt due to Ni dissolution effects")

    print("\nSingle-Atom Catalysts (SACs):")
    print("-" * 45)
    print("  Isolated metal atoms on support")
    print("  Break scaling through unique coordination")
    print()
    print("  M-N₄ sites (M = Fe, Co, Ni)")
    print("  Different from bulk M surface")

    print("\nNanostructuring:")
    print("-" * 45)
    print("  Shape controls exposed facets")
    print("  Size affects coordination numbers")
    print()
    print("  Cubes: (100) facets")
    print("  Octahedra: (111) facets")
    print("  High-index: Steps, kinks")

    print("\nSupport Effects:")
    print("-" * 45)
    print("  Strong metal-support interaction (SMSI)")
    print("  Charge transfer from support")
    print("  Bifunctional mechanisms")
    print()
    print("  Examples:")
    print("    Au/TiO₂: Interface sites active")
    print("    Pt/CeO₂: Oxygen spillover")

    print("\nPromoters:")
    print("-" * 45)
    print("  Additives that enhance activity")
    print("  Often electronegative/electropositive elements")
    print()
    print("  K promotion of Fe for NH₃:")
    print("    Donates electrons, stabilizes N₂*")

design_strategies()
```

<Warning>
ML models trained on one class of materials may not transfer to others. Always validate predictions with DFT before experimental work, especially for novel catalyst types.
</Warning>
