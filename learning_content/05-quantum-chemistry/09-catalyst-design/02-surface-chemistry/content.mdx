## 1. Surface Models

<Definition term="Slab Model">
A periodic representation of a surface where a finite number of atomic layers represents the bulk, separated by vacuum to prevent interaction between periodic images.
</Definition>

```python

def surface_energy():

    print("=" * 60)

    print("  E_slab: Energy of slab")

    print("  E_bulk: Energy per atom in bulk")
```

## 3. d-Band Theory

```python
def dband_theory():
    """d-band model for transition metal reactivity."""

    print("d-Band Theory of Surface Reactivity")
    print("=" * 60)

    print("\nCore Concept:")
    print("-" * 45)
    print("  Adsorbate interacts with metal d-band")
    print("  Creates bonding/antibonding states")
    print()
    print("  ε_d = d-band center (relative to Fermi)")
    print()
    print("  Higher ε_d → stronger adsorption")
    print("  Lower ε_d → weaker adsorption")

    print("\nd-Band Center:")
    print("-" * 45)
    print("  ε_d = ∫ε × n_d(ε) dε / ∫n_d(ε) dε")
    print()
    print("  n_d(ε) = d-projected density of states")

    print("\nTrends Across Metals:")
    print("-" * 45)
    dband_centers = [
        ("Au", -3.5, "Most noble"),
        ("Ag", -4.3, "Noble"),
        ("Cu", -2.7, "Intermediate"),
        ("Pt", -2.3, "Catalytic"),
        ("Pd", -1.8, "Reactive"),
        ("Ni", -1.3, "Very reactive"),
    ]
    print(f"  {'Metal':<8} {'ε_d (eV)':<12} {'Reactivity'}")
    print("  " + "-" * 35)
    for metal, eps_d, react in dband_centers:
        print(f"  {metal:<8} {eps_d:<12.1f} {react}")

    print("\nEffect of Coordination:")
    print("-" * 45)
    print("  Lower coordination → higher ε_d → stronger binding")
    print()
    print("  Sites (decreasing ε_d):")
    print("    Adatom > Step > Kink > Terrace")
    print()
    print("  This explains higher reactivity at defects")

    print("\nEffect of Strain:")
    print("-" * 45)
    print("  Tensile strain: d-band narrowing → ε_d up")
    print("  Compressive: d-band broadening → ε_d down")
    print()
    print("  → Used to tune catalyst activity!")

dband_theory()
```

<Formula name="d-Band Center">
$$\varepsilon_d = \frac{\int_{-\infty}^{E_F} \varepsilon \cdot n_d(\varepsilon) d\varepsilon}{\int_{-\infty}^{E_F} n_d(\varepsilon) d\varepsilon}$$

Where $n_d(\varepsilon)$ is the d-projected density of states and $E_F$ is the Fermi level.
</Formula>

## 6. Work Function and Charge Transfer

```python
def work_function():
    """Work function and surface charge transfer."""

    print("Work Function and Charge Transfer")
    print("=" * 60)

    print("\nWork Function Definition:")
    print("-" * 45)
    print("  Φ = V_vacuum - E_Fermi")
    print()
    print("  Minimum energy to remove electron to infinity")

    print("\nTypical Work Functions:")
    print("-" * 45)
    wf_values = [
        ("Cs", 2.1, "Lowest"),
        ("Na", 2.8, "Alkali"),
        ("Al", 4.3, "Main group"),
        ("Cu", 4.7, "Transition"),
        ("Pt", 5.7, "High"),
        ("Au", 5.3, "Noble"),
    ]
    print(f"  {'Metal':<8} {'Φ (eV)':<10} {'Notes'}")
    print("  " + "-" * 30)
    for metal, wf, notes in wf_values:
        print(f"  {metal:<8} {wf:<10.1f} {notes}")

    print("\nAdsorbate Effect on Φ:")
    print("-" * 45)
    print("  Electropositive (K, Na): Φ decreases")
    print("  Electronegative (O, F): Φ increases")
    print()
    print("  Helmholtz equation:")
    print("  ΔΦ = -θ × μ / (ε₀ × A)")
    print("  μ = dipole moment, A = area per site")

    print("\nCharge Transfer Analysis:")
    print("-" * 45)
    print("  Bader charge analysis")
    print("  Charge density difference:")
    print("    Δρ = ρ(ads/surf) - ρ(surf) - ρ(ads)")
    print()
    print("  Positive Δρ on adsorbate → accepts electrons")
    print("  Negative Δρ → donates electrons")

work_function()
```
