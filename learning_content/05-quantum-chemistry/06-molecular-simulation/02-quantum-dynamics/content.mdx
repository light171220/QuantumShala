## 1. Beyond Classical Force Fields

<Definition term="Ab Initio Molecular Dynamics (AIMD)">
Molecular dynamics where forces are computed from quantum mechanical electronic structure calculations at each time step, rather than from parameterized force fields.
</Definition>

```python
def beyond_force_fields():
    """Motivation for quantum molecular dynamics."""

    print("Beyond Classical Force Fields")
    print("=" * 60)

    print("\nLimitations of Force Fields:")
    print("-" * 45)
    limitations = [
        ("Fixed bonding", "Can't break/form bonds"),
        ("Parameterization", "Limited to trained systems"),
        ("Electronic effects", "No charge transfer, polarization"),
        ("Excited states", "Ground state only"),
    ]
    for limit, issue in limitations:
        print(f"  {limit}:")
        print(f"    {issue}")

    print("\nWhen Quantum is Needed:")
    print("-" * 45)
    print("  • Chemical reactions (bond breaking/forming)")
    print("  • Charge transfer processes")
    print("  • Metal coordination chemistry")
    print("  • Excited state dynamics")
    print("  • Novel materials (no parameters available)")

    print("\nAb Initio MD Concept:")
    print("-" * 45)
    print("  At each MD step:")
    print("    1. Compute electronic structure (DFT/HF)")
    print("    2. Get forces: F = -∂E/∂R")
    print("    3. Propagate nuclei (classical)")
    print("    4. Repeat")
```

## 2. Born-Oppenheimer Molecular Dynamics

<Formula>
$$M_I \frac{d^2\mathbf{R}_I}{dt^2} = -\nabla_I E_0(\{\mathbf{R}\})$$
</Formula>

```python
def born_oppenheimer_md():
    """Born-Oppenheimer molecular dynamics."""

    print("Born-Oppenheimer Molecular Dynamics (BOMD)")
    print("=" * 60)

    print("\nBorn-Oppenheimer Approximation:")
    print("-" * 45)
    print("  Electrons instantaneously adjust to nuclear positions")
    print("  Solve: Ĥₑ|ψ₀⟩ = E₀(R)|ψ₀⟩ at each geometry")
    print()
    print("  Nuclear motion on ground state PES")

    print("\nBOMD Algorithm:")
    print("-" * 45)
    steps = [
        "1. Given nuclear positions R",
        "2. Solve electronic Schrödinger equation → E₀(R), |ψ₀(R)⟩",
        "3. Compute Hellmann-Feynman forces:",
        "   Fᵢ = -∂E₀/∂Rᵢ = -⟨ψ₀|∂Ĥ/∂Rᵢ|ψ₀⟩",
        "4. Propagate nuclei: R(t+Δt) from forces",
        "5. Repeat from step 2",
    ]
    for step in steps:
        print(f"  {step}")

    print("\nHellmann-Feynman Theorem:")
    print("-" * 45)
    print("  dE/dλ = ⟨ψ|∂Ĥ/∂λ|ψ⟩")
    print()
    print("  For nuclear coordinate Rᵢ:")
    print("  Fᵢ = -⟨ψ₀|∂Ĥ/∂Rᵢ|ψ₀⟩")
    print()
    print("  Only the explicit Ĥ dependence matters!")
    print("  (If ψ is exact eigenfunction)")

    print("\nAdvantages of BOMD:")
    print("-" * 45)
    print("  • Conceptually simple")
    print("  • Any electronic structure method")
    print("  • Born-Oppenheimer PES well-defined")

    print("\nDisadvantages of BOMD:")
    print("-" * 45)
    print("  • Full SCF at each step (expensive)")
    print("  • Small time step (0.5-1 fs)")
    print("  • No non-adiabatic effects")

born_oppenheimer_md()
```

## 3. Car-Parrinello Molecular Dynamics

```python
def car_parrinello_md():
    """Car-Parrinello molecular dynamics method."""

    print("Car-Parrinello Molecular Dynamics (CPMD)")
    print("=" * 60)

    print("\nKey Idea:")
    print("  Treat electronic orbitals as dynamical variables")
    print("  Propagate electrons AND nuclei together")
    print("  Avoid full SCF at each step!")

    print("\nExtended Lagrangian:")
    print("-" * 45)
    print("  L = ½ Σᵢ Mᵢ Ṙᵢ² + ½ μ Σⱼ ⟨φ̇ⱼ|φ̇ⱼ⟩")
    print("      - E[{φⱼ}, {Rᵢ}]")
    print("      + constraints (orthonormality)")
    print()
    print("  μ = fictitious electron mass")
    print("  φⱼ = Kohn-Sham orbitals")

    print("\nEquations of Motion:")
    print("-" * 45)
    print("  Nuclei:    Mᵢ R̈ᵢ = -∂E/∂Rᵢ")
    print("  Electrons: μ φ̈ⱼ = -δE/δφⱼ* + constraints")
    print()
    print("  Electrons 'follow' nuclei adiabatically")
    print("  if μ is small enough")

    print("\nAdiabatic Separation:")
    print("-" * 45)
    print("  Need: ω_electron >> ω_nuclear")
    print("  Electron frequency: ω_e ~ √(E_gap/μ)")
    print("  Nuclear frequency: ω_n ~ √(k/M)")
    print()
    print("  Requires μ ~ 200-500 a.u. (light)")
    print("  Time step ~ 0.1-0.2 fs (smaller than BOMD)")

    print("\nAdvantages of CPMD:")
    print("-" * 45)
    print("  • No SCF convergence at each step")
    print("  • Very stable dynamics")
    print("  • Efficient for large systems")

    print("\nDisadvantages of CPMD:")
    print("-" * 45)
    print("  • Very small time step")
    print("  • Electron 'heating' possible")
    print("  • μ parameter sensitivity")

car_parrinello_md()
```

<Tip>
Car-Parrinello MD trades smaller time steps for avoiding full SCF at each step. This is most efficient for DFT with plane-wave basis sets where iterative diagonalization is expensive.
</Tip>

## 4. Forces and Pulay Terms

```python
def forces_pulay():
    """Computing forces in AIMD."""

    print("Forces in Ab Initio MD")
    print("=" * 60)

    print("\nHellmann-Feynman Forces:")
    print("-" * 45)
    print("  F_HF = -⟨ψ|∂Ĥ/∂R|ψ⟩")
    print()
    print("  Valid when:")
    print("    1. ψ is exact eigenfunction of Ĥ")
    print("    2. Or: basis set is complete/independent of R")

    print("\nPulay (Wavefunction) Forces:")
    print("-" * 45)
    print("  With atom-centered basis: χ_μ(r-R)")
    print("  Basis functions depend on nuclear positions!")
    print()
    print("  Full force: F = F_HF + F_Pulay")
    print()
    print("  F_Pulay = -Σ_μν P_μν ∂S_μν/∂R × ε")
    print()
    print("  P = density matrix")
    print("  S = overlap matrix")
    print("  ε = orbital energies")

    print("\nBasis Set Considerations:")
    print("-" * 45)
    basis_types = [
        ("Plane waves", "No Pulay", "F = F_HF only"),
        ("Gaussian", "Pulay present", "Need derivatives"),
        ("Numerical", "Pulay present", "Need derivatives"),
    ]
    print(f"  {'Basis':<15} {'Pulay force':<15} {'Notes'}")
    print("  " + "-" * 45)
    for basis, pulay, notes in basis_types:
        print(f"  {basis:<15} {pulay:<15} {notes}")

    print("\nNumerical vs Analytical Forces:")
    print("-" * 45)
    print("  Numerical: F ≈ -[E(R+δ) - E(R-δ)]/(2δ)")
    print("    Pros: Simple, no implementation")
    print("    Cons: Expensive (2N evaluations)")
    print()
    print("  Analytical: Derived gradient expressions")
    print("    Pros: Efficient, exact")
    print("    Cons: Complex implementation")

forces_pulay()
```

```python
    print("    4. Rescale velocity for energy conservation")

    print("-" * 45)
```
