## 1. Introduction to Monte Carlo

<Definition term="Monte Carlo Simulation">
A computational method that uses random sampling to estimate properties of a system by generating configurations according to their Boltzmann probability distribution.
</Definition>

```python
import numpy as np

def monte_carlo_intro():
    """Introduction to Monte Carlo methods."""

    print("Introduction to Monte Carlo Methods")
    print("=" * 60)

    print("\nKey Idea:")
    print("  Sample configurations from probability distribution")
    print("  Compute averages over samples")

    print("\nBoltzmann Distribution:")
    print("-" * 45)
    print("  P(x) ∝ exp(-E(x)/kT)")
    print()
    print("  Most probable: low energy configurations")
    print("  Goal: sample efficiently from this distribution")

    print("\nThermodynamic Average:")
    print("-" * 45)
    print("  ⟨A⟩ = ∫ A(x) P(x) dx / ∫ P(x) dx")
    print()
    print("  Monte Carlo estimate:")
    print("  ⟨A⟩ ≈ (1/N) Σᵢ A(xᵢ)")
    print()
    print("  where xᵢ are samples from P(x)")

    print("\nMC vs MD:")
    print("-" * 45)
    comparison = [
        ("Dynamics", "No real dynamics", "Real trajectories"),
        ("Sampling", "Jump between configs", "Continuous motion"),
        ("Efficiency", "Non-physical moves OK", "Physical moves only"),
        ("Parallelization", "Independent samples", "Sequential"),
    ]
    print(f"  {'Aspect':<18} {'MC':<20} {'MD'}")
    print("  " + "-" * 55)
    for aspect, mc, md in comparison:
        print(f"  {aspect:<18} {mc:<20} {md}")

monte_carlo_intro()
```

## 2. Metropolis Algorithm

<Formula>
$$P_{\text{accept}} = \min\left(1, e^{-\Delta E / k_B T}\right)$$
</Formula>

```python
def metropolis_algorithm():
    """The Metropolis-Hastings algorithm."""

    print("Metropolis Algorithm")
    print("=" * 60)

    print("\nAlgorithm Steps:")
    print("-" * 45)
    steps = [
        "1. Start with configuration x",
        "2. Propose trial move: x → x'",
        "3. Compute ΔE = E(x') - E(x)",
        "4. If ΔE < 0: accept (downhill)",
        "5. If ΔE > 0: accept with P = exp(-ΔE/kT)",
        "6. Repeat from step 2",
    ]
    for step in steps:
        print(f"  {step}")

    print("\nAcceptance Criterion:")
    print("-" * 45)
    print("  P_accept = min(1, exp(-ΔE/kT))")
    print()
    print("  Generate random r ∈ [0,1)")
    print("  Accept if r < exp(-ΔE/kT)")

    print("\nDetailed Balance:")
    print("-" * 45)
    print("  Key requirement for equilibrium sampling")
    print("  P(x)T(x→x') = P(x')T(x'→x)")
    print()
    print("  T = transition probability")
    print("  Ensures correct Boltzmann distribution")

    print("\nExample: 1D Particle")
    print("-" * 45)
    print("  V(x) = ½kx² (harmonic)")
    print("  Move: x' = x + δ (random step)")
    print("  ΔE = ½k(x'² - x²)")

    print("\nTypical Acceptance Rates:")
    print("-" * 45)
    print("  Target: 20-50% acceptance")
    print("  Too high: steps too small (slow exploration)")
    print("  Too low: many rejections (inefficient)")

metropolis_algorithm()
```

<Tip>
Adjust the maximum step size to achieve ~30-50% acceptance rate. This balances between making large moves (good exploration) and having moves accepted (efficiency).
</Tip>

## 4. Ensemble Sampling

```python
def ensemble_sampling():
    """Sampling different ensembles with MC."""

    print("Monte Carlo Ensemble Sampling")
    print("=" * 60)

    print("\nNVT (Canonical):")
    print("-" * 45)
    print("  Fixed: N, V, T")
    print("  Moves: Particle displacement")
    print("  Accept: exp(-ΔE/kT)")

    print("\nNPT (Isothermal-Isobaric):")
    print("-" * 45)
    print("  Fixed: N, P, T")
    print("  Moves: Particle + Volume")
    print()
    print("  Volume move acceptance:")
    print("  P = exp(-[ΔE + PΔV - NkT ln(V'/V)]/kT)")

    print("\nμVT (Grand Canonical):")
    print("-" * 45)
    print("  Fixed: μ (chemical potential), V, T")
    print("  Moves: Displacement + Insertion/Deletion")
    print()
    print("  Insertion: P = min(1, V/(N+1) × exp((μ-ΔE)/kT))")
    print("  Deletion: P = min(1, N/V × exp((-μ+ΔE)/kT))")

    print("\nGibbs Ensemble:")
    print("-" * 45)
    print("  Two boxes in equilibrium")
    print("  For phase coexistence")
    print("  Moves: Particle transfer between boxes")

ensemble_sampling()
```
