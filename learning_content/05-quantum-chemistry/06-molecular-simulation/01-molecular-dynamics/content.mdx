## 1. Introduction to Molecular Dynamics

<Definition term="Molecular Dynamics (MD)">
A computational method that simulates the time evolution of a molecular system by solving Newton's equations of motion for all atoms, generating trajectories that reveal dynamical and thermodynamic properties.
</Definition>


## 2. Newton's Equations of Motion

<Formula>
$$m_i \frac{d^2\mathbf{r}_i}{dt^2} = \mathbf{F}_i = -\nabla_i V(\mathbf{r}_1, \mathbf{r}_2, ..., \mathbf{r}_N)$$
</Formula>

```python
import numpy as np

def equations_of_motion():
    """Newton's equations for molecular systems."""

    print("Newton's Equations of Motion")
    print("=" * 60)

    print("\nFor Each Atom i:")
    print("  mᵢ d²rᵢ/dt² = Fᵢ = -∇ᵢV")
    print()
    print("  mᵢ = mass of atom i")
    print("  rᵢ = position vector (x, y, z)")
    print("  Fᵢ = force on atom i")
    print("  V = potential energy function")

    print("\nFirst-Order Form:")
    print("-" * 45)
    print("  drᵢ/dt = vᵢ         (velocity)")
    print("  dvᵢ/dt = Fᵢ/mᵢ      (acceleration)")
    print()
    print("  Coupled ODEs for 3N coordinates")

    print("\nConserved Quantities:")
    print("-" * 45)
    print("  Total energy: E = K + V = constant")
    print("    K = Σᵢ ½mᵢvᵢ²  (kinetic)")
    print("    V = V(r₁...rₙ)   (potential)")
    print()
    print("  Total momentum: P = Σᵢ mᵢvᵢ = constant")
    print("  (if no external forces)")

    print("\nPhase Space:")
    print("-" * 45)
    print("  Full state: (r₁...rₙ, v₁...vₙ)")
    print("  6N dimensional phase space")
    print("  Trajectory: path through phase space")

    print("\nExample: 2-Atom System")
    print("-" * 45)
    print("  m₁ a₁ = -dV/dr₁")
    print("  m₂ a₂ = -dV/dr₂")
    print()
    print("  If V = ½k(r₂-r₁-r₀)² (harmonic spring)")
    print("  F₁ = -k(r₁-r₂+r₀)")
    print("  F₂ = -k(r₂-r₁-r₀)")

equations_of_motion()
```

## 4. Integration Algorithms

```python
def integration_algorithms():
    """Numerical integration for MD."""

    print("Integration Algorithms for MD")
    print("=" * 60)

    print("\nThe Challenge:")
    print("  Solve: d²r/dt² = F(r)/m")
    print("  Need: stable, time-reversible, energy-conserving")

    print("\nVerlet Algorithm:")
    print("-" * 45)
    print("  r(t+Δt) = 2r(t) - r(t-Δt) + (Δt)²F(t)/m")
    print()
    print("  Properties:")
    print("    ✓ Time-reversible")
    print("    ✓ Symplectic (conserves phase space)")
    print("    ✓ Simple, stable")
    print("    ✗ Velocities computed indirectly")

    print("\nVelocity Verlet (most common):")
    print("-" * 45)
    print("  r(t+Δt) = r(t) + v(t)Δt + ½a(t)(Δt)²")
    print("  v(t+Δt) = v(t) + ½[a(t) + a(t+Δt)]Δt")
    print()
    print("  Algorithm:")
    print("    1. r_new = r + v*dt + 0.5*a*dt²")
    print("    2. Compute F_new, a_new = F_new/m")
    print("    3. v_new = v + 0.5*(a + a_new)*dt")

    print("\nLeapfrog Algorithm:")
    print("-" * 45)
    print("  v(t+Δt/2) = v(t-Δt/2) + a(t)Δt")
    print("  r(t+Δt) = r(t) + v(t+Δt/2)Δt")
    print()
    print("  Positions and velocities at offset times")
    print("  Equivalent to Verlet, different formulation")

    print("\nTime Step Selection:")
    print("-" * 45)
    print("  Rule: Δt < τ/10 where τ = fastest vibration period")
    print()
    print("  Bond vibrations: ~10-30 fs period")
    print("  → Δt = 1-2 fs typical")
    print()
    print("  With constraints (SHAKE): Δt up to 4 fs")
    print("  Heavy hydrogen: Δt up to 4 fs")

integration_algorithms()
```

## 5. Temperature and Pressure Control

```python
def thermostats_barostats():
    """Controlling temperature and pressure in MD."""

    print("Thermostats and Barostats")
    print("=" * 60)

    print("\nWhy Control T and P?")
    print("  MD naturally samples NVE (microcanonical)")
    print("  Experiments often at constant T, P")
    print("  Need thermostats (T) and barostats (P)")

    print("\nTemperature from Kinetic Energy:")
    print("-" * 45)
    print("  ⟨K⟩ = ½ Nf kB T")
    print()
    print("  Nf = degrees of freedom = 3N - constraints")
    print("  Instantaneous T from velocities")

    print("\nBerendsen Thermostat:")
    print("-" * 45)
    print("  Scale velocities toward target T")
    print("  λ = [1 + (Δt/τT)(T₀/T - 1)]^½")
    print("  v_new = λ × v")
    print()
    print("  Simple but not rigorous NVT ensemble")

    print("\nNosé-Hoover Thermostat:")
    print("-" * 45)
    print("  Extended Lagrangian with heat bath")
    print("  Properly samples canonical ensemble")
    print("  Adds friction term to equations of motion")

    print("\nVelocity Rescaling (Bussi):")
    print("-" * 45)
    print("  Stochastic velocity rescaling")
    print("  Correct canonical ensemble")
    print("  Combines Berendsen efficiency + correctness")

    print("\nPressure Control:")
    print("-" * 45)
    print("  Pressure from virial:")
    print("  P = (NkBT + Σrᵢⱼ·Fᵢⱼ/3) / V")
    print()
    print("  Berendsen barostat: scale box")
    print("  Parrinello-Rahman: extended Lagrangian")

    print("\nCommon Ensembles:")
    print("-" * 45)
    ensembles = [
        ("NVE", "Microcanonical", "No thermostat"),
        ("NVT", "Canonical", "Thermostat"),
        ("NPT", "Isothermal-isobaric", "Thermo + baro"),
```

## 6. Periodic Boundary Conditions

```python
def periodic_boundaries():
    """Periodic boundary conditions in MD."""

    print("Periodic Boundary Conditions (PBC)")
    print("=" * 60)

    print("\nThe Problem:")
    print("  Finite system: surface effects dominate")
    print("  Want to simulate bulk behavior")

    print("\nSolution: PBC")
    print("-" * 45)
    print("  Replicate simulation box infinitely")
    print("  Particle leaving one side enters opposite")
    print("  No surfaces, bulk-like environment")

    print("\nMinimum Image Convention:")
    print("-" * 45)
    print("  For particle pair (i, j):")
    print("  Use closest periodic image of j")
    print()
    print("  rᵢⱼ = rⱼ - rᵢ")
    print("  If |rᵢⱼ| > L/2: rᵢⱼ -= L × sign(rᵢⱼ)")

    print("\nCutoff for Interactions:")
    print("-" * 45)
    print("  Short-range: cutoff at r_cut < L/2")
    print("  Long-range (electrostatics): Ewald/PME")
    print()
    print("  Typical: r_cut = 10-12 Å for vdW")

    print("\nBox Shapes:")
    print("-" * 45)
    shapes = [
        ("Cubic", "Simple", "Most common"),
        ("Rectangular", "Different L_x, L_y, L_z", "Membranes"),
        ("Triclinic", "General parallelepiped", "Crystals"),
        ("Octahedral", "Truncated octahedron", "Efficient"),
    ]
    print(f"  {'Shape':<15} {'Description':<25} {'Use'}")
    print("  " + "-" * 50)
    for shape, desc, use in shapes:
        print(f"  {shape:<15} {desc:<25} {use}")

    print("\nExample: Wrapping Coordinates")
    print("-" * 45)
    print("  def wrap(r, L):")
    print("      return r - L * np.floor(r/L + 0.5)")
    print()
    print("  Keeps particle in central box [-L/2, L/2)")

periodic_boundaries()
```

<Warning>
The simulation box must be large enough that the cutoff distance is less than half the smallest box dimension. Otherwise, particles interact with their own periodic images.
</Warning>

## 7. Trajectory Analysis

```python
def trajectory_analysis():
    """Analyzing molecular dynamics trajectories."""

    print("Trajectory Analysis")
    print("=" * 60)

    print("\nStructural Properties:")
    print("-" * 45)
    print("  Radial distribution function g(r):")

    print("    g(r) = ⟨ρ(r)⟩ / ρ_bulk")

    print("\nDynamic Properties:")
    print("-" * 45)
    print("  Mean Square Displacement (MSD):")
    print("    MSD(t) = ⟨|r(t) - r(0)|²⟩")
    print("    Diffusion: D = lim(t→∞) MSD(t)/(6t)")
    print()
    print("  Velocity Autocorrelation:")
    print("    C_vv(t) = ⟨v(0)·v(t)⟩")
    print("    Related to diffusion and spectra")

    print("\nThermodynamic Properties:")
    print("-" * 45)
    print("  Temperature: T = 2⟨K⟩/(3NkB)")
    print("  Pressure: from virial")
    print("  Energy: average and fluctuations")

    print("\nTime Correlation Functions:")
    print("-" * 45)
    print("  General: C(t) = ⟨A(0)B(t)⟩")
    print()
    print("  Examples:")
    print("    • Dipole-dipole → IR spectra")
    print("    • Bond vector → NMR relaxation")
    print("    • Density-density → scattering")
```
