## 1. Plane Wave Basis Sets

<Definition term="Plane Wave Basis">
A complete set of periodic functions exp(i(k+G)·r) used to expand Bloch wavefunctions in periodic DFT calculations, controlled by a kinetic energy cutoff.
</Definition>

```python
import numpy as np

def plane_wave_basis():
    """Plane wave basis for periodic DFT."""

    print("Plane Wave Basis Sets")
    print("=" * 60)

    print("\nBloch Function Expansion:")
    print("-" * 45)
    print("  ψ_nk(r) = Σ_G c_{n,k+G} e^{i(k+G)·r}")
    print()
    print("  G = reciprocal lattice vectors")
    print("  k = crystal momentum (in 1st BZ)")

    print("\nEnergy Cutoff:")
    print("-" * 45)
    print("  Include plane waves with:")
    print("  ℏ²|k+G|²/(2m) < E_cut")
    print()
    print("  Higher E_cut → more plane waves → more accurate")

    # Example: number of plane waves
    print("\nNumber of Plane Waves:")
    print("-" * 45)
    def n_pw(Ecut_eV, volume_A3):
        """Estimate number of plane waves."""
        # N_PW ≈ (V/4π²) × (2mE_cut/ℏ²)^(3/2)
        Ecut = Ecut_eV * 1.6e-19  # J
        V = volume_A3 * 1e-30  # m³
        m = 9.11e-31
        hbar = 1.055e-34
        k_max = np.sqrt(2 * m * Ecut) / hbar
        N = V * k_max**3 / (6 * np.pi**2)
        return int(N)

    for Ecut in [200, 400, 600, 800]:
        N = n_pw(Ecut, 100)  # 100 Å³ cell
        print(f"  E_cut = {Ecut} eV → ~{N} plane waves")

    print("\nAdvantages:")
    print("-" * 45)
    print("  • Systematic convergence")
    print("  • No basis set superposition error")
    print("  • Efficient FFT algorithms")
    print("  • Natural for periodic systems")

    print("\nDisadvantages:")
    print("-" * 45)
    print("  • Many functions needed for core electrons")
    print("  • → Requires pseudopotentials")
    print("  • Not efficient for vacuum (slabs, molecules)")

plane_wave_basis()
```

## 2. Pseudopotentials

```python
def pseudopotentials():

    print("\nTypes of Pseudopotentials:")

    types = [

    ]
    for pp_type, feature, advantage in types:
        print(f"  {pp_type}:")
        print(f"    Feature: {feature}")
        print(f"    Advantage: {advantage}")
        print()

    print("\nKey Properties:")
    print("-" * 45)
    print("  1. Same scattering as all-electron")
    print("  2. Smooth inside cutoff radius r_c")
    print("  3. Correct valence eigenvalues")
    print("  4. Transferable between environments")

    print("\nPseudopotential Libraries:")
    print("-" * 45)
    libraries = [
        ("SSSP", "Standard Solid State", "QE recommended"),
        ("GBRV", "Garrity-Bennett-Rabe-Vanderbilt", "Accurate"),
        ("PseudoDojo", "Careful validation", "Well tested"),
        ("SG15", "Schlipf-Gygi", "NC optimized"),
    ]
    print(f"  {'Library':<12} {'Full Name':<28} {'Notes'}")
    print("  " + "-" * 50)

pseudopotentials()
```

<Tip>
Always use tested pseudopotential libraries (SSSP, PseudoDojo) for production calculations. Untested pseudopotentials can give qualitatively wrong results.
</Tip>

## 5. The Band Gap Problem

```python
def band_gap_problem():
    """The band gap problem and solutions."""

    print("The DFT Band Gap Problem")
    print("=" * 60)

    print("\nThe Problem:")
    print("-" * 45)
    print("  Kohn-Sham eigenvalues ≠ quasiparticle energies")
    print("  DFT systematically underestimates band gaps")
    print()
    gaps = [
        ("Si", 1.17, 0.5, "LDA"),
        ("GaAs", 1.42, 0.3, "LDA"),
        ("ZnO", 3.4, 0.8, "LDA"),
        ("MgO", 7.8, 4.5, "LDA"),
    ]
    print(f"  {'Material':<10} {'Exp (eV)':<12} {'DFT (eV)':<12} {'Functional'}")
    print("  " + "-" * 45)
    for mat, exp, dft, func in gaps:
        print(f"  {mat:<10} {exp:<12.1f} {dft:<12.1f} {func}")

    print("\nPhysical Reason:")
    print("-" * 45)
    print("  DFT: Ground state theory")
    print("  Band gap: Excited state property")
    print()
    print("  Missing: Derivative discontinuity Δxc")
    print("  E_gap = ε_CB - ε_VB + Δxc")

    print("\nSolutions:")
    print("-" * 45)
    solutions = [
        ("Hybrid functionals", "HSE, PBE0", "Mix exact exchange"),
        ("GW approximation", "G₀W₀, GW", "Quasiparticle theory"),
        ("Meta-GGA", "SCAN, mBJ", "Improved functionals"),
        ("DFT+U", "LDA+U", "For localized electrons"),
        ("Scissors operator", "Rigid shift", "Empirical correction"),
    ]
    for method, example, approach in solutions:
        print(f"  {method}:")
        print(f"    Example: {example}")
        print(f"    Approach: {approach}")

band_gap_problem()
```

<Formula name="HSE Hybrid Functional">
$$E_{xc}^{HSE} = \frac{1}{4}E_x^{HF,SR}(\omega) + \frac{3}{4}E_x^{PBE,SR}(\omega) + E_x^{PBE,LR}(\omega) + E_c^{PBE}$$

Where SR/LR denote short/long range parts separated by screening parameter $\omega$ (typically 0.2 Å⁻¹).
</Formula>
