## 1. Overview of Slater-Condon Rules

<Definition term="Slater-Condon Rules">
A set of formulas for computing matrix elements of one- and two-electron operators between Slater determinants, based on how many spin-orbitals differ between the determinants.
</Definition>

```python
def slater_condon_overview():
    """Overview of Slater-Condon rules."""

    print("Slater-Condon Rules Overview")
    print("=" * 60)

    print("\nPurpose:")
    print("  Compute ⟨Φ|Ô|Φ'⟩ for Slater determinants")
    print("  without expanding the full antisymmetrized integral")

    print("  between |Φ⟩ and |Φ'⟩")

    print("\nCases:")
    print("-" * 45)
    cases = [
        (0, "Same determinant", "Diagonal element"),
        (1, "Single excitation", "One orbital different"),

        ("≥3", "Triple+ excitation", "Matrix element = 0"),
    ]

    print(f"{'Diff':<6} {'Name':<20} {'Description':<25}")
    print("-" * 55)
    for diff, name, desc in cases:
        print(f"{str(diff):<6} {name:<20} {desc:<25}")

    print("\nWhy Maximum of 2 Differences?")
    print("  • One-electron operator: connects states differing by 1")
    print("  • Two-electron operator: connects states differing by 2")
    print("  • No operator connects states differing by ≥3")
```

## 2. Rule 0: Diagonal Elements (Same Determinant)

<Formula>
$$\langle\Phi|\hat{H}|\Phi\rangle = \sum_{i \in \text{occ}} h_{ii} + \frac{1}{2}\sum_{i,j \in \text{occ}} \left[(ii|jj) - (ij|ji)\right]$$
</Formula>

```python
def rule_zero():
    """Slater-Condon rule for diagonal elements."""

    print("Rule 0: Same Determinant (Diagonal)")
    print("=" * 60)

    print("\nOne-Electron Part:")
    print("  ⟨Φ|Ĥ₁|Φ⟩ = Σᵢ∈occ hᵢᵢ")
    print()
    print("  Sum of one-electron integrals over occupied orbitals")

    print("\nTwo-Electron Part:")
    print("  ⟨Φ|Ĥ₂|Φ⟩ = ½Σᵢ,ⱼ∈occ [(ii|jj) - (ij|ji)]")
    print()
    print("  = ½Σᵢ,ⱼ∈occ [Jᵢⱼ - Kᵢⱼ]")
    print()
    print("  Jᵢⱼ = (ii|jj) = Coulomb integral")
    print("  Kᵢⱼ = (ij|ji) = Exchange integral")

    print("\nAlternative Form:")
    print("  ⟨Φ|Ĥ₂|Φ⟩ = Σᵢ<ⱼ [(ii|jj) - (ij|ji)]")
    print()
    print("  (No factor of ½ with restricted sum)")

    print("\nTotal HF Energy:")
    print("-" * 45)
    print("  E = Σᵢ hᵢᵢ + ½Σᵢⱼ (Jᵢⱼ - Kᵢⱼ) + V_nn")

    print("\nPhysical Interpretation:")
    print("  • hᵢᵢ: energy of electron in orbital i")
    print("  • Jᵢⱼ: Coulomb repulsion between i and j")
    print("  • Kᵢⱼ: Exchange stabilization (quantum effect)")

rule_zero()
```

## 3. Rule 1: Single Excitation

<Formula>
$$\langle\Phi|\hat{H}|\Phi_a^i\rangle = h_{ai} + \sum_{j \in \text{occ}} \left[(ai|jj) - (aj|ji)\right]$$
</Formula>

```python
def rule_one():
    """Slater-Condon rule for single excitations."""

    print("Rule 1: Single Excitation (One Orbital Different)")
    print("=" * 60)

    print("\nNotation:")
    print("  |Φₐⁱ⟩ = excitation from occupied i to virtual a")
    print("  |Φₐⁱ⟩ = c†ₐcᵢ|Φ⟩")

    print("\nMatrix Element:")
    print("-" * 45)
    print("  ⟨Φ|Ĥ|Φₐⁱ⟩ = hₐᵢ + Σⱼ∈occ [(ai|jj) - (aj|ji)]")
    print()
    print("  = hₐᵢ + Σⱼ∈occ [Jₐᵢⱼ - Kₐᵢⱼ]")

    print("\nThis is the Fock Matrix Element:")
    print("  ⟨Φ|Ĥ|Φₐⁱ⟩ = Fₐᵢ")

    print("\nBrillouin's Theorem:")
    print("-" * 45)
    print("  For optimized HF orbitals: Fₐᵢ = 0")
    print()
    print("  The HF ground state doesn't couple to singles!")
    print("  (This is a variational condition)")

    print("\nImplication for CI:")
    print("  • CIS built on HF: ⟨HF|Ĥ|singles⟩ = 0")
    print("  • Singles don't lower ground state energy")
    print("  • But they do affect excited states")

rule_one()
```

## 4. Rule 2: Double Excitation

<Formula>
$$\langle\Phi|\hat{H}|\Phi_{ab}^{ij}\rangle = (ai|bj) - (aj|bi) = \langle ab||ij\rangle$$
</Formula>

```python
def rule_two():
    """Slater-Condon rule for double excitations."""

    print("Rule 2: Double Excitation (Two Orbitals Different)")
    print("=" * 60)

    print("\nNotation:")
    print("  |Φₐᵦⁱʲ⟩ = double excitation i→a, j→b")
    print("  |Φₐᵦⁱʲ⟩ = c†ₐc†ᵦcⱼcᵢ|Φ⟩")

    print("\nMatrix Element:")
    print("-" * 45)
    print("  ⟨Φ|Ĥ|Φₐᵦⁱʲ⟩ = (ai|bj) - (aj|bi)")
    print()
    print("  In physicist's notation:")
    print("  ⟨Φ|Ĥ|Φₐᵦⁱʲ⟩ = ⟨ab||ij⟩  (antisymmetrized)")

    print("\nNo One-Electron Contribution:")
    print("  The one-electron operator can't connect")
    print("  states that differ by two orbitals!")

    print("\nSign Conventions:")
    print("-" * 45)
    print("  Careful with orbital ordering!")
    print("  Standard: i < j for occupied, a < b for virtual")
    print()
    print("  Additional sign if |Φₐᵦⁱʲ⟩ requires reordering")

    print("\nImplication for CI:")
    print("  • Doubles directly couple to HF ground state")
    print("  • Major contribution to correlation energy")
    print("  • MP2, CCSD focus on doubles")

rule_two()
```

<Tip>
Brillouin's theorem says singles don't mix with the HF ground state. But doubles do! This is why double excitations are crucial for capturing electron correlation.
</Tip>

## 8. Implementation

```python
import numpy as np

def slater_condon_implementation():
    """Implementing Slater-Condon rules."""

    print("Slater-Condon Implementation")
    print("=" * 60)

    # Example functions
    def count_differences(occ1, occ2):
        """Count orbital differences between determinants."""
        set1, set2 = set(occ1), set(occ2)
        unique1 = set1 - set2
        unique2 = set2 - set1
        return list(unique1), list(unique2)

    def diagonal_element(occ, h1, h2):
        """Compute diagonal matrix element (Rule 0)."""
        E = 0.0
        # One-electron
        for i in occ:
            E += h1[i, i]
        # Two-electron
        for i in occ:
            for j in occ:
                if i != j:
                    E += 0.5 * (h2[i, i, j, j] - h2[i, j, j, i])
        return E

    def single_excitation_element(occ, i, a, h1, h2):
        """Compute single excitation element (Rule 1)."""
        F = h1[a, i]
        for j in occ:
            if j != i:
                F += h2[a, i, j, j] - h2[a, j, j, i]
        return F

    def double_excitation_element(i, j, a, b, h2):
        """Compute double excitation element (Rule 2)."""
        return h2[a, i, b, j] - h2[a, j, b, i]

    print("Example: 4 orbitals, 2 electrons")
    print("-" * 45)

    # Simple integrals
    n_orb = 4
    h1 = np.diag([-1.0, -0.5, 0.5, 1.0])
    h2 = np.zeros((n_orb, n_orb, n_orb, n_orb))
    h2[0, 0, 1, 1] = 0.5
    h2[1, 1, 0, 0] = 0.5

    occ = [0, 1]  # Ground state

    E_diag = diagonal_element(occ, h1, h2)
    print(f"Diagonal ⟨Φ|H|Φ⟩ = {E_diag:.4f}")

    F_02 = single_excitation_element(occ, 0, 2, h1, h2)
    print(f"Single ⟨Φ|H|Φ₂⁰⟩ = {F_02:.4f}")

    V_0123 = double_excitation_element(0, 1, 2, 3, h2)
    print(f"Double ⟨Φ|H|Φ₂₃⁰¹⟩ = {V_0123:.4f}")

slater_condon_implementation()
```

## Summary

In this lesson, you learned:

1. **Slater-Condon rules** give matrix elements based on orbital differences
2. **Rule 0** (diagonal): sum of one-electron + Coulomb - Exchange
3. **Rule 1** (single): gives Fock matrix element Fₐᵢ
4. **Rule 2** (double): gives antisymmetrized integral ⟨ab||ij⟩
5. **Rule 3+**: matrix element is exactly zero
6. **Brillouin's theorem**: HF doesn't couple to singles
7. **Spin integration** affects formulas for closed/open shells
