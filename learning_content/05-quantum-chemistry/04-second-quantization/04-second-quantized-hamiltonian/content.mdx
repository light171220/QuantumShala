## 1. The Electronic Hamiltonian

<Definition term="Electronic Hamiltonian">
The Hamiltonian describing electrons in a fixed nuclear framework (Born-Oppenheimer approximation), including kinetic energy, nuclear attraction, and electron-electron repulsion.
</Definition>

```python
def electronic_hamiltonian():
    """The electronic Hamiltonian in first quantization."""

    print("Electronic Hamiltonian (First Quantization)")
    print("=" * 60)

    print("\nFull Form:")
    print("  Ĥ = T̂ + V̂_ne + V̂_ee + V_nn")
    print()
    print("  T̂ = Σᵢ (-½∇ᵢ²)           Kinetic energy")
    print("  V̂_ne = Σᵢ Σₐ (-Zₐ/rᵢₐ)   Nuclear-electron attraction")
    print("  V̂_ee = Σᵢ<ⱼ (1/rᵢⱼ)       Electron-electron repulsion")
    print("  V_nn = Σₐ<ᵦ (ZₐZᵦ/Rₐᵦ)    Nuclear-nuclear repulsion")

    print("\nGrouping Terms:")
    print("-" * 45)
    print("  Ĥ = Σᵢ ĥ(i) + Σᵢ<ⱼ ĝ(i,j) + V_nn")
    print()
    print("  ĥ(i) = -½∇ᵢ² - Σₐ Zₐ/rᵢₐ  (one-electron)")
    print("  ĝ(i,j) = 1/rᵢⱼ              (two-electron)")

    print("\nChallenge with First Quantization:")
    print("  • Must antisymmetrize wavefunction")
    print("  • Integrals over all electron coordinates")
    print("  • Complicated for many electrons")

electronic_hamiltonian()
```

## 2. One-Electron Integrals

<Formula>
$$h_{pq} = \int \phi_p^*(\mathbf{r}) \left[-\frac{1}{2}\nabla^2 - \sum_A \frac{Z_A}{|\mathbf{r} - \mathbf{R}_A|}\right] \phi_q(\mathbf{r}) d\mathbf{r}$$
</Formula>

```python
def one_electron_integrals():
    """One-electron integrals in second quantization."""

    print("One-Electron Integrals")
    print("=" * 60)

    print("\nDefinition:")
    print("  hₚₑ = ⟨p|ĥ|q⟩ = ∫ φₚ*(r) ĥ φq(r) dr")
    print()
    print("  where ĥ = -½∇² - Σₐ Zₐ/rₐ")

    print("\nComponents:")
    print("-" * 45)
    print("  hₚₑ = Tₚₑ + Vₚₑ")
    print()
    print("  Tₚₑ = ⟨p|-½∇²|q⟩     (kinetic integral)")
    print("  Vₚₑ = ⟨p|-Σₐ Zₐ/rₐ|q⟩ (nuclear attraction)")

    print("\nProperties:")
    print("  • Hermitian: hₚₑ = hqp*")
    print("  • Real for real orbitals: hₚₑ = hqp")
    print("  • Matrix dimension: M × M (M = # of orbitals)")

    print("\nPhysical Meaning:")
    print("  hₚₚ = orbital energy in nuclear field (no e-e)")
    print("  hₚₑ = coupling between orbitals p and q")

    print("\nComputation:")
    print("  • Analytic formulas for Gaussian basis")
    print("  • Standard in all quantum chemistry codes")
    print("  • O(M²) integrals to compute")

one_electron_integrals()
```

## 3. Two-Electron Integrals

<Formula>
$$(pq|rs) = \int \int \phi_p^*(\mathbf{r}_1) \phi_q(\mathbf{r}_1) \frac{1}{|\mathbf{r}_1 - \mathbf{r}_2|} \phi_r^*(\mathbf{r}_2) \phi_s(\mathbf{r}_2) d\mathbf{r}_1 d\mathbf{r}_2$$
</Formula>

```python
def two_electron_integrals():
    """Two-electron integrals in second quantization."""

    print("Two-Electron Integrals")
    print("=" * 60)

    print("\nChemist's Notation (Mulliken):")
    print("  (pq|rs) = ∫∫ φₚ*(r₁)φq(r₁) (1/r₁₂) φᵣ*(r₂)φₛ(r₂) dr₁dr₂")
    print()
    print("  Indices: (bra1, ket1 | bra2, ket2)")

    print("\nPhysicist's Notation (Dirac):")
    print("  ⟨pq|rs⟩ = ∫∫ φₚ*(r₁)φq*(r₂) (1/r₁₂) φᵣ(r₁)φₛ(r₂) dr₁dr₂")
    print()
    print("  ⟨pq|rs⟩ = (pr|qs)  (different ordering!)")

    print("\nSymmetries (real orbitals):")
    print("-" * 45)
    print("  (pq|rs) = (qp|rs) = (pq|sr) = (qp|sr)")
    print("  (pq|rs) = (rs|pq)")
    print()
    print("  8-fold symmetry → O(M⁴/8) unique integrals")

    print("\nSpecial Cases:")
    print("  (pp|pp) = Coulomb self-interaction")
    print("  (pq|pq) = Coulomb integral Jpq")
    print("  (pq|qp) = Exchange integral Kpq")

    print("\nComputational Cost:")
    print("  • O(M⁴) integrals naively")
    print("  • Dominant cost for large systems")
    print("  • Integral screening reduces this")

two_electron_integrals()
```

<Warning>
Be careful with notation conventions! Chemist's notation (pq|rs) and physicist's notation ⟨pq|rs⟩ have different index orderings. Always check which convention a paper or code uses.
</Warning>

## 4. Second-Quantized Form

<Definition term="Second-Quantized Hamiltonian">
The molecular Hamiltonian expressed using creation and annihilation operators, naturally incorporating antisymmetry through the fermionic algebra.
</Definition>

```python
def second_quantized_hamiltonian():
    """The Hamiltonian in second quantization."""

    print("Second-Quantized Hamiltonian")
    print("=" * 60)

    print("\nGeneral Form:")
    print("  Ĥ = Σₚₑ hₚₑ c†ₚcq + ½Σₚₑᵣₛ (pq|rs) c†ₚc†ᵣcₛcq + V_nn")

    print("\nOne-Electron Part:")
    print("-" * 45)
    print("  Ĥ₁ = Σₚₑ hₚₑ c†ₚcq")
    print()
    print("  • Sums over all orbital pairs")
    print("  • hₚₑ = one-electron integral")
    print("  • c†ₚcq = transfer electron from q to p")

    print("\nTwo-Electron Part:")
    print("-" * 45)
    print("  Ĥ₂ = ½Σₚₑᵣₛ (pq|rs) c†ₚc†ᵣcₛcq")
    print()
    print("  • Factor of ½ avoids double counting")
    print("  • Note operator ordering: c†c†cc")
    print("  • (pq|rs) in chemist's notation")

    print("\nAlternative Form (physicist's notation):")
    print("  Ĥ₂ = ½Σₚₑᵣₛ ⟨pq||rs⟩ c†ₚc†qcₛcᵣ")
    print()
    print("  ⟨pq||rs⟩ = ⟨pq|rs⟩ - ⟨pq|sr⟩ (antisymmetrized)")

    print("\nTotal Hamiltonian:")
    print("  Ĥ = Ĥ₁ + Ĥ₂ + V_nn")

second_quantized_hamiltonian()
```

## 5. Spin Orbitals vs Spatial Orbitals

```python
def spin_orbitals():
    """Spin orbital formulation."""

    print("Spin Orbitals vs Spatial Orbitals")
    print("=" * 60)

    print("\nSpin Orbital Basis:")
    print("-" * 45)
    print("  χₚ(x) = φₚ(r) σ(ω)")
    print()
    print("  x = (r, ω) = spatial + spin coordinate")
    print("  σ = α (spin up) or β (spin down)")
    print()
    print("  Hamiltonian indices run over spin orbitals")

    print("\nSpatial Orbital Basis:")
    print("-" * 45)
    print("  Work with φₚ(r) only")
    print("  Explicitly track spin: p↑, p↓")
    print()
    print("  Ĥ = Σₚₑ hₚₑ (c†ₚ↑cq↑ + c†ₚ↓cq↓)")
    print("    + ½Σₚₑᵣₛ (pq|rs) Σσσ' c†ₚσc†ᵣσ'cₛσ'cqσ")

    print("\nRestricted vs Unrestricted:")
    print("-" * 45)
    print("  Restricted: same spatial orbital for α and β")
    print("    φₚ↑(r) = φₚ↓(r)")
    print("    Simpler, good for closed shells")
    print()
    print("  Unrestricted: different orbitals for α and β")
    print("    φₚ↑(r) ≠ φₚ↓(r)")
    print("    More flexible, needed for open shells")

spin_orbitals()
```

## 6. Integral Transformations

```python
import numpy as np

def integral_transformation():
    """Transform integrals between bases."""

    print("Integral Transformations")
    print("=" * 60)

    print("\nAO to MO Transformation:")
    print("-" * 45)
    print("  Given: AO integrals hμν, (μν|λσ)")
    print("  Want: MO integrals hpq, (pq|rs)")
    print()
    print("  MOs: φₚ = Σμ Cμₚ χμ")

    print("\nOne-Electron Transformation:")
    print("  hₚₑ = Σμν Cμₚ* hμν Cνq")
    print("  h_MO = C† h_AO C")
    print()
    print("  Cost: O(M³)")

    print("\nTwo-Electron Transformation:")
    print("  (pq|rs) = Σμνλσ Cμₚ* Cνq Cλᵣ* Cσₛ (μν|λσ)")
    print()
    print("  Four-index transformation!")
    print("  Cost: O(M⁵) with efficient algorithm")

    print("\nEfficient Algorithm (Yoshimine):")
    print("-" * 45)
    print("  Transform one index at a time:")
    print("  1. (μν|λσ) → (pν|λσ)  O(M⁵)")
    print("  2. (pν|λσ) → (pq|λσ)  O(M⁴)")
    print("  3. (pq|λσ) → (pq|rσ)  O(M⁴)")
    print("  4. (pq|rσ) → (pq|rs)  O(M⁴)")
    print()
    print("  Total: O(M⁵), limited by step 1")

integral_transformation()
```

## 7. Hamiltonian Matrix Elements

```python
def hamiltonian_matrix_elements():
    """Computing Hamiltonian matrix elements."""

    print("Hamiltonian Matrix Elements")
    print("=" * 60)

    print("\nBetween Slater Determinants:")
    print("  ⟨Φ|Ĥ|Φ'⟩ = ?")
    print()
    print("  Use second quantization + Wick's theorem")
    print("  Or: Slater-Condon rules (next lesson)")

    print("\nDiagonal Elements (same determinant):")
    print("-" * 45)
    print("  ⟨Φ|Ĥ|Φ⟩ = Σᵢ∈occ hᵢᵢ + ½Σᵢⱼ∈occ ((ii|jj) - (ij|ji))")
    print()
    print("  = Σᵢ hᵢᵢ + ½Σᵢⱼ (Jᵢⱼ - Kᵢⱼ)")

    print("\nSingle Excitation (one orbital different):")
    print("-" * 45)
    print("  |Φₐⁱ⟩ = c†ₐcᵢ|Φ⟩  (excite i → a)")
    print()
    print("  ⟨Φ|Ĥ|Φₐⁱ⟩ = Fₐᵢ")
    print("  (Fock matrix element)")

    print("\nDouble Excitation (two orbitals different):")
    print("-" * 45)
    print("  |Φₐᵦⁱʲ⟩ = c†ₐc†ᵦcⱼcᵢ|Φ⟩")
    print()
    print("  ⟨Φ|Ĥ|Φₐᵦⁱʲ⟩ = (ai|bj) - (aj|bi)")
    print()
    print("  = ⟨ab||ij⟩ (antisymmetrized integral)")

    print("\nHigher Excitations:")
    print("  ⟨Φ|Ĥ|Φ⟩ = 0 for triple and higher")
    print("  (Two-body operator can't connect them)")

hamiltonian_matrix_elements()
```

## 8. Implementation Example

```python
import numpy as np

def hamiltonian_construction():
    """Example of constructing molecular Hamiltonian."""

    print("Hamiltonian Construction Example")
    print("=" * 60)

    # Minimal example: H2 in minimal basis (2 orbitals)
    print("\nExample: H₂ in minimal basis")
    print("-" * 45)

    # One-electron integrals (Hartree)
    h = np.array([[-1.25, -0.48],
                  [-0.48, -1.25]])

    print("One-electron integrals h:")
    print(h)

    # Two-electron integrals (pq|rs) - simplified
    # Only showing unique non-zero integrals
    g = np.zeros((2, 2, 2, 2))
    g[0,0,0,0] = 0.67  # (00|00)
    g[1,1,1,1] = 0.67  # (11|11)
    g[0,0,1,1] = 0.66  # (00|11)
    g[1,1,0,0] = 0.66  # (11|00)
    g[0,1,0,1] = 0.18  # (01|01)
    g[1,0,1,0] = 0.18  # (10|10)
    g[0,1,1,0] = 0.18  # (01|10)
    g[1,0,0,1] = 0.18  # (10|01)

    print("\nKey two-electron integrals:")
    print(f"  (00|00) = {g[0,0,0,0]:.2f}")
    print(f"  (00|11) = {g[0,0,1,1]:.2f}")
    print(f"  (01|01) = {g[0,1,0,1]:.2f}")

    # HF ground state energy
    print("\nHF ground state (both in orbital 0):")
    E_1e = 2 * h[0, 0]
    E_2e = g[0,0,0,0]  # Both electrons in same orbital
    print(f"  E_1e = 2 × h₀₀ = {E_1e:.3f}")
    print(f"  E_2e = (00|00) = {E_2e:.3f}")
    print(f"  E_HF = {E_1e + E_2e:.3f} (+ nuclear repulsion)")

hamiltonian_construction()
```

<Tip>
Quantum chemistry packages (PySCF, Psi4) provide functions to compute and transform integrals. Focus on understanding the structure; let the software handle the numerical work.
</Tip>

## Summary

In this lesson, you learned:

1. **Electronic Hamiltonian** = kinetic + nuclear + electron-electron
2. **One-electron integrals** hₚₑ are O(M²) quantities
3. **Two-electron integrals** (pq|rs) are O(M⁴) quantities
4. **Second-quantized form** Ĥ = Σhₚₑc†c + ½Σ(pq|rs)c†c†cc
5. **Integral transformation** AO→MO costs O(M⁵)
6. **Matrix elements** follow from Wick's theorem
7. **Slater-Condon rules** (next lesson) give explicit formulas
