## 1. Rotation Gates Overview

Rotation gates perform rotations around the X, Y, or Z axis of the Bloch sphere by an arbitrary angle θ. They generalize the Pauli gates (which are π rotations).

<Definition term="Rotation Gate">
A rotation gate R_n(θ) rotates the qubit state around axis n by angle θ on the Bloch sphere. The matrix form is R_n(θ) = exp(-iθn·σ/2) where σ = (X, Y, Z) are the Pauli matrices.
</Definition>

<Formula>
$$R_x(\theta) = e^{-i\theta X/2}, \quad R_y(\theta) = e^{-i\theta Y/2}, \quad R_z(\theta) = e^{-i\theta Z/2}$$
</Formula>

## 2. The Rx Gate (X-axis Rotation)

<Formula>
$$R_x(\theta) = \begin{pmatrix} \cos(\theta/2) & -i\sin(\theta/2) \\ -i\sin(\theta/2) & \cos(\theta/2) \end{pmatrix}$$
</Formula>

```python
import numpy as np

def Rx(theta):
    """Rotation around X-axis by angle theta."""
    c = np.cos(theta / 2)
    s = np.sin(theta / 2)
    return np.array([
        [c, -1j * s],
        [-1j * s, c]
    ])

# Verify: Rx(π) = -iX (up to global phase)
X = np.array([[0, 1], [1, 0]])
Rx_pi = Rx(np.pi)

print("Rx(π):")
print(np.round(Rx_pi, 4))
print(f"\nRx(π) ≈ -iX: {np.allclose(Rx_pi, -1j * X)}")

# Action on |0⟩
ket_0 = np.array([1, 0])
for theta_deg in [0, 90, 180, 270]:
    theta = np.radians(theta_deg)
    result = Rx(theta) @ ket_0
    print(f"Rx({theta_deg}°)|0⟩ = {np.round(result, 3)}")
```

<Tip>
Rx(π) is equivalent to the Pauli X gate up to a global phase of -i. In general, Rx(π) = -iX, Ry(π) = -iY, Rz(π) = -iZ.
</Tip>

## 3. The Ry Gate (Y-axis Rotation)

<Formula>
$$R_y(\theta) = \begin{pmatrix} \cos(\theta/2) & -\sin(\theta/2) \\ \sin(\theta/2) & \cos(\theta/2) \end{pmatrix}$$
</Formula>

```python
import numpy as np

def Ry(theta):
    """Rotation around Y-axis by angle theta."""
    c = np.cos(theta / 2)
    s = np.sin(theta / 2)
    return np.array([
        [c, -s],
        [s, c]
    ])

# Ry is special: it has real entries!
print("Ry(π/2):")
print(np.round(Ry(np.pi/2), 4))

# Ry(π/2) rotates |0⟩ to |+⟩
ket_0 = np.array([1, 0])
result = Ry(np.pi/2) @ ket_0
print(f"\nRy(π/2)|0⟩ = {np.round(result, 4)}")
print("This is (|0⟩ + |1⟩)/√2 = |+⟩")

# Ry can create any state on the Bloch sphere's XZ plane
print("\nRy creates states on XZ plane (y=0):")
for theta_deg in [0, 45, 90, 135, 180]:
    theta = np.radians(theta_deg)
    state = Ry(theta) @ ket_0
    # Bloch coordinates
    z = np.abs(state[0])**2 - np.abs(state[1])**2
    x = 2 * np.real(state[0] * state[1].conj())
    print(f"  Ry({theta_deg:>3}°)|0⟩ → (x={x:>6.3f}, z={z:>6.3f})")
```

## 4. The Rz Gate (Z-axis Rotation)

<Formula>
$$R_z(\theta) = \begin{pmatrix} e^{-i\theta/2} & 0 \\ 0 & e^{i\theta/2} \end{pmatrix}$$
</Formula>

```python
import numpy as np

def Rz(theta):
    """Rotation around Z-axis by angle theta."""
    return np.array([
        [np.exp(-1j * theta / 2), 0],
        [0, np.exp(1j * theta / 2)]
    ])

# Compare with phase gate P(θ)
def P(theta):
    return np.array([[1, 0], [0, np.exp(1j * theta)]])

theta = np.pi / 2
print(f"Rz(π/2):\n{np.round(Rz(theta), 4)}")
print(f"\nP(π/2):\n{np.round(P(theta), 4)}")
print(f"\nRz(θ) = e^(-iθ/2) · P(θ): {np.allclose(Rz(theta), np.exp(-1j*theta/2) * P(theta))}")
```

## 5. Relationship to Pauli Gates

The Pauli gates are π rotations:

<Formula>
$$R_x(\pi) = -iX, \quad R_y(\pi) = -iY, \quad R_z(\pi) = -iZ$$
</Formula>

```python
import numpy as np

def Rx(theta):
    c, s = np.cos(theta/2), np.sin(theta/2)
    return np.array([[c, -1j*s], [-1j*s, c]])

def Ry(theta):
    c, s = np.cos(theta/2), np.sin(theta/2)
    return np.array([[c, -s], [s, c]])

def Rz(theta):
    return np.array([[np.exp(-1j*theta/2), 0], [0, np.exp(1j*theta/2)]])

X = np.array([[0, 1], [1, 0]])
Y = np.array([[0, -1j], [1j, 0]])
Z = np.array([[1, 0], [0, -1]])

print("Rotation-Pauli Relationships (at θ = π):")
print(f"Rx(π) = -iX: {np.allclose(Rx(np.pi), -1j * X)}")
print(f"Ry(π) = -iY: {np.allclose(Ry(np.pi), -1j * Y)}")
print(f"Rz(π) = -iZ: {np.allclose(Rz(np.pi), -1j * Z)}")
```

## 6. Bloch Sphere Visualization

Each rotation gate rotates the Bloch sphere around its respective axis:

```python
import numpy as np

def state_to_bloch(state):
    """Convert quantum state to Bloch coordinates."""
    state = state / np.linalg.norm(state)
    X = np.array([[0, 1], [1, 0]])
    Y = np.array([[0, -1j], [1j, 0]])
    Z = np.array([[1, 0], [0, -1]])
    
    x = np.real(state.conj() @ X @ state)
    y = np.real(state.conj() @ Y @ state)
    z = np.real(state.conj() @ Z @ state)
    return np.array([x, y, z])

def Rx(theta):
    c, s = np.cos(theta/2), np.sin(theta/2)
    return np.array([[c, -1j*s], [-1j*s, c]])

def Ry(theta):
    c, s = np.cos(theta/2), np.sin(theta/2)
    return np.array([[c, -s], [s, c]])

def Rz(theta):
    return np.array([[np.exp(-1j*theta/2), 0], [0, np.exp(1j*theta/2)]])

# Start with |+⟩ state
plus = np.array([1, 1]) / np.sqrt(2)

print("Rotating |+⟩ state around different axes:")
print(f"{'Gate':<12} {'x':>8} {'y':>8} {'z':>8}")
print("-" * 40)

initial = state_to_bloch(plus)
print(f"{'Initial':<12} {initial[0]:>8.3f} {initial[1]:>8.3f} {initial[2]:>8.3f}")

for theta_deg in [90]:
    theta = np.radians(theta_deg)
    
    for name, gate in [('Rx(90°)', Rx), ('Ry(90°)', Ry), ('Rz(90°)', Rz)]:
        rotated = gate(theta) @ plus
        coords = state_to_bloch(rotated)
        print(f"{name:<12} {coords[0]:>8.3f} {coords[1]:>8.3f} {coords[2]:>8.3f}")
```

## 7. The ZYZ Decomposition

Any single-qubit unitary can be decomposed into rotations:

<Definition term="ZYZ Decomposition">
Any single-qubit unitary U can be written as U = e^(iα) Rz(β) Ry(γ) Rz(δ) for some angles α, β, γ, δ.
</Definition>

<Formula>
$$U = e^{i\alpha} R_z(\beta) R_y(\gamma) R_z(\delta)$$
</Formula>

```python
import numpy as np

def Ry(theta):
    c, s = np.cos(theta/2), np.sin(theta/2)
    return np.array([[c, -s], [s, c]])

def Rz(theta):
    return np.array([[np.exp(-1j*theta/2), 0], [0, np.exp(1j*theta/2)]])

def zyz_decompose(U):
    """
    Decompose U = e^(iα) Rz(β) Ry(γ) Rz(δ)
    Returns (alpha, beta, gamma, delta)
    """
    # Extract global phase
    det = np.linalg.det(U)
    alpha = np.angle(det) / 2
    
    # Remove global phase
    V = U * np.exp(-1j * alpha)
    
    # Now V is in SU(2), use standard decomposition
    # This is a simplified version
    gamma = 2 * np.arccos(np.clip(np.abs(V[0, 0]), 0, 1))
    
    if np.abs(np.sin(gamma/2)) < 1e-10:
        beta = np.angle(V[0, 0]) + np.angle(V[1, 1])
        delta = 0
    else:
        beta = np.angle(-V[0, 1]) + np.angle(V[1, 0])
        delta = np.angle(V[0, 1]) + np.angle(V[1, 0])
    
    return alpha, beta, gamma, delta

def reconstruct_from_zyz(alpha, beta, gamma, delta):
    """Reconstruct U from ZYZ decomposition."""
    return np.exp(1j * alpha) * Rz(beta) @ Ry(gamma) @ Rz(delta)

# Test with Hadamard
H = np.array([[1, 1], [1, -1]]) / np.sqrt(2)

alpha, beta, gamma, delta = zyz_decompose(H)
H_reconstructed = reconstruct_from_zyz(alpha, beta, gamma, delta)

print("ZYZ Decomposition of Hadamard:")
print(f"α = {np.degrees(alpha):.1f}°")
print(f"β = {np.degrees(beta):.1f}°")
print(f"γ = {np.degrees(gamma):.1f}°")
print(f"δ = {np.degrees(delta):.1f}°")
print(f"\nReconstruction matches: {np.allclose(H, H_reconstructed)}")
```

## 8. Creating Arbitrary States

Using Ry and Rz, we can create any single-qubit state:

<Formula>
$$|\psi\rangle = R_z(\phi) R_y(\theta) |0\rangle = \cos(\theta/2)|0\rangle + e^{i\phi}\sin(\theta/2)|1\rangle$$
</Formula>

```python
import numpy as np

def Ry(theta):
    c, s = np.cos(theta/2), np.sin(theta/2)
    return np.array([[c, -s], [s, c]])

def Rz(theta):
    return np.array([[np.exp(-1j*theta/2), 0], [0, np.exp(1j*theta/2)]])

def create_state(theta, phi):
    """
    Create state with Bloch sphere angles (θ, φ).
    |ψ⟩ = cos(θ/2)|0⟩ + e^(iφ)sin(θ/2)|1⟩
    """
    ket_0 = np.array([1, 0])
    return Rz(phi) @ Ry(theta) @ ket_0

# Create states at different Bloch sphere positions
print("Creating states at various Bloch sphere positions:")
print(f"{'(θ, φ)':<15} {'State':<30} {'Position'}")
print("-" * 70)

positions = [
    (0, 0, "North pole |0⟩"),
    (np.pi, 0, "South pole |1⟩"),
    (np.pi/2, 0, "|+⟩ on +X axis"),
    (np.pi/2, np.pi, "|-⟩ on -X axis"),
    (np.pi/2, np.pi/2, "|+i⟩ on +Y axis"),
]

for theta, phi, desc in positions:
    state = create_state(theta, phi)
    state_str = f"[{state[0]:.3f}, {state[1]:.3f}]"
    print(f"({np.degrees(theta):>3.0f}°, {np.degrees(phi):>3.0f}°)   {state_str:<30} {desc}")
```

## 9. Rotation Composition

Rotations around different axes don't commute in general:

```python
import numpy as np

def Rx(theta):
    c, s = np.cos(theta/2), np.sin(theta/2)
    return np.array([[c, -1j*s], [-1j*s, c]])

def Ry(theta):
    c, s = np.cos(theta/2), np.sin(theta/2)
    return np.array([[c, -s], [s, c]])

def Rz(theta):
    return np.array([[np.exp(-1j*theta/2), 0], [0, np.exp(1j*theta/2)]])

theta = np.pi / 2

# Check commutativity
RxRy = Rx(theta) @ Ry(theta)
RyRx = Ry(theta) @ Rx(theta)

print("Do rotations commute?")
print(f"Rx(π/2) Ry(π/2) = Ry(π/2) Rx(π/2)? {np.allclose(RxRy, RyRx)}")

# Same axis rotations DO commute
Rx1Rx2 = Rx(np.pi/4) @ Rx(np.pi/2)
Rx2Rx1 = Rx(np.pi/2) @ Rx(np.pi/4)
print(f"Rx(π/4) Rx(π/2) = Rx(π/2) Rx(π/4)? {np.allclose(Rx1Rx2, Rx2Rx1)}")

# Same axis rotations add
Rx_sum = Rx(np.pi/4 + np.pi/2)
print(f"Rx(π/4) Rx(π/2) = Rx(3π/4)? {np.allclose(Rx1Rx2, Rx_sum)}")
```

## Summary

In this lesson, you learned:

1. **Rx(θ)**: Rotation around X-axis with off-diagonal -i·sin(θ/2)
2. **Ry(θ)**: Rotation around Y-axis (real matrix)
3. **Rz(θ)**: Rotation around Z-axis (diagonal matrix)
4. **Pauli Relation**: R_n(π) = -i·σ_n for each axis
5. **ZYZ Decomposition**: U = e^(iα) Rz(β) Ry(γ) Rz(δ)
6. **State Creation**: Ry(θ) Rz(φ) |0⟩ creates any state
7. **Non-commutativity**: Different axis rotations don't commute

<Tip>
Rotation gates are fundamental for variational quantum algorithms (VQE, QAOA) where the angles θ are parameters optimized by a classical computer!
</Tip>
