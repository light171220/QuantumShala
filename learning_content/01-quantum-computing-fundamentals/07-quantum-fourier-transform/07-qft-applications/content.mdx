## 2. Quantum Phase Estimation

QPE uses inverse QFT to extract the phase φ from U|ψ⟩ = e^(2πiφ)|ψ⟩:

<Formula>
$$\text{QPE}: |0\rangle^{\otimes n}|ψ\rangle \xrightarrow{\text{controlled-}U^{2^k}} \xrightarrow{\text{QFT}^{-1}} |\tilde{\phi}\rangle|ψ\rangle$$
</Formula>

```python
import numpy as np

def simulate_qpe_core(n_bits, true_phase):
    """
    Simulate the core of QPE (simplified).
    Shows how QFT^-1 extracts the phase.
    """
    N = 2**n_bits
    
    # After controlled-U operations, counting register is in state:
    # |ψ⟩ = (1/√N) Σⱼ e^(2πi·j·φ)|j⟩
    state = np.zeros(N, dtype=complex)
    for j in range(N):
        state[j] = np.exp(2j * np.pi * j * true_phase)
    state /= np.sqrt(N)
    
    # Apply inverse QFT
    omega = np.exp(-2j * np.pi / N)  # Note: negative for inverse
    qft_inv = np.array([[omega**(j*k) for k in range(N)] 
                        for j in range(N)]) / np.sqrt(N)
    
    result = qft_inv @ state
    probs = np.abs(result)**2
    
    # Find peak
    measured = np.argmax(probs)
    estimated_phase = measured / N
    
    return estimated_phase, probs

print("QPE Simulation")
print("=" * 40)

for n in [3, 4, 5]:
    true_phase = 0.375  # 3/8
    estimated, probs = simulate_qpe_core(n, true_phase)
    error = abs(estimated - true_phase)
    print(f"\nn = {n} bits:")
    print(f"  True phase: {true_phase}")
    print(f"  Estimated:  {estimated}")
    print(f"  Error: {error:.4f}")
```

## 3. Period Finding

QFT converts periodic signals in the computational basis to peaked distributions:

```python
import numpy as np

def demonstrate_period_finding(n_qubits, period):
    """
    Show how QFT reveals period of a periodic state.
    """
    N = 2**n_qubits
    
    # Create periodic state: superposition over 0, r, 2r, 3r, ...
    state = np.zeros(N, dtype=complex)
    for j in range(0, N, period):
        state[j] = 1
    state /= np.linalg.norm(state)
    
    # Apply QFT
    omega = np.exp(2j * np.pi / N)
    qft = np.array([[omega**(j*k) for k in range(N)] 
                    for j in range(N)]) / np.sqrt(N)
    
    transformed = qft @ state
    probs = np.abs(transformed)**2
    
    # Find peaks
    threshold = 1 / (N * 2)
    peaks = np.where(probs > threshold)[0]
    
    if len(peaks) > 1:
        spacing = peaks[1] - peaks[0]
        detected_period = N // spacing
    else:
        detected_period = None
    
    return peaks, detected_period, probs

print("Period Finding via QFT")
print("=" * 50)

n = 5  # 32 states
for period in [2, 4, 8]:
    peaks, detected, probs = demonstrate_period_finding(n, period)
    print(f"\nActual period: {period}")
    print(f"  QFT peaks at: {list(peaks)}")
    print(f"  Peak spacing: {peaks[1] - peaks[0] if len(peaks) > 1 else 'N/A'}")
    print(f"  Detected period: {detected}")
    print(f"  Correct: {'✓' if detected == period else '✗'}")
```

## 4. Role in Shor's Algorithm

Shor's algorithm for factoring uses QFT in a crucial way:

```python
import numpy as np

def shors_period_finding_step():
    """
    Illustrate the QFT step in Shor's algorithm.
    """
    print("Shor's Algorithm Structure")
    print("=" * 50)
    print("""
1. Choose random a < N
2. Create superposition: |0⟩|0⟩ → Σⱼ|j⟩|0⟩
3. Compute f(j) = aʲ mod N: Σⱼ|j⟩|aʲ mod N⟩
4. Measure second register → collapse to periodic state
5. Apply QFT to first register ← KEY STEP
6. Measure → get multiple of N/r
7. Use continued fractions to find period r
    """)
    
    # Simplified simulation
    N_factor = 15
    a = 7
    
    # Period of 7^j mod 15
    values = [pow(a, j, N_factor) for j in range(20)]
    
    print(f"\nExample: N = {N_factor}, a = {a}")
    print(f"Sequence 7^j mod 15: {values[:10]}...")
    
    # Find period
    for r in range(1, 20):
        if pow(a, r, N_factor) == 1:
            print(f"Period r = {r}")
            break
    
    print(f"\nQFT reveals this period from the quantum state!")

shors_period_finding_step()
```

## 5. Quantum Arithmetic (Addition)

QFT enables efficient quantum addition:

```python
import numpy as np

def qft_based_addition_concept():
    """
    Explain QFT-based addition.
    """
    print("QFT-Based Quantum Addition")
    print("=" * 50)
    print("""
The Draper adder uses QFT for addition:

1. Encode number a in quantum register |a⟩
2. Apply QFT: |a⟩ → (1/√N) Σⱼ e^(2πi·j·a/N)|j⟩
3. Add phase for b: → (1/√N) Σⱼ e^(2πi·j·(a+b)/N)|j⟩
4. Apply QFT⁻¹: → |a + b mod N⟩

Key insight: In Fourier basis, addition = phase rotation!
    """)
    
    # Demonstrate
    n = 3
    N = 2**n
    a, b = 3, 5
    
    # Simulate
    # Start with |a⟩
    state = np.zeros(N, dtype=complex)
    state[a] = 1
    
    # QFT
    omega = np.exp(2j * np.pi / N)
    qft = np.array([[omega**(j*k) for k in range(N)] 
                    for j in range(N)]) / np.sqrt(N)
    state = qft @ state
    
    # Add phase for b (this is the "addition" in Fourier space)
    for j in range(N):
        state[j] *= np.exp(2j * np.pi * j * b / N)
    
    # Inverse QFT
    qft_inv = qft.conj().T
    state = qft_inv @ state
    
    # Result
    result = np.argmax(np.abs(state)**2)
    print(f"\nExample: {a} + {b} = {result} (mod {N})")
    print(f"Correct: {(a + b) % N}")

qft_based_addition_concept()
```
