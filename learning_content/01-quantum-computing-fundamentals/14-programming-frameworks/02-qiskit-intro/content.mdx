# Introduction to Qiskit

## Introduction

Qiskit is IBM's open-source quantum computing framework and the most widely used quantum programming SDK. It provides tools for creating, simulating, and running quantum circuits on real IBM Quantum hardware.

## Your First Circuit

```python
from qiskit import QuantumCircuit

# Create a quantum circuit with 2 qubits
qc = QuantumCircuit(2)

# Add gates
qc.h(0)           # Hadamard on qubit 0
qc.cx(0, 1)       # CNOT: control=0, target=1

# Add measurements
qc.measure_all()

# Display the circuit
print("Bell State Circuit:")
print(qc.draw())

# ASCII output:
"""
     ┌───┐      ░ ┌─┐
q_0: ┤ H ├──■───░─┤M├───
     └───┘┌─┴─┐ ░ └╥┘┌─┐

          └───┘ ░  ║ └╥┘
c: 2/══════════════╩══╩═
                   0  1
"""
```

## Common Gates

```python
from qiskit import QuantumCircuit
import numpy as np

qc = QuantumCircuit(2)

# SINGLE-QUBIT GATES
print("Single-Qubit Gates:")
print("=" * 40)

# Pauli gates
qc.x(0)    # X gate (bit flip)
qc.y(0)    # Y gate
qc.z(0)    # Z gate (phase flip)

# Hadamard
qc.h(0)    # Creates superposition

# Phase gates
qc.s(0)    # S gate (√Z)
qc.t(0)    # T gate (√S)

# Rotation gates
qc.rx(np.pi/4, 0)   # Rotate around X
qc.ry(np.pi/4, 0)   # Rotate around Y
qc.rz(np.pi/4, 0)   # Rotate around Z

# TWO-QUBIT GATES
print("\nTwo-Qubit Gates:")
print("=" * 40)

qc.cx(0, 1)    # CNOT (controlled-X)
qc.cz(0, 1)    # Controlled-Z
qc.swap(0, 1)  # SWAP

# General controlled gates
qc.cy(0, 1)    # Controlled-Y
qc.ch(0, 1)    # Controlled-Hadamard

print("""
Gate Quick Reference:
    x(q)         : Bit flip
    h(q)         : Hadamard
    cx(c, t)     : CNOT
    rz(θ, q)     : Z-rotation by θ
    measure(q, c): Measure qubit to classical bit
""")
```

## Running on Simulator

```python
from qiskit import QuantumCircuit
from qiskit.primitives import Sampler

# Create Bell state circuit
qc = QuantumCircuit(2)
qc.h(0)
qc.cx(0, 1)
qc.measure_all()

# Create sampler and run
sampler = Sampler()
job = sampler.run(qc, shots=1000)
result = job.result()

# Get counts
quasi_dist = result.quasi_dists[0]
print("Results:")
print("=" * 40)
for bitstring, probability in quasi_dist.items():
    print(f"  {bin(bitstring)[2:].zfill(2)}: {probability:.3f}")

# Expected output:
# 00: ~0.500
# 11: ~0.500
```

## Circuit Manipulation

```python
from qiskit import QuantumCircuit

# Create circuits
qc1 = QuantumCircuit(2, name='Bell')
qc1.h(0)
qc1.cx(0, 1)

qc2 = QuantumCircuit(2, name='Measure')
qc2.measure_all()

# Compose circuits
combined = qc1.compose(qc2)

# Barrier (for grouping)
qc = QuantumCircuit(3)
qc.h([0, 1, 2])
qc.barrier()          # Visual separator
qc.cx(0, 1)
qc.cx(1, 2)

# Circuit properties
print("Circuit Properties:")
print("=" * 40)
print(f"  Depth: {qc.depth()}")
print(f"  Width: {qc.width()}")
print(f"  Num qubits: {qc.num_qubits}")
print(f"  Gate counts: {qc.count_ops()}")

# Inverse circuit
qc_inv = qc.inverse()

# Copy circuit
qc_copy = qc.copy()
```
