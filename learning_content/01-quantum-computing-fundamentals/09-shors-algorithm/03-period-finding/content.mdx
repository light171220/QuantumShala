## 1. The Key Insight

Shor's algorithm reduces factoring to **period finding**. The period of a modular exponential function reveals information about the factors.

<Definition term="Period Finding Problem">
Given f(x) = a^x mod N, find the smallest positive integer r (the period) such that:

$$f(x + r) = f(x) \quad \text{for all } x$$

Equivalently: find r such that a^r ≡ 1 (mod N)
</Definition>

```python
def find_period_classical(a, N):
    """
    Find the period r of f(x) = a^x mod N.
    Returns r such that a^r ≡ 1 (mod N).
    """
    x = 1
    r = 0
    while True:
        r += 1
        x = (x * a) % N
        if x == 1:
            return r
        if r > N:  # Safety limit
            return None

# Example
N = 15
print(f"Periods of a^x mod {N}:")
print(f"{'a':>4} {'Period r':>10} {'Verification':>20}")
print("-" * 40)

for a in [2, 4, 7, 8, 11, 13, 14]:
    r = find_period_classical(a, N)
    verification = f"{a}^{r} mod {N} = {pow(a, r, N)}"
    print(f"{a:>4} {r:>10} {verification:>20}")
```

## 2. From Period to Factors

The magic happens when the period r is **even** and a^(r/2) ≠ ±1 (mod N):

<Formula>
$$a^r \equiv 1 \pmod{N}$$
$$\Rightarrow a^r - 1 \equiv 0 \pmod{N}$$
$$\Rightarrow (a^{r/2} - 1)(a^{r/2} + 1) \equiv 0 \pmod{N}$$
</Formula>

This means N divides (a^(r/2) - 1)(a^(r/2) + 1), so:

<Formula>
$$\gcd(a^{r/2} - 1, N) \quad \text{or} \quad \gcd(a^{r/2} + 1, N)$$
</Formula>

gives a non-trivial factor of N!

```python
import math

def factor_from_period(a, r, N):
    """
    Given period r of a^x mod N, try to find factors.
    
    Returns:
        (p, q) if successful
        None if this a,r combination doesn't work
    """
    # Check if r is even
    if r % 2 != 0:
        return None, "Period is odd"
    
    # Compute a^(r/2) mod N
    x = pow(a, r // 2, N)
    
    # Check if a^(r/2) ≡ ±1 (mod N)
    if x == 1 or x == N - 1:
        return None, f"a^(r/2) = {x} (trivial)"
    
    # Compute GCDs
    factor1 = math.gcd(x - 1, N)
    factor2 = math.gcd(x + 1, N)
    
    # Check for non-trivial factors
    for f in [factor1, factor2]:
        if 1 < f < N:
            return (f, N // f), "Success!"
    
    return None, "No non-trivial factor found"

# Demonstration
N = 15  # = 3 × 5

print(f"Factoring N = {N} using period finding")
print("=" * 60)

for a in [2, 4, 7, 8, 11, 13]:
    if math.gcd(a, N) > 1:
        print(f"a = {a}: gcd(a, N) = {math.gcd(a, N)} (trivial factor!)")
        continue
    
    r = find_period_classical(a, N)
    factors, status = factor_from_period(a, r, N)
    
    print(f"\na = {a}:")
    print(f"  Period r = {r}")
    print(f"  r even? {r % 2 == 0}")
    
    if r % 2 == 0:
        x = pow(a, r // 2, N)
        print(f"  a^(r/2) mod N = {a}^{r//2} mod {N} = {x}")
        print(f"  gcd({x}-1, {N}) = {math.gcd(x-1, N)}")
        print(f"  gcd({x}+1, {N}) = {math.gcd(x+1, N)}")
    
    print(f"  Result: {factors} - {status}")
```

## 3. The Complete Algorithm

```python
import math
import random

def shors_algorithm_classical(N):
    """
    Classical simulation of Shor's algorithm.
    (The quantum part finds the period efficiently)
    """
    print(f"\nShor's Algorithm for N = {N}")
    print("=" * 50)
    
    # Step 1: Check if N is even
    if N % 2 == 0:
        return 2, N // 2, "N is even"
    
    # Step 2: Check if N = a^k for some a, k
    for k in range(2, int(math.log2(N)) + 1):
        a = round(N ** (1/k))
        if a ** k == N:
            return a, N // a, f"N = {a}^{k}"
    
    # Step 3: Choose random a
    max_attempts = 10
    for attempt in range(max_attempts):
        a = random.randint(2, N - 2)
        print(f"\nAttempt {attempt + 1}: a = {a}")
        
        # Step 4: Check gcd(a, N)
        g = math.gcd(a, N)
        if g > 1:
            print(f"  Lucky! gcd({a}, {N}) = {g}")
            return g, N // g, "Found factor from gcd"
        
        # Step 5: Find period (quantum computer does this!)
        r = find_period_classical(a, N)
        print(f"  Period r = {r}")
        
        # Step 6: Check if r is useful
        if r % 2 != 0:
            print("  r is odd, trying another a")
            continue
        
        x = pow(a, r // 2, N)
        if x == N - 1:
            print(f"  a^(r/2) = {x} = -1 mod N, trying another a")
            continue
        
        # Step 7: Compute factors
        p = math.gcd(x - 1, N)
        q = math.gcd(x + 1, N)
        
        if 1 < p < N:
            print(f"  Found factor: {p}")
            return p, N // p, "Success"
        if 1 < q < N:
            print(f"  Found factor: {q}")
            return q, N // q, "Success"
    
    return None, None, "Failed after max attempts"

# Test on various semiprimes
test_numbers = [15, 21, 35, 77, 91, 143]

print("Factoring with Shor's Algorithm (Classical Simulation)")
print("=" * 60)

for N in test_numbers:
    p, q, status = shors_algorithm_classical(N)
    if p and q:
        print(f"\n✓ {N} = {p} × {q}")
    else:
        print(f"\n✗ Failed to factor {N}")
```

## 4. Success Probability

Not every choice of a works. The probability analysis:

```python
import math
from collections import Counter

def analyze_success_probability(N):
    """
    Analyze which values of a lead to successful factoring.
    """
    p, q = None, None
    # Find factors for reference
    for i in range(2, int(N**0.5) + 1):
        if N % i == 0:
            p, q = i, N // i
            break
    
    results = {"success": 0, "odd_period": 0, "trivial": 0, "gcd_factor": 0}
    
    for a in range(2, N):
        g = math.gcd(a, N)
        if g > 1:
            results["gcd_factor"] += 1
            continue
        
        r = find_period_classical(a, N)
        
        if r % 2 != 0:
            results["odd_period"] += 1
            continue
        
        x = pow(a, r // 2, N)
        if x == 1 or x == N - 1:
            results["trivial"] += 1
            continue
        
        results["success"] += 1
    
    total = N - 2  # a from 2 to N-1
    
    print(f"\nSuccess Analysis for N = {N} = {p} × {q}")
    print("-" * 40)
    for outcome, count in results.items():
        pct = 100 * count / total
        print(f"  {outcome:15}: {count:3} ({pct:5.1f}%)")
    
    print(f"\n  Total valid a: {total}")
    print(f"  Success rate: {100 * results['success'] / total:.1f}%")

# Analyze a few numbers
for N in [15, 21, 35]:
    analyze_success_probability(N)
```

## 5. Why This Works: Number Theory

The reduction works because of **order theory** in modular arithmetic:

<Definition term="Order">
The order of a modulo N is the smallest positive integer r such that a^r ≡ 1 (mod N).
</Definition>

Key theorem: If N = pq and gcd(a, N) = 1, then the order r of a modulo N divides:
- lcm(p-1, q-1) (Carmichael function λ(N))
- φ(N) = (p-1)(q-1) (Euler's totient)

```python
import math

def order_analysis(N, p, q):
    """Analyze orders modulo N = p × q."""
    print(f"\nOrder Analysis for N = {N} = {p} × {q}")
    print("=" * 50)
    
    phi_N = (p - 1) * (q - 1)
    lambda_N = math.lcm(p - 1, q - 1)
    
    print(f"φ(N) = (p-1)(q-1) = {phi_N}")
    print(f"λ(N) = lcm(p-1, q-1) = {lambda_N}")
    print()
    
    print(f"{'a':>4} {'order(a)':>10} {'divides φ(N)?':>15} {'divides λ(N)?':>15}")
    print("-" * 50)
    
    for a in range(2, min(N, 15)):
        if math.gcd(a, N) > 1:
            continue
        r = find_period_classical(a, N)
        div_phi = "Yes" if phi_N % r == 0 else "No"
        div_lambda = "Yes" if lambda_N % r == 0 else "No"
        print(f"{a:>4} {r:>10} {div_phi:>15} {div_lambda:>15}")

order_analysis(15, 3, 5)
order_analysis(21, 3, 7)
```

## Summary

In this lesson, you learned:

- **Period Finding**: Find r such that a^r ≡ 1 (mod N)
- **Factoring Reduction**: Period r reveals factors via gcd(a^(r/2) ± 1, N)
- **Success Conditions**: r must be even and a^(r/2) ≢ ±1 (mod N)
- **Probability**: Works with probability ≥ 1/2 for random a
- **Quantum Advantage**: Classical period finding is exponential, quantum is polynomial!

## Key Formula

<Formula>
$$\text{If } a^r \equiv 1 \pmod{N} \text{ and } r \text{ is even:}$$
$$\gcd(a^{r/2} - 1, N) \times \gcd(a^{r/2} + 1, N) = N$$
</Formula>
