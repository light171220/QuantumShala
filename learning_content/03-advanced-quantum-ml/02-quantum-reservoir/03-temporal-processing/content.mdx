## 1. Temporal Memory in Quantum Systems

Quantum reservoirs process temporal information through the evolution of quantum states, where past inputs influence current states through coherent quantum dynamics.

<Definition term="Quantum Memory Capacity">
Quantum memory capacity measures the ability of a quantum reservoir to retain and recall past inputs. It quantifies how much information about previous time steps can be linearly recovered from current observable measurements.
</Definition>

### The Memory-Nonlinearity Trade-off

<Formula>
$$MC_{total} = \sum_{k=1}^{\infty} MC_k = \sum_{k=1}^{\infty} \max_{W} \text{cor}^2(W \cdot \mathbf{x}(t), u(t-k))$$
</Formula>

```python
import numpy as np
from scipy.linalg import expm

class TemporalQuantumReservoir:
    """Quantum reservoir optimized for temporal processing."""

    def __init__(self, n_qubits, coupling='ising', memory_bias=0.5):
        """
        Initialize temporal quantum reservoir.

        Args:
            n_qubits: Number of qubits
            coupling: Coupling type ('ising', 'heisenberg', 'xy')
            memory_bias: Balance between memory and nonlinearity (0-1)
        """
        self.n_qubits = n_qubits
        self.dim = 2**n_qubits
        self.memory_bias = memory_bias

        # Build Hamiltonian based on coupling type
        self.H = self._build_hamiltonian(coupling)

        # Initialize state
        self.reset()

        # Observable operators for feature extraction
        self.observables = self._generate_observables()

    def _build_hamiltonian(self, coupling):
        """Build Hamiltonian with tunable memory properties."""
        H = np.zeros((self.dim, self.dim), dtype=complex)

        # Pauli matrices
        I = np.eye(2, dtype=complex)
        X = np.array([[0, 1], [1, 0]], dtype=complex)
        Y = np.array([[0, -1j], [1j, 0]], dtype=complex)
        Z = np.array([[1, 0], [0, -1]], dtype=complex)

        def tensor_product(ops):
            result = ops[0]
            for op in ops[1:]:
                result = np.kron(result, op)
            return result

        # Memory-biased field strength
        h_field = 1.0 - self.memory_bias  # Lower field = more memory
        J_coupling = self.memory_bias  # Higher coupling = more complexity

        if coupling == 'ising':
            # Transverse field Ising model
            for i in range(self.n_qubits):
                ops = [I] * self.n_qubits
                ops[i] = X
                H += h_field * np.random.uniform(0.5, 1.5) * tensor_product(ops)

            for i in range(self.n_qubits - 1):
                ops = [I] * self.n_qubits
                ops[i] = Z
                ops[i + 1] = Z
                H += J_coupling * np.random.uniform(0.5, 1.5) * tensor_product(ops)

        elif coupling == 'heisenberg':
            # Heisenberg XXX model
            for i in range(self.n_qubits - 1):
                for P in [X, Y, Z]:
                    ops = [I] * self.n_qubits
                    ops[i] = P
                    ops[i + 1] = P
                    H += J_coupling * tensor_product(ops)

        elif coupling == 'xy':
            # XY model
            for i in range(self.n_qubits - 1):
                for P in [X, Y]:
                    ops = [I] * self.n_qubits
                    ops[i] = P
                    ops[i + 1] = P
                    H += J_coupling * tensor_product(ops)

        return H

    def _generate_observables(self):
        """Generate measurement observables."""
        I = np.eye(2, dtype=complex)
        paulis = [
            np.array([[1, 0], [0, -1]], dtype=complex),  # Z
            np.array([[0, 1], [1, 0]], dtype=complex),   # X
            np.array([[0, -1j], [1j, 0]], dtype=complex) # Y
        ]

        observables = []

        # Single-qubit observables
        for i in range(self.n_qubits):
            for P in paulis:
                ops = [I] * self.n_qubits
                ops[i] = P
                O = ops[0]
                for op in ops[1:]:
                    O = np.kron(O, op)
                observables.append(O)

        return observables

    def reset(self):
        """Reset to initial state."""
        self.rho = np.zeros((self.dim, self.dim), dtype=complex)
        self.rho[0, 0] = 1.0  # |00...0⟩

    def step(self, u, dt=1.0):
        """
        Process one time step.

        Args:
            u: Input value at current time
            dt: Evolution time

        Returns:
            Feature vector from observable measurements
        """
        # Encode input
        theta = u * np.pi
        Ry = np.array([
            [np.cos(theta/2), -np.sin(theta/2)],
            [np.sin(theta/2), np.cos(theta/2)]
        ], dtype=complex)

        I_rest = np.eye(2**(self.n_qubits - 1), dtype=complex)
        U_enc = np.kron(Ry, I_rest)

        # Apply encoding
        self.rho = U_enc @ self.rho @ U_enc.conj().T

        # Time evolution
        U_evol = expm(-1j * self.H * dt)
        self.rho = U_evol @ self.rho @ U_evol.conj().T

        # Measure observables
        features = []
        for O in self.observables:
            exp_val = np.real(np.trace(O @ self.rho))
            features.append(exp_val)

        return np.array(features)

    def process_sequence(self, inputs, dt=1.0):
        """
        Process entire input sequence.

        Args:
            inputs: Sequence of input values
            dt: Evolution time per step

        Returns:
            Matrix of features (time steps × observables)
        """
        self.reset()
        features = []

        for u in inputs:
            feat = self.step(u, dt)
            features.append(feat)

        return np.array(features)
```

## 2. Memory Capacity Analysis

Systematic measurement of quantum reservoir memory capacity reveals how quantum properties enhance temporal processing.

<Definition term="Linear Memory Capacity">
Linear memory capacity MC_k for delay k measures the squared correlation between the optimal linear reconstruction of input u(t-k) from current features x(t). Total capacity is bounded by the number of independent features.
</Definition>

### Measuring Memory Capacity

```python
def measure_memory_capacity(reservoir, max_delay=30, n_samples=1000, washout=100):
    """
    Measure memory capacity of quantum reservoir.

    Args:
        reservoir: TemporalQuantumReservoir instance
        max_delay: Maximum delay to test
        n_samples: Number of input samples
        washout: Initial samples to discard

    Returns:
        dict with total MC and per-delay values
    """
    # Generate random input sequence
    np.random.seed(42)
    u = np.random.uniform(-1, 1, n_samples)

    # Collect reservoir features
    features = reservoir.process_sequence(u)

    # Discard washout period
    X = features[washout:]

    # Measure capacity for each delay
    mc_values = []

    for delay in range(1, max_delay + 1):
        if washout + delay >= n_samples:
            break

        # Target: input at time (t - delay)
        y = u[washout - delay:-delay] if delay < len(u) - washout else u[washout - delay:]

        # Ensure matching lengths
        min_len = min(len(X), len(y))
        X_d = X[:min_len]
        y_d = y[:min_len]

        # Ridge regression
        reg = 1e-6
        XTX = X_d.T @ X_d + reg * np.eye(X_d.shape[1])
        W = np.linalg.solve(XTX, X_d.T @ y_d)

        # Compute reconstruction
        y_pred = X_d @ W

        # Memory capacity for this delay
        if np.std(y_d) > 1e-10 and np.std(y_pred) > 1e-10:
            corr = np.corrcoef(y_pred, y_d)[0, 1]
            mc_k = corr**2 if not np.isnan(corr) else 0
        else:
            mc_k = 0

        mc_values.append(mc_k)

    return {
        'total_mc': sum(mc_values),
        'mc_per_delay': mc_values,
        'delays': list(range(1, len(mc_values) + 1))
    }


def compare_reservoir_memory(n_qubits_list=[3, 4, 5], coupling='ising'):
    """Compare memory capacity across different reservoir sizes."""
    results = {}

    for n_qubits in n_qubits_list:
        reservoir = TemporalQuantumReservoir(n_qubits, coupling)
        mc_result = measure_memory_capacity(reservoir)
        results[n_qubits] = mc_result

        print(f"Qubits: {n_qubits}")
        print(f"  Total MC: {mc_result['total_mc']:.3f}")
        print(f"  Feature dimension: {3 * n_qubits}")
        print()

    return results
```

<Tip>
The total memory capacity of a reservoir is theoretically bounded by the number of independent features. For quantum reservoirs with N qubits measuring all single-qubit Pauli observables, this gives 3N features, but quantum correlations can enhance effective capacity.
</Tip>

## 3. Time Series Prediction

Quantum reservoirs excel at predicting chaotic and complex time series where high-dimensional feature spaces provide advantages.

### Mackey-Glass Time Series

<Formula>
$$\frac{dx}{dt} = \beta \frac{x(t-\tau)}{1 + x(t-\tau)^n} - \gamma x(t)$$
</Formula>

```python
def generate_mackey_glass(n_samples, tau=17, beta=0.2, gamma=0.1, n=10, dt=1.0):
    """
    Generate Mackey-Glass chaotic time series.

    Args:
        n_samples: Number of samples to generate
        tau: Delay parameter (tau > 17 for chaos)
        beta, gamma, n: System parameters
        dt: Time step

    Returns:
        Time series array
    """
    # Initialize with random history
    history_len = int(tau / dt) + 1
    x = np.random.uniform(0.5, 1.5, history_len).tolist()

    for _ in range(n_samples):
        x_tau = x[-history_len]
        dx = beta * x_tau / (1 + x_tau**n) - gamma * x[-1]
        x.append(x[-1] + dx * dt)

    return np.array(x[history_len:])


class QuantumTimeSeriesPredictor:
    """Time series prediction using quantum reservoir."""

    def __init__(self, n_qubits, prediction_horizon=1):
        """
        Initialize predictor.

        Args:
            n_qubits: Number of qubits in reservoir
            prediction_horizon: Steps ahead to predict
        """
        self.reservoir = TemporalQuantumReservoir(n_qubits, coupling='ising')
        self.horizon = prediction_horizon
        self.W_out = None

    def fit(self, time_series, train_ratio=0.7, washout=50, reg=1e-4):
        """
        Train the predictor.

        Args:
            time_series: Input time series
            train_ratio: Fraction for training
            washout: Initial steps to discard
            reg: Regularization strength
        """
        # Normalize input
        self.mean = np.mean(time_series)
        self.std = np.std(time_series)
        normalized = (time_series - self.mean) / self.std

        # Split data
        n_train = int(len(normalized) * train_ratio)
        train_data = normalized[:n_train]

        # Collect features
        features = self.reservoir.process_sequence(train_data)

        # Prepare training matrices
        X = features[washout:-self.horizon]
        y = train_data[washout + self.horizon:]

        # Ridge regression
        XTX = X.T @ X + reg * np.eye(X.shape[1])
        self.W_out = np.linalg.solve(XTX, X.T @ y)

        # Training error
        y_pred = X @ self.W_out
        train_mse = np.mean((y_pred - y)**2)

        return {'train_mse': train_mse, 'train_samples': len(y)}

    def predict(self, time_series, n_steps=None):
        """
        Make predictions on time series.

        Args:
            time_series: Input sequence
            n_steps: Number of future steps (None = one-step ahead)

        Returns:
            Predictions (denormalized)
        """
        if self.W_out is None:
            raise ValueError("Model not trained. Call fit() first.")

        normalized = (time_series - self.mean) / self.std
        features = self.reservoir.process_sequence(normalized)

        if n_steps is None:
            # One-step ahead prediction
            predictions = features @ self.W_out
        else:
            # Multi-step prediction (autoregressive)
            predictions = []
            current_features = features[-1:]

            for _ in range(n_steps):
                pred = current_features @ self.W_out
                predictions.append(pred[0])

                # Feed prediction back
                self.reservoir.step(pred[0])
                new_features = []
                for O in self.reservoir.observables:
                    new_features.append(np.real(np.trace(O @ self.reservoir.rho)))
                current_features = np.array([new_features])

            predictions = np.array(predictions)

        # Denormalize
        return predictions * self.std + self.mean

    def evaluate(self, time_series, train_ratio=0.7, washout=50):
        """
        Evaluate prediction performance.

        Returns:
            Dictionary with MSE, NMSE, correlation
        """
        normalized = (time_series - self.mean) / self.std
        n_train = int(len(normalized) * train_ratio)

        # Test on remaining data
        test_data = normalized[n_train:]
        features = self.reservoir.process_sequence(test_data)

        X = features[washout:-self.horizon]
        y_true = test_data[washout + self.horizon:]

        y_pred = X @ self.W_out

        mse = np.mean((y_pred - y_true)**2)
        nmse = mse / np.var(y_true)
        corr = np.corrcoef(y_pred, y_true)[0, 1]

        return {
            'mse': mse,
            'nmse': nmse,
            'correlation': corr,
            'test_samples': len(y_true)
        }
```

## 4. Quantum Fading Memory

Quantum systems exhibit unique fading memory properties due to decoherence and unitary dynamics.

<Definition term="Quantum Fading Memory">
In quantum reservoirs, fading memory arises from two sources: (1) natural damping from decoherence processes, and (2) the mixing of information through unitary evolution. The interplay determines the memory timescale.
</Definition>

<Warning>
Unlike classical reservoirs where fading memory is guaranteed by the echo state property, quantum reservoirs require careful Hamiltonian design to ensure old inputs are gradually forgotten while maintaining sufficient memory for the task.
</Warning>

```python
def analyze_fading_memory(reservoir, perturbation_strength=0.1, n_steps=50):
    """
    Analyze fading memory by measuring state divergence over time.

    Args:
        reservoir: TemporalQuantumReservoir instance
        perturbation_strength: Initial state perturbation
        n_steps: Number of evolution steps

    Returns:
        Trace distance over time
    """
    # Generate reference trajectory
    reservoir.reset()
    inputs = np.random.uniform(-1, 1, n_steps)

    ref_states = []
    for u in inputs:
        reservoir.step(u)
        ref_states.append(reservoir.rho.copy())

    # Generate perturbed trajectory (different initial state)
    reservoir.reset()
    # Apply small rotation to initial state
    theta = perturbation_strength * np.pi
    Ry = np.array([
        [np.cos(theta/2), -np.sin(theta/2)],
        [np.sin(theta/2), np.cos(theta/2)]
    ], dtype=complex)
    I_rest = np.eye(2**(reservoir.n_qubits - 1), dtype=complex)
    U_pert = np.kron(Ry, I_rest)
    reservoir.rho = U_pert @ reservoir.rho @ U_pert.conj().T

    # Evolve with same inputs
    trace_distances = []
    for t, u in enumerate(inputs):
        reservoir.step(u)

        # Trace distance: D(ρ, σ) = 0.5 * Tr|ρ - σ|
        diff = reservoir.rho - ref_states[t]
        eigenvalues = np.linalg.eigvalsh(diff @ diff.conj().T)
        trace_dist = 0.5 * np.sum(np.sqrt(np.maximum(eigenvalues, 0)))
        trace_distances.append(trace_dist)

    return {
        'trace_distances': trace_distances,
        'memory_time': estimate_memory_time(trace_distances)
    }


def estimate_memory_time(distances, threshold=0.1):
    """Estimate characteristic memory time from convergence."""
    initial = distances[0] if distances else 0
    for t, d in enumerate(distances):
        if d < initial * threshold:
            return t
    return len(distances)
```

## 5. Sequential Task Processing

Quantum reservoirs can handle various sequential learning tasks beyond time series prediction.

### Sequence Classification

```python
class SequenceClassifier:
    """Classify sequences using quantum reservoir."""

    def __init__(self, n_qubits, n_classes):
        """
        Initialize classifier.

        Args:
            n_qubits: Qubits in reservoir
            n_classes: Number of output classes
        """
        self.reservoir = TemporalQuantumReservoir(n_qubits)
        self.n_classes = n_classes
        self.W_out = None

    def _extract_sequence_features(self, sequences):
        """Extract features from each sequence."""
        all_features = []

        for seq in sequences:
            self.reservoir.reset()

            # Process sequence
            for u in seq:
                self.reservoir.step(u)

            # Final state features
            final_features = []
            for O in self.reservoir.observables:
                final_features.append(np.real(np.trace(O @ self.reservoir.rho)))

            all_features.append(final_features)

        return np.array(all_features)

    def fit(self, sequences, labels, reg=1e-3):
        """
        Train classifier.

        Args:
            sequences: List of input sequences
            labels: Integer class labels
            reg: Regularization
        """
        X = self._extract_sequence_features(sequences)

        # One-hot encode labels
        Y = np.zeros((len(labels), self.n_classes))
        for i, label in enumerate(labels):
            Y[i, label] = 1

        # Train multi-output regression
        XTX = X.T @ X + reg * np.eye(X.shape[1])
        self.W_out = np.linalg.solve(XTX, X.T @ Y)

        # Training accuracy
        predictions = self.predict(sequences)
        accuracy = np.mean(predictions == labels)

        return {'train_accuracy': accuracy}

    def predict(self, sequences):
        """Predict class labels."""
        X = self._extract_sequence_features(sequences)
        scores = X @ self.W_out
        return np.argmax(scores, axis=1)

    def predict_proba(self, sequences):
        """Predict class probabilities."""
        X = self._extract_sequence_features(sequences)
        scores = X @ self.W_out
        # Softmax
        exp_scores = np.exp(scores - np.max(scores, axis=1, keepdims=True))
        return exp_scores / np.sum(exp_scores, axis=1, keepdims=True)
```

### Temporal Pattern Generation

```python
def train_pattern_generator(reservoir, target_pattern, n_epochs=100, reg=1e-4):
    """
    Train reservoir to generate temporal pattern.

    Args:
        reservoir: TemporalQuantumReservoir
        target_pattern: Target sequence to generate
        n_epochs: Training epochs
        reg: Regularization

    Returns:
        Trained output weights
    """
    pattern_length = len(target_pattern)

    # Collect features for each position
    X_all = []

    for epoch in range(n_epochs):
        reservoir.reset()

        for t, target in enumerate(target_pattern):
            # Use previous output as input (teacher forcing)
            if t == 0:
                u = 0.0
            else:
                u = target_pattern[t-1]

            features = reservoir.step(u)
            X_all.append(features)

    X = np.array(X_all)
    Y = np.tile(target_pattern, n_epochs)

    # Train
    XTX = X.T @ X + reg * np.eye(X.shape[1])
    W_out = np.linalg.solve(XTX, X.T @ Y)

    return W_out


def generate_pattern(reservoir, W_out, length, seed_input=0.0):
    """Generate pattern autonomously."""
    reservoir.reset()

    generated = []
    u = seed_input

    for _ in range(length):
        features = reservoir.step(u)
        output = features @ W_out
        generated.append(output)
        u = output  # Feed output back as input

    return np.array(generated)
```

## 6. Information Processing Capacity

Beyond memory, quantum reservoirs can compute nonlinear functions of past inputs.

<Definition term="Total Information Processing Capacity">
Total IPC measures all recoverable information from the reservoir, including linear memory and nonlinear transformations of past inputs. For quantum reservoirs, this includes both classical and quantum correlations.
</Definition>

<Formula>
$$IPC_{total} = \sum_{i} \max_W \text{cor}^2(W \cdot \mathbf{x}(t), f_i(u(t-1), u(t-2), ...))$$
</Formula>

```python
def measure_nonlinear_capacity(reservoir, degree=2, max_delay=10, n_samples=800):
    """
    Measure nonlinear processing capacity.

    Args:
        reservoir: TemporalQuantumReservoir
        degree: Polynomial degree for nonlinear targets
        max_delay: Maximum memory depth
        n_samples: Number of samples

    Returns:
        Nonlinear capacity metrics
    """
    # Generate input
    u = np.random.uniform(-1, 1, n_samples)

    # Collect features
    features = reservoir.process_sequence(u)

    washout = 50
    X = features[washout + max_delay:]

    capacities = {}

    # Linear memory (degree 1)
    capacities['linear'] = []
    for delay in range(1, max_delay + 1):
        y = u[washout + max_delay - delay:-delay]
        min_len = min(len(X), len(y))

        reg = 1e-6
        XTX = X[:min_len].T @ X[:min_len] + reg * np.eye(X.shape[1])
        W = np.linalg.solve(XTX, X[:min_len].T @ y[:min_len])

        y_pred = X[:min_len] @ W
        corr = np.corrcoef(y_pred, y[:min_len])[0, 1]
        capacities['linear'].append(corr**2 if not np.isnan(corr) else 0)

    # Quadratic capacity (degree 2)
    if degree >= 2:
        capacities['quadratic'] = []
        for delay in range(1, max_delay + 1):
            y = u[washout + max_delay - delay:-delay]**2
            min_len = min(len(X), len(y))

            XTX = X[:min_len].T @ X[:min_len] + reg * np.eye(X.shape[1])
            W = np.linalg.solve(XTX, X[:min_len].T @ y[:min_len])

            y_pred = X[:min_len] @ W
            corr = np.corrcoef(y_pred, y[:min_len])[0, 1]
            capacities['quadratic'].append(corr**2 if not np.isnan(corr) else 0)

    # Cross-term capacity (products of different delays)
    capacities['cross_terms'] = []
    for d1 in range(1, min(5, max_delay)):
        for d2 in range(d1 + 1, min(6, max_delay + 1)):
            y = u[washout + max_delay - d1:-d1] * u[washout + max_delay - d2:-d2]
            min_len = min(len(X), len(y))

            XTX = X[:min_len].T @ X[:min_len] + reg * np.eye(X.shape[1])
            W = np.linalg.solve(XTX, X[:min_len].T @ y[:min_len])

            y_pred = X[:min_len] @ W
            corr = np.corrcoef(y_pred, y[:min_len])[0, 1]
            capacities['cross_terms'].append(corr**2 if not np.isnan(corr) else 0)

    # Total capacities
    total = {
        'linear_total': sum(capacities['linear']),
        'quadratic_total': sum(capacities.get('quadratic', [])),
        'cross_total': sum(capacities['cross_terms']),
        'details': capacities
    }

    return total
```

<Tip>
Quantum reservoirs often show enhanced nonlinear capacity compared to classical ones of similar size, particularly for cross-term computations where quantum correlations provide additional computational power.
</Tip>
