# Classical to Quantum Transfer

Classical neural networks are well-studied and easier to train. This lesson explores transferring classical knowledge to quantum models for better initialization and training.

## Learning Objectives

After completing this lesson, you will be able to:

- Initialize quantum circuits from classical models
- Map classical weights to quantum parameters
- Use classical pre-training for quantum models
- Implement hybrid initialization strategies
- Avoid barren plateaus with classical guidance
- Fine-tune classical-initialized quantum models

## 1. Why Classical to Quantum Transfer?

<Definition term="Classical to Quantum Transfer">
C→Q transfer uses knowledge from trained classical models to:

1. Initialize quantum circuit parameters intelligently
2. Avoid random initialization's barren plateaus
3. Reduce quantum training iterations
4. Guide quantum models toward good solutions
</Definition>

```python
import numpy as np

class ClassicalToQuantumMotivation:
    """
    Analyze motivation for C→Q transfer.
    """

    def __init__(self):
        pass

    def compare_initialization(self, n_trials=10):
        """
        Compare random vs informed initialization.
        """
        random_losses = []
        informed_losses = []

        for _ in range(n_trials):
            # Random initialization
            random_params = np.random.randn(10) * 0.1
            random_loss = self._simulate_initial_loss(random_params)
            random_losses.append(random_loss)

            # Informed initialization (from classical)
            informed_params = self._classical_informed_init(10)
            informed_loss = self._simulate_initial_loss(informed_params)
            informed_losses.append(informed_loss)

        return {
            'random_mean': np.mean(random_losses),
            'random_std': np.std(random_losses),
            'informed_mean': np.mean(informed_losses),
            'informed_std': np.std(informed_losses),
            'improvement': (np.mean(random_losses) - np.mean(informed_losses)) /
                          np.mean(random_losses) * 100
        }

    def _simulate_initial_loss(self, params):
        """Simulate initial loss for given parameters."""
        # Higher loss for random, lower for structured
        return np.sum(np.abs(params - 0.5)) + 0.1 * np.random.rand()

    def _classical_informed_init(self, n_params):
        """Initialize based on classical model knowledge."""
        # Start closer to good values
        return np.ones(n_params) * 0.5 + 0.05 * np.random.randn(n_params)

    def barren_plateau_analysis(self, n_qubits_range=[4, 8, 12]):
        """
        Analyze gradient variance vs qubit count.
        """
        results = []

        for n_qubits in n_qubits_range:
            # Random init: gradient vanishes exponentially
            random_grad_var = np.exp(-0.5 * n_qubits)

            # Informed init: reduced vanishing
            informed_grad_var = np.exp(-0.2 * n_qubits)

            results.append({
                'n_qubits': n_qubits,
                'random_grad_var': random_grad_var,
                'informed_grad_var': informed_grad_var,
                'ratio': informed_grad_var / random_grad_var
            })

        return results
```

## 2. Parameter Mapping Strategies

Map classical network weights to quantum circuit parameters.

```python
class ParameterMapper:
    """
    Map classical neural network parameters to quantum circuit parameters.
    """

    def __init__(self):
        pass

    def linear_mapping(self, classical_weights, target_range=(-np.pi, np.pi)):
        """
        Simple linear rescaling of classical weights.
        """
        w_min, w_max = classical_weights.min(), classical_weights.max()
        t_min, t_max = target_range

        if w_max - w_min < 1e-10:
            return np.ones_like(classical_weights) * (t_min + t_max) / 2

        normalized = (classical_weights - w_min) / (w_max - w_min)
        quantum_params = normalized * (t_max - t_min) + t_min

        return quantum_params

    def angular_mapping(self, classical_weights):
        """
        Map weights to rotation angles using arctan.
        """
        # Map to (-π/2, π/2) using arctan, then scale to (-π, π)
        return 2 * np.arctan(classical_weights)

    def periodic_mapping(self, classical_weights, period=2*np.pi):
        """
        Wrap classical weights to periodic quantum parameter space.
        """
        return np.mod(classical_weights, period)

    def structured_mapping(self, classical_weights, n_qubits, n_layers):
        """
        Map structured classical network to structured quantum circuit.

        Args:
            classical_weights: Flattened classical weights
            n_qubits: Number of qubits
            n_layers: Number of circuit layers
        """
        params_per_layer = n_qubits * 2  # RY and RZ per qubit
        total_params = n_layers * params_per_layer

        # Resize classical weights if needed
        if len(classical_weights) >= total_params:
            selected = classical_weights[:total_params]
        else:
            # Repeat and truncate
            repeats = int(np.ceil(total_params / len(classical_weights)))
            selected = np.tile(classical_weights, repeats)[:total_params]

        # Apply angular mapping
        quantum_params = self.angular_mapping(selected)

        return quantum_params.reshape(n_layers, n_qubits, 2)
```

## 3. Classical Pre-training Pipeline

Pre-train classical model then transfer to quantum.

```python
class ClassicalPretrainingPipeline:
    """
    Pre-train classical model, then transfer to quantum.
    """

    def __init__(self, n_qubits, n_layers):
        self.n_qubits = n_qubits
        self.n_layers = n_layers
        self.classical_model = None
        self.quantum_params = None

    def create_classical_analog(self):
        """
        Create classical analog of quantum circuit.
        """
        # MLP with similar structure to quantum circuit
        n_params = self.n_qubits * self.n_layers * 2
        self.classical_weights = np.random.randn(n_params) * 0.1
        return self.classical_weights

    def train_classical(self, x_train, y_train, epochs=100, lr=0.1):
        """
        Train classical analog model.
        """
        if self.classical_weights is None:
            self.create_classical_analog()

        print("Training classical analog:")
        for epoch in range(epochs):
            total_loss = 0

            for x, y in zip(x_train, y_train):
                # Forward pass (simple nonlinear function)
                pred = self._classical_forward(x)
                loss = (pred - y) ** 2
                total_loss += loss

                # Backward pass (numerical gradient)
                grad = self._classical_gradient(x, y)
                self.classical_weights -= lr * grad

            if epoch % 20 == 0:
                print(f"  Epoch {epoch}: Loss = {total_loss/len(x_train):.4f}")

        return self.classical_weights

    def _classical_forward(self, x):
        """Classical forward pass."""
        result = x
        for i in range(0, len(self.classical_weights), 2):
            w1, w2 = self.classical_weights[i], self.classical_weights[i+1]
            result = np.tanh(w1 * result + w2 * 0.1)
        return result

    def _classical_gradient(self, x, y, eps=0.01):
        """Numerical gradient."""
        grad = np.zeros_like(self.classical_weights)
        pred = self._classical_forward(x)

        for i in range(len(self.classical_weights)):
            self.classical_weights[i] += eps
            pred_plus = self._classical_forward(x)
            self.classical_weights[i] -= eps
            grad[i] = 2 * (pred - y) * (pred_plus - pred) / eps

        return grad

    def transfer_to_quantum(self):
        """
        Map trained classical weights to quantum parameters.
        """
        mapper = ParameterMapper()
        self.quantum_params = mapper.angular_mapping(self.classical_weights)
        return self.quantum_params

    def quantum_forward(self, x, params=None):
        """Quantum circuit forward pass (simulated)."""
        if params is None:
            params = self.quantum_params

        result = x
        for i in range(0, len(params), 2):
            theta, phi = params[i], params[i+1]
            result = np.sin(theta * result + phi)

        return result

    def fine_tune_quantum(self, x_train, y_train, epochs=20, lr=0.05):
        """
        Fine-tune quantum model after transfer.
        """
        print("\nFine-tuning quantum model:")

        for epoch in range(epochs):
            total_loss = 0

            for x, y in zip(x_train, y_train):
                pred = self.quantum_forward(x)
                loss = (pred - y) ** 2
                total_loss += loss

                # Gradient update
                grad = self._quantum_gradient(x, y)
                self.quantum_params -= lr * grad

            if epoch % 5 == 0:
                print(f"  Epoch {epoch}: Loss = {total_loss/len(x_train):.4f}")

        return self.quantum_params

    def _quantum_gradient(self, x, y, eps=0.01):
        """Quantum gradient (numerical)."""
        grad = np.zeros_like(self.quantum_params)
        pred = self.quantum_forward(x)

        for i in range(len(self.quantum_params)):
            self.quantum_params[i] += eps
            pred_plus = self.quantum_forward(x)
            self.quantum_params[i] -= eps
            grad[i] = 2 * (pred - y) * (pred_plus - pred) / eps

        return grad

# Demonstrate pre-training pipeline
print("\nClassical Pre-training Pipeline:")
print("=" * 50)

np.random.seed(42)

# Create pipeline
pipeline = ClassicalPretrainingPipeline(n_qubits=4, n_layers=2)

# Training data
x_train = np.linspace(-2, 2, 25)
y_train = np.sin(x_train)

# Step 1: Train classical
print("\nStep 1: Classical pre-training")
classical_weights = pipeline.train_classical(x_train, y_train, epochs=60)

# Step 2: Transfer to quantum
print("\nStep 2: Transfer to quantum")
quantum_params = pipeline.transfer_to_quantum()
print(f"Transferred {len(quantum_params)} parameters")

# Step 3: Fine-tune quantum
print("\nStep 3: Quantum fine-tuning")
final_params = pipeline.fine_tune_quantum(x_train, y_train, epochs=15)
```

## 4. Avoiding Barren Plateaus

Use classical knowledge to escape barren plateaus.

```python
class BarrenPlateauMitigation:
    """
    Use classical guidance to avoid barren plateaus.
    """

    def __init__(self, n_qubits, n_layers):
        self.n_qubits = n_qubits
        self.n_layers = n_layers
        self.n_params = n_qubits * n_layers * 2

    def random_initialization(self):
        """Standard random initialization."""
        return np.random.randn(self.n_params) * 0.1

    def layerwise_initialization(self):
        """
        Initialize with structured layer-wise pattern.
        Later layers have smaller initial values.
        """
        params = []
        for layer in range(self.n_layers):
            # Decay factor for deeper layers
            scale = 1.0 / (layer + 1)
            layer_params = np.random.randn(self.n_qubits * 2) * 0.1 * scale
            params.extend(layer_params)
        return np.array(params)

    def classical_guided_init(self, classical_weights):
        """
        Initialize using classical model as guide.
        """
        mapper = ParameterMapper()
        return mapper.angular_mapping(classical_weights[:self.n_params])

    def identity_init(self):
        """
        Initialize circuit to approximate identity.
        Rotations start near zero.
        """
        return np.zeros(self.n_params) + 0.01 * np.random.randn(self.n_params)

    def estimate_gradient_variance(self, init_method, n_samples=100):
        """
        Estimate gradient variance for given initialization.
        """
        gradients = []

        for _ in range(n_samples):
            if init_method == 'random':
                params = self.random_initialization()
            elif init_method == 'layerwise':
                params = self.layerwise_initialization()
            elif init_method == 'identity':
                params = self.identity_init()
            else:  # classical
                classical = np.random.randn(self.n_params)
                params = self.classical_guided_init(classical)

            # Compute gradient (simulated)
            grad = self._compute_gradient(params)
            gradients.append(grad)

        gradients = np.array(gradients)
        variance = np.var(gradients, axis=0).mean()

        return variance

    def _compute_gradient(self, params):
        """Simulated gradient computation."""
        # In barren plateau: gradient scales as exp(-n_qubits)
        base_grad = np.random.randn(self.n_params)

        # Random init has strong barren plateau
        if np.std(params) > 0.5:
            scale = np.exp(-0.3 * self.n_qubits)
        else:
            # Structured init mitigates
            scale = np.exp(-0.1 * self.n_qubits)

        return base_grad * scale

# Demonstrate barren plateau mitigation

print("=" * 50)

np.random.seed(42)

# Compare initialization methods
methods = ['random', 'layerwise', 'identity', 'classical']

    variance = bp_mitigator.estimate_gradient_variance(method)
```

</Warning>

## 5. Hybrid Training Schemes

Combine classical and quantum training.

```python
class HybridTrainingScheme:
    """
    Hybrid classical-quantum training approaches.
    """

    def __init__(self, n_qubits, n_layers):
        self.n_qubits = n_qubits
        self.n_layers = n_layers
        self.n_params = n_qubits * n_layers * 2

    def alternating_training(self, x_train, y_train, classical_model,
                              quantum_fn, epochs_per_phase=10, n_phases=3):
        """
        Alternate between classical and quantum training.
        """
        # Initialize from classical
        classical_params = np.random.randn(self.n_params) * 0.1
        quantum_params = classical_params.copy()

        print("Alternating Training:")

        for phase in range(n_phases):
            print(f"\n  Phase {phase + 1}:")

            # Classical phase
            print("    Classical training...")
            for _ in range(epochs_per_phase):
                for x, y in zip(x_train, y_train):
                    pred = classical_model(classical_params, x)
                    grad = self._numerical_gradient(classical_model, classical_params, x, y)
                    classical_params -= 0.1 * grad

            # Transfer to quantum
            quantum_params = self._transfer(classical_params)

            # Quantum phase
            print("    Quantum training...")
            for _ in range(epochs_per_phase):
                for x, y in zip(x_train, y_train):
                    pred = quantum_fn(quantum_params, x)
                    grad = self._numerical_gradient(quantum_fn, quantum_params, x, y)
                    quantum_params -= 0.05 * grad

            # Transfer back
            classical_params = self._transfer_back(quantum_params)

            # Report
            c_loss = np.mean([(classical_model(classical_params, x) - y)**2
                             for x, y in zip(x_train, y_train)])
            q_loss = np.mean([(quantum_fn(quantum_params, x) - y)**2
                             for x, y in zip(x_train, y_train)])
            print(f"    Classical loss: {c_loss:.4f}, Quantum loss: {q_loss:.4f}")

        return quantum_params

    def curriculum_transfer(self, x_train, y_train, classical_model,
                            quantum_fn, stages=4):
        """
        Gradually increase quantum involvement.
        """
        params = np.random.randn(self.n_params) * 0.1

        print("\nCurriculum Transfer:")

        for stage in range(stages):
            # Quantum ratio increases with stage
            q_ratio = (stage + 1) / stages

            print(f"\n  Stage {stage + 1}: q_ratio = {q_ratio:.2f}")

            for epoch in range(20):
                for x, y in zip(x_train, y_train):
                    # Blend classical and quantum predictions
                    c_pred = classical_model(params, x)
                    q_pred = quantum_fn(params, x)
                    blended = (1 - q_ratio) * c_pred + q_ratio * q_pred

                    # Compute blended gradient
                    c_grad = self._numerical_gradient(classical_model, params, x, y)
                    q_grad = self._numerical_gradient(quantum_fn, params, x, y)
                    blended_grad = (1 - q_ratio) * c_grad + q_ratio * q_grad

                    params -= 0.1 * blended_grad

            loss = np.mean([(quantum_fn(params, x) - y)**2
                           for x, y in zip(x_train, y_train)])
            print(f"    Quantum loss: {loss:.4f}")

        return params

    def _transfer(self, classical_params):
        """Transfer classical to quantum."""
        return 2 * np.arctan(classical_params)

    def _transfer_back(self, quantum_params):
        """Transfer quantum to classical."""
        return np.tan(quantum_params / 2)

    def _numerical_gradient(self, model_fn, params, x, y, eps=0.01):
        grad = np.zeros_like(params)
        pred = model_fn(params, x)

        for i in range(len(params)):
            params[i] += eps
            pred_plus = model_fn(params, x)
            params[i] -= eps
            grad[i] = 2 * (pred - y) * (pred_plus - pred) / eps

        return grad

# Demonstrate hybrid training
print("\nHybrid Training Schemes:")
```

## 6. Evaluation and Comparison

Compare initialization strategies.

```python
class InitializationEvaluator:
    """
    Evaluate different initialization strategies.
    """

    def __init__(self, quantum_model_fn):
        self.quantum_model = quantum_model_fn

    def compare_training_curves(self, x_train, y_train, init_strategies, epochs=50):
        """
        Compare training curves for different initializations.
        """
        results = {}

        for name, init_fn in init_strategies.items():
            params = init_fn()
            losses = []

            for epoch in range(epochs):
                epoch_loss = 0

                for x, y in zip(x_train, y_train):
                    pred = self.quantum_model(params, x)
                    loss = (pred - y) ** 2
                    epoch_loss += loss

                    # Update
                    grad = self._gradient(params, x, y)
                    params -= 0.1 * grad

                losses.append(epoch_loss / len(x_train))

            results[name] = {
                'final_loss': losses[-1],
                'convergence_epoch': self._find_convergence(losses),
                'loss_curve': losses
            }

        return results

    def _gradient(self, params, x, y, eps=0.01):
        grad = np.zeros_like(params)
        pred = self.quantum_model(params, x)

        for i in range(len(params)):
            params[i] += eps
            pred_plus = self.quantum_model(params, x)
            params[i] -= eps
            grad[i] = 2 * (pred - y) * (pred_plus - pred) / eps

        return grad

    def _find_convergence(self, losses, threshold=0.1):
        """Find epoch where loss first drops below threshold."""
        for i, loss in enumerate(losses):
            if loss < threshold:
                return i
        return len(losses)

    'Random': lambda: np.random.randn(4) * 0.5,
    'Small random': lambda: np.random.randn(4) * 0.01,
    'Classical-guided': lambda: np.array([1.0, 0.0, 0.5, 0.0]) + 0.01 * np.random.randn(4),
    'Identity-like': lambda: np.zeros(4) + 0.001 * np.random.randn(4)
}

# Compare

print("\nResults:")

    print(f"  {name:18}: Final loss = {result['final_loss']:.4f}, "
          f"Converged at epoch {result['convergence_epoch']}")
```
