# Multi-Task Quantum Learning

Multi-task learning trains a single model on multiple related tasks, enabling knowledge sharing. This lesson covers multi-task quantum architectures and training strategies.

## Learning Objectives

After completing this lesson, you will be able to:

- Design multi-task quantum architectures
- Share parameters across quantum tasks
- Balance task losses during training
- Implement task-specific heads
- Evaluate multi-task performance
- Leverage task relationships for transfer

## 1. Multi-Task Architecture Design

<Definition term="Multi-Task Learning">
Multi-task learning (MTL) trains a model on multiple tasks simultaneously:

min_θ Σᵢ λᵢ·Lᵢ(θ_shared, θᵢ)

where θ_shared is shared across tasks and θᵢ are task-specific parameters.
</Definition>

```python
import numpy as np

class MultiTaskQuantumModel:
    """
    Quantum model with shared encoder and task-specific heads.
    """

    def __init__(self, n_qubits, n_shared_layers, n_tasks):
        self.n_qubits = n_qubits
        self.n_shared_layers = n_shared_layers
        self.n_tasks = n_tasks

        # Shared encoder parameters
        self.shared_params = np.random.randn(n_qubits * n_shared_layers) * 0.1

        # Task-specific head parameters
        self.task_heads = {
            i: np.random.randn(n_qubits) * 0.1
            for i in range(n_tasks)
        }

    def shared_encoder(self, x):
        """
        Shared feature extraction across tasks.
        """
        result = x
        for p in self.shared_params:
            result = np.sin(p * result)
        return result

    def task_head(self, features, task_id):
        """
        Task-specific prediction head.
        """
        head_params = self.task_heads[task_id]
        result = features
        for p in head_params:
            result = np.tanh(p * result)
        return result

    def forward(self, x, task_id):
        """
        Full forward pass for a specific task.
        """
        features = self.shared_encoder(x)
        return self.task_head(features, task_id)

    def get_all_params(self):
        """Get all parameters as a single vector."""
        all_params = list(self.shared_params)
        for tid in range(self.n_tasks):
            all_params.extend(self.task_heads[tid])
        return np.array(all_params)

    def set_all_params(self, params):
        """Set all parameters from a single vector."""
        n_shared = len(self.shared_params)
        n_per_task = self.n_qubits

        self.shared_params = params[:n_shared]
        idx = n_shared
        for tid in range(self.n_tasks):
            self.task_heads[tid] = params[idx:idx + n_per_task]
            idx += n_per_task

    def describe(self):
        """Describe model architecture."""
        return {
            'n_qubits': self.n_qubits,
            'n_shared_layers': self.n_shared_layers,
            'n_tasks': self.n_tasks,
            'shared_params': len(self.shared_params),
            'task_params_each': self.n_qubits,
            'total_params': len(self.get_all_params())
        }

# Demonstrate multi-task architecture

print("=" * 50)
```

## 2. Parameter Sharing Strategies

Different approaches to sharing parameters across tasks.

```python
class ParameterSharingStrategies:
    """
    Different parameter sharing approaches for multi-task learning.
    """

    def __init__(self, n_qubits, n_layers):
        self.n_qubits = n_qubits
        self.n_layers = n_layers

    def hard_sharing(self, n_tasks):
        """
        Hard parameter sharing: all tasks share same encoder.

        Structure:
        - Shared: layers 0 to n_layers-1
        - Task-specific: one head per task
        """
        shared_params = self.n_qubits * (self.n_layers - 1)
        task_params = self.n_qubits  # One layer per task

        return {
            'strategy': 'hard_sharing',
            'shared_params': shared_params,
            'task_params': task_params * n_tasks,
            'total': shared_params + task_params * n_tasks,
            'sharing_ratio': shared_params / (shared_params + task_params * n_tasks)
        }

    def soft_sharing(self, n_tasks, alpha=0.1):
        """
        Soft parameter sharing: separate params with regularization.

        Each task has its own parameters, but they are regularized
        to be similar.
        """
        params_per_task = self.n_qubits * self.n_layers

        return {
            'strategy': 'soft_sharing',
            'params_per_task': params_per_task,
            'total': params_per_task * n_tasks,
            'regularization_strength': alpha,
            'regularization': f'||θᵢ - θⱼ||² for all task pairs'
        }

    def cross_stitch(self, n_tasks):
        """
        Cross-stitch networks: learnable combination of task features.

        Each layer has a cross-stitch unit that learns to combine
        features from different tasks.
        """
        base_params = self.n_qubits * self.n_layers * n_tasks
        cross_stitch_params = n_tasks * n_tasks * self.n_layers  # Mixing weights

        return {
            'strategy': 'cross_stitch',
            'base_params': base_params,
            'cross_stitch_params': cross_stitch_params,
            'total': base_params + cross_stitch_params,
            'description': 'Learnable linear combination of task features'
        }

    def progressive_networks(self, n_tasks):
        """
        Progressive networks: add capacity for new tasks.

        Each new task gets its own column but can use lateral
        connections to previous tasks.
        """
        # First task: full network
        # Each subsequent task: full network + lateral connections
        base_per_task = self.n_qubits * self.n_layers
        lateral_per_layer = self.n_qubits  # Connection to each previous task

        total = base_per_task  # First task
        for t in range(1, n_tasks):
            total += base_per_task + lateral_per_layer * t * self.n_layers

        return {
            'strategy': 'progressive',
            'base_per_task': base_per_task,
            'total': total,
            'description': 'New tasks can access features from previous tasks'
        }
```

## 3. Multi-Task Training

Train on multiple tasks simultaneously.

```python
class MultiTaskTrainer:
    """
    Train multi-task quantum model.
    """

    def __init__(self, model, task_weights=None):
        self.model = model
        self.n_tasks = model.n_tasks

        # Task weights for loss balancing
        if task_weights is None:
            self.task_weights = np.ones(self.n_tasks) / self.n_tasks
        else:
            self.task_weights = np.array(task_weights)
            self.task_weights /= self.task_weights.sum()

    def compute_task_loss(self, task_id, x_data, y_data):
        """Compute loss for a single task."""
        total_loss = 0
        for x, y in zip(x_data, y_data):
            pred = self.model.forward(x, task_id)
            total_loss += (pred - y) ** 2
        return total_loss / len(x_data)

    def compute_total_loss(self, task_datasets):
        """
        Compute weighted sum of task losses.

        Args:
            task_datasets: Dict mapping task_id to (x_data, y_data)
        """
        total = 0
        task_losses = {}

        for task_id, (x_data, y_data) in task_datasets.items():
            loss = self.compute_task_loss(task_id, x_data, y_data)
            task_losses[task_id] = loss
            total += self.task_weights[task_id] * loss

        return total, task_losses

    def train(self, task_datasets, epochs=100, lr=0.1):
        """
        Train on all tasks.
        """
        print(f"Multi-Task Training ({self.n_tasks} tasks)")
        print(f"Task weights: {self.task_weights}")

        all_params = self.model.get_all_params()

        for epoch in range(epochs):
            # Compute gradient over all tasks
            grad = np.zeros_like(all_params)

            for task_id, (x_data, y_data) in task_datasets.items():
                task_grad = self._compute_gradient(task_id, x_data, y_data)
                grad += self.task_weights[task_id] * task_grad

            # Update
            all_params -= lr * grad
            self.model.set_all_params(all_params)

            if epoch % 20 == 0:
                total_loss, task_losses = self.compute_total_loss(task_datasets)
                loss_str = ", ".join([f"T{t}:{l:.4f}" for t, l in task_losses.items()])
                print(f"  Epoch {epoch}: Total={total_loss:.4f} ({loss_str})")

        return self.model

    def _compute_gradient(self, task_id, x_data, y_data, eps=0.01):
        """Compute gradient for a specific task."""
        all_params = self.model.get_all_params()
        grad = np.zeros_like(all_params)

        for x, y in zip(x_data, y_data):
            pred = self.model.forward(x, task_id)

            for i in range(len(all_params)):
                all_params[i] += eps
                self.model.set_all_params(all_params)
                pred_plus = self.model.forward(x, task_id)
                all_params[i] -= eps
                self.model.set_all_params(all_params)

                grad[i] += 2 * (pred - y) * (pred_plus - pred) / eps

        return grad / len(x_data)

# Demonstrate multi-task training
print("\nMulti-Task Training:")
print("=" * 50)

np.random.seed(42)

# Create model
model = MultiTaskQuantumModel(n_qubits=3, n_shared_layers=2, n_tasks=3)

# Create task datasets
# Task 0: sin(x)
# Task 1: cos(x)
# Task 2: sin(2x)

    0: (x_data, np.sin(x_data)),
    1: (x_data, np.cos(x_data)),
    2: (x_data, np.sin(2 * x_data))

# Train

# Evaluate

    print(f"  Task {task_id}: pred={pred:.3f}, target={targets[task_id]:.3f}")
```

## 4. Loss Balancing

Balance losses across tasks with different scales.

```python
class LossBalancer:
    """
    Balance task losses during multi-task training.
    """

    def __init__(self, n_tasks):
        self.n_tasks = n_tasks
        self.loss_history = {i: [] for i in range(n_tasks)}

    def uniform_weighting(self):
        """Equal weights for all tasks."""
        return np.ones(self.n_tasks) / self.n_tasks

    def loss_scale_weighting(self, task_losses):
        """
        Weight inversely proportional to loss magnitude.

        Tasks with larger losses get smaller weights to prevent domination.
        """
        losses = np.array([task_losses.get(i, 1.0) for i in range(self.n_tasks)])
        weights = 1.0 / (losses + 1e-10)
        return weights / weights.sum()

    def gradient_normalization(self, task_grads):
        """
        Normalize gradients to have similar magnitudes.
        """
        normalized = {}
        for task_id, grad in task_grads.items():
            norm = np.linalg.norm(grad)
            if norm > 1e-10:
                normalized[task_id] = grad / norm
            else:
                normalized[task_id] = grad
        return normalized

    def uncertainty_weighting(self, log_vars):
        """
        Uncertainty weighting: learn task weights.

        Uses learned log variance for each task.
        """
        # Weights = exp(-log_var) / Σ exp(-log_var)
        weights = np.exp(-log_vars)
        return weights / weights.sum()

    def dynamic_weighting(self, task_losses, alpha=0.5):
        """
        Dynamic weight adjustment based on loss change.

        Increase weight for tasks with slower improvement.
        """
        weights = np.ones(self.n_tasks)

        for task_id in range(self.n_tasks):
            self.loss_history[task_id].append(task_losses.get(task_id, 0))

            if len(self.loss_history[task_id]) >= 2:
                recent = self.loss_history[task_id][-2:]
                improvement = recent[-2] - recent[-1]

                if improvement < 0:  # Getting worse
                    weights[task_id] *= (1 + alpha)
                elif improvement < 0.01:  # Slow improvement
                    weights[task_id] *= (1 + alpha * 0.5)

        return weights / weights.sum()

print(f"\nTask losses: {task_losses}")

print(f"   → Higher weight to smaller loss task")

    # Simulate changing losses
    task_losses = {
        0: 0.1 * (1 - i * 0.1),    # Improving
        1: 1.5 * (1 - i * 0.01),   # Slow improvement
        2: 0.5 * (1 - i * 0.05)    # Medium improvement
    }
    dynamic = balancer.dynamic_weighting(task_losses)
    print(f"   Iter {i}: weights = {dynamic}")
```

<Warning>
Imbalanced task losses can cause one task to dominate training. Use loss balancing techniques, especially when task scales differ significantly.
</Warning>

## 5. Auxiliary Tasks

```python
class AuxiliaryTaskLearning:
    """
    Use auxiliary tasks to improve main task.
    """

    def __init__(self, n_qubits, n_layers):
        self.n_qubits = n_qubits
        self.n_layers = n_layers

    def design_auxiliary_tasks(self, main_task):
        """
        Design auxiliary tasks related to main task.
        """
        auxiliaries = []

        if main_task == 'classification':
            auxiliaries = [
                {
                    'name': 'reconstruction',
                    'description': 'Reconstruct input from features',
                    'relation': 'Encourages informative representations'
                },
                {
                    'name': 'contrastive',
                    'description': 'Distinguish similar/dissimilar inputs',
                    'relation': 'Learns discriminative features'
                },
                {
                    'name': 'clustering',
                    'description': 'Predict cluster assignment',
                    'relation': 'Structures feature space'
                }
            ]
        elif main_task == 'regression':
            auxiliaries = [
                {
                    'name': 'gradient_prediction',
                    'description': 'Predict derivative of target',
                    'relation': 'Captures local structure'
                },
                {
                    'name': 'binned_classification',
                    'description': 'Classify into target bins',
                    'relation': 'Provides discrete supervision'
                }
            ]

        return auxiliaries

    def train_with_auxiliary(self, model, main_data, aux_data, aux_weight=0.2):
        """
        Train with main task + auxiliary task.
        """
        # Combine losses
        # L = L_main + aux_weight * L_aux

        print(f"Training with auxiliary task (weight={aux_weight})")
        print(f"  Main samples: {len(main_data[0])}")
        print(f"  Aux samples: {len(aux_data[0])}")

        # Simulated training improvement
        baseline_loss = 0.5
        with_aux_loss = baseline_loss * (1 - 0.15 * aux_weight)

        return {
            'baseline': baseline_loss,
            'with_auxiliary': with_aux_loss,
            'improvement': (baseline_loss - with_aux_loss) / baseline_loss * 100
        }

    def task_relationship_analysis(self, task_features):
        """
        Analyze relationships between tasks.
        """
        n_tasks = len(task_features)
        similarity_matrix = np.zeros((n_tasks, n_tasks))

        for i in range(n_tasks):
            for j in range(n_tasks):
                if i == j:
                    similarity_matrix[i, j] = 1.0
                else:
                    # Correlation of features
                    corr = np.corrcoef(task_features[i], task_features[j])[0, 1]
                    similarity_matrix[i, j] = abs(corr)

        return similarity_matrix
```

## 6. Evaluation Metrics

Evaluate multi-task model performance.

```python
class MultiTaskEvaluator:
    """
    Evaluate multi-task learning performance.
    """

    def __init__(self, model, baseline_models=None):
        self.model = model
        self.baseline_models = baseline_models or {}

    def evaluate_all_tasks(self, test_datasets):
        """
        Evaluate performance on all tasks.
        """
        results = {}

        for task_id, (x_test, y_test) in test_datasets.items():
            predictions = [self.model.forward(x, task_id) for x in x_test]
            mse = np.mean((np.array(predictions) - y_test) ** 2)

            results[task_id] = {
                'mse': mse,
                'rmse': np.sqrt(mse)
            }

        # Aggregate metrics
        avg_mse = np.mean([r['mse'] for r in results.values()])

        return {
            'per_task': results,
            'avg_mse': avg_mse,
            'avg_rmse': np.sqrt(avg_mse)
        }

    def compare_to_single_task(self, test_datasets, single_task_results):
        """
        Compare multi-task to single-task baselines.
        """
        mtl_results = self.evaluate_all_tasks(test_datasets)

        comparison = {}
        for task_id in test_datasets.keys():
            mtl_mse = mtl_results['per_task'][task_id]['mse']
            stl_mse = single_task_results.get(task_id, {}).get('mse', mtl_mse)

            improvement = (stl_mse - mtl_mse) / stl_mse * 100

            comparison[task_id] = {
                'mtl_mse': mtl_mse,
                'stl_mse': stl_mse,
                'improvement': improvement,
                'mtl_better': improvement > 0
            }

        return comparison

    def transfer_gain_analysis(self, test_datasets, isolated_model):
        """
        Analyze transfer gain from multi-task learning.
        """
        mtl_results = self.evaluate_all_tasks(test_datasets)

        gains = {}
        for task_id, (x_test, y_test) in test_datasets.items():
            # Isolated performance
            iso_preds = [isolated_model.forward(x, task_id) for x in x_test]
            iso_mse = np.mean((np.array(iso_preds) - y_test) ** 2)

            # MTL performance
            mtl_mse = mtl_results['per_task'][task_id]['mse']

            transfer_gain = (iso_mse - mtl_mse) / iso_mse * 100

            gains[task_id] = {
                'isolated_mse': iso_mse,
                'mtl_mse': mtl_mse,
                'transfer_gain': transfer_gain
            }

        # Average gain
        avg_gain = np.mean([g['transfer_gain'] for g in gains.values()])

        return {
            'per_task': gains,
            'avg_transfer_gain': avg_gain
        }
```
