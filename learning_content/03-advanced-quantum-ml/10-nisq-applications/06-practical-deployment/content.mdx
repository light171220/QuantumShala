## 2. Circuit Transpilation

Converting abstract circuits to hardware-executable form.

```python
import numpy as np

class CircuitTranspiler:
    """
    Demonstrate circuit transpilation concepts.
    """

    def __init__(self, connectivity, native_gates):
        """
        Args:
            connectivity: List of (qubit1, qubit2) pairs that can interact
            native_gates: Set of native gate names
        """
        self.connectivity = set(tuple(sorted(e)) for e in connectivity)
        self.native_gates = set(native_gates)

    def check_connectivity(self, circuit):
        """
        Check if circuit respects hardware connectivity.
        """
        violations = []
        for gate in circuit:
            if len(gate['qubits']) == 2:
                edge = tuple(sorted(gate['qubits']))
                if edge not in self.connectivity:
                    violations.append(gate)
        return violations

    def decompose_gate(self, gate_name):
        """
        Decompose gate into native gates.
        """
        decompositions = {
            'CNOT': ['RZ', 'SX', 'CZ', 'SX', 'RZ'],
            'RY': ['RZ', 'SX', 'RZ'],
            'RX': ['RZ', 'SX', 'RZ'],
            'SWAP': ['CZ', 'SX', 'CZ', 'SX', 'CZ'],
            'H': ['RZ', 'SX', 'RZ']
        }
        return decompositions.get(gate_name, [gate_name])

    def estimate_swap_overhead(self, qubit1, qubit2, n_qubits):
        """
        Estimate SWAPs needed to connect non-adjacent qubits.
        """
        # Simplified: assume linear connectivity for estimation
        distance = abs(qubit1 - qubit2)
        if distance == 1:
            return 0
        return distance - 1  # SWAPs needed

    def analyze_circuit(self, circuit):
        """
        Analyze circuit for transpilation impact.
        """
        stats = {
            'original_gates': len(circuit),
            'native_gates': 0,
            'non_native_gates': 0,
            'connectivity_violations': 0,
            'estimated_swaps': 0
        }

        for gate in circuit:
            if gate['name'] in self.native_gates:
                stats['native_gates'] += 1
            else:
                stats['non_native_gates'] += 1
                stats['native_gates'] += len(self.decompose_gate(gate['name']))

            if len(gate['qubits']) == 2:
                edge = tuple(sorted(gate['qubits']))
                if edge not in self.connectivity:
                    stats['connectivity_violations'] += 1
                    stats['estimated_swaps'] += self.estimate_swap_overhead(
                        gate['qubits'][0], gate['qubits'][1], 5
                    )

        return stats

# Demonstration
print("Circuit Transpilation Analysis:")
print("=" * 50)

# Heavy-hex-like connectivity (subset)
connectivity = [(0, 1), (1, 2), (2, 3), (3, 4), (1, 5), (3, 6)]
native_gates = {'RZ', 'SX', 'CZ', 'X'}

transpiler = CircuitTranspiler(connectivity, native_gates)

# Sample QML circuit
circuit = [
    {'name': 'RY', 'qubits': [0]},
    {'name': 'RY', 'qubits': [1]},
    {'name': 'RY', 'qubits': [2]},
    {'name': 'CNOT', 'qubits': [0, 1]},
    {'name': 'CNOT', 'qubits': [1, 2]},
    {'name': 'CNOT', 'qubits': [0, 2]},  # Non-adjacent
    {'name': 'RY', 'qubits': [0]},
    {'name': 'RY', 'qubits': [1]},
    {'name': 'RY', 'qubits': [2]},
]

print(f"\nConnectivity: {sorted(connectivity)}")
print(f"Native gates: {sorted(native_gates)}")

stats = transpiler.analyze_circuit(circuit)

print("\n\nCircuit Analysis:")
print("-" * 40)
print(f"  Original gates: {stats['original_gates']}")
print(f"  Native gates needed: {stats['native_gates']}")
print(f"  Connectivity violations: {stats['connectivity_violations']}")
print(f"  Estimated SWAPs needed: {stats['estimated_swaps']}")

# Gate count explosion
print("\n\nGate Count After Transpilation:")
print("-" * 40)

decomposed_count = 0
for gate in circuit:
    decomposed = transpiler.decompose_gate(gate['name'])
    decomposed_count += len(decomposed)
    if gate['name'] not in native_gates:
        print(f"  {gate['name']} → {' + '.join(decomposed)}")

swap_gates = stats['estimated_swaps'] * 3  # Each SWAP = 3 CZ
total_gates = decomposed_count + swap_gates
print(f"\n  Decomposed gates: {decomposed_count}")
print(f"  SWAP overhead: {swap_gates} gates")
print(f"  Total estimated: {total_gates} gates")
```

<Warning>
Transpilation can dramatically increase circuit depth. A 10-gate abstract circuit might become 50+ native gates after decomposition and SWAP insertion. Always check transpiled circuit depth before running.
</Warning>

## 3. Job Management and Queuing

Handling the realities of cloud quantum access.

```python
import numpy as np
import time

class QuantumJobManager:
    """
    Manage quantum job submission and monitoring.
    """

    def __init__(self, max_shots=100000, max_circuits_per_job=300):
        self.max_shots = max_shots
        self.max_circuits_per_job = max_circuits_per_job
        self.jobs = []

    def estimate_queue_time(self, backend_queue, priority='standard'):
        """
        Estimate time to execution based on queue.
        """
        priority_multipliers = {
            'premium': 0.1,
            'standard': 1.0,
            'fairshare': 2.0
        }

        base_time = backend_queue * 5  # 5 min per job in queue
        return base_time * priority_multipliers.get(priority, 1.0)

    def batch_circuits(self, circuits, shots_per_circuit):
        """
        Batch circuits into jobs respecting limits.
        """
        batches = []
        current_batch = []
        current_shots = 0

        for circuit in circuits:
            if (len(current_batch) >= self.max_circuits_per_job or
                current_shots + shots_per_circuit > self.max_shots):
                if current_batch:
                    batches.append(current_batch)
                current_batch = []
                current_shots = 0

            current_batch.append(circuit)
            current_shots += shots_per_circuit

        if current_batch:
            batches.append(current_batch)

        return batches

    def optimize_execution(self, n_circuits, shots, backend_info):
        """
        Suggest optimal execution strategy.
        """
        recommendations = []

        # Shot optimization
        if shots > 10000:
            recommendations.append({
                'type': 'shot_reduction',
                'suggestion': 'Consider starting with fewer shots',
                'reason': 'Many VQE iterations benefit from lower shots initially'
            })

        # Batching
        n_jobs = np.ceil(n_circuits / self.max_circuits_per_job)
        if n_jobs > 5:
            recommendations.append({
                'type': 'reduce_circuits',
                'suggestion': f'Consider reducing to {self.max_circuits_per_job * 5} circuits',
                'reason': f'Currently requires {int(n_jobs)} jobs'
            })

        # Queue awareness
        if backend_info.get('queue_length', 0) > 20:
            recommendations.append({
                'type': 'backend_selection',
                'suggestion': 'Consider alternative backend',
                'reason': f"Queue has {backend_info['queue_length']} jobs"
            })

        return recommendations

# Demonstration
```

## 4. Hybrid Workflow Design

Structuring classical-quantum workflows for production.

```python
import numpy as np

class HybridWorkflow:
    """
    Design and manage hybrid classical-quantum workflows.
    """

    def __init__(self, classical_budget_ratio=0.8):
        """
        Args:
            classical_budget_ratio: Fraction of work to do classically
        """
        self.classical_ratio = classical_budget_ratio
        self.workflow_steps = []

    def add_classical_preprocessing(self, name, compute_cost):
        self.workflow_steps.append({
            'type': 'classical',
            'name': name,
            'cost': compute_cost,
            'stage': 'preprocessing'
        })

    def add_quantum_step(self, name, circuits, shots):
        self.workflow_steps.append({
            'type': 'quantum',
            'name': name,
            'circuits': circuits,
            'shots': shots,
            'stage': 'execution'
        })

    def add_classical_postprocessing(self, name, compute_cost):
        self.workflow_steps.append({
            'type': 'classical',
            'name': name,
            'cost': compute_cost,
            'stage': 'postprocessing'
        })

    def analyze_workflow(self):
        """
        Analyze workflow efficiency.
        """
        classical_cost = sum(s['cost'] for s in self.workflow_steps
                           if s['type'] == 'classical')
        quantum_circuits = sum(s['circuits'] for s in self.workflow_steps
                             if s['type'] == 'quantum')
        quantum_shots = sum(s['circuits'] * s['shots'] for s in self.workflow_steps
                          if s['type'] == 'quantum')

        return {
            'classical_compute': classical_cost,
            'quantum_circuits': quantum_circuits,
            'quantum_shots': quantum_shots,
            'n_steps': len(self.workflow_steps)
        }

# Demonstration
print("Hybrid Workflow Design:")
print("=" * 50)

# Example: VQE workflow
workflow = HybridWorkflow()

# Preprocessing
workflow.add_classical_preprocessing('molecule_geometry', 10)
workflow.add_classical_preprocessing('hamiltonian_construction', 50)
workflow.add_classical_preprocessing('qubit_mapping', 20)

# Quantum execution (multiple iterations)
for i in range(5):  # Simplified: 5 optimization iterations
    workflow.add_quantum_step(f'vqe_iteration_{i}', circuits=20, shots=4096)
    workflow.add_classical_postprocessing(f'gradient_estimation_{i}', 5)
    workflow.add_classical_postprocessing(f'parameter_update_{i}', 2)

# Final analysis
workflow.add_classical_postprocessing('result_analysis', 30)

analysis = workflow.analyze_workflow()

print("\nVQE Workflow Analysis:")
print("-" * 40)
print(f"  Workflow steps: {analysis['n_steps']}")
print(f"  Classical compute units: {analysis['classical_compute']}")
print(f"  Quantum circuits: {analysis['quantum_circuits']}")
print(f"  Total quantum shots: {analysis['quantum_shots']:,}")

# Best practices
print("\n\nHybrid Workflow Best Practices:")
print("-" * 40)

practices = [
    {
        'practice': 'Maximize classical preprocessing',
        'benefit': 'Reduce quantum circuit complexity'
    },
    {
        'practice': 'Batch quantum jobs',
        'benefit': 'Minimize queue overhead'
    },
    {
        'practice': 'Implement caching',
        'benefit': 'Avoid redundant quantum calls'
    },
    {
        'practice': 'Use asynchronous execution',
        'benefit': 'Overlap classical and quantum work'
    },
    {
        'practice': 'Progressive refinement',
        'benefit': 'Start with low shots, increase as needed'
    },
    {
        'practice': 'Checkpoint regularly',
        'benefit': 'Recover from failures'
    }
]

for p in practices:
    print(f"\n  {p['practice']}")
    print(f"    → {p['benefit']}")
```

<Tip>
Structure workflows so that quantum device time is minimized. Do as much classical pre-processing and post-processing as possible to reduce the number of quantum circuit executions.
</Tip>

## Summary

Deploying QML on real hardware requires careful attention to practical details:

1. **Hardware Selection**: Match hardware capabilities to application requirements
2. **Transpilation**: Understand gate decomposition and routing overhead
3. **Job Management**: Handle queuing, batching, and timeouts
4. **Hybrid Workflows**: Minimize quantum device time
5. **Cost Optimization**: Use shots efficiently, consider alternatives
6. **Production Readiness**: Follow comprehensive checklists

The gap between simulation and real hardware is significant—plan accordingly.
