# Federated Learning Basics

Federated learning enables collaborative training across distributed devices while keeping data local. This lesson introduces federated learning concepts and their application to quantum ML.

## 1. What is Federated Learning?

<Definition term="Federated Learning">
Federated Learning (FL) trains a shared model across multiple decentralized nodes that hold local data:

1. Server distributes model to clients
2. Clients train on local data
3. Clients send model updates (not data) to server
4. Server aggregates updates into global model
5. Repeat until convergence
</Definition>

## 2. Federated Averaging (FedAvg)

The standard aggregation algorithm for federated learning.

<Formula>
FedAvg aggregates client models weighted by dataset size:

w_global = Σᵢ (nᵢ/n) · wᵢ

where nᵢ is client i's data size, n = Σnᵢ, and wᵢ are local parameters.
</Formula>

```python
class FederatedAveraging:
    """
    Implement Federated Averaging (FedAvg) algorithm.
    """

    def __init__(self, n_clients, n_params):
        self.n_clients = n_clients
        self.n_params = n_params
        self.global_params = np.random.randn(n_params) * 0.1

    def initialize_clients(self, data_sizes):
        """Initialize clients with data sizes."""
        self.data_sizes = data_sizes
        self.total_data = sum(data_sizes)
        self.client_params = [self.global_params.copy() for _ in range(self.n_clients)]

    def distribute_model(self):
        """Send global model to all clients."""
        for i in range(self.n_clients):
            self.client_params[i] = self.global_params.copy()
        return self.client_params

    def client_update(self, client_id, local_data, local_labels, epochs=5, lr=0.1):
        """
        Train client model on local data.
        """
        params = self.client_params[client_id].copy()

        for _ in range(epochs):
            for x, y in zip(local_data, local_labels):
                pred = self._forward(params, x)
                grad = self._gradient(params, x, y)
                params -= lr * grad

        return params

    def _forward(self, params, x):
        result = x
        for p in params:
            result = np.tanh(p * result)
        return result

    def _gradient(self, params, x, y, eps=0.01):
        grad = np.zeros_like(params)
        pred = self._forward(params, x)

        for i in range(len(params)):
            params[i] += eps
            pred_plus = self._forward(params, x)
            params[i] -= eps
            grad[i] = 2 * (pred - y) * (pred_plus - pred) / eps

        return grad

    def aggregate(self, client_updates):
        """
        FedAvg: Weighted average of client parameters.
        """
        new_params = np.zeros(self.n_params)

        for i, (params, data_size) in enumerate(zip(client_updates, self.data_sizes)):
            weight = data_size / self.total_data
            new_params += weight * params

        self.global_params = new_params
        return new_params

    def federated_round(self, client_data, client_labels, local_epochs=5, lr=0.1):
        """
        Complete one round of federated learning.
        """
        # Distribute model
        self.distribute_model()

        # Client updates
        client_updates = []
        for i in range(self.n_clients):
            updated_params = self.client_update(
                i, client_data[i], client_labels[i],
                epochs=local_epochs, lr=lr
            )
            client_updates.append(updated_params)

        # Aggregate
        self.aggregate(client_updates)

        return self.global_params

# Demonstrate FedAvg
print("\nFederated Averaging (FedAvg):")
print("=" * 50)

np.random.seed(42)

# Setup
n_clients = 5
n_params = 4

fed_avg = FederatedAveraging(n_clients=n_clients, n_params=n_params)

# Create client data (non-IID: different distributions)
client_data = []
client_labels = []
data_sizes = []

for i in range(n_clients):
    n_samples = np.random.randint(10, 30)
    # Each client has slightly different data distribution
    x = np.random.randn(n_samples) + 0.5 * i
    y = np.sin(x)
    client_data.append(x)
    client_labels.append(y)
    data_sizes.append(n_samples)

fed_avg.initialize_clients(data_sizes)

print(f"\nClients: {n_clients}")
print(f"Data sizes: {data_sizes}")
print(f"Initial global params: {fed_avg.global_params}")

# Run federated rounds
print("\nFederated training:")
for round_num in range(5):
    fed_avg.federated_round(client_data, client_labels, local_epochs=3, lr=0.1)

    # Evaluate global model
    total_loss = 0
    for i in range(n_clients):
        for x, y in zip(client_data[i], client_labels[i]):
            pred = fed_avg._forward(fed_avg.global_params, x)
            total_loss += (pred - y) ** 2

    avg_loss = total_loss / sum(data_sizes)
    print(f"  Round {round_num + 1}: Global loss = {avg_loss:.4f}")
```

<Tip>
FedAvg works best when client data is similarly distributed (IID). For non-IID data, consider more sophisticated aggregation methods.
</Tip>

## 3. Non-IID Data Challenges

Handle heterogeneous data distributions across clients.

```python
class NonIIDAnalysis:
    """
    Analyze and handle non-IID data in federated learning.
    """

    def __init__(self):
        pass

    def create_iid_data(self, n_clients, samples_per_client):
        """Create IID data across clients."""
        client_data = []
        for _ in range(n_clients):
            x = np.random.randn(samples_per_client)
            y = np.sin(x)
            client_data.append((x, y))
        return client_data

    def create_non_iid_data(self, n_clients, samples_per_client, shift_scale=2.0):
        """Create non-IID data with distribution shifts."""
        client_data = []
        for i in range(n_clients):
            # Each client has shifted distribution
            x = np.random.randn(samples_per_client) + shift_scale * i
            y = np.sin(x)
            client_data.append((x, y))
        return client_data

    def measure_heterogeneity(self, client_data):
        """
        Measure data heterogeneity across clients.
        """
        means = [np.mean(data[0]) for data in client_data]
        stds = [np.std(data[0]) for data in client_data]

        # Variance of means (measures distribution shift)
        mean_variance = np.var(means)

        # Average std (measures within-client variance)
        avg_std = np.mean(stds)

        # Heterogeneity score
        heterogeneity = mean_variance / (avg_std ** 2 + 1e-10)

        return {
            'client_means': means,
            'mean_variance': mean_variance,
            'avg_within_std': avg_std,
            'heterogeneity_score': heterogeneity,
            'level': 'high' if heterogeneity > 1 else 'moderate' if heterogeneity > 0.1 else 'low'
        }

    def mitigation_strategies(self):
        """Strategies for handling non-IID data."""
        return [
            {
                'name': 'Data sharing',
                'description': 'Share small subset of data to balance distributions',
                'trade_off': 'Reduces privacy'
            },
            {
                'name': 'FedProx',
                'description': 'Add proximal term to keep local updates close to global',
                'trade_off': 'May slow convergence'
            },
            {
                'name': 'Client clustering',
                'description': 'Group similar clients, train separate models',
                'trade_off': 'Multiple models to maintain'
            },
            {
                'name': 'Personalization',
                'description': 'Allow local adaptation after global training',
                'trade_off': 'More complex architecture'
            }
        ]
```

## 4. Quantum Federated Learning Setup

Apply federated learning to quantum settings.

```python
class QuantumFederatedSetup:
    """
    Setup for quantum federated learning.
    """

    def __init__(self, n_clients, n_qubits, n_layers):
        self.n_clients = n_clients
        self.n_qubits = n_qubits
        self.n_layers = n_layers
        self.n_params = n_qubits * n_layers * 2

        # Global quantum model parameters
        self.global_params = np.random.randn(self.n_params) * 0.1

        # Client-specific noise models (different quantum devices)
        self.client_noise = [np.random.uniform(0.01, 0.1) for _ in range(n_clients)]

    def quantum_circuit(self, params, x, noise_level=0.0):
        """
        Simulated variational quantum circuit with noise.
        """
        # Encode input
        result = x

        # Apply variational layers
        for i in range(0, len(params), 2):
            theta, phi = params[i], params[i+1]
            result = np.sin(theta * result + phi)

            # Add noise
            result += noise_level * np.random.randn()

        return result

    def client_train(self, client_id, local_x, local_y, epochs=5, lr=0.05):
        """
        Train quantum model on client's local data.
        """
        params = self.global_params.copy()
        noise = self.client_noise[client_id]

        for _ in range(epochs):
            for x, y in zip(local_x, local_y):
                # Forward with client's noise
                pred = self.quantum_circuit(params, x, noise)

                # Gradient (noisy)
                grad = self._noisy_gradient(params, x, y, noise)
                params -= lr * grad

        return params

    def _noisy_gradient(self, params, x, y, noise, eps=0.01):
        """Compute gradient with noise."""
        grad = np.zeros_like(params)
        pred = self.quantum_circuit(params, x, noise)

        for i in range(len(params)):
            params[i] += eps
            pred_plus = self.quantum_circuit(params, x, noise)
            params[i] -= eps
            grad[i] = 2 * (pred - y) * (pred_plus - pred) / eps

        return grad

    def federated_aggregation(self, client_params, weights=None):
        """
        Aggregate quantum parameters from clients.
        """
        if weights is None:
            weights = np.ones(self.n_clients) / self.n_clients

        new_params = np.zeros(self.n_params)
        for i, params in enumerate(client_params):
            new_params += weights[i] * params

        self.global_params = new_params
        return new_params

    def describe_setup(self):
        """Describe the quantum federated setup."""
        return {
            'n_clients': self.n_clients,
            'n_qubits': self.n_qubits,
            'n_layers': self.n_layers,
            'params_per_client': self.n_params,
            'client_noise_levels': self.client_noise
        }

# Demonstrate quantum federated setup
print("\nQuantum Federated Learning Setup:")
print("=" * 50)

np.random.seed(42)

qfl = QuantumFederatedSetup(n_clients=4, n_qubits=3, n_layers=2)

print("\nSetup:")
setup = qfl.describe_setup()
for key, value in setup.items():
    print(f"  {key}: {value}")

# Create local data for each client
client_data = []
for i in range(4):
    n_samples = 15
    x = np.linspace(-2 + i * 0.5, 2 + i * 0.5, n_samples)
    y = np.sin(x)
    client_data.append((x, y))

# Run quantum federated round
print("\nQuantum Federated Training:")
for round_num in range(3):
    # Client training
    client_updates = []
    for i in range(4):
        updated = qfl.client_train(i, client_data[i][0], client_data[i][1])
        client_updates.append(updated)

    # Aggregate
    qfl.federated_aggregation(client_updates)

    # Evaluate
    total_loss = 0
    for i in range(4):
        for x, y in zip(client_data[i][0], client_data[i][1]):
            pred = qfl.quantum_circuit(qfl.global_params, x, noise_level=0)
            total_loss += (pred - y) ** 2

    avg_loss = total_loss / (4 * 15)
    print(f"  Round {round_num + 1}: Global loss = {avg_loss:.4f}")
```

## 5. Communication Considerations

Reduce communication overhead in federated learning.

```python
class CommunicationEfficiency:
    """
    Techniques for reducing communication in federated learning.
    """

    def __init__(self, n_params):
        self.n_params = n_params

    def full_precision(self, params):
        """Full precision: 64 bits per parameter."""
        return {
            'params': params,
            'bits_per_param': 64,
            'total_bits': 64 * len(params),
            'method': 'full_precision'
        }

    def quantization(self, params, bits=8):
        """Quantize parameters to fewer bits."""
        # Scale to int range
        min_val, max_val = params.min(), params.max()
        scale = (2 ** bits - 1) / (max_val - min_val + 1e-10)

        quantized = np.round((params - min_val) * scale).astype(int)
        # Dequantize
        dequantized = quantized / scale + min_val

        return {
            'quantized': quantized,
            'dequantized': dequantized,
            'bits_per_param': bits,
            'total_bits': bits * len(params) + 64 * 2,  # Plus scale info
            'error': np.mean((params - dequantized) ** 2),
            'method': 'quantization'
        }

    def sparsification(self, gradient, top_k_percent=10):
        """Send only top-k% of gradient values."""
        k = max(1, int(len(gradient) * top_k_percent / 100))

        indices = np.argsort(np.abs(gradient))[-k:]
        values = gradient[indices]

        # Sparse representation
        sparse_grad = np.zeros_like(gradient)
        sparse_grad[indices] = values

        return {
            'indices': indices,
            'values': values,
            'k': k,
            'sparsity': 1 - k / len(gradient),
            'bits_transmitted': k * (64 + 32),  # Value + index
            'method': 'sparsification'
        }

    def gradient_compression(self, gradient, bits=8, top_k_percent=20):
        """Combined: sparsify then quantize."""
        # First sparsify
        sparse = self.sparsification(gradient, top_k_percent)

        # Then quantize the values
        if len(sparse['values']) > 0:
            quant = self.quantization(sparse['values'], bits)
            total_bits = len(sparse['values']) * (bits + 32)
        else:
            total_bits = 0

        return {
            'method': 'combined',
            'total_bits': total_bits,
            'compression_ratio': (64 * len(gradient)) / max(1, total_bits)
        }

# Demonstrate communication efficiency
print("\nCommunication Efficiency Techniques:")
print("=" * 50)

params = np.random.randn(n_params)

print(f"\n1. Full Precision:")

# Quantization

print(f"\n2. Quantization (8-bit):")
print(f"   Bits: {quant['total_bits']}")
print(f"   Compression: {full['total_bits'] / quant['total_bits']:.1f}x")
print(f"   Quantization error: {quant['error']:.6f}")

# Sparsification

print(f"\n3. Top-10% Sparsification:")
print(f"   Values transmitted: {sparse['k']}/{n_params}")
print(f"   Sparsity: {sparse['sparsity']:.1%}")

# Combined

print(f"\n4. Combined (top-20% + 8-bit):")
```
