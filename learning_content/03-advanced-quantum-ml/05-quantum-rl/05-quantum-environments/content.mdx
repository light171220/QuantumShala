# Quantum RL Environments

Quantum reinforcement learning can be applied to quantum control tasks where the environment itself is quantum. This lesson explores how to design environments for quantum state preparation, gate synthesis, and other quantum control problems.

## Learning Objectives

After completing this lesson, you will be able to:

- Design quantum control environments
- Implement quantum state preparation tasks
- Create quantum gate synthesis environments
- Handle quantum measurement and observation
- Build hybrid quantum-classical simulators
- Benchmark quantum RL environments

## 1. Quantum Environment Fundamentals

A quantum RL environment models the dynamics of a quantum system that an agent learns to control.

<Definition term="Quantum RL Environment">
An environment where the state is a quantum state, actions correspond to quantum operations (gates, pulses), and rewards measure how well a quantum objective is achieved.
</Definition>

### Environment Framework

```python
import numpy as np

class QuantumEnvironment:
    """
    Base class for quantum RL environments.

    Follows OpenAI Gym-like interface adapted for quantum systems.
    """

    def __init__(self, n_qubits, max_steps=50):
        self.n_qubits = n_qubits
        self.dim = 2 ** n_qubits
        self.max_steps = max_steps

        # State space: quantum state vector
        self.state = None
        self.step_count = 0

        # Target (for goal-based tasks)
        self.target_state = None

        # Available actions (quantum gates)
        self.action_space = self._build_action_space()
        self.n_actions = len(self.action_space)

    def _build_action_space(self):
        """Define available quantum gates as actions."""
        actions = []

        # Single-qubit gates for each qubit
        for qubit in range(self.n_qubits):
            actions.append(('H', qubit))      # Hadamard
            actions.append(('X', qubit))      # Pauli-X
            actions.append(('Z', qubit))      # Pauli-Z
            actions.append(('T', qubit))      # T gate
            actions.append(('RX', qubit))     # RX(pi/8)
            actions.append(('RY', qubit))     # RY(pi/8)

        # Two-qubit gates
        for i in range(self.n_qubits - 1):
            actions.append(('CNOT', i, i+1))
            actions.append(('CZ', i, i+1))

        return actions

    def reset(self, target_state=None):
        """Reset environment to initial state."""
        # Initialize to |0...0⟩
        self.state = np.zeros(self.dim, dtype=complex)
        self.state[0] = 1.0

        # Set target state
        if target_state is not None:
            self.target_state = target_state
        else:
            self.target_state = self._generate_random_target()

        self.step_count = 0
        return self._get_observation()

    def _generate_random_target(self):
        """Generate a random target state."""
        target = np.random.randn(self.dim) + 1j * np.random.randn(self.dim)
        return target / np.linalg.norm(target)

    def _get_observation(self):
        """Convert quantum state to observation for classical agent."""
        # Return real and imaginary parts
        obs = np.concatenate([self.state.real, self.state.imag])
        return obs

    def step(self, action_idx):
        """Execute action and return (observation, reward, done, info)."""
        action = self.action_space[action_idx]

        # Apply quantum gate
        self.state = self._apply_gate(action)

        # Compute reward
        reward = self._compute_reward()

        self.step_count += 1
        done = self.step_count >= self.max_steps or self._check_success()

        info = {
            'fidelity': self._compute_fidelity(),
            'steps': self.step_count
        }

        return self._get_observation(), reward, done, info

    def _apply_gate(self, action):
        """Apply a quantum gate to the state."""
        gate_name = action[0]

        if gate_name == 'H':
            return self._apply_hadamard(action[1])
        elif gate_name == 'X':
            return self._apply_pauli_x(action[1])
        elif gate_name == 'Z':
            return self._apply_pauli_z(action[1])
        elif gate_name == 'T':
            return self._apply_t_gate(action[1])
        elif gate_name == 'RX':
            return self._apply_rx(action[1], np.pi/8)
        elif gate_name == 'RY':
            return self._apply_ry(action[1], np.pi/8)
        elif gate_name == 'CNOT':
            return self._apply_cnot(action[1], action[2])
        elif gate_name == 'CZ':
            return self._apply_cz(action[1], action[2])

        return self.state

    def _apply_hadamard(self, qubit):
        """Apply Hadamard gate."""
        H = np.array([[1, 1], [1, -1]]) / np.sqrt(2)
        return self._apply_single_qubit_gate(H, qubit)

    def _apply_pauli_x(self, qubit):
        """Apply Pauli-X gate."""
        X = np.array([[0, 1], [1, 0]])
        return self._apply_single_qubit_gate(X, qubit)

    def _apply_pauli_z(self, qubit):
        """Apply Pauli-Z gate."""
        Z = np.array([[1, 0], [0, -1]])
        return self._apply_single_qubit_gate(Z, qubit)

    def _apply_t_gate(self, qubit):
        """Apply T gate."""
        T = np.array([[1, 0], [0, np.exp(1j * np.pi / 4)]])
        return self._apply_single_qubit_gate(T, qubit)

    def _apply_rx(self, qubit, angle):
        """Apply RX rotation."""
        c, s = np.cos(angle/2), np.sin(angle/2)
        RX = np.array([[c, -1j*s], [-1j*s, c]])
        return self._apply_single_qubit_gate(RX, qubit)

    def _apply_ry(self, qubit, angle):
        """Apply RY rotation."""
        c, s = np.cos(angle/2), np.sin(angle/2)
        RY = np.array([[c, -s], [s, c]])
        return self._apply_single_qubit_gate(RY, qubit)

    def _apply_single_qubit_gate(self, gate, qubit):
        """Apply single-qubit gate to the state."""
        new_state = np.zeros(self.dim, dtype=complex)

        for i in range(self.dim):
            bit = (i >> qubit) & 1
            i_flipped = i ^ (1 << qubit)

            if bit == 0:
                new_state[i] += gate[0, 0] * self.state[i]
                new_state[i_flipped] += gate[1, 0] * self.state[i]
            else:
                new_state[i] += gate[1, 1] * self.state[i]
                new_state[i_flipped] += gate[0, 1] * self.state[i]

        return new_state

    def _apply_cnot(self, control, target):
        """Apply CNOT gate."""
        new_state = np.zeros(self.dim, dtype=complex)

        for i in range(self.dim):
            if (i >> control) & 1:
                new_state[i ^ (1 << target)] = self.state[i]
            else:
                new_state[i] = self.state[i]

        return new_state

    def _apply_cz(self, control, target):
        """Apply CZ gate."""
        new_state = self.state.copy()

        for i in range(self.dim):
            if ((i >> control) & 1) and ((i >> target) & 1):
                new_state[i] *= -1

        return new_state

    def _compute_fidelity(self):
        """Compute fidelity with target state."""
        overlap = np.abs(np.vdot(self.target_state, self.state)) ** 2
        return overlap

    def _compute_reward(self):
        """Compute reward based on fidelity improvement."""
        fidelity = self._compute_fidelity()
        return fidelity  # Simple reward: current fidelity

    def _check_success(self, threshold=0.99):
        """Check if target state is reached."""
        return self._compute_fidelity() >= threshold

# Example usage
env = QuantumEnvironment(n_qubits=2, max_steps=20)
obs = env.reset()
print(f"Observation shape: {obs.shape}")
print(f"Number of actions: {env.n_actions}")
print(f"Initial fidelity: {env._compute_fidelity():.4f}")

# Take random actions
for _ in range(5):
    action = np.random.randint(env.n_actions)
    obs, reward, done, info = env.step(action)
    print(f"Action: {env.action_space[action]}, Fidelity: {info['fidelity']:.4f}")
```

<Tip>
Design your observation space to capture all relevant information while being compatible with your RL agent. For quantum states, combining real and imaginary parts works well for classical agents.
</Tip>

## 2. State Preparation Environments

State preparation is a fundamental quantum control task where the goal is to prepare a specific quantum state.

<Formula>
State Preparation Objective:
$$\max_{\{U_t\}} F(|\psi_\text{final}\rangle, |\psi_\text{target}\rangle)$$

where $F = |\langle\psi_\text{target}|\psi_\text{final}\rangle|^2$ is the fidelity.
</Formula>

### Implementation

```python
class StatePreparationEnv(QuantumEnvironment):
    """
    Environment for quantum state preparation tasks.
    """

    def __init__(self, n_qubits=2, target_type='bell', max_steps=30):
        super().__init__(n_qubits, max_steps)
        self.target_type = target_type

        # Define common target states
        self.target_states = {
            'bell': self._create_bell_state(),
            'ghz': self._create_ghz_state(),
            'w': self._create_w_state(),
            'random': None  # Generated on reset
        }

        # Reward shaping parameters
        self.fidelity_threshold = 0.99
        self.step_penalty = -0.01

    def _create_bell_state(self):
        """Create Bell state |Φ+⟩ = (|00⟩ + |11⟩)/√2."""
        state = np.zeros(self.dim, dtype=complex)
        state[0] = 1 / np.sqrt(2)  # |00⟩
        state[self.dim - 1] = 1 / np.sqrt(2)  # |11...1⟩
        return state

    def _create_ghz_state(self):
        """Create GHZ state (|00...0⟩ + |11...1⟩)/√2."""
        state = np.zeros(self.dim, dtype=complex)
        state[0] = 1 / np.sqrt(2)
        state[self.dim - 1] = 1 / np.sqrt(2)
        return state

    def _create_w_state(self):
        """Create W state (|100..0⟩ + |010..0⟩ + ... + |00..01⟩)/√n."""
        state = np.zeros(self.dim, dtype=complex)
        for i in range(self.n_qubits):
            idx = 1 << i
            state[idx] = 1 / np.sqrt(self.n_qubits)
        return state

    def reset(self, target_type=None):
        """Reset with specific target type."""
        self.state = np.zeros(self.dim, dtype=complex)
        self.state[0] = 1.0

        if target_type:
            self.target_type = target_type

        if self.target_type == 'random':
            self.target_state = self._generate_random_target()
        else:
            self.target_state = self.target_states.get(
                self.target_type, self._create_bell_state()
            )

        self.step_count = 0
        self.prev_fidelity = self._compute_fidelity()

        return self._get_observation()

    def _compute_reward(self):
        """Shaped reward for state preparation."""
        fidelity = self._compute_fidelity()

        # Reward for fidelity improvement
        fidelity_delta = fidelity - self.prev_fidelity
        self.prev_fidelity = fidelity

        # Base reward: current fidelity
        reward = fidelity

        # Bonus for improvement
        reward += fidelity_delta * 10

        # Small step penalty to encourage efficiency
        reward += self.step_penalty

        # Large bonus for success
        if fidelity >= self.fidelity_threshold:
            reward += 10.0

        return reward

    def get_optimal_circuit(self):
        """Return optimal circuit for known target states."""
        if self.target_type == 'bell':
            return [('H', 0), ('CNOT', 0, 1)]
        elif self.target_type == 'ghz':
            circuit = [('H', 0)]
            for i in range(self.n_qubits - 1):
                circuit.append(('CNOT', i, i+1))
            return circuit
        return None

# Example: Training on Bell state preparation
env = StatePreparationEnv(n_qubits=2, target_type='bell', max_steps=20)
obs = env.reset()

print("Target: Bell state (|00⟩ + |11⟩)/√2")
print(f"Optimal circuit: {env.get_optimal_circuit()}")
print(f"Initial fidelity: {env._compute_fidelity():.4f}")

# Execute optimal circuit
optimal = env.get_optimal_circuit()
for gate in optimal:
    action_idx = env.action_space.index(gate)
    obs, reward, done, info = env.step(action_idx)
    print(f"After {gate}: Fidelity = {info['fidelity']:.4f}")
```

<Warning>
Reward shaping can accelerate learning but may also introduce local optima. Balance immediate rewards with the final objective to avoid suboptimal policies.
</Warning>

## 3. Gate Synthesis Environments

Gate synthesis involves finding a sequence of available gates to approximate a target unitary transformation.

<Definition term="Gate Synthesis">
The task of decomposing a target unitary operation $U_\text{target}$ into a sequence of gates from a universal gate set, minimizing circuit depth while maintaining accuracy.
</Definition>

### Implementation

```python
class GateSynthesisEnv:
    """
    Environment for quantum gate synthesis.

    Goal: Learn to implement target unitary using available gates.
    """

    def __init__(self, n_qubits=1, max_gates=20):
        self.n_qubits = n_qubits
        self.dim = 2 ** n_qubits
        self.max_gates = max_gates

        # Current accumulated unitary
        self.current_unitary = None

        # Target unitary
        self.target_unitary = None

        # Available gates
        self.gate_set = self._build_gate_set()
        self.n_actions = len(self.gate_set)

        self.step_count = 0

    def _build_gate_set(self):
        """Build universal gate set."""
        gates = {}

        # Single-qubit gates for each qubit
        H = np.array([[1, 1], [1, -1]]) / np.sqrt(2)
        T = np.array([[1, 0], [0, np.exp(1j * np.pi / 4)]])
        S = np.array([[1, 0], [0, 1j]])
        X = np.array([[0, 1], [1, 0]])

        for q in range(self.n_qubits):
            gates[f'H_{q}'] = self._embed_single_gate(H, q)
            gates[f'T_{q}'] = self._embed_single_gate(T, q)
            gates[f'Tdag_{q}'] = self._embed_single_gate(T.conj().T, q)
            gates[f'S_{q}'] = self._embed_single_gate(S, q)
            gates[f'X_{q}'] = self._embed_single_gate(X, q)

        # CNOT gates
        if self.n_qubits >= 2:
            for i in range(self.n_qubits - 1):
                gates[f'CNOT_{i}_{i+1}'] = self._build_cnot(i, i+1)

        return gates

    def _embed_single_gate(self, gate, qubit):
        """Embed single-qubit gate into full Hilbert space."""
        if self.n_qubits == 1:
            return gate

        full_gate = np.eye(self.dim, dtype=complex)

        for i in range(self.dim):
            for j in range(self.dim):
                # Check if this matrix element is affected by the gate
                i_bit = (i >> qubit) & 1
                j_bit = (j >> qubit) & 1
                i_rest = i ^ (i_bit << qubit)
                j_rest = j ^ (j_bit << qubit)

                if i_rest == j_rest:
                    full_gate[i, j] = gate[i_bit, j_bit]
                else:
                    full_gate[i, j] = 0

        return full_gate

    def _build_cnot(self, control, target):
        """Build CNOT matrix."""
        cnot = np.eye(self.dim, dtype=complex)

        for i in range(self.dim):
            if (i >> control) & 1:
                j = i ^ (1 << target)
                cnot[i, i] = 0
                cnot[j, i] = 1

        return cnot

    def reset(self, target_unitary=None):
        """Reset environment."""
        self.current_unitary = np.eye(self.dim, dtype=complex)

        if target_unitary is not None:
            self.target_unitary = target_unitary
        else:
            self.target_unitary = self._generate_random_unitary()

        self.step_count = 0
        return self._get_observation()

    def _generate_random_unitary(self):
        """Generate random unitary using QR decomposition."""
        real = np.random.randn(self.dim, self.dim)
        imag = np.random.randn(self.dim, self.dim)
        A = real + 1j * imag
        Q, R = np.linalg.qr(A)
        # Make it properly random unitary
        D = np.diag(np.diag(R) / np.abs(np.diag(R)))
        return Q @ D

    def _get_observation(self):
        """Return observation for the agent."""
        # Flatten current and target unitaries
        current_flat = np.concatenate([
            self.current_unitary.real.flatten(),
            self.current_unitary.imag.flatten()
        ])
        target_flat = np.concatenate([
            self.target_unitary.real.flatten(),
            self.target_unitary.imag.flatten()
        ])

        # Include fidelity and step count
        fidelity = self._compute_fidelity()
        progress = self.step_count / self.max_gates

        return np.concatenate([
            current_flat, target_flat, [fidelity, progress]
        ])

    def step(self, action_idx):
        """Apply gate and compute reward."""
        gate_name = list(self.gate_set.keys())[action_idx]
        gate_matrix = self.gate_set[gate_name]

        # Apply gate: U_new = G @ U_current
        self.current_unitary = gate_matrix @ self.current_unitary

        self.step_count += 1

        # Compute reward
        fidelity = self._compute_fidelity()
        reward = self._compute_reward(fidelity)

        done = self.step_count >= self.max_gates or fidelity >= 0.999

        info = {
            'fidelity': fidelity,
            'gate': gate_name,
            'steps': self.step_count
        }

        return self._get_observation(), reward, done, info

    def _compute_fidelity(self):
        """Compute fidelity between current and target unitary."""
        # Average gate fidelity
        trace = np.trace(self.target_unitary.conj().T @ self.current_unitary)
        fidelity = (np.abs(trace) ** 2) / (self.dim ** 2)
        return fidelity

    def _compute_reward(self, fidelity):
        """Compute shaped reward."""
        # Base reward: fidelity
        reward = fidelity

        # Penalty for each gate used
        reward -= 0.01

        # Bonus for high fidelity
        if fidelity >= 0.99:
            reward += 5.0
        if fidelity >= 0.999:
            reward += 10.0

        return reward

# Example
env = GateSynthesisEnv(n_qubits=1, max_gates=10)
obs = env.reset()

print(f"Observation size: {len(obs)}")
print(f"Available gates: {list(env.gate_set.keys())}")
print(f"Initial fidelity: {env._compute_fidelity():.4f}")

# Random policy test
total_reward = 0
for _ in range(10):
    action = np.random.randint(env.n_actions)
    obs, reward, done, info = env.step(action)
    total_reward += reward
    if done:
        break

print(f"Final fidelity: {info['fidelity']:.4f}")
print(f"Total reward: {total_reward:.4f}")
```

## 4. Measurement and Observation Design

Designing appropriate observations is crucial for quantum RL, balancing information content with classical processability.

<Formula>
Observation Types:
- Full state: $o = (\text{Re}(|\psi\rangle), \text{Im}(|\psi\rangle))$ — $2 \times 2^n$ values
- Expectation values: $o = (\langle Z_1\rangle, \langle Z_2\rangle, \ldots)$ — $n$ values
- Tomography: Measurements in multiple bases
</Formula>

### Implementation

```python
class ObservationModes:
    """
    Different observation strategies for quantum environments.
    """

    def __init__(self, n_qubits):
        self.n_qubits = n_qubits
        self.dim = 2 ** n_qubits

    def full_state_observation(self, state):
        """
        Return full quantum state as observation.

        Pros: Complete information
        Cons: Exponential size, not physically realizable
        """
        return np.concatenate([state.real, state.imag])

    def pauli_expectation_observation(self, state):
        """
        Return Pauli expectation values.

        Pros: Constant size per qubit, physically measurable
        Cons: Incomplete information
        """
        expectations = []

        for qubit in range(self.n_qubits):
            # ⟨Z⟩
            exp_z = self._measure_z(state, qubit)
            expectations.append(exp_z)

            # ⟨X⟩
            exp_x = self._measure_x(state, qubit)
            expectations.append(exp_x)

            # ⟨Y⟩
            exp_y = self._measure_y(state, qubit)
            expectations.append(exp_y)

        return np.array(expectations)

    def _measure_z(self, state, qubit):
        """Measure ⟨Z⟩ for a qubit."""
        exp_z = 0.0
        for i in range(self.dim):
            prob = np.abs(state[i]) ** 2
            bit = (i >> qubit) & 1
            exp_z += prob * (1 - 2 * bit)
        return exp_z

    def _measure_x(self, state, qubit):
        """Measure ⟨X⟩ for a qubit."""
        # X = |+⟩⟨+| - |-⟩⟨-|
        exp_x = 0.0
        for i in range(self.dim):
            j = i ^ (1 << qubit)
            exp_x += np.real(np.conj(state[i]) * state[j])
        return exp_x

    def _measure_y(self, state, qubit):
        """Measure ⟨Y⟩ for a qubit."""
        exp_y = 0.0
        for i in range(self.dim):
            j = i ^ (1 << qubit)
            bit = (i >> qubit) & 1
            sign = 1 - 2 * bit
            exp_y += sign * np.imag(np.conj(state[i]) * state[j])
        return exp_y

    def bloch_sphere_observation(self, state):
        """
        Return Bloch sphere coordinates for each qubit.

        For single qubit: (x, y, z) = (⟨X⟩, ⟨Y⟩, ⟨Z⟩)
        """
        bloch_coords = []

        for qubit in range(self.n_qubits):
            x = self._measure_x(state, qubit)
            y = self._measure_y(state, qubit)
            z = self._measure_z(state, qubit)
            bloch_coords.extend([x, y, z])

        return np.array(bloch_coords)

    def correlation_observation(self, state):
        """
        Include two-qubit correlations ⟨Z_i Z_j⟩.
        """
        obs = self.pauli_expectation_observation(state).tolist()

        # Add correlations
        for i in range(self.n_qubits):
            for j in range(i + 1, self.n_qubits):
                zz = self._measure_zz(state, i, j)
                obs.append(zz)

        return np.array(obs)

    def _measure_zz(self, state, qubit1, qubit2):
        """Measure ⟨Z_i Z_j⟩ correlation."""
        exp_zz = 0.0
        for i in range(self.dim):
            prob = np.abs(state[i]) ** 2
            bit1 = (i >> qubit1) & 1
            bit2 = (i >> qubit2) & 1
            parity = 1 - 2 * (bit1 ^ bit2)
            exp_zz += prob * parity
        return exp_zz

    def noisy_observation(self, state, noise_level=0.1, n_shots=1000):
        """
        Simulate realistic noisy measurements.
        """
        expectations = self.pauli_expectation_observation(state)

        # Add shot noise
        shot_noise = np.sqrt((1 - expectations**2) / n_shots)
        noisy_obs = expectations + noise_level * shot_noise * np.random.randn(len(expectations))

        # Clip to valid range
        noisy_obs = np.clip(noisy_obs, -1, 1)

        return noisy_obs

# Example
obs_modes = ObservationModes(n_qubits=2)

# Create a test state (Bell state)
bell = np.zeros(4, dtype=complex)
bell[0] = 1/np.sqrt(2)
bell[3] = 1/np.sqrt(2)

print("Bell state observations:")
print(f"Full state: {obs_modes.full_state_observation(bell)}")
print(f"Pauli expectations: {obs_modes.pauli_expectation_observation(bell)}")
print(f"Bloch coordinates: {obs_modes.bloch_sphere_observation(bell)}")
print(f"With correlations: {obs_modes.correlation_observation(bell)}")
print(f"Noisy observation: {obs_modes.noisy_observation(bell)}")
```

<Tip>
For environments meant to run on real quantum hardware, use expectation value observations since full state tomography is exponentially expensive. Include correlations for entangled states.
</Tip>

## 5. Hybrid Quantum-Classical Simulators

Realistic quantum RL environments need to model noise, decoherence, and hardware constraints.

<Definition term="Hybrid Simulator">
A simulation environment that models both ideal quantum dynamics and realistic hardware effects like gate errors, measurement noise, and decoherence.
</Definition>

### Implementation

```python
class NoisyQuantumEnv:
    """
    Quantum environment with realistic noise models.
    """

    def __init__(self, n_qubits=2, noise_model='depolarizing', max_steps=30):
        self.n_qubits = n_qubits
        self.dim = 2 ** n_qubits
        self.max_steps = max_steps
        self.noise_model = noise_model

        # Noise parameters
        self.single_gate_error = 0.001
        self.two_gate_error = 0.01
        self.measurement_error = 0.02
        self.T1 = 50.0  # Relaxation time (microseconds)
        self.T2 = 30.0  # Dephasing time (microseconds)
        self.gate_time = 0.1  # Gate duration (microseconds)

        # State (density matrix for mixed states)
        self.rho = None
        self.target_state = None
        self.total_time = 0.0
        self.step_count = 0

        # Build action space
        self.action_space = self._build_action_space()

    def _build_action_space(self):
        """Define available gates."""
        actions = []
        for q in range(self.n_qubits):
            actions.extend([
                ('H', q), ('X', q), ('Z', q),
                ('RX', q, np.pi/4), ('RY', q, np.pi/4)
            ])
        for i in range(self.n_qubits - 1):
            actions.append(('CNOT', i, i+1))
        return actions

    def reset(self, target_state=None):
        """Reset to ground state."""
        # Initialize as pure state density matrix |0⟩⟨0|
        self.rho = np.zeros((self.dim, self.dim), dtype=complex)
        self.rho[0, 0] = 1.0

        if target_state is not None:
            self.target_state = np.outer(target_state, target_state.conj())
        else:
            psi = self._random_pure_state()
            self.target_state = np.outer(psi, psi.conj())

        self.total_time = 0.0
        self.step_count = 0

        return self._get_observation()

    def _random_pure_state(self):
        """Generate random pure state."""
        psi = np.random.randn(self.dim) + 1j * np.random.randn(self.dim)
        return psi / np.linalg.norm(psi)

    def step(self, action_idx):
        """Execute noisy quantum operation."""
        action = self.action_space[action_idx]

        # Apply ideal gate
        U = self._get_gate_unitary(action)
        self.rho = U @ self.rho @ U.conj().T

        # Apply gate error
        error_rate = self.two_gate_error if action[0] == 'CNOT' else self.single_gate_error
        self.rho = self._apply_depolarizing_noise(self.rho, error_rate)

        # Apply decoherence
        self.total_time += self.gate_time
        self.rho = self._apply_decoherence(self.rho)

        # Compute fidelity and reward
        fidelity = self._compute_fidelity()
        reward = fidelity - 0.01  # Step penalty

        self.step_count += 1
        done = self.step_count >= self.max_steps or fidelity >= 0.99

        info = {
            'fidelity': fidelity,
            'purity': self._compute_purity(),
            'total_time': self.total_time
        }

        return self._get_observation(), reward, done, info

    def _get_gate_unitary(self, action):
        """Get unitary matrix for action."""
        gate_name = action[0]

        if gate_name == 'H':
            U = np.array([[1, 1], [1, -1]]) / np.sqrt(2)
            return self._embed_gate(U, action[1])
        elif gate_name == 'X':
            U = np.array([[0, 1], [1, 0]])
            return self._embed_gate(U, action[1])
        elif gate_name == 'Z':
            U = np.array([[1, 0], [0, -1]])
            return self._embed_gate(U, action[1])
        elif gate_name == 'RX':
            angle = action[2]
            c, s = np.cos(angle/2), np.sin(angle/2)
            U = np.array([[c, -1j*s], [-1j*s, c]])
            return self._embed_gate(U, action[1])
        elif gate_name == 'RY':
            angle = action[2]
            c, s = np.cos(angle/2), np.sin(angle/2)
            U = np.array([[c, -s], [s, c]])
            return self._embed_gate(U, action[1])
        elif gate_name == 'CNOT':
            return self._build_cnot(action[1], action[2])

        return np.eye(self.dim, dtype=complex)

    def _embed_gate(self, gate, qubit):
        """Embed single-qubit gate."""
        full = np.eye(self.dim, dtype=complex)

        for i in range(self.dim):
            for j in range(self.dim):
                i_bit = (i >> qubit) & 1
                j_bit = (j >> qubit) & 1
                i_rest = i ^ (i_bit << qubit)
                j_rest = j ^ (j_bit << qubit)

                if i_rest == j_rest:
                    full[i, j] = gate[i_bit, j_bit]
                else:
                    full[i, j] = 0

        return full

    def _build_cnot(self, control, target):
        """Build CNOT gate matrix."""
        cnot = np.zeros((self.dim, self.dim), dtype=complex)

        for i in range(self.dim):
            if (i >> control) & 1:
                j = i ^ (1 << target)
                cnot[j, i] = 1
            else:
                cnot[i, i] = 1

        return cnot

    def _apply_depolarizing_noise(self, rho, p):
        """Apply depolarizing noise channel."""
        I = np.eye(self.dim) / self.dim
        return (1 - p) * rho + p * I

    def _apply_decoherence(self, rho):
        """Apply T1 and T2 decoherence."""
        t = self.gate_time

        # T1 decay (amplitude damping)
        gamma1 = 1 - np.exp(-t / self.T1)

        # T2 dephasing
        gamma2 = 1 - np.exp(-t / self.T2)

        # Simplified model: decay off-diagonal elements
        decay_factor = np.exp(-t / self.T2)

        new_rho = rho.copy()
        for i in range(self.dim):
            for j in range(self.dim):
                if i != j:
                    new_rho[i, j] *= decay_factor

        return new_rho

    def _compute_fidelity(self):
        """Compute state fidelity with target."""
        # Fidelity for density matrices
        sqrt_target = self._matrix_sqrt(self.target_state)
        product = sqrt_target @ self.rho @ sqrt_target
        sqrt_product = self._matrix_sqrt(product)
        fidelity = np.real(np.trace(sqrt_product)) ** 2
        return np.clip(fidelity, 0, 1)

    def _matrix_sqrt(self, M):
        """Compute matrix square root."""
        eigenvalues, eigenvectors = np.linalg.eigh(M)
        eigenvalues = np.maximum(eigenvalues, 0)  # Ensure positivity
        sqrt_eigenvalues = np.sqrt(eigenvalues)
        return eigenvectors @ np.diag(sqrt_eigenvalues) @ eigenvectors.conj().T

    def _compute_purity(self):
        """Compute purity Tr(ρ²)."""
        return np.real(np.trace(self.rho @ self.rho))

    def _get_observation(self):
        """Return observation from density matrix."""
        # Diagonal elements (populations) and selected off-diagonal (coherences)
        obs = []

        # Populations
        for i in range(self.dim):
            obs.append(np.real(self.rho[i, i]))

        # Main off-diagonal coherences
        for i in range(min(self.dim - 1, 4)):
            obs.append(np.real(self.rho[i, i+1]))
            obs.append(np.imag(self.rho[i, i+1]))

        obs.append(self._compute_purity())
        obs.append(self.total_time / (self.max_steps * self.gate_time))

        return np.array(obs)

# Example with noise
env = NoisyQuantumEnv(n_qubits=2, max_steps=20)
obs = env.reset()

print("Noisy quantum environment:")
print(f"Initial purity: {env._compute_purity():.4f}")

```

## 6. Environment Benchmarks

Standardized benchmarks enable comparison of different quantum RL algorithms.

<Definition term="Benchmark Suite">
A collection of standardized quantum control tasks with well-defined metrics for evaluating RL agent performance.
</Definition>

### Implementation

```python
class QuantumRLBenchmark:
    """
    Benchmark suite for quantum RL algorithms.
    """

    def __init__(self):
        self.benchmarks = {
            'state_prep_bell': self._create_bell_benchmark,
            'state_prep_ghz3': self._create_ghz3_benchmark,
            'gate_synth_toffoli': self._create_toffoli_benchmark,
            'variational': self._create_variational_benchmark
        }

        self.results = {}

    def _create_bell_benchmark(self):
        """Bell state preparation benchmark."""
        return {
            'name': 'Bell State Preparation',
            'env_class': StatePreparationEnv,
            'env_kwargs': {'n_qubits': 2, 'target_type': 'bell', 'max_steps': 20},
            'optimal_steps': 2,
            'optimal_fidelity': 1.0,
            'metrics': ['final_fidelity', 'steps_to_99', 'total_reward']
        }

    def _create_ghz3_benchmark(self):
        """3-qubit GHZ state preparation."""
        return {
            'name': '3-Qubit GHZ State',
            'env_class': StatePreparationEnv,
            'env_kwargs': {'n_qubits': 3, 'target_type': 'ghz', 'max_steps': 30},
            'optimal_steps': 3,
            'optimal_fidelity': 1.0,
            'metrics': ['final_fidelity', 'steps_to_99', 'total_reward']
        }

    def _create_toffoli_benchmark(self):
        """Toffoli gate synthesis benchmark."""
        return {
            'name': 'Toffoli Gate Synthesis',
            'env_class': GateSynthesisEnv,
            'env_kwargs': {'n_qubits': 3, 'max_gates': 50},
            'optimal_steps': 15,  # Approximate
            'optimal_fidelity': 0.999,
            'metrics': ['final_fidelity', 'gate_count', 'total_reward']
        }

    def _create_variational_benchmark(self):
        """Variational state preparation."""
        return {
            'name': 'Random State Preparation',
            'env_class': StatePreparationEnv,
            'env_kwargs': {'n_qubits': 2, 'target_type': 'random', 'max_steps': 50},
            'optimal_steps': None,  # Unknown
            'optimal_fidelity': 0.99,
            'metrics': ['final_fidelity', 'steps_to_99', 'total_reward']
        }

    def run_benchmark(self, benchmark_name, agent, n_episodes=100):
        """
        Run a benchmark with given agent.

        Args:
            benchmark_name: Name of benchmark
            agent: RL agent with select_action(obs) method
            n_episodes: Number of evaluation episodes

        Returns:
            metrics: Dictionary of performance metrics
        """
        if benchmark_name not in self.benchmarks:
            raise ValueError(f"Unknown benchmark: {benchmark_name}")

        config = self.benchmarks[benchmark_name]()
        EnvClass = config['env_class']
        env = EnvClass(**config['env_kwargs'])

        metrics = {
            'final_fidelities': [],
            'steps_to_99': [],
            'total_rewards': [],
            'gate_counts': []
        }

        for episode in range(n_episodes):
            obs = env.reset()
            total_reward = 0
            steps_to_99 = None

            for step in range(config['env_kwargs']['max_steps']):
                action = agent.select_action(obs)
                obs, reward, done, info = env.step(action)
                total_reward += reward

                if steps_to_99 is None and info['fidelity'] >= 0.99:
                    steps_to_99 = step + 1

                if done:
                    break

            metrics['final_fidelities'].append(info['fidelity'])
            metrics['steps_to_99'].append(steps_to_99 if steps_to_99 else step + 1)
            metrics['total_rewards'].append(total_reward)
            metrics['gate_counts'].append(step + 1)

        # Compute summary statistics
        summary = {
            'benchmark': config['name'],
            'avg_fidelity': np.mean(metrics['final_fidelities']),
            'std_fidelity': np.std(metrics['final_fidelities']),
            'avg_steps_to_99': np.mean(metrics['steps_to_99']),
            'success_rate': np.mean(np.array(metrics['final_fidelities']) >= 0.99),
            'avg_reward': np.mean(metrics['total_rewards']),
            'optimal_steps': config['optimal_steps']
        }

        self.results[benchmark_name] = summary
        return summary

    def print_results(self):
        """Print benchmark results."""
        print("\n" + "="*60)
        print("Quantum RL Benchmark Results")
        print("="*60)

        for name, results in self.results.items():
            print(f"\n{results['benchmark']}:")
            print(f"  Average Fidelity: {results['avg_fidelity']:.4f} ± {results['std_fidelity']:.4f}")
            print(f"  Success Rate (≥99%): {results['success_rate']*100:.1f}%")
            print(f"  Avg Steps to 99%: {results['avg_steps_to_99']:.1f}")
            if results['optimal_steps']:
                print(f"  Optimal Steps: {results['optimal_steps']}")
            print(f"  Avg Total Reward: {results['avg_reward']:.2f}")
```

## Summary

Quantum RL environments provide the interface between RL agents and quantum control tasks:

1. **Environment framework**: Gym-like interface adapted for quantum states, gates, and fidelity-based rewards
2. **State preparation**: Goal-directed tasks for preparing specific quantum states
3. **Gate synthesis**: Learning optimal gate sequences for target unitaries
4. **Observation design**: Trading off information content vs. physical realizability
5. **Noisy simulation**: Realistic models including depolarizing noise and decoherence
6. **Benchmarks**: Standardized tasks for fair algorithm comparison

Well-designed environments are crucial for developing practical quantum RL agents.
