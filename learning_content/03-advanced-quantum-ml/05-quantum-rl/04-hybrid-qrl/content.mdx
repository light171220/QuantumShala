# Hybrid Quantum-Classical RL Systems

Practical quantum reinforcement learning often requires combining quantum circuits with classical neural networks. This lesson explores hybrid architectures that leverage quantum advantages while maintaining classical flexibility for real-world RL applications.

- Implement quantum-enhanced DQN systems
- Combine classical preprocessing with quantum networks
- Optimize hybrid training pipelines

## 1. Hybrid Architecture Principles

Hybrid quantum-classical RL systems strategically partition computation between classical and quantum processors.

<Definition term="Hybrid QRL">
A reinforcement learning system that combines classical neural networks for tasks like feature extraction and memory with quantum circuits for policy or value representation.
</Definition>

### Architecture Design Patterns

```python
import numpy as np

class HybridQRLArchitecture:
    """
    Hybrid quantum-classical RL architecture.

    Pattern types:
    - Type A: Classical preprocessing → Quantum policy
    - Type B: Quantum feature extraction → Classical head
    - Type C: Parallel quantum-classical ensemble
    """

    def __init__(self, n_qubits, classical_hidden, pattern='A'):
        self.n_qubits = n_qubits
        self.classical_hidden = classical_hidden
        self.pattern = pattern

        # Classical preprocessing network (for Type A)
        self.classical_encoder = self._init_classical_encoder()

        # Quantum circuit parameters
        self.n_layers = 4
        self.quantum_params = np.random.randn(self.n_layers, n_qubits, 3) * 0.1

        # Classical head (for Type B)
        self.classical_head = self._init_classical_head()

    def _init_classical_encoder(self):
        """Initialize classical preprocessing network."""
        return {
            'W1': np.random.randn(self.classical_hidden, 64) * 0.1,
            'b1': np.zeros(self.classical_hidden),
            'W2': np.random.randn(self.n_qubits, self.classical_hidden) * 0.1,
            'b2': np.zeros(self.n_qubits)
        }

    def _init_classical_head(self):
        """Initialize classical decision head."""
        return {
            'W': np.random.randn(4, self.n_qubits) * 0.1,  # 4 actions
            'b': np.zeros(4)
        }

    def classical_preprocess(self, state):
        """Classical feature extraction and compression."""
        h = np.tanh(self.classical_encoder['W1'] @ state +
                   self.classical_encoder['b1'])
        encoded = np.tanh(self.classical_encoder['W2'] @ h +
                         self.classical_encoder['b2'])
        return encoded

    def quantum_circuit(self, encoded_input):
        """Simulate variational quantum circuit."""
        n_qubits = self.n_qubits
        dim = 2 ** n_qubits

        # Initialize state
        state = np.zeros(dim, dtype=complex)
        state[0] = 1.0

        # Encoding layer
        for i in range(n_qubits):
            angle = np.pi * encoded_input[i]
            ry = np.array([[np.cos(angle/2), -np.sin(angle/2)],
                          [np.sin(angle/2), np.cos(angle/2)]])
            state = self._apply_single_qubit_gate(state, ry, i, n_qubits)

        # Variational layers
        for layer in range(self.n_layers):
            # Rotation gates
            for i in range(n_qubits):
                params = self.quantum_params[layer, i]
                state = self._apply_rotation(state, params, i, n_qubits)

            # Entangling gates
            for i in range(n_qubits - 1):
                state = self._apply_cnot(state, i, i + 1, n_qubits)

        return state

    def _apply_rotation(self, state, params, qubit, n_qubits):
        """Apply Rz Ry Rz rotation."""
        for angle, axis in zip(params, ['z', 'y', 'z']):
            if axis == 'z':
                gate = np.array([[np.exp(-1j * angle/2), 0],
                                [0, np.exp(1j * angle/2)]])
            else:
                gate = np.array([[np.cos(angle/2), -np.sin(angle/2)],
                                [np.sin(angle/2), np.cos(angle/2)]])
            state = self._apply_single_qubit_gate(state, gate, qubit, n_qubits)
        return state

    def _apply_single_qubit_gate(self, state, gate, qubit, n_qubits):
        """Apply single-qubit gate."""
        dim = 2 ** n_qubits
        new_state = np.zeros(dim, dtype=complex)
        for i in range(dim):
            bit = (i >> qubit) & 1
            i_flipped = i ^ (1 << qubit)
            if bit == 0:
                new_state[i] += gate[0, 0] * state[i]
                new_state[i_flipped] += gate[1, 0] * state[i]
            else:
                new_state[i] += gate[1, 1] * state[i]
                new_state[i_flipped] += gate[0, 1] * state[i]
        return new_state

    def _apply_cnot(self, state, control, target, n_qubits):
        """Apply CNOT gate."""
        dim = 2 ** n_qubits
        new_state = np.zeros(dim, dtype=complex)
        for i in range(dim):
            control_bit = (i >> control) & 1
            if control_bit == 1:
                i_flipped = i ^ (1 << target)
                new_state[i_flipped] = state[i]
            else:
                new_state[i] = state[i]
        return new_state

    def forward_type_a(self, state):
        """Type A: Classical preprocessing → Quantum policy."""
        encoded = self.classical_preprocess(state)
        quantum_state = self.quantum_circuit(encoded)

        # Measure action probabilities
        probs = np.abs(quantum_state) ** 2
        action_probs = self._extract_action_probs(probs)
        return action_probs

    def _extract_action_probs(self, probs):
        """Extract action probabilities from measurement."""
        n_actions = 4
        action_probs = np.zeros(n_actions)
        states_per_action = len(probs) // n_actions
        for a in range(n_actions):
            action_probs[a] = np.sum(probs[a * states_per_action:
                                          (a + 1) * states_per_action])
        return action_probs / (action_probs.sum() + 1e-8)

# Example
hybrid_agent = HybridQRLArchitecture(n_qubits=4, classical_hidden=32)
state = np.random.randn(64)
action_probs = hybrid_agent.forward_type_a(state)
print(f"Action probabilities: {action_probs}")
```

<Tip>
Use classical preprocessing to reduce high-dimensional observations (images, sensor data) to a quantum-compatible size before encoding into the quantum circuit.
</Tip>

## 2. Hybrid Actor-Critic Architecture

The actor-critic framework naturally partitions into quantum actor and classical critic (or vice versa).

<Formula>
Hybrid Actor-Critic Loss:
$$\mathcal{L} = \mathcal{L}_\text{actor}^\text{quantum} + \beta \mathcal{L}_\text{critic}^\text{classical}$$

Actor (quantum): $\pi_\theta(a|s) = \text{QCircuit}(s; \theta)$

Critic (classical): $V_\phi(s) = \text{NN}(s; \phi)$
</Formula>

### Implementation

```python
class HybridActorCritic:
    """
    Hybrid actor-critic with quantum actor and classical critic.
    """

    def __init__(self, state_dim, n_actions, n_qubits=4):
        self.state_dim = state_dim
        self.n_actions = n_actions
        self.n_qubits = n_qubits

        # Quantum actor parameters
        self.n_actor_layers = 3
        self.actor_params = np.random.randn(
            self.n_actor_layers, n_qubits, 3) * 0.1

        # Classical encoder for state compression
        self.encoder_W = np.random.randn(n_qubits, state_dim) * 0.1
        self.encoder_b = np.zeros(n_qubits)

        # Classical critic network
        self.critic = {
            'W1': np.random.randn(64, state_dim) * 0.1,
            'b1': np.zeros(64),
            'W2': np.random.randn(32, 64) * 0.1,
            'b2': np.zeros(32),
            'W3': np.random.randn(1, 32) * 0.1,
            'b3': np.zeros(1)
        }

        self.gamma = 0.99
        self.actor_lr = 0.01
        self.critic_lr = 0.001

    def encode_state(self, state):
        """Encode state for quantum circuit."""
        encoded = np.tanh(self.encoder_W @ state + self.encoder_b)
        return encoded

    def quantum_actor(self, encoded_state):
        """Quantum policy network."""
        n_qubits = self.n_qubits
        dim = 2 ** n_qubits

        # Initialize superposition
        psi = np.ones(dim, dtype=complex) / np.sqrt(dim)

        # Encoding
        for i in range(n_qubits):
            angle = np.pi * encoded_state[i]
            psi = self._apply_ry(psi, angle, i, n_qubits)

        # Variational layers
        for layer in range(self.n_actor_layers):
            for i in range(n_qubits):
                params = self.actor_params[layer, i]
                psi = self._apply_u3(psi, params, i, n_qubits)

            # Ring entanglement
            for i in range(n_qubits):
                psi = self._apply_cz(psi, i, (i + 1) % n_qubits, n_qubits)

        # Compute action probabilities
        probs = np.abs(psi) ** 2
        action_probs = np.zeros(self.n_actions)

        for a in range(self.n_actions):
            # Sum probabilities where first qubits encode action
            for i in range(dim):
                if i % self.n_actions == a:
                    action_probs[a] += probs[i]

        return action_probs / (action_probs.sum() + 1e-8)

    def _apply_ry(self, state, angle, qubit, n_qubits):
        """Apply RY rotation."""
        dim = 2 ** n_qubits
        new_state = np.zeros(dim, dtype=complex)
        c, s = np.cos(angle/2), np.sin(angle/2)

        for i in range(dim):
            bit = (i >> qubit) & 1
            i_flipped = i ^ (1 << qubit)
            if bit == 0:
                new_state[i] += c * state[i]
                new_state[i_flipped] += s * state[i]
            else:
                new_state[i] += c * state[i]
                new_state[i_flipped] -= s * state[i]
        return new_state

    def _apply_u3(self, state, params, qubit, n_qubits):
        """Apply general U3 rotation."""
        theta, phi, lam = params
        dim = 2 ** n_qubits
        new_state = np.zeros(dim, dtype=complex)

        # U3 matrix elements
        u00 = np.cos(theta/2)
        u01 = -np.exp(1j * lam) * np.sin(theta/2)
        u10 = np.exp(1j * phi) * np.sin(theta/2)
        u11 = np.exp(1j * (phi + lam)) * np.cos(theta/2)

        for i in range(dim):
            bit = (i >> qubit) & 1
            i_flipped = i ^ (1 << qubit)
            if bit == 0:
                new_state[i] += u00 * state[i]
                new_state[i_flipped] += u10 * state[i]
            else:
                new_state[i] += u11 * state[i]
                new_state[i_flipped] += u01 * state[i]
        return new_state

    def _apply_cz(self, state, control, target, n_qubits):
        """Apply CZ gate."""
        dim = 2 ** n_qubits
        new_state = state.copy()
        for i in range(dim):
            if ((i >> control) & 1) and ((i >> target) & 1):
                new_state[i] *= -1
        return new_state

    def classical_critic(self, state):
        """Classical value network."""
        h1 = np.maximum(0, self.critic['W1'] @ state + self.critic['b1'])
        h2 = np.maximum(0, self.critic['W2'] @ h1 + self.critic['b2'])
        value = (self.critic['W3'] @ h2 + self.critic['b3'])[0]
        return value

    def select_action(self, state):
        """Select action using quantum policy."""
        encoded = self.encode_state(state)
        probs = self.quantum_actor(encoded)
        action = np.random.choice(self.n_actions, p=probs)
        return action, probs[action]

    def update(self, state, action, reward, next_state, done):
        """Update actor and critic."""
        # Compute TD error using classical critic
        value = self.classical_critic(state)
        next_value = 0 if done else self.classical_critic(next_state)
        td_target = reward + self.gamma * next_value
        td_error = td_target - value

        # Update classical critic
        self._update_critic(state, td_target)

        # Update quantum actor using policy gradient
        self._update_actor(state, action, td_error)

        return td_error

    def _update_critic(self, state, target):
        """Update critic network."""
        # Forward pass
        h1 = np.maximum(0, self.critic['W1'] @ state + self.critic['b1'])
        h2 = np.maximum(0, self.critic['W2'] @ h1 + self.critic['b2'])
        pred = (self.critic['W3'] @ h2 + self.critic['b3'])[0]

        # Backward pass (simplified gradient descent)
        error = pred - target

        # Update output layer
        self.critic['W3'] -= self.critic_lr * error * h2.reshape(1, -1)
        self.critic['b3'] -= self.critic_lr * error

        # Backpropagate through hidden layers (simplified)
        grad_h2 = error * self.critic['W3'].T.flatten() * (h2 > 0)
        self.critic['W2'] -= self.critic_lr * np.outer(grad_h2, h1)
        self.critic['b2'] -= self.critic_lr * grad_h2

    def _update_actor(self, state, action, advantage):
        """Update quantum actor using parameter shift."""
        encoded = self.encode_state(state)
        shift = np.pi / 2

        for layer in range(self.n_actor_layers):
            for qubit in range(self.n_qubits):
                for param_idx in range(3):
                    # Parameter shift
                    self.actor_params[layer, qubit, param_idx] += shift
                    probs_plus = self.quantum_actor(encoded)

                    self.actor_params[layer, qubit, param_idx] -= 2 * shift
                    probs_minus = self.quantum_actor(encoded)

                    self.actor_params[layer, qubit, param_idx] += shift

                    # Policy gradient
                    grad = (probs_plus[action] - probs_minus[action]) / 2
                    self.actor_params[layer, qubit, param_idx] += (
                        self.actor_lr * advantage * grad)

# Example usage
agent = HybridActorCritic(state_dim=8, n_actions=4, n_qubits=4)
state = np.random.randn(8)
action, prob = agent.select_action(state)
print(f"Selected action: {action} with probability: {prob:.4f}")
```

<Warning>
When training hybrid actor-critic systems, ensure the learning rates are properly balanced. The quantum actor may require higher learning rates than the classical critic due to different gradient magnitudes.
</Warning>

## 3. Quantum-Enhanced DQN

Deep Q-Networks can incorporate quantum components for enhanced value function approximation.

<Definition term="Quantum DQN">
A DQN architecture where the Q-network is implemented as a variational quantum circuit, potentially with classical preprocessing and postprocessing layers.
</Definition>

### Implementation

```python
class QuantumDQN:
    """
    Quantum Deep Q-Network with experience replay and target network.
    """

    def __init__(self, state_dim, n_actions, n_qubits=4):
        self.state_dim = state_dim
        self.n_actions = n_actions
        self.n_qubits = n_qubits

        # Classical preprocessing
        self.preprocess = {
            'W': np.random.randn(n_qubits, state_dim) * 0.1,
            'b': np.zeros(n_qubits)
        }

        # Quantum Q-network parameters
        self.n_layers = 4
        self.q_params = np.random.randn(self.n_layers, n_qubits, 3) * 0.1

        # Target network parameters (copy of Q-network)
        self.target_params = self.q_params.copy()

        # Classical postprocessing (output scaling)
        self.postprocess = {
            'W': np.random.randn(n_actions, n_qubits) * 0.1,
            'b': np.zeros(n_actions)
        }

        # Experience replay buffer
        self.buffer = []
        self.buffer_size = 10000
        self.batch_size = 32

        # Hyperparameters
        self.gamma = 0.99
        self.lr = 0.01
        self.epsilon = 1.0
        self.epsilon_min = 0.01
        self.epsilon_decay = 0.995
        self.target_update_freq = 100
        self.step_count = 0

    def preprocess_state(self, state):
        """Classical preprocessing for quantum encoding."""
        return np.tanh(self.preprocess['W'] @ state + self.preprocess['b'])

    def quantum_forward(self, encoded_state, params):
        """Forward pass through quantum circuit."""
        n_qubits = self.n_qubits
        dim = 2 ** n_qubits

        # Initialize
        psi = np.zeros(dim, dtype=complex)
        psi[0] = 1.0

        # Encoding
        for i in range(n_qubits):
            angle = np.arctan(encoded_state[i]) * 2
            psi = self._ry_gate(psi, angle, i, n_qubits)

        # Variational layers
        for layer in range(self.n_layers):
            for i in range(n_qubits):
                p = params[layer, i]
                psi = self._rz_gate(psi, p[0], i, n_qubits)
                psi = self._ry_gate(psi, p[1], i, n_qubits)
                psi = self._rz_gate(psi, p[2], i, n_qubits)

            for i in range(n_qubits - 1):
                psi = self._cnot_gate(psi, i, i + 1, n_qubits)

        # Measure expectation values for each qubit
        expectations = []
        for i in range(n_qubits):
            exp_z = self._measure_z(psi, i, n_qubits)
            expectations.append(exp_z)

        return np.array(expectations)

    def _ry_gate(self, state, angle, qubit, n_qubits):
        """Apply RY gate."""
        dim = 2 ** n_qubits
        new_state = np.zeros(dim, dtype=complex)
        c, s = np.cos(angle/2), np.sin(angle/2)

        for i in range(dim):
            bit = (i >> qubit) & 1
            i_f = i ^ (1 << qubit)
            if bit == 0:
                new_state[i] += c * state[i]
                new_state[i_f] += s * state[i]
            else:
                new_state[i] += c * state[i]
                new_state[i_f] -= s * state[i]
        return new_state

    def _rz_gate(self, state, angle, qubit, n_qubits):
        """Apply RZ gate."""
        dim = 2 ** n_qubits
        new_state = np.zeros(dim, dtype=complex)

        for i in range(dim):
            bit = (i >> qubit) & 1
            phase = np.exp(-1j * angle / 2) if bit == 0 else np.exp(1j * angle / 2)
            new_state[i] = phase * state[i]
        return new_state

    def _cnot_gate(self, state, control, target, n_qubits):
        """Apply CNOT gate."""
        dim = 2 ** n_qubits
        new_state = np.zeros(dim, dtype=complex)

        for i in range(dim):
            if (i >> control) & 1:
                new_state[i ^ (1 << target)] = state[i]
            else:
                new_state[i] = state[i]
        return new_state

    def _measure_z(self, state, qubit, n_qubits):
        """Measure Z expectation value."""
        dim = 2 ** n_qubits
        exp_z = 0.0

        for i in range(dim):
            prob = np.abs(state[i]) ** 2
            bit = (i >> qubit) & 1
            exp_z += prob * (1 - 2 * bit)  # +1 for |0⟩, -1 for |1⟩

        return exp_z

    def get_q_values(self, state, use_target=False):
        """Get Q-values for all actions."""
        encoded = self.preprocess_state(state)
        params = self.target_params if use_target else self.q_params
        quantum_out = self.quantum_forward(encoded, params)

        # Classical postprocessing
        q_values = self.postprocess['W'] @ quantum_out + self.postprocess['b']
        return q_values

    def select_action(self, state):
        """Epsilon-greedy action selection."""
        if np.random.random() < self.epsilon:
            return np.random.randint(self.n_actions)

        q_values = self.get_q_values(state)
        return np.argmax(q_values)

    def store_transition(self, state, action, reward, next_state, done):
        """Store transition in replay buffer."""
        if len(self.buffer) >= self.buffer_size:
            self.buffer.pop(0)
        self.buffer.append((state, action, reward, next_state, done))

    def train_step(self):
        """Train on a batch from replay buffer."""
        if len(self.buffer) < self.batch_size:
            return 0.0

        # Sample batch
        indices = np.random.choice(len(self.buffer), self.batch_size, replace=False)
        batch = [self.buffer[i] for i in indices]

        total_loss = 0.0

        for state, action, reward, next_state, done in batch:
            # Compute target
            if done:
                target = reward
            else:
                next_q = self.get_q_values(next_state, use_target=True)
                target = reward + self.gamma * np.max(next_q)

            # Current Q-value
            q_values = self.get_q_values(state)
            td_error = target - q_values[action]
            total_loss += td_error ** 2

            # Update parameters using parameter shift
            self._update_parameters(state, action, td_error)

        # Update target network periodically
        self.step_count += 1
        if self.step_count % self.target_update_freq == 0:
            self.target_params = self.q_params.copy()

        # Decay epsilon
        self.epsilon = max(self.epsilon_min, self.epsilon * self.epsilon_decay)

        return total_loss / self.batch_size

    def _update_parameters(self, state, action, td_error):
        """Update quantum parameters using semi-gradient TD."""
        encoded = self.preprocess_state(state)
        shift = np.pi / 2

        for layer in range(self.n_layers):
            for qubit in range(self.n_qubits):
                for param_idx in range(3):
                    # Compute gradient via parameter shift
                    self.q_params[layer, qubit, param_idx] += shift
                    q_plus = self.get_q_values(state)[action]

                    self.q_params[layer, qubit, param_idx] -= 2 * shift
                    q_minus = self.get_q_values(state)[action]

                    self.q_params[layer, qubit, param_idx] += shift

                    grad = (q_plus - q_minus) / 2

                    # TD update
                    self.q_params[layer, qubit, param_idx] += (
                        self.lr * td_error * grad)

# Example
dqn = QuantumDQN(state_dim=4, n_actions=2, n_qubits=4)
state = np.random.randn(4)
action = dqn.select_action(state)
print(f"Q-values: {dqn.get_q_values(state)}")
print(f"Selected action: {action}")
```

## 4. Classical-Quantum Feature Exchange

Effective hybrid systems require careful design of information flow between classical and quantum components.

<Formula>
Feature Exchange Protocol:
$$z_\text{classical} = f_\text{CNN}(s) \in \mathbb{R}^{n_\text{qubits}}$$
$$z_\text{quantum} = \text{Measure}(\text{QCircuit}(z_\text{classical})) \in \mathbb{R}^{n_\text{qubits}}$$
$$\pi(a|s) = \text{Softmax}(W \cdot z_\text{quantum} + b)$$
</Formula>

### Implementation

```python
class FeatureExchangeAgent:
    """
    Agent with bidirectional classical-quantum feature exchange.
    """

    def __init__(self, input_dim, n_qubits=4, n_actions=4):
        self.input_dim = input_dim
        self.n_qubits = n_qubits
        self.n_actions = n_actions

        # Classical feature extractor (simulating CNN-like processing)
        self.classical_layers = [
            {'W': np.random.randn(32, input_dim) * 0.1, 'b': np.zeros(32)},
            {'W': np.random.randn(16, 32) * 0.1, 'b': np.zeros(16)},
            {'W': np.random.randn(n_qubits, 16) * 0.1, 'b': np.zeros(n_qubits)}
        ]

        # Quantum processing parameters
        self.n_q_layers = 3
        self.q_params = np.random.randn(self.n_q_layers, n_qubits, 3) * 0.1

        # Classical output head
        self.output_head = {
            'W': np.random.randn(n_actions, n_qubits) * 0.1,
            'b': np.zeros(n_actions)
        }

        self.lr = 0.01

    def classical_forward(self, x):
        """Forward through classical layers."""
        h = x
        activations = [x]

        for layer in self.classical_layers[:-1]:
            h = np.maximum(0, layer['W'] @ h + layer['b'])  # ReLU
            activations.append(h)

        # Final layer with tanh for quantum encoding
        h = np.tanh(self.classical_layers[-1]['W'] @ h +
                   self.classical_layers[-1]['b'])
        activations.append(h)

        return h, activations

    def quantum_processing(self, classical_features):
        """Process features through quantum circuit."""
        n_qubits = self.n_qubits
        dim = 2 ** n_qubits

        # Initialize state
        psi = np.zeros(dim, dtype=complex)
        psi[0] = 1.0

        # Encode classical features
        for i in range(n_qubits):
            angle = classical_features[i] * np.pi
            psi = self._apply_ry(psi, angle, i)

        # Variational processing
        for layer in range(self.n_q_layers):
            # Single-qubit rotations
            for i in range(n_qubits):
                psi = self._apply_rz(psi, self.q_params[layer, i, 0], i)
                psi = self._apply_ry(psi, self.q_params[layer, i, 1], i)
                psi = self._apply_rz(psi, self.q_params[layer, i, 2], i)

            # Entanglement
            for i in range(n_qubits - 1):
                psi = self._apply_cnot(psi, i, i + 1)

        # Extract quantum features (expectation values)
        quantum_features = np.array([
            self._expectation_z(psi, i) for i in range(n_qubits)
        ])

        return quantum_features

    def _apply_ry(self, psi, angle, qubit):
        """Apply RY gate."""
        dim = len(psi)
        n_qubits = int(np.log2(dim))
        new_psi = np.zeros(dim, dtype=complex)
        c, s = np.cos(angle/2), np.sin(angle/2)

        for i in range(dim):
            bit = (i >> qubit) & 1
            i_f = i ^ (1 << qubit)
            if bit == 0:
                new_psi[i] += c * psi[i]
                new_psi[i_f] += s * psi[i]
            else:
                new_psi[i] += c * psi[i]
                new_psi[i_f] -= s * psi[i]
        return new_psi

    def _apply_rz(self, psi, angle, qubit):
        """Apply RZ gate."""
        dim = len(psi)
        new_psi = np.zeros(dim, dtype=complex)

        for i in range(dim):
            bit = (i >> qubit) & 1
            phase = np.exp(-1j * angle/2) if bit == 0 else np.exp(1j * angle/2)
            new_psi[i] = phase * psi[i]
        return new_psi

    def _apply_cnot(self, psi, control, target):
        """Apply CNOT gate."""
        dim = len(psi)
        new_psi = np.zeros(dim, dtype=complex)

        for i in range(dim):
            if (i >> control) & 1:
                new_psi[i ^ (1 << target)] = psi[i]
            else:
                new_psi[i] = psi[i]
        return new_psi

    def _expectation_z(self, psi, qubit):
        """Compute ⟨Z⟩ for a qubit."""
        dim = len(psi)
        exp_z = 0.0

        for i in range(dim):
            prob = np.abs(psi[i]) ** 2
            bit = (i >> qubit) & 1
            exp_z += prob * (1 - 2 * bit)

        return exp_z

    def forward(self, state):
        """Full forward pass through hybrid network."""
        # Classical feature extraction
        classical_features, _ = self.classical_forward(state)

        # Quantum processing
        quantum_features = self.quantum_processing(classical_features)

        # Classical output
        logits = self.output_head['W'] @ quantum_features + self.output_head['b']
        probs = self._softmax(logits)

        return probs, quantum_features

    def _softmax(self, x):
        """Compute softmax probabilities."""
        exp_x = np.exp(x - np.max(x))
        return exp_x / exp_x.sum()

    def select_action(self, state):
        """Sample action from policy."""
        probs, _ = self.forward(state)
        return np.random.choice(self.n_actions, p=probs)

    def compute_gradient_info(self, state, action, advantage):
        """Compute gradient information for training."""
        probs, quantum_features = self.forward(state)
        log_prob = np.log(probs[action] + 1e-8)

        return {
            'probs': probs,
            'quantum_features': quantum_features,
            'log_prob': log_prob,
            'loss': -log_prob * advantage
        }

print(f"Quantum features: {q_features}")
```

## 5. Training Pipeline Optimization

Efficient training of hybrid systems requires careful coordination of classical and quantum updates.

<Definition term="Coordinated Training">
A training strategy that alternates or jointly optimizes classical and quantum components while managing different learning dynamics and gradient scales.
</Definition>

### Implementation

```python
class OptimizedHybridTrainer:
    """
    Optimized training pipeline for hybrid quantum-classical RL.
    """

    def __init__(self, agent, env_step_fn):
        self.agent = agent
        self.env_step = env_step_fn

        # Separate optimizers for classical and quantum
        self.classical_lr = 0.001
        self.quantum_lr = 0.01

        # Gradient accumulation
        self.gradient_accumulation_steps = 4
        self.accumulated_grads = None

        # Learning rate scheduling
        self.lr_decay = 0.999
        self.min_lr = 1e-5

        # Gradient clipping
        self.max_grad_norm = 1.0

        # Training statistics
        self.episode_rewards = []
        self.losses = []
        self.grad_norms = []

    def compute_returns(self, rewards, gamma=0.99):
        """Compute discounted returns."""
        returns = np.zeros_like(rewards, dtype=np.float64)
        running_return = 0

        for t in reversed(range(len(rewards))):
            running_return = rewards[t] + gamma * running_return
            returns[t] = running_return

        return returns

    def normalize_advantages(self, advantages):
        """Normalize advantages for stable training."""
        mean = np.mean(advantages)
        std = np.std(advantages) + 1e-8
        return (advantages - mean) / std

    def collect_episode(self, max_steps=200):
        """Collect an episode of experience."""
        states, actions, rewards, log_probs = [], [], [], []

        state = np.random.randn(self.agent.input_dim)  # Reset env

        for step in range(max_steps):
            probs, _ = self.agent.forward(state)
            action = np.random.choice(len(probs), p=probs)
            log_prob = np.log(probs[action] + 1e-8)

            # Simulate environment step
            next_state = np.random.randn(self.agent.input_dim)
            reward = np.random.randn() * 0.1  # Simulated reward
            done = step == max_steps - 1

            states.append(state)
            actions.append(action)
            rewards.append(reward)
            log_probs.append(log_prob)

            state = next_state

            if done:
                break

        return {
            'states': np.array(states),
            'actions': np.array(actions),
            'rewards': np.array(rewards),
            'log_probs': np.array(log_probs)
        }

    def compute_quantum_gradients(self, state, action, advantage):
        """Compute gradients for quantum parameters using parameter shift."""
        grads = np.zeros_like(self.agent.q_params)
        shift = np.pi / 2

        for layer in range(self.agent.n_q_layers):
            for qubit in range(self.agent.n_qubits):
                for param_idx in range(3):
                    # Positive shift
                    self.agent.q_params[layer, qubit, param_idx] += shift
                    probs_plus, _ = self.agent.forward(state)

                    # Negative shift
                    self.agent.q_params[layer, qubit, param_idx] -= 2 * shift
                    probs_minus, _ = self.agent.forward(state)

                    # Restore
                    self.agent.q_params[layer, qubit, param_idx] += shift

                    # Gradient estimate
                    grad = (np.log(probs_plus[action] + 1e-8) -
                           np.log(probs_minus[action] + 1e-8)) / 2
                    grads[layer, qubit, param_idx] = -advantage * grad

        return grads

    def clip_gradients(self, grads):
        """Clip gradient norm."""
        grad_norm = np.sqrt(np.sum(grads ** 2))
        self.grad_norms.append(grad_norm)

        if grad_norm > self.max_grad_norm:
            grads = grads * (self.max_grad_norm / grad_norm)

        return grads

    def update_quantum_params(self, grads):
        """Update quantum parameters with clipped gradients."""
        clipped_grads = self.clip_gradients(grads)
        self.agent.q_params -= self.quantum_lr * clipped_grads

    def train_episode(self):
        """Train on one episode."""
        # Collect experience
        episode = self.collect_episode()

        # Compute returns and advantages
        returns = self.compute_returns(episode['rewards'])
        baseline = np.mean(returns)
        advantages = self.normalize_advantages(returns - baseline)

        # Update policy
        total_grad = np.zeros_like(self.agent.q_params)

        for t in range(len(episode['states'])):
            grads = self.compute_quantum_gradients(
                episode['states'][t],
                episode['actions'][t],
                advantages[t]
            )
            total_grad += grads

        # Average gradients
        total_grad /= len(episode['states'])

        # Apply update
        self.update_quantum_params(total_grad)

        # Decay learning rate
        self.quantum_lr = max(self.min_lr,
                              self.quantum_lr * self.lr_decay)

        # Track statistics
        episode_reward = np.sum(episode['rewards'])
        self.episode_rewards.append(episode_reward)

        return episode_reward

    def train(self, n_episodes=100, verbose=True):
        """Train for multiple episodes."""
        for episode in range(n_episodes):
            reward = self.train_episode()

            if verbose and episode % 10 == 0:
                avg_reward = np.mean(self.episode_rewards[-10:])
                avg_grad = np.mean(self.grad_norms[-10:]) if self.grad_norms else 0
                print(f"Episode {episode}: Reward={reward:.3f}, "
                      f"Avg={avg_reward:.3f}, GradNorm={avg_grad:.4f}")

        return self.episode_rewards

# Example
agent = FeatureExchangeAgent(input_dim=16, n_qubits=4, n_actions=4)
trainer = OptimizedHybridTrainer(agent, None)
print("Training hybrid agent...")
rewards = trainer.train(n_episodes=20, verbose=True)
```

<Warning>
Monitor gradient norms for both classical and quantum components. Quantum gradients can be much smaller due to barren plateaus, requiring adaptive learning rates or gradient scaling.
</Warning>

## 6. Deployment Considerations

Deploying hybrid quantum RL systems requires balancing quantum circuit calls with real-time decision making.

<Formula>
Latency Budget:
$$t_\text{total} = t_\text{classical} + t_\text{quantum} + t_\text{communication}$$

For real-time control: $t_\text{total} < t_\text{decision}$
</Formula>

### Implementation

```python
class DeployableHybridAgent:
    """
    Production-ready hybrid quantum RL agent with caching and fallback.
    """

    def __init__(self, state_dim, n_actions, n_qubits=4):
        self.state_dim = state_dim
        self.n_actions = n_actions
        self.n_qubits = n_qubits

        # Quantum policy
        self.quantum_params = np.random.randn(3, n_qubits, 3) * 0.1

        # Classical fallback policy
        self.classical_policy = {
            'W1': np.random.randn(32, state_dim) * 0.1,
            'b1': np.zeros(32),
            'W2': np.random.randn(n_actions, 32) * 0.1,
            'b2': np.zeros(n_actions)
        }

        # State encoder
        self.encoder = {
            'W': np.random.randn(n_qubits, state_dim) * 0.1,
            'b': np.zeros(n_qubits)
        }

        # Caching
        self.cache = {}
        self.cache_size = 1000
        self.cache_hits = 0
        self.cache_misses = 0

        # Deployment configuration
        self.use_quantum = True
        self.quantum_timeout_ms = 100
        self.fallback_threshold = 0.9  # Use fallback if confidence < threshold

    def encode_state(self, state):
        """Encode state for quantum processing."""
        return np.tanh(self.encoder['W'] @ state + self.encoder['b'])

    def state_hash(self, state, precision=2):
        """Create hash for state caching."""
        discretized = np.round(state, precision)
        return tuple(discretized.tolist())

    def quantum_policy(self, encoded_state):
        """Quantum policy forward pass."""
        n_qubits = self.n_qubits
        dim = 2 ** n_qubits

        psi = np.zeros(dim, dtype=complex)
        psi[0] = 1.0

        # Encoding
        for i in range(n_qubits):
            angle = encoded_state[i] * np.pi
            psi = self._apply_ry(psi, angle, i)

        # Variational layers
        for layer in range(3):
            for i in range(n_qubits):
                params = self.quantum_params[layer, i]
                psi = self._apply_rotation_sequence(psi, params, i)
            for i in range(n_qubits - 1):
                psi = self._apply_cnot(psi, i, i + 1)

        # Extract action probabilities
        probs = np.abs(psi) ** 2
        action_probs = np.zeros(self.n_actions)

        chunk_size = dim // self.n_actions
        for a in range(self.n_actions):
            action_probs[a] = np.sum(probs[a * chunk_size:(a + 1) * chunk_size])

        return action_probs / (action_probs.sum() + 1e-8)

    def _apply_ry(self, psi, angle, qubit):
        """Apply RY gate."""
        dim = len(psi)
        new_psi = np.zeros(dim, dtype=complex)
        c, s = np.cos(angle/2), np.sin(angle/2)

        for i in range(dim):
            bit = (i >> qubit) & 1
            i_f = i ^ (1 << qubit)
            if bit == 0:
                new_psi[i] += c * psi[i]
                new_psi[i_f] += s * psi[i]
            else:
                new_psi[i] += c * psi[i]
                new_psi[i_f] -= s * psi[i]
        return new_psi

    def _apply_rotation_sequence(self, psi, params, qubit):
        """Apply Rz-Ry-Rz sequence."""
        dim = len(psi)

        for angle, axis in zip(params, ['z', 'y', 'z']):
            new_psi = np.zeros(dim, dtype=complex)

            for i in range(dim):
                bit = (i >> qubit) & 1
                i_f = i ^ (1 << qubit)

                if axis == 'z':
                    phase = np.exp(-1j * angle/2) if bit == 0 else np.exp(1j * angle/2)
                    new_psi[i] = phase * psi[i]
                else:  # y
                    c, s = np.cos(angle/2), np.sin(angle/2)
                    if bit == 0:
                        new_psi[i] += c * psi[i]
                        new_psi[i_f] += s * psi[i]
                    else:
                        new_psi[i] += c * psi[i]
                        new_psi[i_f] -= s * psi[i]
            psi = new_psi
        return psi

    def _apply_cnot(self, psi, control, target):
        """Apply CNOT gate."""
        dim = len(psi)
        new_psi = np.zeros(dim, dtype=complex)

        for i in range(dim):
            if (i >> control) & 1:
                new_psi[i ^ (1 << target)] = psi[i]
            else:
                new_psi[i] = psi[i]
        return new_psi

    def classical_fallback(self, state):
        """Classical policy for fallback."""
        h = np.maximum(0, self.classical_policy['W1'] @ state +
                       self.classical_policy['b1'])
        logits = self.classical_policy['W2'] @ h + self.classical_policy['b2']
        probs = np.exp(logits - np.max(logits))
        return probs / probs.sum()

    def get_action_probs(self, state):
        """Get action probabilities with caching and fallback."""
        state_key = self.state_hash(state)

        # Check cache
        if state_key in self.cache:
            self.cache_hits += 1
            return self.cache[state_key]

        self.cache_misses += 1

        if self.use_quantum:
            encoded = self.encode_state(state)
            probs = self.quantum_policy(encoded)

            # Check confidence
            max_prob = np.max(probs)
            if max_prob < self.fallback_threshold:
                # Blend with classical fallback
                classical_probs = self.classical_fallback(state)
                probs = 0.5 * probs + 0.5 * classical_probs
        else:
            probs = self.classical_fallback(state)

        # Update cache
        if len(self.cache) >= self.cache_size:
            # Remove oldest entry
            oldest_key = next(iter(self.cache))
            del self.cache[oldest_key]
        self.cache[state_key] = probs

        return probs

    def select_action(self, state, deterministic=False):
        """Select action for deployment."""
        probs = self.get_action_probs(state)

        if deterministic:
            return np.argmax(probs)
        return np.random.choice(self.n_actions, p=probs)

    def get_cache_stats(self):
        """Get cache performance statistics."""
        total = self.cache_hits + self.cache_misses
        hit_rate = self.cache_hits / total if total > 0 else 0
        return {
            'hits': self.cache_hits,
            'misses': self.cache_misses,
            'hit_rate': hit_rate,
            'cache_size': len(self.cache)
        }

    def save_model(self, path):
        """Save model parameters."""
        model_data = {
            'quantum_params': self.quantum_params,
            'classical_policy': self.classical_policy,
            'encoder': self.encoder
        }
        np.savez(path, **model_data)
        print(f"Model saved to {path}")

    def load_model(self, path):
        """Load model parameters."""
        data = np.load(path, allow_pickle=True)
        self.quantum_params = data['quantum_params']
        self.classical_policy = data['classical_policy'].item()
        self.encoder = data['encoder'].item()
        print(f"Model loaded from {path}")

# Deployment example
agent = DeployableHybridAgent(state_dim=8, n_actions=4, n_qubits=4)

print("Testing deployment...")
for i in range(100):
    state = np.random.randn(8)
    action = agent.select_action(state)

stats = agent.get_cache_stats()
print(f"\nCache statistics:")
print(f"  Hits: {stats['hits']}, Misses: {stats['misses']}")
print(f"  Hit rate: {stats['hit_rate']:.2%}")
print(f"  Cache size: {stats['cache_size']}")
```

<Tip>
For real-time applications, implement state discretization and caching to avoid redundant quantum circuit executions. Use classical fallback policies when quantum latency exceeds decision-time requirements.
</Tip>

## Summary

Hybrid quantum-classical RL systems combine the strengths of both paradigms:

1. **Architecture patterns**: Classical preprocessing with quantum cores, or quantum feature extractors with classical heads
2. **Actor-critic**: Quantum actors for policy representation, classical critics for stable value estimation
3. **Quantum DQN**: Quantum value networks with experience replay and target networks
4. **Feature exchange**: Bidirectional information flow between classical and quantum components
5. **Training optimization**: Separate learning rates, gradient clipping, and coordinated updates
6. **Deployment**: Caching, fallback policies, and latency management for real-time operation

The key to successful hybrid systems is careful interface design and component balancing.
