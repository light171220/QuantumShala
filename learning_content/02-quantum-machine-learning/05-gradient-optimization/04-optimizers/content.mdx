# Classical Optimizers for VQAs

Variational quantum algorithms rely on classical optimizers to tune circuit parameters. Choosing the right optimizer significantly impacts convergence.

## Learning Objectives

After completing this lesson, you will be able to:

- Implement gradient descent variants
- Apply momentum and adaptive learning rates
- Use gradient-free optimization
- Compare optimizer performance
- Choose optimizers for VQAs

## 1. Gradient Descent Variants

From basic gradient descent to advanced variants.

```python
import numpy as np

def gradient_descent_variants():
    """
    Overview of gradient descent variants.
    """
    print("Gradient Descent Variants:")
    print("=" * 50)

    class GradientDescent:
        """Basic gradient descent."""
        def __init__(self, lr=0.01):
            self.lr = lr

        def step(self, params, gradient):
            return params - self.lr * gradient

    class MomentumGD:
        """Gradient descent with momentum."""
        def __init__(self, lr=0.01, momentum=0.9):
            self.lr = lr
            self.momentum = momentum
            self.velocity = None

        def step(self, params, gradient):
            if self.velocity is None:
                self.velocity = np.zeros_like(params)
            self.velocity = self.momentum * self.velocity - self.lr * gradient
            return params + self.velocity

    class NesterovMomentum:
        """Nesterov accelerated gradient."""
        def __init__(self, lr=0.01, momentum=0.9):
            self.lr = lr
            self.momentum = momentum
            self.velocity = None

        def step(self, params, gradient):
            if self.velocity is None:
                self.velocity = np.zeros_like(params)
            v_prev = self.velocity.copy()
            self.velocity = self.momentum * self.velocity - self.lr * gradient
            return params - self.momentum * v_prev + (1 + self.momentum) * self.velocity

    # Test function
    def rosenbrock(params):
        x, y = params
        return (1 - x)**2 + 100 * (y - x**2)**2

    def rosenbrock_grad(params):
        x, y = params
        dx = -2*(1-x) - 400*x*(y - x**2)
        dy = 200*(y - x**2)
        return np.array([dx, dy])

    print("\n  Optimization on Rosenbrock function:")
    print("-" * 40)

    optimizers = {
        'Vanilla GD': GradientDescent(lr=0.001),
        'Momentum': MomentumGD(lr=0.001, momentum=0.9),
        'Nesterov': NesterovMomentum(lr=0.001, momentum=0.9)
    }

    for name, opt in optimizers.items():
        params = np.array([-1.0, 1.0])
        for _ in range(1000):
            grad = rosenbrock_grad(params)
            params = opt.step(params, grad)
        print(f"  {name}: final params = {params.round(3)}, cost = {rosenbrock(params):.4f}")

gradient_descent_variants()
```

<Definition term="Momentum">
A technique that accumulates gradient history to smooth updates and accelerate convergence: v_t = μv_{t-1} + η∇f, where μ is the momentum coefficient.
</Definition>

## 2. Adaptive Learning Rate Methods

Optimizers that automatically adjust learning rates.

```python
import numpy as np

class AdaGrad:
    """Adaptive gradient with per-parameter learning rates."""
    def __init__(self, lr=0.1, epsilon=1e-8):
        self.lr = lr
        self.epsilon = epsilon
        self.cache = None

    def step(self, params, gradient):
        if self.cache is None:
            self.cache = np.zeros_like(params)
        self.cache += gradient**2
        return params - self.lr * gradient / (np.sqrt(self.cache) + self.epsilon)

class RMSprop:
    """RMSprop with exponential moving average."""
    def __init__(self, lr=0.01, decay=0.9, epsilon=1e-8):
        self.lr = lr
        self.decay = decay
        self.epsilon = epsilon
        self.cache = None

    def step(self, params, gradient):
        if self.cache is None:
            self.cache = np.zeros_like(params)
        self.cache = self.decay * self.cache + (1 - self.decay) * gradient**2
        return params - self.lr * gradient / (np.sqrt(self.cache) + self.epsilon)

class Adam:
    """Adam optimizer combining momentum and adaptive rates."""
    def __init__(self, lr=0.001, beta1=0.9, beta2=0.999, epsilon=1e-8):
        self.lr = lr
        self.beta1 = beta1
        self.beta2 = beta2
        self.epsilon = epsilon
        self.m = None
        self.v = None
        self.t = 0

    def step(self, params, gradient):
        if self.m is None:
            self.m = np.zeros_like(params)
            self.v = np.zeros_like(params)

        self.t += 1
        self.m = self.beta1 * self.m + (1 - self.beta1) * gradient
        self.v = self.beta2 * self.v + (1 - self.beta2) * gradient**2

        # Bias correction
        m_hat = self.m / (1 - self.beta1**self.t)
        v_hat = self.v / (1 - self.beta2**self.t)

        return params - self.lr * m_hat / (np.sqrt(v_hat) + self.epsilon)

# Comparison
print("Adaptive Learning Rate Methods:")
print("=" * 50)

def noisy_quadratic(params):
    return np.sum(params**2) + np.random.normal(0, 0.1)

def noisy_gradient(params):
    return 2 * params + np.random.normal(0, 0.1, size=len(params))

np.random.seed(42)
n_steps = 100

optimizers = {
    'AdaGrad': AdaGrad(lr=0.5),
    'RMSprop': RMSprop(lr=0.1),
    'Adam': Adam(lr=0.1)
}

print("\n  Convergence with noisy gradients:")
print("-" * 40)

for name, opt in optimizers.items():
    params = np.array([1.0, 2.0, 3.0])
    history = []
    for _ in range(n_steps):
        history.append(np.sum(params**2))
        grad = noisy_gradient(params)
        params = opt.step(params, grad)
    print(f"  {name}: final |params|² = {np.sum(params**2):.4f}")
```

<Tip>
Adam is often a good default choice for VQAs due to its robustness to noisy gradients and automatic learning rate adjustment.
</Tip>

## 3. Gradient-Free Optimizers

When gradients are expensive or unavailable.

```python
import numpy as np

def gradient_free_optimizers():
    """
    Gradient-free optimization methods.
    """
    print("Gradient-Free Optimizers:")
    print("=" * 50)

    class NelderMead:
        """Simplex-based optimization."""
        def __init__(self, n_params, step=0.5):
            self.n = n_params
            self.step = step
            self.simplex = None

        def initialize(self, x0):
            self.simplex = [x0.copy()]
            for i in range(self.n):
                x = x0.copy()
                x[i] += self.step
                self.simplex.append(x)

        def optimize(self, cost_fn, x0, n_iters=100):
            self.initialize(x0)

            for _ in range(n_iters):
                # Sort by cost
                costs = [cost_fn(x) for x in self.simplex]
                order = np.argsort(costs)
                self.simplex = [self.simplex[i] for i in order]

                # Centroid (excluding worst)
                centroid = np.mean(self.simplex[:-1], axis=0)

                # Reflection
                reflected = centroid + (centroid - self.simplex[-1])
                f_r = cost_fn(reflected)

                if costs[0] <= f_r < costs[-2]:
                    self.simplex[-1] = reflected
                elif f_r < costs[0]:
                    # Expansion
                    expanded = centroid + 2*(reflected - centroid)
                    if cost_fn(expanded) < f_r:
                        self.simplex[-1] = expanded
                    else:
                        self.simplex[-1] = reflected
                else:
                    # Contraction
                    contracted = centroid + 0.5*(self.simplex[-1] - centroid)
                    if cost_fn(contracted) < costs[-1]:
                        self.simplex[-1] = contracted
                    else:
                        # Shrink
                        for i in range(1, len(self.simplex)):
                            self.simplex[i] = self.simplex[0] + 0.5*(self.simplex[i] - self.simplex[0])

            return self.simplex[0]

    class COBYLA:
        """Simplified COBYLA-like trust region."""
        def __init__(self, rhobeg=0.5, rhoend=1e-4):
            self.rhobeg = rhobeg
            self.rhoend = rhoend

        def optimize(self, cost_fn, x0, n_iters=100):
            x = x0.copy()
            rho = self.rhobeg

            for _ in range(n_iters):
                if rho < self.rhoend:
                    break

                # Try random directions
                best_x = x.copy()
                best_cost = cost_fn(x)

                for _ in range(2 * len(x)):
                    direction = np.random.randn(len(x))
                    direction /= np.linalg.norm(direction)
                    x_new = x + rho * direction
                    cost_new = cost_fn(x_new)

                    if cost_new < best_cost:
                        best_x = x_new
                        best_cost = cost_new

                if np.array_equal(best_x, x):
                    rho *= 0.5
                else:
                    x = best_x

            return x

    # Test
    def cost(params):
        return np.sum((params - 1)**2)

    print("\n  Test on simple quadratic:")
    print("-" * 40)

    x0 = np.array([0.0, 0.0, 0.0])

    nm = NelderMead(3)
    result_nm = nm.optimize(cost, x0, n_iters=100)
    print(f"  Nelder-Mead: {result_nm.round(3)}, cost = {cost(result_nm):.4f}")

    cb = COBYLA()
    result_cb = cb.optimize(cost, x0, n_iters=200)
    print(f"  COBYLA-like: {result_cb.round(3)}, cost = {cost(result_cb):.4f}")

gradient_free_optimizers()
```

<Warning>
Gradient-free methods typically require more function evaluations than gradient-based methods. Use them when gradients are very expensive or unavailable.
</Warning>

## 5. Rotosolve: Quantum-Native Optimization

A gradient-free method exploiting quantum gate structure.

```python
import numpy as np

def rotosolve_optimizer():
    """
    Rotosolve: exact 1D optimization for quantum gates.
    """
    print("Rotosolve Optimizer:")
    print("=" * 50)

    print("\n1. Key Insight:")
    print("-" * 40)
    print("  For Pauli rotation gates, the cost function is:")
    print("  f(θ) = A sin(θ) + B cos(θ) + C")
    print()
    print("  This can be minimized analytically!")
    print("  θ* = arctan2(-A, B) - π/2")

    def rotosolve_step(cost_fn, params, param_idx):
        """
        Optimize one parameter using Rotosolve.

        Args:
            cost_fn: Cost function
            params: Current parameters
            param_idx: Which parameter to optimize

        Returns:
            Optimal value for params[param_idx]
        """
        # Evaluate at three points
        p0 = params.copy(); p0[param_idx] = 0
        p1 = params.copy(); p1[param_idx] = np.pi/2
        p2 = params.copy(); p2[param_idx] = np.pi

        f0 = cost_fn(p0)
        f1 = cost_fn(p1)
        f2 = cost_fn(p2)

        # Solve for A, B, C
        # f(0) = B + C = f0
        # f(π/2) = A + C = f1
        # f(π) = -B + C = f2

        C = (f0 + f2) / 2
        B = f0 - C
        A = f1 - C

        # Optimal angle
        theta_opt = np.arctan2(-A, B) - np.pi/2

        return theta_opt

    def rotosolve(cost_fn, initial_params, n_sweeps=10):
        """Full Rotosolve optimization."""
        params = initial_params.copy()
        history = [cost_fn(params)]

        for _ in range(n_sweeps):
            for i in range(len(params)):
                params[i] = rotosolve_step(cost_fn, params, i)
            history.append(cost_fn(params))

        return params, history

    # Test
    print("\n\n2. Example Optimization:")
    print("-" * 40)

    def quantum_cost(params):
        """Simulated quantum cost with sinusoidal structure."""
        return 0.5 * (2 - np.cos(params[0]) * np.cos(params[1]) -
                      np.sin(params[0]) * np.sin(params[1]))

    initial = np.array([1.0, 2.0])
    final, history = rotosolve(quantum_cost, initial, n_sweeps=5)

    print(f"  Initial: params={initial}, cost={history[0]:.4f}")
    print(f"  Final: params={final.round(3)}, cost={history[-1]:.4f}")

    print("\n\n3. Advantages:")
    print("-" * 40)
    print("  • Exact 1D minimization (no learning rate)")
    print("  • Only 3 evaluations per parameter")
    print("  • Works well for deep circuits")
    print("  • Guaranteed progress each sweep")

rotosolve_optimizer()
```

<Tip>
Rotosolve is particularly effective for VQE and QAOA where the cost function has the expected sinusoidal structure from Pauli rotation gates.
</Tip>

## 6. Optimizer Comparison

Benchmarking optimizers on a VQA task.

```python
import numpy as np

def optimizer_benchmark():
    """
    Benchmark different optimizers on VQA-like task.
    """
    print("Optimizer Benchmark:")
    print("=" * 50)

    # Simulated VQE-like cost with noise
    def vqa_cost(params, noise_std=0.01):
        clean = np.sum(np.sin(params)**2) + 0.5 * np.prod(np.cos(params[:2]))
        return clean + np.random.normal(0, noise_std)

    def gradient(cost_fn, params, eps=0.01):
        grad = np.zeros(len(params))
        for i in range(len(params)):
            p_plus = params.copy(); p_plus[i] += eps
            p_minus = params.copy(); p_minus[i] -= eps
            grad[i] = (cost_fn(p_plus) - cost_fn(p_minus)) / (2 * eps)
        return grad

    n_params = 6
    n_steps = 100
    n_trials = 5

    results = {}

    # Adam
    for trial in range(n_trials):
        np.random.seed(trial)
        params = np.random.uniform(0, 2*np.pi, n_params)
        m = np.zeros(n_params)
        v = np.zeros(n_params)
        lr, b1, b2 = 0.1, 0.9, 0.999

        history = []
        for t in range(1, n_steps + 1):
            history.append(vqa_cost(params, noise_std=0))
            g = gradient(vqa_cost, params)
            m = b1 * m + (1-b1) * g
            v = b2 * v + (1-b2) * g**2
            m_hat = m / (1 - b1**t)
            v_hat = v / (1 - b2**t)
            params = params - lr * m_hat / (np.sqrt(v_hat) + 1e-8)

        if 'Adam' not in results:
            results['Adam'] = []
        results['Adam'].append(history)

    # Vanilla GD
    for trial in range(n_trials):
        np.random.seed(trial)
        params = np.random.uniform(0, 2*np.pi, n_params)
        lr = 0.1

        history = []
        for _ in range(n_steps):
            history.append(vqa_cost(params, noise_std=0))
            g = gradient(vqa_cost, params)
            params = params - lr * g

        if 'Vanilla GD' not in results:
            results['Vanilla GD'] = []
        results['Vanilla GD'].append(history)

    # Summary
    print("\n  Final Cost Comparison (mean ± std):")
    print("-" * 40)

    for name, trials in results.items():
        final_costs = [h[-1] for h in trials]
        print(f"  {name}: {np.mean(final_costs):.4f} ± {np.std(final_costs):.4f}")

optimizer_benchmark()
```

## Summary

Choosing the right optimizer for VQAs:

1. **Gradient Descent**: Basic but often unstable
2. **Momentum**: Smooths updates, faster convergence
3. **Adam**: Robust default for noisy quantum gradients
4. **COBYLA/Nelder-Mead**: When gradients unavailable
5. **Rotosolve**: Quantum-native, exact 1D optimization
6. **SPSA**: Low evaluation cost per step

Match the optimizer to your specific constraints and circuit properties.
