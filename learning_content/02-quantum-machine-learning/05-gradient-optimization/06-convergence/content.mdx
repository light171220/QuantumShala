# Convergence Analysis

Understanding when and how VQA optimization converges is crucial for efficient training and reliable results.

## Learning Objectives

After completing this lesson, you will be able to:

- Define convergence criteria for VQAs
- Implement stopping conditions
- Diagnose convergence issues
- Handle local minima and plateaus
- Analyze optimization trajectories

## 1. Convergence Criteria

Different ways to determine if optimization has converged.

```python

    criteria = [

    print("\n\n3. Parameter-Based Criteria:")

    print("  • ||θ_new - θ_old|| < ε: Parameters stopped moving")

    print("\n\n4. Practical VQA Thresholds:")
    print("-" * 40)

    thresholds = [
        ("Cost improvement", "1e-6 to 1e-4"),
        ("Gradient norm", "1e-4 to 1e-2"),
        ("Parameter change", "1e-5 to 1e-3"),
        ("Consecutive patience", "10-50 steps"),
    ]

    for criterion, threshold in thresholds:
        print(f"  {criterion}: {threshold}")

convergence_criteria()
```

<Definition term="Convergence">
The state where optimization has effectively stopped making meaningful progress, either by reaching an optimum or by becoming stuck. Proper criteria distinguish between these cases.
</Definition>

## 2. Stopping Conditions

Implementing robust stopping logic.

```python
import numpy as np

class ConvergenceChecker:
    """
    Monitor optimization and detect convergence.
    """

    def __init__(self, cost_tol=1e-6, grad_tol=1e-4, param_tol=1e-5,
                 patience=20, min_steps=50, max_steps=1000):
        self.cost_tol = cost_tol
        self.grad_tol = grad_tol
        self.param_tol = param_tol
        self.patience = patience
        self.min_steps = min_steps
        self.max_steps = max_steps

        self.cost_history = []
        self.grad_norms = []
        self.param_history = []
        self.best_cost = float('inf')
        self.steps_without_improvement = 0

    def check(self, cost, gradient=None, params=None, step=0):
        """
        Check convergence criteria.

        Returns:
            (converged: bool, reason: str or None)
        """
        self.cost_history.append(cost)

        if gradient is not None:
            self.grad_norms.append(np.linalg.norm(gradient))

        if params is not None:
            self.param_history.append(params.copy())

        # Track improvement
        if cost < self.best_cost - self.cost_tol:
            self.best_cost = cost
            self.steps_without_improvement = 0
        else:
            self.steps_without_improvement += 1

        # Don't stop before minimum steps
        if step < self.min_steps:
            return False, None

        # Max steps reached
        if step >= self.max_steps:
            return True, "max_steps"

        # Patience exhausted
        if self.steps_without_improvement >= self.patience:
            return True, "patience"

        # Gradient norm
        if gradient is not None and self.grad_norms[-1] < self.grad_tol:
            return True, "gradient"

        # Parameter change
        if params is not None and len(self.param_history) >= 2:
            param_change = np.linalg.norm(
                self.param_history[-1] - self.param_history[-2])
            if param_change < self.param_tol:
                return True, "parameters"

        return False, None

# Demonstration
print("Convergence Checker:")
print("=" * 50)

checker = ConvergenceChecker(patience=5, min_steps=10, max_steps=100)

# Simulate optimization
np.random.seed(42)
params = np.array([2.0, 2.0])

for step in range(100):
    cost = np.sum(params**2) + np.random.normal(0, 0.01)
    gradient = 2 * params + np.random.normal(0, 0.1, 2)
    params = params - 0.1 * gradient

    converged, reason = checker.check(cost, gradient, params, step)

    if converged:
        print(f"\nConverged at step {step}!")
        print(f"Reason: {reason}")
        print(f"Final cost: {cost:.6f}")
        break

    if step % 20 == 0:
        print(f"Step {step}: cost = {cost:.4f}, |grad| = {np.linalg.norm(gradient):.4f}")
```

<Tip>
Always use patience-based stopping rather than checking a single condition. Noise can cause temporary false signals.
</Tip>

## 3. Diagnosing Convergence Issues

Identifying and fixing common problems.

```python
import numpy as np

def diagnose_convergence():
    """
    Common convergence issues and diagnostics.
    """
    print("Diagnosing Convergence Issues:")
    print("=" * 50)

    print("\n1. Issue: Loss Not Decreasing")
    print("-" * 40)

    symptoms = [
        ("Flat loss curve", "LR too small or barren plateau"),
        ("Oscillating loss", "LR too large"),
        ("Loss increasing", "LR way too large or bug"),
    ]

    for symptom, cause in symptoms:
        print(f"  {symptom}: {cause}")

    print("\n\n2. Issue: Stuck at Poor Solution")
    print("-" * 40)

    solutions = [
        ("Try different initialization", "Random restarts"),
        ("Increase LR temporarily", "Escape local minimum"),
        ("Use momentum", "Overcome flat regions"),
        ("Try Rotosolve", "Exact 1D optimization"),
    ]

    for approach, description in solutions:
        print(f"  {approach}: {description}")

    print("\n\n3. Diagnostic Metrics:")
    print("-" * 40)

    def compute_diagnostics(cost_history, grad_history):
        """Compute diagnostic metrics."""
        diagnostics = {}

        # Loss statistics
        costs = np.array(cost_history[-50:])  # Last 50
        diagnostics['loss_mean'] = np.mean(costs)
        diagnostics['loss_std'] = np.std(costs)
        diagnostics['loss_trend'] = np.polyfit(range(len(costs)), costs, 1)[0]

        # Gradient statistics
        if grad_history:
            grads = np.array(grad_history[-50:])
            diagnostics['grad_mean'] = np.mean(grads)
            diagnostics['grad_std'] = np.std(grads)

        # Progress ratio
        if len(cost_history) > 10:
            early_mean = np.mean(cost_history[:10])
            late_mean = np.mean(cost_history[-10:])
            diagnostics['progress_ratio'] = (early_mean - late_mean) / (early_mean + 1e-10)

        return diagnostics

    # Example
    np.random.seed(42)
    cost_history = [1.0 - 0.5 * (1 - np.exp(-i/20)) + np.random.normal(0, 0.02)
                   for i in range(100)]
    grad_history = [0.5 * np.exp(-i/30) + np.random.normal(0, 0.05)
                   for i in range(100)]

    diag = compute_diagnostics(cost_history, grad_history)

    print("\n  Example diagnostics:")
    for key, value in diag.items():
        print(f"    {key}: {value:.4f}")

    print("\n\n4. Health Check Interpretation:")
    print("-" * 40)

    interpretations = [
        ("loss_trend > 0", "Loss increasing - reduce LR"),
        ("loss_std / loss_mean > 0.1", "High variance - increase shots or reduce LR"),
        ("grad_mean < 1e-6", "Near stationary - may be converged or stuck"),
        ("progress_ratio < 0.1", "Slow progress - consider restart"),
    ]

    for condition, action in interpretations:
        print(f"  If {condition}:")
        print(f"    → {action}")

diagnose_convergence()
```

## 4. Local Minima and Plateaus

Strategies for handling non-global optima.

```python
import numpy as np

def local_minima_strategies():
    """
    Strategies for escaping local minima.
    """
    print("Handling Local Minima and Plateaus:")
    print("=" * 50)

    print("\n1. Multi-Start Optimization:")
    print("-" * 40)

    def multi_start_optimize(cost_fn, grad_fn, n_params, n_starts=5, n_steps=100, lr=0.1):
        """Run optimization from multiple random starts."""
        best_cost = float('inf')
        best_params = None

        for start in range(n_starts):
            params = np.random.uniform(0, 2*np.pi, n_params)
            for _ in range(n_steps):
                grad = grad_fn(params)
                params = params - lr * grad
            cost = cost_fn(params)
            if cost < best_cost:
                best_cost = cost
                best_params = params.copy()

        return best_params, best_cost

    # Example
    def multi_modal_cost(params):
        return np.sum(np.sin(params)**2) + 0.1 * np.sum(np.cos(2*params))

    def multi_modal_grad(params):
        return 2 * np.sin(params) * np.cos(params) - 0.2 * np.sin(2*params)

    print("\n  Running multi-start optimization...")
    np.random.seed(42)
    best_params, best_cost = multi_start_optimize(
        multi_modal_cost, multi_modal_grad, 3, n_starts=10)
    print(f"  Best cost: {best_cost:.6f}")
    print(f"  Best params: {best_params.round(3)}")

    print("\n\n2. Perturbation and Restart:")
    print("-" * 40)

    def perturb_and_continue(cost_fn, grad_fn, params, n_steps=50, lr=0.1, noise=0.5):
        """Perturb current solution and continue optimization."""
        perturbed = params + np.random.normal(0, noise, len(params))
        for _ in range(n_steps):
            grad = grad_fn(perturbed)
            perturbed = perturbed - lr * grad
        return perturbed, cost_fn(perturbed)

    current = np.array([0.5, 0.5, 0.5])
    new_params, new_cost = perturb_and_continue(
        multi_modal_cost, multi_modal_grad, current)
    print(f"  After perturbation: cost = {new_cost:.6f}")

    print("\n\n3. Simulated Annealing-like Approach:")
    print("-" * 40)

    def annealing_optimize(cost_fn, grad_fn, params, n_steps=200):
        """Gradient descent with decaying noise."""
        best_params = params.copy()
        best_cost = cost_fn(params)

        for step in range(n_steps):
            temperature = 1.0 * (1 - step / n_steps)
            noise = np.random.normal(0, temperature, len(params))
            grad = grad_fn(params)
            params = params - 0.1 * grad + 0.1 * noise

            cost = cost_fn(params)
            if cost < best_cost:
                best_cost = cost
                best_params = params.copy()

        return best_params, best_cost

    params0 = np.array([2.0, 2.0, 2.0])
    annealed, annealed_cost = annealing_optimize(
        multi_modal_cost, multi_modal_grad, params0)
    print(f"  Annealing result: cost = {annealed_cost:.6f}")

local_minima_strategies()
```

<Warning>
VQA landscapes often have many local minima. Always run multiple random initializations and compare results.
</Warning>

## 5. Trajectory Analysis

Visualizing and understanding optimization paths.

```python
import numpy as np

def trajectory_analysis():
    """
    Analyze optimization trajectories.
    """
    print("Trajectory Analysis:")
    print("=" * 50)

    class TrajectoryRecorder:
        """Record and analyze optimization trajectory."""

        def __init__(self):
            self.params_history = []
            self.cost_history = []
            self.grad_history = []

        def record(self, params, cost, gradient=None):
            self.params_history.append(params.copy())
            self.cost_history.append(cost)
            if gradient is not None:
                self.grad_history.append(gradient.copy())

        def analyze(self):
            """Compute trajectory statistics."""
            params = np.array(self.params_history)
            costs = np.array(self.cost_history)

            analysis = {
                'total_steps': len(costs),
                'initial_cost': costs[0],
                'final_cost': costs[-1],
                'best_cost': np.min(costs),
                'cost_reduction': (costs[0] - costs[-1]) / costs[0],
            }

            # Path length
            diffs = np.diff(params, axis=0)
            path_length = np.sum(np.linalg.norm(diffs, axis=1))
            analysis['path_length'] = path_length

            # Euclidean distance start to end
            direct_distance = np.linalg.norm(params[-1] - params[0])
            analysis['direct_distance'] = direct_distance

            # Efficiency (direct / path)
            analysis['path_efficiency'] = direct_distance / (path_length + 1e-10)

            # Oscillation metric
            if len(costs) > 3:
                cost_diffs = np.diff(costs)
                sign_changes = np.sum(np.diff(np.sign(cost_diffs)) != 0)
                analysis['oscillation_count'] = sign_changes

            return analysis

    # Run example optimization
    print("\n1. Recording Optimization:")
    print("-" * 40)

    def cost(p):
        return 0.5 * (p[0]**2 + 10 * p[1]**2)

    def grad(p):
        return np.array([p[0], 10 * p[1]])

    recorder = TrajectoryRecorder()
    params = np.array([5.0, 5.0])
    lr = 0.05

    for step in range(100):
        c = cost(params)
        g = grad(params)
        recorder.record(params, c, g)
        params = params - lr * g

    analysis = recorder.analyze()

    print("\n  Trajectory Analysis:")
    for key, value in analysis.items():
        if isinstance(value, float):
            print(f"    {key}: {value:.4f}")
        else:
            print(f"    {key}: {value}")

    print("\n\n2. Interpretation:")
    print("-" * 40)

    interpretations = [
        ("Path efficiency < 0.3", "Zigzag path - consider momentum"),
        ("Oscillation count > n/3", "Unstable - reduce LR"),
        ("Cost reduction < 0.5", "Poor convergence - check setup"),
        ("Path length >> direct", "Inefficient route"),
    ]

    for metric, advice in interpretations:
        print(f"  {metric}:")
        print(f"    → {advice}")

trajectory_analysis()
```

## Summary

Convergence analysis is essential for reliable VQA results:

1. **Criteria**: Cost, gradient, and parameter-based checks
2. **Stopping**: Use patience and multiple conditions
3. **Diagnostics**: Track loss trend, variance, gradient norm
4. **Local Minima**: Multi-start, perturbation, annealing
5. **Trajectory**: Path length, efficiency, oscillation
6. **Guarantees**: Practical rather than theoretical for VQAs

Always validate convergence with multiple initializations and physical intuition.
