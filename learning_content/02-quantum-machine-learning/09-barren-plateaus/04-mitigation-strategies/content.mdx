# Barren Plateau Mitigation Strategies

Once barren plateaus are detected, several strategies can help recover trainability. This lesson covers practical mitigation techniques.

## Smart Initialization

Initializing parameters to create near-identity circuits avoids the barren plateau at the start of training.

<Definition>
**Identity Initialization**:

Initialize parameters such that $U(\theta_0) \approx I$, making the initial state close to the input state. This preserves gradient signal in early training.
</Definition>

```python
import numpy as np

class IdentityInitializer:
    """Initialize circuits near identity transformation."""

    def __init__(self, n_qubits: int, n_layers: int):
        self.n_qubits = n_qubits
        self.n_layers = n_layers
        self.dim = 2 ** n_qubits

    def identity_init(self, scale: float = 0.01) -> np.ndarray:
        """Initialize near identity (small random angles)."""
        n_params = 3 * self.n_qubits * self.n_layers
        return np.random.uniform(-scale, scale, n_params)

    def zero_init(self) -> np.ndarray:
        """Initialize all parameters to zero."""
        n_params = 3 * self.n_qubits * self.n_layers
        return np.zeros(n_params)

    def alternating_init(self, scale: float = 0.01) -> np.ndarray:
        """Alternate positive/negative to cancel rotations."""
        n_params = 3 * self.n_qubits * self.n_layers
        params = np.random.uniform(0, scale, n_params)
        # Alternate signs
        params[1::2] *= -1
        return params

    def block_identity_init(self) -> np.ndarray:
        """Initialize each layer to be identity."""
        params = []
        for layer in range(self.n_layers):
            for q in range(self.n_qubits):
                # RX, RY, RZ each at 0 gives identity
                params.extend([0.0, 0.0, 0.0])
        return np.array(params) + np.random.uniform(-0.01, 0.01, len(params))

def compare_initializations():
    """Compare gradient variance for different initializations."""
    from typing import Callable

    print("Initialization Strategy Comparison")
    print("=" * 60)

    n_qubits, n_layers = 6, 6
    initializer = IdentityInitializer(n_qubits, n_layers)
    dim = 2 ** n_qubits
    n_params = 3 * n_qubits * n_layers

    def circuit(params):
        state = np.zeros(dim, dtype=complex)
        state[0] = 1
        param_idx = 0

        for _ in range(n_layers):
            for q in range(n_qubits):
                # RX
                theta = params[param_idx]
                c, s = np.cos(theta/2), np.sin(theta/2)
                new_state = np.zeros_like(state)
                for z in range(dim):
                    z_flip = z ^ (1 << (n_qubits - 1 - q))
                    new_state[z] = c * state[z] - 1j * s * state[z_flip]
                state = new_state
                param_idx += 1

                # RY
                theta = params[param_idx]
                c, s = np.cos(theta/2), np.sin(theta/2)
                new_state = np.zeros_like(state)
                for z in range(dim):
                    z_flip = z ^ (1 << (n_qubits - 1 - q))
                    bit = (z >> (n_qubits - 1 - q)) & 1
                    new_state[z] = c * state[z] + (1-2*bit) * s * state[z_flip]
                state = new_state
                param_idx += 1

                # RZ
                theta = params[param_idx]
                for z in range(dim):
                    bit = (z >> (n_qubits - 1 - q)) & 1
                    phase = np.exp(1j*theta/2) if bit else np.exp(-1j*theta/2)
                    state[z] *= phase
                param_idx += 1

            # CNOT chain
            for q in range(n_qubits - 1):
                new_state = np.zeros_like(state)
                for z in range(dim):
                    ctrl = (z >> (n_qubits - 1 - q)) & 1
                    z_new = z ^ (1 << (n_qubits - 1 - (q+1))) if ctrl else z
                    new_state[z_new] += state[z]
                state = new_state

        return state

    def cost(params):
        state = circuit(params)
        c = 0
        for z in range(dim):
            prob = np.abs(state[z])**2
            parity = bin(z).count('1') % 2
            c += prob * (1 - 2 * parity)
        return c

    strategies = [
        ('Random [0, 2π]', lambda: np.random.uniform(0, 2*np.pi, n_params)),
        ('Identity (scale=0.01)', lambda: initializer.identity_init(0.01)),
        ('Identity (scale=0.1)', lambda: initializer.identity_init(0.1)),
        ('Zero + noise', lambda: initializer.zero_init() + np.random.randn(n_params)*0.01),
    ]

    for name, init_fn in strategies:
        gradients = []
        for _ in range(50):
            params = init_fn()
            shift = np.pi / 2
            params_p, params_m = params.copy(), params.copy()
            params_p[0] += shift
            params_m[0] -= shift
            grad = (cost(params_p) - cost(params_m)) / 2
            gradients.append(grad)

        var = np.var(gradients)
        mean_abs = np.mean(np.abs(gradients))
        print(f"{name:<25}: Var={var:.4e}, Mean|grad|={mean_abs:.4f}")

compare_initializations()
```

## Layer-wise Training

Train the circuit layer by layer, adding new layers only after previous ones converge.

<Tip>
Layer-wise training prevents deep circuits from immediately falling into barren plateaus by building up complexity gradually.
</Tip>

```python
import numpy as np

class LayerwiseTrainer:
    """Train circuit layer by layer to avoid barren plateaus."""

    def __init__(self, n_qubits: int, max_layers: int):
        self.n_qubits = n_qubits
        self.max_layers = max_layers
        self.dim = 2 ** n_qubits

    def _circuit(self, params: np.ndarray, n_layers: int) -> np.ndarray:
        """Execute circuit with specified number of layers."""
        state = np.zeros(self.dim, dtype=complex)
        state[0] = 1
        param_idx = 0

        for _ in range(n_layers):
            for q in range(self.n_qubits):
                theta = params[param_idx]
                c, s = np.cos(theta/2), np.sin(theta/2)
                new_state = np.zeros_like(state)
                for z in range(self.dim):
                    z_flip = z ^ (1 << (self.n_qubits - 1 - q))
                    new_state[z] = c * state[z] - s * state[z_flip]
                    new_state[z_flip] = s * state[z] + c * state[z_flip]
                state = new_state
                param_idx += 1

            for q in range(self.n_qubits - 1):
                new_state = np.zeros_like(state)
                for z in range(self.dim):
                    ctrl = (z >> (self.n_qubits - 1 - q)) & 1
                    z_new = z ^ (1 << (self.n_qubits - 1 - (q+1))) if ctrl else z
                    new_state[z_new] += state[z]
                state = new_state

        return state

    def _cost(self, params: np.ndarray, n_layers: int) -> float:
        """Compute cost function."""
        state = self._circuit(params, n_layers)
        c = 0
        for z in range(self.dim):
            prob = np.abs(state[z])**2
            parity = bin(z).count('1') % 2
            c += prob * (1 - 2 * parity)
        return c

    def _gradient(self, params: np.ndarray, n_layers: int,
                  param_idx: int) -> float:
        """Compute gradient for one parameter."""
        shift = np.pi / 2
        params_p, params_m = params.copy(), params.copy()
        params_p[param_idx] += shift
        params_m[param_idx] -= shift
        return (self._cost(params_p, n_layers) -
                self._cost(params_m, n_layers)) / 2

    def train_layerwise(self, n_iterations_per_layer: int = 50,
                        learning_rate: float = 0.1) -> dict:
        """Train circuit layer by layer."""
        all_params = np.zeros(self.n_qubits * self.max_layers)
        history = []

        for n_layers in range(1, self.max_layers + 1):
            n_params_active = self.n_qubits * n_layers

            # Initialize new layer near identity
            if n_layers > 1:
                start_idx = self.n_qubits * (n_layers - 1)
                all_params[start_idx:n_params_active] = np.random.uniform(
                    -0.01, 0.01, self.n_qubits
                )

            costs = []
            for iteration in range(n_iterations_per_layer):
                # Compute gradients for active parameters
                gradients = np.zeros(n_params_active)
                for i in range(n_params_active):
                    gradients[i] = self._gradient(all_params, n_layers, i)

                # Update parameters
                all_params[:n_params_active] -= learning_rate * gradients

                # Record cost
                cost = self._cost(all_params, n_layers)
                costs.append(cost)

            history.append({
                'n_layers': n_layers,
                'final_cost': costs[-1],
                'cost_reduction': costs[0] - costs[-1] if len(costs) > 1 else 0
            })

            print(f"Layer {n_layers}: Final cost = {costs[-1]:.4f}")

        return {
            'final_params': all_params,
            'history': history
        }

# Demonstrate layer-wise training
print("\nLayer-wise Training Demo")
print("=" * 60)

trainer = LayerwiseTrainer(n_qubits=4, max_layers=4)
result = trainer.train_layerwise(n_iterations_per_layer=30, learning_rate=0.3)

print("\nTraining Summary:")
for h in result['history']:
    print(f"  Layer {h['n_layers']}: cost={h['final_cost']:.4f}, "
          f"reduction={h['cost_reduction']:.4f}")
```

## Local Cost Functions

Using local cost functions dramatically reduces barren plateau severity.

```python
import numpy as np

def compare_local_global_training():
    """Compare training with local vs global cost functions."""

    print("\nLocal vs Global Cost Function Training")
    print("=" * 60)

    n_qubits = 6
    n_layers = 4
    n_params = n_qubits * n_layers
    dim = 2 ** n_qubits
    n_iterations = 100
    learning_rate = 0.5

    def circuit(params):
        state = np.zeros(dim, dtype=complex)
        state[0] = 1
        param_idx = 0

        for _ in range(n_layers):
            for q in range(n_qubits):
                theta = params[param_idx]
                c, s = np.cos(theta/2), np.sin(theta/2)
                new_state = np.zeros_like(state)
                for z in range(dim):
                    z_flip = z ^ (1 << (n_qubits - 1 - q))
                    new_state[z] = c * state[z] - s * state[z_flip]
                    new_state[z_flip] = s * state[z] + c * state[z_flip]
                state = new_state
                param_idx += 1

            for q in range(n_qubits - 1):
                new_state = np.zeros_like(state)
                for z in range(dim):
                    ctrl = (z >> (n_qubits - 1 - q)) & 1
                    z_new = z ^ (1 << (n_qubits - 1 - (q+1))) if ctrl else z
                    new_state[z_new] += state[z]
                state = new_state

        return state

    def cost_global(params):
        state = circuit(params)
        c = 0
        for z in range(dim):
            prob = np.abs(state[z])**2
            parity = bin(z).count('1') % 2
            c += prob * (1 - 2 * parity)
        return c

    def cost_local(params):
        """Sum of local Z expectations."""
        state = circuit(params)
        total = 0
        for q in range(n_qubits):
            for z in range(dim):
                prob = np.abs(state[z])**2
                bit = (z >> (n_qubits - 1 - q)) & 1
                total += prob * (1 - 2 * bit)
        return total / n_qubits

    for cost_type, cost_fn in [('Global', cost_global), ('Local', cost_local)]:
        params = np.random.uniform(-0.1, 0.1, n_params)
        costs = []

        for _ in range(n_iterations):
            # Compute gradients
            gradients = np.zeros(n_params)
            for i in range(n_params):
                shift = np.pi / 2
                params_p, params_m = params.copy(), params.copy()
                params_p[i] += shift
                params_m[i] -= shift
                gradients[i] = (cost_fn(params_p) - cost_fn(params_m)) / 2

            # Update
            params -= learning_rate * gradients
            costs.append(cost_fn(params))

        print(f"\n{cost_type} Cost:")
        print(f"  Initial: {costs[0]:.4f}")
        print(f"  Final:   {costs[-1]:.4f}")
        print(f"  Mean |grad|: {np.mean(np.abs(gradients)):.4e}")

compare_local_global_training()
```

## Parameter Correlation

Using correlated parameters reduces effective dimensionality.

```python
import numpy as np

def demonstrate_parameter_correlation():
    """Show how parameter correlation can help."""

    print("\nParameter Correlation Strategy")
    print("=" * 60)

    n_qubits = 6
    n_layers = 4
    dim = 2 ** n_qubits

    def circuit_independent(params):
        """All parameters independent."""
        state = np.zeros(dim, dtype=complex)
        state[0] = 1
        param_idx = 0

        for _ in range(n_layers):
            for q in range(n_qubits):
                theta = params[param_idx]
                c, s = np.cos(theta/2), np.sin(theta/2)
                new_state = np.zeros_like(state)
                for z in range(dim):
                    z_flip = z ^ (1 << (n_qubits - 1 - q))
                    new_state[z] = c * state[z] - s * state[z_flip]
                    new_state[z_flip] = s * state[z] + c * state[z_flip]
                state = new_state
                param_idx += 1

            for q in range(n_qubits - 1):
                new_state = np.zeros_like(state)
                for z in range(dim):
                    ctrl = (z >> (n_qubits - 1 - q)) & 1
                    z_new = z ^ (1 << (n_qubits - 1 - (q+1))) if ctrl else z
                    new_state[z_new] += state[z]
                state = new_state

        return state

    def circuit_correlated(params):
        """Same angle for all qubits in each layer."""
        state = np.zeros(dim, dtype=complex)
        state[0] = 1

        for layer_idx in range(n_layers):
            theta = params[layer_idx]  # One param per layer

            for q in range(n_qubits):
                c, s = np.cos(theta/2), np.sin(theta/2)
                new_state = np.zeros_like(state)
                for z in range(dim):
                    z_flip = z ^ (1 << (n_qubits - 1 - q))
                    new_state[z] = c * state[z] - s * state[z_flip]
                    new_state[z_flip] = s * state[z] + c * state[z_flip]
                state = new_state

            for q in range(n_qubits - 1):
                new_state = np.zeros_like(state)
                for z in range(dim):
                    ctrl = (z >> (n_qubits - 1 - q)) & 1
                    z_new = z ^ (1 << (n_qubits - 1 - (q+1))) if ctrl else z
                    new_state[z_new] += state[z]
                state = new_state

        return state

    def cost(state):
        c = 0
        for z in range(dim):
            prob = np.abs(state[z])**2
            parity = bin(z).count('1') % 2
            c += prob * (1 - 2 * parity)
        return c

    # Compare gradient variance
    for name, circuit_fn, n_params in [
        ('Independent', circuit_independent, n_qubits * n_layers),
        ('Correlated', circuit_correlated, n_layers)
    ]:
        gradients = []
        for _ in range(100):
            params = np.random.uniform(0, 2*np.pi, n_params)
            shift = np.pi / 2
            params_p, params_m = params.copy(), params.copy()
            params_p[0] += shift
            params_m[0] -= shift

            grad = (cost(circuit_fn(params_p)) - cost(circuit_fn(params_m))) / 2
            gradients.append(grad)

        var = np.var(gradients)
        print(f"{name} ({n_params} params): Var[grad] = {var:.4e}")

demonstrate_parameter_correlation()
```

## Summary

Key mitigation strategies for barren plateaus:

1. **Identity Initialization**: Start with parameters near zero
2. **Layer-wise Training**: Build up circuit depth gradually
3. **Local Cost Functions**: Measure only few qubits
4. **Parameter Correlation**: Reduce effective parameter dimension
5. **Shallow Circuits**: Limit circuit depth

<Warning>
No single strategy works universally. Often, combining multiple techniques is necessary for successful training.
</Warning>
