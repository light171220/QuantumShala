# Expressibility of Variational Circuits

Expressibility quantifies how uniformly a parameterized circuit can cover the quantum state space, which directly impacts the circuit's ability to represent target states.

## Learning Objectives

After completing this lesson, you will be able to:

- Define and measure expressibility
- Understand Haar random distributions
- Compute expressibility metrics
- Analyze expressibility vs depth tradeoffs
- Design expressibility-aware circuits

## 1. What is Expressibility?

Expressibility measures how well an ansatz can approximate arbitrary quantum states.

<Definition term="Expressibility">
A measure of how uniformly the states generated by a parameterized quantum circuit are distributed over the Hilbert space, compared to Haar-random states.
</Definition>

```python

```

<Tip>
Perfect expressibility (reaching any state) isn't always desirable—it can lead to barren plateaus. The goal is sufficient expressibility for your target problem.
</Tip>

## 2. Haar Random Distribution

The Haar measure defines the uniform distribution over quantum states.

```python
import numpy as np

def haar_random_analysis():
    """
    Analyze properties of Haar-random states.
    """
    print("Haar Random Distribution:")
    print("=" * 50)

    def random_haar_state(dim):
        """Generate a Haar-random state."""
        # Method: QR decomposition of random complex matrix
        real = np.random.randn(dim, dim)
        imag = np.random.randn(dim, dim)
        z = (real + 1j * imag) / np.sqrt(2)
        q, r = np.linalg.qr(z)
        d = np.diag(r)
        ph = d / np.abs(d)
        q = q @ np.diag(ph)
        return q[:, 0]

    def fidelity_distribution_haar(dim, n_samples=1000):
        """Sample fidelity distribution for Haar random states."""
        fidelities = []
        for _ in range(n_samples):
            psi1 = random_haar_state(dim)
            psi2 = random_haar_state(dim)
            fid = np.abs(np.vdot(psi1, psi2))**2
            fidelities.append(fid)
        return np.array(fidelities)

    # Theoretical distribution
    print("\n1. Theoretical Fidelity Distribution:")
    print("-" * 40)
    print("  For Haar-random states in d dimensions:")
    print("  P(F) = (d-1)(1-F)^(d-2)")
    print("  Mean fidelity: 1/d")

    # Empirical verification
    print("\n2. Empirical Verification:")
    print("-" * 40)

    for n_qubits in [1, 2, 3, 4]:
        dim = 2**n_qubits
        fidelities = fidelity_distribution_haar(dim, 500)
        mean_f = np.mean(fidelities)
        theoretical = 1.0 / dim
        print(f"  {n_qubits} qubits (d={dim}): "
              f"⟨F⟩ = {mean_f:.4f} (theory: {theoretical:.4f})")

    # Concentration effect
    print("\n3. Concentration for Large Systems:")
    print("-" * 40)
    print("  As d→∞, fidelities concentrate near 0")
    print("  Random states become nearly orthogonal")

    for n_qubits in [4, 6, 8, 10]:
        dim = 2**n_qubits
        mean_fidelity = 1.0 / dim
        print(f"  {n_qubits} qubits: ⟨F⟩ = {mean_fidelity:.6f}")

haar_random_analysis()
```

## 3. Computing Expressibility

Practical methods for measuring ansatz expressibility.

```python
import numpy as np

class ExpressibilityAnalyzer:
    """
    Compute expressibility metrics for quantum circuits.
    """

    def __init__(self, n_qubits):
        self.n_qubits = n_qubits
        self.dim = 2**n_qubits

    def sample_fidelities(self, ansatz_fn, n_params, n_samples=500):
        """Sample pairwise fidelities from ansatz."""
        fidelities = []

        for _ in range(n_samples):
            params1 = np.random.uniform(0, 2*np.pi, n_params)
            params2 = np.random.uniform(0, 2*np.pi, n_params)

            state1 = ansatz_fn(params1)
            state2 = ansatz_fn(params2)

            fid = np.abs(np.vdot(state1, state2))**2
            fidelities.append(fid)

        return np.array(fidelities)

    def haar_pdf(self, f):
        """Haar fidelity probability density."""
        d = self.dim
        if f < 0 or f > 1:
            return 0
        return (d - 1) * (1 - f)**(d - 2)

    def compute_kl_divergence(self, fidelities, n_bins=50):
        """Compute KL divergence from Haar distribution."""
        # Histogram of ansatz fidelities
        hist, bin_edges = np.histogram(fidelities, bins=n_bins,
                                       range=(0, 1), density=True)

        # Compute Haar distribution at bin centers
        bin_centers = (bin_edges[:-1] + bin_edges[1:]) / 2
        haar_probs = np.array([self.haar_pdf(f) for f in bin_centers])

        # Normalize
        bin_width = 1.0 / n_bins
        haar_probs = haar_probs * bin_width
        hist_probs = hist * bin_width

        # KL divergence (avoid log(0))
        kl = 0
        for p, q in zip(hist_probs, haar_probs):
            if p > 1e-10 and q > 1e-10:
                kl += p * np.log(p / q)

        return kl

    def compute_expressibility(self, ansatz_fn, n_params, n_samples=500):
        """Compute expressibility metric."""
        fidelities = self.sample_fidelities(ansatz_fn, n_params, n_samples)

        kl_div = self.compute_kl_divergence(fidelities)
        mean_fid = np.mean(fidelities)
        haar_mean = 1.0 / self.dim

        return {
            'kl_divergence': kl_div,
            'mean_fidelity': mean_fid,
            'haar_mean': haar_mean,
            'fidelity_std': np.std(fidelities)
        }

# Demonstration
print("Expressibility Computation:")
print("=" * 50)

n_qubits = 2
analyzer = ExpressibilityAnalyzer(n_qubits)

# Simple ansatz: just RY rotations
def simple_ansatz(params):
    state = np.zeros(4, dtype=complex)
    state[0] = 1.0

    # Simple RY on each qubit
    for q in range(2):
        c, s = np.cos(params[q]/2), np.sin(params[q]/2)
        new_state = np.zeros_like(state)
        for i in range(4):
            bit = (i >> (1 - q)) & 1
            partner = i ^ (1 << (1 - q))
            if bit == 0:
                new_state[i] += c * state[i] - s * state[partner]
                new_state[partner] += s * state[i] + c * state[partner]
        state = new_state

    return state

# Highly expressive ansatz
def expressive_ansatz(params):
    # Random unitary parameterized (simplified)
    state = np.zeros(4, dtype=complex)
    state[0] = 1.0

    # Multiple layers
    for layer in range(3):
        for q in range(2):
            theta = params[layer * 4 + q * 2]
            phi = params[layer * 4 + q * 2 + 1]
            # RY-RZ
            c, s = np.cos(theta/2), np.sin(theta/2)
            new_state = np.zeros_like(state)
            for i in range(4):
                bit = (i >> (1 - q)) & 1
                partner = i ^ (1 << (1 - q))
                if bit == 0:
                    new_state[i] += c * state[i] - s * state[partner]
                    new_state[partner] += s * state[i] + c * state[partner]
            state = new_state

            # RZ
            for i in range(4):
                bit = (i >> (1 - q)) & 1
                phase = np.exp(1j * phi * (1 if bit else -1) / 2)
                state[i] *= phase

    return state

np.random.seed(42)

print("\nSimple ansatz (2 params, RY only):")
result1 = analyzer.compute_expressibility(simple_ansatz, 2)
print(f"  KL divergence: {result1['kl_divergence']:.4f}")
print(f"  Mean fidelity: {result1['mean_fidelity']:.4f} (Haar: {result1['haar_mean']:.4f})")

print("\nExpressive ansatz (12 params, 3 layers):")
result2 = analyzer.compute_expressibility(expressive_ansatz, 12)
print(f"  KL divergence: {result2['kl_divergence']:.4f}")
print(f"  Mean fidelity: {result2['mean_fidelity']:.4f} (Haar: {result2['haar_mean']:.4f})")
```

<Warning>
Lower KL divergence means higher expressibility (closer to Haar). But very high expressibility can indicate potential trainability problems.
</Warning>

## 4. Expressibility vs Depth

How expressibility changes with circuit depth.

```python
import numpy as np

def expressibility_vs_depth():
    """
    Analyze how expressibility varies with circuit depth.
    """
    print("Expressibility vs Circuit Depth:")

    print("\n1. General Trend:")

    print("  • Deep circuits: Approaches Haar random")
    print("  • Very deep: Saturates (no further improvement)")

    # Simulated expressibility data
    print("\n2. Typical Expressibility Scaling:")
    print("-" * 40)

    depths = [1, 2, 4, 8, 16, 32]
    # Simulated KL divergence (decreasing with depth)
    kl_values = [2.5, 1.2, 0.5, 0.15, 0.05, 0.02]

    print(f"  {'Depth':<10} {'KL Div':<12} {'Expressibility':<15}")
    print("  " + "-" * 35)

    for d, kl in zip(depths, kl_values):
        expr = "Low" if kl > 1 else "Medium" if kl > 0.1 else "High"
        print(f"  {d:<10} {kl:<12.4f} {expr:<15}")
```

## Summary

Expressibility is a key metric for variational circuits:

1. **Definition**: How uniformly the ansatz covers state space
2. **Haar Random**: The ideal uniform distribution
3. **Metrics**: KL divergence from Haar, mean fidelity
4. **Depth Scaling**: Expressibility increases with depth
5. **Design Trade-offs**: Balance with trainability and hardware

Understanding expressibility helps design better variational algorithms.
