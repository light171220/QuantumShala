# Hardware-Efficient Ansätze

Hardware-efficient ansätze are designed to work within the constraints of near-term quantum devices, minimizing circuit depth while maintaining sufficient expressibility.

## Learning Objectives

After completing this lesson, you will be able to:

- Understand hardware constraints on ansatz design
- Implement hardware-efficient circuits
- Minimize circuit depth
- Handle limited connectivity
- Optimize for noise resilience

## 3. Implementing Hardware-Efficient Ansätze

Building circuits that directly use native operations.

```python
import numpy as np

class HardwareEfficientAnsatz:
    """
    Hardware-efficient ansatz using native gate patterns.
    """

    def __init__(self, n_qubits, connectivity='linear'):
        self.n_qubits = n_qubits
        self.dim = 2**n_qubits
        self.connectivity = connectivity

    def get_entangling_pairs(self):
        """Get qubit pairs based on connectivity."""
        if self.connectivity == 'linear':
            return [(i, i+1) for i in range(self.n_qubits - 1)]
        elif self.connectivity == 'circular':
            pairs = [(i, i+1) for i in range(self.n_qubits - 1)]
            pairs.append((self.n_qubits - 1, 0))
            return pairs
        else:
            raise ValueError(f"Unknown connectivity: {self.connectivity}")

    def rotation_layer(self, state, params, offset=0):
        """Apply parameterized rotations to all qubits."""
        for q in range(self.n_qubits):
            # RZ rotation
            state = self._rz(state, q, params[offset + 2*q])
            # RY rotation (or SX-RZ-SX decomposition)
            state = self._ry(state, q, params[offset + 2*q + 1])
        return state

    def entangling_layer(self, state):
        """Apply entangling gates based on connectivity."""
        for ctrl, tgt in self.get_entangling_pairs():
            state = self._cz(state, ctrl, tgt)
        return state

    def forward(self, params, n_layers):
        """Execute the full ansatz."""
        state = np.zeros(self.dim, dtype=complex)
        state[0] = 1.0

        params_per_layer = 2 * self.n_qubits

        for layer in range(n_layers):
            offset = layer * params_per_layer
            state = self.rotation_layer(state, params, offset)
            if layer < n_layers - 1:  # No entanglement after last layer
                state = self.entangling_layer(state)

        return state

    def _rz(self, state, qubit, theta):
        new_state = state.copy()
        for i in range(self.dim):
            bit = (i >> (self.n_qubits - 1 - qubit)) & 1
            phase = np.exp(-1j * theta / 2) if bit == 0 else np.exp(1j * theta / 2)
            new_state[i] *= phase
        return new_state

    def _ry(self, state, qubit, theta):
        c, s = np.cos(theta/2), np.sin(theta/2)
        new_state = np.zeros_like(state)
        for i in range(self.dim):
            bit = (i >> (self.n_qubits - 1 - qubit)) & 1
            partner = i ^ (1 << (self.n_qubits - 1 - qubit))
            if bit == 0:
                new_state[i] += c * state[i] - s * state[partner]
                new_state[partner] += s * state[i] + c * state[partner]
        return new_state

    def _cz(self, state, q1, q2):
        new_state = state.copy()
        for i in range(self.dim):
            b1 = (i >> (self.n_qubits - 1 - q1)) & 1
            b2 = (i >> (self.n_qubits - 1 - q2)) & 1
            if b1 == 1 and b2 == 1:
                new_state[i] *= -1
        return new_state

# Demonstration
print("Hardware-Efficient Ansatz Implementation:")
print("=" * 50)

n_qubits = 4
n_layers = 3

ansatz = HardwareEfficientAnsatz(n_qubits, connectivity='linear')

# Parameter count
n_params = 2 * n_qubits * n_layers
print(f"\nConfiguration:")
print(f"  Qubits: {n_qubits}")
print(f"  Layers: {n_layers}")
print(f"  Parameters: {n_params}")
print(f"  Connectivity: linear")

# Test with random parameters
np.random.seed(42)
params = np.random.uniform(0, 2*np.pi, n_params)

state = ansatz.forward(params, n_layers)
print(f"\nOutput state norm: {np.linalg.norm(state):.6f}")

# Compute entanglement
def entanglement_entropy(state, n_qubits):
    mid = n_qubits // 2
    reshaped = state.reshape(2**mid, 2**(n_qubits - mid))
    s = np.linalg.svd(reshaped, compute_uv=False)
    s = s[s > 1e-10]
    return -np.sum(s**2 * np.log2(s**2 + 1e-15))

entropy = entanglement_entropy(state, n_qubits)
print(f"Entanglement entropy: {entropy:.4f} (max: {n_qubits//2})")
```

<Tip>
CZ gates are often preferred over CNOT gates on some hardware because they're symmetric and may have better fidelity.
</Tip>

## 5. Noise-Aware Design

Designing ansätze that are resilient to noise.

```python
import numpy as np

def noise_aware_design():
    """
    Design considerations for noise resilience.
    """
    print("Noise-Aware Ansatz Design:")
    print("=" * 50)

    print("\n1. Error Accumulation Model:")
    print("-" * 40)

    def circuit_fidelity(n_1q_gates, n_2q_gates, e1=0.001, e2=0.01):
        """Estimate circuit fidelity from gate errors."""
        return (1 - e1)**n_1q_gates * (1 - e2)**n_2q_gates

    configs = [
        (20, 5, "Shallow circuit"),
        (40, 10, "Medium circuit"),
        (80, 20, "Deep circuit"),
        (160, 40, "Very deep circuit")
    ]

    print(f"\n  {'Config':<20} {'1Q gates':<10} {'2Q gates':<10} {'Fidelity':<10}")
    print("  " + "-" * 50)

    for n1q, n2q, name in configs:
        fid = circuit_fidelity(n1q, n2q)
        print(f"  {name:<20} {n1q:<10} {n2q:<10} {fid:.4f}")

    print("\n\n2. Design Principles for Noise Resilience:")
    print("-" * 40)

    principles = [
        "Minimize two-qubit gate count (highest error)",
        "Prefer symmetric gates (CZ over CNOT when possible)",
        "Use error-mitigated measurement",
        "Consider dynamical decoupling for idle qubits",
        "Balance expressibility vs depth"
    ]

    for p in principles:
        print(f"  • {p}")

    print("\n\n3. Effective Depth Budget:")
    print("-" * 40)

    # Calculate effective depth given error rate
    target_fidelity = 0.5  # Minimum useful fidelity
    e2q = 0.01  # 1% two-qubit error

    max_2q_gates = int(np.log(target_fidelity) / np.log(1 - e2q))
    print(f"  Target fidelity: {target_fidelity}")
    print(f"  Two-qubit error: {e2q:.1%}")
    print(f"  Maximum 2Q gates: ~{max_2q_gates}")

    # For different layer counts
    print("\n  Layer budget for 4-qubit ansatz (3 CZ per layer):")
    for n_layers in [1, 2, 5, 10, 20]:
        n_cz = 3 * n_layers
        fid = (1 - e2q)**n_cz
        status = "✓" if fid > target_fidelity else "✗"
        print(f"    {n_layers} layers ({n_cz} CZ): fidelity = {fid:.3f} {status}")

noise_aware_design()
```

<Warning>
The number of two-qubit gates is typically the limiting factor. Each layer might add 3-10 two-qubit gates, quickly consuming the error budget.
</Warning>

## Summary

Hardware-efficient ansätze are essential for near-term quantum computing:

1. **Constraints**: Connectivity, native gates, coherence, errors
2. **Native Gates**: Design directly with hardware gate set
3. **Depth Optimization**: Parallelize, prune, use efficient patterns
4. **Noise Awareness**: Budget gates based on error rates
5. **Tradeoffs**: Balance expressibility against noise resilience

The best hardware-efficient ansatz minimizes depth while achieving sufficient expressibility for the target problem.
