# Data Encoding Strategies

Encoding classical data into quantum states is crucial for QML. The choice of encoding significantly impacts algorithm performance.

## Overview of Encoding Methods

Different encoding strategies offer different trade-offs.

<Definition>
**Data Encoding**: The process of mapping classical data x ∈ ℝ^n into a quantum state |ψ(x)⟩ that can be processed by quantum circuits.
</Definition>

```python
import numpy as np

def encoding_comparison():
    """Compare different encoding methods."""

    methods = {
        'Basis Encoding': {
            'qubits': 'O(log N) for N data points',
            'gates': 'O(log N)',
            'data_type': 'Discrete/integer',
            'example': '|x⟩ = |binary(x)⟩'
        },
        'Amplitude Encoding': {
            'qubits': 'O(log N) for N features',
            'gates': 'O(N) for general data',
            'data_type': 'Continuous vectors',
            'example': '|ψ⟩ = Σ x_i |i⟩ / ||x||'
        },
        'Angle Encoding': {
            'qubits': 'O(N) for N features',
            'gates': 'O(N)',
            'data_type': 'Continuous, bounded',
            'example': '|ψ⟩ = ⊗ RY(x_i)|0⟩'
        },
        'Data Re-uploading': {
            'qubits': 'O(1) or O(few)',
            'gates': 'O(N × L) for L layers',
            'data_type': 'Any continuous',
            'example': 'Multiple encoding layers'
        }
    }

    print("Data Encoding Methods Comparison")
    print("=" * 60)

    for method, info in methods.items():
        print(f"\n{method}:")
        for key, value in info.items():
            print(f"  {key}: {value}")

encoding_comparison()
```

## Amplitude Encoding

Encode data vector as quantum state amplitudes.

<Formula>
**Amplitude Encoding**:

$$|x\rangle = \frac{1}{\|x\|} \sum_{i=0}^{N-1} x_i |i\rangle$$

where N must be a power of 2 (pad with zeros if needed).
</Formula>

```python
import numpy as np

def amplitude_encoding(data):
    """Encode data vector as quantum state amplitudes.

    Args:
        data: Classical data vector of length N

    Returns:
        Quantum state of log2(N) qubits
    """
    # Pad to power of 2
    n = len(data)
    n_qubits = int(np.ceil(np.log2(n)))
    dim = 2 ** n_qubits

    state = np.zeros(dim, dtype=complex)
    state[:n] = data

    # Normalize
    norm = np.linalg.norm(state)
    if norm > 0:
        state = state / norm

    return state, n_qubits

def amplitude_decode(state):
    """Decode amplitudes back to data (up to normalization)."""
    return np.real(state)

# Example
print("Amplitude Encoding")
print("=" * 50)

data = np.array([1, 2, 3, 4, 5, 6, 7, 8])
state, n_qubits = amplitude_encoding(data)

print(f"Original data: {data}")
print(f"Data dimension: {len(data)}")
print(f"Qubits required: {n_qubits}")
print(f"\nEncoded state:")
for i, amp in enumerate(state):
    if np.abs(amp) > 0.001:
        print(f"  |{format(i, f'0{n_qubits}b')}⟩: {amp:.4f}")

# Verify normalization
print(f"\nNormalized: {np.isclose(np.sum(np.abs(state)**2), 1)}")
```

## Angle Encoding

Encode features as rotation angles.

<Tip>
Angle encoding is hardware-efficient (one qubit per feature) and preserves locality of features in the parameter space.
</Tip>

```python
import numpy as np

def angle_encoding(data, encoding_type='ry'):
    """Encode data as rotation angles.

    Args:
        data: Classical data vector
        encoding_type: 'ry', 'rz', or 'rx'

    Returns:
        Quantum state
    """
    n_qubits = len(data)
    dim = 2 ** n_qubits

    # Build state by applying rotations
    state = np.zeros(dim, dtype=complex)
    state[0] = 1  # Start from |0...0⟩

    for q, x in enumerate(data):
        # Scale data to [0, π] if needed
        theta = np.pi * x if np.max(np.abs(data)) <= 1 else x

        new_state = np.zeros_like(state)
        c = np.cos(theta / 2)
        s = np.sin(theta / 2)

        for z in range(dim):
            z_flip = z ^ (1 << (n_qubits - 1 - q))

            if encoding_type == 'ry':
                new_state[z] += c * state[z] - s * state[z_flip]
            elif encoding_type == 'rx':
                new_state[z] += c * state[z] - 1j * s * state[z_flip]
            elif encoding_type == 'rz':
                bit = (z >> (n_qubits - 1 - q)) & 1
                phase = np.exp(-1j * theta / 2) if bit == 0 else np.exp(1j * theta / 2)
                new_state[z] = phase * state[z]

        state = new_state

    return state

# Example
print("Angle Encoding")
print("=" * 50)

data = np.array([0.2, 0.5, 0.8])  # 3 features in [0, 1]
state = angle_encoding(data, 'ry')

print(f"Data: {data}")
print(f"Qubits: {len(data)}")
print("\nEncoded state (non-zero amplitudes):")
for z in range(8):
    if np.abs(state[z]) > 0.01:
        print(f"  |{format(z, '03b')}⟩: {state[z]:.4f}")

# Compare encoding types
print("\nComparing encoding types for data [0.5]:")
for enc_type in ['ry', 'rx', 'rz']:
    s = angle_encoding([0.5], enc_type)
    print(f"  {enc_type}: |0⟩={s[0]:.3f}, |1⟩={s[1]:.3f}")
```

## Basis Encoding

Encode discrete values directly as computational basis states.

```python
import numpy as np

def basis_encoding(integer, n_qubits):
    """Encode integer as basis state.

    Args:
        integer: Non-negative integer to encode
        n_qubits: Number of qubits

    Returns:
        Quantum state |integer⟩
    """
    dim = 2 ** n_qubits
    if integer >= dim:
        raise ValueError(f"Integer {integer} too large for {n_qubits} qubits")

    state = np.zeros(dim, dtype=complex)
    state[integer] = 1.0

    return state

def superposition_basis_encoding(integers, n_qubits):
    """Encode multiple integers as superposition.

    Args:
        integers: List of integers
        n_qubits: Number of qubits

    Returns:
        Uniform superposition over basis states
    """
    dim = 2 ** n_qubits
    state = np.zeros(dim, dtype=complex)

    for idx in integers:
        if idx < dim:
            state[idx] = 1.0

    return state / np.linalg.norm(state)

# Example
print("Basis Encoding")
print("=" * 50)

# Single integer
state = basis_encoding(5, 3)  # |101⟩
print(f"Encoding 5 with 3 qubits: |{format(5, '03b')}⟩")
print(f"State: {np.where(state != 0)[0]}")

# Superposition
data = [0, 2, 5, 7]
super_state = superposition_basis_encoding(data, 3)
print(f"\nSuperposition over {data}:")
for z in range(8):
    if np.abs(super_state[z]) > 0.01:
        print(f"  |{format(z, '03b')}⟩: {super_state[z]:.4f}")
```

## Data Re-uploading

Encode data multiple times throughout the circuit.

<Formula>
**Data Re-uploading**:

$$|ψ(x,θ)⟩ = \prod_{l=1}^{L} U_l(θ_l) S(x) |0⟩$$

where S(x) is the encoding layer repeated L times with trainable layers U_l between.
</Formula>

```python
import numpy as np

class DataReuploading:
    """Data re-uploading circuit."""

    def __init__(self, n_qubits, n_layers):
        self.n_qubits = n_qubits
        self.n_layers = n_layers
        self.dim = 2 ** n_qubits
        # Parameters: 3 per qubit per layer (for trainable rotations)
        self.n_params = 3 * n_qubits * n_layers

    def _apply_encoding(self, state, data):
        """Apply data encoding layer."""
        for q in range(min(len(data), self.n_qubits)):
            theta = data[q]
            c, s = np.cos(theta/2), np.sin(theta/2)
            new_state = np.zeros_like(state)

            for z in range(self.dim):
                z_flip = z ^ (1 << (self.n_qubits - 1 - q))
                new_state[z] += c * state[z] - s * state[z_flip]

            state = new_state
        return state

    def _apply_trainable(self, state, params):
        """Apply trainable rotation layer."""
        param_idx = 0
        for q in range(self.n_qubits):
            # RX
            theta = params[param_idx]
            c, s = np.cos(theta/2), np.sin(theta/2)
            new_state = np.zeros_like(state)
            for z in range(self.dim):
                z_flip = z ^ (1 << (self.n_qubits - 1 - q))
                new_state[z] += c * state[z] - 1j * s * state[z_flip]
            state = new_state
            param_idx += 1

            # RY
            theta = params[param_idx]
            c, s = np.cos(theta/2), np.sin(theta/2)
            new_state = np.zeros_like(state)
            for z in range(self.dim):
                z_flip = z ^ (1 << (self.n_qubits - 1 - q))
                new_state[z] += c * state[z] - s * state[z_flip]
            state = new_state
            param_idx += 1

            # RZ
            theta = params[param_idx]
            new_state = np.zeros_like(state)
            for z in range(self.dim):
                bit = (z >> (self.n_qubits - 1 - q)) & 1
                phase = np.exp(-1j * theta/2) if bit == 0 else np.exp(1j * theta/2)
                new_state[z] = phase * state[z]
            state = new_state
            param_idx += 1

        return state

    def forward(self, data, params):
        """Execute data re-uploading circuit."""
        state = np.zeros(self.dim, dtype=complex)
        state[0] = 1

        params_per_layer = 3 * self.n_qubits

        for l in range(self.n_layers):
            # Encoding layer
            state = self._apply_encoding(state, data)
            # Trainable layer
            layer_params = params[l * params_per_layer : (l+1) * params_per_layer]
            state = self._apply_trainable(state, layer_params)

        return state

# Example
print("Data Re-uploading")
print("=" * 50)

circuit = DataReuploading(n_qubits=2, n_layers=3)
data = np.array([0.5, 0.8])  # 2 features
params = np.random.uniform(0, 2*np.pi, circuit.n_params)

print(f"Qubits: {circuit.n_qubits}")
print(f"Layers: {circuit.n_layers}")
print(f"Parameters: {circuit.n_params}")
print(f"Data dimension: {len(data)}")

state = circuit.forward(data, params)
print(f"\nOutput state normalized: {np.isclose(np.sum(np.abs(state)**2), 1)}")
```

## Choosing the Right Encoding

Select encoding based on data characteristics.

```python
def encoding_selection_guide(n_features, data_type, qubit_budget):
    """Suggest appropriate encoding method."""

    recommendations = []

    if data_type == 'discrete':
        recommendations.append({
            'method': 'Basis Encoding',
            'reason': 'Natural for discrete/integer data',
            'qubits': int(np.ceil(np.log2(n_features + 1)))
        })

    if qubit_budget >= n_features:
        recommendations.append({
            'method': 'Angle Encoding',
            'reason': 'Simple, hardware-efficient',
            'qubits': n_features
        })

    if qubit_budget >= int(np.ceil(np.log2(n_features))):
        recommendations.append({
            'method': 'Amplitude Encoding',
            'reason': 'Exponential compression',
            'qubits': int(np.ceil(np.log2(n_features)))
        })

    if qubit_budget < n_features:
        recommendations.append({
            'method': 'Data Re-uploading',
            'reason': 'Works with few qubits',
            'qubits': min(qubit_budget, 4)
        })

    return recommendations

# Example
print("\nEncoding Selection Examples")
print("=" * 50)

scenarios = [
    (10, 'continuous', 4),
    (100, 'continuous', 10),
    (8, 'discrete', 3),
    (50, 'continuous', 2)
]

for n_feat, dtype, budget in scenarios:
    recs = encoding_selection_guide(n_feat, dtype, budget)
    print(f"\n{n_feat} features, {dtype}, {budget} qubits available:")
    for rec in recs:
        print(f"  {rec['method']}: {rec['qubits']} qubits - {rec['reason']}")
```

## Summary

Key data encoding strategies:

1. **Amplitude Encoding**: Efficient qubit usage, expensive state prep
2. **Angle Encoding**: Simple, one qubit per feature
3. **Basis Encoding**: Natural for discrete data
4. **Data Re-uploading**: Flexible, works with few qubits
