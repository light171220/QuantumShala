## 1. The Angle Encoding Concept

Angle encoding maps classical data values directly to rotation angles of quantum gates.

<Definition term="Angle Encoding">
A data encoding method where classical feature values x are mapped to rotation angles θ = f(x) of parameterized quantum gates, typically Rx, Ry, or Rz rotations.
</Definition>

<Formula>
$$|ψ(x)\rangle = \prod_{i} R_{\alpha_i}(f(x_i))|0\rangle^{\otimes n}$$

where $R_{\alpha}$ is a rotation gate around axis $\alpha$ and $f$ is a scaling function.
</Formula>

```python
import numpy as np

def angle_encode_single(x, gate_type='Ry'):
    """
    Encode a single feature as a rotation angle.

    Args:
        x: Feature value (typically normalized to [0, 1] or [-1, 1])
        gate_type: 'Rx', 'Ry', or 'Rz'

    Returns:
        Quantum state after rotation
    """
    # Common scaling: x -> π*x or x -> 2*arcsin(x)
    theta = np.pi * x

    # Build rotation matrix
    if gate_type == 'Rx':
        gate = np.array([
            [np.cos(theta/2), -1j*np.sin(theta/2)],
            [-1j*np.sin(theta/2), np.cos(theta/2)]
        ])
    elif gate_type == 'Ry':
        gate = np.array([
            [np.cos(theta/2), -np.sin(theta/2)],
            [np.sin(theta/2), np.cos(theta/2)]
        ])
    elif gate_type == 'Rz':
        gate = np.array([
            [np.exp(-1j*theta/2), 0],
            [0, np.exp(1j*theta/2)]
        ])

    # Apply to |0⟩
    initial_state = np.array([1, 0], dtype=complex)
    final_state = gate @ initial_state

    return final_state

# Example
print("Angle Encoding Examples:")
print("=" * 50)
for x in [0, 0.25, 0.5, 0.75, 1.0]:
    state = angle_encode_single(x, 'Ry')
    probs = np.abs(state) ** 2
    print(f"x={x:.2f} -> Ry(π*{x:.2f}) -> |0⟩: {probs[0]:.4f}, |1⟩: {probs[1]:.4f}")
```

## 2. Scaling Functions

Different scaling functions map features to angles in various ways.

```python
import numpy as np

def scaling_functions():
    """Compare different scaling functions for angle encoding."""

    scalings = {
        'Linear': lambda x: np.pi * x,
        'Arcsin': lambda x: 2 * np.arcsin(np.clip(x, -1, 1)),
        'Arctan': lambda x: 2 * np.arctan(x),
        'Full rotation': lambda x: 2 * np.pi * x
    }

    print("Scaling Functions for Angle Encoding:")
    print("=" * 60)
    print(f"{'x':<8}", end="")
    for name in scalings:
        print(f"{name:<15}", end="")
    print()
    print("-" * 60)

    for x in [0, 0.25, 0.5, 0.75, 1.0]:
        print(f"{x:<8.2f}", end="")
        for name, func in scalings.items():
            angle = func(x)
            print(f"{angle:<15.4f}", end="")
        print()

    print("\nKey Properties:")
    print("• Linear (πx): Simple, maps [0,1] to [0,π]")
    print("• Arcsin: Creates uniform distribution on Bloch sphere")
    print("• Arctan: Handles unbounded features, compresses outliers")
    print("• Full rotation: Uses full Bloch sphere, but 0≡2π issue")

scaling_functions()
```

## 3. Multi-Feature Angle Encoding

Encoding multiple features requires multiple qubits or repeated applications.

```python
import numpy as np

def multi_feature_angle_encode(features, encoding_type='parallel'):
    """
    Encode multiple features using angle encoding.

    Args:
        features: List of feature values
        encoding_type: 'parallel' (one qubit per feature) or
                      'sequential' (multiple rotations per qubit)

    Returns:
        Quantum state vector
    """
    n_features = len(features)

    if encoding_type == 'parallel':
        # One qubit per feature
        n_qubits = n_features
        dim = 2 ** n_qubits

        # Start with |0...0⟩
        state = np.zeros(dim, dtype=complex)
        state[0] = 1.0

        # Apply Ry rotation on each qubit
        for i, x in enumerate(features):
            theta = np.pi * x
            # Build full operator
            ops = []
            for j in range(n_qubits):
                if j == i:
                    Ry = np.array([
                        [np.cos(theta/2), -np.sin(theta/2)],
                        [np.sin(theta/2), np.cos(theta/2)]
                    ])
                    ops.append(Ry)
                else:
                    ops.append(np.eye(2))

            # Tensor product
            full_op = ops[0]
            for op in ops[1:]:
                full_op = np.kron(full_op, op)

            state = full_op @ state

        return state, n_qubits

    elif encoding_type == 'sequential':
        # Multiple rotations on single qubit
        state = np.array([1, 0], dtype=complex)

        for i, x in enumerate(features):
            # Alternate between Rx, Ry, Rz
            theta = np.pi * x
            gates = ['Ry', 'Rz', 'Ry']
            gate_type = gates[i % 3]

            if gate_type == 'Ry':
                gate = np.array([
                    [np.cos(theta/2), -np.sin(theta/2)],
                    [np.sin(theta/2), np.cos(theta/2)]
                ])
            elif gate_type == 'Rz':
                gate = np.array([
                    [np.exp(-1j*theta/2), 0],
                    [0, np.exp(1j*theta/2)]
                ])

            state = gate @ state

        return state, 1

# Example
features = [0.3, 0.7, 0.5]

print("Multi-Feature Angle Encoding:")
print("=" * 50)

state_p, n_p = multi_feature_angle_encode(features, 'parallel')
print(f"\nParallel encoding ({n_p} qubits):")
print(f"Features: {features}")
print(f"State dimension: {len(state_p)}")
probs = np.abs(state_p) ** 2
for i, p in enumerate(probs):
    if p > 0.01:
        print(f"  |{format(i, f'0{n_p}b')}⟩: {p:.4f}")

state_s, n_s = multi_feature_angle_encode(features, 'sequential')
print(f"\nSequential encoding ({n_s} qubit):")
print(f"State: {np.round(state_s, 4)}")
print(f"Probabilities: |0⟩={np.abs(state_s[0])**2:.4f}, |1⟩={np.abs(state_s[1])**2:.4f}")
```

## 4. Feature Maps and Expressibility

Angle encoding creates a feature map from classical to quantum space.

```python
import numpy as np

def analyze_feature_map(encoding_func, n_samples=100):
    """
    Analyze the feature space created by angle encoding.

    Examines how well-separated different inputs become in quantum space.
    """
    # Generate sample points
    x_samples = np.linspace(0, 1, n_samples)

    # Compute quantum states
    states = [encoding_func(x) for x in x_samples]

    # Compute pairwise inner products (kernel)
    kernel_matrix = np.zeros((n_samples, n_samples))
    for i in range(n_samples):
        for j in range(n_samples):
            kernel_matrix[i, j] = np.abs(np.vdot(states[i], states[j])) ** 2

    return kernel_matrix, x_samples

def simple_angle_encode(x):
    """Simple Ry encoding."""
    theta = np.pi * x
    return np.array([np.cos(theta/2), np.sin(theta/2)])

def repeated_angle_encode(x, reps=2):
    """Repeated encoding with entanglement layers."""
    theta = np.pi * x
    state = np.array([1, 0], dtype=complex)

    for _ in range(reps):
        # Ry rotation
        Ry = np.array([
            [np.cos(theta/2), -np.sin(theta/2)],
            [np.sin(theta/2), np.cos(theta/2)]
        ])
        state = Ry @ state

        # Rz rotation
        Rz = np.array([
            [np.exp(-1j*theta/2), 0],
            [0, np.exp(1j*theta/2)]
        ])
        state = Rz @ state

    return state

print("Feature Map Analysis:")
print("=" * 50)

# Simple encoding
kernel_simple, _ = analyze_feature_map(simple_angle_encode, 10)
print("\nSimple Ry encoding - kernel samples:")
print("k(0,0):", kernel_simple[0, 0])
print("k(0,5):", kernel_simple[0, 5])
print("k(0,9):", kernel_simple[0, 9])

# Repeated encoding
kernel_repeated, _ = analyze_feature_map(
    lambda x: repeated_angle_encode(x, 2), 10
)
print("\nRepeated encoding - kernel samples:")
print("k(0,0):", kernel_repeated[0, 0])
print("k(0,5):", kernel_repeated[0, 5])
print("k(0,9):", kernel_repeated[0, 9])

print("\n→ More repetitions can increase expressibility")
```

## 5. Angle Encoding in Variational Circuits

Combining angle encoding with variational layers.

```python
import numpy as np

def variational_circuit_with_encoding(x, params):
    """
    A simple variational circuit with angle encoding.

    Structure: Encoding -> Variational Layer -> Measurement

    Args:
        x: Input feature
        params: Variational parameters [θ1, θ2]

    Returns:
        Expectation value of Z measurement
    """
    # Encoding layer
    theta_enc = np.pi * x
    Ry_enc = np.array([
        [np.cos(theta_enc/2), -np.sin(theta_enc/2)],
        [np.sin(theta_enc/2), np.cos(theta_enc/2)]
    ])

    # Variational layer
    theta1, theta2 = params
    Ry_var1 = np.array([
        [np.cos(theta1/2), -np.sin(theta1/2)],
        [np.sin(theta1/2), np.cos(theta1/2)]
    ])
    Rz_var = np.array([
        [np.exp(-1j*theta2/2), 0],
        [0, np.exp(1j*theta2/2)]
    ])

    # Apply circuit
    state = np.array([1, 0], dtype=complex)
    state = Ry_enc @ state
    state = Ry_var1 @ state
    state = Rz_var @ state

    # Z measurement expectation
    Z = np.array([[1, 0], [0, -1]])
    expectation = np.real(np.vdot(state, Z @ state))

    return expectation

# Demonstrate circuit behavior
print("Variational Circuit with Angle Encoding:")
print("=" * 50)

params = [0.5, 0.3]  # Fixed variational parameters

print(f"Variational params: θ1={params[0]:.2f}, θ2={params[1]:.2f}")
print("\nInput x -> ⟨Z⟩:")
for x in np.linspace(0, 1, 6):
    exp_val = variational_circuit_with_encoding(x, params)
    print(f"  x={x:.2f} -> ⟨Z⟩={exp_val:+.4f}")

print("\nDifferent params for same input:")
x = 0.5
for p1 in [0, 0.5, 1.0]:
    exp_val = variational_circuit_with_encoding(x, [p1, 0.3])
    print(f"  θ1={p1:.1f} -> ⟨Z⟩={exp_val:+.4f}")
```

## 6. Comparison and Best Practices

When and how to use angle encoding effectively.

```python
def angle_encoding_guidelines():
    """Guidelines for using angle encoding."""

    guidelines = {
        "Advantages": [
            "Efficient: n features need only n qubits (parallel) or 1 qubit (sequential)",
            "Natural for variational algorithms - same parameterized gates",
            "Easy to implement on real hardware",
            "Works well with gradient-based optimization",
            "Low circuit depth possible"
        ],
        "Disadvantages": [
            "Limited expressibility with simple encodings",
            "Feature scaling significantly affects performance",
            "Periodic nature can cause aliasing issues",
            "May need repetitions for complex decision boundaries"
        ],
        "Best Practices": [
            "Normalize features to [0, 1] or [-1, 1]",
            "Consider arcsin scaling for bounded data",
            "Use multiple encoding layers for richer feature maps",
            "Combine different rotation axes (Rx, Ry, Rz)",
            "Add entanglement between feature qubits"
        ],
        "Recommended For": [
            "Variational quantum classifiers",
            "Quantum neural networks",
            "NISQ-era applications",
            "Small to medium feature counts",
            "When circuit depth is limited"
        ]
    }

    print("Angle Encoding Guidelines:")
    print("=" * 60)

    for category, items in guidelines.items():
        print(f"\n{category}:")
        for item in items:
            print(f"  • {item}")

angle_encoding_guidelines()
```

In this lesson, you learned about angle encoding:

1. **Concept**: Map features to rotation gate angles
2. **Scaling**: Linear (πx), arcsin, arctan for different data ranges

<Warning>

</Warning>
