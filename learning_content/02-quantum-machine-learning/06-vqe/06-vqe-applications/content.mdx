# VQE Applications

VQE has emerged as a leading algorithm for near-term quantum advantage. This lesson explores practical applications and the path toward industrially relevant quantum chemistry.

## Molecular Ground State Calculations

The primary application of VQE is computing molecular ground state energies.

<Definition>
**Ground State Energy**: The lowest eigenvalue of the molecular Hamiltonian, determining:
- Molecular stability
- Bond strengths
- Reaction energetics
- Spectroscopic properties
</Definition>

```python
import numpy as np

class MolecularVQE:
    """
    VQE for molecular systems.
    """

    def __init__(self, molecule_name, basis='sto-3g'):
        """
        Initialize molecular VQE.

        Args:
            molecule_name: Name of molecule
            basis: Basis set for calculation
        """
        self.molecule = molecule_name
        self.basis = basis

        # Load pre-computed molecular data
        self.data = self._load_molecular_data()

    def _load_molecular_data(self):
        """Load molecular Hamiltonian data."""
        # Simplified molecular data for demonstration
        molecules = {
            'H2': {
                'n_qubits': 4,
                'n_electrons': 2,
                'nuclear_repulsion': 0.7199,
                'fci_energy': -1.1373,  # Exact ground state
                'hf_energy': -1.1167    # Hartree-Fock
            },
            'LiH': {
                'n_qubits': 12,
                'n_electrons': 4,
                'nuclear_repulsion': 0.9924,
                'fci_energy': -7.8825,
                'hf_energy': -7.8633
            },
            'H2O': {
                'n_qubits': 14,
                'n_electrons': 10,
                'nuclear_repulsion': 9.1897,
                'fci_energy': -75.7299,
                'hf_energy': -75.6340
            }
        }
        return molecules.get(self.molecule, molecules['H2'])

    def correlation_energy(self):
        """Compute correlation energy (FCI - HF)."""
        return self.data['fci_energy'] - self.data['hf_energy']

    def report(self):
        """Print molecular information."""
        print(f"\n{self.molecule} in {self.basis} basis:")
        print(f"  Qubits required: {self.data['n_qubits']}")
        print(f"  Electrons: {self.data['n_electrons']}")
        print(f"  HF energy: {self.data['hf_energy']:.4f} Ha")
        print(f"  FCI energy: {self.data['fci_energy']:.4f} Ha")
        print(f"  Correlation energy: {self.correlation_energy():.4f} Ha")
        print(f"  Correlation energy: {self.correlation_energy() * 627.5:.2f} kcal/mol")

# Example
for mol in ['H2', 'LiH', 'H2O']:
    vqe = MolecularVQE(mol)
    vqe.report()
```

## Potential Energy Surfaces

VQE can compute potential energy surfaces (PES) for chemical reactions.

<Tip>
Computing PES at multiple geometries reveals reaction barriers, equilibrium structures, and transition states—critical for understanding chemical reactivity.
</Tip>

```python
import numpy as np

def compute_h2_pes(bond_lengths):
    """
    Compute H2 potential energy surface.

    Args:
        bond_lengths: Array of H-H distances (Angstrom)

    Returns:
        Array of energies at each geometry
    """
    # Simplified H2 model (Morse potential fit to VQE results)
    D_e = 0.1745  # Dissociation energy (Ha)
    r_e = 0.741   # Equilibrium distance (Å)
    a = 1.44      # Morse parameter

    energies = []
    for r in bond_lengths:
        # Morse potential: V(r) = D_e * (1 - exp(-a*(r-r_e)))^2 - D_e
        E = D_e * (1 - np.exp(-a * (r - r_e)))**2 - D_e
        # Add nuclear repulsion (1/r in atomic units)
        E += 1.0 / (r / 0.529177)  # Convert Å to Bohr
        energies.append(E)

    return np.array(energies)

def analyze_pes(bond_lengths, energies):
    """Analyze potential energy surface."""
    # Find equilibrium
    min_idx = np.argmin(energies)
    r_eq = bond_lengths[min_idx]
    E_eq = energies[min_idx]

    # Estimate dissociation energy
    E_dissoc = energies[-1] - E_eq

    # Force constant (second derivative at equilibrium)
    if min_idx > 0 and min_idx < len(energies) - 1:
        dr = bond_lengths[1] - bond_lengths[0]
        k = (energies[min_idx+1] - 2*E_eq + energies[min_idx-1]) / (dr**2)
    else:
        k = 0

    return {
        'equilibrium_distance': r_eq,
        'equilibrium_energy': E_eq,
        'dissociation_energy': E_dissoc,
        'force_constant': k
    }

# Compute H2 PES
bond_lengths = np.linspace(0.4, 3.0, 50)
energies = compute_h2_pes(bond_lengths)

analysis = analyze_pes(bond_lengths, energies)

print("H2 Potential Energy Surface Analysis")
print("=" * 45)
print(f"Equilibrium distance: {analysis['equilibrium_distance']:.3f} Å")
print(f"Equilibrium energy: {analysis['equilibrium_energy']:.4f} Ha")
print(f"Dissociation energy: {analysis['dissociation_energy']:.4f} Ha")
print(f"                   = {analysis['dissociation_energy'] * 627.5:.2f} kcal/mol")
```

## Chemical Accuracy

Chemical accuracy (1 kcal/mol ≈ 1.6 mHa) is the target for useful quantum chemistry.

<Definition>
**Chemical Accuracy**: Errors of 1 kcal/mol or less, sufficient to:
- Predict reaction outcomes
- Guide catalyst design
- Support drug discovery
- Validate computational methods
</Definition>

```python
import numpy as np

class AccuracyAnalysis:
    """
    Analyze VQE accuracy for chemistry applications.
    """

    # Conversion factors
    HARTREE_TO_KCAL = 627.5094
    HARTREE_TO_EV = 27.2114
    CHEMICAL_ACCURACY_HA = 1.6e-3  # 1 kcal/mol in Hartree

    def __init__(self, exact_energy, vqe_energy):
        self.exact = exact_energy
        self.vqe = vqe_energy
        self.error = abs(vqe_energy - exact_energy)

    def is_chemically_accurate(self):
        """Check if VQE achieves chemical accuracy."""
        return self.error <= self.CHEMICAL_ACCURACY_HA

    def report(self):
        """Print accuracy analysis."""
        print(f"VQE Accuracy Analysis")
        print("=" * 40)
        print(f"Exact energy: {self.exact:.6f} Ha")
        print(f"VQE energy: {self.vqe:.6f} Ha")
        print(f"Error: {self.error:.6f} Ha")
        print(f"     = {self.error * self.HARTREE_TO_KCAL:.4f} kcal/mol")
        print(f"     = {self.error * self.HARTREE_TO_EV * 1000:.2f} meV")
        print(f"Chemical accuracy: {'✓ YES' if self.is_chemically_accurate() else '✗ NO'}")

    @classmethod
    def required_precision(cls, target_kcal=1.0):
        """Calculate required precision for target accuracy."""
        precision_ha = target_kcal / cls.HARTREE_TO_KCAL
        print(f"For {target_kcal} kcal/mol accuracy:")
        print(f"  Need precision: {precision_ha:.6f} Ha")
        print(f"                = {precision_ha * 1000:.3f} mHa")

# Example analyses
print("Example 1: High-quality VQE")
analysis1 = AccuracyAnalysis(exact_energy=-1.1373, vqe_energy=-1.1365)
analysis1.report()

print("\nExample 2: Noisy VQE")
analysis2 = AccuracyAnalysis(exact_energy=-1.1373, vqe_energy=-1.1200)
analysis2.report()

print("\n" + "=" * 40)
AccuracyAnalysis.required_precision(1.0)
```

## Complete VQE Application Example

A full workflow for computing molecular properties.

```python
import numpy as np
from scipy.optimize import minimize

class CompleteVQEApplication:
    """
    Complete VQE application for molecular properties.
    """

    def __init__(self, molecule_data):
        self.data = molecule_data
        self.results = {}

    def run_vqe(self, hamiltonian, ansatz_fn, n_params, max_iter=100):
        """Run VQE optimization."""
        def cost_function(params):
            state = ansatz_fn(params)
            return np.real(state.conj() @ hamiltonian @ state)

        initial_params = np.random.uniform(0, 2*np.pi, n_params)
        result = minimize(cost_function, initial_params, method='COBYLA',
                         options={'maxiter': max_iter})

        return result.fun, result.x

    def compute_properties(self, optimal_energy, optimal_params, ansatz_fn):
        """Compute molecular properties from optimized state."""
        properties = {
            'ground_energy': optimal_energy,
            'correlation_energy': optimal_energy - self.data.get('hf_energy', 0),
            'total_energy': optimal_energy + self.data.get('nuclear_repulsion', 0)
        }

        # Could add: dipole moment, charge distribution, etc.
        return properties

    def full_workflow(self, hamiltonian, ansatz_fn, n_params):
        """Execute complete VQE workflow."""
        print("VQE Application Workflow")
        print("=" * 40)

        # Step 1: Run VQE
        print("\n1. Running VQE optimization...")
        energy, params = self.run_vqe(hamiltonian, ansatz_fn, n_params)
        print(f"   Converged energy: {energy:.6f} Ha")

        # Step 2: Compute properties
        print("\n2. Computing molecular properties...")
        properties = self.compute_properties(energy, params, ansatz_fn)

        # Step 3: Accuracy analysis
        print("\n3. Accuracy analysis...")
        if 'exact_energy' in self.data:
            error = abs(energy - self.data['exact_energy'])
            error_kcal = error * 627.5
            chem_accurate = error_kcal < 1.0

            properties['error_ha'] = error
            properties['error_kcal'] = error_kcal
            properties['chemical_accuracy'] = chem_accurate

            print(f"   Error: {error_kcal:.4f} kcal/mol")
            print(f"   Chemical accuracy: {'YES' if chem_accurate else 'NO'}")

        # Step 4: Report
        print("\n4. Final Results:")
        for key, value in properties.items():
            if isinstance(value, float):
                print(f"   {key}: {value:.6f}")
            else:
                print(f"   {key}: {value}")

        self.results = properties
        return properties

# Example usage

    'name': 'H2',
    'hf_energy': -1.1167,
    'exact_energy': -1.1373,
    'nuclear_repulsion': 0.7199
}

# Simple 2-qubit Hamiltonian (simplified H2)
H = np.array([
    [0.5, 0.2],
    [0.2, -0.8]
])

def simple_ansatz(params):
    theta = params[0]
    return np.array([np.cos(theta/2), np.sin(theta/2)])

# Note: Using simplified Hamiltonian for demonstration

```
