# Quantum Clustering

Clustering groups similar data points without labeled examples. Quantum computing offers advantages through quantum distance calculations, kernel methods, and optimization algorithms for finding cluster assignments.

## Quantum Distance Calculation

Quantum computers can efficiently compute distances between data points encoded as quantum states.

<Definition>
**Quantum Distance**:

For states $|\psi_x\rangle$ and $|\psi_y\rangle$ encoding data points $x$ and $y$:

$$d(x, y) = \sqrt{2(1 - |\langle\psi_x|\psi_y\rangle|^2)}$$

This is related to the Euclidean distance in the feature space defined by the quantum encoding.
</Definition>

```python
import numpy as np

class QuantumDistanceCalculator:
    """Compute distances using quantum state overlaps."""

    def __init__(self, n_qubits: int):
        self.n_qubits = n_qubits
        self.dim = 2 ** n_qubits

    def encode_point(self, x: np.ndarray) -> np.ndarray:
        """Encode data point into quantum state."""
        state = np.zeros(self.dim, dtype=complex)
        state[0] = 1

        # Amplitude encoding with normalization
        for i, val in enumerate(x[:self.n_qubits]):
            theta = np.pi * val / 2  # Scale to [0, π/2]
            c, s = np.cos(theta), np.sin(theta)
            new_state = np.zeros_like(state)
            for z in range(self.dim):
                z_flip = z ^ (1 << (self.n_qubits - 1 - i))
                new_state[z] = c * state[z] - s * state[z_flip]
                new_state[z_flip] = s * state[z] + c * state[z_flip]
            state = new_state

        return state

    def quantum_distance(self, x1: np.ndarray, x2: np.ndarray) -> float:
        """Compute quantum distance between two points."""
        state1 = self.encode_point(x1)
        state2 = self.encode_point(x2)

        # Fidelity (squared overlap)
        fidelity = np.abs(np.vdot(state1, state2))**2

        # Convert to distance
        return np.sqrt(2 * (1 - fidelity))

    def distance_matrix(self, X: np.ndarray) -> np.ndarray:
        """Compute pairwise quantum distance matrix."""
        n = len(X)
        D = np.zeros((n, n))
        for i in range(n):
            for j in range(i + 1, n):
                D[i, j] = self.quantum_distance(X[i], X[j])
                D[j, i] = D[i, j]
        return D
```

## Quantum K-Means Clustering

Quantum k-means uses quantum distance calculations and can leverage quantum optimization for centroid updates.

```python
import numpy as np

class QuantumKMeans:
    """K-means clustering with quantum distance calculations."""

    def __init__(self, n_clusters: int, n_qubits: int, max_iter: int = 100):
        self.n_clusters = n_clusters
        self.n_qubits = n_qubits
        self.max_iter = max_iter
        self.dim = 2 ** n_qubits
        self.centroids = None

    def encode_point(self, x: np.ndarray) -> np.ndarray:
        """Encode data point as quantum state."""
        state = np.zeros(self.dim, dtype=complex)
        state[0] = 1

        for i, val in enumerate(x[:self.n_qubits]):
            theta = np.arctan(val)  # Bounded encoding
            c, s = np.cos(theta), np.sin(theta)
            new_state = np.zeros_like(state)
            for z in range(self.dim):
                z_flip = z ^ (1 << (self.n_qubits - 1 - i))
                new_state[z] = c * state[z] - s * state[z_flip]
                new_state[z_flip] = s * state[z] + c * state[z_flip]
            state = new_state

        return state

    def quantum_distance(self, x: np.ndarray, centroid: np.ndarray) -> float:
        """Compute quantum distance to centroid."""
        state_x = self.encode_point(x)
        state_c = self.encode_point(centroid)
        fidelity = np.abs(np.vdot(state_x, state_c))**2
        return np.sqrt(2 * (1 - fidelity))

    def assign_clusters(self, X: np.ndarray) -> np.ndarray:
        """Assign each point to nearest centroid."""
        labels = np.zeros(len(X), dtype=int)
        for i, x in enumerate(X):
            distances = [self.quantum_distance(x, c) for c in self.centroids]
            labels[i] = np.argmin(distances)
        return labels

    def update_centroids(self, X: np.ndarray, labels: np.ndarray) -> np.ndarray:
        """Update centroids as mean of assigned points."""
        new_centroids = np.zeros_like(self.centroids)
        for k in range(self.n_clusters):
            mask = labels == k
            if np.any(mask):
                new_centroids[k] = X[mask].mean(axis=0)
            else:
                new_centroids[k] = self.centroids[k]
        return new_centroids

    def fit(self, X: np.ndarray) -> np.ndarray:
        """Fit k-means clustering."""
        n_samples, n_features = X.shape

        # Initialize centroids randomly
        indices = np.random.choice(n_samples, self.n_clusters, replace=False)
        self.centroids = X[indices].copy()

        for iteration in range(self.max_iter):
            # Assign clusters
            labels = self.assign_clusters(X)

            # Update centroids
            new_centroids = self.update_centroids(X, labels)

            # Check convergence
            if np.allclose(self.centroids, new_centroids):
                print(f"Converged at iteration {iteration}")
                break

            self.centroids = new_centroids

        return self.assign_clusters(X)

    def inertia(self, X: np.ndarray, labels: np.ndarray) -> float:
        """Compute sum of squared distances to centroids."""
        total = 0
        for i, x in enumerate(X):
            d = self.quantum_distance(x, self.centroids[labels[i]])
            total += d**2
        return total


# Example
print("\nQuantum K-Means Clustering")
print("=" * 50)

np.random.seed(42)

# Create clustered data
cluster1 = np.random.randn(15, 3) + np.array([2, 2, 0])
cluster2 = np.random.randn(15, 3) + np.array([-2, -2, 0])
cluster3 = np.random.randn(15, 3) + np.array([0, 0, 3])
X = np.vstack([cluster1, cluster2, cluster3])

# True labels for evaluation
true_labels = np.array([0]*15 + [1]*15 + [2]*15)

qkm = QuantumKMeans(n_clusters=3, n_qubits=3, max_iter=50)
labels = qkm.fit(X)

print(f"Inertia: {qkm.inertia(X, labels):.4f}")
print(f"Centroids:\n{qkm.centroids.round(2)}")

# Simple accuracy (accounting for label permutation)
from itertools import permutations
best_acc = 0
for perm in permutations([0, 1, 2]):
    perm_labels = np.array([perm[l] for l in labels])
    acc = np.mean(perm_labels == true_labels)
    best_acc = max(best_acc, acc)
print(f"Best accuracy: {best_acc:.2f}")
```

## Quantum Spectral Clustering

Use quantum kernels to build similarity graphs for spectral clustering.

```python
import numpy as np

class QuantumSpectralClustering:
    """Spectral clustering using quantum kernel similarity."""

    def __init__(self, n_clusters: int, n_qubits: int):
        self.n_clusters = n_clusters
        self.n_qubits = n_qubits
        self.dim = 2 ** n_qubits

    def quantum_feature_map(self, x: np.ndarray) -> np.ndarray:
        """Create quantum feature state."""
        state = np.zeros(self.dim, dtype=complex)
        state[0] = 1

        # First rotation layer
        for i, val in enumerate(x[:self.n_qubits]):
            theta = val
            c, s = np.cos(theta/2), np.sin(theta/2)
            new_state = np.zeros_like(state)
            for z in range(self.dim):
                z_flip = z ^ (1 << (self.n_qubits - 1 - i))
                new_state[z] = c * state[z] - s * state[z_flip]
                new_state[z_flip] = s * state[z] + c * state[z_flip]
            state = new_state

        # Entanglement layer
        for q in range(self.n_qubits - 1):
            new_state = np.zeros_like(state)
            for z in range(self.dim):
                ctrl = (z >> (self.n_qubits - 1 - q)) & 1
                z_new = z ^ (1 << (self.n_qubits-1-(q+1))) if ctrl else z
                new_state[z_new] += state[z]
            state = new_state

        # Second rotation layer with feature interaction
        for i in range(self.n_qubits):
            for j in range(i + 1, min(i + 2, len(x))):
                if j < len(x):
                    theta = x[i] * x[j]
                    for z in range(self.dim):
                        bit = (z >> (self.n_qubits - 1 - i)) & 1
                        state[z] *= np.exp(1j*theta/2) if bit else np.exp(-1j*theta/2)

        return state

    def quantum_kernel(self, x1: np.ndarray, x2: np.ndarray) -> float:
        """Compute quantum kernel (similarity)."""
        state1 = self.quantum_feature_map(x1)
        state2 = self.quantum_feature_map(x2)
        return np.abs(np.vdot(state1, state2))**2

    def build_similarity_matrix(self, X: np.ndarray) -> np.ndarray:
        """Build similarity matrix using quantum kernel."""
        n = len(X)
        S = np.zeros((n, n))
        for i in range(n):
            for j in range(i, n):
                S[i, j] = self.quantum_kernel(X[i], X[j])
                S[j, i] = S[i, j]
        return S

    def fit(self, X: np.ndarray) -> np.ndarray:
        """Perform spectral clustering."""
        n = len(X)

        # Build similarity matrix
        S = self.build_similarity_matrix(X)

        # Compute normalized Laplacian
        D = np.diag(S.sum(axis=1))
        D_inv_sqrt = np.diag(1.0 / np.sqrt(np.diag(D) + 1e-10))
        L_norm = np.eye(n) - D_inv_sqrt @ S @ D_inv_sqrt

        # Compute eigenvectors
        eigenvalues, eigenvectors = np.linalg.eigh(L_norm)

        # Use first k eigenvectors (smallest eigenvalues)
        V = eigenvectors[:, :self.n_clusters]

        # Normalize rows
        V_norm = V / (np.linalg.norm(V, axis=1, keepdims=True) + 1e-10)

        # K-means on eigenvector representation
        from numpy.linalg import norm

        # Simple k-means
        centroids = V_norm[np.random.choice(n, self.n_clusters, replace=False)]
        for _ in range(50):
            # Assign
            labels = np.array([np.argmin([norm(v - c) for c in centroids])
                              for v in V_norm])
            # Update
            new_centroids = np.array([V_norm[labels == k].mean(axis=0)
                                      if np.any(labels == k) else centroids[k]
                                      for k in range(self.n_clusters)])
            if np.allclose(centroids, new_centroids):
                break
            centroids = new_centroids

        return labels


# Example
print("\nQuantum Spectral Clustering")
print("=" * 50)

np.random.seed(42)

# Create non-linearly separable data (two moons)
n_samples = 30
t = np.linspace(0, np.pi, n_samples // 2)
moon1 = np.column_stack([np.cos(t), np.sin(t)])
moon2 = np.column_stack([1 - np.cos(t), 0.5 - np.sin(t)])
X = np.vstack([moon1, moon2])
X += np.random.randn(*X.shape) * 0.1
true_labels = np.array([0] * (n_samples // 2) + [1] * (n_samples // 2))

qsc = QuantumSpectralClustering(n_clusters=2, n_qubits=2)
labels = qsc.fit(X)

# Compute accuracy (with possible label flip)
acc1 = np.mean(labels == true_labels)
acc2 = np.mean(labels == (1 - true_labels))
accuracy = max(acc1, acc2)

print(f"Clustering accuracy: {accuracy:.2f}")
print(f"Label distribution: {np.bincount(labels)}")
```

## Clustering Evaluation Metrics

```python
import numpy as np

def silhouette_score(X: np.ndarray, labels: np.ndarray) -> float:
    """Compute silhouette score for clustering quality."""
    n = len(X)
    unique_labels = np.unique(labels)

    if len(unique_labels) < 2:
        return 0.0

    silhouette_vals = []

    for i in range(n):
        # a(i): mean distance to points in same cluster
        same_cluster = labels == labels[i]
        same_cluster[i] = False
        if np.sum(same_cluster) > 0:
            a_i = np.mean([np.linalg.norm(X[i] - X[j])
                          for j in range(n) if same_cluster[j]])
        else:
            a_i = 0

        # b(i): min mean distance to points in other clusters
        b_i = np.inf
        for label in unique_labels:
            if label != labels[i]:
                other_cluster = labels == label
                if np.sum(other_cluster) > 0:
                    mean_dist = np.mean([np.linalg.norm(X[i] - X[j])
                                        for j in range(n) if other_cluster[j]])
                    b_i = min(b_i, mean_dist)

        if b_i == np.inf:
            b_i = 0

        # Silhouette coefficient
        s_i = (b_i - a_i) / max(a_i, b_i) if max(a_i, b_i) > 0 else 0
        silhouette_vals.append(s_i)

    return np.mean(silhouette_vals)


def davies_bouldin_score(X: np.ndarray, labels: np.ndarray) -> float:
    """Compute Davies-Bouldin index (lower is better)."""
    unique_labels = np.unique(labels)
    n_clusters = len(unique_labels)

    if n_clusters < 2:
        return 0.0

    # Compute centroids and within-cluster scatter
    centroids = []
    scatters = []
    for label in unique_labels:
        mask = labels == label
        cluster_points = X[mask]
        centroid = cluster_points.mean(axis=0)
        centroids.append(centroid)
        scatter = np.mean([np.linalg.norm(p - centroid) for p in cluster_points])
        scatters.append(scatter)

    centroids = np.array(centroids)
    scatters = np.array(scatters)

    # Compute DB index
    db_vals = []
    for i in range(n_clusters):
        max_ratio = 0
        for j in range(n_clusters):
            if i != j:
                ratio = (scatters[i] + scatters[j]) / (np.linalg.norm(centroids[i] - centroids[j]) + 1e-10)
                max_ratio = max(max_ratio, ratio)
        db_vals.append(max_ratio)

    return np.mean(db_vals)


# Example
print("\nClustering Evaluation Metrics")
print("=" * 50)

np.random.seed(42)
X = np.vstack([
    np.random.randn(20, 2) + [2, 2],
    np.random.randn(20, 2) + [-2, -2]
])
labels = np.array([0]*20 + [1]*20)

sil = silhouette_score(X, labels)
db = davies_bouldin_score(X, labels)

print(f"Silhouette Score: {sil:.4f} (higher is better, range [-1, 1])")
print(f"Davies-Bouldin Index: {db:.4f} (lower is better)")

# Compare with random labels
random_labels = np.random.randint(0, 2, len(X))
sil_random = silhouette_score(X, random_labels)
db_random = davies_bouldin_score(X, random_labels)
print(f"\nRandom labels - Silhouette: {sil_random:.4f}, DB: {db_random:.4f}")
```

## Summary

Quantum clustering approaches:

1. **Quantum Distances**: Efficient distance computation via state overlaps
2. **Quantum K-Means**: Classical algorithm with quantum distance metric
3. **Spectral Clustering**: Quantum kernels for similarity graphs
4. **Hybrid Methods**: Combine quantum and classical components

<Tip>
Quantum clustering benefits most from the quantum kernel's ability to capture complex similarities. For simple datasets with linear separability, classical methods may be more efficient.
</Tip>
