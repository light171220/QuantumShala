# Industry Applications of Quantum ML

Quantum machine learning is being actively explored across multiple industries. This lesson covers practical applications in finance, drug discovery, materials science, and optimization, examining where quantum approaches may provide advantages.

## Quantum Finance Applications

Financial institutions are exploring quantum ML for portfolio optimization, risk analysis, and fraud detection.

<Definition>
**Quantum Portfolio Optimization**:

Maximize returns while minimizing risk using quantum approaches:

$$\max_w \left( \mu^T w - \lambda w^T \Sigma w \right)$$

where $w$ are portfolio weights, $\mu$ expected returns, $\Sigma$ the covariance matrix, and $\lambda$ the risk aversion parameter.
</Definition>

```python
import numpy as np

class QuantumPortfolioOptimizer:
    """Quantum-inspired portfolio optimization using VQE approach."""

    def __init__(self, n_assets: int, n_layers: int = 2):
        self.n_assets = n_assets
        self.n_qubits = n_assets
        self.n_layers = n_layers
        self.dim = 2 ** n_qubits
        self.n_params = n_qubits * n_layers * 2

    def encode_returns(self, expected_returns: np.ndarray) -> np.ndarray:
        """Encode expected returns into quantum state."""
        state = np.zeros(self.dim, dtype=complex)
        state[0] = 1

        for i, ret in enumerate(expected_returns):
            theta = np.pi * ret
            c, s = np.cos(theta/2), np.sin(theta/2)
            new_state = np.zeros_like(state)
            for z in range(self.dim):
                z_flip = z ^ (1 << (self.n_qubits - 1 - i))
                new_state[z] = c * state[z] - s * state[z_flip]
                new_state[z_flip] = s * state[z] + c * state[z_flip]
            state = new_state

        return state

    def variational_circuit(self, state: np.ndarray,
                            params: np.ndarray) -> np.ndarray:
        """Apply variational ansatz."""
        param_idx = 0

        for layer in range(self.n_layers):
            for q in range(self.n_qubits):
                theta = params[param_idx]
                c, s = np.cos(theta/2), np.sin(theta/2)
                new_state = np.zeros_like(state)
                for z in range(self.dim):
                    z_flip = z ^ (1 << (self.n_qubits - 1 - q))
                    bit = (z >> (self.n_qubits - 1 - q)) & 1
                    new_state[z] = c * state[z] + (1-2*bit) * s * state[z_flip]
                state = new_state
                param_idx += 1

                theta = params[param_idx]
                for z in range(self.dim):
                    bit = (z >> (self.n_qubits - 1 - q)) & 1
                    state[z] *= np.exp(1j*theta/2) if bit else np.exp(-1j*theta/2)
                param_idx += 1

            # Entanglement for correlation modeling
            for q in range(self.n_qubits - 1):
                new_state = np.zeros_like(state)
                for z in range(self.dim):
                    ctrl = (z >> (self.n_qubits - 1 - q)) & 1
                    z_new = z ^ (1 << (self.n_qubits-1-(q+1))) if ctrl else z
                    new_state[z_new] += state[z]
                state = new_state

        return state

    def get_weights(self, params: np.ndarray,
                    expected_returns: np.ndarray) -> np.ndarray:
        """Get portfolio weights from circuit output."""
        state = self.encode_returns(expected_returns)
        state = self.variational_circuit(state, params)

        # Extract weights from measurement probabilities
        weights = np.zeros(self.n_assets)
        for z in range(self.dim):
            prob = np.abs(state[z])**2
            for i in range(self.n_assets):
                if (z >> (self.n_qubits - 1 - i)) & 1:
                    weights[i] += prob

        # Normalize to sum to 1
        weights /= weights.sum() + 1e-10
        return weights

    def portfolio_objective(self, weights: np.ndarray,
                           expected_returns: np.ndarray,
                           cov_matrix: np.ndarray,
                           risk_aversion: float) -> float:
        """Compute portfolio objective (higher is better)."""
        expected_return = np.dot(weights, expected_returns)
        portfolio_risk = np.dot(weights.T, np.dot(cov_matrix, weights))
        return expected_return - risk_aversion * portfolio_risk

    def optimize(self, expected_returns: np.ndarray,
                 cov_matrix: np.ndarray,
                 risk_aversion: float = 0.5,
                 n_epochs: int = 100,
                 lr: float = 0.1) -> np.ndarray:
        """Optimize portfolio allocation."""
        params = np.random.uniform(-np.pi, np.pi, self.n_params)

        for epoch in range(n_epochs):
            gradients = np.zeros_like(params)

            for i in range(len(params)):
                params_plus = params.copy()
                params_plus[i] += np.pi/2
                params_minus = params.copy()
                params_minus[i] -= np.pi/2

                w_plus = self.get_weights(params_plus, expected_returns)
                w_minus = self.get_weights(params_minus, expected_returns)

                obj_plus = self.portfolio_objective(w_plus, expected_returns,
                                                    cov_matrix, risk_aversion)
                obj_minus = self.portfolio_objective(w_minus, expected_returns,
                                                     cov_matrix, risk_aversion)

                gradients[i] = (obj_plus - obj_minus) / 2

            params += lr * gradients  # Gradient ascent (maximize)

            if epoch % 25 == 0:
                weights = self.get_weights(params, expected_returns)
                obj = self.portfolio_objective(weights, expected_returns,
                                              cov_matrix, risk_aversion)
                print(f"Epoch {epoch}: Objective = {obj:.4f}")

        return self.get_weights(params, expected_returns)

    [0.04, 0.01, 0.005, 0.002],

print(f"Portfolio Risk (Std Dev): {risk*100:.2f}%")
print(f"Sharpe Ratio: {expected_return/risk:.2f}")
```

## Drug Discovery and Molecular Simulation

Quantum ML accelerates drug discovery through molecular property prediction and compound screening.

```python
import numpy as np

class QuantumMolecularPredictor:
    """Predict molecular properties using quantum features."""

    def __init__(self, n_qubits: int, n_layers: int = 2):
        self.n_qubits = n_qubits
        self.n_layers = n_layers
        self.dim = 2 ** n_qubits
        self.n_params = n_qubits * n_layers * 2

    def encode_molecular_features(self, features: np.ndarray) -> np.ndarray:
        """Encode molecular descriptors into quantum state."""
        state = np.zeros(self.dim, dtype=complex)
        state[0] = 1

        # Normalize features
        features = np.array(features[:self.n_qubits])
        features = features / (np.linalg.norm(features) + 1e-10)

        # Amplitude-like encoding
        for i, val in enumerate(features):
            theta = np.arccos(np.clip(val, -1, 1))
            c, s = np.cos(theta/2), np.sin(theta/2)
            new_state = np.zeros_like(state)
            for z in range(self.dim):
                z_flip = z ^ (1 << (self.n_qubits - 1 - i))
                new_state[z] = c * state[z] - s * state[z_flip]
                new_state[z_flip] = s * state[z] + c * state[z_flip]
            state = new_state

        return state

    def quantum_kernel(self, x1: np.ndarray, x2: np.ndarray,
                       params: np.ndarray) -> float:
        """Compute quantum kernel between two molecules."""
        state1 = self.encode_molecular_features(x1)
        state2 = self.encode_molecular_features(x2)

        # Apply variational transformation
        for state in [state1, state2]:
            param_idx = 0
            for _ in range(self.n_layers):
                for q in range(self.n_qubits):
                    theta = params[param_idx]
                    c, s = np.cos(theta/2), np.sin(theta/2)
                    new_state = np.zeros_like(state)
                    for z in range(self.dim):
                        z_flip = z ^ (1 << (self.n_qubits - 1 - q))
                        bit = (z >> (self.n_qubits - 1 - q)) & 1
                        new_state[z] = c * state[z] + (1-2*bit) * s * state[z_flip]
                    state[:] = new_state
                    param_idx += 1

                    theta = params[param_idx]
                    for z in range(self.dim):
                        bit = (z >> (self.n_qubits - 1 - q)) & 1
                        state[z] *= np.exp(1j*theta/2) if bit else np.exp(-1j*theta/2)
                    param_idx += 1

        return np.abs(np.vdot(state1, state2))**2

    def predict_property(self, X_train: np.ndarray, y_train: np.ndarray,
                        X_test: np.ndarray, params: np.ndarray,
                        reg: float = 0.1) -> np.ndarray:
        """Predict molecular properties using kernel regression."""
        n_train = len(X_train)

        # Build kernel matrix
        K_train = np.zeros((n_train, n_train))
        for i in range(n_train):
            for j in range(i, n_train):
                K_train[i, j] = self.quantum_kernel(X_train[i], X_train[j], params)
                K_train[j, i] = K_train[i, j]

        # Solve kernel regression
        alpha = np.linalg.solve(K_train + reg * np.eye(n_train), y_train)

        # Predict
        predictions = []
        for x_test in X_test:
            k_vec = np.array([self.quantum_kernel(x_test, x_train, params)
                             for x_train in X_train])
            predictions.append(k_vec @ alpha)

        return np.array(predictions)
```

## Materials Science Applications

Quantum ML helps predict material properties and discover new materials with desired characteristics.

```python
import numpy as np

class QuantumMaterialsPredictor:
    """Predict material properties using quantum feature encoding."""

    def __init__(self, n_qubits: int, n_layers: int = 2):
        self.n_qubits = n_qubits
        self.n_layers = n_layers
        self.dim = 2 ** n_qubits
        self.n_params = n_qubits * n_layers * 2

    def encode_composition(self, composition: np.ndarray) -> np.ndarray:
        """Encode material composition into quantum state."""
        state = np.zeros(self.dim, dtype=complex)
        state[0] = 1

        # Composition encoding (e.g., element fractions)
        for i, frac in enumerate(composition[:self.n_qubits]):
            theta = np.pi * frac
            c, s = np.cos(theta/2), np.sin(theta/2)
            new_state = np.zeros_like(state)
            for z in range(self.dim):
                z_flip = z ^ (1 << (self.n_qubits - 1 - i))
                new_state[z] = c * state[z] - s * state[z_flip]
                new_state[z_flip] = s * state[z] + c * state[z_flip]
            state = new_state

        return state

    def apply_variational_circuit(self, state: np.ndarray,
                                  params: np.ndarray) -> np.ndarray:
        """Apply variational circuit for feature extraction."""
        param_idx = 0

        for _ in range(self.n_layers):
            for q in range(self.n_qubits):
                theta = params[param_idx]
                c, s = np.cos(theta/2), np.sin(theta/2)
                new_state = np.zeros_like(state)
                for z in range(self.dim):
                    z_flip = z ^ (1 << (self.n_qubits - 1 - q))
                    bit = (z >> (self.n_qubits - 1 - q)) & 1
                    new_state[z] = c * state[z] + (1-2*bit) * s * state[z_flip]
                state = new_state
                param_idx += 1

                theta = params[param_idx]
                for z in range(self.dim):
                    bit = (z >> (self.n_qubits - 1 - q)) & 1
                    state[z] *= np.exp(1j*theta/2) if bit else np.exp(-1j*theta/2)
                param_idx += 1

            # Entanglement (model element interactions)
            for q in range(self.n_qubits - 1):
                new_state = np.zeros_like(state)
                for z in range(self.dim):
                    ctrl = (z >> (self.n_qubits - 1 - q)) & 1
                    z_new = z ^ (1 << (self.n_qubits-1-(q+1))) if ctrl else z
                    new_state[z_new] += state[z]
                state = new_state

        return state

    def predict_property(self, composition: np.ndarray,
                        params: np.ndarray) -> float:
        """Predict material property from composition."""
        state = self.encode_composition(composition)
        state = self.apply_variational_circuit(state, params)

        # Output: expectation value
        exp_val = 0
        for z in range(self.dim):
            bit = (z >> (self.n_qubits - 1)) & 1
            exp_val += (1 - 2*bit) * np.abs(state[z])**2

        return exp_val
```

## Optimization Applications

Quantum ML solves combinatorial optimization problems in logistics, scheduling, and resource allocation.

```python
import numpy as np

class QuantumOptimizer:
    """QAOA-inspired optimizer for combinatorial problems."""

    def __init__(self, n_variables: int, n_layers: int = 2):
        self.n_variables = n_variables
        self.n_qubits = n_variables
        self.n_layers = n_layers
        self.dim = 2 ** n_qubits
        self.n_params = 2 * n_layers  # gamma and beta for each layer

    def cost_function(self, bitstring: int, weights: np.ndarray) -> float:
        """Evaluate cost for a given bitstring."""
        cost = 0
        for i in range(self.n_variables):
            bit_i = (bitstring >> (self.n_variables - 1 - i)) & 1
            cost += weights[i] * bit_i

            # Pairwise interactions
            for j in range(i + 1, self.n_variables):
                bit_j = (bitstring >> (self.n_variables - 1 - j)) & 1
                cost += 0.1 * bit_i * bit_j

        return cost

    def qaoa_circuit(self, params: np.ndarray,
                    weights: np.ndarray) -> np.ndarray:
        """Run QAOA circuit."""
        # Initialize in superposition
        state = np.ones(self.dim, dtype=complex) / np.sqrt(self.dim)

        for layer in range(self.n_layers):
            gamma = params[2 * layer]
            beta = params[2 * layer + 1]

            # Cost unitary: exp(-i * gamma * C)
            for z in range(self.dim):
                cost = self.cost_function(z, weights)
                state[z] *= np.exp(-1j * gamma * cost)

            # Mixer unitary: exp(-i * beta * B)
            for q in range(self.n_qubits):
                new_state = np.zeros_like(state)
                c, s = np.cos(beta), np.sin(beta)

                for z in range(self.dim):
                    z_flip = z ^ (1 << (self.n_qubits - 1 - q))
                    new_state[z] += c * state[z] - 1j * s * state[z_flip]

                state = new_state
                state /= np.linalg.norm(state)

        return state

    def expectation(self, params: np.ndarray, weights: np.ndarray) -> float:
        """Compute expected cost."""
        state = self.qaoa_circuit(params, weights)
        exp_cost = 0
        for z in range(self.dim):
            prob = np.abs(state[z])**2
            cost = self.cost_function(z, weights)
            exp_cost += prob * cost
        return exp_cost

    def optimize(self, weights: np.ndarray,
                 n_epochs: int = 100, lr: float = 0.1) -> tuple:
        """Find optimal solution."""
        params = np.random.uniform(0, np.pi, self.n_params)

        for epoch in range(n_epochs):
            gradients = np.zeros_like(params)

            for i in range(len(params)):
                params_plus = params.copy()
                params_plus[i] += np.pi/2
                params_minus = params.copy()
                params_minus[i] -= np.pi/2

                exp_plus = self.expectation(params_plus, weights)
                exp_minus = self.expectation(params_minus, weights)
                gradients[i] = (exp_plus - exp_minus) / 2

            params -= lr * gradients

        # Get solution
        state = self.qaoa_circuit(params, weights)
        probs = np.abs(state)**2
        best_bitstring = np.argmax(probs)

        return best_bitstring, self.cost_function(best_bitstring, weights)
```
