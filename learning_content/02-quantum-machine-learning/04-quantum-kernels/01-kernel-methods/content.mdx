# Introduction to Kernel Methods

Kernel methods are a class of algorithms that use kernel functions to operate in high-dimensional feature spaces without explicitly computing the coordinates in those spaces.

## Learning Objectives

After completing this lesson, you will be able to:

- Understand the kernel trick and its importance
- Define positive definite kernels
- Apply Mercer's theorem
- Implement common kernel functions
- Recognize when kernels are beneficial

## 1. The Kernel Trick

The kernel trick allows algorithms to compute inner products in high-dimensional feature spaces efficiently.

<Definition term="Kernel Function">
A kernel function K(x, y) computes the inner product between two data points in a (possibly infinite-dimensional) feature space: K(x, y) = ⟨φ(x), φ(y)⟩, where φ is a feature map.
</Definition>

```python
import numpy as np

def kernel_trick_demonstration():
    """
    Demonstrate how kernels compute inner products efficiently.
    """
    print("The Kernel Trick:")
    print("=" * 50)

    # Example: Polynomial feature map for 2D data
    def explicit_polynomial_features(x, degree=2):
        """Explicitly compute polynomial features."""
        x1, x2 = x
        # For degree 2: [1, x1, x2, x1², x1x2, x2²]
        return np.array([
            1,
            np.sqrt(2) * x1,
            np.sqrt(2) * x2,
            x1**2,
            np.sqrt(2) * x1 * x2,
            x2**2
        ])

    def polynomial_kernel(x, y, degree=2):
        """Compute polynomial kernel directly."""
        return (1 + np.dot(x, y))**degree

    # Test data
    x = np.array([1.0, 2.0])
    y = np.array([3.0, 4.0])

    print("\n1. Explicit Feature Space Method:")
    print("-" * 40)
    phi_x = explicit_polynomial_features(x)
    phi_y = explicit_polynomial_features(y)
    explicit_inner_product = np.dot(phi_x, phi_y)
    print(f"  φ(x) dimension: {len(phi_x)}")
    print(f"  ⟨φ(x), φ(y)⟩ = {explicit_inner_product:.4f}")

    print("\n2. Kernel Trick Method:")
    print("-" * 40)
    kernel_value = polynomial_kernel(x, y)
    print(f"  K(x, y) = (1 + x·y)² = {kernel_value:.4f}")

    print(f"\n  Both methods give the same result!")
    print(f"  But kernel: O(d) vs explicit: O(d^degree)")

kernel_trick_demonstration()
```

<Tip>
The kernel trick is especially powerful when the feature space is infinite-dimensional (like the RBF kernel), making explicit computation impossible.
</Tip>

## 2. Positive Definite Kernels

Not every function can serve as a valid kernel. Valid kernels must be positive definite.

```python
import numpy as np

def positive_definite_kernels():
    """
    Explore properties of positive definite kernels.
    """
    print("Positive Definite Kernels:")
    print("=" * 50)

    def is_positive_semidefinite(matrix, tol=1e-10):
        """Check if a matrix is positive semi-definite."""
        eigenvalues = np.linalg.eigvalsh(matrix)
        return np.all(eigenvalues >= -tol)

    def compute_gram_matrix(X, kernel_fn):
        """Compute Gram matrix K_ij = K(x_i, x_j)."""
        n = len(X)
        K = np.zeros((n, n))
        for i in range(n):
            for j in range(n):
                K[i, j] = kernel_fn(X[i], X[j])
        return K

    # Sample data
    np.random.seed(42)
    X = np.random.randn(5, 2)

    print("\n1. Valid Kernels (Positive Definite):")
    print("-" * 40)

    kernels = {
        'Linear': lambda x, y: np.dot(x, y),
        'Polynomial (d=2)': lambda x, y: (1 + np.dot(x, y))**2,
        'RBF (γ=1)': lambda x, y: np.exp(-np.sum((x-y)**2)),
    }

    for name, kernel in kernels.items():
        K = compute_gram_matrix(X, kernel)
        is_psd = is_positive_semidefinite(K)
        min_eig = np.min(np.linalg.eigvalsh(K))
        print(f"\n  {name}:")
        print(f"    PSD: {is_psd}")
        print(f"    Min eigenvalue: {min_eig:.6f}")

    print("\n\n2. Invalid 'Kernel' Example:")
    print("-" * 40)

    def invalid_kernel(x, y):
        """Not a valid kernel - can produce negative eigenvalues."""
        return np.dot(x, y) - 2  # Subtracting constant breaks PSD property

    K_invalid = compute_gram_matrix(X, invalid_kernel)
    is_psd = is_positive_semidefinite(K_invalid)
    min_eig = np.min(np.linalg.eigvalsh(K_invalid))
    print(f"  K(x,y) = x·y - 2:")
    print(f"    PSD: {is_psd}")
    print(f"    Min eigenvalue: {min_eig:.6f}")
    print(f"    ⚠ Not a valid kernel!")

positive_definite_kernels()
```

<Formula>
Positive Definite Condition:
$$\sum_{i,j} c_i c_j K(x_i, x_j) \geq 0$$
for all finite sets {x₁, ..., xₙ} and all real numbers c₁, ..., cₙ.
</Formula>

## 3. Mercer's Theorem

Mercer's theorem provides the theoretical foundation for kernel methods.

```python
import numpy as np

def mercers_theorem():
    """
    Illustrate Mercer's theorem concepts.
    """
    print("Mercer's Theorem:")
    print("=" * 50)

    print("\n1. Statement:")
    print("-" * 40)
    print("  If K(x,y) is a continuous, symmetric, positive")
    print("  definite kernel, then there exists a feature map φ")
    print("  such that K(x,y) = ⟨φ(x), φ(y)⟩")

    print("\n\n2. Eigenfunction Expansion:")
    print("-" * 40)
    print("  K(x,y) = Σᵢ λᵢ ψᵢ(x) ψᵢ(y)")
    print("  where λᵢ ≥ 0 are eigenvalues")
    print("  and ψᵢ are orthonormal eigenfunctions")

    # Demonstrate with RBF kernel approximation
    print("\n\n3. RBF Kernel Example:")
    print("-" * 40)

    def rbf_kernel(x, y, gamma=1.0):
        return np.exp(-gamma * np.sum((x - y)**2))

    # Approximate feature map using Random Fourier Features
    def random_fourier_features(X, n_features=100, gamma=1.0):
        """Approximate RBF kernel via random Fourier features."""
        d = X.shape[1]
        np.random.seed(42)

        # Sample from N(0, 2*gamma)
        W = np.random.randn(d, n_features) * np.sqrt(2 * gamma)
        b = np.random.uniform(0, 2*np.pi, n_features)

        # Compute features: sqrt(2/D) * cos(Wx + b)
        Z = np.sqrt(2 / n_features) * np.cos(X @ W + b)
        return Z

    # Test approximation
    X = np.random.randn(5, 3)
    x1, x2 = X[0], X[1]

    exact_kernel = rbf_kernel(x1, x2)

    for n_features in [10, 50, 100, 500]:
        Z = random_fourier_features(X, n_features)
        approx_kernel = np.dot(Z[0], Z[1])
        error = abs(exact_kernel - approx_kernel)
        print(f"  D={n_features:3d}: K̂={approx_kernel:.4f}, Error={error:.4f}")

    print(f"\n  Exact K(x₁,x₂) = {exact_kernel:.4f}")

mercers_theorem()
```

<Warning>
Mercer's theorem guarantees existence of a feature map but doesn't provide it explicitly. The feature space may be infinite-dimensional.
</Warning>

## 4. Common Kernel Functions

Several kernel functions are widely used in practice.

```python
import numpy as np

def common_kernels():
    """
    Overview of commonly used kernel functions.
    """
    print("Common Kernel Functions:")
    print("=" * 50)

    class KernelLibrary:
        """Collection of standard kernels."""

        @staticmethod
        def linear(x, y):
            """Linear kernel: K(x,y) = x·y"""
            return np.dot(x, y)

        @staticmethod
        def polynomial(x, y, degree=3, coef0=1):
            """Polynomial kernel: K(x,y) = (γx·y + c)^d"""
            gamma = 1.0 / len(x)
            return (gamma * np.dot(x, y) + coef0)**degree

        @staticmethod
        def rbf(x, y, gamma=1.0):
            """RBF/Gaussian kernel: K(x,y) = exp(-γ||x-y||²)"""
            return np.exp(-gamma * np.sum((x - y)**2))

        @staticmethod
        def sigmoid(x, y, gamma=1.0, coef0=0):
            """Sigmoid kernel: K(x,y) = tanh(γx·y + c)"""
            return np.tanh(gamma * np.dot(x, y) + coef0)

        @staticmethod
        def laplacian(x, y, gamma=1.0):
            """Laplacian kernel: K(x,y) = exp(-γ||x-y||₁)"""
            return np.exp(-gamma * np.sum(np.abs(x - y)))

    # Test on sample data
    np.random.seed(42)
    x = np.array([1.0, 2.0, 3.0])
    y = np.array([2.0, 3.0, 1.0])

    print("\nKernel Values for x=[1,2,3], y=[2,3,1]:")
    print("-" * 40)

    kernels = [
        ("Linear", KernelLibrary.linear),
        ("Polynomial (d=3)", lambda a,b: KernelLibrary.polynomial(a,b,3)),
        ("RBF (γ=0.5)", lambda a,b: KernelLibrary.rbf(a,b,0.5)),
        ("Laplacian (γ=0.5)", lambda a,b: KernelLibrary.laplacian(a,b,0.5)),
    ]

    for name, kernel in kernels:
        value = kernel(x, y)
        print(f"  {name:<20}: {value:.4f}")

    # Properties comparison
    print("\n\nKernel Properties:")
    print("-" * 40)

    properties = [
        ("Linear", "Finite (d)", "Linear boundaries", "Fastest"),
        ("Polynomial", "Finite (d+n choose n)", "Polynomial boundaries", "Fast"),
        ("RBF", "Infinite", "Complex boundaries", "Moderate"),
        ("Laplacian", "Infinite", "Sharp boundaries", "Moderate"),
    ]

    print(f"  {'Kernel':<12} {'Feature Dim':<20} {'Decision Boundary':<20}")
    print("  " + "-" * 52)
    for name, dim, boundary, _ in properties:
        print(f"  {name:<12} {dim:<20} {boundary:<20}")

common_kernels()
```

## 5. Kernel Properties and Combinations

Kernels can be combined to create new valid kernels.

```python
import numpy as np

def kernel_combinations():
    """
    Demonstrate valid kernel combinations.
    """
    print("Kernel Combinations:")
    print("=" * 50)

    def linear_kernel(x, y):
        return np.dot(x, y)

    def rbf_kernel(x, y, gamma=1.0):
        return np.exp(-gamma * np.sum((x - y)**2))

    print("\n1. Valid Combination Rules:")
    print("-" * 40)

    rules = [
        ("Sum", "K₁ + K₂", "Valid if both K₁, K₂ valid"),
        ("Product", "K₁ × K₂", "Valid if both K₁, K₂ valid"),
        ("Scalar", "c × K (c > 0)", "Valid if K valid"),
        ("Polynomial", "p(K) (pos. coeffs)", "Valid if K valid"),
        ("Exponential", "exp(K)", "Valid if K valid"),
    ]

    for name, formula, condition in rules:
        print(f"  {name:<12}: {formula:<15} - {condition}")

    # Demonstrate combinations
    print("\n\n2. Example Combinations:")
    print("-" * 40)

    x = np.array([1.0, 2.0])
    y = np.array([2.0, 1.0])

    k_linear = linear_kernel(x, y)
    k_rbf = rbf_kernel(x, y, 0.5)

    print(f"\n  Base kernels:")
    print(f"    K_linear(x,y) = {k_linear:.4f}")
    print(f"    K_rbf(x,y) = {k_rbf:.4f}")

    print(f"\n  Combined kernels:")
    print(f"    K_sum = K_linear + K_rbf = {k_linear + k_rbf:.4f}")
    print(f"    K_prod = K_linear × K_rbf = {k_linear * k_rbf:.4f}")
    print(f"    K_exp = exp(K_linear) = {np.exp(k_linear):.4f}")

    # Multi-scale kernel
    print("\n\n3. Multi-Scale RBF Kernel:")
    print("-" * 40)

    def multiscale_rbf(x, y, gammas=[0.1, 1.0, 10.0], weights=None):
        """Combine RBF kernels at multiple scales."""
        if weights is None:
            weights = np.ones(len(gammas)) / len(gammas)

        total = 0
        for gamma, weight in zip(gammas, weights):
            total += weight * rbf_kernel(x, y, gamma)
        return total

    k_multi = multiscale_rbf(x, y)
    print(f"  K_multiscale(x,y) = {k_multi:.4f}")
    print(f"  Combines: γ ∈ {{0.1, 1.0, 10.0}}")

kernel_combinations()
```

<Tip>
Multi-scale kernels are useful when your data has features at different scales or when you're uncertain about the optimal kernel parameter.
</Tip>

## Summary

Kernel methods provide powerful tools for non-linear machine learning:

1. **Kernel Trick**: Compute inner products without explicit feature maps
2. **Positive Definite**: Valid kernels ensure well-defined optimization
3. **Mercer's Theorem**: Theoretical foundation linking kernels to features
4. **Common Kernels**: Linear, polynomial, RBF, Laplacian
5. **Combinations**: Sum, product, and other valid operations
6. **Use Cases**: Best for small-medium datasets with complex patterns

Understanding classical kernels is essential before exploring quantum kernels.
