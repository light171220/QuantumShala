## 1. Quantum Support Vector Machine

Using quantum kernels with SVM.

<Definition term="Quantum SVM (QSVM)">
A support vector machine that uses a quantum kernel to compute similarities between data points. The kernel matrix is computed on a quantum computer, while the optimization is performed classically.
</Definition>

```python
import numpy as np

class QuantumSVM:
    """Support Vector Machine with quantum kernel."""

    def __init__(self, n_qubits=2, C=1.0):
        self.n_qubits = n_qubits
        self.dim = 2 ** n_qubits
        self.C = C  # Regularization
        self.alpha = None
        self.support_vectors = None
        self.support_labels = None
        self.b = 0

    def quantum_kernel(self, x1, x2):
        """Compute quantum kernel value."""
        state1 = self._feature_map(x1)
        state2 = self._feature_map(x2)
        return np.abs(np.vdot(state1, state2)) ** 2

    def _feature_map(self, x):
        """ZZ-type feature map."""
        state = np.ones(self.dim, dtype=complex) / np.sqrt(self.dim)

        for i in range(self.dim):
            bits = format(i, f'0{self.n_qubits}b')
            phase = 0

            for q in range(self.n_qubits):
                if bits[q] == '1' and q < len(x):
                    phase += x[q]

            for q1 in range(self.n_qubits):
                for q2 in range(q1 + 1, self.n_qubits):
                    if bits[q1] == '1' and bits[q2] == '1':
                        if q1 < len(x) and q2 < len(x):
                            phase += x[q1] * x[q2]

            state[i] *= np.exp(1j * phase)

        return state

    def build_kernel_matrix(self, X):
        """Build full kernel matrix."""
        n = len(X)
        K = np.zeros((n, n))
        for i in range(n):
            for j in range(i, n):
                K[i, j] = self.quantum_kernel(X[i], X[j])
                K[j, i] = K[i, j]
        return K

    def fit(self, X, y, max_iter=1000, tol=1e-5):
        """
        Train SVM using simplified SMO-like algorithm.
        """
        n = len(X)
        self.X_train = X
        self.y_train = np.array(y)

        # Build kernel matrix
        K = self.build_kernel_matrix(X)

        # Initialize
        self.alpha = np.zeros(n)

        # Simplified optimization (coordinate ascent)
        for iteration in range(max_iter):
            alpha_old = self.alpha.copy()

            for i in range(n):
                # Compute gradient
                grad = 1 - y[i] * np.sum(self.alpha * y * K[i, :])

                # Update with constraint
                self.alpha[i] = max(0, min(self.C, self.alpha[i] + grad / (K[i, i] + 1e-8)))

            # Check convergence
            if np.max(np.abs(self.alpha - alpha_old)) < tol:
                break

        # Find support vectors
        sv_mask = self.alpha > 1e-7
        self.support_vectors = X[sv_mask]
        self.support_labels = y[sv_mask]
        self.support_alpha = self.alpha[sv_mask]
        self.support_K = K[sv_mask][:, sv_mask]

        # Compute bias
        self.b = np.mean(self.support_labels -
                        np.sum(self.support_alpha * self.support_labels *
                              self.support_K, axis=1))

        return self

    def predict(self, X):
        """Predict class labels."""
        predictions = []

        for x in X:
            k_vals = np.array([self.quantum_kernel(x, sv) for sv in self.support_vectors])
            decision = np.sum(self.support_alpha * self.support_labels * k_vals) + self.b
            predictions.append(np.sign(decision))

        return np.array(predictions)

    def score(self, X, y):
        """Compute accuracy."""
        predictions = self.predict(X)
        return np.mean(predictions == y)
```

```python
import numpy as np

class QuantumKernelRidge:
    """Kernel Ridge Regression with quantum kernel."""

    def __init__(self, n_qubits=2, alpha=1.0):
        self.n_qubits = n_qubits
        self.dim = 2 ** n_qubits
        self.alpha = alpha  # Regularization
        self.weights = None
        self.X_train = None

    def quantum_kernel(self, x1, x2):
        """Compute quantum kernel."""
        state1 = self._feature_map(x1)
        state2 = self._feature_map(x2)
        return np.abs(np.vdot(state1, state2)) ** 2

    def _feature_map(self, x):
        state = np.ones(self.dim, dtype=complex) / np.sqrt(self.dim)

        for i in range(self.dim):
            bits = format(i, f'0{self.n_qubits}b')
            phase = 0
            for q in range(self.n_qubits):
                if bits[q] == '1' and q < len(x):
                    phase += x[q]
            state[i] *= np.exp(1j * phase)

        return state

    def build_kernel_matrix(self, X1, X2=None):
        """Build kernel matrix between X1 and X2."""
        if X2 is None:
            X2 = X1

        K = np.array([[self.quantum_kernel(x1, x2) for x2 in X2] for x1 in X1])
        return K

    def fit(self, X, y):
        """Fit kernel ridge regression."""
        self.X_train = X
        n = len(X)

        K = self.build_kernel_matrix(X)

        # Kernel ridge: w = (K + αI)^{-1} y
        self.weights = np.linalg.solve(K + self.alpha * np.eye(n), y)

        return self

    def predict(self, X):
        """Predict target values."""
        K_test = self.build_kernel_matrix(X, self.X_train)
        return K_test @ self.weights

    def score(self, X, y):
        """Compute R² score."""
        predictions = self.predict(X)
        ss_res = np.sum((y - predictions) ** 2)
        ss_tot = np.sum((y - np.mean(y)) ** 2)
        return 1 - ss_res / ss_tot
```

## 3. Anomaly Detection with Quantum Kernels

Using kernel methods for outlier detection.

```python
import numpy as np

class QuantumOneClassSVM:
    """One-Class SVM with quantum kernel for anomaly detection."""

    def __init__(self, n_qubits=2, nu=0.1):
        self.n_qubits = n_qubits
        self.dim = 2 ** n_qubits
        self.nu = nu  # Fraction of outliers
        self.threshold = None
        self.X_train = None

    def quantum_kernel(self, x1, x2):
        state1 = self._feature_map(x1)
        state2 = self._feature_map(x2)
        return np.abs(np.vdot(state1, state2)) ** 2

    def _feature_map(self, x):
        state = np.ones(self.dim, dtype=complex) / np.sqrt(self.dim)
        for i in range(self.dim):
            bits = format(i, f'0{self.n_qubits}b')
            phase = sum(x[q] for q in range(self.n_qubits) if bits[q] == '1' and q < len(x))
            state[i] *= np.exp(1j * phase)
        return state

    def fit(self, X):
        """Fit one-class SVM."""
        self.X_train = X
        n = len(X)

        # Build kernel matrix
        K = np.array([[self.quantum_kernel(X[i], X[j])
                      for j in range(n)] for i in range(n)])

        # Compute average kernel distance to all training points
        self.mean_K = np.mean(K, axis=0)

        # Decision function: similarity to training distribution
        scores = np.array([self._decision_score(X[i]) for i in range(n)])

        # Set threshold based on nu (fraction of outliers)
        self.threshold = np.percentile(scores, 100 * self.nu)

        return self

    def _decision_score(self, x):
        """Compute decision score (higher = more normal)."""
        k_vals = np.array([self.quantum_kernel(x, x_train)
                         for x_train in self.X_train])
        return np.mean(k_vals)

    def predict(self, X):
        """Predict: 1 for normal, -1 for anomaly."""
        predictions = []
        for x in X:
            score = self._decision_score(x)
            predictions.append(1 if score >= self.threshold else -1)
        return np.array(predictions)

    def decision_function(self, X):
        """Return decision scores."""
        return np.array([self._decision_score(x) for x in X])

def anomaly_detection_demo():
    """Demonstrate quantum kernel anomaly detection."""

    print("Quantum Kernel Anomaly Detection")
    print("=" * 60)

    np.random.seed(42)

    # Normal data (cluster)
    n_normal = 50
    X_normal = np.random.randn(n_normal, 2) * 0.5 + [0, 0]

    # Anomalies (scattered)
    n_anomalies = 10
    X_anomalies = np.random.uniform(-3, 3, (n_anomalies, 2))

    # Train on normal data only
    print("\nTraining on normal data only...")
    detector = QuantumOneClassSVM(n_qubits=2, nu=0.1)
    detector.fit(X_normal)

    # Test on mixed data
    X_test = np.vstack([X_normal[:10], X_anomalies])
    y_true = np.array([1] * 10 + [-1] * n_anomalies)

    predictions = detector.predict(X_test)
    scores = detector.decision_function(X_test)

    # Evaluate
    tp = np.sum((predictions == -1) & (y_true == -1))  # Anomalies detected
    fp = np.sum((predictions == -1) & (y_true == 1))   # False alarms
    tn = np.sum((predictions == 1) & (y_true == 1))    # Normal correct
    fn = np.sum((predictions == 1) & (y_true == -1))   # Missed anomalies

    precision = tp / (tp + fp) if (tp + fp) > 0 else 0
    recall = tp / (tp + fn) if (tp + fn) > 0 else 0

    print(f"\nResults:")
    print(f"  Threshold: {detector.threshold:.4f}")
    print(f"  True Positives (anomalies detected): {tp}")
    print(f"  False Positives (false alarms): {fp}")
    print(f"  Precision: {precision:.2%}")
    print(f"  Recall: {recall:.2%}")

    print("\nSample decision scores:")
    print("  Normal samples:", np.round(scores[:5], 4))
    print("  Anomaly samples:", np.round(scores[-5:], 4))
```

## 4. Quantum vs Classical Kernel Comparison

When do quantum kernels provide advantage?

```python
import numpy as np

def kernel_comparison_study():
    """Compare quantum and classical kernels on various datasets."""

    print("Quantum vs Classical Kernel Comparison")
    print("=" * 60)

    np.random.seed(42)

    # Define classical kernels
    def rbf_kernel(X, gamma=1.0):
        sq_dist = np.sum((X[:, None, :] - X[None, :, :]) ** 2, axis=2)
        return np.exp(-gamma * sq_dist)

    def polynomial_kernel(X, degree=2):
        return (X @ X.T + 1) ** degree

    # Define quantum kernel
    def quantum_kernel_matrix(X, n_qubits=2):
        dim = 2 ** n_qubits
        n = len(X)

        def feature_map(x):
            state = np.ones(dim, dtype=complex) / np.sqrt(dim)
            for i in range(dim):
                bits = format(i, f'0{n_qubits}b')
                phase = sum(x[q] for q in range(n_qubits) if bits[q] == '1' and q < len(x))
                for q1 in range(n_qubits):
                    for q2 in range(q1 + 1, n_qubits):
                        if bits[q1] == '1' and bits[q2] == '1':
                            if q1 < len(x) and q2 < len(x):
                                phase += x[q1] * x[q2]
                state[i] *= np.exp(1j * phase)
            return state

        K = np.zeros((n, n))
        for i in range(n):
            for j in range(i, n):
                s1, s2 = feature_map(X[i]), feature_map(X[j])
                K[i, j] = np.abs(np.vdot(s1, s2)) ** 2
                K[j, i] = K[i, j]
        return K

    # Kernel SVM helper
    def kernel_svm_accuracy(K_train, y_train, K_test, y_test, alpha=0.1):
        n = len(K_train)
        w = np.linalg.solve(K_train + alpha * np.eye(n), y_train)
        predictions = np.sign(K_test @ w)
        return np.mean(predictions == y_test)

    # Test datasets
    datasets = {}

    # Dataset 1: Linear
    n = 40
    X1 = np.random.randn(n, 2)
    y1 = np.sign(X1[:, 0] + X1[:, 1])
    datasets["Linear"] = (X1, y1)

    # Dataset 2: XOR
    X2 = np.random.randn(n, 2)
    y2 = np.sign(X2[:, 0] * X2[:, 1])
    datasets["XOR"] = (X2, y2)

    # Dataset 3: Circular
    r = np.random.uniform(0, 2, n)
    theta = np.random.uniform(0, 2 * np.pi, n)
    X3 = np.column_stack([r * np.cos(theta), r * np.sin(theta)])
    y3 = np.sign(r - 1)
    datasets["Circular"] = (X3, y3)

    # Compare on each dataset
    print("\nAccuracy Comparison (train/test split):")
    print("-" * 70)
    print(f"{'Dataset':<12} {'RBF':<15} {'Polynomial':<15} {'Quantum':<15}")
    print("-" * 70)

    for name, (X, y) in datasets.items():
        # Train/test split
        idx = np.random.permutation(n)
        n_train = int(0.75 * n)
        X_train, X_test = X[idx[:n_train]], X[idx[n_train:]]
        y_train, y_test = y[idx[:n_train]], y[idx[n_train:]]

        # Build kernels
        K_rbf_train = rbf_kernel(X_train, gamma=1.0)
        K_rbf_test = rbf_kernel(X_test, gamma=1.0)
        # Cross-kernel for test
        sq_dist = np.sum((X_test[:, None, :] - X_train[None, :, :]) ** 2, axis=2)
        K_rbf_test = np.exp(-1.0 * sq_dist)

        K_poly_train = polynomial_kernel(X_train, degree=2)
        K_poly_test = (X_test @ X_train.T + 1) ** 2

        K_q_train = quantum_kernel_matrix(X_train)
        # Cross quantum kernel
        dim = 4
        def qfm(x):
            state = np.ones(dim, dtype=complex) / np.sqrt(dim)
            for i in range(dim):
                bits = format(i, '02b')
                phase = sum(x[q] for q in range(2) if bits[q] == '1' and q < len(x))
                if bits[0] == '1' and bits[1] == '1' and len(x) >= 2:
                    phase += x[0] * x[1]
                state[i] *= np.exp(1j * phase)
            return state
        K_q_test = np.array([[np.abs(np.vdot(qfm(x1), qfm(x2)))**2
                            for x2 in X_train] for x1 in X_test])

        # Accuracies
        acc_rbf = kernel_svm_accuracy(K_rbf_train, y_train, K_rbf_test, y_test)
        acc_poly = kernel_svm_accuracy(K_poly_train, y_train, K_poly_test, y_test)
        acc_q = kernel_svm_accuracy(K_q_train, y_train, K_q_test, y_test)

        print(f"{name:<12} {acc_rbf:<15.2%} {acc_poly:<15.2%} {acc_q:<15.2%}")

    print("\nNote: Quantum advantage depends on data structure and feature map choice.")
```

```python
    print("\n" + "=" * 60)
```

```python
    n_qubits = 2
    dim = 4

    def qk(x1, x2):
        def fm(x):
            state = np.ones(dim, dtype=complex) / np.sqrt(dim)
            for i in range(dim):
                bits = format(i, '02b')
                phase = sum(x[q] for q in range(2) if bits[q] == '1' and q < len(x))
                state[i] *= np.exp(1j * phase)
            return state
        return np.abs(np.vdot(fm(x1), fm(x2))) ** 2

    K = np.array([[qk(X_scaled[i], X_scaled[j])

    diag_ones = np.allclose(np.diag(K), 1)

    print(f"  Diagonal = 1: {diag_ones}")
```
