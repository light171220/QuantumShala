from dataclasses import dataclass
from typing import Dict, List, Optional
from datetime import date
from enum import Enum
import numpy as np

class MilestoneType(Enum):
    QUBIT_COUNT = "qubit_count"
    ERROR_RATE = "error_rate"
    COHERENCE_TIME = "coherence_time"
    LOGICAL_QUBITS = "logical_qubits"
    QUANTUM_VOLUME = "quantum_volume"

@dataclass
class HardwareMilestone:
    provider: str
    milestone_type: MilestoneType
    value: float
    date_achieved: date
    date_projected: Optional[date] = None  # For future milestones

@dataclass
class HardwareRoadmap:
    provider: str
    technology: str
    current_qubits: int
    projected_qubits_2025: int
    projected_qubits_2030: int
    current_error_rate: float
    target_error_rate: float
    fault_tolerance_timeline: int  # Years from now

class HardwareTracker:
    def __init__(self):
        self.milestones: List[HardwareMilestone] = []
        self.roadmaps: Dict[str, HardwareRoadmap] = {}

    def add_milestone(self, milestone: HardwareMilestone):
        self.milestones.append(milestone)

    def add_roadmap(self, roadmap: HardwareRoadmap):
        self.roadmaps[roadmap.provider] = roadmap

    def project_capability(self, year: int) -> Dict[str, Dict]:
        """Project hardware capabilities for a given year."""
        projections = {}
        current_year = 2025

        for provider, roadmap in self.roadmaps.items():
            years_ahead = year - current_year

            # Exponential qubit growth model
            growth_rate = (roadmap.projected_qubits_2030 / roadmap.current_qubits) ** (1/5)
            projected_qubits = int(roadmap.current_qubits * (growth_rate ** years_ahead))

            # Logarithmic error improvement
            error_improvement = 0.5 ** (years_ahead / 3)  # Halve every 3 years
            projected_error = roadmap.current_error_rate * error_improvement

            projections[provider] = {
                'qubits': min(projected_qubits, 1000000),  # Cap at 1M
                'error_rate': max(projected_error, roadmap.target_error_rate),
                'fault_tolerant': years_ahead >= roadmap.fault_tolerance_timeline
            }

        return projections

    def quantum_advantage_timeline(self, application: str) -> Dict:
        """Estimate when quantum advantage might be achieved."""
        requirements = {
            'portfolio_optimization': {'qubits': 1000, 'error_rate': 0.001},
            'derivatives_pricing': {'qubits': 5000, 'error_rate': 0.0001},
            'risk_analysis': {'qubits': 2000, 'error_rate': 0.001},
            'machine_learning': {'qubits': 10000, 'error_rate': 0.0001}
        }

        req = requirements.get(application, {'qubits': 5000, 'error_rate': 0.001})

        for year in range(2025, 2040):
            capabilities = self.project_capability(year)
            for provider, cap in capabilities.items():
                if cap['qubits'] >= req['qubits'] and cap['error_rate'] <= req['error_rate']:
                    return {
                        'application': application,
                        'estimated_year': year,
                        'enabling_provider': provider,
                        'requirements': req
                    }

        return {
            'application': application,
            'estimated_year': 'beyond_2040',
            'requirements': req
        }

from dataclasses import dataclass
from typing import Dict, List
from enum import Enum

class MaturityLevel(Enum):
    THEORETICAL = 1
    RESEARCH = 2
    PROOF_OF_CONCEPT = 3
    PILOT = 4
    PRODUCTION = 5

@dataclass
class EmergingApplication:
    name: str
    description: str
    current_maturity: MaturityLevel
    estimated_production_year: int
    potential_speedup: str
    key_challenges: List[str]
    enabling_technologies: List[str]

class EmergingApplicationTracker:
    def __init__(self):
        self.applications: List[EmergingApplication] = []

    def add_application(self, app: EmergingApplication):
        self.applications.append(app)

    def by_maturity(self) -> Dict[str, List[str]]:
        """Group applications by maturity level."""
        result = {}
        for level in MaturityLevel:
            apps = [a.name for a in self.applications if a.current_maturity == level]
            if apps:
                result[level.name] = apps
        return result

    def near_term_opportunities(self, horizon_years: int = 5) -> List[Dict]:
        """Identify applications likely to reach production soon."""
        current_year = 2025
        target_year = current_year + horizon_years

        opportunities = []
        for app in self.applications:
            if app.estimated_production_year <= target_year:
                opportunities.append({
                    'name': app.name,
                    'years_away': app.estimated_production_year - current_year,
                    'speedup': app.potential_speedup,
                    'readiness': app.current_maturity.value / 5
                })

        return sorted(opportunities, key=lambda x: x['years_away'])

    def technology_requirements(self) -> Dict[str, List[str]]:
        """Map technologies to applications they enable."""
        tech_map = {}
        for app in self.applications:
            for tech in app.enabling_technologies:
                if tech not in tech_map:
                    tech_map[tech] = []
                tech_map[tech].append(app.name)
        return tech_map

from dataclasses import dataclass
from typing import Dict, List
from enum import Enum
from datetime import date

class CryptoType(Enum):
    SYMMETRIC = "symmetric"
    HASH = "hash"
    PUBLIC_KEY = "public_key"
    SIGNATURE = "signature"
    KEY_EXCHANGE = "key_exchange"

class QuantumThreatLevel(Enum):
    NONE = "none"
    LOW = "low"
    MEDIUM = "medium"
    HIGH = "high"
    BROKEN = "broken"

@dataclass
class CryptographicAsset:
    algorithm: str
    crypto_type: CryptoType
    current_key_size: int
    quantum_threat: QuantumThreatLevel
    grover_impact: str  # Effect of Grover's algorithm
    shor_impact: str    # Effect of Shor's algorithm
    pqc_replacement: str

class PostQuantumMigration:
    def __init__(self):
        self.assets: List[CryptographicAsset] = []
        self.migration_status: Dict[str, str] = {}

    def add_asset(self, asset: CryptographicAsset):
        self.assets.append(asset)

    def vulnerability_assessment(self) -> Dict[str, List[str]]:
        """Assess vulnerabilities by threat level."""
        assessment = {}
        for level in QuantumThreatLevel:
            vulnerable = [a.algorithm for a in self.assets if a.quantum_threat == level]
            if vulnerable:
                assessment[level.value] = vulnerable
        return assessment

    def migration_priority(self) -> List[Dict]:
        """Prioritize migration based on threat and usage."""
        priorities = []
        threat_scores = {
            QuantumThreatLevel.BROKEN: 5,
            QuantumThreatLevel.HIGH: 4,
            QuantumThreatLevel.MEDIUM: 3,
            QuantumThreatLevel.LOW: 2,
            QuantumThreatLevel.NONE: 1

            score = threat_scores[asset.quantum_threat]
            priorities.append({
                'algorithm': asset.algorithm,
                'type': asset.crypto_type.value,
                'priority_score': score,
                'replacement': asset.pqc_replacement,
                'action': 'urgent' if score >= 4 else 'planned' if score >= 3 else 'monitor'

        return sorted(priorities, key=lambda x: x['priority_score'], reverse=True)

from dataclasses import dataclass

class StrategicPriority(Enum):

    expected_benefits: List[str]

class QuantumStrategyPlanner:
    def __init__(self, organization_name: str):
        self.organization = organization_name
        self.initiatives: List[StrategicInitiative] = []

    def add_initiative(self, initiative: StrategicInitiative):
        self.initiatives.append(initiative)

    def build_roadmap(self) -> Dict[str, List[Dict]]:
        """Build strategic roadmap by time horizon."""
        roadmap = {}
        priority_names = {
            StrategicPriority.IMMEDIATE: '0-1 Years',
            StrategicPriority.SHORT_TERM: '1-3 Years',
            StrategicPriority.MEDIUM_TERM: '3-5 Years',
            StrategicPriority.LONG_TERM: '5+ Years'
        }

        for priority in StrategicPriority:
            initiatives = [i for i in self.initiatives if i.priority == priority]
            if initiatives:
                roadmap[priority_names[priority]] = [
                    {
                        'name': i.name,
                        'investment': i.estimated_investment,
                        'benefits': i.expected_benefits[:2]
                    }
                    for i in initiatives
                ]

        return roadmap

        """Analyze total investment requirements."""
        by_priority = {}
        for priority in StrategicPriority:
            initiatives = [i for i in self.initiatives if i.priority == priority]
            by_priority[priority.name] = sum(i.estimated_investment for i in initiatives)

            'total_investment': sum(i.estimated_investment for i in self.initiatives),
            'by_priority': by_priority,
            'annual_budget': sum(i.estimated_investment for i in self.initiatives) / 5

from dataclasses import dataclass
from typing import Dict, List
from enum import Enum

class TrendCategory(Enum):
    HARDWARE = "hardware"
    SOFTWARE = "software"
    APPLICATIONS = "applications"
    ECOSYSTEM = "ecosystem"
    REGULATION = "regulation"

@dataclass
class IndustryTrend:
    name: str
    category: TrendCategory
    description: str
    impact_level: int  # 1-5
    confidence: float  # 0-1
    timeframe: str

class TrendAnalyzer:
    def __init__(self):
        self.trends: List[IndustryTrend] = []

    def add_trend(self, trend: IndustryTrend):
        self.trends.append(trend)

    def high_impact_trends(self, min_impact: int = 4) -> List[Dict]:
        """Identify high-impact trends."""
        high_impact = [t for t in self.trends if t.impact_level >= min_impact]
        return [
            {
                'name': t.name,
                'category': t.category.value,
                'impact': t.impact_level,
                'confidence': t.confidence,
                'timeframe': t.timeframe
            }
            for t in sorted(high_impact, key=lambda x: x.impact_level, reverse=True)
        ]

    def by_category(self) -> Dict[str, List[str]]:
        """Group trends by category."""
        result = {}
        for cat in TrendCategory:
            trends = [t.name for t in self.trends if t.category == cat]
            if trends:
                result[cat.value] = trends
        return result

    def prepare_for_trends(self) -> List[Dict]:
        """Generate preparation recommendations."""
        recommendations = []
        for trend in self.trends:
            if trend.impact_level >= 4 and trend.confidence >= 0.6:
                recommendations.append({
                    'trend': trend.name,
                    'action': f"Prepare for {trend.name}",
                    'priority': 'high' if trend.impact_level == 5 else 'medium',
                    'timeframe': trend.timeframe
                })
        return recommendations

1. **Hardware evolution** - Path to fault-tolerant quantum computing
