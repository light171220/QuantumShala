```python
        """Aggregate metrics by application domain."""
        summary = {}
        for domain in ApplicationDomain:
            domain_studies = [s for s in self.studies if s.domain == domain]
            if domain_studies:
                summary[domain.value] = {
                    'count': len(domain_studies),
                    'total_investment': sum(s.investment_estimate for s in domain_studies),
                    'avg_team_size': sum(s.team_size for s in domain_studies) / len(domain_studies),
                    'institutions': [s.institution for s in domain_studies],
                    'most_advanced_phase': max(s.phase.value for s in domain_studies)
                }
        return summary

    def extract_common_lessons(self) -> Dict[str, int]:
        """Find most frequently mentioned lessons learned."""
        lesson_counts = {}
        for study in self.studies:
            for lesson in study.lessons_learned:
                # Simplify to key themes
                keywords = ['data', 'talent', 'integration', 'hardware', 'algorithms', 'partnership']
                for keyword in keywords:
                    if keyword.lower() in lesson.lower():
                        lesson_counts[keyword] = lesson_counts.get(keyword, 0) + 1
        return dict(sorted(lesson_counts.items(), key=lambda x: x[1], reverse=True))

    def success_factors(self) -> List[Dict]:
        """Identify factors correlated with project advancement."""
        advanced = [s for s in self.studies if s.phase.value in ['pilot', 'production', 'scaled']]
        early = [s for s in self.studies if s.phase.value in ['research', 'proof_of_concept']]

        if not advanced or not early:
            return []

        return [
            {
                'factor': 'Average Investment',
                'advanced': sum(s.investment_estimate for s in advanced) / len(advanced),
                'early': sum(s.investment_estimate for s in early) / len(early)
            },
            {
                'factor': 'Average Team Size',
                'advanced': sum(s.team_size for s in advanced) / len(advanced),
                'early': sum(s.team_size for s in early) / len(early)
            },
            {
                'factor': 'Published Research',
                'advanced': sum(s.published_papers for s in advanced) / len(advanced),
                'early': sum(s.published_papers for s in early) / len(early)
            }
        ]
```

## Portfolio Optimization Case Study

### JPMorgan Chase QOALA Framework

```python
from dataclasses import dataclass
from typing import Dict, List, Tuple
import numpy as np

@dataclass
class QOALAImplementation:
    """
    Simulation of JPMorgan's QOALA approach to portfolio optimization.
    QOALA: Quantum Optimization for Asset Allocation and Liabilities Assignment
    """
    num_assets: int
    discretization_levels: int = 4  # Bits per asset weight

    def encode_portfolio(self, continuous_weights: np.ndarray) -> np.ndarray:
        """Encode continuous weights to binary representation."""
        # Discretize weights to fit in qubits
        levels = 2 ** self.discretization_levels
        discrete = np.round(continuous_weights * (levels - 1)).astype(int)
        discrete = np.clip(discrete, 0, levels - 1)

        # Convert to binary
        binary = []
        for weight in discrete:
            bits = format(weight, f'0{self.discretization_levels}b')
            binary.extend([int(b) for b in bits])
        return np.array(binary)

    def decode_portfolio(self, binary_weights: np.ndarray) -> np.ndarray:
        """Decode binary representation back to weights."""
        n_bits = self.discretization_levels
        levels = 2 ** n_bits

        weights = []
        for i in range(self.num_assets):
            bits = binary_weights[i*n_bits:(i+1)*n_bits]
            decimal = sum(b * (2 ** (n_bits - 1 - j)) for j, b in enumerate(bits))
            weights.append(decimal / (levels - 1))

        weights = np.array(weights)
        return weights / weights.sum() if weights.sum() > 0 else weights

    def build_qubo_matrix(
        self,
        returns: np.ndarray,
        covariance: np.ndarray,
        risk_aversion: float
    ) -> np.ndarray:
        """Build QUBO matrix for portfolio optimization."""
        n_bits = self.discretization_levels
        n_qubits = self.num_assets * n_bits
        Q = np.zeros((n_qubits, n_qubits))

        levels = 2 ** n_bits

        # Linear terms (returns)
        for i in range(self.num_assets):
            for k in range(n_bits):
                qubit_idx = i * n_bits + k
                bit_value = 2 ** (n_bits - 1 - k) / (levels - 1)
                Q[qubit_idx, qubit_idx] -= returns[i] * bit_value

        # Quadratic terms (covariance)
        for i in range(self.num_assets):
            for j in range(self.num_assets):
                for ki in range(n_bits):
                    for kj in range(n_bits):
                        qi = i * n_bits + ki
                        qj = j * n_bits + kj
                        bit_i = 2 ** (n_bits - 1 - ki) / (levels - 1)
                        bit_j = 2 ** (n_bits - 1 - kj) / (levels - 1)
                        if qi <= qj:
                            Q[qi, qj] += risk_aversion * covariance[i, j] * bit_i * bit_j

        return Q

    def simulate_qaoa_optimization(
        self,
        Q: np.ndarray,
        num_iterations: int = 100
    ) -> Tuple[np.ndarray, float]:
        """Simulate QAOA optimization (simplified)."""
        n_qubits = Q.shape[0]
        best_solution = None
        best_energy = float('inf')

        for _ in range(num_iterations):
            # Random sampling (simulating quantum measurement)
            solution = np.random.randint(0, 2, n_qubits)

            # Calculate energy
            energy = solution @ Q @ solution

            if energy < best_energy:
                best_energy = energy
                best_solution = solution

        return best_solution, best_energy

class QOALABenchmark:
    def __init__(self, qoala: QOALAImplementation):
        self.qoala = qoala

    def benchmark_vs_classical(
        self,
        returns: np.ndarray,
        covariance: np.ndarray,
        risk_aversion: float = 1.0
    ) -> Dict:
        """Compare QOALA approach with classical optimization."""
        # QOALA approach
        Q = self.qoala.build_qubo_matrix(returns, covariance, risk_aversion)
        binary_solution, quantum_energy = self.qoala.simulate_qaoa_optimization(Q)
        quantum_weights = self.qoala.decode_portfolio(binary_solution)

        # Classical approach (simple mean-variance)
        inv_cov = np.linalg.pinv(covariance)
        classical_weights = inv_cov @ returns
        classical_weights = np.maximum(classical_weights, 0)
        classical_weights = classical_weights / classical_weights.sum()

        # Compare results
        quantum_return = np.dot(returns, quantum_weights)
        quantum_risk = np.sqrt(np.dot(quantum_weights, covariance @ quantum_weights))

        classical_return = np.dot(returns, classical_weights)
        classical_risk = np.sqrt(np.dot(classical_weights, covariance @ classical_weights))

        return {
            'quantum': {
                'weights': quantum_weights.tolist(),
                'return': quantum_return,
                'risk': quantum_risk,
                'sharpe': quantum_return / quantum_risk if quantum_risk > 0 else 0
            },
            'classical': {
                'weights': classical_weights.tolist(),
                'return': classical_return,
                'risk': classical_risk,
                'sharpe': classical_return / classical_risk if classical_risk > 0 else 0
            },
            'comparison': {
                'return_difference': quantum_return - classical_return,
                'risk_difference': quantum_risk - classical_risk,
                'discretization_error': np.linalg.norm(quantum_weights - classical_weights)
            }
        }

# Example usage
qoala = QOALAImplementation(num_assets=5, discretization_levels=3)
benchmark = QOALABenchmark(qoala)

np.random.seed(42)
returns = np.array([0.10, 0.08, 0.12, 0.06, 0.09])
covariance = np.array([
    [0.04, 0.01, 0.02, 0.01, 0.01],
    [0.01, 0.03, 0.01, 0.02, 0.01],
    [0.02, 0.01, 0.05, 0.01, 0.02],
    [0.01, 0.02, 0.01, 0.02, 0.01],
    [0.01, 0.01, 0.02, 0.01, 0.03]
])

results = benchmark.benchmark_vs_classical(returns, covariance)
print("Quantum Sharpe:", results['quantum']['sharpe'])
print("Classical Sharpe:", results['classical']['sharpe'])
```

## Derivatives Pricing Case Study

### Goldman Sachs Quantum Monte Carlo

```python
from dataclasses import dataclass
from typing import Dict, List, Callable
import numpy as np

@dataclass
class QuantumMonteCarloCaseStudy:
    """
    Implementation based on Goldman Sachs' quantum Monte Carlo research.
    Demonstrates amplitude estimation for option pricing.
    """
    num_qubits: int
    num_shots: int

    def classical_monte_carlo(
        self,
        spot: float,
        strike: float,
        rate: float,
        vol: float,
        T: float,
        num_paths: int
    ) -> Dict:
        """Classical Monte Carlo for European call option."""
        # Generate paths
        dt = T
        Z = np.random.standard_normal(num_paths)
        ST = spot * np.exp((rate - 0.5*vol**2)*dt + vol*np.sqrt(dt)*Z)

        # Calculate payoffs
        payoffs = np.maximum(ST - strike, 0)
        price = np.exp(-rate * T) * np.mean(payoffs)
        std_error = np.exp(-rate * T) * np.std(payoffs) / np.sqrt(num_paths)

        return {
            'price': price,
            'std_error': std_error,
            'num_paths': num_paths,
            'method': 'classical_monte_carlo'
        }

    def simulated_quantum_monte_carlo(
        self,
        spot: float,
        strike: float,
        rate: float,
        vol: float,
        T: float
    ) -> Dict:
        """
        Simulated quantum amplitude estimation for option pricing.
        Demonstrates quadratic speedup in principle.
        """
        # Number of "quantum samples" with quadratic advantage
        # Classical: O(1/ε²) samples for error ε
        # Quantum: O(1/ε) amplitude estimation queries
        effective_samples = self.num_shots ** 2  # Quadratic advantage

        # Simulate with effective sample count
        dt = T
        Z = np.random.standard_normal(effective_samples)
        ST = spot * np.exp((rate - 0.5*vol**2)*dt + vol*np.sqrt(dt)*Z)

        payoffs = np.maximum(ST - strike, 0)
        price = np.exp(-rate * T) * np.mean(payoffs)

        # Error scales as O(1/num_shots) for quantum
        std_error = np.exp(-rate * T) * np.std(payoffs) / effective_samples

        return {
            'price': price,
            'std_error': std_error,
            'effective_samples': effective_samples,
            'quantum_shots': self.num_shots,
            'method': 'quantum_amplitude_estimation',
            'speedup_factor': self.num_shots  # Quadratic advantage
        }

    def compare_methods(
        self,
        spot: float,
        strike: float,
        rate: float,
        vol: float,
        T: float
    ) -> Dict:
        """Compare classical and quantum Monte Carlo."""
        classical = self.classical_monte_carlo(
            spot, strike, rate, vol, T, self.num_shots
        )
        quantum = self.simulated_quantum_monte_carlo(
            spot, strike, rate, vol, T
        )

        return {
            'classical': classical,
            'quantum': quantum,
            'price_difference': abs(classical['price'] - quantum['price']),
            'error_reduction': classical['std_error'] / quantum['std_error'] if quantum['std_error'] > 0 else float('inf'),
            'theoretical_speedup': 'O(1/ε) vs O(1/ε²)'
        }

class QuantumMonteCarloBenchmark:
    def __init__(self):
        self.results: List[Dict] = []

    def convergence_analysis(
        self,
        spot: float = 100,
        strike: float = 100,
        rate: float = 0.05,
        vol: float = 0.2,
        T: float = 1.0
    ) -> List[Dict]:
        """Analyze convergence for different shot counts."""
        shot_counts = [100, 500, 1000, 5000, 10000]
        results = []

        for shots in shot_counts:
            qmc = QuantumMonteCarloCaseStudy(num_qubits=10, num_shots=shots)
            comparison = qmc.compare_methods(spot, strike, rate, vol, T)

            results.append({
                'shots': shots,
                'classical_error': comparison['classical']['std_error'],
                'quantum_error': comparison['quantum']['std_error'],
                'error_ratio': comparison['error_reduction']
            })

        return results

# Example usage

    spot=100, strike=100, rate=0.05, vol=0.2, T=1.0

print(f"Error reduction factor: {comparison['error_reduction']:.2f}x")
```

```python
from dataclasses import dataclass
from typing import Dict, List, Tuple
import numpy as np

@dataclass
class CreditRiskCaseStudy:
    """
    Simulation based on BBVA's quantum computing credit risk research.
    Focus on quantum-enhanced credit portfolio analysis.
    """
    num_obligors: int
    correlation_model: str = "gaussian_copula"

    def simulate_classical_credit_var(
        self,
        pds: np.ndarray,  # Probability of default
        lgds: np.ndarray,  # Loss given default
        exposures: np.ndarray,
        correlation: float,
        num_scenarios: int
    ) -> Dict:
        """Classical credit VaR simulation."""
        # Gaussian copula model
        n = len(pds)
        losses = []

        for _ in range(num_scenarios):
            # Systematic factor
            Z = np.random.standard_normal()

            # Idiosyncratic factors
            epsilon = np.random.standard_normal(n)

            # Correlated default indicators
            latent = np.sqrt(correlation) * Z + np.sqrt(1 - correlation) * epsilon
            from scipy.stats import norm
            default_threshold = norm.ppf(pds)
            defaults = latent < default_threshold

            # Calculate portfolio loss
            scenario_loss = np.sum(defaults * lgds * exposures)
            losses.append(scenario_loss)

        losses = np.array(losses)

        return {
            'expected_loss': np.mean(losses),
            'var_95': np.percentile(losses, 95),
            'var_99': np.percentile(losses, 99),
            'es_99': np.mean(losses[losses >= np.percentile(losses, 99)]),
            'num_scenarios': num_scenarios,
            'method': 'classical_monte_carlo'
        }

    def simulate_quantum_credit_var(
        self,
        pds: np.ndarray,
        lgds: np.ndarray,
        exposures: np.ndarray,
        correlation: float,
        num_shots: int
    ) -> Dict:
        """
        Simulated quantum credit VaR using amplitude estimation.
        Demonstrates potential for tail risk measurement.
        """
        n = len(pds)

        # Quantum advantage: quadratic speedup for tail probability estimation
        effective_scenarios = num_shots ** 2

        losses = []
        from scipy.stats import norm

        for _ in range(min(effective_scenarios, 100000)):  # Cap for simulation
            Z = np.random.standard_normal()
            epsilon = np.random.standard_normal(n)
            latent = np.sqrt(correlation) * Z + np.sqrt(1 - correlation) * epsilon
            default_threshold = norm.ppf(pds)
            defaults = latent < default_threshold
            scenario_loss = np.sum(defaults * lgds * exposures)
            losses.append(scenario_loss)

        losses = np.array(losses)

        # Quantum provides better precision for tail estimates
        return {
            'expected_loss': np.mean(losses),
            'var_95': np.percentile(losses, 95),
            'var_99': np.percentile(losses, 99),
            'es_99': np.mean(losses[losses >= np.percentile(losses, 99)]),
            'quantum_shots': num_shots,
            'effective_scenarios': effective_scenarios,
            'method': 'quantum_amplitude_estimation',
            'tail_precision_improvement': np.sqrt(num_shots)  # Relative to classical
        }

    def compare_tail_risk_estimation(
        self,
        pds: np.ndarray,
        lgds: np.ndarray,
        exposures: np.ndarray,
        correlation: float
    ) -> Dict:
        """Compare classical and quantum tail risk estimation."""
        classical = self.simulate_classical_credit_var(
            pds, lgds, exposures, correlation, num_scenarios=10000
        )

        quantum = self.simulate_quantum_credit_var(
            pds, lgds, exposures, correlation, num_shots=100
        )

        return {
            'classical': classical,
            'quantum': quantum,
            'var_99_difference': abs(classical['var_99'] - quantum['var_99']),
            'computation_ratio': quantum['effective_scenarios'] / classical['num_scenarios']
        }

# Example usage
np.random.seed(42)

# Generate synthetic portfolio
```

```python
class SuccessEvaluator:
    def __init__(self):
        self.projects: List[ProjectMetrics] = []

    def add_project(self, project: ProjectMetrics):
        self.projects.append(project)

    def calculate_composite_score(self, project: ProjectMetrics) -> float:
        """Calculate weighted success score."""
        weights = {
            'phase': 0.25,
            'technical': 0.25,
            'talent': 0.20,
            'partnership': 0.15,
            'research': 0.15
        }

        score = (
            weights['phase'] * project.phase_progression +
            weights['technical'] * project.technical_achievement +
            weights['talent'] * project.talent_development +
            weights['partnership'] * project.partnership_strength +
            weights['research'] * project.publication_impact
        )
        return score

    def benchmark_analysis(self) -> Dict:
        """Analyze all projects against benchmarks."""
        if not self.projects:
            return {}

        scores = [self.calculate_composite_score(p) for p in self.projects]

        return {
            'average_score': np.mean(scores),
            'top_performer': self.projects[np.argmax(scores)].project_name,
            'score_distribution': {
                'min': min(scores),
                'max': max(scores),
                'median': np.median(scores)
            },
            'recommendations': self._generate_recommendations()
        }

    def _generate_recommendations(self) -> List[str]:
        """Generate recommendations based on analysis."""
        recommendations = []

        avg_talent = np.mean([p.talent_development for p in self.projects])
        if avg_talent < 0.5:
            recommendations.append("Increase investment in quantum talent development")

        avg_partnership = np.mean([p.partnership_strength for p in self.projects])
        if avg_partnership < 0.6:
            recommendations.append("Strengthen hardware provider partnerships")

        return recommendations

# Example evaluation
evaluator = SuccessEvaluator()

projects = [

    ProjectMetrics("Credit VaR", 0.4, 0.5, 0.6, 0.7, 0.5, 2, 7.0),
]

for p in projects:

print("Benchmark Analysis:", evaluator.benchmark_analysis())
```
