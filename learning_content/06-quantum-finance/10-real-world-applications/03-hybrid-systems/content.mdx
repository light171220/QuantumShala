# Hybrid Quantum-Classical Systems

Hybrid systems combine quantum and classical computing to leverage the strengths of each. This approach is essential for practical quantum finance applications today and in the near term.

## Hybrid Architecture Design

### Core Hybrid Framework

```python
from abc import ABC, abstractmethod
from dataclasses import dataclass
from typing import Dict, Any, Callable, Optional, List
import numpy as np
from enum import Enum

class ExecutionMode(Enum):
    QUANTUM = "quantum"
    CLASSICAL = "classical"
    HYBRID = "hybrid"

@dataclass
class HybridConfig:
    max_quantum_shots: int = 10000
    classical_timeout_seconds: float = 60.0
    quantum_timeout_seconds: float = 300.0
    fallback_enabled: bool = True
    min_quantum_quality: float = 0.8

class HybridExecutor(ABC):
    def __init__(self, config: HybridConfig):
        self.config = config
        self.execution_log: List[Dict] = []

    @abstractmethod
    def quantum_solve(self, problem: Dict) -> Dict:
        """Quantum solution method."""
        pass

    @abstractmethod
    def classical_solve(self, problem: Dict) -> Dict:
        """Classical fallback method."""
        pass

    def execute(self, problem: Dict, mode: ExecutionMode = ExecutionMode.HYBRID) -> Dict:
        """Execute with automatic fallback."""
        import time

        if mode == ExecutionMode.CLASSICAL:
            return self._execute_classical(problem)

        if mode == ExecutionMode.QUANTUM:
            return self._execute_quantum(problem)

        # Hybrid mode: try quantum, fallback to classical
        start = time.time()
        try:
            result = self._execute_quantum(problem)
            if result.get('quality', 1.0) < self.config.min_quantum_quality:
                if self.config.fallback_enabled:
                    classical_result = self._execute_classical(problem)
                    result['classical_fallback'] = classical_result
                    result['used_fallback'] = True
            return result
        except Exception as e:
            if self.config.fallback_enabled:
                result = self._execute_classical(problem)
                result['quantum_error'] = str(e)
                result['used_fallback'] = True
                return result
            raise

    def _execute_quantum(self, problem: Dict) -> Dict:
        import time
        start = time.time()
        result = self.quantum_solve(problem)
        result['execution_time'] = time.time() - start
        result['mode'] = 'quantum'
        self.execution_log.append(result)
        return result

    def _execute_classical(self, problem: Dict) -> Dict:
        import time
        start = time.time()
        result = self.classical_solve(problem)
        result['execution_time'] = time.time() - start
        result['mode'] = 'classical'
        self.execution_log.append(result)
        return result

class PortfolioHybridExecutor(HybridExecutor):
    def __init__(self, config: HybridConfig, num_assets: int):
        super().__init__(config)
        self.num_assets = num_assets

    def quantum_solve(self, problem: Dict) -> Dict:
        """Simulated quantum portfolio optimization."""
        returns = np.array(problem['returns'])
        cov = np.array(problem['covariance'])
        risk_aversion = problem.get('risk_aversion', 1.0)

        # Simulate QAOA-like optimization
        n_iterations = 100
        best_weights = np.random.dirichlet(np.ones(self.num_assets))
        best_objective = float('inf')

        for _ in range(n_iterations):
            # Simulate quantum sampling
            weights = np.random.dirichlet(np.ones(self.num_assets))
            # Add some noise to simulate quantum behavior
            weights += np.random.normal(0, 0.01, self.num_assets)
            weights = np.maximum(weights, 0)
            weights /= weights.sum()

            portfolio_return = np.dot(returns, weights)
            portfolio_risk = np.sqrt(np.dot(weights, np.dot(cov, weights)))
            objective = -portfolio_return + risk_aversion * portfolio_risk

            if objective < best_objective:
                best_objective = objective
                best_weights = weights

        return {
            'weights': best_weights.tolist(),
            'expected_return': float(np.dot(returns, best_weights)),
            'risk': float(np.sqrt(np.dot(best_weights, np.dot(cov, best_weights)))),
            'quality': 0.85 + np.random.uniform(-0.1, 0.1),
            'iterations': n_iterations
        }

    def classical_solve(self, problem: Dict) -> Dict:
        """Mean-variance optimization using quadratic programming."""
        returns = np.array(problem['returns'])
        cov = np.array(problem['covariance'])
        risk_aversion = problem.get('risk_aversion', 1.0)

        # Simple analytical solution for unconstrained case
        inv_cov = np.linalg.pinv(cov)
        weights = inv_cov @ returns
        weights = np.maximum(weights, 0)
        weights /= weights.sum()

        return {
            'weights': weights.tolist(),
            'expected_return': float(np.dot(returns, weights)),
            'risk': float(np.sqrt(np.dot(weights, np.dot(cov, weights)))),
            'quality': 1.0,
            'method': 'mean_variance'
        }

# Example usage

executor = PortfolioHybridExecutor(config, num_assets=5)

problem = {
    'returns': [0.10, 0.08, 0.12, 0.06, 0.09],
    'covariance': np.eye(5).tolist(),
    'risk_aversion': 1.0
}

result = executor.execute(problem, ExecutionMode.HYBRID)
print(f"Weights: {result['weights']}")
print(f"Mode: {result['mode']}")
```

### VQE for Portfolio Optimization

```python
from dataclasses import dataclass
from typing import Callable, Tuple, List
import numpy as np

@dataclass
class VQEConfig:
    num_qubits: int
    num_layers: int
    max_iterations: int
    learning_rate: float
    shots: int

class VariationalQuantumEigensolver:
    def __init__(self, config: VQEConfig):
        self.config = config
        self.convergence_history: List[float] = []

    def create_ansatz_params(self) -> np.ndarray:
        """Initialize variational parameters."""
        num_params = self.config.num_qubits * self.config.num_layers * 3
        return np.random.uniform(0, 2 * np.pi, num_params)

    def simulate_circuit(self, params: np.ndarray, hamiltonian: np.ndarray) -> float:
        """Simulate parameterized circuit and measure energy."""
        n = self.config.num_qubits

        # Simplified simulation of variational circuit
        # In practice, this would run on quantum hardware
        state = np.zeros(2**n)
        state[0] = 1  # |00...0>

        # Apply parameterized gates (simplified)
        param_idx = 0
        for layer in range(self.config.num_layers):
            for qubit in range(n):
                # RY rotation
                angle = params[param_idx]
                param_idx += 1
                # Simplified rotation effect
                state = self._apply_ry(state, qubit, angle)

        # Measure energy expectation
        probs = np.abs(state) ** 2
        energies = np.diag(hamiltonian)[:len(probs)]
        energy = np.dot(probs, energies)

        # Add shot noise
        energy += np.random.normal(0, 0.01 / np.sqrt(self.config.shots))

        return energy

    def _apply_ry(self, state: np.ndarray, qubit: int, angle: float) -> np.ndarray:
        """Simplified RY gate application."""
        n = int(np.log2(len(state)))
        c, s = np.cos(angle/2), np.sin(angle/2)
        new_state = state.copy()
        for i in range(len(state)):
            if (i >> qubit) & 1 == 0:
                j = i | (1 << qubit)
                new_state[i] = c * state[i] - s * state[j]
                new_state[j] = s * state[i] + c * state[j]
        return new_state

    def optimize(
        self,
        hamiltonian: np.ndarray,
        initial_params: Optional[np.ndarray] = None
    ) -> Tuple[np.ndarray, float]:
        """Run VQE optimization loop."""
        params = initial_params if initial_params is not None else self.create_ansatz_params()
        best_params = params.copy()
        best_energy = float('inf')

        for iteration in range(self.config.max_iterations):
            # Compute gradient using parameter shift
            gradient = self._compute_gradient(params, hamiltonian)

            # Update parameters
            params = params - self.config.learning_rate * gradient

            # Evaluate energy
            energy = self.simulate_circuit(params, hamiltonian)
            self.convergence_history.append(energy)

            if energy < best_energy:
                best_energy = energy
                best_params = params.copy()

        return best_params, best_energy

    def _compute_gradient(self, params: np.ndarray, hamiltonian: np.ndarray) -> np.ndarray:
        """Parameter shift rule for gradient estimation."""
        gradient = np.zeros_like(params)
        shift = np.pi / 2

        for i in range(len(params)):
            params_plus = params.copy()
            params_plus[i] += shift
            params_minus = params.copy()
            params_minus[i] -= shift

            energy_plus = self.simulate_circuit(params_plus, hamiltonian)
            energy_minus = self.simulate_circuit(params_minus, hamiltonian)

            gradient[i] = (energy_plus - energy_minus) / 2

        return gradient

class VQEPortfolioOptimizer:
    def __init__(self, num_assets: int, vqe_config: VQEConfig):
        self.num_assets = num_assets
        self.vqe = VariationalQuantumEigensolver(vqe_config)

    def build_portfolio_hamiltonian(
        self,
        returns: np.ndarray,
        covariance: np.ndarray,
        risk_aversion: float
    ) -> np.ndarray:
        """Convert portfolio problem to Hamiltonian."""
        n = self.num_assets

        # Simplified QUBO to Ising conversion
        # In practice, this requires proper encoding
        dim = 2 ** n
        H = np.zeros((dim, dim))

        for i in range(min(dim, len(returns))):
            # Diagonal elements represent portfolio configurations
            binary = format(i, f'0{n}b')
            weights = np.array([int(b) for b in binary]) / n

            portfolio_return = np.dot(returns, weights)
            portfolio_risk = np.dot(weights, np.dot(covariance, weights))
            H[i, i] = -portfolio_return + risk_aversion * portfolio_risk

        return H

    def optimize(
        self,
        returns: np.ndarray,
        covariance: np.ndarray,
        risk_aversion: float = 1.0
    ) -> Dict:
        """Run VQE portfolio optimization."""
        hamiltonian = self.build_portfolio_hamiltonian(returns, covariance, risk_aversion)
        optimal_params, min_energy = self.vqe.optimize(hamiltonian)

        # Extract portfolio weights from optimal state
        final_probs = self._get_measurement_probs(optimal_params, hamiltonian)
        best_state = np.argmax(final_probs)
        binary = format(best_state, f'0{self.num_assets}b')
        weights = np.array([int(b) for b in binary])
        weights = weights / weights.sum() if weights.sum() > 0 else np.ones(self.num_assets) / self.num_assets

        return {
            'weights': weights.tolist(),
            'energy': min_energy,
            'convergence': self.vqe.convergence_history,
            'iterations': len(self.vqe.convergence_history)
        }

    def _get_measurement_probs(self, params: np.ndarray, hamiltonian: np.ndarray) -> np.ndarray:
        """Get measurement probability distribution."""
        # Simplified - would use actual circuit measurement
        n = self.num_assets
        probs = np.zeros(2**n)
        energies = np.diag(hamiltonian)
        probs = np.exp(-energies / 0.1)  # Boltzmann-like distribution
        return probs / probs.sum()
```

## QAOA for Combinatorial Finance Problems

### QAOA Implementation

```python
from dataclasses import dataclass
from typing import List, Tuple
import numpy as np

@dataclass
class QAOAConfig:
    num_layers: int  # p parameter
    max_iterations: int
    shots: int

class QAOAOptimizer:
    def __init__(self, config: QAOAConfig):
        self.config = config
        self.energy_history: List[float] = []

    def solve_maxcut_portfolio(
        self,
        adjacency: np.ndarray,
        asset_values: np.ndarray
    ) -> Tuple[np.ndarray, float]:
        """
        Solve asset selection as MaxCut-like problem.
        adjacency: Correlation/covariance between assets
        asset_values: Expected returns
        """
        n = len(asset_values)

        # Initialize QAOA parameters
        gamma = np.random.uniform(0, np.pi, self.config.num_layers)
        beta = np.random.uniform(0, np.pi, self.config.num_layers)

        best_solution = None
        best_objective = float('-inf')

        for iteration in range(self.config.max_iterations):
            # Simulate QAOA circuit
            probs = self._simulate_qaoa(gamma, beta, adjacency, asset_values)

            # Sample solutions
            for _ in range(self.config.shots):
                solution = self._sample_solution(probs)
                objective = self._evaluate_objective(solution, adjacency, asset_values)

                if objective > best_objective:
                    best_objective = objective
                    best_solution = solution

            # Update parameters (simplified gradient descent)
            gamma, beta = self._update_params(gamma, beta, adjacency, asset_values)
            self.energy_history.append(-best_objective)

        return best_solution, best_objective

    def _simulate_qaoa(
        self,
        gamma: np.ndarray,
        beta: np.ndarray,
        adjacency: np.ndarray,
        values: np.ndarray
    ) -> np.ndarray:
        """Simulate QAOA circuit output probabilities."""
        n = len(values)
        dim = 2 ** n

        # Start with uniform superposition
        probs = np.ones(dim) / dim

        # Apply QAOA layers (simplified classical simulation)
        for p in range(self.config.num_layers):
            # Cost layer effect (gamma)
            for i in range(dim):
                solution = np.array([(i >> j) & 1 for j in range(n)])
                cost = self._evaluate_objective(solution, adjacency, values)
                probs[i] *= np.exp(-1j * gamma[p] * cost).real ** 2

            # Mixer layer effect (beta)
            # Simplified: slight redistribution
            mixing = 0.1 * np.sin(beta[p])
            probs = (1 - mixing) * probs + mixing / dim

        probs = np.abs(probs)
        return probs / probs.sum()

    def _sample_solution(self, probs: np.ndarray) -> np.ndarray:
        """Sample a solution from probability distribution."""
        n = int(np.log2(len(probs)))
        idx = np.random.choice(len(probs), p=probs)
        return np.array([(idx >> j) & 1 for j in range(n)])

    def _evaluate_objective(
        self,
        solution: np.ndarray,
        adjacency: np.ndarray,
        values: np.ndarray
    ) -> float:
        """Evaluate portfolio objective."""
        # Return contribution
        return_val = np.dot(solution, values)

        # Diversification penalty (prefer diverse portfolios)
        n = len(solution)
        diversity = 0
        for i in range(n):
            for j in range(i+1, n):
                if solution[i] != solution[j]:
                    diversity += adjacency[i, j]

        return return_val + 0.1 * diversity

    def _update_params(
        self,
        gamma: np.ndarray,
        beta: np.ndarray,
        adjacency: np.ndarray,
        values: np.ndarray
    ) -> Tuple[np.ndarray, np.ndarray]:
        """Simple parameter update using finite differences."""
        eps = 0.1
        lr = 0.05

        for p in range(self.config.num_layers):
            # Update gamma
            gamma_plus = gamma.copy()
            gamma_plus[p] += eps
            gamma_minus = gamma.copy()
            gamma_minus[p] -= eps

            grad_gamma = (
                self._estimate_cost(gamma_plus, beta, adjacency, values) -
                self._estimate_cost(gamma_minus, beta, adjacency, values)
            ) / (2 * eps)
            gamma[p] -= lr * grad_gamma

            # Update beta
            beta_plus = beta.copy()
            beta_plus[p] += eps
            beta_minus = beta.copy()
            beta_minus[p] -= eps

            grad_beta = (
                self._estimate_cost(gamma, beta_plus, adjacency, values) -
                self._estimate_cost(gamma, beta_minus, adjacency, values)
            ) / (2 * eps)
            beta[p] -= lr * grad_beta

        return gamma, beta

    def _estimate_cost(
        self,
        gamma: np.ndarray,
        beta: np.ndarray,
        adjacency: np.ndarray,
        values: np.ndarray
    ) -> float:
        """Estimate expected cost for given parameters."""
        probs = self._simulate_qaoa(gamma, beta, adjacency, values)
        n = int(np.log2(len(probs)))

        expected_cost = 0
        for i in range(len(probs)):
            solution = np.array([(i >> j) & 1 for j in range(n)])
            cost = -self._evaluate_objective(solution, adjacency, values)
            expected_cost += probs[i] * cost

        return expected_cost

# Example usage
qaoa_config = QAOAConfig(num_layers=3, max_iterations=20, shots=100)
qaoa = QAOAOptimizer(qaoa_config)

# Asset correlation matrix

    [0, 0.3, 0.1, 0.2],
    [0.3, 0, 0.4, 0.1],
    [0.1, 0.4, 0, 0.3],
    [0.2, 0.1, 0.3, 0]
])

values = np.array([0.12, 0.08, 0.15, 0.10])

print(f"Selected assets: {solution}")
print(f"Objective: {objective:.4f}")
```

```python
from dataclasses import dataclass
from typing import Dict, List, Any
import numpy as np
import time

@dataclass
class CommunicationMetrics:
    data_size_bytes: int
    latency_ms: float
    compression_ratio: float
    encoding_time_ms: float
    decoding_time_ms: float

class QuantumDataEncoder:
    def __init__(self, num_qubits: int):
        self.num_qubits = num_qubits

    def encode_portfolio_data(self, weights: np.ndarray) -> Dict[str, Any]:
        """Encode classical portfolio data for quantum circuit."""
        start = time.time()

        # Normalize weights to quantum amplitudes
        norm = np.linalg.norm(weights)
        amplitudes = weights / norm if norm > 0 else weights

        # Convert to angles for rotation gates
        angles = 2 * np.arcsin(np.sqrt(np.abs(amplitudes)))

        encoding_time = (time.time() - start) * 1000

        return {
            'angles': angles.tolist(),
            'normalization': norm,
            'num_qubits_required': len(weights),
            'encoding_time_ms': encoding_time
        }

    def decode_quantum_result(self, measurement_counts: Dict[str, int]) -> Dict[str, Any]:
        """Decode quantum measurement results to classical format."""
        start = time.time()

        total_shots = sum(measurement_counts.values())
        probabilities = {k: v/total_shots for k, v in measurement_counts.items()}

        # Extract most likely state
        best_state = max(probabilities, key=probabilities.get)

        # Convert to portfolio weights
        n = len(best_state)
        weights = np.zeros(n)
        for i, bit in enumerate(best_state):
            weights[i] = int(bit)

        # Normalize
        if weights.sum() > 0:
            weights = weights / weights.sum()
        else:
            weights = np.ones(n) / n

        decoding_time = (time.time() - start) * 1000

        return {
            'weights': weights.tolist(),
            'confidence': probabilities[best_state],
            'all_probabilities': probabilities,
            'decoding_time_ms': decoding_time
        }

class HybridCommunicationOptimizer:
    def __init__(self):
        self.metrics_history: List[CommunicationMetrics] = []

    def batch_problems(
        self,
        problems: List[Dict],
        batch_size: int
    ) -> List[List[Dict]]:
        """Batch problems to reduce communication overhead."""
        return [
            problems[i:i+batch_size]
            for i in range(0, len(problems), batch_size)
        ]

    def compress_data(self, data: Dict) -> Tuple[bytes, float]:
        """Compress data for transmission."""
        import json
        import zlib

        json_str = json.dumps(data)
        original_size = len(json_str.encode())
        compressed = zlib.compress(json_str.encode())
        compressed_size = len(compressed)

        ratio = compressed_size / original_size if original_size > 0 else 1.0

        return compressed, ratio

    def optimize_circuit_parameters(
        self,
        problem_size: int,
        available_qubits: int
    ) -> Dict[str, int]:
        """Recommend circuit parameters to minimize communication."""
        # Minimize shots while maintaining accuracy
        base_shots = 1000
        scaling = min(1, available_qubits / problem_size)
        recommended_shots = int(base_shots / (scaling ** 2))

        # Batch multiple measurements
        measurements_per_batch = min(100, problem_size)

        return {
            'shots': recommended_shots,
            'measurements_per_batch': measurements_per_batch,
            'estimated_round_trips': problem_size // measurements_per_batch + 1,
            'estimated_data_mb': recommended_shots * problem_size * 8 / 1e6
        }
```

## Production Hybrid Workflow

```python
from dataclasses import dataclass
from typing import Dict, List, Optional, Callable
from enum import Enum
import time
import numpy as np

class WorkflowStatus(Enum):
    PENDING = "pending"
    RUNNING = "running"
    COMPLETED = "completed"
    FAILED = "failed"
    FALLBACK = "fallback"

@dataclass
class WorkflowStep:
    name: str
    executor: Callable
    timeout_seconds: float
    fallback: Optional[Callable] = None
    critical: bool = True

class HybridWorkflow:
    def __init__(self, name: str):
        self.name = name
        self.steps: List[WorkflowStep] = []
        self.results: Dict[str, Dict] = {}
        self.status: WorkflowStatus = WorkflowStatus.PENDING

    def add_step(self, step: WorkflowStep):
        self.steps.append(step)

    def execute(self, initial_data: Dict) -> Dict:
        """Execute the full workflow with fallback handling."""
        self.status = WorkflowStatus.RUNNING
        current_data = initial_data.copy()

        for step in self.steps:
            try:
                start = time.time()
                result = step.executor(current_data)

                if time.time() - start > step.timeout_seconds:
                    raise TimeoutError(f"Step {step.name} timed out")

                self.results[step.name] = {
                    'status': 'success',
                    'result': result,
                    'time': time.time() - start
                }
                current_data.update(result)

            except Exception as e:
                if step.fallback:
                    try:
                        result = step.fallback(current_data)
                        self.results[step.name] = {
                            'status': 'fallback',
                            'result': result,
                            'error': str(e)
                        }
                        current_data.update(result)
                    except Exception as fallback_error:
                        if step.critical:
                            self.status = WorkflowStatus.FAILED
                            return {'error': str(fallback_error), 'step': step.name}
                elif step.critical:
                    self.status = WorkflowStatus.FAILED
                    return {'error': str(e), 'step': step.name}

        self.status = WorkflowStatus.COMPLETED
        return current_data

class ProductionPortfolioWorkflow:
    def __init__(self):
        self.workflow = HybridWorkflow("portfolio_optimization")
        self._setup_steps()

    def _setup_steps(self):
        # Step 1: Data preprocessing
        self.workflow.add_step(WorkflowStep(
            name="preprocess",
            executor=self._preprocess,
            timeout_seconds=10.0,
            critical=True
        ))

        # Step 2: Quantum optimization
        self.workflow.add_step(WorkflowStep(
            name="quantum_optimize",
            executor=self._quantum_optimize,
            timeout_seconds=300.0,
            fallback=self._classical_optimize,
            critical=True
        ))

        # Step 3: Post-processing
        self.workflow.add_step(WorkflowStep(
            name="postprocess",
            executor=self._postprocess,
            timeout_seconds=5.0,
            critical=True
        ))

        # Step 4: Validation
        self.workflow.add_step(WorkflowStep(
            name="validate",
            executor=self._validate,
            timeout_seconds=5.0,
            critical=False
        ))

    def _preprocess(self, data: Dict) -> Dict:
        """Prepare data for optimization."""
        returns = np.array(data['returns'])
        covariance = np.array(data['covariance'])

        # Normalize
        returns_normalized = (returns - returns.mean()) / returns.std()

        return {
            'returns_normalized': returns_normalized.tolist(),
            'normalization_params': {
                'mean': float(returns.mean()),
                'std': float(returns.std())
            }
        }

    def _quantum_optimize(self, data: Dict) -> Dict:
        """Quantum portfolio optimization."""
        returns = np.array(data['returns_normalized'])

        # Simulated quantum optimization
        n = len(returns)
        weights = np.random.dirichlet(np.ones(n))

        # Simulate some optimization iterations
        for _ in range(10):
            gradient = returns - 2 * weights
            weights += 0.1 * gradient
            weights = np.maximum(weights, 0)
            weights /= weights.sum()

        return {
            'raw_weights': weights.tolist(),
            'method': 'quantum'
        }

    def _classical_optimize(self, data: Dict) -> Dict:
        """Classical fallback optimization."""
        returns = np.array(data['returns_normalized'])
        n = len(returns)

        # Simple mean-variance
        weights = np.ones(n) / n

        return {
            'raw_weights': weights.tolist(),
            'method': 'classical_fallback'
        }

    def _postprocess(self, data: Dict) -> Dict:
        """Apply constraints and denormalize."""
        weights = np.array(data['raw_weights'])

        # Apply minimum weight constraint
        min_weight = 0.05
        weights = np.maximum(weights, min_weight)
        weights /= weights.sum()

        return {
            'final_weights': weights.tolist()
        }

    def _validate(self, data: Dict) -> Dict:
        """Validate the solution."""
        weights = np.array(data['final_weights'])

        checks = {
            'sum_to_one': abs(weights.sum() - 1.0) < 0.001,
            'all_positive': all(w >= 0 for w in weights),
            'within_bounds': all(w <= 0.5 for w in weights)
        }

        return {
            'validation': checks,
            'is_valid': all(checks.values())
        }

    def run(self, returns: List[float], covariance: List[List[float]]) -> Dict:
        """Run the complete workflow."""
        initial_data = {
            'returns': returns,
            'covariance': covariance
        }
        return self.workflow.execute(initial_data)
```

Hybrid quantum-classical systems are essential for practical quantum finance:
