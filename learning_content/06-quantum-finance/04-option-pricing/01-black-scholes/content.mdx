## 1. Introduction to Option Pricing

Options are financial derivatives that give the holder the right, but not the obligation, to buy or sell an underlying asset.

<Definition term="European Option">
An option that can only be exercised at expiration. A call option gives the right to buy; a put option gives the right to sell at the strike price K.
</Definition>

### Option Payoffs at Expiration

```python
import numpy as np
import matplotlib.pyplot as plt
from scipy.stats import norm

def call_payoff(S, K):
    """Call option payoff at expiration."""
    return np.maximum(S - K, 0)

def put_payoff(S, K):
    """Put option payoff at expiration."""
    return np.maximum(K - S, 0)

# Visualize payoffs
K = 100  # Strike price
S = np.linspace(50, 150, 100)

call_pf = call_payoff(S, K)
put_pf = put_payoff(S, K)

print("Option Payoffs at Expiration:")
print(f"  If S = 120, Call payoff = {call_payoff(120, K):.0f}")
print(f"  If S = 80, Call payoff = {call_payoff(80, K):.0f}")
print(f"  If S = 80, Put payoff = {put_payoff(80, K):.0f}")
print(f"  If S = 120, Put payoff = {put_payoff(120, K):.0f}")
```

## 2. Black-Scholes Assumptions

The Black-Scholes model relies on several key assumptions:

<Warning>
These assumptions are never perfectly true in real markets, but the model remains useful when deviations are small or accounted for.
</Warning>

1. **Log-normal returns**: Stock prices follow geometric Brownian motion
2. **Constant volatility**: σ is constant over the option's life
3. **No dividends**: The underlying pays no dividends
4. **No arbitrage**: Risk-free profit opportunities don't exist
5. **Continuous trading**: Markets are always open
6. **No transaction costs**: Trading is frictionless
7. **Constant risk-free rate**: r is constant and known

## 3. The Black-Scholes Formula

<Formula>
Black-Scholes European Call Price:
$$C = S_0 N(d_1) - K e^{-rT} N(d_2)$$

where:
$$d_1 = \frac{\ln(S_0/K) + (r + \sigma^2/2)T}{\sigma\sqrt{T}}$$
$$d_2 = d_1 - \sigma\sqrt{T}$$

and $N(\cdot)$ is the cumulative standard normal distribution.
</Formula>

```python
class BlackScholes:
    """
    Black-Scholes option pricing model.
    """

    def __init__(self, S, K, T, r, sigma):
        """
        Args:
            S: Current stock price
            K: Strike price
            T: Time to expiration (years)
            r: Risk-free rate (annual)
            sigma: Volatility (annual)
        """
        self.S = S
        self.K = K
        self.T = T
        self.r = r
        self.sigma = sigma

        # Calculate d1 and d2
        self.d1 = (np.log(S/K) + (r + sigma**2/2)*T) / (sigma*np.sqrt(T))
        self.d2 = self.d1 - sigma*np.sqrt(T)

    def call_price(self):
        """Calculate European call option price."""
        return (self.S * norm.cdf(self.d1) -
                self.K * np.exp(-self.r * self.T) * norm.cdf(self.d2))

    def put_price(self):
        """Calculate European put option price."""
        return (self.K * np.exp(-self.r * self.T) * norm.cdf(-self.d2) -
                self.S * norm.cdf(-self.d1))

    def call_put_parity(self):
        """
        Verify put-call parity: C - P = S - K*e^(-rT)
        """
        C = self.call_price()
        P = self.put_price()
        forward_diff = self.S - self.K * np.exp(-self.r * self.T)
        return {
            'C - P': C - P,
            'S - K*e^(-rT)': forward_diff,
            'parity_holds': np.isclose(C - P, forward_diff)
        }

# Example pricing
bs = BlackScholes(
    S=100,      # Stock at $100
    K=100,      # Strike at $100 (at-the-money)
    T=0.5,      # 6 months
    r=0.05,     # 5% risk-free rate
    sigma=0.20  # 20% volatility
)

print("Black-Scholes Option Prices:")
print(f"  Stock Price: ${bs.S}")
print(f"  Strike: ${bs.K}")
print(f"  Time to Expiry: {bs.T} years")
print(f"  Risk-free Rate: {bs.r:.1%}")
print(f"  Volatility: {bs.sigma:.1%}")
print(f"\n  d1 = {bs.d1:.4f}")
print(f"  d2 = {bs.d2:.4f}")
print(f"\n  Call Price: ${bs.call_price():.2f}")
print(f"  Put Price: ${bs.put_price():.2f}")

parity = bs.call_put_parity()
print(f"\nPut-Call Parity Check:")
print(f"  C - P = {parity['C - P']:.4f}")
print(f"  S - Ke^(-rT) = {parity['S - K*e^(-rT)']:.4f}")
print(f"  Parity holds: {parity['parity_holds']}")
```

## 4. Sensitivity Analysis

Understanding how option prices change with inputs.

```python
def analyze_option_sensitivities(base_S=100, base_K=100, base_T=0.5,
                                   base_r=0.05, base_sigma=0.2):
    """
    Analyze how option prices change with each parameter.
    """
    results = {}

    # Vary stock price
    S_range = np.linspace(80, 120, 5)
    call_by_S = [BlackScholes(S, base_K, base_T, base_r, base_sigma).call_price()
                 for S in S_range]
    results['stock_price'] = {'values': S_range, 'call_prices': call_by_S}

    # Vary volatility
    sigma_range = np.linspace(0.10, 0.40, 5)
    call_by_sigma = [BlackScholes(base_S, base_K, base_T, base_r, s).call_price()
                     for s in sigma_range]
    results['volatility'] = {'values': sigma_range, 'call_prices': call_by_sigma}

    # Vary time to expiry
    T_range = np.linspace(0.1, 1.0, 5)
    call_by_T = [BlackScholes(base_S, base_K, t, base_r, base_sigma).call_price()
                 for t in T_range]
    results['time'] = {'values': T_range, 'call_prices': call_by_T}

    # Vary interest rate
    r_range = np.linspace(0.01, 0.10, 5)
    call_by_r = [BlackScholes(base_S, base_K, base_T, r, base_sigma).call_price()
                 for r in r_range]
    results['rate'] = {'values': r_range, 'call_prices': call_by_r}

    return results

sens = analyze_option_sensitivities()

print("Sensitivity Analysis (ATM Call, Base Price = $6.89):")
print("\nStock Price Effect:")
for s, c in zip(sens['stock_price']['values'], sens['stock_price']['call_prices']):
    print(f"  S = ${s:.0f}: Call = ${c:.2f}")

print("\nVolatility Effect:")
for v, c in zip(sens['volatility']['values'], sens['volatility']['call_prices']):
    print(f"  σ = {v:.0%}: Call = ${c:.2f}")
```

<Tip>
Option prices increase with volatility regardless of direction - higher uncertainty means higher option value since the downside is limited.
</Tip>

## 5. Implied Volatility

Market-implied volatility is the σ that makes Black-Scholes match observed prices.

```python
from scipy.optimize import brentq

def implied_volatility(market_price, S, K, T, r, option_type='call'):
    """
    Calculate implied volatility using Brent's method.

    Args:
        market_price: Observed option price
        S, K, T, r: Black-Scholes parameters
        option_type: 'call' or 'put'

    Returns:
        Implied volatility
    """
    def objective(sigma):
        bs = BlackScholes(S, K, T, r, sigma)
        if option_type == 'call':
            return bs.call_price() - market_price
        else:
            return bs.put_price() - market_price

    # Bounds for volatility search
    try:
        iv = brentq(objective, 0.001, 3.0)
        return iv
    except ValueError:
        return np.nan

# Example: Find implied volatility
market_call = 8.50  # Observed market price
S, K, T, r = 100, 100, 0.5, 0.05

iv = implied_volatility(market_call, S, K, T, r)
print(f"Implied Volatility Calculation:")
print(f"  Market Call Price: ${market_call:.2f}")
print(f"  Implied Volatility: {iv:.1%}")

# Verify
bs_verify = BlackScholes(S, K, T, r, iv)
print(f"  Verify BS Price: ${bs_verify.call_price():.2f}")

# Volatility smile example
strikes = [90, 95, 100, 105, 110]
market_prices = [12.50, 9.20, 6.80, 4.90, 3.50]  # Hypothetical

ivs = [implied_volatility(p, S, k, T, r) for p, k in zip(market_prices, strikes)]

print("\nImplied Volatility Smile:")
for k, p, iv in zip(strikes, market_prices, ivs):
    print(f"  K={k}: Price=${p:.2f}, IV={iv:.1%}")
```

## 6. Model Limitations

<Warning>
The Black-Scholes model has known limitations that practitioners must understand:
</Warning>

```python
class BlackScholesLimitations:
    """
    Demonstrate Black-Scholes limitations.
    """

    @staticmethod
    def volatility_smile_issue():
        """
        BS assumes constant volatility, but markets show a 'smile'.
        """
        return {
            'assumption': 'Constant volatility σ',
            'reality': 'Implied vol varies with strike (smile/skew)',
            'impact': 'OTM puts often underpriced by BS',
            'solution': 'Use local volatility or stochastic volatility models'
        }

    @staticmethod
    def jump_risk_issue():
        """
        BS assumes continuous price paths, missing jumps.
        """
        return {
            'assumption': 'Continuous geometric Brownian motion',
            'reality': 'Prices can jump (earnings, events)',
            'impact': 'Short-dated OTM options underpriced',
            'solution': 'Jump-diffusion models (Merton, Kou)'
        }

    @staticmethod
    def early_exercise_issue():
        """
        BS only works for European options.
        """
        return {
            'assumption': 'Exercise only at expiration',
            'reality': 'American options can exercise early',
            'impact': 'BS underprices American calls on dividends',
            'solution': 'Binomial trees, finite difference methods'
        }

    @staticmethod
    def discrete_dividends_issue():
        """
        Standard BS assumes no dividends.
        """
        return {
            'assumption': 'No dividends during option life',
            'reality': 'Stocks pay discrete dividends',
            'impact': 'Call prices overestimated, put prices underestimated',
            'solution': 'Adjust stock price or use dividend models'
        }

# Display limitations
limits = BlackScholesLimitations()
print("Black-Scholes Model Limitations:")
print("\n1. Volatility Smile:")
for k, v in limits.volatility_smile_issue().items():
    print(f"   {k}: {v}")

print("\n2. Jump Risk:")
for k, v in limits.jump_risk_issue().items():
    print(f"   {k}: {v}")

print("\n3. Early Exercise:")
for k, v in limits.early_exercise_issue().items():
    print(f"   {k}: {v}")
```

## 7. Black-Scholes with Dividends

Adjust for continuous dividend yield.

```python
class BlackScholesDividend:
    """
    Black-Scholes with continuous dividend yield.
    """

    def __init__(self, S, K, T, r, sigma, q):
        """
        Args:
            q: Continuous dividend yield
        """
        self.S = S
        self.K = K
        self.T = T
        self.r = r
        self.sigma = sigma
        self.q = q

        # Adjusted d1 and d2
        self.d1 = (np.log(S/K) + (r - q + sigma**2/2)*T) / (sigma*np.sqrt(T))
        self.d2 = self.d1 - sigma*np.sqrt(T)

    def call_price(self):
        """European call with dividends."""
        return (self.S * np.exp(-self.q * self.T) * norm.cdf(self.d1) -
                self.K * np.exp(-self.r * self.T) * norm.cdf(self.d2))

    def put_price(self):
        """European put with dividends."""
        return (self.K * np.exp(-self.r * self.T) * norm.cdf(-self.d2) -
                self.S * np.exp(-self.q * self.T) * norm.cdf(-self.d1))

# Compare with and without dividends
bs_no_div = BlackScholes(100, 100, 1.0, 0.05, 0.20)
bs_div = BlackScholesDividend(100, 100, 1.0, 0.05, 0.20, q=0.02)

print("Impact of Dividends (2% yield):")
print(f"\n  Without dividends:")
print(f"    Call: ${bs_no_div.call_price():.2f}")
print(f"    Put: ${bs_no_div.put_price():.2f}")
print(f"\n  With 2% dividend yield:")
print(f"    Call: ${bs_div.call_price():.2f}")
print(f"    Put: ${bs_div.put_price():.2f}")
print(f"\n  Call decreases: ${bs_no_div.call_price() - bs_div.call_price():.2f}")
print(f"  Put increases: ${bs_div.put_price() - bs_no_div.put_price():.2f}")
```
