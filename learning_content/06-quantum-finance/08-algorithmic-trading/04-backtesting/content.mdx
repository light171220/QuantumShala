# Strategy Backtesting

Backtesting simulates how a strategy would have performed on historical data. Proper backtesting methodology is crucial for avoiding false confidence in strategies.

## Backtesting Framework Architecture

```python
"""
Robust Backtesting Framework
"""

from dataclasses import dataclass, field
from typing import Dict, List, Optional, Callable
from datetime import datetime
from enum import Enum
from abc import ABC, abstractmethod
import numpy as np

class OrderSide(Enum):
    BUY = "buy"
    SELL = "sell"

@dataclass
class Order:
    symbol: str
    side: OrderSide
    quantity: float
    order_type: str = "market"
    limit_price: Optional[float] = None

@dataclass
class Fill:
    order: Order
    fill_price: float
    fill_quantity: float
    timestamp: datetime
    commission: float = 0.0
    slippage: float = 0.0

@dataclass
class Position:
    symbol: str
    quantity: float
    avg_cost: float
    unrealized_pnl: float = 0.0

@dataclass
class PortfolioState:
    cash: float
    positions: Dict[str, Position]
    timestamp: datetime
    equity: float = 0.0

@dataclass
class MarketBar:
    symbol: str
    timestamp: datetime
    open: float
    high: float
    low: float
    close: float
    volume: float

class Strategy(ABC):
    """Base class for trading strategies."""

    @abstractmethod
    def on_bar(self, bar: MarketBar, portfolio: PortfolioState) -> List[Order]:
        """Called on each bar. Returns list of orders."""
        pass

    def on_fill(self, fill: Fill):
        """Called when an order is filled."""
        pass


class BacktestEngine:
    """
    Event-driven backtesting engine.
    """

    def __init__(self,
                 initial_capital: float = 100000,
                 commission_rate: float = 0.001,
                 slippage_model: str = "fixed"):
        self.initial_capital = initial_capital
        self.commission_rate = commission_rate
        self.slippage_model = slippage_model

        # State
        self.cash = initial_capital
        self.positions: Dict[str, Position] = {}
        self.trades: List[Fill] = []
        self.equity_curve: List[Dict] = []
        self.orders_pending: List[Order] = []

    def run(self, strategy: Strategy,
            data: Dict[str, List[MarketBar]]) -> Dict:
        """
        Run backtest with given strategy and data.
        """
        # Combine all bars and sort by timestamp
        all_bars = []
        for symbol, bars in data.items():
            all_bars.extend(bars)
        all_bars.sort(key=lambda b: b.timestamp)

        # Current prices for each symbol
        current_prices: Dict[str, float] = {}

        for bar in all_bars:
            current_prices[bar.symbol] = bar.close

            # Calculate portfolio state
            portfolio = self._get_portfolio_state(current_prices, bar.timestamp)

            # Record equity curve
            self.equity_curve.append({
                "timestamp": bar.timestamp,
                "equity": portfolio.equity,
                "cash": portfolio.cash
            })

            # Get strategy orders
            orders = strategy.on_bar(bar, portfolio)

            # Execute orders
            for order in orders:
                fill = self._execute_order(order, bar)
                if fill:
                    self.trades.append(fill)
                    strategy.on_fill(fill)

        # Generate results
        return self._generate_results()

    def _get_portfolio_state(self, prices: Dict[str, float],
                              timestamp: datetime) -> PortfolioState:
        """Calculate current portfolio state."""
        positions = {}
        position_value = 0.0

        for symbol, pos in self.positions.items():
            current_price = prices.get(symbol, pos.avg_cost)
            unrealized = (current_price - pos.avg_cost) * pos.quantity

            positions[symbol] = Position(
                symbol=symbol,
                quantity=pos.quantity,
                avg_cost=pos.avg_cost,
                unrealized_pnl=unrealized
            )
            position_value += current_price * pos.quantity

        equity = self.cash + position_value

        return PortfolioState(
            cash=self.cash,
            positions=positions,
            timestamp=timestamp,
            equity=equity
        )

    def _execute_order(self, order: Order, bar: MarketBar) -> Optional[Fill]:
        """Execute an order with slippage and commission."""
        # Calculate fill price with slippage
        slippage = self._calculate_slippage(order, bar)

        if order.side == OrderSide.BUY:
            fill_price = bar.close + slippage
        else:
            fill_price = bar.close - slippage

        # Calculate commission
        commission = abs(fill_price * order.quantity * self.commission_rate)

        # Check if we have enough cash for buys
        if order.side == OrderSide.BUY:
            total_cost = fill_price * order.quantity + commission
            if total_cost > self.cash:
                return None  # Reject order

        # Execute
        fill = Fill(
            order=order,
            fill_price=fill_price,
            fill_quantity=order.quantity,
            timestamp=bar.timestamp,
            commission=commission,
            slippage=slippage * order.quantity
        )

        # Update positions and cash
        self._update_positions(fill)

        return fill

    def _calculate_slippage(self, order: Order, bar: MarketBar) -> float:
        """Calculate slippage based on model."""
        if self.slippage_model == "fixed":
            return 0.01  # $0.01 per share
        elif self.slippage_model == "percentage":
            return bar.close * 0.0001  # 1 basis point
        elif self.slippage_model == "volatility":
            # Higher slippage in volatile bars
            bar_range = bar.high - bar.low
            return bar_range * 0.1
        return 0.0

    def _update_positions(self, fill: Fill):
        """Update positions and cash after fill."""
        symbol = fill.order.symbol

        if fill.order.side == OrderSide.BUY:
            self.cash -= (fill.fill_price * fill.fill_quantity + fill.commission)

            if symbol in self.positions:
                pos = self.positions[symbol]
                # Calculate new average cost
                total_cost = pos.avg_cost * pos.quantity + fill.fill_price * fill.fill_quantity
                new_qty = pos.quantity + fill.fill_quantity
                pos.avg_cost = total_cost / new_qty
                pos.quantity = new_qty
            else:
                self.positions[symbol] = Position(
                    symbol=symbol,
                    quantity=fill.fill_quantity,
                    avg_cost=fill.fill_price
                )
        else:
            self.cash += (fill.fill_price * fill.fill_quantity - fill.commission)

            if symbol in self.positions:
                pos = self.positions[symbol]
                pos.quantity -= fill.fill_quantity
                if pos.quantity <= 0:
                    del self.positions[symbol]

    def _generate_results(self) -> Dict:
        """Generate backtest results."""
        if not self.equity_curve:
            return {}

        equities = [e["equity"] for e in self.equity_curve]

        # Calculate returns
        returns = np.diff(equities) / equities[:-1]

        # Calculate metrics
        total_return = (equities[-1] / equities[0] - 1) * 100
        sharpe = np.mean(returns) / np.std(returns) * np.sqrt(252) if np.std(returns) > 0 else 0
        max_dd = self._calculate_max_drawdown(equities)

        # Trade analysis
        wins = [t for t in self.trades if self._is_winning_trade(t)]
        win_rate = len(wins) / len(self.trades) * 100 if self.trades else 0

        total_commission = sum(t.commission for t in self.trades)
        total_slippage = sum(t.slippage for t in self.trades)

        return {
            "total_return_pct": total_return,
            "sharpe_ratio": sharpe,
            "max_drawdown_pct": max_dd,
            "total_trades": len(self.trades),
            "win_rate_pct": win_rate,
            "total_commission": total_commission,
            "total_slippage": total_slippage,
            "final_equity": equities[-1]
        }

    def _calculate_max_drawdown(self, equities: List[float]) -> float:
        """Calculate maximum drawdown percentage."""
        peak = equities[0]
        max_dd = 0.0

        for equity in equities:
            if equity > peak:
                peak = equity
            dd = (peak - equity) / peak * 100
            max_dd = max(max_dd, dd)

        return max_dd

    def _is_winning_trade(self, fill: Fill) -> bool:
        """Determine if a trade was winning (simplified)."""
        return True  # Would need to track round trips


# Example strategy
class SimpleMAStrategy(Strategy):
    def __init__(self, fast_period: int = 10, slow_period: int = 30):
        self.fast_period = fast_period
        self.slow_period = slow_period
        self.prices: Dict[str, List[float]] = {}

    def on_bar(self, bar: MarketBar, portfolio: PortfolioState) -> List[Order]:
        symbol = bar.symbol

        if symbol not in self.prices:
            self.prices[symbol] = []
        self.prices[symbol].append(bar.close)

        if len(self.prices[symbol]) < self.slow_period:
            return []

        fast_ma = np.mean(self.prices[symbol][-self.fast_period:])
        slow_ma = np.mean(self.prices[symbol][-self.slow_period:])

        current_position = portfolio.positions.get(symbol)
        has_position = current_position and current_position.quantity > 0

        orders = []

        if fast_ma > slow_ma and not has_position:
            # Buy signal
            qty = portfolio.cash * 0.95 / bar.close  # 95% of cash
            orders.append(Order(symbol, OrderSide.BUY, qty))

        elif fast_ma < slow_ma and has_position:
            # Sell signal
            orders.append(Order(symbol, OrderSide.SELL, current_position.quantity))

        return orders


# Run backtest

        low=price * 0.99,

    ))

from datetime import timedelta
```

## Common Backtesting Biases

Understanding and avoiding biases is critical for reliable backtests.

```python
from enum import Enum
from dataclasses import dataclass
from typing import List, Dict

class BiasType(Enum):
    SURVIVORSHIP = "survivorship"
    LOOK_AHEAD = "look_ahead"
    OVERFITTING = "overfitting"
    SELECTION = "selection"
    TRANSACTION_COST = "transaction_cost"

@dataclass
class BiasDescription:
    bias_type: BiasType
    description: str
    example: str
    mitigation: str

COMMON_BIASES = [
    BiasDescription(
        bias_type=BiasType.SURVIVORSHIP,
        description="Only testing on assets that survived to present day",
        example="Testing on current S&P 500 members ignores delisted companies",
        mitigation="Use point-in-time universe data that includes delisted assets"
    ),
    BiasDescription(
        bias_type=BiasType.LOOK_AHEAD,
        description="Using information not available at the time of the decision",
        example="Using revised GDP numbers instead of initial estimates",
        mitigation="Use as-reported data, ensure data timestamps are respected"
    ),
    BiasDescription(
        bias_type=BiasType.OVERFITTING,
        description="Fitting strategy to historical noise rather than real patterns",
        example="Adding parameters until backtest looks perfect",
        mitigation="Out-of-sample testing, cross-validation, regularization"
    ),
    BiasDescription(
        bias_type=BiasType.SELECTION,
        description="Cherry-picking time periods or assets that look good",
        example="Only showing backtest results for 2020 bull market",
        mitigation="Test across multiple market regimes and time periods"
    ),
    BiasDescription(
        bias_type=BiasType.TRANSACTION_COST,
        description="Ignoring or underestimating trading costs",
        example="Assuming zero slippage on large orders",
        mitigation="Model realistic commission, slippage, and market impact"
    )
]

class BiasDetector:
    """Detect potential biases in backtests."""

    def check_survivorship_bias(self,
                                 symbols_tested: List[str],
                                 delisted_in_period: List[str]) -> Dict:
        """Check for survivorship bias."""
        missing_delisted = [s for s in delisted_in_period if s not in symbols_tested]

        return {
            "bias_detected": len(missing_delisted) > 0,
            "severity": len(missing_delisted) / len(delisted_in_period) if delisted_in_period else 0,
            "missing_symbols": missing_delisted,
            "recommendation": "Include delisted symbols in backtest universe"
        }

    def check_overfitting(self,
                          n_parameters: int,
                          n_trades: int,
                          n_assets: int) -> Dict:
        """Check for potential overfitting."""
        # Rule of thumb: need many trades per parameter
        trades_per_param = n_trades / n_parameters if n_parameters > 0 else float('inf')

        return {
            "bias_detected": trades_per_param < 30,
            "trades_per_parameter": trades_per_param,
            "recommendation": f"Need ~30 trades per parameter, have {trades_per_param:.1f}",
            "severity": max(0, 1 - trades_per_param / 30)
        }

    def check_cost_realism(self,
                           assumed_commission: float,
                           assumed_slippage: float,
                           typical_commission: float = 0.001,
                           typical_slippage: float = 0.0005) -> Dict:
        """Check if transaction costs are realistic."""
        commission_ratio = assumed_commission / typical_commission if typical_commission > 0 else 0
        slippage_ratio = assumed_slippage / typical_slippage if typical_slippage > 0 else 0

        return {
            "bias_detected": commission_ratio < 0.5 or slippage_ratio < 0.5,
            "commission_ratio": commission_ratio,
            "slippage_ratio": slippage_ratio,
            "recommendation": "Consider higher transaction costs for conservatism"
        }


# Display biases

    print(f"\n{bias.bias_type.value.upper()}")
    print(f"  Description: {bias.description}")
    print(f"  Example: {bias.example}")
    print(f"  Mitigation: {bias.mitigation}")
```

## Out-of-Sample Testing

Properly validate strategies with train/test splits.

```python
import numpy as np
from typing import Dict, List, Tuple
from dataclasses import dataclass

@dataclass
class BacktestPeriod:
    name: str
    start_idx: int
    end_idx: int
    purpose: str  # "train", "validation", "test"

class WalkForwardAnalyzer:
    """
    Walk-forward analysis for robust strategy validation.
    """

    def __init__(self,
                 train_window: int = 252,  # 1 year training
                 test_window: int = 63,    # 3 months testing
                 n_splits: int = 4):
        self.train_window = train_window
        self.test_window = test_window
        self.n_splits = n_splits

    def create_periods(self, total_length: int) -> List[BacktestPeriod]:
        """Create train/test periods for walk-forward analysis."""
        periods = []
        window_size = self.train_window + self.test_window

        for i in range(self.n_splits):
            start = i * self.test_window
            train_end = start + self.train_window
            test_end = train_end + self.test_window

            if test_end > total_length:
                break

            periods.append(BacktestPeriod(
                name=f"Train_{i+1}",
                start_idx=start,
                end_idx=train_end,
                purpose="train"
            ))

            periods.append(BacktestPeriod(
                name=f"Test_{i+1}",
                start_idx=train_end,
                end_idx=test_end,
                purpose="test"
            ))

        return periods

    def analyze_degradation(self,
                            train_results: List[float],
                            test_results: List[float]) -> Dict:
        """Analyze performance degradation from train to test."""
        if not train_results or not test_results:
            return {}

        avg_train = np.mean(train_results)
        avg_test = np.mean(test_results)

        degradation = (avg_train - avg_test) / abs(avg_train) * 100 if avg_train != 0 else 0

        return {
            "avg_train_return": avg_train,
            "avg_test_return": avg_test,
            "degradation_pct": degradation,
            "is_concerning": degradation > 50,
            "recommendation": self._get_recommendation(degradation)
        }

    def _get_recommendation(self, degradation: float) -> str:
        if degradation > 80:
            return "Strategy likely overfit. Consider simpler approach."
        elif degradation > 50:
            return "Significant degradation. Review parameter stability."
        elif degradation > 20:
            return "Moderate degradation. Monitor in paper trading."
        else:
            return "Acceptable degradation. Proceed with caution."


class MonteCarloValidator:
    """
    Monte Carlo simulation for strategy validation.
    """

    def __init__(self, n_simulations: int = 1000):
        self.n_simulations = n_simulations

    def shuffle_returns(self, returns: np.ndarray) -> np.ndarray:
        """Randomly shuffle returns to create synthetic path."""
        return np.random.permutation(returns)

    def run_validation(self,
                       strategy_returns: np.ndarray,
                       strategy_metric: float) -> Dict:
        """
        Run Monte Carlo validation.
        Tests if strategy metric could be achieved by chance.
        """
        random_metrics = []

        for _ in range(self.n_simulations):
            shuffled = self.shuffle_returns(strategy_returns)
            # Calculate same metric on shuffled returns
            metric = np.sum(shuffled)  # Simplified: total return
            random_metrics.append(metric)

        random_metrics = np.array(random_metrics)

        # Calculate percentile of actual strategy
        percentile = np.mean(random_metrics < strategy_metric) * 100

        return {
            "strategy_metric": strategy_metric,
            "random_mean": np.mean(random_metrics),
            "random_std": np.std(random_metrics),
            "percentile": percentile,
            "is_significant": percentile > 95,
            "p_value": 1 - percentile / 100
        }


# Demonstrate walk-forward analysis
analyzer = WalkForwardAnalyzer(train_window=100, test_window=25, n_splits=4)
periods = analyzer.create_periods(total_length=252)

print("\nWalk-Forward Analysis Periods")
print("=" * 40)
for period in periods:
    print(f"{period.name}: {period.start_idx} to {period.end_idx} ({period.purpose})")

# Simulate train/test results
train_results = [0.15, 0.12, 0.18, 0.14]  # Good train performance
test_results = [0.05, 0.03, 0.08, 0.04]   # Degraded test performance

degradation = analyzer.analyze_degradation(train_results, test_results)
print(f"\nDegradation Analysis:")
print(f"  Train avg: {degradation['avg_train_return']*100:.1f}%")
print(f"  Test avg: {degradation['avg_test_return']*100:.1f}%")
print(f"  Degradation: {degradation['degradation_pct']:.1f}%")
print(f"  Recommendation: {degradation['recommendation']}")
```

## Summary

Robust backtesting requires:

1. **Proper Framework**: Event-driven architecture with realistic execution
2. **Bias Awareness**: Avoid survivorship, look-ahead, and overfitting biases
3. **Cost Modeling**: Include commission, slippage, and market impact
4. **Validation**: Walk-forward analysis and Monte Carlo testing
5. **Skepticism**: Treat backtest results as upper bounds on real performance
