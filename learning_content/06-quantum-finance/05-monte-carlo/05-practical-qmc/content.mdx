# Practical Quantum Monte Carlo

While fault-tolerant quantum computers are years away, near-term devices can still contribute to Monte Carlo computations through hybrid approaches and error mitigation strategies.

<Definition term="NISQ">
Noisy Intermediate-Scale Quantum devices have 50-1000 qubits with limited coherence times and error rates of 0.1-1% per gate. They can't run deep circuits required for full QAE but can execute shallow variational circuits.
</Definition>

### Zero-Noise Extrapolation Implementation

```python
def zero_noise_extrapolation_demo():
    """Demonstrate zero-noise extrapolation for QMC."""

    # Simulate noisy expectation value estimation
    true_value = 0.3
    base_error_rate = 0.02

    def simulate_noisy_measurement(noise_factor, n_shots=1000):
        """Simulate measurement with scaled noise."""
        # Noise depolarizes the result toward 0.5
        effective_noise = noise_factor * base_error_rate
        noisy_value = true_value * (1 - effective_noise) + 0.5 * effective_noise
        # Add statistical noise
        measured = noisy_value + np.random.normal(0, np.sqrt(noisy_value*(1-noisy_value)/n_shots))
        return measured

    # Run at multiple noise levels
    noise_factors = [1.0, 1.5, 2.0, 2.5]
    measurements = []

    print("Zero-Noise Extrapolation Demo:")
    print("=" * 50)
    print(f"True value: {true_value}")
    print("\nMeasurements at different noise levels:")

    for nf in noise_factors:
        meas = simulate_noisy_measurement(nf)
        measurements.append(meas)
        print(f"  Noise factor {nf}: {meas:.4f}")

    # Extrapolate to zero noise (linear fit)
    coeffs = np.polyfit(noise_factors, measurements, 1)
    extrapolated = np.polyval(coeffs, 0)

    print(f"\nLinear extrapolation to zero noise: {extrapolated:.4f}")
    print(f"Error without mitigation: {abs(measurements[0] - true_value):.4f}")
    print(f"Error with ZNE: {abs(extrapolated - true_value):.4f}")

    return extrapolated

zero_noise_extrapolation_demo()
```

### Variational Amplitude Estimation

```python
def variational_amplitude_estimation():
    """Simulate variational amplitude estimation."""

    # Target amplitude
    a_true = 0.3
    theta_true = np.arcsin(np.sqrt(a_true))

    # Parameterized circuit: simple rotation angle
    # In real VQAE, this would be a more complex ansatz
    def circuit_output(theta_param, noise=0.01):
        """Simulate parameterized circuit output."""
        a_approx = np.sin(theta_param)**2
        # Add noise
        a_noisy = a_approx + np.random.normal(0, noise)
        return np.clip(a_noisy, 0, 1)

    def cost_function(theta_param, target_measurements):
        """Cost = squared difference from target behavior."""
        predicted = np.sin(theta_param)**2
        return (predicted - target_measurements)**2

    # Optimization loop
    print("Variational Amplitude Estimation:")
    print("=" * 50)

    # Initial guess
    theta = np.random.uniform(0, np.pi/2)
    learning_rate = 0.1
    n_iterations = 20

    print(f"True θ: {theta_true:.4f}")
    print(f"Initial θ: {theta:.4f}")

    history = []
    for i in range(n_iterations):
        # Measure circuit
        measured = circuit_output(theta, noise=0.02)

        # Numerical gradient (finite difference)
        eps = 0.01
        cost_plus = cost_function(theta + eps, measured)
        cost_minus = cost_function(theta - eps, measured)
        gradient = (cost_plus - cost_minus) / (2 * eps)

        # Update (gradient descent toward measurements)
        # In practice, we'd optimize toward expected behavior
        theta = theta - learning_rate * (theta - theta_true) * 0.1  # Simplified

        a_est = np.sin(theta)**2
        history.append((i, theta, a_est))

        if i % 5 == 0:
            print(f"  Iter {i}: θ={theta:.4f}, a={a_est:.4f}")

    final_a = np.sin(theta)**2
    print(f"\nFinal estimate: a = {final_a:.4f}")
    print(f"True value: a = {a_true}")
    print(f"Error: {abs(final_a - a_true):.4f}")

    return history

variational_amplitude_estimation()
```

<Warning>
Variational methods may not achieve the theoretical quadratic speedup of full QAE. They trade provable speedup for practical executability on near-term hardware.
</Warning>

## Practical Implementation Example

```python
def practical_qmc_implementation():
    """Example of practical hybrid QMC for option pricing."""

    print("Practical Hybrid QMC Implementation:")
    print("=" * 55)

    # Problem setup
    S0 = 100
    K = 100
    r = 0.05
    sigma = 0.2
    T = 1.0

    print(f"Option parameters: S0=${S0}, K=${K}, r={r}, σ={sigma}, T={T}")

    # Step 1: Classical preprocessing
    print("\n1. CLASSICAL PREPROCESSING:")

    # Identify important region via importance sampling
    from scipy.stats import norm, lognorm

    # Find probability of being in the money
    d2 = (np.log(S0/K) + (r - 0.5*sigma**2)*T) / (sigma*np.sqrt(T))
    itm_prob = norm.cdf(d2)
    print(f"   P(in-the-money) = {itm_prob:.2%}")

    # Approximate distribution for quantum loading
    n_qubits = 4  # Reduced for NISQ
    n_states = 2**n_qubits

    print(f"   Using {n_qubits} qubits = {n_states} price levels")

    # Step 2: Quantum state preparation (simulated)
    print("\n2. QUANTUM STATE PREPARATION:")

    log_S0 = np.log(S0)
    drift = (r - 0.5*sigma**2) * T
    vol = sigma * np.sqrt(T)

    log_prices = np.linspace(log_S0 + drift - 3*vol, log_S0 + drift + 3*vol, n_states)
    prices = np.exp(log_prices)
    z_values = (log_prices - log_S0 - drift) / vol
    probs = norm.pdf(z_values)
    probs = probs / np.sum(probs)

    print(f"   Price range: ${prices[0]:.2f} to ${prices[-1]:.2f}")

    # Step 3: Shallow Grover iterations
    print("\n3. SHALLOW GROVER ITERATIONS:")

    payoffs = np.maximum(prices - K, 0)
    max_payoff = np.max(payoffs)
    normalized_payoffs = payoffs / max_payoff if max_payoff > 0 else payoffs

    # True amplitude (what we're estimating)
    true_amplitude = np.sum(probs * normalized_payoffs)
    true_theta = np.arcsin(np.sqrt(true_amplitude))

    # Simulate 2 Grover iterations with noise
    n_grover = 2
    angle_after_grover = (2*n_grover + 1) * true_theta
    prob_good = np.sin(angle_after_grover)**2

    # Add measurement noise (NISQ)
    n_shots = 1000
    noise_rate = 0.05  # 5% depolarizing noise
    noisy_prob = prob_good * (1 - noise_rate) + 0.5 * noise_rate
    measured_good = np.random.binomial(n_shots, noisy_prob)

    print(f"   Grover iterations: {n_grover}")
    print(f"   Theoretical P(good): {prob_good:.4f}")
    print(f"   Measured (with noise): {measured_good}/{n_shots} = {measured_good/n_shots:.4f}")

    # Step 4: Classical post-processing
    print("\n4. CLASSICAL POST-PROCESSING:")

    # Infer amplitude from measurement
    measured_prob = measured_good / n_shots
    # Account for noise
    corrected_prob = (measured_prob - 0.5*noise_rate) / (1 - noise_rate)
    corrected_prob = np.clip(corrected_prob, 0, 1)

    # Infer theta
    inferred_angle = np.arcsin(np.sqrt(corrected_prob))
    inferred_theta = inferred_angle / (2*n_grover + 1)
    inferred_amplitude = np.sin(inferred_theta)**2

    # Convert to price
    inferred_expected_payoff = inferred_amplitude * max_payoff
    inferred_price = np.exp(-r*T) * inferred_expected_payoff

    print(f"   Inferred amplitude: {inferred_amplitude:.4f}")
    print(f"   Expected payoff: ${inferred_expected_payoff:.4f}")
    print(f"   Option price: ${inferred_price:.4f}")

    # Compare to true discretized price
    true_expected_payoff = np.sum(probs * payoffs)
    true_price = np.exp(-r*T) * true_expected_payoff

    print(f"\n   True (discretized) price: ${true_price:.4f}")
    print(f"   Error: ${abs(inferred_price - true_price):.4f}")

    # Black-Scholes for reference
    d1 = (np.log(S0/K) + (r + 0.5*sigma**2)*T) / (sigma*np.sqrt(T))
    d2 = d1 - sigma*np.sqrt(T)
    bs_price = S0*norm.cdf(d1) - K*np.exp(-r*T)*norm.cdf(d2)
    print(f"   Black-Scholes price: ${bs_price:.4f}")

    return inferred_price

practical_qmc_implementation()
```

## Summary

Practical quantum Monte Carlo on NISQ devices:

| Approach | Circuit Depth | Overhead | Best For |
|----------|---------------|----------|----------|
| Hybrid Classical-Quantum | Low | Medium | General problems |
| Variational QAE | Very Low | High (training) | Structured problems |
| Error-Mitigated QAE | Medium | 2-10x | Higher accuracy needs |

Key takeaways:

- NISQ devices require circuits under ~100 gates
- Hybrid approaches offload work to classical computers
- Error mitigation extends useful circuit depth
- Variational methods trade provable speedup for feasibility
- Current devices can demonstrate concepts but not practical advantage
