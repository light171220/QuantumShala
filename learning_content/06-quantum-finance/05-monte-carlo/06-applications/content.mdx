### Exotic Option Example

```python
def quantum_exotic_pricing():
    """Example: Quantum pricing for Asian option."""

    # Parameters
    S0 = 100
    K = 100
    r = 0.05
    sigma = 0.2
    T = 1.0
    n_averaging = 12  # Monthly averaging

    print("Quantum Asian Option Pricing:")
    print("=" * 50)
    print(f"Parameters: S0=${S0}, K=${K}, r={r}, σ={sigma}, T={T}")
    print(f"Averaging: {n_averaging} observation points")

    # Classical Monte Carlo estimate
    n_paths = 100000
    dt = T / n_averaging

    np.random.seed(42)
    paths = np.zeros((n_paths, n_averaging + 1))
    paths[:, 0] = S0

    for t in range(n_averaging):
        Z = np.random.standard_normal(n_paths)
        paths[:, t+1] = paths[:, t] * np.exp((r - 0.5*sigma**2)*dt + sigma*np.sqrt(dt)*Z)

    averages = np.mean(paths, axis=1)
    payoffs = np.maximum(averages - K, 0)
    classical_price = np.exp(-r*T) * np.mean(payoffs)
    classical_std = np.exp(-r*T) * np.std(payoffs) / np.sqrt(n_paths)

    print(f"\nClassical MC ({n_paths:,} paths):")
    print(f"  Price: ${classical_price:.4f} ± ${1.96*classical_std:.4f}")
    print(f"  Computational cost: O({n_paths * n_averaging:,}) operations")

    # Quantum complexity comparison
    target_accuracy = classical_std
    quantum_queries = int(1 / target_accuracy)

    print(f"\nQuantum complexity for same accuracy:")
    print(f"  Queries: O({quantum_queries:,})")
    print(f"  Theoretical speedup: {n_paths / quantum_queries:.0f}x")

    return classical_price

quantum_exotic_pricing()
```

### Quantum VaR Implementation

```python
def quantum_var_implementation():
    """Implement quantum approach to VaR calculation."""

    # Portfolio setup
    n_assets = 3
    weights = np.array([0.4, 0.35, 0.25])
    returns_mean = np.array([0.08, 0.06, 0.04])  # Annual
    returns_std = np.array([0.20, 0.15, 0.10])
    correlation = np.array([
        [1.0, 0.5, 0.3],
        [0.5, 1.0, 0.4],
        [0.3, 0.4, 1.0]
    ])

    # Covariance matrix
    std_diag = np.diag(returns_std)
    cov = std_diag @ correlation @ std_diag

    # Portfolio statistics
    portfolio_mean = weights @ returns_mean
    portfolio_var = weights @ cov @ weights
    portfolio_std = np.sqrt(portfolio_var)

    print("Quantum VaR Calculation:")
    print("=" * 50)
    print(f"Portfolio: {n_assets} assets")
    print(f"Expected return: {portfolio_mean*100:.2f}%")
    print(f"Portfolio volatility: {portfolio_std*100:.2f}%")

    # VaR at 99% confidence (parametric)
    alpha = 0.01
    z = norm.ppf(alpha)
    parametric_var = -(portfolio_mean + z * portfolio_std)

    print(f"\nParametric VaR (99%): {parametric_var*100:.2f}%")

    # Monte Carlo VaR
    n_simulations = 100000
    portfolio_value = 1000000  # $1M portfolio

    np.random.seed(42)
    L = np.linalg.cholesky(cov)
    Z = np.random.standard_normal((n_simulations, n_assets))
    asset_returns = returns_mean + (Z @ L.T)
    portfolio_returns = asset_returns @ weights

    mc_var = -np.percentile(portfolio_returns, alpha * 100) * portfolio_value
    mc_es = -np.mean(portfolio_returns[portfolio_returns <= np.percentile(portfolio_returns, alpha * 100)]) * portfolio_value

    print(f"Monte Carlo VaR (99%): ${mc_var:,.0f}")
    print(f"Expected Shortfall: ${mc_es:,.0f}")

    # Quantum complexity
    target_accuracy = 0.001  # 0.1% accuracy
    quantum_queries = int(1 / target_accuracy)
    classical_samples = int(1 / target_accuracy**2)

    print(f"\nComplexity for {target_accuracy*100:.1f}% accuracy:")
    print(f"  Classical: {classical_samples:,} samples")
    print(f"  Quantum: {quantum_queries:,} queries")
    print(f"  Speedup: {classical_samples / quantum_queries:.0f}x")

    return mc_var, mc_es

quantum_var_implementation()
```

<Tip>
Risk calculations often require real-time updates. Quantum advantage is most valuable when very high accuracy is needed for regulatory reporting or when running many scenarios.
</Tip>

## Credit Risk Modeling

Credit portfolio risk is computationally intensive due to correlation modeling.

```python
def quantum_credit_risk():
    """Quantum approaches to credit risk modeling."""

    print("Quantum Credit Risk Applications:")
    print("=" * 55)

    # Credit portfolio setup
    n_obligors = 1000
    default_probs = np.random.beta(2, 50, n_obligors)  # 0-10% default probs
    exposures = np.random.lognormal(10, 2, n_obligors)  # Random exposures
    lgd = 0.45  # Loss given default

    print(f"Portfolio: {n_obligors:,} obligors")
    print(f"Total exposure: ${np.sum(exposures)/1e6:.1f}M")
    print(f"Average PD: {np.mean(default_probs)*100:.2f}%")

    # Classical simulation (simplified)
    n_simulations = 100000
    correlation = 0.2  # Single factor model

    np.random.seed(42)
    systematic = np.random.standard_normal(n_simulations)
    idiosyncratic = np.random.standard_normal((n_simulations, n_obligors))

    factor_loadings = np.sqrt(correlation)
    specific_loadings = np.sqrt(1 - correlation)

    # Asset returns
    asset_returns = (factor_loadings * systematic[:, np.newaxis] +
                    specific_loadings * idiosyncratic)

    # Default thresholds
    default_thresholds = norm.ppf(default_probs)

    # Defaults
    defaults = asset_returns < default_thresholds

    # Portfolio losses
    losses = defaults * exposures * lgd
    portfolio_losses = np.sum(losses, axis=1)

    expected_loss = np.mean(portfolio_losses)
    var_99 = np.percentile(portfolio_losses, 99)
    es_99 = np.mean(portfolio_losses[portfolio_losses >= var_99])

    print(f"\nCredit Risk Metrics:")
    print(f"  Expected Loss: ${expected_loss:,.0f}")
    print(f"  VaR (99%): ${var_99:,.0f}")
    print(f"  Expected Shortfall: ${es_99:,.0f}")

    # Quantum advantage potential
    print("\nQuantum Advantage:")
    print("  - High-dimensional correlation modeling")
    print("  - Tail probability estimation for rare defaults")
    print("  - CVA calculations with default correlation")
    print(f"  - Potential {int(np.sqrt(n_simulations))}x speedup")

    return expected_loss, var_99

quantum_credit_risk()
```

<Warning>
Credit risk models involve complex correlation structures. Quantum advantage depends on being able to efficiently load the joint default distribution, which may require approximations.
</Warning>
