# Classical Monte Carlo Methods

Monte Carlo methods are computational algorithms that use random sampling to obtain numerical results. In finance, they are essential for pricing derivatives, measuring risk, and simulating market scenarios.

<Definition term="Monte Carlo Method">
A computational technique that uses repeated random sampling to estimate numerical quantities. Named after the Monte Carlo Casino, the method relies on the law of large numbers to converge to true values.
</Definition>

## The Law of Large Numbers

The theoretical foundation of Monte Carlo is the law of large numbers: sample means converge to population means.

<Formula expression="\\bar{X}_n = \\frac{1}{n}\\sum_{i=1}^{n} X_i \\xrightarrow{p} \\mathbb{E}[X] \\text{ as } n \\to \\infty" description="Sample mean converges in probability to expected value" />

```python
import numpy as np
import matplotlib.pyplot as plt
from scipy import stats

def demonstrate_law_of_large_numbers():
    """Show convergence of sample mean to true mean."""
    np.random.seed(42)

    # True mean of exponential distribution
    true_mean = 1.0  # For exp(rate=1)

    # Generate samples and track running mean
    n_samples = 10000
    samples = np.random.exponential(true_mean, n_samples)

    running_mean = np.cumsum(samples) / np.arange(1, n_samples + 1)

    # Calculate running standard error
    running_var = np.zeros(n_samples)
    for i in range(n_samples):
        running_var[i] = np.var(samples[:i+1], ddof=1) if i > 0 else 0
    running_se = np.sqrt(running_var / np.arange(1, n_samples + 1))

    print(f"True mean: {true_mean}")
    print(f"Sample mean (n=100): {running_mean[99]:.4f}")
    print(f"Sample mean (n=1000): {running_mean[999]:.4f}")
    print(f"Sample mean (n=10000): {running_mean[-1]:.4f}")

    return running_mean, running_se

running_mean, running_se = demonstrate_law_of_large_numbers()
```

## Monte Carlo Integration

The core application is estimating integrals (expected values) via random sampling.

```python
def monte_carlo_integration():
    """Demonstrate Monte Carlo integration for expected values."""

    # Example: E[max(S_T - K, 0)] for European call option
    S0 = 100
    K = 100
    r = 0.05
    sigma = 0.2
    T = 1.0

    def simulate_call_payoff(n_samples, seed=None):
        """Simulate terminal payoffs."""
        if seed is not None:
            np.random.seed(seed)

        # Generate terminal stock prices under risk-neutral measure
        Z = np.random.standard_normal(n_samples)
        S_T = S0 * np.exp((r - 0.5*sigma**2)*T + sigma*np.sqrt(T)*Z)

        # Calculate payoffs
        payoffs = np.maximum(S_T - K, 0)

        return payoffs

    # Monte Carlo estimation
    n_samples = 100000
    payoffs = simulate_call_payoff(n_samples, seed=42)

    # Discounted expected value
    mc_price = np.exp(-r*T) * np.mean(payoffs)
    mc_std_err = np.exp(-r*T) * np.std(payoffs) / np.sqrt(n_samples)

    print(f"Monte Carlo call price: ${mc_price:.4f}")
    print(f"Standard error: ${mc_std_err:.4f}")
    print(f"95% CI: [${mc_price - 1.96*mc_std_err:.4f}, ${mc_price + 1.96*mc_std_err:.4f}]")

    # Black-Scholes analytical price for comparison
    from scipy.stats import norm
    d1 = (np.log(S0/K) + (r + 0.5*sigma**2)*T) / (sigma*np.sqrt(T))
    d2 = d1 - sigma*np.sqrt(T)
    bs_price = S0*norm.cdf(d1) - K*np.exp(-r*T)*norm.cdf(d2)

    print(f"\nBlack-Scholes price: ${bs_price:.4f}")
    print(f"Error: ${abs(mc_price - bs_price):.4f}")

    return mc_price, bs_price

mc_price, bs_price = monte_carlo_integration()
```

## Convergence Rate

The central limit theorem governs Monte Carlo convergence.

<Formula expression="\\sqrt{n}(\\bar{X}_n - \\mu) \\xrightarrow{d} N(0, \\sigma^2)" description="CLT: Monte Carlo error is O(1/√n)" />

```python
def analyze_convergence_rate():
    """Analyze Monte Carlo convergence with sample size."""

    # True value (known integral)
    # Integral of x^2 from 0 to 1 = 1/3
    true_value = 1/3

    sample_sizes = [10, 50, 100, 500, 1000, 5000, 10000, 50000, 100000]
    n_trials = 100

    results = {
        'n': sample_sizes,
        'mean_error': [],
        'rmse': [],
        'theoretical_se': []
    }

    for n in sample_sizes:
        errors = []
        for trial in range(n_trials):
            # Monte Carlo estimate of integral of x^2
            samples = np.random.uniform(0, 1, n)
            estimate = np.mean(samples**2)
            errors.append(estimate - true_value)

        results['mean_error'].append(np.mean(np.abs(errors)))
        results['rmse'].append(np.sqrt(np.mean(np.array(errors)**2)))
        # Theoretical SE: var(x^2) = E[x^4] - E[x^2]^2 = 1/5 - 1/9 = 4/45
        results['theoretical_se'].append(np.sqrt(4/45 / n))

    # Verify O(1/sqrt(n)) convergence
    print("Convergence Analysis:")
    print(f"{'n':>8} {'RMSE':>10} {'Theory':>10} {'Ratio':>8}")
    print("-" * 40)
    for i, n in enumerate(sample_sizes):
        ratio = results['rmse'][i] / results['theoretical_se'][i]
        print(f"{n:>8} {results['rmse'][i]:>10.6f} {results['theoretical_se'][i]:>10.6f} {ratio:>8.2f}")

    return results

convergence = analyze_convergence_rate()
```

<Warning>
Monte Carlo converges at O(1/√n), meaning you need 100× more samples to reduce error by 10×. This is the fundamental limitation that quantum computing aims to improve.
</Warning>

## Variance Reduction Techniques

Several techniques can reduce variance without increasing sample size.

### Antithetic Variates

Use negative correlation between sample pairs.

```python
def antithetic_variates():
    """Implement antithetic variates for variance reduction."""

    S0, K, r, sigma, T = 100, 100, 0.05, 0.2, 1.0
    n_samples = 50000

    np.random.seed(42)

    # Standard Monte Carlo
    Z = np.random.standard_normal(n_samples)
    S_T = S0 * np.exp((r - 0.5*sigma**2)*T + sigma*np.sqrt(T)*Z)
    payoffs_std = np.maximum(S_T - K, 0)

    std_price = np.exp(-r*T) * np.mean(payoffs_std)
    std_se = np.exp(-r*T) * np.std(payoffs_std) / np.sqrt(n_samples)

    # Antithetic variates (use Z and -Z)
    n_pairs = n_samples // 2
    Z = np.random.standard_normal(n_pairs)

    S_T_pos = S0 * np.exp((r - 0.5*sigma**2)*T + sigma*np.sqrt(T)*Z)
    S_T_neg = S0 * np.exp((r - 0.5*sigma**2)*T + sigma*np.sqrt(T)*(-Z))

    payoff_pos = np.maximum(S_T_pos - K, 0)
    payoff_neg = np.maximum(S_T_neg - K, 0)
    payoffs_anti = (payoff_pos + payoff_neg) / 2

    anti_price = np.exp(-r*T) * np.mean(payoffs_anti)
    anti_se = np.exp(-r*T) * np.std(payoffs_anti) / np.sqrt(n_pairs)

    print("Antithetic Variates Comparison:")
    print(f"Standard MC: ${std_price:.4f} ± ${1.96*std_se:.4f}")
    print(f"Antithetic:  ${anti_price:.4f} ± ${1.96*anti_se:.4f}")
    print(f"Variance reduction: {(1 - (anti_se/std_se)**2)*100:.1f}%")

    return std_price, anti_price, std_se, anti_se

antithetic_variates()
```

### Control Variates

Use known expectations to reduce variance.

```python
def control_variates():
    """Implement control variates for variance reduction."""

    S0, K, r, sigma, T = 100, 100, 0.05, 0.2, 1.0
    n_samples = 50000

    np.random.seed(42)

    # Generate samples
    Z = np.random.standard_normal(n_samples)
    S_T = S0 * np.exp((r - 0.5*sigma**2)*T + sigma*np.sqrt(T)*Z)

    # Target: Call payoff
    Y = np.maximum(S_T - K, 0)

    # Control variate: Terminal stock price (known expectation)
    X = S_T
    E_X = S0 * np.exp(r*T)  # Known under risk-neutral measure

    # Optimal beta
    cov_XY = np.cov(X, Y)[0, 1]
    var_X = np.var(X)
    beta = cov_XY / var_X

    # Control variate adjustment
    Y_cv = Y - beta * (X - E_X)

    # Prices
    std_price = np.exp(-r*T) * np.mean(Y)
    cv_price = np.exp(-r*T) * np.mean(Y_cv)

    std_se = np.exp(-r*T) * np.std(Y) / np.sqrt(n_samples)
    cv_se = np.exp(-r*T) * np.std(Y_cv) / np.sqrt(n_samples)

    print("Control Variates Comparison:")
    print(f"Standard MC: ${std_price:.4f} ± ${1.96*std_se:.4f}")
    print(f"Control var: ${cv_price:.4f} ± ${1.96*cv_se:.4f}")
    print(f"Optimal beta: {beta:.4f}")
    print(f"Variance reduction: {(1 - (cv_se/std_se)**2)*100:.1f}%")

    return std_price, cv_price

control_variates()
```

### Importance Sampling

Sample from a different distribution to reduce variance in regions of interest.

```python
def importance_sampling():
    """Implement importance sampling for rare event estimation."""

    # Estimate probability of large losses
    # Target: P(X > threshold) where X ~ N(0, 1)
    threshold = 3.0  # Rare event

    n_samples = 100000
    np.random.seed(42)

    # Standard Monte Carlo
    X_std = np.random.standard_normal(n_samples)
    indicator_std = X_std > threshold
    p_std = np.mean(indicator_std)
    se_std = np.std(indicator_std) / np.sqrt(n_samples)

    # Importance sampling: sample from N(threshold, 1)
    mu_is = threshold  # Shift mean to region of interest
    X_is = np.random.normal(mu_is, 1, n_samples)

    # Likelihood ratio (importance weight)
    # p(x)/q(x) = exp(-x^2/2) / exp(-(x-mu)^2/2)
    log_weight = -0.5*X_is**2 + 0.5*(X_is - mu_is)**2
    weight = np.exp(log_weight)

    # Importance sampling estimate
    indicator_is = X_is > threshold
    weighted_indicator = indicator_is * weight
    p_is = np.mean(weighted_indicator)
    se_is = np.std(weighted_indicator) / np.sqrt(n_samples)

    # True probability
    from scipy.stats import norm
    p_true = 1 - norm.cdf(threshold)

    print("Importance Sampling for Rare Events:")
    print(f"True P(X > {threshold}): {p_true:.6f}")
    print(f"Standard MC: {p_std:.6f} ± {1.96*se_std:.6f}")
    print(f"Importance Sampling: {p_is:.6f} ± {1.96*se_is:.6f}")
    print(f"Variance reduction: {(1 - (se_is/se_std)**2)*100:.1f}%")

    return p_true, p_std, p_is

importance_sampling()
```

<Tip>
Combine multiple variance reduction techniques for maximum effect. Antithetic variates with control variates is a common combination in option pricing.
</Tip>

## Computational Complexity

Understanding Monte Carlo's computational limits reveals the opportunity for quantum speedup.

```python
def computational_analysis():
    """Analyze computational requirements for different accuracy levels."""

    # For accuracy ε, Monte Carlo needs O(1/ε²) samples
    target_accuracies = [0.1, 0.01, 0.001, 0.0001]

    # Assume variance σ² = 1 (normalized)
    sigma = 1.0

    print("Classical Monte Carlo Complexity:")
    print(f"{'Accuracy ε':>12} {'Samples needed':>15} {'Relative cost':>15}")
    print("-" * 45)

    base_samples = (sigma / target_accuracies[0])**2

    for eps in target_accuracies:
        n_samples = (sigma / eps)**2
        relative_cost = n_samples / base_samples
        print(f"{eps:>12.4f} {int(n_samples):>15,} {relative_cost:>15.0f}x")

    print("\nKey insight: 10× better accuracy requires 100× more samples!")

    return target_accuracies

computational_analysis()
```

## Multi-dimensional Monte Carlo

Monte Carlo excels in high dimensions where grid methods fail.

```python
def multidimensional_mc():
    """Demonstrate Monte Carlo in high dimensions."""

    def basket_option_price(n_assets, n_samples=100000, seed=42):
        """Price a basket option on n correlated assets."""
        np.random.seed(seed)

        S0 = 100
        K = 100
        r = 0.05
        sigma = 0.2
        T = 1.0
        rho = 0.5  # Correlation between assets

        # Correlation matrix
        corr = np.full((n_assets, n_assets), rho)
        np.fill_diagonal(corr, 1.0)

        # Cholesky decomposition for correlated samples
        L = np.linalg.cholesky(corr)

        # Generate correlated returns
        Z = np.random.standard_normal((n_samples, n_assets))
        Z_corr = Z @ L.T

        # Terminal prices
        S_T = S0 * np.exp((r - 0.5*sigma**2)*T + sigma*np.sqrt(T)*Z_corr)

        # Basket payoff (average of assets)
        basket_avg = np.mean(S_T, axis=1)
        payoffs = np.maximum(basket_avg - K, 0)

        price = np.exp(-r*T) * np.mean(payoffs)
        std_err = np.exp(-r*T) * np.std(payoffs) / np.sqrt(n_samples)

        return price, std_err

    print("Basket Option Pricing (High-Dimensional MC):")
    print(f"{'Assets':>8} {'Price':>10} {'Std Err':>10}")
    print("-" * 30)

    for n_assets in [2, 5, 10, 20, 50]:
        price, std_err = basket_option_price(n_assets)
        print(f"{n_assets:>8} ${price:>9.4f} ${std_err:>9.4f}")

    print("\nNote: MC error is independent of dimension!")

    return True

multidimensional_mc()
```

## Quasi-Monte Carlo

Low-discrepancy sequences can improve convergence.

```python
def quasi_monte_carlo():
    """Compare Monte Carlo with Quasi-Monte Carlo using Sobol sequences."""
    from scipy.stats.qmc import Sobol, scale

    # Integral of f(x,y) = x*y over [0,1]^2
    # True value = 0.25
    true_value = 0.25

    def f(x):
        return x[:, 0] * x[:, 1]

    sample_sizes = [100, 500, 1000, 5000, 10000]
    n_trials = 50

    mc_errors = []
    qmc_errors = []

    for n in sample_sizes:
        mc_trial_errors = []
        qmc_trial_errors = []

        for trial in range(n_trials):
            # Standard Monte Carlo
            x_mc = np.random.uniform(0, 1, (n, 2))
            mc_estimate = np.mean(f(x_mc))
            mc_trial_errors.append(abs(mc_estimate - true_value))

            # Quasi-Monte Carlo (Sobol sequence)
            sampler = Sobol(d=2, scramble=True, seed=trial)
            x_qmc = sampler.random(n)
            qmc_estimate = np.mean(f(x_qmc))
            qmc_trial_errors.append(abs(qmc_estimate - true_value))

        mc_errors.append(np.mean(mc_trial_errors))
        qmc_errors.append(np.mean(qmc_trial_errors))

    print("Monte Carlo vs Quasi-Monte Carlo:")
    print(f"{'n':>8} {'MC Error':>12} {'QMC Error':>12} {'Improvement':>12}")
    print("-" * 50)

    for i, n in enumerate(sample_sizes):
        improvement = mc_errors[i] / qmc_errors[i]
        print(f"{n:>8} {mc_errors[i]:>12.6f} {qmc_errors[i]:>12.6f} {improvement:>11.1f}x")

    return mc_errors, qmc_errors

quasi_monte_carlo()
```

## Summary

Classical Monte Carlo methods provide:

| Aspect | Description |
|--------|-------------|
| Convergence | O(1/√n) - fundamental limitation |
| Dimension independence | Error doesn't grow with dimensions |
| Flexibility | Works for any integrable function |
| Parallelization | Embarrassingly parallel |

Key variance reduction techniques:

- **Antithetic variates**: Use negative correlation
- **Control variates**: Exploit known expectations
- **Importance sampling**: Sample rare events efficiently
- **Stratification**: Force representative sampling

The O(1/√n) convergence rate is the key limitation that quantum algorithms aim to improve to O(1/n), providing quadratic speedup.
