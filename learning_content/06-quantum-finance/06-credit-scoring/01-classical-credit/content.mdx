# Classical Credit Scoring

Credit scoring is one of the most important applications of machine learning in finance. This lesson covers traditional methods that quantum approaches aim to improve.

<Definition term="Credit Scoring">
A statistical technique that transforms borrower characteristics into a numerical score representing the probability of default or creditworthiness. Used by lenders to make loan approval and pricing decisions.
</Definition>

## Logistic Regression for Credit Scoring

Logistic regression is the foundational model for credit scoring.

```python
from scipy.special import expit as sigmoid

def logistic_regression_credit():
    """Implement logistic regression for credit scoring."""

    print("Logistic Regression Credit Model")
    print("=" * 50)

    # Generate synthetic credit data
    np.random.seed(42)
    n_samples = 1000

    # Features
    income = np.random.lognormal(10.5, 0.5, n_samples)  # Annual income
    debt_ratio = np.random.beta(2, 5, n_samples)  # Debt-to-income
    credit_history = np.random.randint(0, 10, n_samples)  # Years of credit
    num_defaults = np.random.poisson(0.3, n_samples)  # Past defaults

    # True default probability (logistic model)
    log_odds = (-3 +
                -0.00002 * income +      # Higher income = lower risk
                5 * debt_ratio +          # Higher debt = higher risk
                -0.2 * credit_history +   # Longer history = lower risk
                2 * num_defaults)         # Past defaults = higher risk

    true_prob = sigmoid(log_odds)
    default = np.random.binomial(1, true_prob)

    # Create dataset
    X = np.column_stack([income, debt_ratio, credit_history, num_defaults])
    y = default

    print(f"Dataset: {n_samples} samples")
    print(f"Default rate: {y.mean()*100:.1f}%")
    print(f"Features: Income, Debt Ratio, Credit History, Past Defaults")

    # Split data
    X_train, X_test, y_train, y_test = train_test_split(
        X, y, test_size=0.3, random_state=42
    )

    # Standardize features
    scaler = StandardScaler()
    X_train_scaled = scaler.fit_transform(X_train)
    X_test_scaled = scaler.transform(X_test)

    # Simple gradient descent for logistic regression
    def fit_logistic(X, y, learning_rate=0.01, epochs=1000):
        n_features = X.shape[1]
        weights = np.zeros(n_features)
        bias = 0

        for epoch in range(epochs):
            # Forward pass
            linear = X @ weights + bias
            predictions = sigmoid(linear)

            # Compute gradients
            error = predictions - y
            grad_w = X.T @ error / len(y)
            grad_b = np.mean(error)

            # Update weights
            weights -= learning_rate * grad_w
            bias -= learning_rate * grad_b

        return weights, bias

    weights, bias = fit_logistic(X_train_scaled, y_train)

    print(f"\nLearned Coefficients (standardized):")
    feature_names = ['Income', 'Debt Ratio', 'Credit History', 'Past Defaults']
    for name, w in zip(feature_names, weights):
        direction = "↓ risk" if w < 0 else "↑ risk"
        print(f"  {name}: {w:.3f} ({direction})")

    # Evaluate
    test_probs = sigmoid(X_test_scaled @ weights + bias)
    test_preds = (test_probs > 0.5).astype(int)
    accuracy = np.mean(test_preds == y_test)

    print(f"\nTest Accuracy: {accuracy*100:.1f}%")

    return weights, accuracy

logistic_regression_credit()
```

## Decision Trees and Random Forests

Modern credit scoring often uses ensemble methods.

```python
def tree_based_credit_scoring():
    """Decision tree and random forest for credit scoring."""

    print("Tree-Based Credit Scoring")
    print("=" * 50)

    # Generate data
    np.random.seed(42)
    n_samples = 2000

    # Features
    income = np.random.lognormal(10.5, 0.5, n_samples)
    debt_ratio = np.random.beta(2, 5, n_samples)
    employment_years = np.random.exponential(5, n_samples)
    num_accounts = np.random.poisson(5, n_samples)
    credit_utilization = np.random.beta(3, 4, n_samples)

    # Non-linear decision rules
    default_prob = np.zeros(n_samples)

    # Rule 1: High debt ratio is risky
    default_prob += 0.3 * (debt_ratio > 0.4)

    # Rule 2: Low income with high utilization
    default_prob += 0.25 * ((income < 40000) & (credit_utilization > 0.7))

    # Rule 3: Short employment
    default_prob += 0.2 * (employment_years < 2)

    # Rule 4: Too many accounts
    default_prob += 0.15 * (num_accounts > 8)

    # Add noise
    default_prob = np.clip(default_prob + np.random.normal(0, 0.1, n_samples), 0, 1)
    default = np.random.binomial(1, default_prob)

    X = np.column_stack([income, debt_ratio, employment_years,
                         num_accounts, credit_utilization])
    y = default

    print(f"Dataset: {n_samples} samples, Default rate: {y.mean()*100:.1f}%")

    # Implement simple decision tree (single split)
    def find_best_split(X, y, feature_idx):
        """Find best split point for a feature."""
        feature = X[:, feature_idx]
        thresholds = np.percentile(feature, np.arange(10, 100, 10))

        best_gini = float('inf')
        best_threshold = None

        for t in thresholds:
            left_mask = feature <= t
            right_mask = ~left_mask

            if left_mask.sum() < 10 or right_mask.sum() < 10:
                continue

            # Gini impurity
            left_gini = 1 - (y[left_mask].mean()**2 + (1-y[left_mask].mean())**2)
            right_gini = 1 - (y[right_mask].mean()**2 + (1-y[right_mask].mean())**2)

            weighted_gini = (left_mask.sum() * left_gini +
                           right_mask.sum() * right_gini) / len(y)

            if weighted_gini < best_gini:
                best_gini = weighted_gini
                best_threshold = t

        return best_threshold, best_gini

    # Find best feature and split
    feature_names = ['Income', 'Debt Ratio', 'Empl. Years',
                     'Num Accounts', 'Credit Util.']

    print("\nDecision Tree Analysis (Single Split):")
    for i, name in enumerate(feature_names):
        threshold, gini = find_best_split(X, y, i)
        if threshold is not None:
            print(f"  {name}: split at {threshold:.2f}, Gini={gini:.4f}")

    # Simple random forest approximation
    def bootstrap_sample(X, y):
        n = len(y)
        indices = np.random.choice(n, n, replace=True)
        return X[indices], y[indices]

    n_trees = 10
    predictions = np.zeros((len(y), n_trees))

    print(f"\nRandom Forest ({n_trees} trees):")
    for t in range(n_trees):
        X_boot, y_boot = bootstrap_sample(X, y)

        # Find best feature for this tree (with feature subsampling)
        features_to_try = np.random.choice(5, 3, replace=False)
        best_feature = None
        best_gini = float('inf')
        best_thresh = None

        for f in features_to_try:
            thresh, gini = find_best_split(X_boot, y_boot, f)
            if thresh is not None and gini < best_gini:
                best_gini = gini
                best_feature = f
                best_thresh = thresh

        # Predict based on split
        if best_feature is not None:
            mask = X[:, best_feature] <= best_thresh
            y_boot_split = y_boot[X_boot[:, best_feature] <= best_thresh]
            predictions[mask, t] = y_boot_split.mean() if len(y_boot_split) > 0 else 0.5
            predictions[~mask, t] = 1 - predictions[mask, t].mean()

    # Ensemble prediction
    ensemble_pred = predictions.mean(axis=1)
    ensemble_class = (ensemble_pred > 0.5).astype(int)
    accuracy = np.mean(ensemble_class == y)

    print(f"  Ensemble Accuracy: {accuracy*100:.1f}%")

    return accuracy

tree_based_credit_scoring()
```

## Model Evaluation Metrics

Credit scoring requires specialized evaluation metrics.

```python
def credit_scoring_metrics():
    """Specialized metrics for credit scoring evaluation."""

    print("Credit Scoring Evaluation Metrics")
    print("=" * 55)

    # Simulated predictions
    np.random.seed(42)
    n = 1000
    y_true = np.random.binomial(1, 0.15, n)  # 15% default rate
    y_prob = np.clip(y_true * 0.6 + np.random.normal(0.2, 0.15, n), 0, 1)
    y_pred = (y_prob > 0.5).astype(int)

    # Confusion matrix
    tp = np.sum((y_pred == 1) & (y_true == 1))
    fp = np.sum((y_pred == 1) & (y_true == 0))
    tn = np.sum((y_pred == 0) & (y_true == 0))
    fn = np.sum((y_pred == 0) & (y_true == 1))

    print("Confusion Matrix:")
    print(f"  True Positives (caught defaults): {tp}")
    print(f"  False Positives (rejected good): {fp}")
    print(f"  True Negatives (approved good): {tn}")
    print(f"  False Negatives (missed defaults): {fn}")

    # Key metrics
    accuracy = (tp + tn) / (tp + tn + fp + fn)
    precision = tp / (tp + fp) if (tp + fp) > 0 else 0
    recall = tp / (tp + fn) if (tp + fn) > 0 else 0
    f1 = 2 * precision * recall / (precision + recall) if (precision + recall) > 0 else 0

    print(f"\nStandard Metrics:")
    print(f"  Accuracy: {accuracy*100:.1f}%")
    print(f"  Precision: {precision*100:.1f}%")
    print(f"  Recall: {recall*100:.1f}%")
    print(f"  F1 Score: {f1:.3f}")

    # ROC-AUC (simplified calculation)
    def calculate_auc(y_true, y_prob):
        """Calculate AUC using trapezoidal rule."""
        thresholds = np.linspace(0, 1, 100)
        tpr_list = []
        fpr_list = []

        for t in thresholds:
            pred = (y_prob >= t).astype(int)
            tp = np.sum((pred == 1) & (y_true == 1))
            fp = np.sum((pred == 1) & (y_true == 0))
            fn = np.sum((pred == 0) & (y_true == 1))
            tn = np.sum((pred == 0) & (y_true == 0))

            tpr = tp / (tp + fn) if (tp + fn) > 0 else 0
            fpr = fp / (fp + tn) if (fp + tn) > 0 else 0

            tpr_list.append(tpr)
            fpr_list.append(fpr)

        # Sort by FPR
        sorted_pairs = sorted(zip(fpr_list, tpr_list))
        fpr_sorted = [p[0] for p in sorted_pairs]
        tpr_sorted = [p[1] for p in sorted_pairs]

        # Trapezoidal AUC
        auc = 0
        for i in range(1, len(fpr_sorted)):
            auc += (fpr_sorted[i] - fpr_sorted[i-1]) * (tpr_sorted[i] + tpr_sorted[i-1]) / 2

        return auc

    auc = calculate_auc(y_true, y_prob)
    print(f"\nROC-AUC: {auc:.3f}")

    # Gini coefficient
    gini = 2 * auc - 1
    print(f"Gini Coefficient: {gini:.3f}")

    # KS statistic (simplified)
    def calculate_ks(y_true, y_prob):
        """Calculate Kolmogorov-Smirnov statistic."""
        sorted_indices = np.argsort(y_prob)[::-1]
        y_sorted = y_true[sorted_indices]

        n_defaults = y_true.sum()
        n_non_defaults = len(y_true) - n_defaults

        cum_defaults = np.cumsum(y_sorted) / n_defaults
        cum_non_defaults = np.cumsum(1 - y_sorted) / n_non_defaults

        ks = np.max(np.abs(cum_defaults - cum_non_defaults))
        return ks

    ks = calculate_ks(y_true, y_prob)
    print(f"KS Statistic: {ks:.3f}")

    # Business interpretation
    print("\nBusiness Interpretation:")
    print(f"  • Model identifies {recall*100:.0f}% of eventual defaults")
    print(f"  • {precision*100:.0f}% of flagged applications would default")
    print(f"  • KS of {ks:.2f} indicates {'good' if ks > 0.3 else 'moderate'} separation")

    return {'auc': auc, 'gini': gini, 'ks': ks}

credit_scoring_metrics()
```

<Tip>
In credit scoring, the KS (Kolmogorov-Smirnov) statistic is often more relevant than accuracy because it measures how well the model separates good and bad borrowers across the entire score distribution.
</Tip>

## Summary

Classical credit scoring fundamentals:

| Method | Strengths | Limitations |
|--------|-----------|-------------|
| Logistic Regression | Interpretable, stable | Linear only |
| Decision Trees | Handles non-linearity | Overfits |
| Random Forest | Robust, accurate | Less interpretable |
| Gradient Boosting | State-of-the-art | Slow training |
| Neural Networks | Any function | Black box |

Key takeaways:

- Credit scoring predicts default probability from borrower features
- Logistic regression remains foundational for its interpretability
- Ensemble methods (RF, GBM) often achieve best classical performance
- Specialized metrics (AUC, KS, Gini) are essential for evaluation
- Classical methods face computational limits with high-dimensional data
- Quantum approaches may improve optimization and feature learning
