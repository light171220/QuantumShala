## 1. Introduction to Stress Testing

Stress testing evaluates portfolio performance under extreme but plausible conditions.

<Definition term="Stress Testing">
A risk management technique that tests portfolio resilience against hypothetical extreme market scenarios or historical crisis events. Unlike VaR, stress tests focus on specific scenarios rather than probability distributions.
</Definition>

### Why Stress Test?

| VaR Limitation | Stress Test Advantage |
|----------------|----------------------|
| Based on normal conditions | Tests extreme conditions |
| Historical window limited | Can test unprecedented events |
| Single number output | Rich scenario information |
| May miss tail dependencies | Captures crisis correlations |

## 2. Historical Stress Testing

Use past market crises to stress current portfolios.

```python
def apply_historical_stress(portfolio_weights, asset_exposures, scenario_name):
    """
    Apply historical stress scenario to portfolio.

    Args:
        portfolio_weights: Dict of {asset: weight}
        asset_exposures: Dict of {asset: exposure_multiplier}
        scenario_name: Name of historical scenario

    Returns:
        Portfolio P&L under stress
    """
    scenario = HISTORICAL_CRISES[scenario_name]
    portfolio_pnl = 0.0

    for asset, weight in portfolio_weights.items():
        if asset in scenario:
            asset_return = scenario[asset]
            exposure = asset_exposures.get(asset, 1.0)
            portfolio_pnl += weight * asset_return * exposure

    return portfolio_pnl

# Example portfolio

    'S&P 500': 0.60,
    'Bonds': 0.30,
    'Gold': 0.10
}

exposures = {asset: 1.0 for asset in portfolio}

print("Historical Stress Test Results:")
print("=" * 50)
for scenario_name in HISTORICAL_CRISES:
    pnl = apply_historical_stress(portfolio, exposures, scenario_name)
    desc = HISTORICAL_CRISES[scenario_name]['description']
    print(f"{scenario_name}:")
    print(f"  {desc}")
    print(f"  Portfolio P&L: {pnl:.2%}")
    print()
```

## 3. Hypothetical Stress Scenarios

Create custom scenarios for specific risk concerns.

```python
class StressScenario:
    """
    Define custom stress scenarios.
    """

    def __init__(self, name, description):
        self.name = name
        self.description = description
        self.shocks = {}

    def add_shock(self, risk_factor, shock_value, shock_type='absolute'):
        """
        Add a shock to a risk factor.

        Args:
            risk_factor: Name of the risk factor
            shock_value: Shock magnitude
            shock_type: 'absolute' (e.g., -0.20) or 'relative' (e.g., 1.5x)
        """
        self.shocks[risk_factor] = {
            'value': shock_value,
            'type': shock_type
        }

    def apply_to_portfolio(self, portfolio_positions, sensitivities):
        """
        Apply scenario to portfolio.

        Args:
            portfolio_positions: Dict of {asset: position_value}
            sensitivities: Dict of {asset: {risk_factor: sensitivity}}

        Returns:
            Total P&L
        """
        total_pnl = 0.0

        for asset, position in portfolio_positions.items():
            asset_pnl = 0.0

            for factor, sensitivity in sensitivities.get(asset, {}).items():
                if factor in self.shocks:
                    shock = self.shocks[factor]
                    if shock['type'] == 'absolute':
                        asset_pnl += position * sensitivity * shock['value']
                    else:  # relative
                        asset_pnl += position * sensitivity * (shock['value'] - 1)

            total_pnl += asset_pnl

        return total_pnl


# Create hypothetical scenarios
def create_rate_shock_scenario():
```

## 4. Sensitivity-Based Stress Testing

Use Greeks and factor sensitivities for faster stress calculations.

```python
class SensitivityStressTester:
    """
    Stress testing using portfolio sensitivities.
    """

    def __init__(self):
        self.risk_factors = {}
        self.portfolio_sensitivities = {}

    def set_factor_levels(self, factor_name, current_level):
        """Set current level of a risk factor."""
        self.risk_factors[factor_name] = current_level

    def set_sensitivity(self, portfolio_name, factor_name, sensitivity):
        """
        Set portfolio sensitivity to a factor.

        sensitivity = dP/dF (change in portfolio per unit factor change)
        """
        if portfolio_name not in self.portfolio_sensitivities:
            self.portfolio_sensitivities[portfolio_name] = {}
        self.portfolio_sensitivities[portfolio_name][factor_name] = sensitivity

    def stress_test(self, factor_shocks):
        """
        Run stress test with specified factor shocks.

        Args:
            factor_shocks: Dict of {factor: shock_value}

        Returns:
            Dict of {portfolio: pnl}
        """
        results = {}

        for portfolio, sensitivities in self.portfolio_sensitivities.items():
            pnl = 0.0
            for factor, sensitivity in sensitivities.items():
                if factor in factor_shocks:
                    pnl += sensitivity * factor_shocks[factor]
            results[portfolio] = pnl

        return results

    def generate_stress_grid(self, factor1, range1, factor2, range2, steps=5):
        """
        Generate stress results over a grid of two factors.
        """
        f1_values = np.linspace(range1[0], range1[1], steps)
        f2_values = np.linspace(range2[0], range2[1], steps)

        results = np.zeros((steps, steps))

        for i, f1 in enumerate(f1_values):
            for j, f2 in enumerate(f2_values):
                pnl = self.stress_test({factor1: f1, factor2: f2})
                results[i, j] = sum(pnl.values())

        return f1_values, f2_values, results

# Example
tester = SensitivityStressTester()

# Set up sensitivities (example: options portfolio)
tester.set_sensitivity('Portfolio', 'spot_price', 50000)    # Delta
tester.set_sensitivity('Portfolio', 'volatility', 100000)   # Vega
tester.set_sensitivity('Portfolio', 'interest_rate', -20000) # Rho

# Run various stress tests
scenarios = [
    {'spot_price': -0.10, 'volatility': 0.10},   # Down 10%, vol up
    {'spot_price': -0.20, 'volatility': 0.20},   # Down 20%, vol up more
    {'spot_price': 0.10, 'volatility': -0.05},   # Up 10%, vol down
]

print("Sensitivity-Based Stress Tests:")
print("=" * 50)
for i, scenario in enumerate(scenarios):
    results = tester.stress_test(scenario)
    print(f"Scenario {i+1}: {scenario}")
    print(f"  P&L: ${results['Portfolio']:,.0f}")
```

<Tip>
Sensitivity-based stress testing is fast but assumes linearity. For large shocks, consider re-pricing positions using full valuation models.
</Tip>

## 5. Reverse Stress Testing

Start from an unacceptable outcome and find scenarios that cause it.

```python
def reverse_stress_test(portfolio_value, max_loss, sensitivities, factor_ranges):
    """
    Find scenarios that cause losses exceeding max_loss.

    Args:
        portfolio_value: Current portfolio value
        max_loss: Maximum acceptable loss (as fraction)
        sensitivities: Dict of {factor: portfolio_sensitivity}
        factor_ranges: Dict of {factor: (min, max)}

    Returns:
        List of scenarios causing the specified loss
    """
    target_pnl = -max_loss * portfolio_value
    critical_scenarios = []

    # Grid search (in practice, use optimization)
    n_samples = 1000
    np.random.seed(42)

    for _ in range(n_samples):
        scenario = {}
        pnl = 0.0

        for factor, (f_min, f_max) in factor_ranges.items():
            shock = np.random.uniform(f_min, f_max)
            scenario[factor] = shock
            pnl += sensitivities.get(factor, 0) * shock

        if pnl <= target_pnl:
            scenario['pnl'] = pnl
            critical_scenarios.append(scenario)

    return critical_scenarios

# Example
portfolio_value = 10000000
sensitivities = {
    'equity': 0.60 * portfolio_value,
    'rates': -0.30 * portfolio_value * 5,  # Duration 5
    'credit': -0.10 * portfolio_value * 3   # Spread duration 3
}

factor_ranges = {
    'equity': (-0.40, 0.20),
    'rates': (-0.02, 0.03),
    'credit': (-0.01, 0.05)
}

critical = reverse_stress_test(portfolio_value, 0.20, sensitivities, factor_ranges)

print("Reverse Stress Test Results:")
print(f"Finding scenarios causing >{20}% loss")
print("=" * 50)
print(f"Found {len(critical)} critical scenarios")
if critical:
    print("\nExample critical scenarios:")
    for i, s in enumerate(critical[:3]):
        print(f"  Scenario {i+1}:")
        for factor, shock in s.items():
            if factor != 'pnl':
                print(f"    {factor}: {shock:+.2%}")
        print(f"    P&L: ${s['pnl']:,.0f}")
```

## 6. Regulatory Stress Testing

Banks and financial institutions must comply with regulatory requirements.

### Key Regulatory Frameworks

| Framework | Region | Focus |
|-----------|--------|-------|
| CCAR/DFAST | US | Bank capital adequacy |
| EBA Stress Tests | EU | European bank resilience |
| Basel III | Global | Capital requirements |
| FRTB | Global | Trading book risk |

```python
class RegulatoryStressTest:
    """
    Framework for regulatory stress testing.
    """

    def __init__(self, portfolio_value, capital_ratio_current):
        self.portfolio_value = portfolio_value
        self.capital_ratio = capital_ratio_current
        self.results = {}

    def run_scenario(self, scenario_name, factor_shocks, sensitivities):
        """
        Run a regulatory stress scenario.
        """
        pnl = 0.0
        for factor, shock in factor_shocks.items():
            pnl += sensitivities.get(factor, 0) * shock

        new_value = self.portfolio_value + pnl
        loss_rate = pnl / self.portfolio_value

        # Simplified capital impact
        capital_impact = loss_rate * 0.5  # Rough approximation
        new_capital_ratio = self.capital_ratio + capital_impact

        self.results[scenario_name] = {
            'pnl': pnl,
            'loss_rate': loss_rate,
            'capital_impact': capital_impact,
            'new_capital_ratio': new_capital_ratio,
            'passes_minimum': new_capital_ratio >= 0.045  # Basel III minimum
        }

        return self.results[scenario_name]

    def summary_report(self):
        """Generate regulatory summary report."""
        print("\nRegulatory Stress Test Summary")
        print("=" * 60)
        print(f"Initial Capital Ratio: {self.capital_ratio:.2%}")
        print(f"Minimum Required: 4.50%\n")

        for scenario, result in self.results.items():
            status = "✓ PASS" if result['passes_minimum'] else "✗ FAIL"
            print(f"{scenario}:")
            print(f"  P&L: ${result['pnl']:,.0f} ({result['loss_rate']:.2%})")
            print(f"  Post-Stress Capital: {result['new_capital_ratio']:.2%}")
            print(f"  Status: {status}")
            print()

# Example
reg_test = RegulatoryStressTest(
    portfolio_value=100000000,
    capital_ratio_current=0.12
)

sensitivities = {
    'equity': 50000000,
    'credit': -30000000,
    'rates': -20000000
}

# Regulatory scenarios
scenarios = {
    'Baseline': {'equity': -0.05, 'credit': 0.01, 'rates': 0.005},
    'Adverse': {'equity': -0.15, 'credit': 0.02, 'rates': 0.01},
    'Severely Adverse': {'equity': -0.30, 'credit': 0.04, 'rates': 0.02}
}

for name, shocks in scenarios.items():
    reg_test.run_scenario(name, shocks, sensitivities)

reg_test.summary_report()
```
